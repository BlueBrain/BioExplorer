29 OSPRayVolume::OSPRayVolume(const Vector3ui& dimensions, const Vector3f& spacing, const DataType type,
30  const VolumeParameters& params, OSPTransferFunction transferFunction,
31  const std::string& volumeType)
32  : Volume(dimensions, spacing, type)
33  , _parameters(params)
34  , _volume(ospNewVolume(volumeType.c_str()))
35 {
36  osphelper::set(_volume, "dimensions", Vector3i(dimensions));
37  osphelper::set(_volume, "gridSpacing", Vector3f(spacing));
38 
39  switch (type)
40  {
41  case DataType::FLOAT:
42  osphelper::set(_volume, "voxelType", "float");
43  _ospType = OSP_FLOAT;
44  _dataSize = 4;
45  break;
46  case DataType::DOUBLE:
47  osphelper::set(_volume, "voxelType", "double");
48  _ospType = OSP_DOUBLE;
49  _dataSize = 8;
50  break;
51  case DataType::UINT8:
52  osphelper::set(_volume, "voxelType", "uchar");
53  _ospType = OSP_UINT;
54  _dataSize = 1;
55  break;
56  case DataType::UINT16:
57  osphelper::set(_volume, "voxelType", "ushort");
58  _ospType = OSP_UINT2;
59  _dataSize = 2;
60  break;
61  case DataType::INT16:
62  osphelper::set(_volume, "voxelType", "short");
63  _ospType = OSP_INT2;
64  _dataSize = 2;
65  break;
66  case DataType::UINT32:
67  case DataType::INT8:
68  case DataType::INT32:
69  throw std::runtime_error("Unsupported voxel type " + std::to_string(int(type)));
70  }
71 
72  ospSetObject(_volume, "transferFunction", transferFunction);
73 }

121 void OSPRayVolume::commit()
122 {
123  if (_parameters.isModified())
124  {
125  osphelper::set(_volume, "gradientShadingEnabled", _parameters.getGradientShading());
126  osphelper::set(_volume, "adaptiveMaxSamplingRate",
127  static_cast<float>(_parameters.getAdaptiveMaxSamplingRate()));
128  osphelper::set(_volume, "adaptiveSampling", _parameters.getAdaptiveSampling());
129  osphelper::set(_volume, "singleShade", _parameters.getSingleShade());
130  osphelper::set(_volume, "preIntegration", _parameters.getPreIntegration());
131  osphelper::set(_volume, "samplingRate", static_cast<float>(_parameters.getSamplingRate()));
132  osphelper::set(_volume, "specular", Vector3f(_parameters.getSpecular()));
133  osphelper::set(_volume, "volumeClippingBoxLower", Vector3f(_parameters.getClipBox().getMin()));
134  osphelper::set(_volume, "volumeClippingBoxUpper", Vector3f(_parameters.getClipBox().getMax()));
135  }
136  if (isModified() || _parameters.isModified())
137  ospCommit(_volume);
138  resetModified();
139 }
