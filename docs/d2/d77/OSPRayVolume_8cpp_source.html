35 OSPRayVolume::OSPRayVolume(const Vector3ui& dimensions, const Vector3f& spacing, const DataType type,
36  const VolumeParameters& params, OSPTransferFunction transferFunction,
37  const std::string& volumeType)
38  : Volume(dimensions, spacing, type)
39  , _parameters(params)
40  , _volume(ospNewVolume(volumeType.c_str()))
41 {
42  osphelper::set(_volume, OSPRAY_VOLUME_PROPERTY_DIMENSIONS, Vector3i(dimensions));
43  osphelper::set(_volume, OSPRAY_VOLUME_PROPERTY_GRID_SPACING, Vector3f(spacing));
44 
45  switch (type)
46  {
47  case DataType::FLOAT:
48  osphelper::set(_volume, OSPRAY_VOLUME_VOXEL_TYPE, "float");
49  _ospType = OSP_FLOAT;
50  _dataSize = 4;
51  break;
52  case DataType::DOUBLE:
53  osphelper::set(_volume, OSPRAY_VOLUME_VOXEL_TYPE, "double");
54  _ospType = OSP_DOUBLE;
55  _dataSize = 8;
56  break;
57  case DataType::UINT8:
58  osphelper::set(_volume, OSPRAY_VOLUME_VOXEL_TYPE, "uchar");
59  _ospType = OSP_UINT;
60  _dataSize = 1;
61  break;
62  case DataType::UINT16:
63  osphelper::set(_volume, OSPRAY_VOLUME_VOXEL_TYPE, "ushort");
64  _ospType = OSP_UINT2;
65  _dataSize = 2;
66  break;
67  case DataType::INT16:
68  osphelper::set(_volume, OSPRAY_VOLUME_VOXEL_TYPE, "short");
69  _ospType = OSP_INT2;
70  _dataSize = 2;
71  break;
72  case DataType::UINT32:
73  case DataType::INT8:
74  case DataType::INT32:
75  throw std::runtime_error("Unsupported voxel type " + std::to_string(int(type)));
76  }
77 
78  ospSetObject(_volume, RENDERER_PROPERTY_TRANSFER_FUNCTION, transferFunction);
79 }

128 void OSPRayVolume::commit()
129 {
130  if (_parameters.isModified())
131  {
132  osphelper::set(_volume, OSPRAY_VOLUME_GRADIENT_SHADING_ENABLED, _parameters.getGradientShading());
133  osphelper::set(_volume, OSPRAY_VOLUME_GRADIENT_OFFSET, static_cast<float>(_parameters.getGradientOffset()));
134  osphelper::set(_volume, OSPRAY_VOLUME_ADAPTIVE_MAX_SAMPLING_RATE,
135  static_cast<float>(_parameters.getAdaptiveMaxSamplingRate()));
136  osphelper::set(_volume, OSPRAY_VOLUME_ADAPTIVE_SAMPLING, _parameters.getAdaptiveSampling());
137  osphelper::set(_volume, OSPRAY_VOLUME_SINGLE_SHADE, _parameters.getSingleShade());
138  osphelper::set(_volume, OSPRAY_VOLUME_PRE_INTEGRATION, _parameters.getPreIntegration());
139  osphelper::set(_volume, OSPRAY_VOLUME_SAMPLING_RATE, static_cast<float>(_parameters.getSamplingRate()));
140  osphelper::set(_volume, OSPRAY_VOLUME_SPECULAR_EXPONENT, Vector3f(_parameters.getSpecular()));
141  osphelper::set(_volume, OSPRAY_VOLUME_USER_PARAMETERS, Vector3f(_parameters.getUserParameters()));
142  osphelper::set(_volume, OSPRAY_VOLUME_VOLUME_CLIPPING_BOX_LOWER, Vector3f(_parameters.getClipBox().getMin()));
143  osphelper::set(_volume, OSPRAY_VOLUME_VOLUME_CLIPPING_BOX_UPPER, Vector3f(_parameters.getClipBox().getMax()));
144  }
145  if (isModified() || _parameters.isModified())
146  ospCommit(_volume);
147  resetModified();
148 }
