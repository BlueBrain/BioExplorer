67 void OptiXOpenDeckCamera::commit(const OptiXCamera& camera,
68  ::optix::Context context)
69 {
70  const auto pos = camera.getPosition();
71 
72  const Vector3d u =
73  normalize(glm::rotate(camera.getOrientation(), Vector3d(1, 0, 0)));
74  const Vector3d v =
75  normalize(glm::rotate(camera.getOrientation(), Vector3d(0, 1, 0)));
76  const Vector3d w =
77  normalize(glm::rotate(camera.getOrientation(), Vector3d(0, 0, 1)));
78 
79  const auto headPos =
80  camera.getPropertyOrValue<std::array<double, 3>>("headPosition",
81  {{0., 0., 0.}});
82  const auto headRotation =
83  camera.getPropertyOrValue<std::array<double, 4>>("headRotation",
84  {{0., 0., 0., 1.}});
85  const auto headUVec =
86  glm::rotate(Quaterniond(headRotation[3], headRotation[0],
87  headRotation[1], headRotation[2]),
88  Vector3d(1., 0., 0.));
89 
90  context[CUDA_ATTR_CAMERA_SEGMENT_ID]->setUint(
91  camera.getPropertyOrValue<int>("segmentId", 0));
92  context["HALF_IPD"]->setFloat(
93  camera.getPropertyOrValue<double>(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE, 0.065) /
94  2.0);
95  context[CUDA_ATTR_CAMERA_HEAD_POS]->setFloat(headPos[0], headPos[1],
96  headPos[2]);
97  context[CUDA_ATTR_CAMERA_HEAD_UVEC]->setFloat(headUVec.x, headUVec.y,
98  headUVec.z);
99  context[CUDA_ATTR_CAMERA_EYE]->setFloat(pos.x, pos.y, pos.z);
100  context[CUDA_ATTR_CAMERA_U]->setFloat(u.x, u.y, u.z);
101  context[CUDA_ATTR_CAMERA_V]->setFloat(v.x, v.y, v.z);
102  context[CUDA_ATTR_CAMERA_W]->setFloat(w.x, w.y, w.z);
103  context[CUDA_ATTR_CAMERA_APERTURE_RADIUS]->setFloat(
104  camera.getPropertyOrValue<double>(CAMERA_PROPERTY_APERTURE_RADIUS, 0.0));
105  context[CUDA_ATTR_CAMERA_FOCAL_SCALE]->setFloat(
106  camera.getPropertyOrValue<double>(CAMERA_PROPERTY_FOCUS_DISTANCE, 1.0));
107  context[CUDA_ATTR_CAMERA_BAD_COLOR]->setFloat(1.f, 0.f, 1.f);
108  // context[CUDA_ATTR_CAMERA_OFFSET]->setFloat(0, 0);
109 }
