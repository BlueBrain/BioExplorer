95 void AbstractManipulator::rotate(const Vector3d& pivot, const double du, const double dv, AxisMode axisMode)
96 {
97  const Vector3d axisX = glm::rotate(_camera.getOrientation(), Vector3d(1.0, 0.0, 0.0));
98 
99  const Vector3d axisY = axisMode == AxisMode::localY ? glm::rotate(_camera.getOrientation(), Vector3d(0.0, 1.0, 0.0))
100  : Vector3d(0.0, 1.0, 0.0);
101 
102  const Quaterniond deltaU = glm::angleAxis(-du, axisY);
103  const Quaterniond deltaV = glm::angleAxis(-dv, axisX);
104 
105  const Quaterniond final = deltaU * deltaV * _camera.getOrientation();
106  const Vector3d dir = glm::rotate(final, Vector3d(0.0, 0.0, -1.0));
107 
108  const double rotationRadius = glm::length(_camera.getPosition() - pivot);
109  _camera.setPosition(pivot + rotationRadius * -dir);
110  _camera.setOrientation(final);
111  _camera.setTarget(pivot);
112 }
