62 RNASequence::RNASequence(Scene& scene, const RNASequenceDetails& details)
63  : Node()
64 {
65  const auto& sequence = details.contents;
66  const size_t nbElements = sequence.length();
67 
68  auto model = scene.createModel();
69 
70  size_t materialId = 0;
71  for (const auto& nucleotid : nucleotidMap)
72  {
73  auto material =
74  model->createMaterial(materialId, nucleotid.second.name);
75  brayns::PropertyMap props;
76  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
77  static_cast<int>(MaterialShadingMode::basic)});
78  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
79  props.setProperty(
80  {MATERIAL_PROPERTY_CHAMELEON_MODE,
81  static_cast<int>(
82  MaterialChameleonMode::undefined_chameleon_mode)});
83  material->setDiffuseColor(nucleotid.second.color);
84  material->updateProperties(props);
85  ++materialId;
86  }
87 
88  const size_t nbMaterials = materialId;
89 
90  PLUGIN_INFO("Sequence total length: " << nbElements);
91  PLUGIN_INFO("Created " << nbMaterials << " materials");
92 
93  Vector3f U{details.range[0], details.range[1], nbElements};
94  Vector3f V{details.range[0], details.range[1], nbElements};
95 
96  switch (details.shape)
97  {
98  case RNAShape::moebius:
99  U = {2.f * M_PI, 4.f * M_PI, nbElements};
100  V = {-0.4f, 0.4f, 1.f};
101  break;
102  case RNAShape::heart:
103  U = {0.f, 2.f * M_PI, nbElements};
104  V = {0.f, 1.f, 1.f};
105  break;
106  default:
107  break;
108  }
109 
110  const float uStep = (U.y - U.x) / U.z;
111  const float vStep = (V.y - V.x) / V.z;
112 
113  size_t elementId = 0;
114  for (float v(V.x); v < V.y; v += vStep)
115  {
116  for (float u(U.x); u < U.y; u += uStep)
117  {
118  Vector3f src;
119  Vector3f dst;
120  switch (details.shape)
121  {
122  case RNAShape::moebius:
123  {
124  src = _moebius(details.assemblyParams[0], u, v);
125  dst = _moebius(details.assemblyParams[0], u + uStep, v);
126  break;
127  }
128  case RNAShape::torus:
129  {
130  src = _torus(details.assemblyParams[0], u,
131  {details.params[0], details.params[1],
132  details.params[2]});
133  dst = _torus(details.assemblyParams[0], u + uStep,
134  {details.params[0], details.params[1],
135  details.params[2]});
136  break;
137  }
138  case RNAShape::star:
139  {
140  src = _star(details.assemblyParams[0], u);
141  dst = _star(details.assemblyParams[0], u + uStep);
142  break;
143  }
144  case RNAShape::spring:
145  {
146  src = _spring(details.assemblyParams[0], u);
147  dst = _spring(details.assemblyParams[0], u + uStep);
148  break;
149  }
150  case RNAShape::trefoilKnot:
151  {
152  src = _trefoilKnot(details.assemblyParams[0], u,
153  {details.params[0], details.params[1],
154  details.params[2]});
155  dst = _trefoilKnot(details.assemblyParams[0], u + uStep,
156  {details.params[0], details.params[1],
157  details.params[2]});
158  break;
159  }
160  case RNAShape::heart:
161  {
162  src = _heart(details.assemblyParams[0], u);
163  dst = _heart(details.assemblyParams[0], u + uStep);
164  break;
165  }
166  case RNAShape::thing:
167  {
168  src = _thing(details.assemblyParams[0], u,
169  {details.params[0], details.params[1],
170  details.params[2]});
171  dst = _thing(details.assemblyParams[0], u + uStep,
172  {details.params[0], details.params[1],
173  details.params[2]});
174  break;
175  }
176  default:
177  PLUGIN_THROW("Undefined shape");
178  break;
179  }
180 
181  const char letter = sequence[elementId];
182  if (nucleotidMap.find(letter) != nucleotidMap.end())
183  {
184  const auto& codon = nucleotidMap[letter];
185  const auto materialId = codon.index;
186  const auto radius = details.assemblyParams[1];
187  const Vector3f position{details.position[0],
188  details.position[1],
189  details.position[2]};
190  model->addCylinder(materialId,
191  {position + src, position + dst, radius});
192  if (elementId == 0)
193  model->addSphere(materialId, {position + src, radius});
194  if (elementId == nbElements - 1)
195  model->addSphere(materialId, {position + dst, radius});
196  }
197  if (elementId == nbElements)
198  break;
199  ++elementId;
200  }
201  }
202 
203  // Metadata
204  ModelMetadata metadata;
205  metadata[METADATA_ASSEMBLY] = details.assemblyName;
206  metadata["RNA sequence"] = sequence;
207  _modelDescriptor =
208  std::make_shared<ModelDescriptor>(std::move(model), details.name,
209  metadata);
210  if (_modelDescriptor &&
211  !GeneralSettings::getInstance()->getModelVisibilityOnCreation())
212  _modelDescriptor->setVisible(false);
213 }

215 Vector3f RNASequence::_trefoilKnot(const float radius, const float t,
216  const Vector3f& params)
217 {
218  return {radius * (sin(t) + 2.f * sin(params.x * t)),
219  radius * (cos(t) - 2.f * cos(params.y * t)),
220  radius * (-sin(params.z * t))};
221 }

223 Vector3f RNASequence::_torus(const float radius, const float t,
224  const Vector3f& params)
225 {
226  return {radius * (cos(t) + params.x * cos(params.y * t) * cos(t)),
227  radius * (sin(t) + params.x * cos(params.y * t) * sin(t)),
228  radius * params.x * sin(params.y * t)};
229 }

231 Vector3f RNASequence::_star(const float radius, const float t)
232 {
233  return {radius * (2.f * sin(3.f * t) * cos(t)),
234  radius * (2.f * sin(3.f * t) * sin(t)), radius * sin(3.f * t)};
235 }

237 Vector3f RNASequence::_spring(const float radius, const float t)
238 {
239  return {radius * cos(t), radius * sin(t), radius * cos(t)};
240 }

242 Vector3f RNASequence::_heart(const float radius, const float u)
243 {
244  return {radius * 4.f * pow(sin(u), 3.f),
245  radius * 0.25f *
246  (13.f * cos(u) - 5.f * cos(2.f * u) - 2.f * cos(3.f * u) -
247  cos(4.f * u)),
248  0.f};
249 }
250 
251 Vector3f RNASequence::_thing(const float radius, const float t,
252  const Vector3f& params)
253 {
254  return {radius * (sin(t) + params.x * sin(params.y * t)),
255  radius * (cos(t) - params.x * cos(params.y * t)),
256  radius * (-sin(params.z * t))};
257 }

259 Vector3f RNASequence::_moebius(const float radius, const float u, const float v)
260 {
261  return {4.f * radius * (cos(u) + v * cos(u / 2.f) * cos(u)),
262  4.f * radius * (sin(u) + v * cos(u / 2.f) * sin(u)),
263  8.f * radius * (v * sin(u / 2.f))};
264 }
