64 RNASequence::RNASequence(Scene& scene, const RNASequenceDetails& details,
65  const Vector3f& assemblyPosition,
66  const Quaterniond& assemblyRotation)
67  : Node()
68  , _scene(scene)
69  , _details(details)
70  , _assemblyPosition(assemblyPosition)
71  , _assemblyRotation(assemblyRotation)
72 {
73  const bool processAsProtein = !_details.proteinContents.empty();
74  const Quaterniond rotation = floatsToQuaterniond(_details.rotation);
75  const std::string& sequence = _details.contents;
76  const size_t nbElements = sequence.length();
77  PLUGIN_INFO("DNA Sequence length: " << nbElements);
78 
79  const auto& params = _details.assemblyParams;
80  auto randInfo =
81  floatsToRandomizationDetails(params, 0,
82  PositionRandomizationType::radial);
83 
84  Vector3f U{details.range[0], details.range[1], nbElements};
85  Vector3f V{details.range[0], details.range[1], nbElements};
86 
87  switch (details.shape)
88  {
89  case RNAShape::moebius:
90  U = {2.f * M_PI, 4.f * M_PI, nbElements};
91  V = {-0.4f, 0.4f, 1.f};
92  break;
93  case RNAShape::heart:
94  U = {0.f, 2.f * M_PI, nbElements};
95  V = {0.f, 1.f, 1.f};
96  break;
97  default:
98  break;
99  }
100 
101  const float uStep = (U.y - U.x) / U.z;
102  const float vStep = (V.y - V.x) / V.z;
103 
104  if (processAsProtein)
105  _buildRNAAsProteinInstances(U, V, rotation, randInfo);
106  else
107  _buildRNAAsCurve(U, V, rotation, randInfo);
108 }

110 void RNASequence::_buildRNAAsCurve(const Vector3f& U, const Vector3f& V,
111  const Quaterniond& rotation,
112  const RandomizationDetails& randInfo)
113 {
114  auto model = _scene.createModel();
115 
116  size_t materialId = 0;
117  for (const auto& nucleotid : nucleotidMap)
118  {
119  auto material =
120  model->createMaterial(materialId, nucleotid.second.name);
121  brayns::PropertyMap props;
122  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
123  static_cast<int>(MaterialShadingMode::basic)});
124  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
125  props.setProperty(
126  {MATERIAL_PROPERTY_CHAMELEON_MODE,
127  static_cast<int>(
128  MaterialChameleonMode::undefined_chameleon_mode)});
129  material->setDiffuseColor(nucleotid.second.color);
130  material->updateProperties(props);
131  ++materialId;
132  }
133  PLUGIN_INFO("Created " << materialId << " materials");
134 
135  const float uStep = (U.y - U.x) / U.z;
136  const float vStep = (V.y - V.x) / V.z;
137 
138  const std::string& sequence = _details.contents;
139  const size_t nbElements = sequence.length();
140  const auto position = floatsToVector3f(_details.position);
141  const auto radius = _details.assemblyParams[5];
142 
143  size_t elementId = 0;
144  for (float v(V.x); v < V.y; v += vStep)
145  {
146  for (float u(U.x); u < U.y; u += uStep)
147  {
148  Vector3f src, dst;
149  _getSegment(u, v, uStep, src, dst);
150 
151  const char letter = sequence[elementId];
152  if (nucleotidMap.find(letter) != nucleotidMap.end())
153  {
154  const auto& codon = nucleotidMap[letter];
155  const auto materialId = codon.index;
156 
157  const Vector3f assemblyPosition =
158  Vector3f(_assemblyRotation * Vector3d(_assemblyPosition));
159  const Vector3f translationSrc =
160  assemblyPosition + position +
161  Vector3f(_assemblyRotation * rotation * Vector3d(src));
162  const Vector3f translationDst =
163  assemblyPosition + position +
164  Vector3f(_assemblyRotation * rotation * Vector3d(dst));
165 
166  model->addCylinder(materialId,
167  {translationSrc, translationDst, radius});
168  if (elementId == 0)
169  model->addSphere(materialId, {translationSrc, radius});
170  if (elementId == nbElements - 1)
171  model->addSphere(materialId, {translationDst, radius});
172  }
173 
174  if (elementId >= nbElements)
175  break;
176  ++elementId;
177  }
178  }
179 
180  // Metadata
181  ModelMetadata metadata;
182  metadata[METADATA_ASSEMBLY] = _details.assemblyName;
183  metadata["RNA sequence"] = sequence;
184  _modelDescriptor =
185  std::make_shared<ModelDescriptor>(std::move(model), _details.name,
186  metadata);
187  if (_modelDescriptor &&
188  !GeneralSettings::getInstance()->getModelVisibilityOnCreation())
189  _modelDescriptor->setVisible(false);
190 }

192 void RNASequence::_buildRNAAsProteinInstances(
193  const Vector3f& U, const Vector3f& V, const Quaterniond& rotation,
194  const RandomizationDetails& randInfo)
195 {
196  const auto& sequence = _details.contents;
197  const size_t nbElements = sequence.length();
198 
199  // Load protein
200  ModelPtr model{nullptr};
201  const std::string proteinName = _details.assemblyName + "_RNA sequence";
202  ProteinDetails pd;
203  pd.assemblyName = _details.assemblyName;
204  pd.name = proteinName;
205  pd.contents = _details.proteinContents;
206  pd.recenter = true;
207 
208  _protein = ProteinPtr(new Protein(_scene, pd));
209  _modelDescriptor = _protein->getModelDescriptor();
210 
211  const float uStep = (U.y - U.x) / U.z;
212  const float vStep = (V.y - V.x) / V.z;
213 
214  const auto position = floatsToVector3f(_details.position);
215 
216  size_t elementId = 0;
217  for (float v(V.x); v < V.y; v += vStep)
218  {
219  for (float u(U.x); u < U.y; u += uStep)
220  {
221  Vector3f src, dst;
222  _getSegment(u, v, uStep, src, dst);
223 
224  const char letter = sequence[elementId];
225  if (nucleotidMap.find(letter) != nucleotidMap.end())
226  {
227  const Vector3f direction = normalize(dst - src);
228  const Vector3f normal = cross(UP_VECTOR, direction);
229  if (elementId % 50 == 0)
230  {
231  Transformation finalTransformation;
232 
233  float upOffset = 0.f;
234  if (randInfo.positionSeed != 0 &&
235  randInfo.randomizationType ==
236  PositionRandomizationType::radial)
237  upOffset =
238  randInfo.positionStrength *
239  rnd3((randInfo.positionSeed + elementId) * 10);
240 
241  const Vector3f translation =
242  Vector3f(_assemblyRotation *
243  Vector3d(_assemblyPosition)) +
244  position +
245  Vector3f(_assemblyRotation * rotation *
246  Vector3d(src + normal * upOffset));
247  finalTransformation.setTranslation(translation);
248 
249  Quaterniond instanceRotation =
250  glm::quatLookAt(normal, UP_VECTOR);
251  if (randInfo.rotationSeed != 0)
252  instanceRotation =
253  weightedRandomRotation(randInfo.rotationSeed,
254  elementId, instanceRotation,
255  randInfo.rotationStrength);
256  finalTransformation.setRotation(rotation *
257  instanceRotation);
258 
259  if (elementId == 0)
260  _modelDescriptor->setTransformation(
261  finalTransformation);
262  const ModelInstance instance(true, false,
263  finalTransformation);
264  _modelDescriptor->addInstance(instance);
265  }
266  }
267 
268  if (elementId >= nbElements)
269  break;
270  ++elementId;
271  }
272  }

275 void RNASequence::_getSegment(const float u, const float v, const float uStep,
276  Vector3f& src, Vector3f& dst)
277 {
278  const auto radius = _details.assemblyParams[0];
279  const auto params = floatsToVector3f(_details.params);
280  switch (_details.shape)
281  {
282  case RNAShape::moebius:
283  {
284  src = _moebius(radius, u, v);
285  dst = _moebius(radius, u + uStep, v);
286  break;
287  }
288  case RNAShape::torus:
289  {
290  src = _torus(radius, u, params);
291  dst = _torus(radius, u + uStep, params);
292  break;
293  }
294  case RNAShape::star:
295  {
296  src = _star(radius, u);
297  dst = _star(radius, u + uStep);
298  break;
299  }
300  case RNAShape::spring:
301  {
302  src = _spring(radius, u, params);
303  dst = _spring(radius, u + uStep, params);
304  break;
305  }
306  case RNAShape::trefoilKnot:
307  {
308  src = _trefoilKnot(radius, u, params);
309  dst = _trefoilKnot(radius, u + uStep, params);
310  break;
311  }
312  case RNAShape::heart:
313  {
314  src = _heart(radius, u);
315  dst = _heart(radius, u + uStep);
316  break;
317  }
318  case RNAShape::thing:
319  {
320  src = _thing(radius, u, params);
321  dst = _thing(radius, u + uStep, params);
322  break;
323  }
324  default:
325  PLUGIN_THROW("Undefined shape");
326  break;
327  }
328 }

330 Vector3f RNASequence::_trefoilKnot(const float radius, const float t,
331  const Vector3f& params)
332 {
333  return {radius * ((sin(t) + 2.f * sin(params.x * t))) / 3.f,
334  radius * ((cos(t) - 2.f * cos(params.y * t))) / 3.f,
335  radius * (-sin(params.z * t))};
336 }

338 Vector3f RNASequence::_torus(const float radius, const float t,
339  const Vector3f& params)
340 {
341  return {radius * (cos(t) + params.x * cos(params.y * t) * cos(t)),
342  radius * (sin(t) + params.x * cos(params.y * t) * sin(t)),
343  radius * params.x * sin(params.y * t)};
344 }

346 Vector3f RNASequence::_star(const float radius, const float t)
347 {
348  return {radius * (2.f * sin(3.f * t) * cos(t)),
349  radius * (2.f * sin(3.f * t) * sin(t)), radius * sin(3.f * t)};
350 }

352 Vector3f RNASequence::_spring(const float radius, const float t,
353  const Vector3f& params)
354 {
355  return {radius * cos(t) + (radius + params.x * cos(params.y * t)) * cos(t),
356  radius * sin(t) + (radius + params.x * cos(params.y * t)) * sin(t),
357  radius * t + params.x * sin(params.y * t)};
358 }

360 Vector3f RNASequence::_heart(const float radius, const float u)
361 {
362  return {radius * 4.f * pow(sin(u), 3.f),
363  radius * 0.25f *
364  (13.f * cos(u) - 5.f * cos(2.f * u) - 2.f * cos(3.f * u) -
365  cos(4.f * u)),
366  0.f};
367 }

369 Vector3f RNASequence::_thing(const float radius, const float t,
370  const Vector3f& params)
371 {
372  return {radius * (sin(t) + params.x * sin(params.y * t)),
373  radius * (cos(t) - params.x * cos(params.y * t)),
374  radius * (-sin(params.z * t))};
375 }

377 Vector3f RNASequence::_moebius(const float radius, const float u, const float v)
378 {
379  return {4.f * radius * (cos(u) + v * cos(u / 2.f) * cos(u)),
380  4.f * radius * (sin(u) + v * cos(u / 2.f) * sin(u)),
381  8.f * radius * (v * sin(u / 2.f))};
382 }
