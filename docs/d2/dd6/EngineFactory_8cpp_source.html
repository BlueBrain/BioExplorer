51 Engine* EngineFactory::create(const std::string& name)
52 {
53  const auto it = SUPPORTED_ENGINES.find(name);
54 
55  if (it == SUPPORTED_ENGINES.end())
56  CORE_THROW("Unsupported engine: " + name);
57 
58  const auto libraryName = (*it).second;
59  if (_engines.count(libraryName) == 0)
60  return _loadEngine(libraryName.c_str(), _argc, _argv);
61  return _engines[libraryName].get();
62 }

64 Engine* EngineFactory::_loadEngine(const std::string& name, int argc, const char* argv[])
65 {
66  try
67  {
68  DynamicLib library(name);
69  auto createSym = library.getSymbolAddress("core_engine_create");
70  if (!createSym)
71  {
72  throw std::runtime_error(std::string("Plugin '") + name + "' is not a valid Core engine; missing " +
73  "core_engine_create()");
74  }
75 
76  CreateFuncType createFunc = (CreateFuncType)createSym;
77  if (auto plugin = createFunc(argc, argv, _parametersManager))
78  {
79  _engines.emplace(name, std::unique_ptr<Engine>(plugin));
80  _libs.push_back(std::move(library));
81  CORE_INFO("Loaded engine '" << name << "'");
82  return plugin;
83  }
84  }
85  catch (const std::runtime_error& exc)
86  {
87  CORE_ERROR("Failed to load engine " << std::quoted(name) << ": " << exc.what());
88  }
89  return nullptr;
90 }
