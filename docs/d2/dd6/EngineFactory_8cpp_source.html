49 Engine* EngineFactory::create(const std::string& name)
50 {
51  const auto it = SUPPORTED_ENGINES.find(name);
52 
53  if (it == SUPPORTED_ENGINES.end())
54  CORE_THROW("Unsupported engine: " + name);
55 
56  const auto libraryName = (*it).second;
57  if (_engines.count(libraryName) == 0)
58  return _loadEngine(libraryName.c_str(), _argc, _argv);
59  return _engines[libraryName].get();
60 }

62 Engine* EngineFactory::_loadEngine(const std::string& name, int argc, const char* argv[])
63 {
64  try
65  {
66  DynamicLib library(name);
67  auto createSym = library.getSymbolAddress("core_engine_create");
68  if (!createSym)
69  {
70  throw std::runtime_error(std::string("Plugin '") + name + "' is not a valid Core engine; missing " +
71  "core_engine_create()");
72  }
73 
74  CreateFuncType createFunc = (CreateFuncType)createSym;
75  if (auto plugin = createFunc(argc, argv, _parametersManager))
76  {
77  _engines.emplace(name, std::unique_ptr<Engine>(plugin));
78  _libs.push_back(std::move(library));
79  CORE_INFO("Loaded engine '" << name << "'");
80  return plugin;
81  }
82  }
83  catch (const std::runtime_error& exc)
84  {
85  CORE_ERROR("Failed to load engine " << std::quoted(name) << ": " << exc.what());
86  }
87  return nullptr;
88 }
