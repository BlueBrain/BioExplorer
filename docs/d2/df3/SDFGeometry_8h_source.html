103 inline Boxd getSDFBoundingBox(const SDFGeometry& geom)
104 {
105  Boxd bounds;
106  switch (geom.type)
107  {
108  case core::SDFType::Sphere:
109  {
110  bounds.merge(geom.p0 - Vector3f(geom.r0));
111  bounds.merge(geom.p0 + Vector3f(geom.r0));
112  break;
113  }
114  case core::SDFType::Pill:
115  {
116  bounds.merge(geom.p0 - Vector3f(geom.r0));
117  bounds.merge(geom.p0 + Vector3f(geom.r0));
118  bounds.merge(geom.p1 - Vector3f(geom.r0));
119  bounds.merge(geom.p1 + Vector3f(geom.r0));
120  break;
121  }
122  case core::SDFType::ConePill:
123  case core::SDFType::ConePillSigmoid:
124  {
125  bounds.merge(geom.p0 - Vector3f(geom.r0));
126  bounds.merge(geom.p0 + Vector3f(geom.r0));
127  bounds.merge(geom.p1 - Vector3f(geom.r1));
128  bounds.merge(geom.p1 + Vector3f(geom.r1));
129  break;
130  }
131  default:
132  throw std::runtime_error("No bounds found for SDF type.");
133  }
134  return bounds;
135 }
