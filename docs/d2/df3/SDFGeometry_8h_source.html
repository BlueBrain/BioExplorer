146 inline Boxd getSDFBoundingBox(const SDFGeometry& geom)
147 {
148  Boxd bounds;
149  switch (geom.type)
150  {
151  case SDFType::sdf_sphere:
152  case SDFType::sdf_cut_sphere:
153  {
154  bounds.merge(geom.p0 - Vector3f(geom.r0));
155  bounds.merge(geom.p0 + Vector3f(geom.r0));
156  break;
157  }
158  case SDFType::sdf_pill:
159  case SDFType::sdf_vesica:
160  {
161  bounds.merge(geom.p0 - Vector3f(geom.r0));
162  bounds.merge(geom.p0 + Vector3f(geom.r0));
163  bounds.merge(geom.p1 - Vector3f(geom.r0));
164  bounds.merge(geom.p1 + Vector3f(geom.r0));
165  break;
166  }
167  case SDFType::sdf_cone:
168  case SDFType::sdf_cone_pill:
169  case SDFType::sdf_cone_pill_sigmoid:
170  {
171  bounds.merge(geom.p0 - Vector3f(geom.r0));
172  bounds.merge(geom.p0 + Vector3f(geom.r0));
173  bounds.merge(geom.p1 - Vector3f(geom.r1));
174  bounds.merge(geom.p1 + Vector3f(geom.r1));
175  break;
176  }
177  case SDFType::sdf_torus:
178  {
179  bounds.merge(geom.p0 - Vector3f(geom.r0 + geom.r1));
180  bounds.merge(geom.p0 + Vector3f(geom.r0 + geom.r1));
181  break;
182  }
183  default:
184  throw std::runtime_error("No bounds found for SDF type.");
185  }
186  return bounds;
187 }
