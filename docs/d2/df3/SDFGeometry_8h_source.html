159 inline Boxd getSDFBoundingBox(const SDFGeometry& geom)
160 {
161  Boxd bounds;
162  switch (geom.type)
163  {
164  case SDFType::sdf_sphere:
165  case SDFType::sdf_cut_sphere:
166  {
167  bounds.merge(geom.p0 - Vector3f(geom.r0));
168  bounds.merge(geom.p0 + Vector3f(geom.r0));
169  break;
170  }
171  case SDFType::sdf_pill:
172  case SDFType::sdf_vesica:
173  {
174  bounds.merge(geom.p0 - Vector3f(geom.r0));
175  bounds.merge(geom.p0 + Vector3f(geom.r0));
176  bounds.merge(geom.p1 - Vector3f(geom.r0));
177  bounds.merge(geom.p1 + Vector3f(geom.r0));
178  break;
179  }
180  case SDFType::sdf_cone:
181  case SDFType::sdf_cone_pill:
182  case SDFType::sdf_cone_pill_sigmoid:
183  {
184  bounds.merge(geom.p0 - Vector3f(geom.r0));
185  bounds.merge(geom.p0 + Vector3f(geom.r0));
186  bounds.merge(geom.p1 - Vector3f(geom.r1));
187  bounds.merge(geom.p1 + Vector3f(geom.r1));
188  break;
189  }
190  case SDFType::sdf_torus:
191  {
192  bounds.merge(geom.p0 - Vector3f(geom.r0 + geom.r1));
193  bounds.merge(geom.p0 + Vector3f(geom.r0 + geom.r1));
194  break;
195  }
196  case SDFType::sdf_ellipsoid:
197  {
198  bounds.merge(geom.p0 - geom.p1);
199  bounds.merge(geom.p0 + geom.p1);
200  break;
201  }
202  default:
203  throw std::runtime_error("No bounds found for SDF type.");
204  }
205  return bounds;
206 }
