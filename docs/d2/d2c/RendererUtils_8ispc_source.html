25 /**
26  Composes source and destination colors according to specified alpha correction
27  @param src Source RGBA color
28  @param dst Returned destination RGBA color
29  @param alphaCorrection Alpha correction used to blend colors
30 */
31 inline void compose(const varying vec4f& src, varying vec4f& dst, const varying float alphaCorrection)
32 {
33  const float alpha = alphaCorrection * src.w;
34  dst = make_vec4f((1.f - dst.w) * alpha * make_vec3f(src) + dst.w * make_vec3f(dst), dst.w + alpha * (1.f - dst.w));
35 }

37 /**
38  Returns the refracted vector according to the direction of the incident ray,
39  he normal to the surface, and localRefraction indices
40  @param direction Direction of the incident ray
41  @param normal Normal to the surface
42  @param n1 In localRefraction index
43  @param n2 Out localRefraction index
44  @return Refracted vector
45 */
46 inline vec3f refractedVector(const varying vec3f& direction, const varying vec3f& normal, const varying float n1,
47  const varying float n2)
48 {
49  if (n2 == 0.f)
50  return direction;
51  const float eta = n1 / n2;
52  const float cos1 = -dot(direction, normal);
53  const float cos2 = 1.f - eta * eta * (1.f - cos1 * cos1);
54  if (cos2 > 0.f)
55  return eta * direction + (eta * cos1 - sqrt(cos2)) * normal;
56  return direction;
57 }

59 /**
60  Composes source and destination colors according to specified alpha
61  correction
62  @param src Source RGBA color
63  @param dst Returned destination RGBA color
64  @param alphaCorrection Alpha correction used to blend colors
65 */
66 inline void adaptiveCompose(const varying vec4f& src, varying vec4f& dst, const varying float alphaCorrection)
67 {
68  const float alpha = 1.f - pow(1.f - min(src.w, 1.f - 1.f / 256.f), alphaCorrection);
69  const float a = alpha * (1.f - dst.w);
70  dst.x = dst.x + src.x * a;
71  dst.y = dst.y + src.y * a;
72  dst.z = dst.z + src.z * a;
73  dst.w += (alpha * (1.f - dst.w));
74 }
