60 template <typename T>
61 void textureToOptix(T* ptr_dst, const core::Texture2D& texture,
62  const uint8_t face, const uint8_t mipLevel,
63  const bool hasAlpha)
64 {
65  uint16_t width = texture.width;
66  uint16_t height = texture.height;
67  for (uint8_t i = 0; i < mipLevel; ++i)
68  {
69  width /= 2;
70  height /= 2;
71  }
72  size_t idx_src = 0;
73  size_t idx_dst = 0;
74  const auto rawData = texture.getRawData<T>(face, mipLevel);
75  for (uint16_t y = 0; y < height; ++y)
76  {
77  for (uint16_t x = 0; x < width; ++x)
78  {
79  ptr_dst[idx_dst] = rawData[idx_src];
80  ptr_dst[idx_dst + 1u] = rawData[idx_src + 1u];
81  ptr_dst[idx_dst + 2u] = rawData[idx_src + 2u];
82  ptr_dst[idx_dst + 3u] =
83  hasAlpha ? rawData[idx_src + 3u] : white<T>();
84  idx_dst += 4u;
85  idx_src += hasAlpha ? 4u : 3u;
86  }
87  }
88 }

166 ::optix::TextureSampler OptiXContext::createTextureSampler(Texture2DPtr texture)
167 {
168  uint16_t nx = texture->width;
169  uint16_t ny = texture->height;
170  const uint16_t channels = texture->channels;
171  const uint16_t optixChannels = 4;
172  const bool hasAlpha = optixChannels == channels;
173 
174  const bool useFloat = texture->depth == 4;
175  const bool useByte = texture->depth == 1;
176 
177  if (!useFloat && !useByte)
178  throw std::runtime_error("Only byte or float textures are supported");
179 
180  const bool createMipmaps =
181  texture->getMipLevels() == 1 && useByte && !texture->isCubeMap();
182  uint16_t mipMapLevels = texture->getMipLevels();
183  if (createMipmaps)
184  mipMapLevels = texture->getPossibleMipMapsLevels();
185 
186  if (createMipmaps && !useByte)
187  throw std::runtime_error(
188  "Non 8-bits textures are not supported for automatic mipmaps "
189  "generation");
190 
191  RTformat optixFormat =
192  useByte ? RT_FORMAT_UNSIGNED_BYTE4 : RT_FORMAT_FLOAT4;
193 
194  // Create texture sampler
195  ::optix::TextureSampler sampler = _optixContext->createTextureSampler();
196  const auto wrapMode = wrapModeToOptix(texture->getWrapMode());
197  sampler->setWrapMode(0, wrapMode);
198  sampler->setWrapMode(1, wrapMode);
199  sampler->setWrapMode(2, wrapMode);
200  sampler->setIndexingMode(RT_TEXTURE_INDEX_NORMALIZED_COORDINATES);
201  sampler->setReadMode(RT_TEXTURE_READ_NORMALIZED_FLOAT);
202  sampler->setMaxAnisotropy(8.0f);
203 
204  // Create buffer and populate with texture data
205  optix::Buffer buffer;
206  if (texture->isCubeMap())
207  buffer = _optixContext->createCubeBuffer(RT_BUFFER_INPUT, optixFormat,
208  nx, ny, mipMapLevels);
209  else
210  buffer =
211  _optixContext->createMipmappedBuffer(RT_BUFFER_INPUT, optixFormat,
212  nx, ny, mipMapLevels);
213 
214  std::vector<void*> mipMapBuffers(mipMapLevels);
215  for (uint8_t currentLevel = 0u; currentLevel < mipMapLevels; ++currentLevel)
216  mipMapBuffers[currentLevel] = buffer->map(currentLevel);
217 
218  if (createMipmaps)
219  {
220  uint8_t* ptr_dst = (uint8_t*)mipMapBuffers[0];
221  size_t idx_src = 0;
222  size_t idx_dst = 0;
223  const auto rawData = texture->getRawData<unsigned char>();
224  for (uint16_t y = 0; y < ny; ++y)
225  {
226  for (uint16_t x = 0; x < nx; ++x)
227  {
228  ptr_dst[idx_dst] = rawData[idx_src];
229  ptr_dst[idx_dst + 1u] = rawData[idx_src + 1u];
230  ptr_dst[idx_dst + 2u] = rawData[idx_src + 2u];
231  ptr_dst[idx_dst + 3u] = hasAlpha ? rawData[idx_src + 3u] : 255u;
232  idx_dst += 4u;
233  idx_src += hasAlpha ? 4u : 3u;
234  }
235  }
236  ny /= 2u;
237  nx /= 2u;
238 
239  for (uint8_t currentLevel = 1u; currentLevel < mipMapLevels;
240  ++currentLevel)
241  {
242  ptr_dst = (uint8_t*)mipMapBuffers[currentLevel];
243  uint8_t* ptr_src = (uint8_t*)mipMapBuffers[currentLevel - 1u];
244  for (uint16_t y = 0u; y < ny; ++y)
245  {
246  for (uint16_t x = 0u; x < nx; ++x)
247  {
248  ptr_dst[(y * nx + x) * 4u] =
249  (ptr_src[(y * 2u * nx + x) * 8u] +
250  ptr_src[((y * 2u * nx + x) * 2u + 1u) * 4u] +
251  ptr_src[((y * 2u + 1u) * nx + x) * 8u] +
252  ptr_src[(((y * 2u + 1u) * nx + x) * 2u + 1u) * 4u]) /
253  4.0f;
254  ptr_dst[(y * nx + x) * 4u + 1u] =
255  (ptr_src[(y * 2u * nx + x) * 8u + 1u] +
256  ptr_src[((y * 2u * nx + x) * 2u + 1u) * 4u + 1u] +
257  ptr_src[((y * 2u + 1u) * nx + x) * 8u + 1u] +
258  ptr_src[(((y * 2u + 1u) * nx + x) * 2u + 1u) * 4u +
259  1u]) /
260  4.0f;
261  ptr_dst[(y * nx + x) * 4u + 2u] =
262  (ptr_src[(y * 2u * nx + x) * 8u + 2u] +
263  ptr_src[((y * 2u * nx + x) * 2u + 1u) * 4u + 2u] +
264  ptr_src[((y * 2u + 1u) * nx + x) * 8u + 2u] +
265  ptr_src[(((y * 2u + 1u) * nx + x) * 2u + 1u) * 4u +
266  2u]) /
267  4.0f;
268  ptr_dst[(y * nx + x) * 4u + 3u] =
269  (ptr_src[(y * 2u * nx + x) * 8u + 3u] +
270  ptr_src[((y * 2u * nx + x) * 2u + 1u) * 4u + 3u] +
271  ptr_src[((y * 2u + 1u) * nx + x) * 8u + 3u] +
272  ptr_src[(((y * 2u + 1u) * nx + x) * 2u + 1u) * 4u +
273  3u]) /
274  4.0f;
275 
276  if (texture->isNormalMap())
277  {
278  glm::vec3 normalized = glm::normalize(glm::vec3(
279  2.0f * (float)ptr_dst[(y * nx + x) * 4u] / 255.0f -
280  1.0f,
281  2.0f * (float)ptr_dst[(y * nx + x) * 4u + 1u] /
282  255.0f -
283  1.0f,
284  2.0f * (float)ptr_dst[(y * nx + x) * 4u + 2u] /
285  255.0f -
286  1.0f));
287  ptr_dst[(y * nx + x) * 4u] =
288  255.0f * (0.5f * normalized.x + 0.5f);
289  ptr_dst[(y * nx + x) * 4u + 1u] =
290  255.0f * (0.5f * normalized.y + 0.5f);
291  ptr_dst[(y * nx + x) * 4u + 2u] =
292  255.0f * (0.5f * normalized.z + 0.5f);
293  }
294  }
295  }
296  ny /= 2u;
297  nx /= 2u;
298  }
299  }
300  else
301  {
302  for (uint8_t face = 0; face < texture->getNumFaces(); ++face)
303  {
304  auto mipWidth = nx;
305  auto mipHeight = ny;
306  for (uint16_t mip = 0; mip < mipMapLevels; ++mip)
307  {
308  if (useByte)
309  {
310  auto dst = (uint8_t*)mipMapBuffers[mip];
311  dst += face * mipWidth * mipHeight * 4;
312  textureToOptix<uint8_t>(dst, *texture, face, mip, hasAlpha);
313  }
314  else if (useFloat)
315  {
316  auto dst = (float*)mipMapBuffers[mip];
317  dst += face * mipWidth * mipHeight * 4;
318  textureToOptix<float>(dst, *texture, face, mip, hasAlpha);
319  }
320  mipWidth /= 2;
321  mipHeight /= 2;
322  }
323  }
324  }
325 
326  for (uint8_t currentLevel = 0u; currentLevel < mipMapLevels; ++currentLevel)
327  buffer->unmap(currentLevel);
328 
329  // Assign buffer to sampler
330  sampler->setBuffer(buffer);
331  sampler->setFilteringModes(RT_FILTER_LINEAR, RT_FILTER_LINEAR,
332  mipMapLevels > 1 ? RT_FILTER_LINEAR
333  : RT_FILTER_NONE);
334  sampler->validate();
335  return sampler;
336 }

572  _programGroups.data(), static_cast<unsigned int>(_programGroups.size()), log,
573  &sizeof_log, &_state.pipeline));
574 
575  OptixStackSizes stack_sizes = {};

579  uint32_t direct_callable_stack_size_from_traversal;
580  uint32_t direct_callable_stack_size_from_state;

584  &direct_callable_stack_size_from_state, &continuation_stack_size));

586  direct_callable_stack_size_from_state, continuation_stack_size,
