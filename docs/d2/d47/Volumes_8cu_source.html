47 /**
48 A smart way to avoid recursion restrictions with OptiX 6 is to use templates!
49 
50 https://www.thanassis.space/cudarenderer-BVH.html#recursion
51 */
52 #define MAX_RECURSION_DEPTH 30
53 
54 template <int depth>
55 __device__ float treeWalker(const int volumeOctreeIndicesId, const int volumeOctreeValuesId, const float3& point,
56  const float distance, const float cutoff, const uint index)
57 {
58  if (depth >= MAX_RECURSION_DEPTH)
59  return 0.f;
60 
61  const uint begin = optix::rtTex1D<uint32_t>(volumeOctreeIndicesId, index * 2);
62  const uint end = optix::rtTex1D<uint32_t>(volumeOctreeIndicesId, index * 2 + 1);
63  const uint idxData = index * FIELD_POINT_DATA_SIZE;
64 
65  if (begin == 0 && end == 0)
66  {
67  // Leaf
68  const float value = optix::rtTex1D<float>(volumeOctreeValuesId, idxData + FIELD_POINT_OFFSET_VALUE);
69  return value / (distance * distance);
70  }
71 
72  float voxelValue = 0.f;
73  for (uint childIndex = begin; childIndex <= end; ++childIndex)
74  {
75  const uint idx = childIndex * FIELD_POINT_DATA_SIZE;
76  const float3 childPosition =
77  make_float3(optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_POINT_OFFSET_POSITION_X),
78  optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_POINT_OFFSET_POSITION_Y),
79  optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_POINT_OFFSET_POSITION_Z));
80  const float d = length(point - childPosition);
81 
82  if (d >= cutoff)
83  {
84  // Child is further than the cutoff distance, no need to evaluate events in the child node, we take the
85  // precomputed value of node instead
86  const float value = optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_POINT_OFFSET_VALUE);
87  voxelValue += value / (d * d);
88  }
89  else
90  // Dive into the child node and compute its contents
91  voxelValue +=
92  treeWalker<depth + 1>(volumeOctreeIndicesId, volumeOctreeValuesId, point, d, cutoff / 2.f, childIndex);
93  }
94  return voxelValue;
95 }

105 template <int depth>
106 __device__ float3 treeWalker3(const int volumeOctreeIndicesId, const int volumeOctreeValuesId, const float3& point,
107  const float distance, const float cutoff, const uint index)
108 {
109  if (depth >= MAX_RECURSION_DEPTH)
110  return make_float3(0.f);
111 
112  const uint begin = optix::rtTex1D<uint32_t>(volumeOctreeIndicesId, index * 2);
113  const uint end = optix::rtTex1D<uint32_t>(volumeOctreeIndicesId, index * 2 + 1);
114  const uint idxData = index * FIELD_VECTOR_DATA_SIZE;
115 
116  if (begin == 0 && end == 0)
117  {
118  // Leaf
119  const float3 vectorDirection =
120  make_float3(optix::rtTex1D<float>(volumeOctreeValuesId, idxData + FIELD_VECTOR_OFFSET_DIRECTION_X),
121  optix::rtTex1D<float>(volumeOctreeValuesId, idxData + FIELD_VECTOR_OFFSET_DIRECTION_Y),
122  optix::rtTex1D<float>(volumeOctreeValuesId, idxData + FIELD_VECTOR_OFFSET_DIRECTION_Z));
123  return vectorDirection / (distance * distance);
124  }
125 
126  float3 voxelValue = make_float3(0.f);
127  for (uint childIndex = begin; childIndex <= end; ++childIndex)
128  {
129  const uint idx = childIndex * FIELD_VECTOR_DATA_SIZE;
130  const float3 childPosition =
131  make_float3(optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_VECTOR_OFFSET_POSITION_X),
132  optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_VECTOR_OFFSET_POSITION_Y),
133  optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_VECTOR_OFFSET_POSITION_Z));
134  const float d = length(point - childPosition);
135 
136  if (d >= cutoff)
137  {
138  // Child is further than the cutoff distance, no need to evaluate events in the child node, we take the
139  // precomputed value of node instead
140  const float3 vectorDirection =
141  make_float3(optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_VECTOR_OFFSET_DIRECTION_X),
142  optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_VECTOR_OFFSET_DIRECTION_Y),
143  optix::rtTex1D<float>(volumeOctreeValuesId, idx + FIELD_VECTOR_OFFSET_DIRECTION_Z));
144  voxelValue += vectorDirection / (d * d);
145  }
146  else
147  // Dive into the child node and compute its contents
148  voxelValue +=
149  treeWalker3<depth + 1>(volumeOctreeIndicesId, volumeOctreeValuesId, point, d, cutoff / 2.f, childIndex);
150  }
151  return voxelValue;
152 }

162 static __device__ float4 get_voxel_value(const int idx, const float3& point)
163 {
164  const int volumeSamplerId = static_cast<int>(volumes[idx + OFFSET_VOLUME_TEXTURE_SAMPLER_ID]);
165  if (volumeSamplerId != 0)
166  return make_float4(0.f, 0.f, 0.f, optix::rtTex3D<float>(volumeSamplerId, point.x, point.y, point.z));
167 
168  const int volumeOctreeIndicesId = static_cast<int>(volumes[idx + OFFSET_OCTREE_INDICES_SAMPLER_ID]);
169  const int volumeOctreeValuesId = static_cast<int>(volumes[idx + OFFSET_OCTREE_VALUES_SAMPLER_ID]);
170  const int volumeOctreeType = static_cast<int>(volumes[idx + OFFSET_OCTREE_TYPE]);
171  if (volumeOctreeIndicesId != 0 && volumeOctreeValuesId != 0)
172  {
173  const float distance = volumeUserParameters.x;
174  const float cutoff = volumeUserParameters.y;
175  switch (volumeOctreeType)
176  {
177  case OctreeDataType::point:
178  return make_float4(0.f, 0.f, 0.f,
179  treeWalker<0>(volumeOctreeIndicesId, volumeOctreeValuesId, point, distance, cutoff, 0u));
180  case OctreeDataType::vector:
181  const float3 sampleValue =
182  treeWalker3<0>(volumeOctreeIndicesId, volumeOctreeValuesId, point, distance, cutoff, 0u);
183  return make_float4(normalize(sampleValue), length(sampleValue));
184  }
185  }
186  return make_float4(0.f);
187 }

189 template <bool use_robust_method>
190 static __device__ void intersect_volume(int primIdx)
191 {
192  const int idx = primIdx * volume_size;
193  const float3 dimensions = {volumes[idx + OFFSET_DIMENSIONS], volumes[idx + OFFSET_DIMENSIONS + 1],
194  volumes[idx + OFFSET_DIMENSIONS + 2]};
195  const float3 offset = {volumes[idx + OFFSET_OFFSET], volumes[idx + OFFSET_OFFSET + 1],
196  volumes[idx + OFFSET_OFFSET + 2]};
197  const float3 spacing = {volumes[idx + OFFSET_SPACING], volumes[idx + OFFSET_SPACING + 1],
198  volumes[idx + OFFSET_SPACING + 2]};
199  const int transferFunctionSamplerId = static_cast<int>(volumes[idx + OFFSET_TRANSFER_FUNCTION_TEXTURE_SAMPLER_ID]);
200  const float2 valueRange = {volumes[idx + OFFSET_VALUE_RANGE], volumes[idx + OFFSET_VALUE_RANGE + 1]};
201 
202  const float3 boxMin = offset;
203  const float3 boxMax = offset + dimensions * spacing;
204 
205  const float3 a = (boxMin - ray.origin) / ray.direction;
206  const float3 b = (boxMax - ray.origin) / ray.direction;
207  const float3 near = fminf(a, b);
208  const float3 far = fmaxf(a, b);
209  float t0 = fmaxf(near);
210  float t1 = fminf(far);
211 
212  const ::optix::size_t2 screen = output_buffer.size();
213  uint seed = tea<16>(screen.x * launch_index.y + launch_index.x, frame);
214 
215  const float diag = max(spacing.x, max(spacing.y, spacing.z));
216  const float step = max(0.1f, diag / volumeSamplingRate);
217  const float random = rnd(seed) * step;
218 
219  // Apply ray clipping
220  t0 = max(t0, ray.tmin);
221  t1 = min(t1, ray.tmax);
222 
223  if (t0 > 0.f && t0 <= t1)
224  {
225  float t = t0 + random;
226  while (t < t1)
227  {
228  const float3 p = ray.origin + t * ray.direction;
229  const float3 p0 = (p - offset) / spacing;
230  const float4 voxelValue = get_voxel_value(idx, p0);
231  const float4 voxelColor = calcTransferFunctionColor(transferFunctionSamplerId, valueRange, voxelValue.w);
232  if (voxelColor.w > 0.f)
233  if (rtPotentialIntersection(t - sceneEpsilon))
234  {
235  float3 normal = make_float3(voxelValue);
236  if (volumeGradientShadingEnabled)
237  {
238  normal = make_float3(0);
239  const float3 positions[6] = {{-1, 0, 0}, {1, 0, 0}, {0, -1, 0},
240  {0, 1, 0}, {0, 0, -1}, {0, 0, 1}};
241  for (const auto& position : positions)
242  {
243  const float3 p1 = p0 + (position * volumeGradientOffset);
244  const float4 voxelValue = get_voxel_value(idx, p1);
245  normal += voxelValue.w * position;
246  }
247  normal = ::optix::normalize(-1.f * normal);
248  }
249 
250  geometric_normal = shading_normal = normal;
251  userDataIndex = 0;
252  texcoord = make_float2(voxelValue.w, 0.f);
253  texcoord3d = p0;
254  rtReportIntersection(0);
255  break;
256  }
257  t += step;
258  }
259  }
260 }
