37 SpikeSimulationHandler::SpikeSimulationHandler(const std::string& populationName, const uint64_t simulationReportId)
38  : core::AbstractSimulationHandler()
39  , _populationName(populationName)
40  , _simulationReportId(simulationReportId)
41 {
42  const auto& connector = DBConnector::getInstance();
43  _simulationReport = connector.getSimulationReport(_populationName, _simulationReportId);
44  _nbFrames = (_simulationReport.endTime - _simulationReport.startTime) / _simulationReport.timeStep;
45  _dt = _simulationReport.timeStep;
46 
47  const auto spikes = connector.getNeuronSpikeReportValues(_populationName, _simulationReportId,
48  _simulationReport.startTime, _simulationReport.endTime);
49 
50  uint64_t i = 0;
51  for (const auto spike : spikes)
52  {
53  _guidsMapping[spike.first] = i;
54  ++i;
55  }
56  _frameSize = spikes.size();
57  _frameData.resize(_frameSize, _restVoltage);
58 
59  PLUGIN_INFO(1, "---------------------------------------------------------");
60  PLUGIN_INFO(1, "Spike simulation information");
61  PLUGIN_INFO(1, "----------------------------");
62  PLUGIN_INFO(1, "Population name : " << _populationName);
63  PLUGIN_INFO(1, "Number of simulated nodes: " << _frameSize);
64  PLUGIN_INFO(1, "Start time : " << _simulationReport.startTime);
65  PLUGIN_INFO(1, "End time : " << _simulationReport.endTime);
66  PLUGIN_INFO(1, "Number of frames : " << _nbFrames);
67  _logVisualizationSettings();
68  PLUGIN_INFO(1, "---------------------------------------------------------");
69 }

79 void* SpikeSimulationHandler::getFrameData(const uint32_t frame)
80 {
81  const auto& connector = DBConnector::getInstance();
82  const auto boundedFrame = _getBoundedFrame(frame);
83 
84  if (_currentFrame != boundedFrame)
85  {
86  const float ts = _simulationReport.startTime + boundedFrame * _dt;
87  const float startTime = ts - (_spikingVoltage - _restVoltage) / _decaySpeed;
88  const float endTime = _simulationReport.endTime;
89  const auto spikes = connector.getNeuronSpikeReportValues(_populationName, _simulationReportId,
90  std::max(0.f, startTime), std::min(ts, endTime));
91 
92  // Rest all values
93  for (uint64_t i = 0; i < _frameSize; ++i)
94  _frameData[i] = _restVoltage;
95 
96  // Update values of nodes spiking within the specified time range
97  for (const auto spike : spikes)
98  _frameData[_guidsMapping[spike.first]] =
99  std::max(_restVoltage, _spikingVoltage - _decaySpeed * (ts - spike.second));
100 
101  _currentFrame = boundedFrame;
102  }
103 
104  return _frameData.data();
105 }
