43 SpikeSimulationHandler::SpikeSimulationHandler(
44  const std::string& populationName, const uint64_t simulationReportId)
45  : brayns::AbstractSimulationHandler()
46  , _populationName(populationName)
47  , _simulationReportId(simulationReportId)
48 {
49  const auto& connector = DBConnector::getInstance();
50  _simulationReport =
51  connector.getSimulationReport(_populationName, _simulationReportId);
52  _nbFrames = (_simulationReport.endTime - _simulationReport.startTime) /
53  _simulationReport.timeStep;
54  _dt = _simulationReport.timeStep;
55 
56  const auto spikes = connector.getNeuronSpikeReportValues(
57  _populationName, _simulationReportId, _simulationReport.startTime,
58  _simulationReport.endTime + _simulationReport.timeStep);
59 
60  uint64_t i = 0;
61  for (const auto spike : spikes)
62  {
63  _guidsMapping[spike] = i;
64  ++i;
65  }
66  _frameSize = spikes.size();
67  _frameData.resize(_frameSize, DEFAULT_REST_VALUE);
68 
69  PLUGIN_INFO(1, "---------------------------------------------------------");
70  PLUGIN_INFO(1, "Spike simulation information");
71  PLUGIN_INFO(1, "----------------------------");
72  PLUGIN_INFO(1, "Population name : " << _populationName);
73  PLUGIN_INFO(1, "Number of simulated nodes: " << _frameSize);
74  PLUGIN_INFO(1,
75  "Start time : " << _simulationReport.startTime);
76  PLUGIN_INFO(1, "End time : " << _simulationReport.endTime);
77  PLUGIN_INFO(1, "Time interval : " << _simulationReport.timeStep);
78  PLUGIN_INFO(1, "Decay speed : " << DEFAULT_DECAY_SPEED);
79  PLUGIN_INFO(1, "Number of frames : " << _nbFrames);
80  PLUGIN_INFO(1, "---------------------------------------------------------");
81 }

92 void* SpikeSimulationHandler::getFrameData(const uint32_t frame)
93 {
94  const auto& connector = DBConnector::getInstance();
95  const auto boundedFrame = _getBoundedFrame(frame);
96  if (_currentFrame != boundedFrame)
97  {
98  for (size_t i = 0; i < _frameSize; ++i)
99  {
100  if (_frameData[i] > DEFAULT_REST_VALUE)
101  _frameData[i] -= DEFAULT_DECAY_SPEED;
102  else
103  _frameData[i] = DEFAULT_REST_VALUE;
104  }
105 
106  const double ts = _simulationReport.startTime + boundedFrame * _dt;
107  const double endTime = _simulationReport.endTime - _dt;
108  const auto spikes =
109  connector.getNeuronSpikeReportValues(_populationName,
110  _simulationReportId,
111  std::min(ts, endTime),
112  std::min(ts + _dt, endTime));
113 
114  for (const auto spike : spikes)
115  _frameData[_guidsMapping[spike]] = DEFAULT_SPIKING_VALUE;
116 
117  _currentFrame = boundedFrame;
118  }
119 
120  return _frameData.data();
121 }
