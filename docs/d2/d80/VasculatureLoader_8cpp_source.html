72 ModelDescriptorPtr VasculatureLoader::importFromStorage(const std::string& storage, const LoaderProgress& callback,
73  const PropertyMap& properties) const
74 {
75  PropertyMap props = _defaults;
76  props.merge(properties);
77 
78  details::VasculatureDetails details;
79  const auto baseName = std::filesystem::path(storage).filename();
80  details.assemblyName = baseName;
81  details.populationName = baseName;
82  details.sqlFilter = props.getProperty<std::string>(LOADER_PROPERTY_DATABASE_SQL_FILTER.name);
83  details.radiusMultiplier = props.getProperty<double>(LOADER_PROPERTY_RADIUS_MULTIPLIER.name);
84  details.colorScheme = stringToEnum<details::VasculatureColorScheme>(
85  props.getProperty<std::string>(LOADER_PROPERTY_VASCULATURE_COLOR_SCHEME.name));
86  details.realismLevel = 0;
87  details.realismLevel += (props.getProperty<bool>(LOADER_PROPERTY_VASCULATURE_REALISM_LEVEL_SECTIONS.name)
88  ? static_cast<int64_t>(details::VasculatureRealismLevel::section)
89  : 0);
90  details.realismLevel += (props.getProperty<bool>(LOADER_PROPERTY_VASCULATURE_REALISM_LEVEL_BIFURCATIONS.name)
91  ? static_cast<int64_t>(details::VasculatureRealismLevel::bifurcation)
92  : 0);
93  details.representation = stringToEnum<details::VasculatureRepresentation>(
94  props.getProperty<std::string>(LOADER_PROPERTY_VASCULATURE_REPRESENTATION.name));
95  details.alignToGrid = props.getProperty<double>(LOADER_PROPERTY_ALIGN_TO_GRID.name);
96  const auto position = props.getProperty<std::array<double, 3>>(LOADER_PROPERTY_POSITION.name);
97  const Vector3d pos = core::Vector3d(position[0], position[1], position[2]);
98  const auto rotation = props.getProperty<std::array<double, 4>>(LOADER_PROPERTY_ROTATION.name);
99  const Quaterniond rot = core::Quaterniond(rotation[3], rotation[0], rotation[1], rotation[2]);
100  const auto scale = props.getProperty<std::array<double, 3>>(LOADER_PROPERTY_SCALE.name);
101  details.scale = {scale[0], scale[1], scale[2]};
102  Vasculature vasculature(_scene, details, pos, rot, callback);
103  return std::move(vasculature.getModelDescriptor());
104 }
