53 void Vasculature::_applyPaletteToModel(Model& model, const doubles& palette)
54 {
55  if (!palette.empty())
56  {
57  auto& materials = model.getMaterials();
58  if (palette.size() / 3 < materials.size())
59  // Note that the size of the palette can be greater than the number
60  // of materials, depending on how optimized the model is. When being
61  // constructed, some parts of the model might be removed because
62  // they are so small that they become invisible
63  PLUGIN_THROW("Invalid palette size. Expected " +
64  std::to_string(materials.size()) +
65  ", provided: " + std::to_string(palette.size() / 3));
66  uint64_t i = 0;
67  for (auto material : materials)
68  {
69  const Vector3f color{palette[i], palette[i + 1], palette[i + 2]};
70  material.second->setDiffuseColor(color);
71  material.second->setSpecularColor(color);
72  i += 3;
73  }
74  }
75 }

77 void Vasculature::_buildGraphModel(Model& model,
78  const VasculatureColorSchemeDetails& details)
79 {
80  const double radiusMultiplier = _details.radiusMultiplier;
81  size_t materialId = 0;
82 
83  auto& connector = DBConnector::getInstance();
84 
85  PLUGIN_INFO(1, "Loading sections...");
86  const auto sections =
87  connector.getVasculatureSections(_details.populationName,
88  _details.sqlFilter);
89  _nbSections = sections.size();
90 
91  ThreadSafeContainers containers;
92  uint64_t counter = 0;
93  uint64_t index;
94 #pragma omp parallel for
95  for (index = 0; index < _nbSections; ++index)
96  {
97  const auto sectionId = sections[index];
98 
99  const auto filter = "section_guid=" + std::to_string(sectionId);
100  const auto nodes =
101  connector.getVasculatureNodes(_details.populationName, filter);
102 
103  if (nodes.size() < 2)
104  continue;
105 
106  ThreadSafeContainer container(model, _details.useSdf,
107  doublesToVector3d(_details.scale));
108 
109  const auto& srcNode = nodes.begin()->second;
110  const auto& dstNode = (--nodes.end())->second;
111 
112  switch (details.colorScheme)
113  {
114  case VasculatureColorScheme::section:
115  materialId = dstNode.sectionId;
116  break;
117  case VasculatureColorScheme::subgraph:
118  materialId = dstNode.graphId;
119  break;
120  case VasculatureColorScheme::pair:
121  materialId = dstNode.pairId;
122  break;
123  case VasculatureColorScheme::entry_node:
124  materialId = dstNode.entryNodeId;
125  break;
126  }
127 
128  const auto& src = srcNode.position;
129  const auto& dst = dstNode.position;
130  const auto userData = nodes.begin()->first;
131  const auto direction = dst - src;
132  const auto maxRadius = std::max(srcNode.radius, dstNode.radius);
133  const float radius =
134  std::min(length(direction) / 5.0, maxRadius * radiusMultiplier);
135  container.addSphere(src, radius * 0.2, materialId, userData);
136  container.addCone(src, radius * 0.2, Vector3f(src + direction * 0.79),
137  radius * 0.2, materialId, userData);
138  container.addCone(dst, 0.0, Vector3f(src + direction * 0.8), radius,
139  materialId, userData);
140  container.addCone(Vector3f(src + direction * 0.8), radius,
141  Vector3f(src + direction * 0.79), radius * 0.2,
142  materialId, userData);
143 #pragma omp critical
144  containers.push_back(container);
145 
146 #pragma omp critical
147  ++counter;
148 
149 #pragma omp critical
150  _nbNodes += nodes.size();
151 
152 #pragma omp critical
153  PLUGIN_PROGRESS("Loading " << _nbSections << " sections", counter,
154  _nbSections);
155  }
156 
157  for (size_t i = 0; i < containers.size(); ++i)
158  {
159  const float progress = 1.f + i;
160  PLUGIN_PROGRESS("- Compiling 3D geometry...", progress,
161  containers.size());
162  auto& container = containers[i];
163  container.commitToModel();
164  }
165 }

167 void Vasculature::_buildSimpleModel(
168  Model& model, const VasculatureColorSchemeDetails& details,
169  const doubles& radii)
170 {
171  const auto radiusMultiplier = _details.radiusMultiplier;
172  const auto useSdf = _details.useSdf;
173 
174  auto& connector = DBConnector::getInstance();
175 
176  PLUGIN_INFO(1, "Loading sections...");
177  const auto sections =
178  connector.getVasculatureSections(_details.populationName,
179  _details.sqlFilter);
180  _nbSections = sections.size();
181 
182  ThreadSafeContainers containers;
183  uint64_t counter = 0;
184  uint64_t index;
185 #pragma omp parallel
186  for (index = 0; index < _nbSections; ++index)
187  {
188  const auto sectionId = sections[index];
189 
190  const auto filter = "section_guid=" + std::to_string(sectionId);
191  const auto nodes =
192  connector.getVasculatureNodes(_details.populationName, filter);
193 
194  if (nodes.size() < 2)
195  continue;
196 
197  ThreadSafeContainer container(model, _details.useSdf,
198  doublesToVector3d(_details.scale));
199 
200  uint64_t geometryIndex = 0;
201  size_t materialId = 0;
202  const auto srcNode = nodes.begin();
203  const auto userData = srcNode->first;
204 
205  switch (details.colorScheme)
206  {
207  case VasculatureColorScheme::section:
208  materialId = srcNode->second.sectionId;
209  break;
210  case VasculatureColorScheme::subgraph:
211  materialId = srcNode->second.graphId;
212  break;
213  case VasculatureColorScheme::pair:
214  materialId = srcNode->second.pairId;
215  break;
216  case VasculatureColorScheme::entry_node:
217  materialId = srcNode->second.entryNodeId;
218  break;
219  }
220 
221 #pragma omp critical
222  _subGraphs.insert(srcNode->second.graphId);
223 #pragma omp critical
224  _pairs.insert(srcNode->second.pairId);
225 #pragma omp critical
226  _entryNodes.insert(srcNode->second.entryNodeId);
227 
228  const auto& srcPoint = srcNode->second.position;
229  const auto srcRadius =
230  (userData < radii.size() ? radii[userData]
231  : srcNode->second.radius) *
232  radiusMultiplier;
233 
234  const auto& dstNode = (--nodes.end())->second;
235  const auto& dstPoint = dstNode.position;
236  const auto dstRadius =
237  (userData < radii.size() ? radii[userData] : dstNode.radius) *
238  radiusMultiplier;
239 
240  if (!useSdf)
241  container.addSphere(dstPoint, dstRadius, materialId, userData);
242 
243  container.addCone(dstPoint, dstRadius, srcPoint, srcRadius, materialId,
244  userData, {},
245  Vector3f(segmentDisplacementStrength,
246  segmentDisplacementFrequency, 0.f));
247 #pragma omp critical
248  containers.push_back(container);
249 
250 #pragma omp critical
251  ++counter;
252 
253 #pragma omp critical
254  _nbNodes += nodes.size();
255 
256 #pragma omp critical
257  PLUGIN_PROGRESS("Loading " << _nbSections << " sections", counter,
258  _nbSections);
259  }
260  PLUGIN_INFO(1, "");
261 
262  for (size_t i = 0; i < containers.size(); ++i)
263  {
264  PLUGIN_PROGRESS("- Compiling 3D geometry...", 1 + i, containers.size());
265  auto& container = containers[i];
266  container.commitToModel();
267  }
268  PLUGIN_INFO(1, "");
269 
270  _applyPaletteToModel(model, details.palette);
271 
272  PLUGIN_ERROR("Created " + std::to_string(model.getMaterials().size()) +
273  " materials");
274 }

276 void Vasculature::_buildAdvancedModel(
277  Model& model, const VasculatureColorSchemeDetails& details,
278  const doubles& radii)
279 {
280  const auto radiusMultiplier = _details.radiusMultiplier;
281  const auto useSdf = _details.useSdf;
282 
283  auto& connector = DBConnector::getInstance();
284 
285  PLUGIN_INFO(1, "Loading sections...");
286  const auto sections =
287  connector.getVasculatureSections(_details.populationName,
288  _details.sqlFilter);
289  _nbSections = sections.size();
290 
291  ThreadSafeContainers containers;
292  uint64_t counter = 0;
293  uint64_t index;
294 #pragma omp parallel
295  for (index = 0; index < _nbSections; ++index)
296  {
297  const auto sectionId = sections[index];
298 
299  const auto filter = "section_guid=" + std::to_string(sectionId);
300  const auto nodes =
301  connector.getVasculatureNodes(_details.populationName, filter);
302 
303  if (nodes.size() < 2)
304  continue;
305 
306  ThreadSafeContainer container(model, _details.useSdf,
307  doublesToVector3d(_details.scale));
308 
309  uint64_t geometryIndex = 0;
310  size_t materialId = 0;
311  Neighbours neighbours{};
312 
313  uint64_t i = 0;
314  GeometryNode dstNode;
315 
316  for (const auto& node : nodes)
317  {
318  const auto& srcNode = node.second;
319  const auto userData = node.first;
320 
321 #pragma omp critical
322  _subGraphs.insert(srcNode.graphId);
323 #pragma omp critical
324  _pairs.insert(srcNode.pairId);
325 #pragma omp critical
326  _entryNodes.insert(srcNode.entryNodeId);
327 
328  switch (details.colorScheme)
329  {
330  case VasculatureColorScheme::section:
331  materialId = srcNode.sectionId;
332  break;
333  case VasculatureColorScheme::subgraph:
334  materialId = srcNode.graphId;
335  break;
336  case VasculatureColorScheme::node:
337  materialId = userData;
338  break;
339  case VasculatureColorScheme::pair:
340  materialId = srcNode.pairId;
341  break;
342  case VasculatureColorScheme::entry_node:
343  materialId = srcNode.entryNodeId;
344  break;
345  case VasculatureColorScheme::section_gradient:
346  materialId = nodes.size() * (double(i) / double(nodes.size()));
347  break;
348  }
349 
350  const auto& srcPoint = srcNode.position;
351  const auto srcRadius =
352  (userData < radii.size() ? radii[userData] : srcNode.radius) *
353  radiusMultiplier;
354  const auto sectionId = srcNode.sectionId;
355 
356  if (i == 0 && !useSdf)
357  container.addSphere(srcPoint, srcRadius, materialId, userData);
358 
359  if (i > 0)
360  {
361  const auto dstPoint = dstNode.position;
362  const double dstRadius =
363  (userData < radii.size() ? radii[userData]
364  : dstNode.radius) *
365  radiusMultiplier;
366 
367  if (!useSdf)
368  container.addSphere(dstPoint, dstRadius, materialId,
369  userData);
370 
371  geometryIndex =
372  container.addCone(dstPoint, dstRadius, srcPoint, srcRadius,
373  materialId, userData, neighbours,
374  Vector3f(segmentDisplacementStrength,
375  segmentDisplacementFrequency,
376  0.f));
377  neighbours = {geometryIndex};
378  }
379 
380  dstNode = srcNode;
381  ++i;
382  }
383 #pragma omp critical
384  containers.push_back(container);
385 
386 #pragma omp critical
387  ++counter;
388 
389 #pragma omp critical
390  _nbNodes += nodes.size();
391 
392 #pragma omp critical
393  PLUGIN_PROGRESS("Loading " << _nbSections << " sections", counter,
394  _nbSections);
395  }
396  PLUGIN_INFO(1, "");
397 
398  for (size_t i = 0; i < containers.size(); ++i)
399  {
400  PLUGIN_PROGRESS("- Compiling 3D geometry...", 1 + i, containers.size());
401  auto& container = containers[i];
402  container.commitToModel();
403  }
404  PLUGIN_INFO(1, "");
405 
406  _applyPaletteToModel(model, details.palette);
407 
408  PLUGIN_ERROR("Created " + std::to_string(model.getMaterials().size()) +
409  " materials");
410 }

412 void Vasculature::_buildModel(const VasculatureColorSchemeDetails& details,
413  const doubles& radii)
414 {
415  if (_modelDescriptor)
416  _scene.removeModel(_modelDescriptor->getModelID());
417 
418  auto model = _scene.createModel();
419  switch (_details.quality)
420  {
421  case VasculatureQuality::low:
422  _buildGraphModel(*model, details);
423  break;
424  case VasculatureQuality::medium:
425  _buildSimpleModel(*model, details, radii);
426  break;
427  default:
428  _buildAdvancedModel(*model, details, radii);
429  break;
430  }
431 
432  ModelMetadata metadata = {{"Number of nodes", std::to_string(_nbNodes)},
433  {"Number of sections",
434  std::to_string(_nbSections)},
435  {"Number of sub-graphs",
436  std::to_string(_subGraphs.size())}};
437 
438  _modelDescriptor.reset(new brayns::ModelDescriptor(std::move(model),
439  _details.assemblyName,
440  metadata));
441  if (_modelDescriptor)
442  _scene.addModel(_modelDescriptor);
443  else
444  PLUGIN_THROW("Vasculature model could not be created for population " +
445  _details.populationName);
446 }

453 void Vasculature::setRadiusReport(const VasculatureRadiusReportDetails& details)
454 {
455  auto& connector = DBConnector::getInstance();
456  const auto simulationReport =
457  connector.getVasculatureSimulationReport(details.populationName,
458  details.simulationReportId);
459 
460  const size_t nbFrames =
461  (simulationReport.endTime - simulationReport.startTime) /
462  simulationReport.timeStep;
463  if (nbFrames == 0)
464  PLUGIN_THROW("Report does not contain any simulation data: " +
465  simulationReport.description);
466 
467  if (details.frame >= nbFrames)
468  PLUGIN_THROW("Invalid frame specified for report: " +
469  simulationReport.description);
470  const floats radii =
471  connector.getVasculatureSimulationTimeSeries(details.populationName,
472  details.simulationReportId,
473  details.frame);
474  doubles series;
475  for (const double radius : radii)
476  series.push_back(details.amplitude * radius);
477  _buildModel(VasculatureColorSchemeDetails(), series);
478 }
