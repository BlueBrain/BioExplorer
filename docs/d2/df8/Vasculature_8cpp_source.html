90 void Vasculature::_addGraphSection(ThreadSafeContainer& container, const GeometryNode& srcNode,
91  const GeometryNode& dstNode, const size_t materialId)
92 {
93  const auto userData = NO_USER_DATA;
94  const auto useSdf = false;
95  const auto maxRadius = std::max(srcNode.radius, dstNode.radius);
96  const auto src = _animatedPosition(Vector4d(srcNode.position, maxRadius));
97  const auto dst = _animatedPosition(Vector4d(dstNode.position, maxRadius));
98  const auto direction = dst - src;
99 
100  const float radius = std::min(length(direction) / 5.0, _getCorrectedRadius(maxRadius, _details.radiusMultiplier));
101  container.addSphere(src, radius * 0.2, materialId, useSdf, userData);
102  container.addCone(src, radius * 0.2, Vector3f(src + direction * 0.79), radius * 0.2, materialId, useSdf, userData);
103  container.addCone(dst, 0.0, Vector3f(src + direction * 0.8), radius, materialId, useSdf, userData);
104  container.addCone(Vector3f(src + direction * 0.8), radius, Vector3f(src + direction * 0.79), radius * 0.2,
105  materialId, useSdf, userData);
106 }

109  const GeometryNode& dstNode, const size_t materialId, const uint64_t userData)
110 {

113 
114  const auto dstRadius = _getCorrectedRadius(dstNode.radius, _details.radiusMultiplier);
115  const auto& dstPoint = _animatedPosition(Vector4d(dstNode.position, dstRadius));
116 
117  const auto useSdf =
118  andCheck(static_cast<uint32_t>(_details.realismLevel), static_cast<uint32_t>(VasculatureRealismLevel::section));
119  if (!useSdf)
120  {
121  container.addSphere(srcPoint, srcRadius, materialId, useSdf, userData);
122  container.addSphere(dstPoint, dstRadius, materialId, useSdf, userData);
123  }
124 
125  container.addCone(srcPoint, srcRadius, dstPoint, dstRadius, materialId, useSdf, userData, {},
126  Vector3f(_getDisplacementValue(DisplacementElement::vasculature_segment_strength),
127  _getDisplacementValue(DisplacementElement::vasculature_segment_frequency), 0.f));

130 void Vasculature::_addDetailedSection(ThreadSafeContainer& container, const GeometryNodes& nodes,
131  const size_t baseMaterialId, const doubles& radii, const Vector2d& radiusRange)
132 {
133  uint64_t geometryIndex = 0;
134  Neighbours neighbours;
135 
136  const auto useSdf =
137  andCheck(static_cast<uint32_t>(_details.realismLevel), static_cast<uint32_t>(VasculatureRealismLevel::section));
138 
139  GeometryNodes localNodes;
140  switch (_details.representation)
141  {
142  case VasculatureRepresentation::optimized_segment:
143  {
144  double oldRadius = 0.0;
145  double segmentLength = 0.0;
146  for (const auto& node : nodes)
147  {
148  segmentLength += node.second.radius;
149  if (segmentLength < 2.0 * (oldRadius + node.second.radius))
150  {
151  GeometryNode n;
152  n.position = node.second.position;
153  n.radius = node.second.radius;
154  localNodes[node.first] = n;
155  }
156  else
157  segmentLength = 0.0;
158  oldRadius = node.second.radius;
159  }
160  break;
161  }
162  case VasculatureRepresentation::bezier:
163  {
164  Vector4fs points;
165  uint64_ts ids;
166  for (const auto& node : nodes)
167  {
168  points.push_back(
169  Vector4d(node.second.position.x, node.second.position.y, node.second.position.z, node.second.radius));
170  ids.push_back(node.first);
171  }
172  const auto localPoints = _getProcessedSectionPoints(morphology::MorphologyRepresentation::bezier, points);
173 
174  uint64_t i = 0;
175  for (const auto& point : localPoints)
176  {
177  GeometryNode n;
178  n.position = Vector3d(point.x, point.y, point.y);
179  n.radius = point.w;
180  localNodes[ids[i * (nodes.size() / localPoints.size())]] = n;
181  ++i;
182  }
183  }
184  default:
185  localNodes = nodes;
186  }
187 
188  uint64_t i = 0;
189  GeometryNode dstNode;
190  for (const auto& node : localNodes)
191  {
192  const auto& srcNode = node.second;
193  const auto userData = node.first;
194 
195  size_t materialId;
196  switch (_details.colorScheme)
197  {
198  case VasculatureColorScheme::radius:
199  materialId = 256 * ((srcNode.radius - radiusRange.x) / (radiusRange.y - radiusRange.x));
200  break;
201  case VasculatureColorScheme::section_points:
202  materialId = 256 * double(node.first - nodes.begin()->first) / double(nodes.size());
203  break;
204  default:
205  materialId = baseMaterialId;
206  break;
207  }
208 
209  const float srcRadius = _getCorrectedRadius((userData < radii.size() ? radii[userData] : srcNode.radius),
210  _details.radiusMultiplier);
211  const auto srcPosition = _animatedPosition(Vector4d(srcNode.position, srcRadius));
212 
213  if (i == 0)
214  container.addSphere(srcPosition, srcRadius, materialId, useSdf, userData);
215  else
216  {
217  const float dstRadius = _getCorrectedRadius((userData < radii.size() ? radii[userData] : dstNode.radius),
218  _details.radiusMultiplier);
219  const auto dstPosition = _animatedPosition(Vector4d(dstNode.position, dstRadius));
220 
221  geometryIndex =
222  container.addCone(srcPosition, srcRadius, dstPosition, dstRadius, materialId, useSdf, userData,
223  neighbours,
224  Vector3f(_getDisplacementValue(DisplacementElement::vasculature_segment_strength),
225  _getDisplacementValue(DisplacementElement::vasculature_segment_frequency),
226  0.f));
227  neighbours = {geometryIndex};
228 
229  if (!useSdf)
230  neighbours.insert(container.addSphere(srcPosition, srcRadius, materialId, useSdf, userData));
231  }
232 
233  dstNode = srcNode;
234  ++i;
235  }
236 }

238 void Vasculature::_addOrientation(ThreadSafeContainer& container, const GeometryNodes& nodes, const uint64_t sectionId)
239 {
240  const auto nbNodes = nodes.size();
241  if (nbNodes <= 3)
242  return;
243 
244  StreamlinesData streamline;
245 
246  GeometryNode previousNode;
247  const float alpha = 1.f;
248  uint64_t i = 0;
249  for (const auto& node : nodes)
250  {
251  streamline.vertex.push_back(
252  Vector4f(node.second.position, _getCorrectedRadius(node.second.radius, _details.radiusMultiplier)));
253  streamline.vertexColor.push_back(
254  (i == 0 ? Vector4f(0.f, 0.f, 0.f, alpha)
255  : Vector4f(0.5 + 0.5 * normalize(node.second.position - previousNode.position), alpha)));
256  previousNode = node.second;
257  ++i;
258  }
259 
260  container.addStreamline(sectionId, streamline);
261 }

263 void Vasculature::_buildModel(const LoaderProgress& callback, const doubles& radii)
264 {
265  if (_modelDescriptor)
266  _scene.removeModel(_modelDescriptor->getModelID());
267 
268  auto model = _scene.createModel();
269  ThreadSafeContainers containers;
270 
271  PLUGIN_INFO(1, "Identifying nodes...");
272  callback.updateProgress("Identifying nodes...", 1.f);
273  const auto nbDBConnections = DBConnector::getInstance().getNbConnections();
274 
275  _nbNodes = DBConnector::getInstance().getVasculatureNbNodes(_details.populationName, _details.sqlFilter);
276 
277  const auto dbBatchSize = _nbNodes / nbDBConnections;
278  PLUGIN_INFO(1, "DB connections=" << nbDBConnections << ", DB batch size=" << dbBatchSize);
279 
280  Vector2d radiusRange;
281  if (_details.colorScheme == VasculatureColorScheme::radius)
282  radiusRange = DBConnector::getInstance().getVasculatureRadiusRange(_details.populationName, _details.sqlFilter);
283 
284  uint64_t index;
285  volatile bool flag = false;
286  std::string flagMessage;
287 #pragma omp parallel for shared(flag) num_threads(nbDBConnections)
288  for (index = 0; index < nbDBConnections; ++index)
289  {
290  try
291  {
292  if (flag)
293  continue;
294 
295  if (omp_get_thread_num() == 0)
296  {
297  PLUGIN_PROGRESS("Loading sections...", index, nbDBConnections);
298  try
299  {
300  callback.updateProgress("Loading sections...", 0.5f * ((float)index / (float)nbDBConnections));
301  }
302  catch (...)
303  {
304 #pragma omp critical
305  flag = true;
306  }
307  }
308 
309  const auto offset = index * dbBatchSize;
310  const std::string limits = "OFFSET " + std::to_string(offset) + " LIMIT " + std::to_string(dbBatchSize);
311 
312  const auto filter = _details.sqlFilter;
313  const auto nodes = DBConnector::getInstance().getVasculatureNodes(_details.populationName, filter, limits);
314 
315  if (nodes.empty())
316  continue;
317 
318  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation,
319  doublesToVector3d(_details.scale));
320 
321  auto iter = nodes.begin();
322  uint64_t previousSectionId = iter->second.sectionId;
323  do
324  {
325  GeometryNodes sectionNodes;
326  const auto sectionId = iter->second.sectionId;
327  const auto userData = iter->first;
328  while (iter != nodes.end() && iter->second.sectionId == previousSectionId)
329  {
330  sectionNodes[iter->first] = iter->second;
331  ++iter;
332  }
333  previousSectionId = sectionId;
334 
335  if (sectionNodes.size() >= 1)
336  {
337  const auto& srcNode = sectionNodes.begin()->second;
338  auto it = sectionNodes.end();
339  --it;
340  const auto& dstNode = it->second;
341 
342  size_t materialId;
343  switch (_details.colorScheme)
344  {
345  case VasculatureColorScheme::section:
346  materialId = sectionId;
347  break;
348  case VasculatureColorScheme::section_orientation:
349  materialId = getMaterialIdFromOrientation(dstNode.position - srcNode.position);
350  break;
351  case VasculatureColorScheme::subgraph:
352  materialId = dstNode.graphId;
353  break;
354  case VasculatureColorScheme::pair:
355  materialId = dstNode.pairId;
356  break;
357  case VasculatureColorScheme::entry_node:
358  materialId = dstNode.entryNodeId;
359  break;
360  case VasculatureColorScheme::radius:
361  materialId = 256 * ((srcNode.radius - radiusRange.x) / (radiusRange.y - radiusRange.x));
362  break;
363  case VasculatureColorScheme::region:
364  materialId = dstNode.regionId;
365  break;
366  default:
367  materialId = 0;
368  break;
369  }
370 
371  switch (_details.representation)
372  {
373  case VasculatureRepresentation::graph:
374  _addGraphSection(container, srcNode, dstNode, materialId);
375  break;
376  case VasculatureRepresentation::section:
377  _addSimpleSection(container, srcNode, dstNode, materialId, userData);
378  break;
379  default:
380  _addDetailedSection(container, sectionNodes, materialId, radii, radiusRange);
381  break;
382  }
383  }
384  } while (iter != nodes.end());
385 
386 #pragma omp critical
387  containers.push_back(container);
388  }
389  catch (const std::runtime_error& e)
390  {
391 #pragma omp critical
392  flagMessage = e.what();
393 #pragma omp critical
394  flag = true;
395  }
396  catch (...)
397  {
398 #pragma omp critical
399  flagMessage = "Loading was canceled";
400 #pragma omp critical
401  flag = true;
402  }
403  }
404 
405  for (size_t i = 0; i < containers.size(); ++i)
406  {
407  PLUGIN_PROGRESS("- Compiling 3D geometry...", 1 + i, containers.size());
408  callback.updateProgress("Compiling 3D geometry...", 0.5f + 0.5f * (float)(1 + i) / (float)containers.size());
409  auto& container = containers[i];
410  container.commitToModel();
411  }
412  model->applyDefaultColormap();
413 
414  const ModelMetadata metadata = {{"Number of nodes", std::to_string(_nbNodes)}, {"SQL filter", _details.sqlFilter}};
415 
416  _modelDescriptor.reset(new core::ModelDescriptor(std::move(model), _details.assemblyName, metadata));
417 
418  if (!_modelDescriptor)
419  PLUGIN_THROW(
420  "Vasculature model could not be created for "
421  "population " +
422  _details.populationName);
423 }

425 void Vasculature::setRadiusReport(const VasculatureRadiusReportDetails& details)
426 {
427  auto& connector = DBConnector::getInstance();
428  const auto simulationReport = connector.getSimulationReport(details.populationName, details.simulationReportId);
429 
430  const size_t nbFrames = (simulationReport.endTime - simulationReport.startTime) / simulationReport.timeStep;
431  if (nbFrames == 0)
432  PLUGIN_THROW("Report does not contain any simulation data: " + simulationReport.description);
433 
434  if (details.frame >= nbFrames)
435  PLUGIN_THROW("Invalid frame specified for report: " + simulationReport.description);
436  const floats radii =
437  connector.getVasculatureSimulationTimeSeries(details.populationName, details.simulationReportId, details.frame);
438  doubles series;
439  for (const double radius : radii)
440  series.push_back(details.amplitude * radius);
441  _buildModel(LoaderProgress(), series);
442 }
