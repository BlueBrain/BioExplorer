127 Boxd vector_to_bounds(const doubles &lowBounds, const doubles &highBounds)
128 {
129  if (!lowBounds.empty() && lowBounds.size() != 3)
130  PLUGIN_THROW("Invalid low bounds. 3 floats expected");
131  if (!highBounds.empty() && highBounds.size() != 3)
132  PLUGIN_THROW("Invalid high bounds. 3 floats expected");
133 
134  Boxd bounds;
135  if (!lowBounds.empty())
136  bounds.merge({lowBounds[0], lowBounds[1], lowBounds[2]});
137  if (!highBounds.empty())
138  bounds.merge({highBounds[0], highBounds[1], highBounds[2]});
139  return bounds;
140 }

972 Response BioExplorerPlugin::_addEnzymeReaction(const EnzymeReactionDetails &payload) const

975  try
976  {
977  AssemblyPtr enzymeAssembly{nullptr};
978  ProteinPtr enzyme{nullptr};
979 
980  const auto substrateNames = split(payload.substrateNames);
981  const auto productNames = split(payload.productNames);
982  Proteins substrates(substrateNames.size());
983  Proteins products(productNames.size());
984 
985  uint64_t index = 0;
986  for (const auto &substrateName : substrateNames)
987  {
988  for (auto &assembly : _assemblies)
989  {
990  auto substrate = assembly.second->getProtein(substrateName);
991  if (substrate)
992  {
993  substrates[index] = substrate;
994  break;
995  }
996  }
997  ++index;
998  }
999 
1000  index = 0;
1001  for (const auto &productName : productNames)
1002  {
1003  for (auto &assembly : _assemblies)
1004  {
1005  auto product = assembly.second->getProtein(productName);
1006  if (product)
1007  {
1008  products[index] = product;
1009  break;
1010  }
1011  }
1012  ++index;
1013  }
1014 
1015  for (auto &assembly : _assemblies)
1016  {
1017  enzyme = assembly.second->getProtein(payload.enzymeName);
1018  if (enzyme)
1019  {
1020  enzymeAssembly = assembly.second;
1021  break;
1022  }
1023  }
1024 
1025  if (!enzyme)
1026  PLUGIN_THROW("Enzyme " + payload.enzymeName + " could not be found in scene");
1027  if (substrates.size() != substrateNames.size())
1028  PLUGIN_THROW("Some substrates could not be found in scene");
1029  if (products.size() != productNames.size())
1030  PLUGIN_THROW("Some products could not be found in scene");
1031 
1032  auto it = _assemblies.find(payload.assemblyName);
1033  if (it != _assemblies.end())
1034  (*it).second->addEnzymeReaction(payload, enzymeAssembly, enzyme, substrates, products);
1035  else
1036  PLUGIN_THROW("Assembly " + payload.assemblyName + " not found");
1037  }
1038  catch (const std::runtime_error &e)
1039  {
1040  response.status = false;
1041  response.contents = e.what();
1042  }
1043  return response;
