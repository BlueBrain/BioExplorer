102 Boxd vector_to_bounds(const doubles &lowBounds, const doubles &highBounds)
103 {
104  if (!lowBounds.empty() && lowBounds.size() != 3)
105  PLUGIN_THROW("Invalid low bounds. 3 floats expected");
106  if (!highBounds.empty() && highBounds.size() != 3)
107  PLUGIN_THROW("Invalid high bounds. 3 floats expected");
108 
109  Boxd bounds;
110  if (!lowBounds.empty())
111  bounds.merge({lowBounds[0], lowBounds[1], lowBounds[2]});
112  if (!highBounds.empty())
113  bounds.merge({highBounds[0], highBounds[1], highBounds[2]});
114  return bounds;
115 }

914 Response BioExplorerPlugin::_addEnzymeReaction(
915  const EnzymeReactionDetails &payload) const
916 {
917  Response response;
918  try
919  {
920  AssemblyPtr enzymeAssembly{nullptr};
921  ProteinPtr enzyme{nullptr};
922 
923  const auto substrateNames = split(payload.substrateNames);
924  const auto productNames = split(payload.productNames);
925  Proteins substrates(substrateNames.size());
926  Proteins products(productNames.size());
927 
928  uint64_t index = 0;
929  for (const auto &substrateName : substrateNames)
930  {
931  for (auto &assembly : _assemblies)
932  {
933  auto substrate = assembly.second->getProtein(substrateName);
934  if (substrate)
935  {
936  substrates[index] = substrate;
937  break;
938  }
939  }
940  ++index;
941  }
942 
943  index = 0;
944  for (const auto &productName : productNames)
945  {
946  for (auto &assembly : _assemblies)
947  {
948  auto product = assembly.second->getProtein(productName);
949  if (product)
950  {
951  products[index] = product;
952  break;
953  }
954  }
955  ++index;
956  }
957 
958  for (auto &assembly : _assemblies)
959  {
960  enzyme = assembly.second->getProtein(payload.enzymeName);
961  if (enzyme)
962  {
963  enzymeAssembly = assembly.second;
964  break;
965  }
966  }
967 
968  if (!enzyme)
969  PLUGIN_THROW("Enzyme " + payload.enzymeName +
970  " could not be found in scene");
971  if (substrates.size() != substrateNames.size())
972  PLUGIN_THROW("Some substrates could not be found in scene");
973  if (products.size() != productNames.size())
974  PLUGIN_THROW("Some products could not be found in scene");
975 
976  auto it = _assemblies.find(payload.assemblyName);
977  if (it != _assemblies.end())
978  (*it).second->addEnzymeReaction(payload, enzymeAssembly, enzyme,
979  substrates, products);
980  else
981  PLUGIN_THROW("Assembly " + payload.assemblyName + " not found");
982  }
983  catch (const std::runtime_error &e)
984  {
985  response.status = false;
986  response.contents = e.what();
987  }
988  return response;

1096

1098  auto model = scene.createModel();
1099 
1100  const Vector3d red = {1, 0, 0};
1101  const Vector3d green = {0, 1, 0};
1102  const Vector3d blue = {0, 0, 1};
1103  const Vector3d grey = {0.5, 0.5, 0.5};
1104 
1105  PropertyMap props;
1106  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
1107  static_cast<int>(MaterialShadingMode::basic)});
1108  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
1109  props.setProperty(
1110  {MATERIAL_PROPERTY_CHAMELEON_MODE,
1111  static_cast<int>(
1112  MaterialChameleonMode::undefined_chameleon_mode)});
1113 
1114  auto material = model->createMaterial(0, "x");
1115  material->setDiffuseColor(grey);
1116  material->setProperties(props);
1117  const auto position = doublesToVector3d(payload.position);
1118 
1119  const double m = payload.minValue;
1120  const double M = payload.maxValue;
1121  const double s = payload.steps;
1122  const float r = static_cast<float>(payload.radius);
1123 
1125  for (double x = m; x <= M; x += s)
1126  for (double y = m; y <= M; y += s)
1127  {
1128  bool showFullGrid = true;
1129  if (!payload.showFullGrid)
1130  showFullGrid = (fabs(x) < 0.001f || fabs(y) < 0.001f);
1131  if (showFullGrid)
1132  {
1133  model->addCylinder(0, {position + Vector3d(x, y, m),
1134  position + Vector3d(x, y, M), r});
1135  model->addCylinder(0, {position + Vector3d(m, x, y),
1136  position + Vector3d(M, x, y), r});
1137  model->addCylinder(0, {position + Vector3d(x, m, y),
1138  position + Vector3d(x, M, y), r});
1139  }
1140  }
1141 
1142  if (payload.showPlanes)
1143  {
1144  // Planes
1145  material = model->createMaterial(1, "plane_x");
1146  material->setDiffuseColor(payload.useColors ? red : grey);
1147  material->setOpacity(payload.planeOpacity);
1148  material->setProperties(props);
1149  auto &tmx = model->getTriangleMeshes()[1];
1150  tmx.vertices.push_back(position + Vector3d(m, 0, m));
1151  tmx.vertices.push_back(position + Vector3d(M, 0, m));
1152  tmx.vertices.push_back(position + Vector3d(M, 0, M));
1153  tmx.vertices.push_back(position + Vector3d(m, 0, M));
1154  tmx.indices.push_back(Vector3ui(0, 1, 2));
1155  tmx.indices.push_back(Vector3ui(2, 3, 0));
1156 
1157  material = model->createMaterial(2, "plane_y");
1158  material->setDiffuseColor(payload.useColors ? green : grey);
1159  material->setOpacity(payload.planeOpacity);
1160  material->setProperties(props);
1161  auto &tmy = model->getTriangleMeshes()[2];
1162  tmy.vertices.push_back(position + Vector3d(m, m, 0));
1163  tmy.vertices.push_back(position + Vector3d(M, m, 0));
1164  tmy.vertices.push_back(position + Vector3d(M, M, 0));
1165  tmy.vertices.push_back(position + Vector3d(m, M, 0));
1166  tmy.indices.push_back(Vector3ui(0, 1, 2));
1167  tmy.indices.push_back(Vector3ui(2, 3, 0));
1168 
1169  material = model->createMaterial(3, "plane_z");
1170  material->setDiffuseColor(payload.useColors ? blue : grey);
1171  material->setOpacity(payload.planeOpacity);
1172  material->setProperties(props);
1173  auto &tmz = model->getTriangleMeshes()[3];
1174  tmz.vertices.push_back(position + Vector3d(0, m, m));
1175  tmz.vertices.push_back(position + Vector3d(0, m, M));
1176  tmz.vertices.push_back(position + Vector3d(0, M, M));
1177  tmz.vertices.push_back(position + Vector3d(0, M, m));
1178  tmz.indices.push_back(Vector3ui(0, 1, 2));
1179  tmz.indices.push_back(Vector3ui(2, 3, 0));
1180  }
1181 
1182  // Axis
1183  if (payload.showAxis)
1184  {
1185  const double l = M;
1186  const float smallRadius = r * 25.0;
1187  const float largeRadius = r * 50.0;
1188  const double l1 = l * 0.89;
1189  const double l2 = l * 0.90;
1190 
1191  PropertyMap props;
1192  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
1193  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
1194  static_cast<int>(MaterialShadingMode::basic)});
1195  props.setProperty(
1196  {MATERIAL_PROPERTY_CHAMELEON_MODE,
1197  static_cast<int>(
1198  MaterialChameleonMode::undefined_chameleon_mode)});
1199 
1200  // X
1201  material = model->createMaterial(4, "x_axis");
1202  material->setDiffuseColor(red);
1203  material->setProperties(props);
