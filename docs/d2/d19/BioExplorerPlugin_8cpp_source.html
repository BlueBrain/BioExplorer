96Boxd vector_to_bounds(const std::vector<float> &lowBounds,
97 const std::vector<float> &highBounds)
98{
99 if (!lowBounds.empty() && lowBounds.size() != 3)
100 PLUGIN_THROW("Invalid low bounds. 3 floats expected");
101 if (!highBounds.empty() && highBounds.size() != 3)
102 PLUGIN_THROW("Invalid high bounds. 3 floats expected");
103
104 Boxd bounds;
105 if (!lowBounds.empty())
106 bounds.merge({lowBounds[0], lowBounds[1], lowBounds[2]});
107 if (!highBounds.empty())
108 bounds.merge({highBounds[0], highBounds[1], highBounds[2]});
109 return bounds;
110}

494void BioExplorerPlugin::_parseCommandLineArguments(int argc, char **argv)
495{
496 for (size_t i = 0; i < argc; ++i)
497 {
498 const std::string argument = argv[i];
499 std::string key;
500 std::string value;
501 const int pos = argument.find("=");
502 if (pos == std::string::npos)
503 key = argument;
504 else
505 {
506 key = argument.substr(0, pos);
507 value = argument.substr(pos + 1);
508 }
509 _commandLineArguments[key] = value;
510 }

547SceneInformationDetails BioExplorerPlugin::_getSceneInformation() const
548{
549 SceneInformationDetails sceneInfo;
550 const auto &scene = _api->getScene();
551 const auto &modelDescriptors = scene.getModelDescriptors();
552 sceneInfo.nbModels = modelDescriptors.size();
553
554 for (const auto modelDescriptor : modelDescriptors)
555 {
556 const auto &instances = modelDescriptor->getInstances();
557 const auto nbInstances = instances.size();
558 auto &model = modelDescriptor->getModel();
559 for (const auto &spheres : model.getSpheres())
560 sceneInfo.nbSpheres += nbInstances * spheres.second.size();
561 for (const auto &cylinders : model.getCylinders())
562 sceneInfo.nbCylinders += nbInstances * cylinders.second.size();
563 for (const auto &cones : model.getCones())
564 sceneInfo.nbCones += nbInstances * cones.second.size();
565 for (const auto &triangleMesh : model.getTriangleMeshes())
566 {
567 const auto &triangle = triangleMesh.second;
568 sceneInfo.nbIndices += nbInstances * triangle.indices.size();
569 sceneInfo.nbVertices += nbInstances * triangle.vertices.size();
570 sceneInfo.nbNormals += nbInstances * triangle.normals.size();
571 sceneInfo.nbColors += nbInstances * triangle.colors.size();
572 }
573 sceneInfo.nbMaterials +=
574 nbInstances *
575 (model.getSpheres().size() + model.getCylinders().size() +
576 model.getCones().size() + model.getTriangleMeshes().size());
577 }
578 return sceneInfo;
579}

805Response BioExplorerPlugin::_addGrid(const AddGridDetails &payload)
806{
807 Response response;
808 try
809 {
810 PLUGIN_INFO("Building Grid scene");
811
812 auto &scene = _api->getScene();
813 auto model = scene.createModel();
814
815 const Vector3f red = {1, 0, 0};
816 const Vector3f green = {0, 1, 0};
817 const Vector3f blue = {0, 0, 1};
818 const Vector3f grey = {0.5, 0.5, 0.5};
819
820 PropertyMap props;
821 props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
822 static_cast<int>(MaterialShadingMode::basic)});
823 props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
824 props.setProperty(
825 {MATERIAL_PROPERTY_CHAMELEON_MODE,
826 static_cast<int>(
827 MaterialChameleonMode::undefined_chameleon_mode)});
828
829 auto material = model->createMaterial(0, "x");
830 material->setDiffuseColor(grey);
831 material->setProperties(props);
832 const auto position = floatsToVector3f(payload.position);
833
834 const float m = payload.minValue;
835 const float M = payload.maxValue;
836 const float s = payload.steps;
837 const float r = payload.radius;
838
840 for (float x = m; x <= M; x += s)
841 for (float y = m; y <= M; y += s)
842 {
843 bool showFullGrid = true;
844 if (!payload.showFullGrid)
845 showFullGrid = (fabs(x) < 0.001f || fabs(y) < 0.001f);
846 if (showFullGrid)
847 {
848 model->addCylinder(0, {position + Vector3f(x, y, m),
849 position + Vector3f(x, y, M), r});
850 model->addCylinder(0, {position + Vector3f(m, x, y),
851 position + Vector3f(M, x, y), r});
852 model->addCylinder(0, {position + Vector3f(x, m, y),
853 position + Vector3f(x, M, y), r});
854 }
855 }
856
857 if (payload.showPlanes)
858 {
859 // Planes
860 material = model->createMaterial(1, "plane_x");
861 material->setDiffuseColor(payload.useColors ? red : grey);
862 material->setOpacity(payload.planeOpacity);
863 material->setProperties(props);
864 auto &tmx = model->getTriangleMeshes()[1];
865 tmx.vertices.push_back(position + Vector3f(m, 0, m));
866 tmx.vertices.push_back(position + Vector3f(M, 0, m));
867 tmx.vertices.push_back(position + Vector3f(M, 0, M));
868 tmx.vertices.push_back(position + Vector3f(m, 0, M));
869 tmx.indices.push_back(Vector3ui(0, 1, 2));
870 tmx.indices.push_back(Vector3ui(2, 3, 0));
871
872 material = model->createMaterial(2, "plane_y");
873 material->setDiffuseColor(payload.useColors ? green : grey);
874 material->setOpacity(payload.planeOpacity);
875 material->setProperties(props);
876 auto &tmy = model->getTriangleMeshes()[2];
877 tmy.vertices.push_back(position + Vector3f(m, m, 0));
878 tmy.vertices.push_back(position + Vector3f(M, m, 0));
879 tmy.vertices.push_back(position + Vector3f(M, M, 0));
880 tmy.vertices.push_back(position + Vector3f(m, M, 0));
881 tmy.indices.push_back(Vector3ui(0, 1, 2));
882 tmy.indices.push_back(Vector3ui(2, 3, 0));
883
884 material = model->createMaterial(3, "plane_z");
885 material->setDiffuseColor(payload.useColors ? blue : grey);
886 material->setOpacity(payload.planeOpacity);
887 material->setProperties(props);
888 auto &tmz = model->getTriangleMeshes()[3];
889 tmz.vertices.push_back(position + Vector3f(0, m, m));
890 tmz.vertices.push_back(position + Vector3f(0, m, M));
891 tmz.vertices.push_back(position + Vector3f(0, M, M));
892 tmz.vertices.push_back(position + Vector3f(0, M, m));
893 tmz.indices.push_back(Vector3ui(0, 1, 2));
894 tmz.indices.push_back(Vector3ui(2, 3, 0));
895 }
896
897 // Axis
898 if (payload.showAxis)
899 {
900 const float l = M;
901 const float smallRadius = payload.radius * 25.0;
902 const float largeRadius = payload.radius * 50.0;
903 const float l1 = l * 0.89;
904 const float l2 = l * 0.90;
905
906 PropertyMap props;
907 props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
908 props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
909 static_cast<int>(MaterialShadingMode::basic)});
910 props.setProperty(
911 {MATERIAL_PROPERTY_CHAMELEON_MODE,
912 static_cast<int>(
913 MaterialChameleonMode::undefined_chameleon_mode)});
914
915 // X
916 material = model->createMaterial(4, "x_axis");
917 material->setDiffuseColor(red);
918 material->setProperties(props);
919
920 model->addCylinder(4, {position, position + Vector3f(l1, 0, 0),
921 smallRadius});
922 model->addCone(4, {position + Vector3f(l1, 0, 0),
923 position + Vector3f(l2, 0, 0), smallRadius,
924 largeRadius});
925 model->addCone(4, {position + Vector3f(l2, 0, 0),
926 position + Vector3f(M, 0, 0), largeRadius, 0});
927
928 // Y
929 material = model->createMaterial(5, "y_axis");
930 material->setDiffuseColor(green);
931 material->setProperties(props);
932
933 model->addCylinder(5, {position, position + Vector3f(0, l1, 0),
934 smallRadius});
935 model->addCone(5, {position + Vector3f(0, l1, 0),
936 position + Vector3f(0, l2, 0), smallRadius,
937 largeRadius});
938 model->addCone(5, {position + Vector3f(0, l2, 0),
939 position + Vector3f(0, M, 0), largeRadius, 0});
940
941 // Z
942 material = model->createMaterial(6, "z_axis");
943 material->setDiffuseColor(blue);
944 material->setProperties(props);
945
946 model->addCylinder(6, {position, position + Vector3f(0, 0, l1),
947 smallRadius});
948 model->addCone(6, {position + Vector3f(0, 0, l1),
949 position + Vector3f(0, 0, l2), smallRadius,
950 largeRadius});
951 model->addCone(6, {position + Vector3f(0, 0, l2),
952 position + Vector3f(0, 0, M), largeRadius, 0});
953
954 // Origin
955 model->addSphere(0, {position, smallRadius});
956 }
957
958 scene.addModel(
959 std::make_shared<ModelDescriptor>(std::move(model), "Grid"));
960 }
961 CATCH_STD_EXCEPTION()
962 return response;
963}

1102 material->updateProperty(

1104 payload.shadingModes[id]);
1105 if (!payload.userParameters.empty())

1122 }
1123 }
1124 catch (const std::runtime_error &e)
1125 {
1126 PLUGIN_INFO(e.what());
1127 }

1129 }
1130 }
1131 else
1132 PLUGIN_INFO("Model " << modelId << " is not registered");
1133 }
1134 scene.markModified(false);
1135 }

1249 try
1250 {

1252
1253 const auto clipPlanes = getClippingPlanes(scene);
1254
1255 auto model = scene.createModel();
1256
1257 // Material
1258 const size_t materialId = 0;
1259 auto material = model->createMaterial(materialId, "Point cloud");
1260
1261 PropertyMap props;
1262 props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
1263 static_cast<int>(MaterialShadingMode::basic)});
1264 props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});

1272 material->updateProperties(props);

1275 for (const auto modelDescriptor : modelDescriptors)
1276 {
1277 const auto &instances = modelDescriptor->getInstances();
1278 for (const auto &instance : instances)
1279 {

1281 const auto &m = modelDescriptor->getModel();
1282 const auto &spheresMap = m.getSpheres();
1283 for (const auto &spheres : spheresMap)
1284 {

1288 tf.getTranslation() +
1289 tf.getRotation() * (Vector3d(sphere.center) -
1290 tf.getRotationCenter());
1291
1292 const Vector3f c = center;
1293 if (isClipped(c, clipPlanes))
1294 continue;
1295
1296 model->addSphere(materialId,
1297 {c, payload.radius * sphere.radius});
1298 }
1299 }
1300 }
1301 }
1302

1305 scene.addModel(md);
