43 uint64_t ThreadSafeContainer::addSphere(const Vector3f& position,
44  const float radius,
45  const size_t materialId,
46  const uint64_t userData,
47  const Neighbours& neighbours,
48  const Vector3f displacement)
49 {

52  displacement.y / scale.x, displacement.z};

54  return _addSDFGeometry(materialId,

63 uint64_t ThreadSafeContainer::addCone(
64  const Vector3f& sourcePosition, const float sourceRadius,
65  const Vector3f& targetPosition, const float targetRadius,
66  const size_t materialId, const uint64_t userDataOffset,
67  const Neighbours& neighbours, const Vector3f displacement)
68 {
69  const Vector3f scale = _scale;
70  const Vector3f scaledDisplacement{displacement.x * scale.x,
71  displacement.y / scale.x, displacement.z};
72  if (_useSdf)
73  {
74  const auto geom =
75  createSDFConePill(sourcePosition * scale, targetPosition * scale,
76  sourceRadius * scale.x, targetRadius * scale.x,
77  userDataOffset, scaledDisplacement);
78  return _addSDFGeometry(materialId, geom, neighbours);
79  }
80  if (sourceRadius == targetRadius)
81  return _addCylinder(materialId,
82  {sourcePosition * scale, targetPosition * scale,
83  sourceRadius * scale.x, userDataOffset});
84  return _addCone(materialId, {sourcePosition * scale, targetPosition * scale,
85  sourceRadius * scale.x, targetRadius * scale.x,
86  userDataOffset});
87 }

166 void ThreadSafeContainer::_commitMaterials()
167 {
168  for (const auto materialId : _materialIds)
169  {
170  Vector3f color{1.f, 1.f, 1.f};
171  auto material =
172  _model.createMaterial(materialId, std::to_string(materialId));
173 
174  material->setDiffuseColor(color);
175  material->setSpecularColor(color);
176  material->setSpecularExponent(100.f);
177  PropertyMap props;
178  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE, 0});
179  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
180  props.setProperty({MATERIAL_PROPERTY_CHAMELEON_MODE, 0});
181  props.setProperty({MATERIAL_PROPERTY_CAST_USER_DATA, false});
182  props.setProperty({MATERIAL_PROPERTY_NODE_ID, 0});
183  material->updateProperties(props);
184  }
185 }

226 void ThreadSafeContainer::_commitSDFGeometriesToModel()
227 {
228  _finalizeSDFGeometries();
229 
230  const uint64_t numGeoms = _sdfMorphologyData.geometries.size();
231  size_ts localToGlobalIndex(numGeoms, 0);
232 
233  // Add geometries to _model. We do not know the indices of the
234  // neighbours yet so we leave them empty.
235  for (uint64_t i = 0; i < numGeoms; ++i)
236  localToGlobalIndex[i] =
237  _model.addSDFGeometry(_sdfMorphologyData.materials[i],
238  _sdfMorphologyData.geometries[i], {});
239 
240  // Write the neighbours using global indices
241  uint64_ts neighboursTmp;
242  for (uint64_t i = 0; i < numGeoms; ++i)
243  {
244  const uint64_t globalIndex = localToGlobalIndex[i];
245  neighboursTmp.clear();
246 
247  for (auto localNeighbourIndex : _sdfMorphologyData.neighbours[i])
248  neighboursTmp.push_back(localToGlobalIndex[localNeighbourIndex]);
249 
250  _model.updateSDFGeometryNeighbours(globalIndex, neighboursTmp);
251  }
252 
253  for (const auto materialId : _sdfMorphologyData.materials)
254  _materialIds.insert(materialId);
255  _sdfMorphologyData.geometries.clear();
256  _sdfMorphologyData.neighbours.clear();
257  _sdfMorphologyData.materials.clear();
258 }
