46 uint64_t ThreadSafeContainer::addSphere(const Vector3f& position, const float radius, const size_t materialId,
47  const bool useSdf, const uint64_t userDataOffset, const Neighbours& neighbours,
48  const Vector3f displacement)
49 {
50  const Vector3f scale = _scale;
51  const Vector3f scaledPosition =
52  getAlignmentToGrid(_alignToGrid, Vector3f(_position + _rotation * Vector3d(position)) * scale);
53  if (useSdf)
54  {
55  const auto scaledRadius = (radius - displacement.x) * scale.x;
56  _bounds.merge(scaledPosition + scaledRadius);
57  _bounds.merge(scaledPosition - scaledRadius);
58  const Vector3f scaledDisplacement{displacement.x * scale.x, displacement.y / scale.x, displacement.z};
59  return _addSDFGeometry(materialId,
60  createSDFSphere(scaledPosition, scaledRadius, userDataOffset, scaledDisplacement),
61  neighbours);
62  }
63  const auto scaledRadius = radius * scale.x;
64  _bounds.merge(scaledPosition + scaledRadius);
65  _bounds.merge(scaledPosition - scaledRadius);
66  return _addSphere(materialId, {scaledPosition, scaledRadius, userDataOffset});
67 }

87 uint64_t ThreadSafeContainer::addCone(const Vector3f& sourcePosition, const float sourceRadius,
88  const Vector3f& targetPosition, const float targetRadius, const size_t materialId,
89  const bool useSdf, const uint64_t userDataOffset, const Neighbours& neighbours,
90  const Vector3f displacement)
91 {
92  const Vector3f scale = _scale;
93  const Vector3f scaledSrcPosition =
94  getAlignmentToGrid(_alignToGrid, Vector3f(_position + _rotation * Vector3d(sourcePosition)) * scale);
95  const Vector3f scaledDstPosition =
96  getAlignmentToGrid(_alignToGrid, Vector3f(_position + _rotation * Vector3d(targetPosition)) * scale);
97  if (useSdf)
98  {
99  const Vector3f scaledDisplacement{displacement.x * scale.x, displacement.y / scale.x, displacement.z};
100  const auto scaledSrcRadius = (sourceRadius - displacement.x) * scale.x;
101  const auto scaledDstRadius = (targetRadius - displacement.x) * scale.x;
102  const auto geom = createSDFConePill(scaledSrcPosition, scaledDstPosition, scaledSrcRadius, scaledDstRadius,
103  userDataOffset, scaledDisplacement);
104  _bounds.merge(scaledSrcPosition + scaledSrcRadius);
105  _bounds.merge(scaledSrcPosition - scaledSrcRadius);
106  _bounds.merge(scaledDstPosition + scaledDstRadius);
107  _bounds.merge(scaledDstPosition - scaledDstRadius);
108  return _addSDFGeometry(materialId, geom, neighbours);
109  }
110  if (fabs(sourceRadius - targetRadius) < equalityEpsilon)
111  {
112  const auto scaledSrcRadius = sourceRadius * scale.x;
113  const auto scaledDstRadius = targetRadius * scale.x;
114  _bounds.merge(scaledSrcPosition + scaledSrcRadius);
115  _bounds.merge(scaledSrcPosition - scaledSrcRadius);
116  _bounds.merge(scaledDstPosition + scaledDstRadius);
117  _bounds.merge(scaledDstPosition - scaledDstRadius);
118  return _addCylinder(materialId, {scaledSrcPosition, scaledDstPosition, sourceRadius * scale.x, userDataOffset});
119  }
120 
121  const auto scaledSrcRadius = sourceRadius * scale.x;
122  const auto scaledDstRadius = targetRadius * scale.x;
123  _bounds.merge(scaledSrcPosition + scaledSrcRadius);
124  _bounds.merge(scaledSrcPosition - scaledSrcRadius);
125  _bounds.merge(scaledDstPosition + scaledDstRadius);
126  _bounds.merge(scaledDstPosition - scaledDstRadius);
127  return _addCone(materialId, {scaledSrcPosition, scaledDstPosition, sourceRadius * scale.x, targetRadius * scale.x,
128  userDataOffset});
129 }

235 void ThreadSafeContainer::_finalizeSDFGeometries()
236 {
237  const uint64_t numGeoms = _sdfMorphologyData.geometries.size();
238  for (uint64_t i = 0; i < numGeoms; ++i)
239  {
240  const auto& neighbours = _sdfMorphologyData.neighbours[i];
241  for (const auto neighbour : neighbours)
242  if (neighbour != i)
243  _sdfMorphologyData.neighbours[neighbour].insert(i);
244  }

289 void ThreadSafeContainer::_commitSDFGeometriesToModel()
290 {
291  _finalizeSDFGeometries();
292 
293  const uint64_t numGeoms = _sdfMorphologyData.geometries.size();
294  size_ts localToGlobalIndex(numGeoms, 0);
295 
296  // Add geometries to _model. We do not know the indices of the neighbours yet so we leave them empty.
297  for (uint64_t i = 0; i < numGeoms; ++i)
298  localToGlobalIndex[i] =
299  _model.addSDFGeometry(_sdfMorphologyData.materials[i], _sdfMorphologyData.geometries[i], {});
300 
301  // Write the neighbours using global indices
302  uint64_ts neighboursTmp;
303  for (uint64_t i = 0; i < numGeoms; ++i)
304  {
305  const uint64_t globalIndex = localToGlobalIndex[i];
306  neighboursTmp.clear();
307 
308  for (auto localNeighbourIndex : _sdfMorphologyData.neighbours[i])
309  neighboursTmp.push_back(localToGlobalIndex[localNeighbourIndex]);
310 
311  _model.updateSDFGeometryNeighbours(globalIndex, neighboursTmp);
312  }
313 
314  for (const auto materialId : _sdfMorphologyData.materials)
315  _materialIds.insert(materialId);
316  _sdfMorphologyData.geometries.clear();
317  _sdfMorphologyData.neighbours.clear();
318  _sdfMorphologyData.materials.clear();
319 }

321 void ThreadSafeContainer::_commitMeshesToModel()
322 {
323  for (const auto& meshes : _meshesMap)
324  {
325  const auto materialId = meshes.first;
326  _materialIds.insert(materialId);
327  const auto& srcMesh = meshes.second;
328  auto& dstMesh = _model.getTriangleMeshes()[materialId];
329  auto vertexOffset = dstMesh.vertices.size();
330  dstMesh.vertices.insert(dstMesh.vertices.end(), srcMesh.vertices.begin(), srcMesh.vertices.end());
331  auto indexOffset = dstMesh.indices.size();
332  dstMesh.indices.insert(dstMesh.indices.end(), srcMesh.indices.begin(), srcMesh.indices.end());
333  for (uint64_t i = 0; i < srcMesh.indices.size(); ++i)
334  dstMesh.indices[indexOffset + i] += vertexOffset;
335  dstMesh.normals.insert(dstMesh.normals.end(), srcMesh.normals.begin(), srcMesh.normals.end());
336  dstMesh.colors.insert(dstMesh.colors.end(), srcMesh.colors.begin(), srcMesh.colors.end());
337  for (const auto& vertex : srcMesh.vertices)
338  _bounds.merge(vertex);
339  }
340  _meshesMap.clear();
341 }

343 void ThreadSafeContainer::_commitStreamlinesToModel()
344 {
345  const auto materialId = 0;
346  if (_streamlinesMap.find(materialId) == _streamlinesMap.end())
347  return;
348 
349  _materialIds.insert(materialId);
350  auto& modelStreamline = _model.getStreamlines()[materialId];
351  auto modelOffset = modelStreamline.vertex.size();
352 
353  for (const auto& streamline : _streamlinesMap)
354  {
355  modelStreamline.vertex.insert(modelStreamline.vertex.end(), streamline.second.vertex.begin(),
356  streamline.second.vertex.end());
357  modelStreamline.vertexColor.insert(modelStreamline.vertexColor.end(), streamline.second.vertexColor.begin(),
358  streamline.second.vertexColor.end());
359 
360  for (size_t i = 0; i < streamline.second.vertex.size() - 1; ++i)
361  {
362  modelStreamline.indices.push_back(modelOffset + i);
363  _bounds.merge(streamline.second.vertex[i]);
364  }
365  modelOffset += streamline.second.vertex.size();
366  }
367  _streamlinesMap.clear();
368 }
