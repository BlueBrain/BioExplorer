53 uint64_t ThreadSafeContainer::addSphere(const Vector3f& position, const float radius, const size_t materialId,
54  const bool useSdf, const uint64_t userDataOffset, const Neighbours& neighbours,
55  const Vector3f displacement)
56 {
57  const Vector3f scale = _scale;
58  const Vector3f scaledPosition =
59  getAlignmentToGrid(_alignToGrid, Vector3f(_position + _rotation * Vector3d(position)) * scale);
60  if (useSdf)
61  {
62  const auto scaledRadius = (radius - displacement.x) * scale.x;
63  _bounds.merge(scaledPosition + scaledRadius);
64  _bounds.merge(scaledPosition - scaledRadius);
65  const Vector3f scaledDisplacement{displacement.x * scale.x, displacement.y / scale.x, displacement.z};
66  return _addSDFGeometry(materialId,
67  createSDFSphere(scaledPosition, scaledRadius, userDataOffset, scaledDisplacement),
68  neighbours);
69  }
70  const auto scaledRadius = radius * scale.x;
71  _bounds.merge(scaledPosition + scaledRadius);
72  _bounds.merge(scaledPosition - scaledRadius);
73  return _addSphere(materialId, {scaledPosition, scaledRadius, userDataOffset});
74 }

94 uint64_t ThreadSafeContainer::addCone(const Vector3f& sourcePosition, const float sourceRadius,
95  const Vector3f& targetPosition, const float targetRadius, const size_t materialId,
96  const bool useSdf, const uint64_t userDataOffset, const Neighbours& neighbours,
97  const Vector3f displacement)
98 {
99  const Vector3f scale = _scale;
100  const Vector3f scaledSrcPosition =
101  getAlignmentToGrid(_alignToGrid, Vector3f(_position + _rotation * Vector3d(sourcePosition)) * scale);
102  const Vector3f scaledDstPosition =
103  getAlignmentToGrid(_alignToGrid, Vector3f(_position + _rotation * Vector3d(targetPosition)) * scale);
104  if (useSdf)
105  {
106  const Vector3f scaledDisplacement{displacement.x * scale.x, displacement.y / scale.x, displacement.z};
107  const auto scaledSrcRadius = (sourceRadius - displacement.x) * scale.x;
108  const auto scaledDstRadius = (targetRadius - displacement.x) * scale.x;
109  const auto geom = createSDFConePill(scaledSrcPosition, scaledDstPosition, scaledSrcRadius, scaledDstRadius,
110  userDataOffset, scaledDisplacement);
111  _bounds.merge(scaledSrcPosition + scaledSrcRadius);
112  _bounds.merge(scaledSrcPosition - scaledSrcRadius);
113  _bounds.merge(scaledDstPosition + scaledDstRadius);
114  _bounds.merge(scaledDstPosition - scaledDstRadius);
115  return _addSDFGeometry(materialId, geom, neighbours);
116  }
117  if (fabs(sourceRadius - targetRadius) < equalityEpsilon)
118  {
119  const auto scaledSrcRadius = sourceRadius * scale.x;
120  const auto scaledDstRadius = targetRadius * scale.x;
121  _bounds.merge(scaledSrcPosition + scaledSrcRadius);
122  _bounds.merge(scaledSrcPosition - scaledSrcRadius);
123  _bounds.merge(scaledDstPosition + scaledDstRadius);
124  _bounds.merge(scaledDstPosition - scaledDstRadius);
125  return _addCylinder(materialId, {scaledSrcPosition, scaledDstPosition, sourceRadius * scale.x, userDataOffset});
126  }
127 
128  const auto scaledSrcRadius = sourceRadius * scale.x;
129  const auto scaledDstRadius = targetRadius * scale.x;
130  _bounds.merge(scaledSrcPosition + scaledSrcRadius);
131  _bounds.merge(scaledSrcPosition - scaledSrcRadius);
132  _bounds.merge(scaledDstPosition + scaledDstRadius);
133  _bounds.merge(scaledDstPosition - scaledDstRadius);
134  return _addCone(materialId, {scaledSrcPosition, scaledDstPosition, sourceRadius * scale.x, targetRadius * scale.x,
135  userDataOffset});
136 }

228 {

230  for (uint64_t i = 0; i < numGeoms; ++i)

281 void ThreadSafeContainer::_commitSDFGeometriesToModel()
282 {
283  _finalizeSDFGeometries();
284 
285  const uint64_t numGeoms = _sdfMorphologyData.geometries.size();
286  size_ts localToGlobalIndex(numGeoms, 0);
287 
288  // Add geometries to _model. We do not know the indices of the neighbours yet so we leave them empty.
289  for (uint64_t i = 0; i < numGeoms; ++i)
290  localToGlobalIndex[i] =
291  _model.addSDFGeometry(_sdfMorphologyData.materials[i], _sdfMorphologyData.geometries[i], {});
292 
293  // Write the neighbours using global indices
294  uint64_ts neighboursTmp;
295  for (uint64_t i = 0; i < numGeoms; ++i)
296  {
297  const uint64_t globalIndex = localToGlobalIndex[i];
298  neighboursTmp.clear();
299 
300  for (auto localNeighbourIndex : _sdfMorphologyData.neighbours[i])
301  neighboursTmp.push_back(localToGlobalIndex[localNeighbourIndex]);
302 
303  _model.updateSDFGeometryNeighbours(globalIndex, neighboursTmp);
304  }
305 
306  for (const auto materialId : _sdfMorphologyData.materials)
307  _materialIds.insert(materialId);
308  _sdfMorphologyData.geometries.clear();
309  _sdfMorphologyData.neighbours.clear();
310  _sdfMorphologyData.materials.clear();
311 }

313 void ThreadSafeContainer::_commitMeshesToModel()
314 {
315  for (const auto& meshes : _meshesMap)
316  {
317  const auto materialId = meshes.first;
318  _materialIds.insert(materialId);
319  const auto& srcMesh = meshes.second;
320  auto& dstMesh = _model.getTriangleMeshes()[materialId];
321  auto vertexOffset = dstMesh.vertices.size();
322  dstMesh.vertices.insert(dstMesh.vertices.end(), srcMesh.vertices.begin(), srcMesh.vertices.end());
323  auto indexOffset = dstMesh.indices.size();
324  dstMesh.indices.insert(dstMesh.indices.end(), srcMesh.indices.begin(), srcMesh.indices.end());
325  for (uint64_t i = 0; i < srcMesh.indices.size(); ++i)
326  dstMesh.indices[indexOffset + i] += vertexOffset;
327  dstMesh.normals.insert(dstMesh.normals.end(), srcMesh.normals.begin(), srcMesh.normals.end());
328  dstMesh.colors.insert(dstMesh.colors.end(), srcMesh.colors.begin(), srcMesh.colors.end());
329  for (const auto& vertex : srcMesh.vertices)
330  _bounds.merge(vertex);
331  }
332  _meshesMap.clear();
333 }

335 void ThreadSafeContainer::_commitStreamlinesToModel()
336 {
337  const auto materialId = 0;
338  if (_streamlinesMap.find(materialId) == _streamlinesMap.end())
339  return;
340 
341  _materialIds.insert(materialId);
342  auto& modelStreamline = _model.getStreamlines()[materialId];
343  auto modelOffset = modelStreamline.vertex.size();
344 
345  for (const auto& streamline : _streamlinesMap)
346  {
347  modelStreamline.vertex.insert(modelStreamline.vertex.end(), streamline.second.vertex.begin(),
348  streamline.second.vertex.end());
349  modelStreamline.vertexColor.insert(modelStreamline.vertexColor.end(), streamline.second.vertexColor.begin(),
350  streamline.second.vertexColor.end());
351 
352  for (size_t i = 0; i < streamline.second.vertex.size() - 1; ++i)
353  {
354  modelStreamline.indices.push_back(modelOffset + i);
355  _bounds.merge(streamline.second.vertex[i]);
356  }
357  modelOffset += streamline.second.vertex.size();
358  }
359  _streamlinesMap.clear();
360 }
