33 Glycans::Glycans(Scene& scene, const SugarDetails& details)
34  : Molecule(scene, Vector3d(), doublesToQuaterniond(details.rotation), {})
35  , _details(details)
36 {
37  size_t lineIndex{0};
38 
39  std::stringstream lines{_details.contents};
40  std::string line;
41  std::string title;
42  std::string header{details.name};
43 
44  while (getline(lines, line, '\n'))
45  {
46  if (line.find(KEY_ATOM) == 0 || line.find(KEY_HETATM) == 0)
47  _readAtom(line, true);
48  else if (details.loadBonds && line.find(KEY_CONECT) == 0)
49  _readConnect(line);
50  else if (line.find(KEY_HEADER) == 0)
51  header = _readHeader(line);
52  else if (line.find(KEY_TITLE) == 0)
53  title = _readTitle(line);
54  }
55  auto model = scene.createModel();
56 
57  if (_details.recenter)
58  {
59  // Recenter
60  Boxd bounds;
61 
62  // Get current center
63  for (const auto& atom : _atomMap)
64  bounds.merge(atom.second.position);
65  const Vector3d center = bounds.getCenter();
66 
67  const auto firstAtomPosition = _atomMap.begin()->second.position;
68  const auto translation = center - firstAtomPosition;
69 
70  // Translate according to position of first atom
71  for (auto& atom : _atomMap)
72  atom.second.position -= translation;
73  }
74 
75  // Build 3d models according to atoms positions
76  _buildModel(_details.assemblyName, _details.name, _details.pdbId, header, _details.representation,
77  _details.atomRadiusMultiplier, _details.loadBonds);
78 }
