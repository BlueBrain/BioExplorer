59 void WhiteMatter::_buildModel(const LoaderProgress& callback)
60 {
61  if (_modelDescriptor)
62  _scene.removeModel(_modelDescriptor->getModelID());
63 
64  auto model = _scene.createModel();
65  ThreadSafeContainers containers;
66 
67  const auto ompThreads = omp_get_max_threads();
68 
69  const auto nbDBConnections = DBConnector::getInstance().getNbConnections();
70  const auto streamlines =
71  DBConnector::getInstance().getWhiteMatterStreamlines(_details.populationName, _details.sqlFilter);
72 
73  const auto nbStreamlines = streamlines.size();
74  uint64_t index;
75  volatile bool flag = false;
76  std::string flagMessage;
77 #pragma omp parallel for num_threads(nbDBConnections)
78  for (index = 0; index < nbStreamlines; ++index)
79  {
80  try
81  {
82  if (flag)
83  continue;
84 
85  if (omp_get_thread_num() == 0)
86  {
87  PLUGIN_PROGRESS("Loading white matter...", index, nbStreamlines);
88  try
89  {
90  callback.updateProgress("Loading white matter...",
91  0.5f * ((float)index / (float)(nbStreamlines / nbDBConnections)));
92  }
93  catch (...)
94  {
95 #pragma omp critical
96  flag = true;
97  }
98  }
99  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation);
100  StreamlinesData streamline;
101 
102  const float alpha = 1.f;
103  uint64_t j = 0;
104  Vector3f previousPoint;
105  const auto& points = streamlines[index];
106  for (const auto& point : points)
107  {
108  streamline.vertex.push_back({point.x, point.y, point.z, _details.radius});
109  streamline.vertexColor.push_back(
110  (j == 0 ? Vector4f(0.f, 0.f, 0.f, alpha)
111  : Vector4f(0.5f + 0.5f * normalize(point - previousPoint), alpha)));
112  previousPoint = point;
113  ++j;
114  }
115  const size_t materialId = 0;
116  container.addStreamline(materialId, streamline);
117 
118 #pragma omp critical
119  containers.push_back(container);
120  }
121  catch (const std::runtime_error& e)
122  {
123 #pragma omp critical
124  flagMessage = e.what();
125 #pragma omp critical
126  flag = true;
127  }
128  catch (...)
129  {
130 #pragma omp critical
131  flagMessage = "Loading was canceled";
132 #pragma omp critical
133  flag = true;
134  }
135  }
136  for (uint64_t i = 0; i < containers.size(); ++i)
137  {
138  PLUGIN_PROGRESS("- Compiling 3D geometry...", i + 1, containers.size());
139  callback.updateProgress("Compiling 3D geometry...", 0.5f + 0.5f * (float)(1 + i) / (float)containers.size());
140  auto& container = containers[i];
141  container.commitToModel();
142  }
143 
144  const ModelMetadata metadata = {{"Number of streamlines", std::to_string(nbStreamlines)},
145  {"SQL filter", _details.sqlFilter}};
146  _modelDescriptor.reset(new core::ModelDescriptor(std::move(model), _details.assemblyName, metadata));
147  if (!_modelDescriptor)
148  PLUGIN_THROW(
149  "WhiteMatter model could not be created for "
150  "population " +
151  _details.populationName);
152 }
