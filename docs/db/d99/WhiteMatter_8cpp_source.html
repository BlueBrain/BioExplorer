51 void WhiteMatter::_addStreamline(ThreadSafeContainer& container,
52  const Vector3fs& points,
53  const uint64_t materialId)
54 {
55  StreamlinesData streamline;
56 
57  const float alpha = 1.f;
58  uint64_t i = 0;
59  Vector3f previousPoint;
60  for (const auto& point : points)
61  {
62  streamline.vertex.push_back(
63  {point.x, point.y, point.z, _details.radius});
64  streamline.vertexColor.push_back(
65  (i == 0 ? Vector4f(0.f, 0.f, 0.f, alpha)
66  : Vector4f(0.5f + 0.5f * normalize(point - previousPoint),
67  alpha)));
68  previousPoint = point;
69  ++i;
70  }
71 
72  container.addStreamline(materialId, streamline);
73 }

75 void WhiteMatter::_buildModel()
76 {
77  if (_modelDescriptor)
78  _scene.removeModel(_modelDescriptor->getModelID());
79 
80  auto model = _scene.createModel();
81  ThreadSafeContainer container(*model);
82 
83  const auto ompThreads = omp_get_max_threads();
84 
85  const auto streamlines =
86  DBConnector::getInstance().getWhiteMatterStreamlines(
87  _details.populationName, _details.sqlFilter);
88  const auto nbStreamlines = streamlines.size();
89  for (uint64_t i = 0; i < nbStreamlines; ++i)
90  {
91  _addStreamline(container, streamlines[i], i);
92  PLUGIN_PROGRESS("Loading " << i << "/" << nbStreamlines
93  << " streamlines",
94  i, nbStreamlines);
95  }
96 
97  container.commitToModel();
98  PLUGIN_INFO(1, "");
99 
100  const ModelMetadata metadata = {{"Number of streamlines",
101  std::to_string(nbStreamlines)},
102  {"SQL filter", _details.sqlFilter}};
103 
104  _modelDescriptor.reset(new brayns::ModelDescriptor(std::move(model),
105  _details.assemblyName,
106  metadata));
107  if (_modelDescriptor)
108  _scene.addModel(_modelDescriptor);
109  else
110  PLUGIN_THROW(
111  "WhiteMatter model could not be created for "
112  "population " +
113  _details.populationName);
114 }
