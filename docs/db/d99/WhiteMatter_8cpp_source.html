52 void WhiteMatter::_buildModel(const LoaderProgress& callback)
53 {
54  if (_modelDescriptor)
55  _scene.removeModel(_modelDescriptor->getModelID());
56 
57  auto model = _scene.createModel();
58  ThreadSafeContainers containers;
59 
60  const auto ompThreads = omp_get_max_threads();
61 
62  const auto nbDBConnections = DBConnector::getInstance().getNbConnections();
63  const auto streamlines =
64  DBConnector::getInstance().getWhiteMatterStreamlines(_details.populationName, _details.sqlFilter);
65 
66  const auto nbStreamlines = streamlines.size();
67  uint64_t index;
68  volatile bool flag = false;
69  std::string flagMessage;
70 #pragma omp parallel for shared(flag, flagMessage) num_threads(nbDBConnections)
71  for (index = 0; index < nbStreamlines; ++index)
72  {
73  try
74  {
75  if (flag)
76  continue;
77 
78  if (omp_get_thread_num() == 0)
79  {
80  PLUGIN_PROGRESS("Loading white matter...", index, nbStreamlines);
81  try
82  {
83  callback.updateProgress("Loading white matter...",
84  0.5f * ((float)index / (float)(nbStreamlines / nbDBConnections)));
85  }
86  catch (...)
87  {
88 #pragma omp critical
89  {
90  flag = true;
91  }
92  }
93  }
94  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation);
95  StreamlinesData streamline;
96 
97  const float alpha = 1.f;
98  uint64_t j = 0;
99  Vector3f previousPoint;
100  const auto& points = streamlines[index];
101  for (const auto& point : points)
102  {
103  streamline.vertex.push_back({point.x, point.y, point.z, _details.radius});
104  streamline.vertexColor.push_back(
105  (j == 0 ? Vector4f(0.f, 0.f, 0.f, alpha)
106  : Vector4f(0.5f + 0.5f * normalize(point - previousPoint), alpha)));
107  previousPoint = point;
108  ++j;
109  }
110  const size_t materialId = 0;
111  container.addStreamline(materialId, streamline);
112 
113 #pragma omp critical
114  {
115  containers.push_back(container);
116  }
117  }
118  catch (const std::runtime_error& e)
119  {
120 #pragma omp critical
121  {
122  flagMessage = e.what();
123  flag = true;
124  }
125  }
126  catch (...)
127  {
128 #pragma omp critical
129  {
130  flagMessage = "Loading was canceled";
131  flag = true;
132  }
133  }
134  }
135  for (uint64_t i = 0; i < containers.size(); ++i)
136  {
137  PLUGIN_PROGRESS("- Compiling 3D geometry...", i + 1, containers.size());
138  callback.updateProgress("Compiling 3D geometry...", 0.5f + 0.5f * (float)(1 + i) / (float)containers.size());
139  auto& container = containers[i];
140  container.commitToModel();
141  }
142 
143  const ModelMetadata metadata = {{"Number of streamlines", std::to_string(nbStreamlines)},
144  {"SQL filter", _details.sqlFilter}};
145  _modelDescriptor.reset(new core::ModelDescriptor(std::move(model), _details.assemblyName, metadata));
146  if (!_modelDescriptor)
147  PLUGIN_THROW(
148  "WhiteMatter model could not be created for "
149  "population " +
150  _details.populationName);
151 }
152 } // namespace connectomics
153 } // namespace bioexplorer
