45 CompartmentSimulationHandler::CompartmentSimulationHandler(const std::string& populationName,
46  const uint64_t simulationReportId)
47  : core::AbstractSimulationHandler()
48  , _populationName(populationName)
49  , _simulationReportId(simulationReportId)
50 {
51  const auto& connector = DBConnector::getInstance();
52  _simulationReport = connector.getSimulationReport(_populationName, _simulationReportId);
53  _nbFrames = (_simulationReport.endTime - _simulationReport.startTime) / _simulationReport.timeStep;
54  _dt = _simulationReport.timeStep;
55 
56  const auto values = connector.getNeuronCompartmentReportValues(_populationName, _simulationReportId, 0);
57 
58  _frameSize = values.size();
59  _frameData.resize(_frameSize);
60  mempcpy(&_frameData.data()[0], values.data(), sizeof(float) * values.size());
61 
62  PLUGIN_INFO(1, "---------------------------------------------------------");
63  PLUGIN_INFO(1, "Compartment simulation information");
64  PLUGIN_INFO(1, "----------------------------------");
65  PLUGIN_INFO(1, "Population name : " << _populationName);
66  PLUGIN_INFO(1, "Number of simulated nodes: " << _frameSize);
67  PLUGIN_INFO(1, "Start time : " << _simulationReport.startTime);
68  PLUGIN_INFO(1, "End time : " << _simulationReport.endTime);
69  PLUGIN_INFO(1, "Time interval : " << _dt);
70  PLUGIN_INFO(1, "Number of frames : " << _nbFrames);
71  PLUGIN_INFO(1, "---------------------------------------------------------");
72 }

81 void* CompartmentSimulationHandler::getFrameData(const uint32_t frame)
82 {
83  const auto& connector = DBConnector::getInstance();
84  const auto boundedFrame = _getBoundedFrame(frame);
85  if (_currentFrame != boundedFrame)
86  {
87  const auto values = connector.getNeuronCompartmentReportValues(_populationName, _simulationReportId, frame);
88  mempcpy(&_frameData.data()[0], values.data(), sizeof(float) * values.size());
89  _currentFrame = boundedFrame;
90  }
91 
92  return _frameData.data();
93 }
