59 void DBConnector::init(const CommandLineArguments& arguments)
60 {
61  std::string dbHost, dbPort, dbUser, dbPassword, dbName;
62  for (const auto& argument : arguments)
63  {
64  if (argument.first == ARG_DB_HOST)
65  dbHost = argument.second;
66  if (argument.first == ARG_DB_PORT)
67  dbPort = argument.second;
68  if (argument.first == ARG_DB_USER)
69  dbUser = argument.second;
70  if (argument.first == ARG_DB_PASSWORD)
71  dbPassword = argument.second;
72  if (argument.first == ARG_DB_NAME)
73  dbName = argument.second;
74  if (argument.first == ARG_DB_NB_CONNECTIONS)
75  _dbNbConnections = std::stoi(argument.second.c_str());
76  }
77 
78  _connectionString =
79  "host=" + dbHost + " port=" + dbPort + " dbname=" + dbName + " user=" + dbUser + " password=" + dbPassword;
80 
81  PLUGIN_INFO(_connectionString);
82 
83  for (size_t i = 0; i < _dbNbConnections; ++i)
84  {
85  try
86  {
87  _connections.push_back(ConnectionPtr(new pqxx::connection(_connectionString)));
88  }
89  catch (const pqxx::pqxx_exception& e)
90  {
91  PLUGIN_THROW(
92  "Failed to connect to database, check command line "
93  "parameters. " +
94  std::string(e.base().what()));
95  }
96  }
97  _initialized = true;
98  PLUGIN_INFO("Initialized " << _dbNbConnections << " connections to database");
99 }

101 void DBConnector::importCircuitMorphologies(const std::string& populationName, const std::string& source,
102  const std::string& morphologyPath)
103 {
104  CHECK_DB_INITIALIZATION
105 
106  PLUGIN_INFO("- Brion Circuit " << source)
107  const brion::BlueConfig blueConfiguration(source);
108  PLUGIN_INFO("- Brain Circuit")
109  const brain::Circuit circuit(blueConfiguration);
110  PLUGIN_INFO("- GIDs")
111  const auto gids = circuit.getGIDs();
112  PLUGIN_INFO("- " << gids.size() << " URIs")
113  const auto uris = circuit.getMorphologyURIs(gids);
114  brain::neuron::SectionTypes sectionTypes;
115  sectionTypes.push_back(brain::neuron::SectionType::soma);
116  sectionTypes.push_back(brain::neuron::SectionType::axon);
117  sectionTypes.push_back(brain::neuron::SectionType::dendrite);
118  sectionTypes.push_back(brain::neuron::SectionType::apicalDendrite);
119 
120  uint64_t morphologyId = 0;
121 #pragma omp parallel for private(morphologyId) num_threads(_dbNbConnections)
122  for (morphologyId = 0; morphologyId < uris.size(); ++morphologyId)
123  {
124  pqxx::work transaction(*_connections[omp_get_thread_num() % _dbNbConnections]);
125  try
126  {
127  Timer chrono;
128  const std::string path = morphologyPath + uris[morphologyId].getPath();
129  const brion::URI source(path);
130  const brain::neuron::Morphology morphology(source);
131  const auto sections = morphology.getSections(sectionTypes);
132  for (const auto& section : sections)
133  {
134  const Vector4fs& samples = section.getSamples();
135  const pqxx::binarystring points((void*)samples.data(), samples.size() * sizeof(Vector4f));
136  const int sectionId = section.getID();
137  const int parentId = section.hasParent() ? section.getParent().getID() : -1;
138  const int sectionType = static_cast<int>(section.getType());
139  const Vector3f p = samples[0];
140  transaction.exec_params("INSERT INTO " + populationName + ".section VALUES ($1,$2,$3,$4,$5,$6,$7,$8)",
141  morphologyId, sectionId, parentId, sectionType, points, p.x, p.y, p.z);
142  }
143  transaction.commit();
144  }
145  catch (const pqxx::sql_error& e)
146  {
147  transaction.abort();
148  PLUGIN_THROW(e.what());
149  }
150  if (omp_get_thread_num() == 0)
151  {
152  const float progress = (1 + morphologyId) * omp_get_num_threads();
153  PLUGIN_PROGRESS("- Loading cells", progress, uris.size());
154  }
155  }
156 }
