46 /**
47 A smart way to avoid recursion restrictions with OptiX 6 is to use templates!
48 
49 https://www.thanassis.space/cudarenderer-BVH.html#recursion
50 */
51 #define MAX_RECURSION_DEPTH 30
52 
53 template <int depth>
54 __device__ float treeWalker(const int fieldOctreeIndicesId, const int fieldOctreeValuesId, const float3& point,
55  const float distance, const float cutoff, const uint index)
56 {
57  if (depth >= MAX_RECURSION_DEPTH)
58  return 0.f;
59 
60  const uint begin = optix::rtTex1D<uint32_t>(fieldOctreeIndicesId, index * 2);
61  const uint end = optix::rtTex1D<uint32_t>(fieldOctreeIndicesId, index * 2 + 1);
62  const uint idxData = index * FIELD_POINT_DATA_SIZE;
63 
64  if (begin == 0 && end == 0)
65  {
66  // Leaf
67  const float value = optix::rtTex1D<float>(fieldOctreeValuesId, idxData + FIELD_POINT_OFFSET_VALUE);
68  return value / (distance * distance);
69  }
70 
71  float voxelValue = 0.f;
72  for (uint childIndex = begin; childIndex <= end; ++childIndex)
73  {
74  const uint childIdxData = childIndex * FIELD_POINT_DATA_SIZE;
75  const float3 childPosition =
76  make_float3(optix::rtTex1D<float>(fieldOctreeValuesId, childIdxData + FIELD_POINT_OFFSET_POSITION_X),
77  optix::rtTex1D<float>(fieldOctreeValuesId, childIdxData + FIELD_POINT_OFFSET_POSITION_Y),
78  optix::rtTex1D<float>(fieldOctreeValuesId, childIdxData + FIELD_POINT_OFFSET_POSITION_Z));
79  const float d = length(point - childPosition);
80 
81  if (d >= cutoff)
82  {
83  // Child is further than the cutoff distance, no need to evaluate events in the child node, we take the
84  // precomputed value of node instead
85  const float value = optix::rtTex1D<float>(fieldOctreeValuesId, childIdxData + FIELD_POINT_OFFSET_VALUE);
86  voxelValue += value / (d * d);
87  }
88  else
89  // Dive into the child node and compute its contents
90  voxelValue +=
91  treeWalker<depth + 1>(fieldOctreeIndicesId, fieldOctreeValuesId, point, d, cutoff / 2.f, childIndex);
92  }
93  return voxelValue;
94 }

104 template <int depth>
105 __device__ float3 treeWalker3(const int fieldOctreeIndicesId, const int fieldOctreeValuesId, const float3& point,
106  const float distance, const float cutoff, const uint index)
107 {
108  if (depth >= MAX_RECURSION_DEPTH)
109  return make_float3(0.f);
110 
111  const uint begin = optix::rtTex1D<uint32_t>(fieldOctreeIndicesId, index * 2);
112  const uint end = optix::rtTex1D<uint32_t>(fieldOctreeIndicesId, index * 2 + 1);
113  const uint idxData = index * FIELD_VECTOR_DATA_SIZE;
114 
115  if (begin == 0 && end == 0)
116  {
117  // Leaf
118  const float3 vectorDirection =
119  make_float3(optix::rtTex1D<float>(fieldOctreeValuesId, idxData + FIELD_VECTOR_OFFSET_DIRECTION_X),
120  optix::rtTex1D<float>(fieldOctreeValuesId, idxData + FIELD_VECTOR_OFFSET_DIRECTION_Y),
121  optix::rtTex1D<float>(fieldOctreeValuesId, idxData + FIELD_VECTOR_OFFSET_DIRECTION_Z));
122  return vectorDirection / (distance * distance);
123  }
124 
125  float3 voxelValue = make_float3(0.f);
126  for (uint childIndex = begin; childIndex <= end; ++childIndex)
127  {
128  const uint idx = childIndex * FIELD_VECTOR_DATA_SIZE;
129  const float3 childPosition =
130  make_float3(optix::rtTex1D<float>(fieldOctreeValuesId, idx + FIELD_VECTOR_OFFSET_POSITION_X),
131  optix::rtTex1D<float>(fieldOctreeValuesId, idx + FIELD_VECTOR_OFFSET_POSITION_Y),
132  optix::rtTex1D<float>(fieldOctreeValuesId, idx + FIELD_VECTOR_OFFSET_POSITION_Z));
133  const float d = length(point - childPosition);
134 
135  if (d >= cutoff)
136  {
137  // Child is further than the cutoff distance, no need to evaluate events in the child node, we take the
138  // precomputed value of node instead
139  const float3 vectorDirection =
140  make_float3(optix::rtTex1D<float>(fieldOctreeValuesId, idx + FIELD_VECTOR_OFFSET_DIRECTION_X),
141  optix::rtTex1D<float>(fieldOctreeValuesId, idx + FIELD_VECTOR_OFFSET_DIRECTION_Y),
142  optix::rtTex1D<float>(fieldOctreeValuesId, idx + FIELD_VECTOR_OFFSET_DIRECTION_Z));
143  voxelValue += vectorDirection / (d * d);
144  }
145  else
146  // Dive into the child node and compute its contents
147  voxelValue +=
148  treeWalker3<depth + 1>(fieldOctreeIndicesId, fieldOctreeValuesId, point, d, cutoff / 2.f, childIndex);
149  }
150  return voxelValue;
151 }

161 static __device__ float4 get_voxel_value(const int idx, const float3& point)
162 {
163  const int fieldOctreeIndicesId = static_cast<int>(fields[idx + FIELD_OFFSET_OCTREE_INDICES_SAMPLER_ID]);
164  const int fieldOctreeValuesId = static_cast<int>(fields[idx + FIELD_OFFSET_OCTREE_VALUES_SAMPLER_ID]);
165  const int fieldOctreeType = static_cast<int>(fields[idx + FIELD_OFFSET_OCTREE_TYPE]);
166  if (fieldOctreeIndicesId != 0 && fieldOctreeValuesId != 0)
167  {
168  switch (fieldOctreeType)
169  {
170  case OctreeDataType::point:
171  return make_float4(0.f, 0.f, 0.f,
172  treeWalker<0>(fieldOctreeIndicesId, fieldOctreeValuesId, point, fieldDistance,
173  fieldCutoff, 0u));
174  case OctreeDataType::vector:
175  const float3 sampleValue =
176  treeWalker3<0>(fieldOctreeIndicesId, fieldOctreeValuesId, point, fieldDistance, fieldCutoff, 0u);
177  return make_float4(normalize(sampleValue), length(sampleValue));
178  }
179  }
180  return make_float4(0.f);
181 }

183 template <bool use_robust_method>
184 static __device__ void intersect_field(int primIdx)
185 {
186  const int idx = primIdx * field_size;
187  const float3 dimensions = {fields[idx + FIELD_OFFSET_DIMENSIONS], fields[idx + FIELD_OFFSET_DIMENSIONS + 1],
188  fields[idx + FIELD_OFFSET_DIMENSIONS + 2]};
189  const float3 offset = {fields[idx + FIELD_OFFSET_OFFSET], fields[idx + FIELD_OFFSET_OFFSET + 1],
190  fields[idx + FIELD_OFFSET_OFFSET + 2]};
191  const float3 spacing = {fields[idx + FIELD_OFFSET_SPACING], fields[idx + FIELD_OFFSET_SPACING + 1],
192  fields[idx + FIELD_OFFSET_SPACING + 2]};
193  const int transferFunctionSamplerId =
194  static_cast<int>(fields[idx + FIELD_OFFSET_TRANSFER_FUNCTION_TEXTURE_SAMPLER_ID]);
195  const float2 valueRange = {fields[idx + FIELD_OFFSET_VALUE_RANGE], fields[idx + FIELD_OFFSET_VALUE_RANGE + 1]};
196 
197  const float3 boxMin = offset;
198  const float3 boxMax = offset + dimensions * spacing;
199 
200  const float3 a = (boxMin - ray.origin) / ray.direction;
201  const float3 b = (boxMax - ray.origin) / ray.direction;
202  const float3 near = fminf(a, b);
203  const float3 far = fmaxf(a, b);
204  float t0 = fmaxf(near);
205  float t1 = fminf(far);
206 
207  const ::optix::size_t2 screen = output_buffer.size();
208  uint seed = tea<16>(screen.x * launch_index.y + launch_index.x, frame);
209 
210  const float diag = max(spacing.x, max(spacing.y, spacing.z));
211  const float step = max(0.1f, diag / fieldSamplingRate);
212  const float random = rnd(seed) * step;
213 
214  // Apply ray clipping
215  t0 = max(t0, ray.tmin);
216  t1 = min(t1, ray.tmax);
217 
218  if (t0 > 0.f && t0 <= t1)
219  {
220  float t = t0 + random;
221  while (t < t1)
222  {
223  const float3 p = ray.origin + t * ray.direction;
224  const float3 p0 = (p - offset) / spacing;
225  const float4 voxelValue = get_voxel_value(idx, p0);
226  const float4 voxelColor = calcTransferFunctionColor(transferFunctionSamplerId, valueRange, voxelValue.w);
227  if (voxelColor.w > 0.f)
228  if (rtPotentialIntersection(t - sceneEpsilon))
229  {
230  float3 normal = make_float3(voxelValue);
231  if (fieldGradientShadingEnabled)
232  {
233  normal = make_float3(0);
234  const float3 positions[6] = {{-1, 0, 0}, {1, 0, 0}, {0, -1, 0},
235  {0, 1, 0}, {0, 0, -1}, {0, 0, 1}};
236  for (const auto& position : positions)
237  {
238  const float3 p1 = p0 + (position * fieldGradientOffset);
239  const float4 voxelValue = get_voxel_value(idx, p1);
240  normal += voxelValue.w * position;
241  }
242  normal = ::optix::normalize(-1.f * normal);
243  }
244 
245  geometric_normal = shading_normal = normal;
246  userDataIndex = 0;
247  texcoord = make_float2(voxelValue.w, 0.f);
248  texcoord3d = p0;
249  rtReportIntersection(0);
250  break;
251  }
252  t += step;
253  }
254  }
255 }
