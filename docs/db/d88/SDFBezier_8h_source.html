41 inline float bezier(float t, float p0, float c0, float p1, float c1)
42 {
43  float s = 1 - t;
44  float s2 = s * s;
45  float t2 = t * t;
46  return s * s2 * p0 + 3 * s2 * t * c0 + 3 * s * t2 * c1 + t * t2 * p1;
47 }

51 inline Boxd bezierBounds(const SDFBezier& bc)
52 {
53  Boxd bounds;
54 
55  // see https://www.iquilezles.org/www/articles/bezierbbox/bezierbbox.htm
56 
57  // be conservative, use max radii
58  float r = glm::max(bc.r0, bc.r1);
59 
60  // tight bounding box
61  Vector3f mi = glm::min(bc.p0, bc.p1);
62  Vector3f ma = glm::max(bc.p0, bc.p1);
63 
64  const Vector3f c = -1.0f * bc.p0 + bc.c0;
65  const Vector3f b = 1.0f * bc.p0 - 2 * bc.c0 + bc.c1;
66  const Vector3f a = -1.0f * bc.p0 + 3 * bc.c0 - 3 * bc.c1 + bc.p1;
67 
68  Vector3f h = b * b - a * c;
69 
70  if (h.x > 0.0f)
71  {
72  h.x = glm::sqrt(h.x);
73  float t = (-b.x - h.x) / a.x;
74  if (!glm::isnan(t) && t > 0.0f && t < 1.0f)
75  {
76  float q = bezier(t, bc.p0.x, bc.c0.x, bc.p1.x, bc.c1.x);
77  mi.x = glm::min(mi.x, q);
78  ma.x = glm::max(ma.x, q);
79  }
80 
81  t = (-b.x + h.x) / a.x;
82  if (!glm::isnan(t) && t > 0.0f && t < 1.0f)
83  {
84  float q = bezier(t, bc.p0.x, bc.c0.x, bc.p1.x, bc.c1.x);
85  mi.x = glm::min(mi.x, q);
86  ma.x = glm::max(ma.x, q);
87  }
88  }
89 
90  if (h.y > 0.0f)
91  {
92  h.y = glm::sqrt(h.y);
93  float t = (-b.y - h.y) / a.y;
94  if (!glm::isnan(t) && t > 0.0f && t < 1.0f)
95  {
96  float q = bezier(t, bc.p0.y, bc.c0.y, bc.p1.y, bc.c1.y);
97  mi.y = glm::min(mi.y, q);
98  ma.y = glm::max(ma.y, q);
99  }
100  t = (-b.y + h.y) / a.y;
101  if (!glm::isnan(t) && t > 0.0f && t < 1.0f)
102  {
103  float q = bezier(t, bc.p0.y, bc.c0.y, bc.p1.y, bc.c1.y);
104  mi.y = glm::min(mi.y, q);
105  ma.y = glm::max(ma.y, q);
106  }
107  }
108 
109  if (h.z > 0.0f)
110  {
111  h.z = glm::sqrt(h.z);
112  float t = (-b.z - h.z) / a.z;
113  if (!glm::isnan(t) && t > 0.0f && t < 1.0f)
114  {
115  float q = bezier(t, bc.p0.z, bc.c0.z, bc.p1.z, bc.c1.z);
116  mi.z = glm::min(mi.z, q);
117  ma.z = glm::max(ma.z, q);
118  }
119  t = (-b.z + h.z) / a.z;
120  if (!glm::isnan(t) && t > 0.0f && t < 1.0f)
121  {
122  float q = bezier(t, bc.p0.z, bc.c0.z, bc.p1.z, bc.c1.z);
123  mi.z = glm::min(mi.z, q);
124  ma.z = glm::max(ma.z, q);
125  }
126  }
127  bounds.merge(mi - r);
128  bounds.merge(ma + r);
129 
130  return bounds;
131 }
