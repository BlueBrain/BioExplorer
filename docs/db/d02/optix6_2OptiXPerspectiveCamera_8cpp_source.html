45 void OptiXPerspectiveCamera::commit(const OptiXCamera& camera, ::optix::Context context)
46 {
47  auto position = camera.getPosition();
48  const auto stereo = camera.getPropertyOrValue<bool>(CONTEXT_CAMERA_STEREO, false);
49  const auto interpupillaryDistance = camera.getPropertyOrValue<double>(CONTEXT_CAMERA_IPD, 0.0635f);
50  auto aspect = camera.getPropertyOrValue<double>(CONTEXT_CAMERA_ASPECT, 1.0);
51 
52  if (stereo)
53  aspect *= 2.f;
54 
55  const auto up = glm::rotate(camera.getOrientation(), UP_VECTOR);
56 
57  Vector3d u, v, w;
58  float ulen, vlen, wlen;
59  w = camera.getTarget() - position;
60 
61  wlen = glm::length(w);
62  u = normalize(glm::cross(w, up));
63  v = normalize(glm::cross(u, w));
64 
65  vlen = wlen * tanf(0.5f * camera.getPropertyOrValue<double>(CONTEXT_CAMERA_FOVY, 45.0) * M_PI / 180.f);
66  v *= vlen;
67  ulen = vlen * aspect;
68  u *= ulen;
69  const Vector3f ipd_offset = 0.5f * interpupillaryDistance * u;
70 
71  context[CONTEXT_CAMERA_U]->setFloat(u.x, u.y, u.z);
72  context[CONTEXT_CAMERA_V]->setFloat(v.x, v.y, v.z);
73  context[CONTEXT_CAMERA_W]->setFloat(w.x, w.y, w.z);
74 
75  context[CONTEXT_CAMERA_EYE]->setFloat(position.x, position.y, position.z);
76  context[CONTEXT_CAMERA_APERTURE_RADIUS]->setFloat(
77  camera.getPropertyOrValue<double>(CONTEXT_CAMERA_APERTURE_RADIUS, 0.0));
78  context[CONTEXT_CAMERA_FOCAL_SCALE]->setFloat(camera.getPropertyOrValue<double>(CONTEXT_CAMERA_FOCAL_SCALE, 1.0));
79  context[CONTEXT_CAMERA_OFFSET]->setFloat(0, 0);
80 
81  context[CONTEXT_CAMERA_STEREO]->setUint(stereo);
82  context[CONTEXT_CAMERA_IPD_OFFSET]->setFloat(ipd_offset.x, ipd_offset.y, ipd_offset.z);
83 }
