29 float treeWalker(const uniform OctreeVolume* uniform self, const vec3f& point, const varying float distance,
30  const varying float cutoff, const varying int32 index = 0)
31 {
32  const uint64 begin = self->indices[index * 2];
33  const uint64 end = self->indices[index * 2 + 1];
34  if (begin == 0 && end == 0)
35  return self->values[index] / (distance * distance);
36 
37  float voxelValue = 0.f;
38  for (uint64 childIndex = begin; childIndex <= end; ++childIndex)
39  {
40  const uint64 idx = childIndex * 4; // 4 Floats per value (position + value)
41  const vec3f childPosition = make_vec3f(self->values[idx], self->values[idx + 1], self->values[idx + 2]);
42  const float d = length(point - childPosition);
43  if (d >= cutoff)
44  {
45  // Child is further than the cutoff distance, no need to evaluate events in the child node, we take the
46  // precomputed value of node instead
47  voxelValue += self->values[childIndex + 3] / (d * d);
48  }
49  else
50  // Dive into the child node and compute its contents
51  voxelValue += treeWalker(self, point, d, cutoff / 2.f, childIndex);
52  }
53  return voxelValue;
54 }

56 vec3f treeWalker3(const uniform OctreeVolume* uniform self, const vec3f& point, const varying float distance,
57  const varying float cutoff, const varying int32 index = 0)
58 {
59  const uint64 begin = self->indices[index * 2];
60  const uint64 end = self->indices[index * 2 + 1];
61  const uint64 idx = index * 6; // 6 Floats per value (position + direction)
62 
63  if (begin == 0 && end == 0)
64  return make_vec3f(self->values[idx], self->values[idx + 1], self->values[idx + 2]) /
65  (self->distance * self->distance);
66 
67  vec3f voxelValue = make_vec3f(0.f);
68  for (uint64 childIndex = begin; childIndex <= end; ++childIndex)
69  {
70  uint64 idx = childIndex * 3;
71  const vec3f childPosition = make_vec3f(self->values[idx], self->values[idx + 1], self->values[idx + 2]);
72  const float d = length(point - childPosition);
73  if (d >= cutoff)
74  {
75  // Child is further than the cutoff distance, no need to evaluate events in the child node, we take the
76  // precomputed value of node instead
77  const vec3f value = make_vec3f(self->values[idx + 3], self->values[idx + 4], self->values[idx + 5]);
78  voxelValue = voxelValue + value / (d * d);
79  }
80  else
81  // Dive into the child node and compute its contents
82  voxelValue = voxelValue + treeWalker3(self, point, d, cutoff / 2.f, childIndex);
83  }
84  return voxelValue;
85 }

87 float get_voxel_value(const uniform OctreeVolume* uniform self, const vec3f& p)
88 {
89  switch (self->dataType)
90  {
91  case point:
92  return treeWalker(self, p, self->distance, self->cutoff, 0u);
93  case vector:
94  const vec3f sampleValue = treeWalker3(self, p, self->distance, self->cutoff, 0u);
95  return clamp(length(sampleValue), self->transferFunction->valueRange.x, self->transferFunction->valueRange.y);
96  }
97 }
