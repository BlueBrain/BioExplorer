26inline float getRandomValue()
27{
28 float r;
29 int nbMaxTries = 10;
30 while (nbMaxTries >= 0 && rdrand(&r) == false)
31 --nbMaxTries;
32 return r;
33}

35struct PerspectiveStereoCamera
36{

40 vec3f dir_00;

53void PerspectiveStereoCamera_initRay(uniform Camera* uniform _self,
54 varying Ray& ray,
55 const varying CameraSample& sample)
56{
57 uniform PerspectiveStereoCamera* uniform self =
58 (uniform PerspectiveStereoCamera * uniform) _self;
59
60 vec2f screen = sample.screen;
61 vec3f org = self->org;
62
63 if (self->side_by_side)
64 {
65 screen.x *= 2.f;
66 if (screen.x < 1.f)
67 {
68 org = org - self->ipd_offset;
69 }
70 else
71 {
72 org = org + self->ipd_offset;
73 screen.x -= 1.f;
74 }
75 }
76
77 screen = Camera_subRegion(_self, screen);
78
79 vec3f dir =
80 self->dir_00 + screen.x * self->dir_du + screen.y * self->dir_dv;
81
82 if (self->super.doesDOF)
83 {
84 float random = 1.f;
85 random = getRandomValue();
86 random = (random == 0.f ? 0.001f : random);
87 const vec3f llp =
88 uniformSampleDisk(random * self->scaledAperture, sample.lens);
89 // transform local lens point to focal plane (dir_XX are prescaled in
90 // this case)
91 const vec3f lp =
92 (llp.x * self->dir_du) + ((llp.y * self->aspect) * self->dir_dv);
93 org = org + lp;
94 dir = dir - lp;
95 }
96
97 float nearClip = self->super.nearClip;
98 float farClip = inf;
99 const varying vec3f direction = normalize(dir);
100
101 clipRay(self->clipPlanes, self->numClipPlanes, org, direction, nearClip,
102 farClip);
103
104 const float time = Camera_shutterTime(_self, sample.time);
105 setRay(ray, org, direction, nearClip, farClip, time);
106}

108export void* uniform PerspectiveStereoCamera_create(void* uniform cppE)
109{
110 uniform PerspectiveStereoCamera* uniform self =
111 uniform new uniform PerspectiveStereoCamera;
112 self->super.cppEquivalent = cppE;
113 self->super.initRay = PerspectiveStereoCamera_initRay;
114 self->super.doesDOF = false;
115 self->numClipPlanes = 0;
116 return self;
117}

119export void PerspectiveStereoCamera_set(
120 void* uniform _self, const uniform vec3f& org, const uniform vec3f& dir_00,
121 const uniform vec3f& dir_du, const uniform vec3f& dir_dv,
122 const uniform float scaledAperture, const uniform float aspect,
123 const uniform vec3f& ipd_offset, const uniform bool side_by_side,
124 const uniform vec4f clipPlanes[], const uniform unsigned int numClipPlanes)
125{
126 uniform PerspectiveStereoCamera* uniform self =
127 (uniform PerspectiveStereoCamera * uniform) _self;
128 self->org = org;
129 self->dir_00 = dir_00;
130 self->dir_du = dir_du;
131 self->dir_dv = dir_dv;
132 self->scaledAperture = scaledAperture;
133 self->super.doesDOF = scaledAperture > 0.f;
134 self->aspect = aspect;
135 self->side_by_side = side_by_side;
136 self->ipd_offset = ipd_offset;
137 self->clipPlanes = clipPlanes;
138 self->numClipPlanes = numClipPlanes;
139}
