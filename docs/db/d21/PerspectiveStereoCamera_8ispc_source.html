26 inline float getRandomValue()
27 {
28  float r;
29  int nbMaxTries = 10;
30  while (nbMaxTries >= 0 && rdrand(&r) == false)
31  --nbMaxTries;
32  return r;
33 }

53 void PerspectiveStereoCamera_initRay(uniform Camera* uniform _self,
54  varying Ray& ray,
55  const varying CameraSample& sample)
56 {
57  uniform PerspectiveStereoCamera* uniform self =
58  (uniform PerspectiveStereoCamera * uniform) _self;
59 
60  vec2f screen = sample.screen;
61  vec3f org = self->org;
62 
63  if (self->side_by_side)
64  {
65  screen.x *= 2.f;
66  if (screen.x < 1.f)
67  {
68  org = org - self->ipd_offset;
69  }
70  else
71  {
72  org = org + self->ipd_offset;
73  screen.x -= 1.f;
74  }
75  }
76 
77  screen = Camera_subRegion(_self, screen);
78 
79  vec3f dir =
80  self->dir_00 + screen.x * self->dir_du + screen.y * self->dir_dv;
81 
82  if (self->super.doesDOF)
83  {
84  float random = 1.f;
85  random = getRandomValue();
86  random = (random == 0.f ? 0.001f : random);
87  const vec3f llp =
88  uniformSampleDisk(random * self->scaledAperture, sample.lens);
89  // transform local lens point to focal plane (dir_XX are prescaled in
90  // this case)
91  const vec3f lp =
92  (llp.x * self->dir_du) + ((llp.y * self->aspect) * self->dir_dv);
93  org = org + lp;
94  dir = dir - lp;
95  }
96 
97  float nearClip = self->super.nearClip;
98  float farClip = inf;
99  const varying vec3f direction = normalize(dir);
100 
101  clipRay(self->clipPlanes, self->numClipPlanes, org, direction, nearClip,
102  farClip);
103 
104  const float time = Camera_shutterTime(_self, sample.time);
105  setRay(ray, org, direction, nearClip, farClip, time);
106 }
