44 OSPRayEngine::OSPRayEngine(ParametersManager& parametersManager)
45  : Engine(parametersManager)
46 {
47  auto& ap = _parametersManager.getApplicationParameters();
48  try
49  {
50  std::vector<const char*> argv;
51 
52  // Ospray expects but ignores the application name as the first argument
53  argv.push_back("Core");
54 
55  // Setup log and error output
56  argv.push_back("--osp:logoutput");
57  argv.push_back("cout");
58  argv.push_back("--osp:erroroutput");
59  argv.push_back("cerr");
60 
61  if (_parametersManager.getApplicationParameters().getParallelRendering())
62  {
63  argv.push_back("--osp:mpi");
64  }
65 
66  int argc = argv.size();
67  ospInit(&argc, argv.data());
68  }
69  catch (const std::exception& e)
70  {
71  // Note: This is necessary because OSPRay does not yet implement a
72  // ospDestroy API.
73  CORE_ERROR("Error during ospInit(): " << e.what());
74  }
75 
76  for (const auto& module : ap.getOsprayModules())
77  {
78  try
79  {
80  const auto error = ospLoadModule(module.c_str());
81  if (error != OSP_NO_ERROR)
82  throw std::runtime_error(ospDeviceGetLastErrorMsg(ospGetCurrentDevice()));
83  }
84  catch (const std::exception& e)
85  {
86  CORE_ERROR("Error while loading module " << module << ": " << e.what());
87  }
88  }
89 
90  _createRenderers();
91 
92  _scene = std::make_shared<OSPRayScene>(_parametersManager.getAnimationParameters(),
93  _parametersManager.getGeometryParameters(),
94  _parametersManager.getVolumeParameters());
95 
96  _createCameras();
97 
98  _renderer->setScene(_scene);
99  _renderer->setCamera(_camera);
100 }

112 void OSPRayEngine::commit()
113 {
114  Engine::commit();
115 
116  auto device = ospGetCurrentDevice();
117  if (device && _parametersManager.getRenderingParameters().isModified())
118  {
119  const auto useDynamicLoadBalancer = _parametersManager.getApplicationParameters().getDynamicLoadBalancer();
120  if (_useDynamicLoadBalancer != useDynamicLoadBalancer)
121  {
122  ospDeviceSet1i(device, "dynamicLoadBalancer", useDynamicLoadBalancer);
123  ospDeviceCommit(device);
124  _useDynamicLoadBalancer = useDynamicLoadBalancer;
125 
126  CORE_INFO("Using " << (useDynamicLoadBalancer ? "dynamic" : "static") << " load balancer");
127  }
128  }
129 }

136 void OSPRayEngine::_createRenderers()
137 {
138  _renderer = std::make_shared<OSPRayRenderer>(_parametersManager.getAnimationParameters(),
139  _parametersManager.getRenderingParameters());

141  {
142  CORE_INFO("Registering 'advanced' renderer");
143  PropertyMap properties;
144  properties.setProperty({"alphaCorrection", 0.5, 0.001, 1., {"Alpha correction"}});
145  properties.setProperty(
146  {"maxDistanceToSecondaryModel", 30., 0.1, 100., {"Maximum distance to secondary model"}});
147  properties.setProperty({"giDistance", 10000.0, {"Global illumination distance"}});
148  properties.setProperty({"giWeight", 0.0, 1.0, 1.0, {"Global illumination weight"}});
149  properties.setProperty({"giSamples", 0, 0, 64, {"Global illumination samples"}});
150  properties.setProperty({"shadows", 0.0, 0.0, 1.0, {"Shadow intensity"}});
151  properties.setProperty({"softShadows", 0.0, 0.0, 1.0, {"Shadow softness"}});
152  properties.setProperty({"softShadowsSamples", 1, 1, 64, {"Soft shadow samples"}});
153  properties.setProperty({"mainExposure", 1.0, 0.01, 10.0, {"Exposure"}});
154  properties.setProperty({"epsilonFactor", 1.0, 1.0, 1000.0, {"Epsilon factor"}});
155  properties.setProperty({"fogStart", 0.0, 0.0, 1e6, {"Fog start"}});
156  properties.setProperty({"fogThickness", 1e6, 1e6, 1e6, {"Fog thickness"}});
157  properties.setProperty({"maxBounces", 3, 1, 100, {"Maximum number of ray bounces"}});
158  properties.setProperty({"useHardwareRandomizer", false, {"Use hardware accelerated randomizer"}});
159  properties.setProperty({"showBackground", true, {"Show background"}});
160  properties.setProperty({"matrixFilter", false, {"Matrix filter"}});
161  addRendererType("advanced", properties);
162  }
163  {
164  CORE_INFO("Registering 'scivis' renderer");
165  PropertyMap properties;
166  properties.setProperty({"aoDistance", 10000., {"Ambient occlusion distance"}});
167  properties.setProperty({"aoSamples", int32_t(1), int32_t(0), int32_t(128), {"Ambient occlusion samples"}});
168  properties.setProperty({"aoTransparencyEnabled", true, {"Ambient occlusion transparency"}});
169  properties.setProperty({"aoWeight", 0., 0., 1., {"Ambient occlusion weight"}});
170  properties.setProperty({"oneSidedLighting", true, {"One-sided lighting"}});
171  properties.setProperty({"shadowsEnabled", false, {"Shadows"}});
172 
173  addRendererType("scivis", properties);
174  }
175  CORE_INFO("Registering 'basic' renderer");
176  addRendererType("basic");

202 void OSPRayEngine::_createCameras()
203 {
204  _camera = std::make_shared<OSPRayCamera>();
205 
206  const bool isStereo = _parametersManager.getApplicationParameters().isStereo();
207  Property stereoProperty{"stereo", isStereo, {"Stereo"}};
208  Property fovy{"fovy", 45., .1, 360., {"Field of view"}};
209  Property aspect{"aspect", 1., {"Aspect ratio"}};
210  aspect.markReadOnly();
211  Property eyeSeparation{"interpupillaryDistance", 0.0635, {"Eye separation"}};
212  Property enableClippingPlanes{"enableClippingPlanes", true, {"Clipping"}};
213 
214  {
215  PropertyMap properties;
216  properties.setProperty(fovy);
217  properties.setProperty(aspect);
218  properties.setProperty({"apertureRadius", 0., {"Aperture radius"}});
219  properties.setProperty({"focusDistance", 1., {"Focus Distance"}});
220  properties.setProperty({"nearClip", 0., 0., 1e6, {"Near clip"}});
221  properties.setProperty(enableClippingPlanes);
222  properties.setProperty(stereoProperty);
223  properties.setProperty(eyeSeparation);
224  addCameraType("perspective", properties);
225  }
226  {
227  PropertyMap properties;
228  properties.setProperty({"height", 1., {"Height"}});
229  properties.setProperty(aspect);
230  properties.setProperty(enableClippingPlanes);
231  addCameraType("orthographic", properties);
232  }
233  {
234  PropertyMap properties;
235  properties.setProperty(fovy);
236  properties.setProperty(aspect);
237  properties.setProperty(enableClippingPlanes);
238  if (isStereo)
239  {
240  properties.setProperty(stereoProperty);
241  properties.setProperty(eyeSeparation);
242  properties.setProperty({"zeroParallaxPlane", 1., {"Zero parallax plane"}});
243  }
244  addCameraType("perspectiveParallax", properties);
245  }
246  {
247  PropertyMap properties;
248  properties.setProperty(enableClippingPlanes);
249  properties.setProperty({"half", true, {"Half sphere"}});
250  if (isStereo)
251  {
252  properties.setProperty(stereoProperty);
253  properties.setProperty(eyeSeparation);
254  }
255  addCameraType("panoramic", properties);
256  }
257  {
258  PropertyMap properties;
259  properties.setProperty(fovy);
260  properties.setProperty(aspect);
261  properties.setProperty({"apertureRadius", 0., {"Aperture radius"}});
262  properties.setProperty({"focusDistance", 1., {"Focus Distance"}});
263  properties.setProperty(enableClippingPlanes);
264  addCameraType("fisheye", properties);
265  }
266 }
