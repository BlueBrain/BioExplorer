48 OSPRayEngine::OSPRayEngine(ParametersManager& parametersManager)
49  : Engine(parametersManager)
50 {
51  auto& ap = _parametersManager.getApplicationParameters();
52  try
53  {
54  std::vector<const char*> argv;
55 
56  // Ospray expects but ignores the application name as the first argument
57  argv.push_back("Core");
58 
59  // Setup log and error output
60  argv.push_back("--osp:logoutput");
61  argv.push_back("cout");
62  argv.push_back("--osp:erroroutput");
63  argv.push_back("cerr");
64 
65  if (_parametersManager.getApplicationParameters().getParallelRendering())
66  {
67  argv.push_back("--osp:mpi");
68  }
69 
70  int argc = argv.size();
71  ospInit(&argc, argv.data());
72  }
73  catch (const std::exception& e)
74  {
75  // Note: This is necessary because OSPRay does not yet implement a
76  // ospDestroy API.
77  CORE_ERROR("Error during ospInit(): " << e.what());
78  }
79 
80  for (const auto& module : ap.getOsprayModules())
81  {
82  try
83  {
84  const auto error = ospLoadModule(module.c_str());
85  if (error != OSP_NO_ERROR)
86  throw std::runtime_error(ospDeviceGetLastErrorMsg(ospGetCurrentDevice()));
87  }
88  catch (const std::exception& e)
89  {
90  CORE_ERROR("Error while loading module " << module << ": " << e.what());
91  }
92  }
93 
94  _createRenderers();
95 
96  _scene = std::make_shared<OSPRayScene>(_parametersManager.getAnimationParameters(),
97  _parametersManager.getGeometryParameters(),
98  _parametersManager.getVolumeParameters());
99 
100  _createCameras();
101 
102  _renderer->setEngine(this);
103  _renderer->setCamera(_camera);
104 }

116 void OSPRayEngine::commit()
117 {
118  Engine::commit();
119 
120  auto device = ospGetCurrentDevice();
121  if (device && _parametersManager.getRenderingParameters().isModified())
122  {
123  const auto useDynamicLoadBalancer = _parametersManager.getApplicationParameters().getDynamicLoadBalancer();
124  if (_useDynamicLoadBalancer != useDynamicLoadBalancer)
125  {
126  ospDeviceSet1i(device, OSPRAY_ENGINE_PROPERTY_LOAD_BALANCER_DYNAMIC, useDynamicLoadBalancer);
127  ospDeviceCommit(device);
128  _useDynamicLoadBalancer = useDynamicLoadBalancer;
129 
130  PLUGIN_INFO("Using " << (useDynamicLoadBalancer ? "dynamic" : "static") << " load balancer");
131  }
132  }
