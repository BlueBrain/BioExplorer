71 void OptiXFrameBuffer::resize(const Vector2ui& frameSize)
72 {
73  if (_outputBuffer && frameSize == _frameSize)
74  return;
75 
76  if (glm::compMul(frameSize) == 0)
77  throw std::runtime_error("Invalid size for frame buffer resize");
78 
79  _frameSize = frameSize;
80  _cleanup();
81 
82  RTformat format;
83  switch (_frameBufferFormat)
84  {
85  case FrameBufferFormat::rgb_i8:
86  format = RT_FORMAT_UNSIGNED_BYTE3;
87  break;
88  case FrameBufferFormat::rgba_i8:
89  case FrameBufferFormat::bgra_i8:
90  format = RT_FORMAT_UNSIGNED_BYTE4;
91  break;
92  case FrameBufferFormat::rgb_f32:
93  format = RT_FORMAT_FLOAT4;
94  break;
95  default:
96  format = RT_FORMAT_UNKNOWN;
97  }
98 
99  auto context = OptiXContext::get().getOptixContext();
100  _outputBuffer = context->createBuffer(RT_BUFFER_OUTPUT, format, _frameSize.x, _frameSize.y);
101  context[CONTEXT_OUTPUT_BUFFER]->set(_outputBuffer);
102 
103  _accumBuffer = context->createBuffer(RT_BUFFER_INPUT_OUTPUT, RT_FORMAT_FLOAT4, _frameSize.x, _frameSize.y);
104  context[CONTEXT_ACCUMULATION_BUFFER]->set(_accumBuffer);
105 
106  _tonemappedBuffer = context->createBuffer(RT_BUFFER_OUTPUT, RT_FORMAT_FLOAT4, _frameSize.x, _frameSize.y);
107  context[CONTEXT_TONEMAPPED_BUFFER]->set(_tonemappedBuffer);
108 
109  _denoisedBuffer = context->createBuffer(RT_BUFFER_OUTPUT, RT_FORMAT_FLOAT4, _frameSize.x, _frameSize.y);
110  context[CONTEXT_DENOISED_BUFFER]->set(_denoisedBuffer);
111  clear();
112 }

120 void OptiXFrameBuffer::_mapUnsafe()
121 {
122  // Initialization
123  if (!_postprocessingStagesInitialized)
124  _initializePostProcessingStages();
125 
126  rtBufferMap(_outputBuffer->get(), &_colorData);
127 
128  auto context = OptiXContext::get().getOptixContext();
129  context[CONTEXT_FRAME_NUMBER]->setUint(_accumulationFrameNumber);
130 
131  _colorBuffer = (uint8_t*)(_colorData);
132 
133  // Post processing
134  if (_accumulationType == AccumulationType::ai_denoised)
135  {
136  const bool useDenoiser = (_accumulationFrameNumber >= _renderingParameters.getNumNonDenoisedFrames());
137 
138  if (useDenoiser)
139  rtBufferMap(_denoisedBuffer->get(), &_floatData);
140  else
141  rtBufferMap(_tonemappedBuffer->get(), &_floatData);
142 
143  _floatBuffer = (float*)_floatData;
144  }
145  ++_accumulationFrameNumber;
146 }

154 void OptiXFrameBuffer::_unmapUnsafe()
155 {
156  // Post processing stages
157  const bool useDenoiser = (_accumulationFrameNumber >= _renderingParameters.getNumNonDenoisedFrames());
158  if (_accumulationType == AccumulationType::ai_denoised)
159  {
160  if (_commandListWithDenoiser && _commandListWithDenoiserAndToneMapper)
161  {
162  ::optix::Variable(_denoiserStage->queryVariable(CONTEXT_DENOISE_BLEND))
163  ->setFloat(_renderingParameters.getDenoiseBlend());
164  if (useDenoiser)
165  {
166  if (_renderingParameters.getToneMapperExposure() > 0.f)
167  _commandListWithDenoiserAndToneMapper->execute();
168  else
169  _commandListWithDenoiser->execute();
170  }
171  }
172 
173  if (useDenoiser)
174  rtBufferUnmap(_denoisedBuffer->get());
175  rtBufferUnmap(_tonemappedBuffer->get());
176  }
177 
178  rtBufferUnmap(_outputBuffer->get());
179 
180  if (_postprocessingStagesInitialized)
181  {
182  auto context = OptiXContext::get().getOptixContext();
183  context[CONTEXT_TONE_MAPPER_EXPOSURE]->setFloat(_renderingParameters.getToneMapperExposure());
184  context[CONTEXT_TONE_MAPPER_GAMMA]->setFloat(_renderingParameters.getToneMapperGamma());
185  context[CONTEXT_DENOISE_BLEND]->setFloat(_renderingParameters.getDenoiseBlend());
186  }
187 }

198 void OptiXFrameBuffer::_initializePostProcessingStages()
199 {
200  auto context = OptiXContext::get().getOptixContext();
201 
202  _tonemapStage = context->createBuiltinPostProcessingStage(CONTEXT_STAGE_TONE_MAPPER);
203  _tonemapStage->declareVariable(CONTEXT_INPUT_BUFFER)->set(_accumBuffer);
204  _tonemapStage->declareVariable(CONTEXT_OUTPUT_BUFFER)->set(_tonemappedBuffer);
205  _tonemapStage->declareVariable(CONTEXT_TONE_MAPPER_EXPOSURE)
206  ->setFloat(_renderingParameters.getToneMapperExposure());
207  _tonemapStage->declareVariable(CONTEXT_TONE_MAPPER_GAMMA)->setFloat(_renderingParameters.getToneMapperGamma());
208 
209  _denoiserStage = context->createBuiltinPostProcessingStage(CONTEXT_STAGE_DENOISER);
210  _denoiserStage->declareVariable(CONTEXT_INPUT_BUFFER)->set(_accumBuffer);
211  _denoiserStage->declareVariable(CONTEXT_OUTPUT_BUFFER)->set(_denoisedBuffer);
212  _denoiserStage->declareVariable(CONTEXT_DENOISE_BLEND)->setFloat(_renderingParameters.getDenoiseBlend());
213  _denoiserStage->declareVariable(CONTEXT_INPUT_ALBEDO_BUFFER);
214  _denoiserStage->declareVariable(CONTEXT_INPUT_NORMAL_BUFFER);
215 
216  _denoiserWithMappingStage = context->createBuiltinPostProcessingStage(CONTEXT_STAGE_DENOISER);
217  _denoiserWithMappingStage->declareVariable(CONTEXT_INPUT_BUFFER)->set(_tonemappedBuffer);
218  _denoiserWithMappingStage->declareVariable(CONTEXT_OUTPUT_BUFFER)->set(_denoisedBuffer);
219  _denoiserWithMappingStage->declareVariable(CONTEXT_DENOISE_BLEND)->setFloat(_renderingParameters.getDenoiseBlend());
220  _denoiserWithMappingStage->declareVariable(CONTEXT_INPUT_ALBEDO_BUFFER);
221  _denoiserWithMappingStage->declareVariable(CONTEXT_INPUT_NORMAL_BUFFER);
222 
223  // With denoiser
224  _commandListWithDenoiser = context->createCommandList();
225  _commandListWithDenoiser->appendLaunch(0, _frameSize.x, _frameSize.y);
226  _commandListWithDenoiser->appendPostprocessingStage(_denoiserStage, _frameSize.x, _frameSize.y);
227  _commandListWithDenoiser->finalize();
228 
229  // With denoiser and tone mapper
230  _commandListWithDenoiserAndToneMapper = context->createCommandList();
231  _commandListWithDenoiserAndToneMapper->appendLaunch(0, _frameSize.x, _frameSize.y);
232  _commandListWithDenoiserAndToneMapper->appendPostprocessingStage(_tonemapStage, _frameSize.x, _frameSize.y);
233  _commandListWithDenoiserAndToneMapper->appendPostprocessingStage(_denoiserWithMappingStage, _frameSize.x,
234  _frameSize.y);
235  _commandListWithDenoiserAndToneMapper->finalize();
236 
237  _postprocessingStagesInitialized = true;
238 }
