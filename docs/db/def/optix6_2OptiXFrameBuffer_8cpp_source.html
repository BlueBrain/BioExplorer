69 void OptiXFrameBuffer::resize(const Vector2ui& frameSize)
70 {
71  if (_outputBuffer && frameSize == _frameSize)
72  return;
73 
74  if (glm::compMul(frameSize) == 0)
75  throw std::runtime_error("Invalid size for framebuffer resize");
76 
77  _frameSize = frameSize;
78  _cleanup();
79 
80  RTformat format;
81  switch (_frameBufferFormat)
82  {
83  case FrameBufferFormat::rgb_i8:
84  format = RT_FORMAT_UNSIGNED_BYTE3;
85  break;
86  case FrameBufferFormat::rgba_i8:
87  case FrameBufferFormat::bgra_i8:
88  format = RT_FORMAT_UNSIGNED_BYTE4;
89  break;
90  case FrameBufferFormat::rgb_f32:
91  format = RT_FORMAT_FLOAT4;
92  break;
93  default:
94  format = RT_FORMAT_UNKNOWN;
95  }
96 
97  auto context = OptiXContext::get().getOptixContext();
98  _outputBuffer = context->createBuffer(RT_BUFFER_OUTPUT, format, _frameSize.x, _frameSize.y);
99  context[CONTEXT_OUTPUT_BUFFER]->set(_outputBuffer);
100 
101  _accumBuffer = context->createBuffer(RT_BUFFER_INPUT_OUTPUT, RT_FORMAT_FLOAT4, _frameSize.x, _frameSize.y);
102  context[CONTEXT_ACCUMULATION_BUFFER]->set(_accumBuffer);
103 
104  _tonemappedBuffer = context->createBuffer(RT_BUFFER_OUTPUT, RT_FORMAT_FLOAT4, _frameSize.x, _frameSize.y);
105  context[CONTEXT_TONEMAPPED_BUFFER]->set(_tonemappedBuffer);
106 
107  _denoisedBuffer = context->createBuffer(RT_BUFFER_OUTPUT, RT_FORMAT_FLOAT4, _frameSize.x, _frameSize.y);
108  context[CONTEXT_DENOISED_BUFFER]->set(_denoisedBuffer);
109  clear();
110 }

118 void OptiXFrameBuffer::_mapUnsafe()
119 {
120  // Initialization
121  if (!_postprocessingStagesInitialized)
122  _initializePostProcessingStages();
123 
124  rtBufferMap(_outputBuffer->get(), &_colorData);
125 
126  auto context = OptiXContext::get().getOptixContext();
127  context[CONTEXT_FRAME_NUMBER]->setUint(_accumulationFrameNumber);
128 
129  _colorBuffer = (uint8_t*)(_colorData);
130 
131  // Post processing
132  if (_accumulationType == AccumulationType::ai_denoised)
133  {
134  const bool useDenoiser = (_accumulationFrameNumber >= _renderingParameters.getNumNonDenoisedFrames());
135 
136  if (useDenoiser)
137  rtBufferMap(_denoisedBuffer->get(), &_floatData);
138  else
139  rtBufferMap(_tonemappedBuffer->get(), &_floatData);
140 
141  _floatBuffer = (float*)_floatData;
142  }
143  ++_accumulationFrameNumber;
144 }

152 void OptiXFrameBuffer::_unmapUnsafe()
153 {
154  // Post processing stages
155  const bool useDenoiser = (_accumulationFrameNumber >= _renderingParameters.getNumNonDenoisedFrames());
156  if (_accumulationType == AccumulationType::ai_denoised)
157  {
158  if (_commandListWithDenoiser && _commandListWithDenoiserAndToneMapper)
159  {
160  optix::Variable(_denoiserStage->queryVariable(CONTEXT_DENOISE_BLEND))
161  ->setFloat(_renderingParameters.getDenoiseBlend());
162  if (useDenoiser)
163  {
164  if (_renderingParameters.getToneMapperExposure() > 0.f)
165  _commandListWithDenoiserAndToneMapper->execute();
166  else
167  _commandListWithDenoiser->execute();
168  }
169  }
170 
171  if (useDenoiser)
172  rtBufferUnmap(_denoisedBuffer->get());
173  rtBufferUnmap(_tonemappedBuffer->get());
174  }
175 
176  rtBufferUnmap(_outputBuffer->get());
177 
178  if (_postprocessingStagesInitialized)
179  {
180  auto context = OptiXContext::get().getOptixContext();
181  context[CONTEXT_TONE_MAPPER_EXPOSURE]->setFloat(_renderingParameters.getToneMapperExposure());
182  context[CONTEXT_TONE_MAPPER_GAMMA]->setFloat(_renderingParameters.getToneMapperGamma());
183  context[CONTEXT_DENOISE_BLEND]->setFloat(_renderingParameters.getDenoiseBlend());
184  }
185 }

196 void OptiXFrameBuffer::_initializePostProcessingStages()
197 {
198  auto context = OptiXContext::get().getOptixContext();
199 
200  _tonemapStage = context->createBuiltinPostProcessingStage(CONTEXT_STAGE_TONE_MAPPER);
201  _tonemapStage->declareVariable(CONTEXT_INPUT_BUFFER)->set(_accumBuffer);
202  _tonemapStage->declareVariable(CONTEXT_OUTPUT_BUFFER)->set(_tonemappedBuffer);
203  _tonemapStage->declareVariable(CONTEXT_TONE_MAPPER_EXPOSURE)
204  ->setFloat(_renderingParameters.getToneMapperExposure());
205  _tonemapStage->declareVariable(CONTEXT_TONE_MAPPER_GAMMA)->setFloat(_renderingParameters.getToneMapperGamma());
206 
207  _denoiserStage = context->createBuiltinPostProcessingStage(CONTEXT_STAGE_DENOISER);
208  _denoiserStage->declareVariable(CONTEXT_INPUT_BUFFER)->set(_accumBuffer);
209  _denoiserStage->declareVariable(CONTEXT_OUTPUT_BUFFER)->set(_denoisedBuffer);
210  _denoiserStage->declareVariable(CONTEXT_DENOISE_BLEND)->setFloat(_renderingParameters.getDenoiseBlend());
211  _denoiserStage->declareVariable(CONTEXT_INPUT_ALBEDO_BUFFER);
212  _denoiserStage->declareVariable(CONTEXT_INPUT_NORMAL_BUFFER);
213 
214  _denoiserWithMappingStage = context->createBuiltinPostProcessingStage(CONTEXT_STAGE_DENOISER);
215  _denoiserWithMappingStage->declareVariable(CONTEXT_INPUT_BUFFER)->set(_tonemappedBuffer);
216  _denoiserWithMappingStage->declareVariable(CONTEXT_OUTPUT_BUFFER)->set(_denoisedBuffer);
217  _denoiserWithMappingStage->declareVariable(CONTEXT_DENOISE_BLEND)->setFloat(_renderingParameters.getDenoiseBlend());
218  _denoiserWithMappingStage->declareVariable(CONTEXT_INPUT_ALBEDO_BUFFER);
219  _denoiserWithMappingStage->declareVariable(CONTEXT_INPUT_NORMAL_BUFFER);
220 
221  // With denoiser
222  _commandListWithDenoiser = context->createCommandList();
223  _commandListWithDenoiser->appendLaunch(0, _frameSize.x, _frameSize.y);
224  _commandListWithDenoiser->appendPostprocessingStage(_denoiserStage, _frameSize.x, _frameSize.y);
225  _commandListWithDenoiser->finalize();
226 
227  // With denoiser and tone mapper
228  _commandListWithDenoiserAndToneMapper = context->createCommandList();
229  _commandListWithDenoiserAndToneMapper->appendLaunch(0, _frameSize.x, _frameSize.y);
230  _commandListWithDenoiserAndToneMapper->appendPostprocessingStage(_tonemapStage, _frameSize.x, _frameSize.y);
231  _commandListWithDenoiserAndToneMapper->appendPostprocessingStage(_denoiserWithMappingStage, _frameSize.x,
232  _frameSize.y);
233  _commandListWithDenoiserAndToneMapper->finalize();
234 
235  _postprocessingStagesInitialized = true;
236 }
