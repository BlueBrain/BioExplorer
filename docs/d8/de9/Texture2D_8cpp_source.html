58 uint8_t Texture2D::getPossibleMipMapsLevels() const
59 {
60  uint8_t mipMapLevels = 1u;
61  auto nx = width;
62  auto ny = height;
63  while (nx % 2 == 0 && ny % 2 == 0)
64  {
