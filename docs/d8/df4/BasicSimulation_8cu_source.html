53 static __device__ inline void shade(bool textured)
54 {
55  float3 world_shading_normal =
56  optix::normalize(rtTransformNormal(RT_OBJECT_TO_WORLD, shading_normal));
57  float3 world_geometric_normal = optix::normalize(
58  rtTransformNormal(RT_OBJECT_TO_WORLD, geometric_normal));
59 
60  float3 p_normal = optix::faceforward(world_shading_normal, -ray.direction,
61  world_geometric_normal);
62 
63  float3 p_Kd;
64  if (simulation_data.size() > 0)
65  p_Kd = calcTransferFunctionColor(value_range.x, value_range.y,
66  simulation_data[simulation_idx],
67  colors, opacities);
68  else if (textured)
69  p_Kd = make_float3(
70  optix::rtTex2D<float4>(albedoMetallic_map, texcoord.x, texcoord.y));
71  else
72  p_Kd = Kd;
73 
74  prd.result = p_Kd * max(0.f, optix::dot(-ray.direction, p_normal));
75 }
