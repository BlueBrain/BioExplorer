38 OptiXVolume::OptiXVolume(const Vector3ui& dimensions, const Vector3f& spacing, const DataType dataType,
39  const VolumeParameters& params)
40  : Volume(dimensions, spacing, dataType)
41  , _parameters(params)
42 {
43  CORE_INFO("Volume dimension: " << _dimensions);
44  CORE_INFO("Volume spacing : " << _spacing);
45  switch (dataType)
46  {
47  case DataType::INT8:
48  _dataType = RT_FORMAT_BYTE;
49  _dataTypeSize = sizeof(int8_t);
50  CORE_INFO("Volume data type: RT_FORMAT_BYTE");
51  break;
52  case DataType::UINT8:
53  _dataType = RT_FORMAT_UNSIGNED_BYTE;
54  _dataTypeSize = sizeof(uint8_t);
55  CORE_INFO("Volume data type: RT_FORMAT_UNSIGNED_BYTE");
56  break;
57  case DataType::INT16:
58  _dataType = RT_FORMAT_INT;
59  _dataTypeSize = sizeof(int16_t);
60  CORE_INFO("Volume data type: RT_FORMAT_INT");
61  break;
62  case DataType::UINT16:
63  _dataType = RT_FORMAT_UNSIGNED_INT;
64  _dataTypeSize = sizeof(uint16_t);
65  CORE_INFO("Volume data type: RT_FORMAT_UNSIGNED_INT");
66  break;
67  case DataType::FLOAT:
68  _dataType = RT_FORMAT_FLOAT;
69  _dataTypeSize = sizeof(float);
70  CORE_INFO("Volume data type: RT_FORMAT_FLOAT");
71  break;
72  default:
73  throw std::runtime_error("Unsupported voxel type " + std::to_string(int(dataType)));
74  }

85 void OptiXSharedDataVolume::setVoxels(const void* voxels)
86 {
87  const uint64_t volumeSize = _dimensions.x * _dimensions.y * _dimensions.z;
88  floats volumeAsFloats(volumeSize);
89  _valueRange = Vector2f(1e6, -1e6);
90 
91  for (uint64_t i = 0; i < volumeSize; ++i)
92  {
93  switch (_dataType)
94  {
95  case RT_FORMAT_BYTE:
96  {
97  int8_t value;
98  int8_t* v = (int8_t*)voxels;
99  memcpy(&value, v + i, sizeof(int8_t));
100  volumeAsFloats[i] = value;
101  _valueRange.x = std::min(_valueRange.x, (float)value);
102  _valueRange.y = std::max(_valueRange.y, (float)value);
103  break;
104  }
105  case RT_FORMAT_UNSIGNED_BYTE:
106  {
107  uint8_t value;
108  uint8_t* v = (uint8_t*)voxels;
109  memcpy(&value, v + i, sizeof(uint8_t));
110  volumeAsFloats[i] = value;
111  _valueRange.x = std::min(_valueRange.x, (float)value);
112  _valueRange.y = std::max(_valueRange.y, (float)value);
113  break;
114  }
115  case RT_FORMAT_INT:
116  {
117  int16_t value;
118  int16_t* v = (int16_t*)voxels;
119  memcpy(&value, v + i, sizeof(int16_t));
120  volumeAsFloats[i] = value;
121  _valueRange.x = std::min(_valueRange.x, (float)value);
122  _valueRange.y = std::max(_valueRange.y, (float)value);
123  break;
124  }
125  case RT_FORMAT_UNSIGNED_INT:
126  {
127  uint16_t value;
128  uint16_t* v = (uint16_t*)voxels;
129  memcpy(&value, v + i, sizeof(uint16_t));
130  volumeAsFloats[i] = value;
131  _valueRange.x = std::min(_valueRange.x, (float)value);
132  _valueRange.y = std::max(_valueRange.y, (float)value);
133  break;
134  }
135  case RT_FORMAT_FLOAT:
136  {
137  float value;
138  float* v = (float*)voxels;
139  memcpy(&value, v + i, sizeof(float));
140  volumeAsFloats[i] = value;
141  _valueRange.x = std::min(_valueRange.x, (float)value);
142  _valueRange.y = std::max(_valueRange.y, (float)value);
143  break;
144  }
145  }
146  }
147  const size_t bufferSize = volumeAsFloats.size() * sizeof(float);
148  _memoryBuffer.resize(bufferSize);
149  memcpy(_memoryBuffer.data(), volumeAsFloats.data(), bufferSize);
150 }
