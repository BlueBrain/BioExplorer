151 std::string hyphenatedToCamelCase(const std::string& hyphenated)
152 {
153  std::string camel = hyphenated;
154 
155  for (size_t x = 0; x < camel.length(); x++)
156  {
157  if (camel[x] == '-')
158  {
159  std::string tempString = camel.substr(x + 1, 1);
160 
161  transform(tempString.begin(), tempString.end(), tempString.begin(), toupper);
162 
163  camel.erase(x, 2);

210 template <class T, class PRE, class POST>
211 inline bool from_json(
212  T& obj, const std::string& json, PRE preUpdateFunc = [] {}, POST postUpdateFunc = [] {})
213 {
214  staticjson::ParseStatus status;
215 
216  if (!preUpdate<T>(json, preUpdateFunc))
217  return false;
218 
219  const auto success = staticjson::from_json_string(json.c_str(), &obj, &status);
220  if (success)
221  {
222  obj.markModified();
223  if (std::function<void(T&)>(postUpdateFunc))
224  postUpdateFunc(obj);
225  }
226  else
227  CORE_ERROR(status.description());
228  return success;
229 }

643  template <class T>
644  void _handleGET(const std::string& endpoint, T& obj, const int64_t throttleTime = DEFAULT_THROTTLE)
645  {
646  using namespace rockets::http;
647 
648  _rocketsServer->handle(Method::GET, endpoint,
649  [&obj](const Request&)
650  { return make_ready_response(Code::OK, to_json(obj), JSON_TYPE); });
651 
652  _handleObjectSchema(endpoint, obj);
653 
654  const std::string rpcEndpoint = getRequestEndpointName(endpoint);
655 
656  _jsonrpcServer->bind(rpcEndpoint, std::function<const T&()>([&obj]() -> const T& { return obj; }));
657  _handleSchema(rpcEndpoint,
658  buildJsonRpcSchemaRequestReturnOnly({rpcEndpoint, "Get the current state of " + endpoint}, obj));
659 
660  // Create new throttle for that endpoint
661  _throttle[endpoint];
662 
663  obj.onModified(
664  [&, endpoint = getNotificationEndpointName(endpoint), throttleTime](const auto& base)
665  {
666  auto& throttle = _throttle[endpoint];
667 
668  // throttle itself is not thread-safe, but we can get called from different threads (c.f. async model
669  // load)
670  std::lock_guard<std::mutex> lock(throttle.first);
671 
672  const auto& castedObj = static_cast<const T&>(base);
673  const auto notify =
674  [&rocketsServer = _rocketsServer, clientID = _currentClientID, endpoint, json = to_json(castedObj)]
675  {
676  if (rocketsServer->getConnectionCount() == 0)
677  return;
678  try
679  {

794  template <class P, class R>
795  void _handleTask(const RpcParameterDescription& desc,
796  std::function<std::shared_ptr<Task<R>>(P, uintptr_t)> createTask)
797  {
798  // define the action that is executed on every incoming request from the
799  // client:
800  // - create the task that shall be executed
801  // - wire the result of task to the response callback from rockets
802  // - setup progress reporting during the task execution using libuv
803  // - wire the cancel request from rockets to the task
804 
805  auto action = [&, createTask](P params, auto clientID, auto respond, auto progressCb)
806  {
807  // transform task error to rockets error response
808  auto errorCallback = [&, respond](const TaskRuntimeError& error)
809  {
810  const Response response(Response::Error{error.what(), error.code, error.data});
811  this->_delayedNotify([respond, response] { respond(response); });
812  };
813 
814  try
815  {
816  // transform task result to rockets response
817  auto readyCallback = [&, respond](const R& result)
818  {
819  try
820  {
821  this->_delayedNotify([respond, result] { respond({to_json(result)}); });
822  }
823  catch (const std::runtime_error& e)
824  {
825  const Response response(Response::Error{e.what(), TASK_RESULT_TO_JSON_ERROR});
826  this->_delayedNotify([respond, response] { respond(response); });
827  }
828  };
829 
830  // create the task that shall be executed for this request
831  auto task = createTask(std::move(params), clientID);
832 
833  std::function<void()> finishProgress = [task] { task->progress.update("Done", 1.f); };
834 
835 #ifdef USE_NETWORKING
836  // setup periodic progress reporting if we have libuv running
837  if (uvw::Loop::getDefault()->alive())
838  {
839  auto progressUpdate = uvw::Loop::getDefault()->resource<uvw::TimerHandle>();
840 
841  auto sendProgress = [progressCb, &progress = task->progress] { progress.consume(progressCb); };
842  progressUpdate->on<uvw::TimerEvent>([sendProgress](const auto&, auto&) { sendProgress(); });
843 
844  finishProgress = [task, progressUpdate, sendProgress]
845  {
846  task->progress.update("Done", 1.f);
847  sendProgress();
848  progressUpdate->stop();
849  progressUpdate->close();
850  };
851 
852  using ms = std::chrono::milliseconds;
853  progressUpdate->start(ms(0), ms(SLOW_THROTTLE));
854  }
855 #endif
856 
857  // setup the continuation task that handles the result or error
858  // of the task to handle the responses to rockets accordingly.
859  auto responseTask = std::make_shared<async::task<void>>(task->get().then(
860  [&, readyCallback, errorCallback, task, finishProgress](typename Task<R>::Type result)
861  {
862  finishProgress();
863 
864  try
865  {
866  readyCallback(result.get());
867  }
868  catch (const TaskRuntimeError& e)
869  {
870  errorCallback(e);
871  }
872  catch (const std::exception& e)
873  {
874  errorCallback({e.what()});
875  }
876  catch (const async::task_canceled&)
877  {
878  task->finishCancel();
879  }
880 
881 #ifdef USE_NETWORKING
882  if (_processDelayedNotifies)
883  _processDelayedNotifies->send();
884 #endif
885 
886  std::lock_guard<std::mutex> lock(_tasksMutex);
887  _tasks.erase(task);
888  _binaryRequests.removeTask(task);
889  }));
890 
891  std::lock_guard<std::mutex> lock(_tasksMutex);
892  _tasks.emplace(task, responseTask);
893 
894  // forward the cancel request from rockets to the task
895  auto cancel = [task, responseTask](auto done) { task->cancel(done); };
896 
897  task->schedule();
898 
899  return rockets::jsonrpc::CancelRequestCallback(cancel);
900  }
901  // respond errors during the setup of the task
902  catch (const TaskRuntimeError& e)
903  {
904  errorCallback(e);
905  }
906  catch (const std::exception& e)
907  {
908  errorCallback({e.what()});
909  }
910  return rockets::jsonrpc::CancelRequestCallback();
911  };
912  _handleAsyncRPC<P, R>(desc, action);
913  }

1590  FileStats _getFileStats(const std::string& path)
1591  {
1592  FileStats ft;
1593  ft.type = "none";
1594  ft.error = 0;
1595  ft.sizeBytes = 0;
1596 
1597  // Make sure the requested file is within the sandbox path
1598  // TODO: Unhardcode sandbox path and add as braynsService input param
1599  const std::string& sandboxPath = _engine.getParametersManager().getApplicationParameters().getSandboxPath();
1600  size_t pos = path.find(sandboxPath);
1601  if (pos != 0)
1602  {
1603  ft.error = 1;
1604  ft.message = "Path falls outside the sandbox: " + sandboxPath;
1605  return ft;
1606  }
1607 
1608  if (path.find("../") == 0 || path.find("/../") != std::string::npos || path.rfind("/..") == 4)
1609  {
1610  ft.error = 3;
1611  ft.message = "Illegal path detected";
1612  return ft;
1613  }
1614 
1615  struct stat s;
1616  int err = stat(path.c_str(), &s);
1617  if (err == 0)
1618  {
1619  // Path is a regular file
1620  if (s.st_mode & S_IFREG)
1621  {
1622  // Test permissions with fopen
1623  FILE* permissionTest = fopen(path.c_str(), "r");
1624  int openError = errno;
1625  if (permissionTest != nullptr)
1626  fclose(permissionTest);
1627  if (openError == EACCES)
1628  {
1629  ft.error = 2;
1630  ft.message = "Permission for " + path + " denied";
1631  return ft;
1632  }
1633 
1634  // If we reached this point, file is accessible
1635  ft.type = "file";
1636  ft.sizeBytes = s.st_size;
1637  }
1638  // Path is a folder
1639  else if (s.st_mode & S_IFDIR)
1640  {
1641  // Test permissions with access
1642  if (access(path.c_str(), X_OK) < 0)
1643  {
1644  if (errno == EACCES)
1645  {
1646  ft.error = 2;
1647  ft.message = "Permission for " + path + " denied";
1648  return ft;
1649  }
1650  else
1651  {
1652  ft.error = 3;
1653  ft.message = "Unknown error";
1654  return ft;
1655  }
1656  }
1657 
1658  // If we reached this point, folder is accessible
1659  ft.type = "directory";
1660  }
1661  else
1662  {
1663  ft.message = "Unknown type of element";
1664  }
1665  }
1666  else if (err > 0)
1667  {
1668  if (err == EACCES)
1669  {
1670  ft.error = 2;
1671  ft.message = "Permission for " + path + " denied";
1672  }
1673  else
1674  {
1675  ft.error = 3;

1677  }
1678  }
1679 
1680  return ft;
