150 std::string hyphenatedToCamelCase(const std::string& hyphenated)
151 {
152  std::string camel = hyphenated;
153 
154  for (size_t x = 0; x < camel.length(); x++)
155  {
156  if (camel[x] == '-')
157  {
158  std::string tempString = camel.substr(x + 1, 1);
159 
160  transform(tempString.begin(), tempString.end(), tempString.begin(), toupper);
161 
162  camel.erase(x, 2);

189 bool preUpdate(const std::string& json, PRE preUpdateFunc,
190  typename std::enable_if<!std::is_abstract<T>::value>::type* = 0)
191 {
192  if (std::function<bool(const T&)>(preUpdateFunc))

196  return false;
197  if (!preUpdateFunc(temp))
198  return false;
199  }
200  return true;

209 template <class T, class PRE, class POST>
210 inline bool from_json(
211  T& obj, const std::string& json, PRE preUpdateFunc = [] {}, POST postUpdateFunc = [] {})
212 {
213  staticjson::ParseStatus status;
214 
215  if (!preUpdate<T>(json, preUpdateFunc))
216  return false;
217 
218  const auto success = staticjson::from_json_string(json.c_str(), &obj, &status);
219  if (success)
220  {
221  obj.markModified();
222  if (std::function<void(T&)>(postUpdateFunc))
223  postUpdateFunc(obj);
224  }
225  else
226  CORE_ERROR(status.description());
227  return success;
228 }

643  template <class T>
644  void _handleGET(const std::string& endpoint, T& obj, const int64_t throttleTime = DEFAULT_THROTTLE)
645  {
646  using namespace rockets::http;
647 
648  _rocketsServer->handle(Method::GET, endpoint,
649  [&obj](const Request&)
650  { return make_ready_response(Code::OK, to_json(obj), JSON_TYPE); });
651 
652  _handleObjectSchema(endpoint, obj);
653 
654  const std::string rpcEndpoint = getRequestEndpointName(endpoint);
655 
656  _jsonrpcServer->bind(rpcEndpoint, std::function<const T&()>([&obj]() -> const T& { return obj; }));
657  _handleSchema(rpcEndpoint,
658  buildJsonRpcSchemaRequestReturnOnly({rpcEndpoint, "Get the current state of " + endpoint}, obj));
659 
660  // Create new throttle for that endpoint
661  _throttle[endpoint];
662 
663  obj.onModified(
664  [&, endpoint = getNotificationEndpointName(endpoint), throttleTime](const auto& base)
665  {
666  auto& throttle = _throttle[endpoint];
667 
668  // throttle itself is not thread-safe, but we can get called
669  // from different threads (c.f. async model load)
670  std::lock_guard<std::mutex> lock(throttle.first);
671 
672  const auto& castedObj = static_cast<const T&>(base);
673  const auto notify =
674  [&rocketsServer = _rocketsServer, clientID = _currentClientID, endpoint, json = to_json(castedObj)]
675  {
676  if (rocketsServer->getConnectionCount() == 0)
677  return;
678  try
679  {
680  const auto& msg = rockets::jsonrpc::makeNotification(endpoint, json);
681  if (clientID == NO_CURRENT_CLIENT)
682  rocketsServer->broadcastText(msg);
683  else
684  rocketsServer->broadcastText(msg, {clientID});
685  }

785  template <class P, class R>
786  void _handleTask(const RpcParameterDescription& desc,
787  std::function<std::shared_ptr<Task<R>>(P, uintptr_t)> createTask)
788  {
789  // define the action that is executed on every incoming request from the
790  // client:
791  // - create the task that shall be executed
792  // - wire the result of task to the response callback from rockets
793  // - setup progress reporting during the task execution using libuv
794  // - wire the cancel request from rockets to the task
795 
796  auto action = [&, createTask](P params, auto clientID, auto respond, auto progressCb)
797  {
798  // transform task error to rockets error response
799  auto errorCallback = [&, respond](const TaskRuntimeError& error)
800  {
801  const Response response(Response::Error{error.what(), error.code, error.data});
802  this->_delayedNotify([respond, response] { respond(response); });
803  };
804 
805  try
806  {
807  // transform task result to rockets response
808  auto readyCallback = [&, respond](const R& result)
809  {
810  try
811  {
812  this->_delayedNotify([respond, result] { respond({to_json(result)}); });
813  }
814  catch (const std::runtime_error& e)
815  {
816  const Response response(Response::Error{e.what(), TASK_RESULT_TO_JSON_ERROR});
817  this->_delayedNotify([respond, response] { respond(response); });
818  }
819  };
820 
821  // create the task that shall be executed for this request
822  auto task = createTask(std::move(params), clientID);
823 
824  std::function<void()> finishProgress = [task] { task->progress.update("Done", 1.f); };
825 
826 // setup periodic progress reporting if we have libuv running
827 #ifdef BRAYNS_USE_LIBUV
828  if (uvw::Loop::getDefault()->alive())
829  {
830  auto progressUpdate = uvw::Loop::getDefault()->resource<uvw::TimerHandle>();
831 
832  auto sendProgress = [progressCb, &progress = task->progress] { progress.consume(progressCb); };
833  progressUpdate->on<uvw::TimerEvent>([sendProgress](const auto&, auto&) { sendProgress(); });
834 
835  finishProgress = [task, progressUpdate, sendProgress]
836  {
837  task->progress.update("Done", 1.f);
838  sendProgress();
839  progressUpdate->stop();
840  progressUpdate->close();
841  };
842 
843  using ms = std::chrono::milliseconds;
844  progressUpdate->start(ms(0), ms(SLOW_THROTTLE));
845  }
846 #endif
847 
848  // setup the continuation task that handles the result or error
849  // of the task to handle the responses to rockets accordingly.
850  auto responseTask = std::make_shared<async::task<void>>(task->get().then(
851  [&, readyCallback, errorCallback, task, finishProgress](typename Task<R>::Type result)
852  {
853  finishProgress();
854 
855  try
856  {
857  readyCallback(result.get());
858  }
859  catch (const TaskRuntimeError& e)
860  {
861  errorCallback(e);
862  }
863  catch (const std::exception& e)
864  {
865  errorCallback({e.what()});
866  }
867  catch (const async::task_canceled&)
868  {
869  task->finishCancel();
870  }
871 
872 #ifdef BRAYNS_USE_LIBUV
873  if (_processDelayedNotifies)
874  _processDelayedNotifies->send();
875 #endif
876 
877  std::lock_guard<std::mutex> lock(_tasksMutex);
878  _tasks.erase(task);
879  _binaryRequests.removeTask(task);
880  }));
881 
882  std::lock_guard<std::mutex> lock(_tasksMutex);
883  _tasks.emplace(task, responseTask);
884 
885  // forward the cancel request from rockets to the task
886  auto cancel = [task, responseTask](auto done) { task->cancel(done); };
887 
888  task->schedule();
889 
890  return rockets::jsonrpc::CancelRequestCallback(cancel);
891  }
892  // respond errors during the setup of the task
893  catch (const TaskRuntimeError& e)
894  {
895  errorCallback(e);
896  }
897  catch (const std::exception& e)
898  {
899  errorCallback({e.what()});
900  }
901  return rockets::jsonrpc::CancelRequestCallback();
902  };
903  _handleAsyncRPC<P, R>(desc, action);
904  }

1583  FileStats _getFileStats(const std::string& path)
1584  {
1585  FileStats ft;
1586  ft.type = "none";
1587  ft.error = 0;
1588  ft.sizeBytes = 0;
1589 
1590  // Make sure the requested file is within the sandbox path
1591  // TODO: Unhardcode sandbox path and add as braynsService input param
1592  const std::string& sandboxPath = _engine.getParametersManager().getApplicationParameters().getSandboxPath();
1593  size_t pos = path.find(sandboxPath);
1594  if (pos != 0)
1595  {
1596  ft.error = 1;
1597  ft.message = "Path falls outside the sandbox: " + sandboxPath;
1598  return ft;
1599  }
1600 
1601  if (path.find("../") == 0 || path.find("/../") != std::string::npos || path.rfind("/..") == 4)
1602  {
1603  ft.error = 3;
1604  ft.message = "Illegal path detected";
1605  return ft;
1606  }
1607 
1608  struct stat s;
1609  int err = stat(path.c_str(), &s);
1610  if (err == 0)
1611  {
1612  // Path is a regular file
1613  if (s.st_mode & S_IFREG)
1614  {
1615  // Test permissions with fopen
1616  FILE* permissionTest = fopen(path.c_str(), "r");
1617  int openError = errno;
1618  if (permissionTest != nullptr)
1619  fclose(permissionTest);
1620  if (openError == EACCES)
1621  {
1622  ft.error = 2;
1623  ft.message = "Permission for " + path + " denied";
1624  return ft;
1625  }
1626 
1627  // If we reached this point, file is accessible
1628  ft.type = "file";
1629  ft.sizeBytes = s.st_size;
1630  }
1631  // Path is a folder
1632  else if (s.st_mode & S_IFDIR)
1633  {
1634  // Test permissions with access
1635  if (access(path.c_str(), X_OK) < 0)
1636  {
1637  if (errno == EACCES)
1638  {
1639  ft.error = 2;
1640  ft.message = "Permission for " + path + " denied";
1641  return ft;
1642  }
1643  else
1644  {
1645  ft.error = 3;
1646  ft.message = "Unknown error";
1647  return ft;
1648  }
1649  }
1650 
1651  // If we reached this point, folder is accessible
1652  ft.type = "directory";
1653  }
1654  else
1655  {
1656  ft.message = "Unknown type of element";
1657  }
1658  }
1659  else if (err > 0)
1660  {
1661  if (err == EACCES)
1662  {
1663  ft.error = 2;
1664  ft.message = "Permission for " + path + " denied";
1665  }
1666  else
1667  {
1668  ft.error = 3;

1671  }

1673  return ft;
