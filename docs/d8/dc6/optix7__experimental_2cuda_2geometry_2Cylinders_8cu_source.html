33 extern "C" __global__ void __intersection__cylinder()
34 {
35  const GeometryData::Cylinder* cylinder =
36  reinterpret_cast<GeometryData::Cylinder*>(optixGetSbtDataPointer());
37 
38  const float3 v0 = cylinder->center;
39  const float3 v1 = cylinder->up;
40  const float radius = cylinder->radius;
41 
42  const float3 ray_orig = optixGetWorldRayOrigin();
43  const float3 ray_dir = optixGetWorldRayDirection();
44  const float ray_tmin = optixGetRayTmin();
45  const float ray_tmax = optixGetRayTmax();
46 
47  const float3 A = v0 - ray_orig;
48  const float3 B = v1 - ray_orig;
49 
50  const float3 O = make_float3(0.f);
51  const float3 V = ray_dir;
52 
53  const float3 AB = B - A;
54  const float3 AO = O - A;
55 
56  const float3 AOxAB = cross(AO, AB);
57  const float3 VxAB = cross(V, AB);
58  const float ab2 = dot(AB, AB);
59  const float a = dot(VxAB, VxAB);
60  const float b = 2.f * dot(VxAB, AOxAB);
61  const float c = dot(AOxAB, AOxAB) - (radius * radius * ab2);
62 
63  const float radical = b * b - 4.f * a * c;
64  if (radical >= 0.f)
65  {
66  // clip to near and far cap of cylinder
67  const float tA = dot(AB, A) / dot(V, AB);
68  const float tB = dot(AB, B) / dot(V, AB);
69  // const float tAB0 = max( 0.f, min( tA, tB ));
70  // const float tAB1 = min( RT_DEFAULT_MAX, max( tA, tB ));
71  const float tAB0 = min(tA, tB);
72  const float tAB1 = max(tA, tB);
73 
74  const float srad = sqrt(radical);
75 
76  const float t = (-b - srad) / (2.f * a);
77 
78  bool check_second = true;
79  float3 normal;
80  if (t >= tAB0 && t <= tAB1 && t > ray_tmin && t < ray_tmax)
81  {
82  const float3 P = ray_orig + t * ray_dir - v0;
83  const float3 V = cross(P, AB);
84  normal = cross(AB, V);
85  if (optixReportIntersection(t, 0, float3_as_uints(normal),
86  __float_as_uint(radius)))
87  check_second = false;
88  }
89 
90  if (check_second)
91  {
92  const float t = (-b + srad) / (2.f * a);
93  if (t >= tAB0 && t <= tAB1 && t > ray_tmin && t < ray_tmax)
94  {
95  const float3 P = t * ray_dir - A;
96  const float3 V = cross(P, AB);
97  normal = cross(AB, V);
98  optixReportIntersection(t, 0, float3_as_uints(normal),
99  __float_as_uint(radius));
100  }
101  }
102  }
103 }
