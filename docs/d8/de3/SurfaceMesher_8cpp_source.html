66ModelDescriptorPtr SurfaceMesher::generateSurface(brayns::Scene& scene,
67 const std::string& title,
68 const Vector4fs& atoms,
69 const double shrinkfactor)
70{
71 ModelDescriptorPtr modelDescriptor{nullptr};
72 MeshLoader meshLoader(scene);
73 const std::string filename =
74 GeneralSettings::getInstance()->getOffFolder() +
75 title.substr(title.find("_") + 1) + ".off";
76 try
77 {
78 PLUGIN_INFO("Trying to load surface from cache " << filename);
79 modelDescriptor =
80 meshLoader.importFromFile(filename, LoaderProgress(), {});
81 _setMaterialExtraAttributes(modelDescriptor);
82 PLUGIN_INFO("Surface loaded from cache " << filename);
83 return modelDescriptor;
84 }
85 catch (const std::runtime_error& e)
86 {
87 PLUGIN_INFO("Failed to load surface from cache ("
88 << e.what() << "), constructing it...");
89 }
90
91 std::list<Weighted_point> l;
92 for (const auto& atom : atoms)
93 l.push_front(Weighted_point(Point_3(atom.x, atom.y, atom.z), atom.w));
94
95 PLUGIN_INFO("Constructing skin surface from " << l.size() << " atoms");
96
97 Polyhedron polyhedron;
98 Skin_surface_3 skinSurface(l.begin(), l.end(), shrinkfactor);
99
100 PLUGIN_INFO("Meshing skin surface...");
101 CGAL::mesh_skin_surface_3(skinSurface, polyhedron);
102 CGAL::Polygon_mesh_processing::triangulate_faces(polyhedron);
103
104 PLUGIN_INFO("Adding mesh to model");
105 std::ofstream out(filename);
106 out << polyhedron;
107 modelDescriptor = meshLoader.importFromFile(filename, LoaderProgress(), {});
108 _setMaterialExtraAttributes(modelDescriptor);
109 return modelDescriptor;
110}

112ModelDescriptorPtr SurfaceMesher::generateUnionOfBalls(brayns::Scene& scene,
113 const std::string& title,
114 const Vector4fs& atoms)
115{
116 std::list<Weighted_point> l;
117 for (const auto& atom : atoms)
118 l.push_front(Weighted_point(Point_3(atom.x, atom.y, atom.z), atom.w));
119
120 ModelDescriptorPtr modelDescriptor{nullptr};
121 MeshLoader meshLoader(scene);
122 const std::string filename =
123 GeneralSettings::getInstance()->getOffFolder() +
124 title.substr(title.find("_") + 1) + ".off";
125 try
126 {
127 PLUGIN_INFO("Trying to load union of balls from cache " << filename);
128 modelDescriptor =
129 meshLoader.importFromFile(filename, LoaderProgress(), {});
130 _setMaterialExtraAttributes(modelDescriptor);
131 PLUGIN_INFO("Surface loaded from cache " << filename);
132 return modelDescriptor;
133 }
134 catch (const std::runtime_error& e)
135 {
136 PLUGIN_INFO("Failed to load union of balls from cache ("
137 << e.what() << "), constructing it...");
138 }
139
140 PLUGIN_INFO("Constructing union of balls from " << l.size() << " atoms");
141
142 Polyhedron polyhedron;
143 Union_of_balls_3 union_of_balls(l.begin(), l.end());
144 CGAL::mesh_union_of_balls_3(union_of_balls, polyhedron);
145
146 PLUGIN_INFO("Adding mesh to model");
147 std::ofstream out(filename);
148 out << polyhedron;
149 modelDescriptor = meshLoader.importFromFile(filename, LoaderProgress(), {});
150 _setMaterialExtraAttributes(modelDescriptor);
151 return modelDescriptor;
152}

154void SurfaceMesher::_setMaterialExtraAttributes(
155 ModelDescriptorPtr modelDescriptor)
156{
157 auto materials = modelDescriptor->getModel().getMaterials();
158 for (auto& material : materials)
159 {
160 brayns::PropertyMap props;
161 props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
162 static_cast<int>(MaterialShadingMode::basic)});
163 props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
164 props.setProperty({MATERIAL_PROPERTY_CHAMELEON_MODE,
165 static_cast<int>(MaterialChameleonMode::receiver)});
166 props.setProperty({MATERIAL_PROPERTY_NODE_ID, static_cast<int>(_uuid)});
167 material.second->updateProperties(props);
168 }
169}
