61 void FlyingModeManipulator::dragRight(const Vector2i& to, const Vector2i& from)
62 {
63  const float distance = -(to.y - from.y) * DEFAULT_MOUSE_MOTION_SPEED_MULTIPLIER * getMotionSpeed();
