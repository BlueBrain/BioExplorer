27 static __device__ inline void shade()
28 {
29  optix::size_t2 screen = output_buffer.size();
30  uint seed = tea<16>(screen.x * launch_index.y + launch_index.x, frame);
31 
32  const float3 hit_point = ray.origin + t_hit * ray.direction;
33  const float3 world_shading_normal = optix::normalize(rtTransformNormal(RT_OBJECT_TO_WORLD, shading_normal));
34  const float3 world_geometric_normal = optix::normalize(rtTransformNormal(RT_OBJECT_TO_WORLD, geometric_normal));
35  float3 normal = optix::faceforward(world_shading_normal, -ray.direction, world_geometric_normal);
36 
37  // Glossiness
38  if (glossiness < 1.f)
39  normal = optix::normalize(normal + (1.f - glossiness) *
40  make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f));
41 
42  // Diffuse
43  float attenuation = 0.f;
44  for (int i = 0; i < giSamples; ++i)
45  {
46  float3 aa_normal = optix::normalize(make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f));
47  if (::optix::dot(aa_normal, normal) < 0.f)
48  aa_normal = -aa_normal;
49 
50  PerRayData_shadow shadow_prd;
51  shadow_prd.attenuation = make_float3(1.f);
52 
53  float near = sceneEpsilon;
54  float far = giRayLength;
55  applyClippingPlanes(hit_point, aa_normal, near, far);
56 
57  ::optix::Ray shadow_ray(hit_point, aa_normal, shadowRayType, near, far);
58  rtTrace(top_shadower, shadow_ray, shadow_prd);
59 
60  attenuation += ::optix::luminance(shadow_prd.attenuation);
61  }
62  float3 color = make_float3((attenuation) / float(giSamples));
63 
64  const float opacity = fmaxf(Ko);
65  if (opacity > 0.f && prd.depth < maxRayDepth - 1)
66  {
67  const float reflection = fmaxf(Kr);
68  // Reflection
69  if (reflection > 0.f)
70  {
71  PerRayData_radiance reflected_prd;
72  reflected_prd.result = make_float4(0.f);
73  reflected_prd.importance = prd.importance * reflection;
74  reflected_prd.depth = prd.depth + 1;
75 
76  const float3 R = optix::reflect(ray.direction, normal);
77  const optix::Ray reflected_ray(hit_point, R, radianceRayType, sceneEpsilon, giRayLength);
78  rtTrace(top_object, reflected_ray, reflected_prd);
79  color = color * (1.f - reflection) + Kr * make_float3(reflected_prd.result);
80  }
81 
82  // Refraction
83  if (opacity < 1.f)
84  {
85  PerRayData_radiance refracted_prd;
86  refracted_prd.result = make_float4(0.f);
87  refracted_prd.importance = prd.importance * (1.f - opacity);
88  refracted_prd.depth = prd.depth + 1;
89 
90  const float3 refractedNormal = refractedVector(ray.direction, normal, refraction_index, 1.f);
91  const optix::Ray refracted_ray(hit_point, refractedNormal, radianceRayType, sceneEpsilon, giRayLength);
92  rtTrace(top_object, refracted_ray, refracted_prd);
93  color = color * opacity + (1.f - opacity) * make_float3(refracted_prd.result);
94  }
95  }
96 
97  prd.result = make_float4(::optix::clamp(color, 0.f, 1.f), opacity);
98 }
