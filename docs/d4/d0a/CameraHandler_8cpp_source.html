77 void CameraHandler::_buildCameraPath()
78 {
79  Vector3ds origins;
80  Vector3ds directions;
81  Vector3ds ups;
82  doubles aperture_radii;
83  doubles focus_distances;
84  _smoothedKeyFrames.clear();
85  const double numberOfSmoothingSteps = static_cast<double>(_numberOfSmoothingSteps);
86  const double stepsBetweenKeyFrames = static_cast<double>(_stepsBetweenKeyFrames);
87 
88  for (uint64_t s = 0; s < _keyFrames.size() - 1; ++s)
89  {
90  const auto& p0 = _keyFrames[s];
91  const auto& p1 = _keyFrames[s + 1];
92 
93  for (uint64_t i = 0; i < _stepsBetweenKeyFrames; ++i)
94  {
95  Vector3d origin = {0, 0, 0};
96  Vector3d direction = {0, 0, 0};
97  Vector3d up = {0, 0, 0};
98 
99  origin = p0.origin + (p1.origin - p0.origin) * static_cast<double>(i) / stepsBetweenKeyFrames;
100  direction = p0.direction + (p1.direction - p0.direction) * static_cast<double>(i) / stepsBetweenKeyFrames;
101  up = p0.up + (p1.up - p0.up) * static_cast<double>(i) / stepsBetweenKeyFrames;
102 
103  float aperture_radius = p0.apertureRadius + (p1.apertureRadius - p0.apertureRadius) *
104  static_cast<double>(i) / stepsBetweenKeyFrames;
105  float focus_distance = p0.focalDistance + (p1.focalDistance - p0.focalDistance) * static_cast<double>(i) /
106  stepsBetweenKeyFrames;
107 
108  origins.push_back(origin);
109  directions.push_back(direction);
110  ups.push_back(up);
111  aperture_radii.push_back(aperture_radius);
112  focus_distances.push_back(focus_distance);
113  }
114  }
115 
116  size_t nb_frames = origins.size();
117  for (uint64_t i = 0; i < nb_frames; ++i)
118  {
119  Vector3d o;
120  Vector3d d;
121  Vector3d u;
122  double aperture_radius = 0.0;
123  double focus_distance = 0.0;
124 
125  for (int64_t j = 0; j < _numberOfSmoothingSteps; ++j)
126  {
127  const int64_t index =
128  std::max(0l, std::min(static_cast<int64_t>(i) + j - static_cast<int64_t>(_numberOfSmoothingSteps) / 2,
129  static_cast<int64_t>(nb_frames)));
130  o += origins[index];
131  d += directions[index];
132  u += ups[index];
133  aperture_radius += aperture_radii[index];
134  focus_distance += focus_distances[index];
135  }
136 
137  _smoothedKeyFrames.push_back({o / numberOfSmoothingSteps, d / numberOfSmoothingSteps,
138  u / numberOfSmoothingSteps, aperture_radius / numberOfSmoothingSteps,
139  focus_distance / numberOfSmoothingSteps});
140  }
141 }
