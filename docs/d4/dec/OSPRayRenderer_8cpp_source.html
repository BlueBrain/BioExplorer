67 void OSPRayRenderer::commit()
68 {
69  const AnimationParameters& ap = _animationParameters;
70  const RenderingParameters& rp = _renderingParameters;
71  OSPRayScene& scene = static_cast<OSPRayScene&>(_engine->getScene());
72  const bool lightsChanged = (_currLightsData != scene.lightData());
73  const bool rendererChanged = (_currentOSPRenderer != getCurrentType());
74 
75  if (!ap.isModified() && !rp.isModified() && !scene.isModified() && !isModified() && !_camera->isModified() &&
76  !lightsChanged && !rendererChanged)
77  {
78  return;
79  }
80 
81  if (rendererChanged)
82  _createOSPRenderer();
83 
84  toOSPRayProperties(*this, _renderer);
85 
86  if (lightsChanged || rendererChanged)
87  {
88  ospSetData(_renderer, RENDERER_PROPERTY_LIGHTS, scene.lightData());
89  _currLightsData = scene.lightData();
90  }
91 
92  if (isModified() || rendererChanged || scene.isModified())
93  {
94  _commitRendererMaterials();
95 
96  if (auto simulationModel = scene.getSimulatedModel())
97  {
98  auto& model = static_cast<OSPRayModel&>(simulationModel->getModel());
99  ospSetObject(_renderer, RENDERER_PROPERTY_SECONDARY_MODEL, model.getSecondaryModel());
100  ospSetData(_renderer, RENDERER_PROPERTY_USER_DATA, model.simulationData());
101  ospSetObject(_renderer, DEFAULT_COMMON_TRANSFER_FUNCTION, model.transferFunction());
102  }
103  else
104  {
105  // ospRemoveParam leaks objects, so we set it to null first
106  ospSetData(_renderer, RENDERER_PROPERTY_USER_DATA, nullptr);
107  ospRemoveParam(_renderer, RENDERER_PROPERTY_USER_DATA);
108  }
109 
110  // Setting the clip planes in the renderer and the camera
111  Planes planes;
112  for (const auto& clipPlane : scene.getClipPlanes())
113  planes.push_back(clipPlane->getPlane());
114 
115  setClipPlanes(planes);
116  _camera->setClipPlanes(planes);
117  }
118  // Anaglyph property is handled by the camera, but processed by the renderer
119  Vector3f anaglyphIpdOffset;
120  bool anaglyphEnabled = false;
121  if (_camera->getCurrentType() == CAMERA_PROPERTY_TYPE_ANAGLYPH)
122  {
123  const Vector3d dir = _camera->getOrientation() * UP_VECTOR;
124  const double d = dot(dir, UP_VECTOR);
125  const Vector3d dir_du = (d > 0.999 ? Vector3d(1, 0, 0) : normalize(cross(dir, UP_VECTOR))); // Avoid gimble lock
126  const double interpupillaryDistance =
127  _camera->getProperty<double>(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str());
128  anaglyphIpdOffset = 0.5f * interpupillaryDistance * dir_du;
129  anaglyphEnabled = true;
130  }
131 
132  osphelper::set(_renderer, OSPRAY_RENDERER_PROPERTY_ANAGLYPH_IPD_OFFSET, anaglyphIpdOffset);
133  osphelper::set(_renderer, OSPRAY_RENDERER_PROPERTY_ANAGLYPH_ENABLED, static_cast<int>(anaglyphEnabled));
134 
135  _camera->commit();
136 
137  osphelper::set(_renderer, RENDERER_PROPERTY_TIMESTAMP.name.c_str(), static_cast<float>(ap.getFrame()));
138  osphelper::set(_renderer, RENDERER_PROPERTY_RANDOM_NUMBER, rand() % 10000);
139  osphelper::set(_renderer, OSPRAY_RENDERER_PROPERTY_VARIANCE_THRESHOLD,
140  static_cast<float>(rp.getVarianceThreshold()));
141  osphelper::set(_renderer, OSPRAY_RENDERER_PROPERTY_SAMPLES_PER_PIXEL, static_cast<int>(_spp));
142 
143  if (auto material = std::static_pointer_cast<OSPRayMaterial>(scene.getBackgroundMaterial()))
144  {
145  material->setDiffuseColor(_backgroundColor);
146  material->commit(_currentOSPRenderer);
147  ospSetObject(_renderer, RENDERER_PROPERTY_BACKGROUND_MATERIAL, material->getOSPMaterial());
148  }
149 
150  ospSetObject(_renderer, OSPRAY_RENDERER_PROPERTY_CAMERA, _camera->impl());
151  ospSetObject(_renderer, OSPRAY_RENDERER_PROPERTY_WORLD, scene.getModel());
152 
153  // Clip planes
154  if (!_clipPlanes.empty())
155  {
156  const auto clipPlanes = convertVectorToFloat(_clipPlanes);
157  auto clipPlaneData = ospNewData(clipPlanes.size(), OSP_FLOAT4, clipPlanes.data());
158  ospSetData(_renderer, CAMERA_PROPERTY_CLIPPING_PLANES, clipPlaneData);
159  ospRelease(clipPlaneData);
160  }
161  else
162  {
163  // ospRemoveParam leaks objects, so we set it to null first
164  ospSetData(_renderer, CAMERA_PROPERTY_CLIPPING_PLANES, nullptr);
165  ospRemoveParam(_renderer, CAMERA_PROPERTY_CLIPPING_PLANES);
166  }
167 
168  ospCommit(_renderer);
169 }

180 Renderer::PickResult OSPRayRenderer::pick(const Vector2f& pickPos)
181 {
182  OSPPickResult ospResult;
183  osp::vec2f pos{pickPos.x, pickPos.y};
184 
185  // HACK: as the time for picking is set to 0.5 and interpolated in a (default) 0..0 range, the ray.time will be 0.
186  // So all geometries that have a time > 0 (like branches that have distance to the soma for the growing use-case),
187  // cannot be picked. So we make the range as large as possible to make ray.time be as large as possible.
188  osphelper::set(_camera->impl(), OSPRAY_RENDERER_PROPERTY_SHUTTER_CLOSE, INFINITY);
189  ospCommit(_camera->impl());
190 
191  ospPick(&ospResult, _renderer, pos);
192 
193  // UNDO HACK
194  osphelper::set(_camera->impl(), OSPRAY_RENDERER_PROPERTY_SHUTTER_CLOSE, 0.f);
195  ospCommit(_camera->impl());
196 
197  PickResult result;
198  result.hit = ospResult.hit;
199  if (result.hit)
200  {
201  result.pos = {ospResult.position.x, ospResult.position.y, ospResult.position.z};
202  CORE_INFO("OSPRAY Depth: " << pickPos << " - " << result.pos);
203  }
204  return result;
205 }

207 void OSPRayRenderer::_createOSPRenderer()
208 {
209  auto newRenderer = ospNewRenderer(getCurrentType().c_str());
210  if (!newRenderer)
211  throw std::runtime_error(getCurrentType() + " is not a registered renderer");
212  _destroyRenderer();
213  _renderer = newRenderer;
214  if (_camera)
215  ospSetObject(_renderer, OSPRAY_RENDERER_PROPERTY_CAMERA, _camera->impl());
216  _currentOSPRenderer = getCurrentType();
217  markModified(false);
218 }
