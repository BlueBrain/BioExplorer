73 void OSPRayRenderer::commit()
74 {
75  const AnimationParameters& ap = _animationParameters;
76  const RenderingParameters& rp = _renderingParameters;
77  OSPRayScene& scene = static_cast<OSPRayScene&>(_engine->getScene());
78  const bool lightsChanged = (_currLightsData != scene.lightData());
79  const bool rendererChanged = (_currentOSPRenderer != getCurrentType());
80 
81  if (!ap.isModified() && !rp.isModified() && !scene.isModified() && !isModified() && !_camera->isModified() &&
82  !lightsChanged && !rendererChanged)
83  {
84  return;
85  }
86 
87  if (rendererChanged)
88  _createOSPRenderer();
89 
90  toOSPRayProperties(*this, _renderer);
91 
92  if (lightsChanged || rendererChanged)
93  {
94  ospSetData(_renderer, RENDERER_PROPERTY_LIGHTS, scene.lightData());
95  _currLightsData = scene.lightData();
96  }
97 
98  if (isModified() || rendererChanged || scene.isModified())
99  {
100  _commitRendererMaterials();
101 
102  if (auto simulationModel = scene.getSimulatedModel())
103  {
104  auto& model = static_cast<OSPRayModel&>(simulationModel->getModel());
105  ospSetObject(_renderer, RENDERER_PROPERTY_SECONDARY_MODEL, model.getSecondaryModel());
106  ospSetData(_renderer, RENDERER_PROPERTY_USER_DATA, model.simulationData());
107  ospSetObject(_renderer, RENDERER_PROPERTY_TRANSFER_FUNCTION, model.transferFunction());
108  }
109  else
110  {
111  // ospRemoveParam leaks objects, so we set it to null first
112  ospSetData(_renderer, RENDERER_PROPERTY_USER_DATA, nullptr);
113  ospRemoveParam(_renderer, RENDERER_PROPERTY_USER_DATA);
114  }
115 
116  // Setting the clip planes in the renderer and the camera
117  Planes planes;
118  for (const auto& clipPlane : scene.getClipPlanes())
119  planes.push_back(clipPlane->getPlane());
120 
121  setClipPlanes(planes);
122  _camera->setClipPlanes(planes);
123  }
124  // Anaglyph property is handled by the camera, but processed by the renderer
125  Vector3f anaglyphIpdOffset;
126  bool anaglyphEnabled = false;
127  if (_camera->getCurrentType() == CAMERA_PROPERTY_TYPE_ANAGLYPH)
128  {
129  const Vector3d dir = _camera->getOrientation() * UP_VECTOR;
130  const double d = dot(dir, UP_VECTOR);
131  const Vector3d dir_du = (d > 0.999 ? Vector3d(1, 0, 0) : normalize(cross(dir, UP_VECTOR))); // Avoid gimble lock
132  const double interpupillaryDistance =
133  _camera->getProperty<double>(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str());
134  anaglyphIpdOffset = 0.5f * interpupillaryDistance * dir_du;
135  anaglyphEnabled = true;
136  }
137 
138  osphelper::set(_renderer, OSPRAY_RENDERER_PROPERTY_ANAGLYPH_IPD_OFFSET, anaglyphIpdOffset);
139  osphelper::set(_renderer, OSPRAY_RENDERER_PROPERTY_ANAGLYPH_ENABLED, static_cast<int>(anaglyphEnabled));
140 
141  _camera->commit();
142 
143  osphelper::set(_renderer, RENDERER_PROPERTY_TIMESTAMP.name.c_str(), static_cast<float>(ap.getFrame()));
144  osphelper::set(_renderer, RENDERER_PROPERTY_RANDOM_NUMBER, rand() % 10000);
145  osphelper::set(_renderer, OSPRAY_RENDERER_PROPERTY_VARIANCE_THRESHOLD,
146  static_cast<float>(rp.getVarianceThreshold()));
147  osphelper::set(_renderer, OSPRAY_RENDERER_PROPERTY_SAMPLES_PER_PIXEL, static_cast<int>(_spp));
148 
149  if (auto material = std::static_pointer_cast<OSPRayMaterial>(scene.getBackgroundMaterial()))
150  {
151  material->setDiffuseColor(_backgroundColor);
152  material->commit(_currentOSPRenderer);
153  ospSetObject(_renderer, RENDERER_PROPERTY_BACKGROUND_MATERIAL, material->getOSPMaterial());
154  }
155 
156  ospSetObject(_renderer, OSPRAY_RENDERER_PROPERTY_CAMERA, _camera->impl());
157  ospSetObject(_renderer, OSPRAY_RENDERER_PROPERTY_WORLD, scene.getModel());
158 
159  // Clip planes
160  if (!_clipPlanes.empty())
161  {
162  const auto clipPlanes = convertVectorToFloat(_clipPlanes);
163  auto clipPlaneData = ospNewData(clipPlanes.size(), OSP_FLOAT4, clipPlanes.data());
164  ospSetData(_renderer, CAMERA_PROPERTY_CLIPPING_PLANES, clipPlaneData);
165  ospRelease(clipPlaneData);
166  }
167  else
168  {
169  // ospRemoveParam leaks objects, so we set it to null first
170  ospSetData(_renderer, CAMERA_PROPERTY_CLIPPING_PLANES, nullptr);
171  ospRemoveParam(_renderer, CAMERA_PROPERTY_CLIPPING_PLANES);
172  }
173 
174  ospCommit(_renderer);
175 }

186 Renderer::PickResult OSPRayRenderer::pick(const Vector2f& pickPos)
187 {
188  OSPPickResult ospResult;
189  osp::vec2f pos{pickPos.x, pickPos.y};
190 
191  // HACK: as the time for picking is set to 0.5 and interpolated in a (default) 0..0 range, the ray.time will be 0.
192  // So all geometries that have a time > 0 (like branches that have distance to the soma for the growing use-case),
193  // cannot be picked. So we make the range as large as possible to make ray.time be as large as possible.
194  osphelper::set(_camera->impl(), OSPRAY_RENDERER_PROPERTY_SHUTTER_CLOSE, INFINITY);
195  ospCommit(_camera->impl());
196 
197  ospPick(&ospResult, _renderer, pos);
198 
199  // UNDO HACK
200  osphelper::set(_camera->impl(), OSPRAY_RENDERER_PROPERTY_SHUTTER_CLOSE, 0.f);
201  ospCommit(_camera->impl());
202 
203  PickResult result;
204  result.hit = ospResult.hit;
205  if (result.hit)
206  result.pos = {ospResult.position.x, ospResult.position.y, ospResult.position.z};
207  return result;
208 }

210 void OSPRayRenderer::_createOSPRenderer()
211 {
212  auto newRenderer = ospNewRenderer(getCurrentType().c_str());
213  if (!newRenderer)
214  throw std::runtime_error(getCurrentType() + " is not a registered renderer");
215  _destroyRenderer();
216  _renderer = newRenderer;
217  if (_camera)
218  ospSetObject(_renderer, OSPRAY_RENDERER_PROPERTY_CAMERA, _camera->impl());
219  _currentOSPRenderer = getCurrentType();
220  markModified(false);
221 }
