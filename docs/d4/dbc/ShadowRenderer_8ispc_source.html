29 inline vec3f ShadowRenderer_shadeRay(const uniform ShadowRenderer* uniform self, varying ScreenSample& sample)
30 {
31  Ray ray = sample.ray;
32  sample.z = inf;
33  sample.alpha = 0.f;
34 
35  const uniform int sampleCnt = self->softness == 0.f ? 1 : self->samplesPerFrame;
36  const uniform int accumID = reduce_max(sample.sampleID.z) * self->samplesPerFrame;
37 
38  traceRay(self->super.model, ray);
39 
40  if (ray.geomID < 0)
41  // No intersection, use environment map or background color
42  return make_vec3f(0.f);
43 
44  sample.z = ray.t;
45  sample.alpha = 1.f;
46 
47  DifferentialGeometry dg;
48  postIntersect(self->super.model, dg, ray, DG_NS | DG_NG | DG_NORMALIZE | DG_FACEFORWARD);
49 
50  // init TEA RNG //
51  uniform FrameBuffer* uniform fb = self->super.fb;
52  RandomTEA rng_state;
53  varying RandomTEA* const uniform rng = &rng_state;
54  RandomTEA__Constructor(rng, 0x290374, (fb->size.x * sample.sampleID.y) + sample.sampleID.x);
55  const vec2f rot = RandomTEA__getFloats(rng);
56 
57  int hits = 0;
58  vec3f superColor = make_vec3f(1.f);
59 
60  for (uniform int l = 0; l < self->numLights; l++)
61  {
62  const uniform Light* uniform light = self->lights[l];
63 
64  for (uniform int i = 0; i < sampleCnt; i++)
65  {
66  const vec2f halton = HaltonSequence_get2D(sampleCnt * accumID + i + l);
67  const vec2f samplingLocation =
68  self->softness * (make_vec2f(0.5f) + 0.5f * CranleyPattersonRotation(halton, rot));
69  const varying Light_SampleRes lightSample = light->sample(light, dg, samplingLocation);
70  const vec3f lightDirection = lightSample.dir;
71  const linear3f localToWorld = frame(lightDirection);
72 
73  Ray shadow_ray;
74  setRay(shadow_ray, dg.P, lightDirection, dg.epsilon, self->rayLength);
75  traceRay(self->super.model, shadow_ray);
76  if (shadow_ray.geomID >= 0)
77  hits++;
78  }
79  }
80 
81  return superColor * make_vec3f(1.0f - hits / (float)(sampleCnt * self->numLights));
82 }
