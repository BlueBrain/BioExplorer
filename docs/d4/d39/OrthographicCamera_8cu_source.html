27 // Pass 'seed' by reference to keep randomness state
28 __device__ float4 launch(uint& seed, const float2 screen, const bool use_randomness)
29 {
30  // Subpixel jitter: send the ray through a different position inside the
31  // pixel each time, to provide anti-aliasing.
32  const float2 subpixel_jitter =
33  use_randomness ? make_float2(rnd(seed) - 0.5f, rnd(seed) - 0.5f) : make_float2(0.f, 0.f);
34 
35  // Normalized pixel position (from -0.5 to 0.5)
36  const float2 p = (make_float2(launch_index) + subpixel_jitter) / screen * 2.f - 1.f;
37 
38  const float3 ray_origin = W + p.x * U + p.y * V;
39  const float3 ray_direction = dir;
40 
41  PerRayData_radiance prd;
42  prd.importance = 1.f;
43  prd.depth = 0;
44  prd.zDepth = INFINITY;
45 
46  float near = sceneEpsilon;
47  float far = INFINITY;
48 
49  // Clipping planes
50  if (enableClippingPlanes)
51  applyClippingPlanes(ray_origin, ray_direction, near, far);
52 
53  const optix::Ray ray(ray_origin, ray_direction, radiance_ray_type, near, far);
54  rtTrace(top_object, ray, prd);
55 
56  return make_float4(make_float3(prd.result) * mainExposure, prd.result.w);
57 }

59 RT_PROGRAM void orthographicCamera()
60 {
61  const size_t2 screen = output_buffer.size();
62  const float2 screen_f = make_float2(screen);
63 
64  uint seed = tea<16>(screen.x * launch_index.y + launch_index.x, frame);
65 
66  const int num_samples = max(1, samples_per_pixel);
67  // We enable randomness if we are using subpixel sampling or accumulation
68  const bool use_randomness = frame > 0 || num_samples > 1;
69  float4 result = make_float4(0.f);
70  for (int i = 0; i < num_samples; i++)
71  result += launch(seed, screen_f, use_randomness);
72  result /= num_samples;
73 
74  float4 acc_val;
75  if (frame > 0)
76  {
77  acc_val = accum_buffer[launch_index];
78  acc_val = lerp(acc_val, result, 1.0f / static_cast<float>(frame + 1));
79  }
80  else
81  acc_val = result;
82 
83  output_buffer[launch_index] = make_color(acc_val);
84 
85  if (accum_buffer.size().x > 1 && accum_buffer.size().y > 1)
86  accum_buffer[launch_index] = acc_val;
87 }
