38 MeshShape::MeshShape(const Vector4ds& clippingPlanes, const Vector3d& scale,
39  const std::string& contents)
40  : Shape(clippingPlanes)
41 {
42  // Load mesh
43  Assimp::Importer importer;
44  const aiScene* aiScene =
45  importer.ReadFileFromMemory(contents.c_str(), contents.length(),
46  aiProcess_GenSmoothNormals |
47  aiProcess_Triangulate);
48 
49  if (!aiScene)
50  PLUGIN_THROW(importer.GetErrorString());
51 
52  if (!aiScene->HasMeshes())
53  PLUGIN_THROW("No mesh found");
54 
55  // Add protein instances according to membrane topology
56  for (size_t m = 0; m < aiScene->mNumMeshes; ++m)
57  {
58  aiMesh* mesh = aiScene->mMeshes[m];
59 
60  // Determine mesh center
61  Vector3d meshCenter{0.0, 0.0, 0.0};
62  for (uint64_t i = 0; i < mesh->mNumVertices; ++i)
63  meshCenter += _toVector3d(mesh->mVertices[i]);
64  meshCenter /= mesh->mNumVertices;
65 
66  // Recenter mesh and store transformed vertices
67  for (uint64_t i = 0; i < mesh->mNumVertices; ++i)
68  {
69  const auto v = _toVector3d(mesh->mVertices[i], meshCenter, scale);
70  _vertices.push_back(v);
71  _bounds.merge(v);
72  }
73 
74  // Store faces
75  for (size_t f = 0; f < mesh->mNumFaces; ++f)
76  if (mesh->mFaces[f].mNumIndices == 3)
77  {
78  const Vector3ui face(mesh->mFaces[f].mIndices[0],
79  mesh->mFaces[f].mIndices[1],
80  mesh->mFaces[f].mIndices[2]);
81 
82  _faces.push_back(face);
83  const auto faceSurface =
84  _getSurfaceArea(_vertices[face.x], _vertices[face.y],
85  _vertices[face.z]);
86  _faceSurfaces.push_back(faceSurface);
87  _surface += faceSurface;
88  }
89 
90  // Store normals
91  _normals.resize(_vertices.size());
92  if (mesh->HasNormals())
93  {
94  for (const auto& face : _faces)
95  {
96  auto normal =
97  glm::normalize(_toVector3d(mesh->mNormals[face.x]));
98  _normals[face.x] = normal;
99  normal = glm::normalize(_toVector3d(mesh->mNormals[face.y]));
100  _normals[face.y] = normal;
101  normal = glm::normalize(_toVector3d(mesh->mNormals[face.z]));
102  _normals[face.z] = normal;
103  }
104  }
105  else
106  for (const auto& face : _faces)
107  {
108  const auto v0 =
109  glm::normalize(_vertices[face.y] - _vertices[face.x]);
110  const auto v1 =
111  glm::normalize(_vertices[face.z] - _vertices[face.x]);
112  const auto normal = glm::cross(v0, v1);
113  _normals[face.x] = normal;
114  _normals[face.y] = normal;
115  _normals[face.z] = normal;
116  }
117 
118  PLUGIN_INFO(3, "----=== MeshBasedMembrane ===----");
119  PLUGIN_INFO(3, "Scale : " << scale);
120  PLUGIN_INFO(3, "Number of faces : " << _faces.size());
121  PLUGIN_INFO(3, "Mesh surface area : " << _surface);
122  PLUGIN_INFO(3, "Bounds : " << _bounds);
123  }
124 }

126 Transformation MeshShape::getTransformation(
127  const uint64_t occurrence, const uint64_t nbOccurrences,
128  const MolecularSystemAnimationDetails& MolecularSystemAnimationDetails,
129  const double offset) const
130 {
131  if (occurrence == 0)
132  {
133  _faceIndex = 0;
134  _instanceCoveringProcess = 0.0;
135  _surfaceCoveringProcess = _faceSurfaces[_faceIndex];
136  }
137 
138  const double instanceSurface = _surface / nbOccurrences;
139  const double expectedSurfaceCovering = instanceSurface * occurrence;
140 
141  if (_instanceCoveringProcess > _surfaceCoveringProcess)
142  {
143  while (_surfaceCoveringProcess < expectedSurfaceCovering)
144  {
145  ++_faceIndex;
146  _surfaceCoveringProcess += _faceSurfaces[_faceIndex];
147  }
148  }
149  _instanceCoveringProcess += instanceSurface;
150 
151  const auto face = _faces[_faceIndex];
152 
153  Vector2d coordinates{1.0, 1.0};
154  while (coordinates.x + coordinates.y > 1.0)
155  {
156  coordinates.x = 0.5f + rnd1();
157  coordinates.y = 0.5f + rnd1();
158  }
159 
160  const auto v00 = _vertices[face.y] - _vertices[face.x];
161  const auto v01 = _vertices[face.z] - _vertices[face.x];
162  Vector3d pos =
163  _vertices[face.x] + v00 * coordinates.x + v01 * coordinates.y;
164 
165  // Clipping planes
166  if (isClipped(pos, _clippingPlanes))
167  throw std::runtime_error("Instance is clipped");
168 
169  // Normals
170  const auto v10 = _vertices[face.x] - pos;
171  const auto v11 = _vertices[face.y] - pos;
172  const auto v12 = _vertices[face.z] - pos;
173  const Vector3d areas{0.5 * length(glm::cross(v11, v12)),
174  0.5 * length(glm::cross(v10, v12)),
175  0.5 * length(glm::cross(v10, v11))};
176 
177  const auto n0 = _normals[face.x];
178  const auto n1 = _normals[face.y];
179  const auto n2 = _normals[face.z];
180 
181  const Vector3d normal = glm::normalize(
182  Vector4d(glm::normalize((n0 * areas.x + n1 * areas.y + n2 * areas.z) /
183  (areas.x + areas.y + areas.z)),
184  0.0));
185 
186  Quaterniond rot = safeQuatlookAt(normal);
187  if (MolecularSystemAnimationDetails.positionSeed != 0)
188  pos += MolecularSystemAnimationDetails.positionStrength *
189  Vector3d(rnd2(MolecularSystemAnimationDetails.positionSeed),
190  rnd2(MolecularSystemAnimationDetails.positionSeed + 1),
191  rnd2(MolecularSystemAnimationDetails.positionSeed + 2));
192 
193  if (MolecularSystemAnimationDetails.rotationSeed != 0)
194  rot = weightedRandomRotation(
195  rot, MolecularSystemAnimationDetails.rotationSeed, occurrence,
196  MolecularSystemAnimationDetails.rotationStrength);
197 
198  pos += offset * normal;
199 
200  Transformation transformation;
201  transformation.setTranslation(pos);
202  transformation.setRotation(rot);
203  return transformation;
204 }

206 bool MeshShape::isInside(const Vector3d& point) const
207 {
208  const Vector3d& center = _bounds.getCenter();
209  const Vector3d rayDirection = center - point;
210  const double rayLength = length(rayDirection);
211  const Vector3d direction = normalize(rayDirection);
212  for (const auto& face : _faces)
213  {
214  Boxd box;
215  box.merge(_vertices[face.x]);
216  box.merge(_vertices[face.y]);
217  box.merge(_vertices[face.z]);
218  double t;
219  if (rayBoxIntersection(point, direction, box, rayLength / 10.0,
220  rayLength, t))
221  return false;
222  }
223  return true;
224 }

250 double MeshShape::_getSurfaceArea(const Vector3d& v0, const Vector3d& v1,
251  const Vector3d& v2) const
252 {
253  // Compute triangle area
254  const double a = length(v1 - v0);
255  const double b = length(v2 - v0);
256  const double c = length(v2 - v1);
257  const double s = (a + b + c) / 2.0;
258  const double e = s * (s - a) * (s - b) * (s - c);
259  return sqrt(e);
260 }
