41 MeshShape::MeshShape(const Vector4ds& clippingPlanes, const Vector3d& scale, const std::string& contents)
42  : Shape(clippingPlanes)
43 {
44  // Load mesh
45  Assimp::Importer importer;
46  const aiScene* aiScene = importer.ReadFileFromMemory(contents.c_str(), contents.length(),
47  aiProcess_GenSmoothNormals | aiProcess_Triangulate);
48 
49  if (!aiScene)
50  PLUGIN_THROW(importer.GetErrorString());
51 
52  if (!aiScene->HasMeshes())
53  PLUGIN_THROW("No mesh found");
54 
55  // Add protein instances according to membrane topology
56  for (size_t m = 0; m < aiScene->mNumMeshes; ++m)
57  {
58  aiMesh* mesh = aiScene->mMeshes[m];
59 
60  // Determine mesh center
61  Vector3d meshCenter{0.0, 0.0, 0.0};
62  for (uint64_t i = 0; i < mesh->mNumVertices; ++i)
63  meshCenter += _toVector3d(mesh->mVertices[i]);
64  meshCenter /= mesh->mNumVertices;
65 
66  // Recenter mesh and store transformed vertices
67  for (uint64_t i = 0; i < mesh->mNumVertices; ++i)
68  {
69  const auto v = _toVector3d(mesh->mVertices[i], meshCenter, scale);
70  _vertices.push_back(v);
71  _bounds.merge(v);
72  }
73 
74  // Store faces
75  for (size_t f = 0; f < mesh->mNumFaces; ++f)
76  if (mesh->mFaces[f].mNumIndices == 3)
77  {
78  const Vector3ui face(mesh->mFaces[f].mIndices[0], mesh->mFaces[f].mIndices[1],
79  mesh->mFaces[f].mIndices[2]);
80 
81  _faces.push_back(face);
82  const auto faceSurface = _getSurfaceArea(_vertices[face.x], _vertices[face.y], _vertices[face.z]);
83  _faceSurfaces.push_back(faceSurface);
84  _surface += faceSurface;
85  }
86 
87  // Store normals
88  _normals.resize(_vertices.size());
89  if (mesh->HasNormals())
90  {
91  for (const auto& face : _faces)
92  {
93  auto normal = glm::normalize(_toVector3d(mesh->mNormals[face.x]));
94  _normals[face.x] = normal;
95  normal = glm::normalize(_toVector3d(mesh->mNormals[face.y]));
96  _normals[face.y] = normal;
97  normal = glm::normalize(_toVector3d(mesh->mNormals[face.z]));
98  _normals[face.z] = normal;
99  }
100  }
101  else
102  for (const auto& face : _faces)
103  {
104  const auto v0 = glm::normalize(_vertices[face.y] - _vertices[face.x]);
105  const auto v1 = glm::normalize(_vertices[face.z] - _vertices[face.x]);
106  const auto normal = glm::cross(v0, v1);
107  _normals[face.x] = normal;
108  _normals[face.y] = normal;
109  _normals[face.z] = normal;
110  }
111 
112  PLUGIN_INFO(3, "----=== MeshBasedMembrane ===----");
113  PLUGIN_INFO(3, "Scale : " << scale);
114  PLUGIN_INFO(3, "Number of faces : " << _faces.size());
115  PLUGIN_INFO(3, "Mesh surface area : " << _surface);
116  PLUGIN_INFO(3, "Bounds : " << _bounds);
117  }
118 }

120 Transformation MeshShape::getTransformation(const uint64_t occurrence, const uint64_t nbOccurrences,
121  const MolecularSystemAnimationDetails& MolecularSystemAnimationDetails,
122  const double offset) const
123 {
124  if (occurrence == 0)
125  {
126  _faceIndex = 0;
127  _instanceCoveringProcess = 0.0;
128  _surfaceCoveringProcess = _faceSurfaces[_faceIndex];
129  }
130 
131  const double instanceSurface = _surface / nbOccurrences;
132  const double expectedSurfaceCovering = instanceSurface * occurrence;
133 
134  if (_instanceCoveringProcess > _surfaceCoveringProcess)
135  {
136  while (_surfaceCoveringProcess < expectedSurfaceCovering)
137  {
138  ++_faceIndex;
139  _surfaceCoveringProcess += _faceSurfaces[_faceIndex];
140  }
141  }
142  _instanceCoveringProcess += instanceSurface;
143 
144  const auto face = _faces[_faceIndex];
145 
146  Vector2d coordinates{1.0, 1.0};
147  while (coordinates.x + coordinates.y > 1.0)
148  {
149  coordinates.x = 0.5f + rnd1();
150  coordinates.y = 0.5f + rnd1();
151  }
152 
153  const auto v00 = _vertices[face.y] - _vertices[face.x];
154  const auto v01 = _vertices[face.z] - _vertices[face.x];
155  Vector3d pos = _vertices[face.x] + v00 * coordinates.x + v01 * coordinates.y;
156 
157  // Clipping planes
158  if (isClipped(pos, _clippingPlanes))
159  throw std::runtime_error("Instance is clipped");
160 
161  // Normals
162  const auto v10 = _vertices[face.x] - pos;
163  const auto v11 = _vertices[face.y] - pos;
164  const auto v12 = _vertices[face.z] - pos;
165  const Vector3d areas{0.5 * length(glm::cross(v11, v12)), 0.5 * length(glm::cross(v10, v12)),
166  0.5 * length(glm::cross(v10, v11))};
167 
168  const auto n0 = _normals[face.x];
169  const auto n1 = _normals[face.y];
170  const auto n2 = _normals[face.z];
171 
172  const Vector3d normal = glm::normalize(
173  Vector4d(glm::normalize((n0 * areas.x + n1 * areas.y + n2 * areas.z) / (areas.x + areas.y + areas.z)), 0.0));
174 
175  Quaterniond rot = safeQuatlookAt(normal);
176  if (MolecularSystemAnimationDetails.positionSeed != 0)
177  pos += MolecularSystemAnimationDetails.positionStrength *
178  Vector3d(rnd2(MolecularSystemAnimationDetails.positionSeed),
179  rnd2(MolecularSystemAnimationDetails.positionSeed + 1),
180  rnd2(MolecularSystemAnimationDetails.positionSeed + 2));
181 
182  if (MolecularSystemAnimationDetails.rotationSeed != 0)
183  rot = weightedRandomRotation(rot, MolecularSystemAnimationDetails.rotationSeed, occurrence,
184  MolecularSystemAnimationDetails.rotationStrength);
185 
186  pos += offset * normal;
187 
188  Transformation transformation;
189  transformation.setTranslation(pos);
190  transformation.setRotation(rot);
191  return transformation;
192 }

194 bool MeshShape::isInside(const Vector3d& point) const
195 {
196  const Vector3d& center = _bounds.getCenter();
197  const Vector3d rayDirection = center - point;
198  const double rayLength = length(rayDirection);
199  const Vector3d direction = normalize(rayDirection);
200  for (const auto& face : _faces)
201  {
202  Boxd box;
203  box.merge(_vertices[face.x]);
204  box.merge(_vertices[face.y]);
205  box.merge(_vertices[face.z]);
206  double t;
207  if (rayBoxIntersection(point, direction, box, rayLength / 10.0, rayLength, t))
208  return false;
209  }
210  return true;
211 }

235 double MeshShape::_getSurfaceArea(const Vector3d& v0, const Vector3d& v1, const Vector3d& v2) const
236 {
237  // Compute triangle area
238  const double a = length(v1 - v0);
239  const double b = length(v2 - v0);
240  const double c = length(v2 - v1);
241  const double s = (a + b + c) / 2.0;
242  const double e = s * (s - a) * (s - b) * (s - c);
243  return sqrt(e);
244 }
