33 Mesh::Mesh(Scene& scene, const MeshDescriptor& md)
34  : Node()
35 {
36  const auto loader = MeshLoader(scene);
37 
38  // Load protein
39  const Vector3f position = {md.position[0], md.position[1], md.position[2]};
40  const Quaterniond orientation = {md.orientation[0], md.orientation[1],
41  md.orientation[2], md.orientation[3]};
42  const Vector3f scale = {md.scale[0], md.scale[1], md.scale[2]};
43 
44  ProteinDescriptor pd;
45  pd.assemblyName = md.assemblyName;
46  pd.name = md.name;
47  pd.contents = md.proteinContents;
48  pd.representation = ProteinRepresentation::atoms;
49  pd.atomRadiusMultiplier = 1.f;
50  pd.recenter = true;
51  pd.atomRadiusMultiplier = md.atomRadiusMultiplier;
52  pd.representation = md.representation;
53  pd.position = md.position;
54  pd.orientation = md.orientation;
55 
56  // Random seed
57  srand(md.randomSeed);
58 
59  // Create model
60  _protein = ProteinPtr(new Protein(scene, pd));
61  _modelDescriptor = _protein->getModelDescriptor();
62  auto model = &_modelDescriptor->getModel();
63  model->updateBounds();
64  const auto proteinSize = model->getBounds().getSize();
65 
66  // Load mesh
67  Assimp::Importer importer;
68  const aiScene* aiScene =
69  importer.ReadFileFromMemory(md.meshContents.c_str(),
70  md.meshContents.length(),
71  aiProcess_GenSmoothNormals |
72  aiProcess_Triangulate);
73 
74  if (!aiScene)
75  PLUGIN_THROW(std::runtime_error(importer.GetErrorString()));
76 
77  if (!aiScene->HasMeshes())
78  PLUGIN_THROW(std::runtime_error("No meshes found"));
79 
80  const auto trfm = aiScene->mRootNode->mTransformation;
81  const Matrix4f matrix{trfm.a1, trfm.b1, trfm.c1, trfm.d1, trfm.a2, trfm.b2,
82  trfm.c2, trfm.d2, trfm.a3, trfm.b3, trfm.c3, trfm.d3,
83  trfm.a4, trfm.b4, trfm.c4, trfm.d4};
84 
85  // Add protein instances according to mesh topology
86  size_t instanceCount = 0;
87  float meshCoveringProgress = 0.f;
88  float instanceCoveringProgress = 0.f;
89  for (size_t m = 0; m < aiScene->mNumMeshes; ++m)
90  {
91  const auto& mesh = aiScene->mMeshes[m];
92 
93  // Mesh scaling
94  Vector3f meshCenter{0.f, 0.f, 0.f};
95  for (size_t i = 0; i < mesh->mNumVertices; ++i)
96  {
97  const auto& v = mesh->mVertices[i];
98  meshCenter += _toVector3f(v);
99  }
100  meshCenter /= mesh->mNumVertices;
101 
102  // Compute full mesh area
103  std::vector<Vector3ui> faces;
104  for (size_t f = 0; f < mesh->mNumFaces; ++f)
105  if (mesh->mFaces[f].mNumIndices == 3)
106  faces.push_back(Vector3ui(mesh->mFaces[f].mIndices[0],
107  mesh->mFaces[f].mIndices[1],
108  mesh->mFaces[f].mIndices[2]));
109 
110  float meshSurface = 0.f;
111  for (const auto& face : faces)
112  meshSurface += _getSurfaceArea(
113  _toVector3f(mesh->mVertices[face.x], meshCenter, scale),
114  _toVector3f(mesh->mVertices[face.y], meshCenter, scale),
115  _toVector3f(mesh->mVertices[face.z], meshCenter, scale));
116 
117  const float proteinSurface = proteinSize.x * proteinSize.x;
118 
119  // Total number of instance needed to fill the mesh surface
120  const size_t nbInstances = md.density * meshSurface / proteinSurface;
121  const float instanceSurface = meshSurface / nbInstances;
122 
123  PLUGIN_INFO << "----=== Mesh ===----" << std::endl;
124  PLUGIN_INFO << "Number of faces : " << faces.size() << std::endl;
125  PLUGIN_INFO << "Mesh surface area : " << meshSurface << std::endl;
126  PLUGIN_INFO << "Protein surface area : " << proteinSurface << std::endl;
127  PLUGIN_INFO << "Instance surface area: " << instanceSurface
128  << std::endl;
129  PLUGIN_INFO << "Number of instances : " << nbInstances << std::endl;
130 
131  for (const auto& face : faces)
132  {
133  const auto P0 =
134  _toVector3f(mesh->mVertices[face.x], meshCenter, scale);
135  const auto P1 =
136  _toVector3f(mesh->mVertices[face.y], meshCenter, scale);
137  const auto P2 =
138  _toVector3f(mesh->mVertices[face.z], meshCenter, scale);
139 
140  const auto V0 = P1 - P0;
141  const auto V1 = P2 - P0;
142 
143  const Vector3f defaultNormal = glm::cross(V0, V1);
144 
145  // Compute face surface
146  const float faceSurface = _getSurfaceArea(P0, P1, P2);
147 
148  // Estimate number of proteins for current face
149  meshCoveringProgress += faceSurface;
150  const size_t nbProteins =
151  size_t((meshCoveringProgress - instanceCoveringProgress) /
152  instanceSurface);
153 
154  // compute protein positions and orientations
155  for (size_t i = 0; i < nbProteins; ++i)
156  {
157  instanceCoveringProgress += instanceSurface;
158  Vector2f coordinates{1.f, 1.f};
159  while (coordinates.x + coordinates.y > 1.f)
160  {
161  coordinates.x = float(rand() % nbProteins) / nbProteins;
162  coordinates.y = float(rand() % nbProteins) / nbProteins;
163  }
164 
165  Transformation tf;
166  const Vector3f P = P0 + V0 * coordinates.x + V1 * coordinates.y;
167  const Vector3f transformedVertex =
168  matrix * Vector4f(P.x, P.y, P.z, 1.f);
169 
170  const Vector3f position{md.position[0], md.position[1],
171  md.position[2]};
172 
173  const float variableOffset =
174  md.surfaceVariableOffset * (rand() % 1000 / 1000.f - 0.5f);
175 
176  tf.setTranslation(position + transformedVertex +
177  defaultNormal * md.surfaceFixedOffset +
178  defaultNormal * variableOffset);
179 
180  if (mesh->HasNormals())
181  {
182  const auto v0 = P0 - P;
183  const auto v1 = P1 - P;
184  const auto v2 = P2 - P;
185 
186  const Vector3f areas{0.5f * length(glm::cross(v1, v2)),
187  0.5f * length(glm::cross(v0, v2)),
188  0.5f * length(glm::cross(v0, v1))};
189 
190  const auto N0 = _toVector3f(mesh->mNormals[face.x]);
191  const auto N1 = _toVector3f(mesh->mNormals[face.y]);
192  const auto N2 = _toVector3f(mesh->mNormals[face.z]);
193 
194  const Vector3f normal = glm::normalize(
195  matrix *
196  Vector4f(glm::normalize((N0 * areas.x + N1 * areas.y +
197  N2 * areas.z) /
198  (areas.x + areas.y + areas.z)),
199  1.f));
200 
201  if (normal != UP_VECTOR)
202  {
203  const Quaterniond orientation{md.orientation[0],
204  md.orientation[1],
205  md.orientation[2],
206  md.orientation[3]};
207  const Quaterniond rotation =
208  glm::quatLookAt(normal, UP_VECTOR);
209  tf.setRotation(rotation * orientation);
210  }
211  tf.setTranslation(position + transformedVertex +
212  normal * md.surfaceFixedOffset +
213  normal * variableOffset);
214  }
215 
216  if (instanceCount == 0)
217  _modelDescriptor->setTransformation(tf);
218 
219  const ModelInstance instance(true, false, tf);
220  _modelDescriptor->addInstance(instance);
221 
222  ++instanceCount;
223  }
224  }
225  }
226 }

228 float Mesh::_getSurfaceArea(const Vector3f& v0, const Vector3f& v1,
229  const Vector3f& v2) const
230 {
231  // Compute triangle area
232  const float a = length(v1 - v0);
233  const float b = length(v2 - v1);
234  const float c = length(v0 - v2);
235  const float p = (a + b + c) / 2.f;
236  const float e = p * (p - a) * (p - b) * (p - c);
237  if (e < 0)
238  return 0.f;
239 
240  return sqrt(e);
241 }

248 Vector3f Mesh::_toVector3f(const aiVector3D& v, const Vector3f& center,
249  const Vector3f& scale) const
250 {
251  const Vector3f p{v.x, v.y, v.z};
252  const Vector3f a = p - center;
253  const Vector3f b = p + a * scale;
254  return b;
255 }
