42 void OSPRayCamera::commit()
43 {
44  if (isModified())
45  {
46  const bool cameraChanged = _currentOSPCamera != getCurrentType();
47  if (cameraChanged)
48  _createOSPCamera();
49 
50  const auto position = getPosition();
51  const auto dir = glm::rotate(getOrientation(), Vector3d(0., 0., -1.));
52  const auto up = glm::rotate(getOrientation(), Vector3d(0., 1., 0.));
53 
54  osphelper::set(_camera, CAMERA_PROPERTY_POSITION, Vector3f(position));
55  osphelper::set(_camera, CAMERA_PROPERTY_DIRECTION, Vector3f(dir));
56  osphelper::set(_camera, CAMERA_PROPERTY_UP_VECTOR, Vector3f(up));
57  osphelper::set(_camera, CAMERA_PROPERTY_BUFFER_TARGET, getBufferTarget());
58 
59  toOSPRayProperties(*this, _camera);
60 
61  // Clip planes
62  if (!_clipPlanes.empty())
63  {
64  const auto clipPlanes = convertVectorToFloat(_clipPlanes);
65  auto clipPlaneData = ospNewData(clipPlanes.size(), OSP_FLOAT4, clipPlanes.data());
66  ospSetData(_camera, CAMERA_PROPERTY_CLIPPING_PLANES, clipPlaneData);
67  ospRelease(clipPlaneData);
68  }
69  else
70  {
71  // ospRemoveParam leaks objects, so we set it to null first
72  ospSetData(_camera, CAMERA_PROPERTY_CLIPPING_PLANES, nullptr);
73  ospRemoveParam(_camera, CAMERA_PROPERTY_CLIPPING_PLANES);
74  }
75  }
76  ospCommit(_camera);
77 }

93 void OSPRayCamera::_createOSPCamera()
94 {
95  auto newCamera = ospNewCamera(getCurrentType().c_str());
96  if (!newCamera)
97  throw std::runtime_error(getCurrentType() + " is not a registered camera");
98  if (_camera)
99  ospRelease(_camera);
100  _camera = newCamera;
101  _currentOSPCamera = getCurrentType();
102  markModified(false);
103 }
