42 void OSPRayCamera::commit()
43 {
44  // Anaglyph property is handled by the camera, but processed by the renderer
45  Vector3f anaglyphIpdOffset;
46  bool anaglyphEnabled = false;
47  if (getCurrentType() == CAMERA_PROPERTY_TYPE_ANAGLYPH)
48  {
49  const Vector3d dir = getOrientation() * UP_VECTOR;
50  const double d = dot(dir, UP_VECTOR);
51  const Vector3d dir_du = (d > 0.999 ? Vector3d(1, 0, 0) : normalize(cross(dir, UP_VECTOR))); // Avoid gimble lock
52  const double interpupillaryDistance = getProperty<double>(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str());
53  anaglyphIpdOffset = 0.5f * interpupillaryDistance * dir_du;
54  anaglyphEnabled = true;
55  }
56 
57  Renderer& renderer = _engine->getRenderer();
58  OSPRayRenderer* ospRenderer = dynamic_cast<OSPRayRenderer*>(&renderer);
59  if (ospRenderer)
60  if (ospRenderer->impl())
61  {
62  osphelper::set(ospRenderer->impl(), OSPRAY_RENDERER_PROPERTY_ANAGLYPH_IPD_OFFSET, anaglyphIpdOffset);
63  osphelper::set(ospRenderer->impl(), OSPRAY_RENDERER_PROPERTY_ANAGLYPH_ENABLED,
64  static_cast<int>(anaglyphEnabled));
65  }
66 
67  if (isModified())
68  {
69  const bool cameraChanged = _currentOSPCamera != getCurrentType();
70  if (cameraChanged)
71  _createOSPCamera();
72 
73  const auto position = getPosition();
74  const auto dir = glm::rotate(getOrientation(), Vector3d(0., 0., -1.));
75  const auto up = glm::rotate(getOrientation(), Vector3d(0., 1., 0.));
76 
77  osphelper::set(_camera, CAMERA_PROPERTY_POSITION, Vector3f(position));
78  osphelper::set(_camera, CAMERA_PROPERTY_DIRECTION, Vector3f(dir));
79  osphelper::set(_camera, CAMERA_PROPERTY_UP_VECTOR, Vector3f(up));
80  osphelper::set(_camera, CAMERA_PROPERTY_BUFFER_TARGET, getBufferTarget());
81 
82  toOSPRayProperties(*this, _camera);
83 
84  // Clip planes
85  if (!_clipPlanes.empty())
86  {
87  const auto clipPlanes = convertVectorToFloat(_clipPlanes);
88  auto clipPlaneData = ospNewData(clipPlanes.size(), OSP_FLOAT4, clipPlanes.data());
89  ospSetData(_camera, CAMERA_PROPERTY_CLIPPING_PLANES, clipPlaneData);
90  ospRelease(clipPlaneData);
91  }
92  else
93  {
94  // ospRemoveParam leaks objects, so we set it to null first
95  ospSetData(_camera, CAMERA_PROPERTY_CLIPPING_PLANES, nullptr);
96  ospRemoveParam(_camera, CAMERA_PROPERTY_CLIPPING_PLANES);
97  }
98  }
99  ospCommit(_camera);
100 }

116 void OSPRayCamera::_createOSPCamera()
117 {
118  auto newCamera = ospNewCamera(getCurrentType().c_str());
119  if (!newCamera)
120  throw std::runtime_error(getCurrentType() + " is not a registered camera");
121  if (_camera)
122  ospRelease(_camera);
123  _camera = newCamera;
124  _currentOSPCamera = getCurrentType();
125  markModified(false);
126 }
