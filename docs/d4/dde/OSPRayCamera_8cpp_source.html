37 void OSPRayCamera::commit()
38 {
39  if (!isModified())
40  return;
41 
42  const bool cameraChanged = _currentOSPCamera != getCurrentType();
43  if (cameraChanged)
44  _createOSPCamera();
45 
46  const auto position = getPosition();
47  const auto dir = glm::rotate(getOrientation(), Vector3d(0., 0., -1.));
48  const auto up = glm::rotate(getOrientation(), Vector3d(0., 1., 0.));
49 
50  osphelper::set(_camera, "pos", Vector3f(position));
51  osphelper::set(_camera, "dir", Vector3f(dir));
52  osphelper::set(_camera, "up", Vector3f(up));
53  osphelper::set(_camera, "buffer_target", getBufferTarget());
54 
55  toOSPRayProperties(*this, _camera);
56 
57  // Clip planes
58  if (!_clipPlanes.empty())
59  {
60  const auto clipPlanes = convertVectorToFloat(_clipPlanes);
61  auto clipPlaneData = ospNewData(clipPlanes.size(), OSP_FLOAT4, clipPlanes.data());
62  ospSetData(_camera, "clipPlanes", clipPlaneData);
63  ospRelease(clipPlaneData);
64  }
65  else
66  {
67  // ospRemoveParam leaks objects, so we set it to null first
68  ospSetData(_camera, "clipPlanes", nullptr);
69  ospRemoveParam(_camera, "clipPlanes");
70  }
71 
72  ospCommit(_camera);
73 }

89 void OSPRayCamera::_createOSPCamera()
90 {
91  auto newCamera = ospNewCamera(getCurrentType().c_str());
92  if (!newCamera)
93  throw std::runtime_error(getCurrentType() + " is not a registered camera");
94  if (_camera)
95  ospRelease(_camera);
96  _camera = newCamera;
97  _currentOSPCamera = getCurrentType();
