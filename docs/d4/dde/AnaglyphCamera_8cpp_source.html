43 void AnaglyphCamera::commit()
44 {
45  Camera::commit();
46 
47  fieldOfView = getParamf(CAMERA_PROPERTY_FIELD_OF_VIEW.name.c_str(), DEFAULT_CAMERA_FIELD_OF_VIEW);
48  aspect = getParamf(CAMERA_PROPERTY_ASPECT_RATIO.name.c_str(), DEFAULT_CAMERA_ASPECT_RATIO);
49  apertureRadius = getParamf(CAMERA_PROPERTY_APERTURE_RADIUS.name.c_str(), DEFAULT_CAMERA_APERTURE_RADIUS);
50  focalDistance = getParamf(CAMERA_PROPERTY_FOCAL_DISTANCE.name.c_str(), DEFAULT_CAMERA_FOCAL_DISTANCE);
51  nearClip = getParamf(CAMERA_PROPERTY_NEAR_CLIP.name.c_str(), DEFAULT_CAMERA_NEAR_CLIP);
52  interpupillaryDistance =
53  getParamf(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str(), DEFAULT_CAMERA_INTERPUPILLARY_DISTANCE);
54  enableClippingPlanes =
55  getParam(CAMERA_PROPERTY_ENABLE_CLIPPING_PLANES.name.c_str(), DEFAULT_CAMERA_ENABLE_CLIPPING_PLANES);
56  clipPlanes = enableClippingPlanes ? getParamData(CAMERA_PROPERTY_CLIPPING_PLANES, nullptr) : nullptr;
57  useHardwareRandomizer = getParam(COMMON_PROPERTY_USE_HARDWARE_RANDOMIZER.name.c_str(),
58  static_cast<int>(DEFAULT_COMMON_USE_HARDWARE_RANDOMIZER));
59 
60  dir = normalize(dir);
61  ::ospray::vec3f dir_du = normalize(cross(dir, up));
62  ::ospray::vec3f dir_dv = cross(dir_du, dir);
63 
64  ::ospray::vec3f org = pos;
65  double imgPlane_size_y = 2.f * tanf(::ospray::deg2rad(0.5f * fieldOfView));
66  double imgPlane_size_x = imgPlane_size_y * aspect;
67 
68  dir_du *= imgPlane_size_x;
69  dir_dv *= imgPlane_size_y;
70 
71  ::ospray::vec3f dir_00 = dir - 0.5f * dir_du - 0.5f * dir_dv;
72 
73  double scaledAperture = 0.f;
74  if (apertureRadius > 0.f)
75  {
76  dir_du *= focalDistance;
77  dir_dv *= focalDistance;
78  dir_00 *= focalDistance;
79  scaledAperture = apertureRadius / imgPlane_size_x;
80  }
81 
82  const auto clipPlaneData = clipPlanes ? clipPlanes->data : nullptr;
83  const size_t numClipPlanes = clipPlanes ? clipPlanes->numItems : 0;
84 
85  ::ispc::AnaglyphCamera_set(getIE(), (const ::ispc::vec3f&)org, (const ::ispc::vec3f&)dir_00,
86  (const ::ispc::vec3f&)dir_du, (const ::ispc::vec3f&)dir_dv, scaledAperture, aspect,
87  (const ::ispc::vec4f*)clipPlaneData, numClipPlanes, nearClip, useHardwareRandomizer);
88 }
