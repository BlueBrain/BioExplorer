51 {
52  Camera::commit();
53 
54  fieldOfView = getParamf(CAMERA_PROPERTY_FIELD_OF_VIEW.name.c_str(), DEFAULT_CAMERA_FIELD_OF_VIEW);

56  apertureRadius = getParamf(CAMERA_PROPERTY_APERTURE_RADIUS.name.c_str(), DEFAULT_CAMERA_APERTURE_RADIUS);

58  nearClip = getParamf(CAMERA_PROPERTY_NEAR_CLIP.name.c_str(), DEFAULT_CAMERA_NEAR_CLIP);
59  interpupillaryDistance =
60  getParamf(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str(), DEFAULT_CAMERA_INTERPUPILLARY_DISTANCE);
61  enableClippingPlanes =
62  getParam(CAMERA_PROPERTY_ENABLE_CLIPPING_PLANES.name.c_str(), DEFAULT_CAMERA_ENABLE_CLIPPING_PLANES);
63  clipPlanes = enableClippingPlanes ? getParamData(CAMERA_PROPERTY_CLIPPING_PLANES, nullptr) : nullptr;
64  useHardwareRandomizer = getParam(COMMON_PROPERTY_USE_HARDWARE_RANDOMIZER.name.c_str(),
65  static_cast<int>(DEFAULT_COMMON_USE_HARDWARE_RANDOMIZER));
66 
67  dir = normalize(dir);
68  ::ospray::vec3f dir_du = normalize(cross(dir, up));
69  ::ospray::vec3f dir_dv = cross(dir_du, dir);
70 
71  ::ospray::vec3f org = pos;
72  double imgPlane_size_y = 2.f * tanf(::ospray::deg2rad(0.5f * fieldOfView));
73  double imgPlane_size_x = imgPlane_size_y * aspect;
74 
75  dir_du *= imgPlane_size_x;
76  dir_dv *= imgPlane_size_y;
77 
78  ::ospray::vec3f dir_00 = dir - 0.5f * dir_du - 0.5f * dir_dv;
79 
80  double scaledAperture = 0.f;
81  if (apertureRadius > 0.f)
82  {
83  dir_du *= focalDistance;
84  dir_dv *= focalDistance;
85  dir_00 *= focalDistance;
86  scaledAperture = apertureRadius / imgPlane_size_x;
87  }
88 
89  const auto clipPlaneData = clipPlanes ? clipPlanes->data : nullptr;
90  const size_t numClipPlanes = clipPlanes ? clipPlanes->numItems : 0;
91 
92  ::ispc::AnaglyphCamera_set(getIE(), (const ::ispc::vec3f&)org, (const ::ispc::vec3f&)dir_00,
93  (const ::ispc::vec3f&)dir_du, (const ::ispc::vec3f&)dir_dv, scaledAperture, aspect,
94  (const ::ispc::vec4f*)clipPlaneData, numClipPlanes, nearClip, useHardwareRandomizer);
95 }
