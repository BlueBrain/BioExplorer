32 uint32 getLocation(const uniform MetabolismRenderer* uniform self, varying Ray& ray, float& epsilon)
33 {
34  // Back ray
35  int backLocation = extra_cellular_space;
36  Ray backRay = ray;
37  backRay.dir = -1.f * backRay.dir;
38  traceRay(self->super.model, backRay);
39  if (backRay.geomID >= 0)
40  {
41  DifferentialGeometry dg;
42  postIntersect(self->super.model, dg, backRay, DG_NS | DG_NORMALIZE);
43  uniform AdvancedMaterial* objMaterial = (uniform AdvancedMaterial*)dg.material;
44  foreach_unique(mat in objMaterial) backLocation = mat->userParameter;
45  }
46  return backLocation;
47 }

49 inline vec3f MetabolismRenderer_shadeRay(const uniform MetabolismRenderer* uniform self, varying ScreenSample& sample)
50 {
51  vec3f finalColor = make_vec3f(0.f);
52  if (sample.sampleID.z == 0)
53  {
54  sample.alpha = 0.f;
55  sample.z = inf;
56  }
57 
58  if (self->userDataSize == 0)
59  return finalColor;
60 
61  float t = self->nearPlane + ((sample.sampleID.z % self->refinementSteps) * (self->rayStep / self->refinementSteps));
62 
63  // Following intersections
64  float finalAlpha = 0.f;
65  float voxelAlpha = 0.f;
66  vec3f voxelColor = make_vec3f(0.f);
67  while (t < self->farPlane && finalAlpha < 1.f)
68  {
69  Ray r = sample.ray;
70  r.t0 = 0.f;
71  r.t = self->farPlane - t;
72  r.org = sample.ray.org + t * sample.ray.dir;
73  r.primID = -1;
74  r.geomID = -1;
75  r.instID = -1;
76 
77  float geometryEpsilon;
78  const uint32 location = getLocation(self, r, geometryEpsilon);
79 
80  t += self->rayStep;
81 
82  const float concentration = self->userData[1 + location];
83  if (concentration >= 1e38)
84  continue;
85  voxelAlpha = self->transferFunction->getOpacityForValue(self->transferFunction, concentration);
86  if (voxelAlpha > OPACITY_EPSILON)
87  voxelColor = self->transferFunction->getColorForValue(self->transferFunction, concentration);
88 
89  float noiseAlpha = 1.f;
90  if (self->noiseFrequency > 0.f && finalAlpha > OPACITY_EPSILON)
91  {
92  const vec3f p = sample.ray.org + t * sample.ray.dir;
93  noiseAlpha = 1.f + cloud(p * self->noiseFrequency,
94  cos((location + self->timestamp) * 0.05f) +
95  sin((location + self->timestamp) * 0.032f) * self->noiseAmplitude);
96  }
97 
98  const float alpha = clamp(noiseAlpha * self->alphaCorrection * voxelAlpha, 0.f, 1.f);
99  finalColor = finalColor + voxelColor * alpha * (1.f - finalAlpha);
100  finalAlpha += alpha;
101  }
102 
103  sample.alpha = finalAlpha;
104  return clamp(finalColor * self->exposure, make_vec3f(0.f), make_vec3f(1.f));
105 }
