29 bool getNextIntersection(const uniform MetabolismRenderer* uniform self,
30  varying Ray& ray, float& distance, uint32& location,
31  bool& enteringLocation, float& epsilon)
32 {
33  traceRay(self->super.model, ray);
34  if (ray.geomID >= 0)
35  {
36  DifferentialGeometry dg;
37  postIntersect(self->super.model, dg, ray, DG_NS | DG_NORMALIZE);
38 
39  uniform AdvancedMaterial* objMaterial =
40  (uniform AdvancedMaterial*)dg.material;
41 
42  foreach_unique(mat in objMaterial) location = mat->userParameter;
43 
44  distance = ray.t;
45  enteringLocation = dot(ray.dir, dg.Ns) < 0.f;
46  epsilon = dg.epsilon;
47  return true;
48  }
49  return false;
50 }

52 inline vec3f MetabolismRenderer_shadeRay(
53  const uniform MetabolismRenderer* uniform self,
54  varying ScreenSample& sample)
55 {
56  vec3f finalColor = make_vec3f(0.f);
57  if (sample.sampleID.z == 0)
58  {
59  sample.alpha = 0.f;
60  sample.z = inf;
61  }
62 
63  if (self->userDataSize == 0)
64  return finalColor;
65 
66  bool firstVisibleVoxel = true;
67 
68  // Find intersections for current ray
69  float intersectionDistances[NB_MAX_INTERSECTIONS];
70  uint32 intersectionLocations[NB_MAX_INTERSECTIONS];
71  float intersectionOpacities[NB_MAX_INTERSECTIONS];
72  uint32 stackLocations[NB_MAX_INTERSECTIONS];
73  int32 stackCounts[NB_MAX_INTERSECTIONS];
74 
75  for (uint32 i = 0; i < NB_MAX_INTERSECTIONS; ++i)
76  {
77  intersectionDistances[i] = 0.f;
78  intersectionLocations[i] = extra_cellular_space;
79  intersectionOpacities[i] = 0.f;
80  stackLocations[i] = 0;
81  stackCounts[i] = 0;
82  }
83 
84  float t = self->nearPlane + ((sample.sampleID.z % self->refinementSteps) *
85  (self->rayStep / self->refinementSteps));
86 
87  // Initial intersection
88  float distance;
89  uint32 location = extra_cellular_space;
90  bool enteringLocation;
91  float geometryEpsilon;
92 
93  uint32 intersectionIndex = 0;
94  intersectionDistances[intersectionIndex] = t;
95  intersectionLocations[intersectionIndex] = location;
96  intersectionOpacities[intersectionIndex] =
97  self->transferFunction->getOpacityForValue(self->transferFunction,
98  location);
99 
100  uint32 stackIndex = 0;
101  stackCounts[stackIndex] = 1;
102  stackLocations[stackIndex] = location;
103 
104  // Following intersections
105  Ray r = sample.ray;
106  r.t0 = 0.f;
107  r.t = self->farPlane;
108  while (true)
109  {
110  if (getNextIntersection(self, r, distance, location, enteringLocation,
111  geometryEpsilon))
112  {
113  const float locationOpacity =
114  self->transferFunction->getOpacityForValue(
115  self->transferFunction, location);
116  if (locationOpacity > OPACITY_EPSILON)
117  {
118  if (enteringLocation)
119  {
120  // Entering a location
121  if (stackLocations[stackIndex] == location)
122  {
123  // Already inside that location
124  ++stackCounts[stackIndex];
125  }
126  else
127  {
128  // Entering new location, adding intersection
129  ++stackIndex;
130  if (stackIndex > NB_MAX_INTERSECTIONS)
131  return make_vec3f(1, 0, 0);
132 
133  stackCounts[stackIndex] = 1;
134  stackLocations[stackIndex] = location;
135 
136  ++intersectionIndex;
137  if (intersectionIndex > NB_MAX_INTERSECTIONS)
138  return make_vec3f(0, 0, 1);
139 
140  intersectionDistances[intersectionIndex] = distance;
141  intersectionLocations[intersectionIndex] = location;
142  intersectionOpacities[intersectionIndex] =
143  locationOpacity;
144  }
145  }
146  else
147  {
148  // Exiting a location
149  if (stackLocations[stackIndex] == location)
150  {
151  // Already inside that location
152  --stackCounts[stackIndex];
153  if (stackCounts[stackIndex] == 0)
154  {
155  --stackIndex;
156  --stackCounts[stackIndex];
157 
158  ++intersectionIndex;
159  if (intersectionIndex > NB_MAX_INTERSECTIONS)
160  return make_vec3f(0, 1, 0);
161 
162  intersectionDistances[intersectionIndex] = distance;
163  intersectionLocations[intersectionIndex] = location;
164  intersectionOpacities[intersectionIndex] =
165  intersectionOpacities[intersectionIndex - 1];
166  }
167  }
168  // else Strange...
169  }
170  }
171 
172  // r.org = r.org + (distance + geometryEpsilon) * r.dir;
173  r.t0 = distance + geometryEpsilon;
174  r.t = self->farPlane - distance;
175  r.primID = -1;
176  r.geomID = -1;
177  r.instID = -1;
178  }
179  else
180  break;
181  }
182 
183  // How many intersections on that ray?
184  if (intersectionIndex == 0)
185  {
186  // No intersection dectected
187  if (intersectionOpacities[0] < OPACITY_EPSILON)
188  // Location is invisible, returning Background color
189  return finalColor;
190 
191  // Location is visible, ray-march all the way from near to far plane
192  intersectionDistances[0] = self->farPlane;
193  }
194 
195  // Ray marching along identified intersections
196  float finalAlpha = 0.f;
197  uint32 currentIntersection = 0;
198  uint32 lastVisibleIntersection = 0;
199  while (finalAlpha < 1.f && t < self->farPlane)
200  {
201  if (t > intersectionDistances[currentIntersection])
202  ++currentIntersection;
203 
204  const float intersectionOpacity =
205  intersectionOpacities[currentIntersection];
206 
207  if (intersectionOpacity > OPACITY_EPSILON)
208  lastVisibleIntersection = currentIntersection;
209 
210  const float opacity = intersectionOpacities[lastVisibleIntersection];
211 
212  float voxelAlpha = 0.f;
213  vec3f voxelColor = make_vec3f(0.f);
214 
215  if (opacity > OPACITY_EPSILON)
216  {
217  const uint32 location =
218  intersectionLocations[lastVisibleIntersection];
219  const float concentration = (location < self->userDataSize - 1)
220  ? self->userData[1 + location]
221  : 0.f;
222  voxelAlpha = concentration * opacity;
223  if (voxelAlpha > OPACITY_EPSILON)
224  voxelColor = self->transferFunction->getColorForValue(
225  self->transferFunction, location);
226 
227  float noiseAlpha = 1.f;
228  if (self->noiseFrequency > 0.f && opacity > OPACITY_EPSILON)
229  {
230  const vec3f p = sample.ray.org + t * sample.ray.dir;
231  noiseAlpha =
232  1.f + cloud(p * self->noiseFrequency,
233  cos((location + self->timestamp) * 0.05f) +
234  sin((location + self->timestamp) * 0.032f) *
235  self->noiseAmplitude);
236  }
237 
238  const float alpha =
239  clamp(noiseAlpha * self->alphaCorrection * voxelAlpha, 0.f,
240  1.f);
241  finalColor = finalColor + voxelColor * alpha * (1.f - finalAlpha);
242  finalAlpha += alpha;
243  }
244 
245  t += self->rayStep;
246  }
247 
248  sample.alpha = finalAlpha;
249  return clamp(finalColor * self->exposure, make_vec3f(0.f), make_vec3f(1.f));
250 }
