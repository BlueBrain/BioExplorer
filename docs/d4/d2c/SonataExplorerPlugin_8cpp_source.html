159 std::string _sanitizeString(const std::string& input)
160 {
161  static const std::vector<std::string> sanitetizeItems = {"\"", "\\", "'", ";", "&", "|", "`"};
162 
163  std::string result = "";
164 
165  for (size_t i = 0; i < input.size(); i++)
166  {
167  bool found = false;
168  for (const auto& token : sanitetizeItems)
169  {
170  if (std::string(1, input[i]) == token)
171  {
172  result += "\\" + token;
173  found = true;
174  break;
175  }
176  }
177  if (!found)
178  {
179  result += std::string(1, input[i]);
180  }
181  }
182  return result;
183 }

185 std::vector<std::string> _splitString(const std::string& source, const char token)
186 {
187  std::vector<std::string> result;
188  std::string split;
189  std::istringstream ss(source);
190  while (std::getline(ss, split, token))
191  result.push_back(split);
192 
193  return result;
194 }

360 Response SonataExplorerPlugin::_exportModelToMesh(const ExportModelToMesh& details)
361 {
362  Response response;
363  try
364  {
365  auto& scene = _api->getScene();
366  auto modelDescriptor = scene.getModel(details.modelId);
367  if (modelDescriptor)
368  {
369  const auto& model = modelDescriptor->getModel();
370  std::list<Weighted_point> l;
371  for (const auto& spheres : model.getSpheres())
372  {
373  uint64_t count = 0;
374  for (const auto& s : spheres.second)
375  {
376  if (count % details.density == 0)
377  l.push_front(Weighted_point(Point_3(s.center.x, s.center.y, s.center.z),
378  details.radiusMultiplier * s.radius));
379  ++count;
380  }
381  }
382 
383  PLUGIN_INFO("Constructing skin surface from " << l.size() << " spheres");
384 
385  Polyhedron polyhedron;
386  if (details.skin)
387  {
388  Skin_surface_3 skinSurface(l.begin(), l.end(), details.shrinkFactor);
389 
390  PLUGIN_INFO("Meshing skin surface...");
391  CGAL::mesh_skin_surface_3(skinSurface, polyhedron);
392  CGAL::Polygon_mesh_processing::triangulate_faces(polyhedron);
393  }
394  else
395  {
396  Union_of_balls_3 union_of_balls(l.begin(), l.end());
397  CGAL::mesh_union_of_balls_3(union_of_balls, polyhedron);
398  }
399 
400  PLUGIN_INFO("Export mesh to " << details.path);
401  std::ofstream out(details.path);
402  out << polyhedron;
403  }
404  else
405  PLUGIN_THROW("Model " + std::to_string(details.modelId) + " is not registered");
406  }
407  CATCH_STD_EXCEPTION()
408  return response;
409 }

411 Response SonataExplorerPlugin::_setConnectionsPerValue(const ConnectionsPerValue& cpv)
412 {
413  Response response;
414  try
415  {
416  meshing::PointCloud pointCloud;
417 
418  auto& scene = _api->getScene();
419  auto modelDescriptor = scene.getModel(cpv.modelId);
420  if (modelDescriptor)
421  {
422  auto simulationHandler = modelDescriptor->getModel().getSimulationHandler();
423  if (!simulationHandler)
424  PLUGIN_THROW("Scene has not user data handler");
425 
426  auto& model = modelDescriptor->getModel();
427  for (const auto& spheres : model.getSpheres())
428  {
429  for (const auto& s : spheres.second)
430  {
431  const float* data = static_cast<float*>(simulationHandler->getFrameData(cpv.frame));
432 
433  const float value = data[s.userData];
434  if (abs(value - cpv.value) < cpv.epsilon)
435  pointCloud[spheres.first].push_back({s.center.x, s.center.y, s.center.z, s.radius});
436  }
437  }
438 
439  if (!pointCloud.empty())
440  {
441  auto meshModel = scene.createModel();
442  meshing::PointCloudMesher mesher;
443  if (mesher.toConvexHull(*meshModel, pointCloud))
444  {
445  auto modelDesc =
446  std::make_shared<ModelDescriptor>(std::move(meshModel),
447  "Connection for value " + std::to_string(cpv.value));
448  scene.addModel(modelDesc);
449  _markModified();
450  }
451  }
452  else
453  PLUGIN_INFO("No connections added for value " << std::to_string(cpv.value));
454  }
455  else
456  PLUGIN_INFO("Model " << cpv.modelId << " is not registered");
457  }
458  CATCH_STD_EXCEPTION()
459  return response;
460 }

563 Response SonataExplorerPlugin::_addSphere(const AddSphere& details)
564 {
565  Response response;
566  try
567  {
568  if (details.center.size() < 3)
569  PLUGIN_THROW(
570  "Sphere center has the wrong number of parameters (3 "
571  "necessary)");
572  if (details.color.size() < 4)
573  PLUGIN_THROW(
574  "Sphere color has the wrong number of parameters (RGBA, 4 "
575  "necessary)");
576  if (details.radius < 0.f)
577  PLUGIN_THROW("Negative radius passed for sphere creation");
578 
579  auto& scene = _api->getScene();
580  ModelPtr modelptr = scene.createModel();
581 
582  const size_t matId = 1;
583  const Vector3d color(details.color[0], details.color[1], details.color[2]);
584  const double opacity = details.color[3];
585  _createShapeMaterial(modelptr, matId, color, opacity);
586 
587  const Vector3f center(details.center[0], details.center[1], details.center[2]);
588  modelptr->addSphere(matId, {center, details.radius});
589 
590  size_t numModels = scene.getNumModels();
