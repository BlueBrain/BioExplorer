175 // c-array to std.array: https://stackoverflow.com/questions/11205186
176 template <size_t M, typename T>
177 auto toArray(glm::vec<M, T>& vec)
178 {
179  return reinterpret_cast<std::array<T, M>*>(glm::value_ptr(vec));
180 }
181 
182 template <typename T>
183 auto toArray(glm::tquat<T>& quat)
184 {
185  return reinterpret_cast<std::array<T, 4>*>(glm::value_ptr(quat));
186 }
187 
188 template <size_t M, typename T>
189 auto toArray(std::vector<glm::vec<M, T>>& vecVec)
190 {
191  return reinterpret_cast<std::vector<std::array<T, M>>*>(&vecVec);
192 }

681 template <typename T>
682 inline std::pair<bool, core::PropertyMap> fromJSONWithPropertyMap(T& params, const std::string& json,
683  const std::string& propertyMapName)
684 {
685  using namespace rapidjson;
686  Document document;
687  document.Parse(json.c_str());
688 
689  core::PropertyMap propertyMap;
690 
691  if (document.HasMember(propertyMapName.c_str()))
692  {
693  auto& loaderProperties = document[propertyMapName.c_str()];
694  Document propertyDoc;
695  propertyDoc.CopyFrom(loaderProperties, propertyDoc.GetAllocator());
696  rapidjson::StringBuffer buffer;
697  rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
698  propertyDoc.Accept(writer);
699 
700  propertyMap = jsonToPropertyMap(buffer.GetString());
701 
702  // Clear property map to be able to parse rest of values
703  loaderProperties.SetObject();
704  }
705 
706  rapidjson::StringBuffer buffer;
707  rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
708  document.Accept(writer);
709  const auto str = buffer.GetString();
710  return {staticjson::from_json_string(str, &params, nullptr), propertyMap};
711 }
