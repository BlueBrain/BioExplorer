48  int AdvancedRenderer_getBytesPerPrimitive(const void* geometry)
49  {
50  const ospray::Geometry* base =
51  static_cast<const ospray::Geometry*>(geometry);
52  if (dynamic_cast<const ospray::Spheres*>(base))
53  return sizeof(brayns::Sphere);
54  else if (dynamic_cast<const ospray::Cylinders*>(base))
55  return sizeof(brayns::Cylinder);
56  else if (dynamic_cast<const ospray::Cones*>(base))
57  return sizeof(brayns::Cone);
58  else if (dynamic_cast<const ospray::SDFGeometries*>(base))
59  return sizeof(brayns::SDFGeometry);
60  return 0;
61  }
62 }

97  _giStrength = getParam1f("giWeight", 0.f);
98  _giDistance = getParam1f("giDistance", 1e20f);
99  _giSamples = getParam1i("giSamples", 1);
