48  int AdvancedRenderer_getBytesPerPrimitive(const void* geometry)
49  {
50  const ::ospray::Geometry* base = static_cast<const ::ospray::Geometry*>(geometry);
51  if (dynamic_cast<const ::ospray::Spheres*>(base))
52  return sizeof(core::Sphere);
53  else if (dynamic_cast<const ::ospray::Cylinders*>(base))
54  return sizeof(core::Cylinder);
55  else if (dynamic_cast<const ::core::engine::ospray::Cones*>(base))
56  return sizeof(::core::Cone);
57  else if (dynamic_cast<const ::core::engine::ospray::SDFGeometries*>(base))
58  return sizeof(::core::SDFGeometry);
59  return 0;
60  }
61 }
