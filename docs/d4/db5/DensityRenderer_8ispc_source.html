27/**
28 Composes source and destination colors according to specified alpha
29 correction
30 @param src Source RGBA color
31 @param dst Returned destination RGBA color
32 @param alphaCorrection Alpha correction used to blend colors
33*/
34inline void composite(const varying vec4f& src, varying vec4f& dst,
35 const varying float alphaCorrection)
36{
37 const float alpha =
38 1.f - pow(1.f - min(src.w, 1.f - 1.f / 256.f), alphaCorrection);
39 const float a = alpha * (1.f - dst.w);
40 dst.x = dst.x + src.x * a;
41 dst.y = dst.y + src.y * a;
42 dst.z = dst.z + src.z * a;
43 dst.w += (alpha * (1.f - dst.w));
44}

46inline vec3f DensityRenderer_shadeRay(
47 const uniform DensityRenderer* uniform self, varying ScreenSample& sample)
48{
49 const Ray ray = sample.ray;
50
51 if (sample.sampleID.z == 0)
52 {
53 sample.alpha = 0.f;
54 sample.z = inf;
55 }
56
57 // init TEA RNG //
58 uniform FrameBuffer* uniform fb = self->super.fb;
59 RandomTEA rng_state;
60 varying RandomTEA* const uniform rng = &rng_state;
61 RandomTEA__Constructor(rng, 0x290374,
62 (fb->size.x * sample.sampleID.y) +
63 sample.sampleID.x);
64 const vec2f rot = RandomTEA__getFloats(rng);
65
66 vec4f finalColor = make_vec4f(0.f);
67 bool firstVisibleVoxel = true;
68
69 float t = self->rayStep;
70 while (finalColor.w < 0.9f && t < self->farPlane)
71 {
72 const vec3f p = ray.org + t * ray.dir;
73
74 int hits = 0;
75 const vec3f N = ray.dir;
76 const linear3f localToWorld = frame(N);
77
78 for (uniform int i = 0; i < self->sampleCount; ++i)
79 {
80 const vec2f halton =
81 HaltonSequence_get2D(self->sampleCount * sample.sampleID.z + i);
82 const vec2f r = CranleyPattersonRotation(halton, rot);
83 const vec3f local_ao_dir = cosineSampleHemisphere(r);
84 const vec3f ao_dir = localToWorld * local_ao_dir;
85
86 Ray ao_ray;
87 setRay(ao_ray, p, ao_dir, 0.f, self->searchLength);
88 if (isOccluded(self->super.model, ao_ray))
89 ++hits;
90 }
91
92 if (hits > 0)
93 {
94 const float a = (float)hits / (float)self->sampleCount;
95
96 const vec3f sampleColor = make_vec3f(a, a, 1.f - a);
97 const float alpha = 1.f / (float)self->sampleCount;
98
99 composite(make_vec4f(sampleColor, alpha), finalColor, 1.0f);
100
101 if (firstVisibleVoxel)
102 sample.z = min(t, sample.z);
103
104 firstVisibleVoxel = false;
105 }
106
107 t += self->rayStep;
108 }
109
110 sample.alpha = finalColor.w;
111 return clamp(make_vec3f(finalColor) * self->exposure, make_vec3f(0.f),
112 make_vec3f(1.f));
113}

115void DensityRenderer_renderSample(uniform Renderer* uniform _self,
116 void* uniform perFrameData,
117 varying ScreenSample& sample)
118{
119 uniform DensityRenderer* uniform self =
120 (uniform DensityRenderer * uniform) _self;
121 sample.ray.time = self->timestamp;
122 sample.rgb = DensityRenderer_shadeRay(self, sample);
123}
