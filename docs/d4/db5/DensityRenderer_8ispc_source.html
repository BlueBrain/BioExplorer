30 /**
31  Composes source and destination colors according to specified alpha correction
32  @param src Source RGBA color
33  @param dst Returned destination RGBA color
34  @param alphaCorrection Alpha correction used to blend colors
35 */
36 inline void composite(const varying vec4f& src, varying vec4f& dst, const varying float alphaCorrection)
37 {
38  const float alpha = 1.f - pow(1.f - min(src.w, 1.f - 1.f / 256.f), alphaCorrection);
39  const float a = alpha * (1.f - dst.w);
40  dst.x = dst.x + src.x * a;
41  dst.y = dst.y + src.y * a;
42  dst.z = dst.z + src.z * a;
43  dst.w += (alpha * (1.f - dst.w));
44 }

46 inline vec3f DensityRenderer_shadeRay(const uniform DensityRenderer* uniform self, varying ScreenSample& sample)
47 {
48  const Ray ray = sample.ray;
49 
50  if (sample.sampleID.z == 0)
51  {
52  sample.alpha = 0.f;
53  sample.z = inf;
54  }
55 
56  // init TEA RNG //
57  uniform FrameBuffer* uniform fb = self->super.super.fb;
58  RandomTEA rng_state;
59  varying RandomTEA* const uniform rng = &rng_state;
60  RandomTEA__Constructor(rng, 0x290374, (fb->size.x * sample.sampleID.y) + sample.sampleID.x);
61  const vec2f rot = RandomTEA__getFloats(rng);
62 
63  vec4f finalColor = make_vec4f(0.f);
64  bool firstVisibleVoxel = true;
65 
66  float t = self->rayStep;
67  while (finalColor.w < 0.9f && t < self->farPlane)
68  {
69  const vec3f p = ray.org + t * ray.dir;
70 
71  int hits = 0;
72  const vec3f N = ray.dir;
73  const linear3f localToWorld = frame(N);
74 
75  for (uniform int i = 0; i < self->samplesPerFrame; ++i)
76  {
77  const vec2f halton = HaltonSequence_get2D(self->samplesPerFrame * sample.sampleID.z + i);
78  const vec2f r = CranleyPattersonRotation(halton, rot);
79  const vec3f local_ao_dir = cosineSampleHemisphere(r);
80  const vec3f ao_dir = localToWorld * local_ao_dir;
81 
82  Ray ao_ray;
83  setRay(ao_ray, p, ao_dir, 0.f, self->searchLength);
84  if (isOccluded(self->super.super.model, ao_ray))
85  ++hits;
86  }
87 
88  if (hits > 0)
89  {
90  const float a = (float)hits / (float)self->samplesPerFrame;
91 
92  const vec3f sampleColor = make_vec3f(a, a, 1.f - a);
93  const float alpha = 1.f / (float)self->samplesPerFrame;
94 
95  composite(make_vec4f(sampleColor, alpha), finalColor, 1.0f);
96 
97  if (firstVisibleVoxel)
98  sample.z = min(t, sample.z);
99 
100  firstVisibleVoxel = false;
101  }
102 
103  t += self->rayStep;
104  }
105 
106  sample.alpha = finalColor.w;
107  return clamp(make_vec3f(finalColor) * self->exposure, make_vec3f(0.f), make_vec3f(1.f));
108 }

110 void DensityRenderer_renderSample(uniform Renderer* uniform _self, void* uniform perFrameData,
111  varying ScreenSample& sample)
112 {
113  uniform DensityRenderer* uniform self = (uniform DensityRenderer * uniform) _self;
114  sample.ray.time = self->super.timestamp;
115  if (self->super.anaglyphEnabled)
116  {
117  ScreenSample s = sample;
118  s.ray.org = s.ray.org - self->super.anaglyphIpdOffset;
119  const vec3f colorLeft = DensityRenderer_shadeRay(self, s);
120  const vec3f leftAnaglyphColor = getAnaglyphLeftColor(colorLeft);
121 
122  s = sample;
123  s.ray.org = s.ray.org + self->super.anaglyphIpdOffset;
124  const vec3f colorRight = DensityRenderer_shadeRay(self, s);
125  const vec3f rightAnaglyphColor = getAnaglyphRightColor(colorRight);
126 
127  sample.rgb = leftAnaglyphColor + rightAnaglyphColor;
128  }
129  else
130  sample.rgb = DensityRenderer_shadeRay(self, sample);
131 }
