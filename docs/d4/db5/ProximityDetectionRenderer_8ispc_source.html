46 inline vec3f ProximityDetectionRenderer_shadeRay(const uniform ProximityDetectionRenderer* uniform self,
47  varying ScreenSample& sample)
48 {
49  varying Ray ray = sample.ray;
50  vec4f color = make_vec4f(0.f);
51  sample.alpha = 0.f;
52 
53  int depth = 0;
54  while (color.w < 1.f && depth < self->super.maxRayDepth)
55  {
56  traceRay(self->super.super.super.model, ray);
57 
58  if (ray.geomID < 0)
59  {
60  vec4f bgColor = skyboxMapping((Renderer*)self, ray, (AdvancedMaterial*)self->super.super.bgMaterial);
61  bgColor.w = 1.f;
62  compose(bgColor, color, self->alphaCorrection);
63  return make_vec3f(color) * self->super.super.exposure;
64  }
65 
66  sample.alpha = 1.f;
67  if (depth == 0)
68  sample.z = ray.t;
69 
70  DifferentialGeometry dg;
71  postIntersect(self->super.super.super.model, dg, ray,
72  DG_NG | DG_NS | DG_NORMALIZE | DG_FACEFORWARD | DG_MATERIALID | DG_COLOR | DG_TEXCOORD);
73 
74  uniform Material* material = dg.material;
75  uniform AdvancedMaterial* advancedMaterial = (uniform AdvancedMaterial*)material;
76 
77  vec3f normal = dg.Ns;
78  const vec3f P = dg.P + dg.epsilon * dg.Ng;
79 
80  bool processSurfaceShading = true;
81  varying vec3f ao_dir = getRandomVector(self->super.useHardwareRandomizer, self->super.super.super.fb->size.x,
82  sample, normal, self->super.randomNumber);
83 
84  Ray ao_ray;
85  ao_ray.org = P;
86  ao_ray.dir = ao_dir;
87  ao_ray.t0 = max(0.f, dg.epsilon);
88  ao_ray.t = self->detectionDistance;
89  ao_ray.primID = -1;
90  ao_ray.geomID = -1;
91  ao_ray.instID = -1;
92 
93  traceRay(self->super.super.super.model, ao_ray);
94  if (ao_ray.geomID != -1)
95  {
96  // If AO ray hits a geometry, no surface is shading required
97  processSurfaceShading = false;
98 
99  DifferentialGeometry ao_dg;
100  postIntersect(self->super.super.super.model, ao_dg, ao_ray, DG_MATERIALID);
101 
102  const bool doDetectionTest = self->detectionOnDifferentMaterial ? material != ao_dg.material : true;
103  if (doDetectionTest)
104  {
105  const float a = ao_ray.t / self->detectionDistance;
106  const vec4f touchColor = make_vec4f(a > nearFarThreshold ? self->nearColor : self->farColor, 1.f);
107  compose(touchColor, color, self->alphaCorrection);
108  sample.alpha = 1.f;
109  if (depth == 0)
110  sample.z = ray.t;
111  }
112  }
113 
114  if (processSurfaceShading && self->surfaceShadingEnabled)
115  {
116  MaterialShadingMode shadingMode = undefined_shading_mode;
117  vec3f Kd;
118  float Ns = 1.f;
119  float opacity = 1.f;
120  if (!advancedMaterial)
121  {
122  Kd = make_vec3f(dg.color);
123  opacity = dg.color.w;
124  }
125  else
126  foreach_unique(mat in advancedMaterial)
127  {
128  Kd = mat->Kd * make_vec3f(dg.color);
129  shadingMode = mat->shadingMode;
130  Ns = mat->Ns;
131  opacity = dg.color.w * mat->d;
132  }
133 
134  for (uniform int i = 0; self->super.super.lights && i < self->super.super.numLights; i++)
135  {
136  const uniform Light* uniform light = self->super.super.lights[i];
137 
138  const vec2f samplingLocation = make_vec2f(0.f);
139  const Light_SampleRes lightSample = light->sample(light, dg, samplingLocation);
140  const float radiance = reduce_max(lightSample.weight);
141 
142  if (reduce_max(radiance) > 0.f)
143  {
144  // Head-light shading
145  float cosNL = (shadingMode == undefined_shading_mode ? 1.f : max(0.f, dot(neg(ray.dir), dg.Ns)));
146 
147  switch (shadingMode)
148  {
149  case electron:
150  case electron_transparency:
151  cosNL = 1.f - pow(cosNL, Ns);
152  opacity *= (shadingMode == electron_transparency ? cosNL : 1.f);
153  break;
154  case diffuse_transparency:
155  opacity *= cosNL;
156  break;
157  case cartoon:
158  const unsigned int angleAsInt = cosNL * Ns;
159  cosNL = (float)(angleAsInt) / Ns;
160  break;
161  case perlin:
162  const vec3f point = ray.org + ray.t * ray.dir;
163  float n = noise(point);
164  if (ray.t < dg.epsilon * 1.5e4f)
165  n += worleyNoise(point, 2.f);
166  else
167  n *= 2.f;
168 
169  n = clamp(n, 0.f, 1.f);
170  cosNL *= n;
171  break;
172  }
173 
174  const vec4f shadedColor = make_vec4f(Kd * cosNL * radiance, 1.f);
175  compose(shadedColor, color, self->alphaCorrection);
176  }
177  }
178  break;
179  }
180 
181  ray.t0 = max(0.f, ray.t + dg.epsilon);
182  ray.t = inf;
183  ray.primID = -1;
184  ray.geomID = -1;
185  ray.instID = -1;
186  ++depth;
187  }
188 
189  return make_vec3f(color) * self->super.super.exposure;
190 }

192 void ProximityDetectionRenderer_renderSample(uniform Renderer* uniform _self, void* uniform perFrameData,
193  varying ScreenSample& sample)
194 {
195  uniform ProximityDetectionRenderer* uniform self = (uniform ProximityDetectionRenderer * uniform) _self;
196  sample.ray.time = self->super.super.timestamp;
197  if (self->super.super.anaglyphEnabled)
198  {
199  ScreenSample s = sample;
200  s.ray.org = s.ray.org - self->super.super.anaglyphIpdOffset;
201  const vec3f colorLeft = ProximityDetectionRenderer_shadeRay(self, s);
202  const vec3f leftAnaglyphColor = getAnaglyphLeftColor(colorLeft);
203 
204  s = sample;
205  s.ray.org = s.ray.org + self->super.super.anaglyphIpdOffset;
206  const vec3f colorRight = ProximityDetectionRenderer_shadeRay(self, s);
207  const vec3f rightAnaglyphColor = getAnaglyphRightColor(colorRight);
208 
209  sample.rgb = leftAnaglyphColor + rightAnaglyphColor;
210  }
211  else
212  sample.rgb = ProximityDetectionRenderer_shadeRay(self, sample);
213 }
