36 Protein::Protein(Scene& scene, const ProteinDetails& details)
37  : Molecule(scene, details.chainIds)
38  , _details(details)
39 {
40  size_t lineIndex{0};
41 
42  std::stringstream lines{_details.contents};
43  std::string line;
44  std::string title;
45  std::string header{_details.name};
46 
47  while (getline(lines, line, '\n'))
48  {
49  if (line.find(KEY_ATOM) == 0)
50  _readAtom(line, _details.loadHydrogen);
51  else if (details.loadNonPolymerChemicals && line.find(KEY_HETATM) == 0)
52  _readAtom(line, _details.loadHydrogen);
53  else if (line.find(KEY_HEADER) == 0)
54  header = _readHeader(line);
55  else if (line.find(KEY_TITLE) == 0)
56  title = _readTitle(line);
57  else if (details.loadBonds && line.find(KEY_CONECT) == 0)
58  _readConnect(line);
59  else if (line.find(KEY_SEQRES) == 0)
60  _readSequence(line);
61  // else if (line.find(KEY_REMARK) == 0)
62  // _readRemark(line);
63  }
64 
65  if (_residueSequenceMap.empty())
66  {
67  // Build AA sequences from ATOMS if no SEQRES record exists
68  size_t previousReqSeq = std::numeric_limits<size_t>::max();
69  for (const auto& atom : _atomMap)
70  {
71  auto& sequence = _residueSequenceMap[atom.second.chainId];
72  if (previousReqSeq != atom.second.reqSeq)
73  sequence.resNames.push_back(atom.second.resName);
74  previousReqSeq = atom.second.reqSeq;
75  }
76  for (auto& sequence : _residueSequenceMap)
77  sequence.second.numRes = sequence.second.resNames.size();
78  }
79 
80  // Build 3d models according to atoms positions (re-centered to origin)
81  if (details.recenter)
82  {
83  brayns::Boxf newBounds;
84  const auto& center = _bounds.getCenter();
85  for (auto& atom : _atomMap)
86  {
87  atom.second.position -= center;
88  newBounds.merge(atom.second.position);
89  }
90  _bounds = newBounds;
91  }
92 
93  _buildModel(_details.assemblyName, _details.name, _details.pdbId, header,
94  _details.representation, _details.atomRadiusMultiplier,
95  _details.loadBonds);
96 
97  _buildAminoAcidBounds();
98  _computeReqSetOffset();
99 
100  // Trans-membrane information
101  const auto transmembraneParams =
102  doublesToVector2d(details.transmembraneParams);
103  _transMembraneOffset = transmembraneParams.x;
104  _transMembraneRadius = transmembraneParams.y;
105 }

119 void Protein::setColorScheme(const ProteinColorScheme& colorScheme,
120  const Palette& palette, const size_ts& chainIds)
121 {
122  switch (colorScheme)
123  {
124  case ProteinColorScheme::none:
125  for (auto& atom : _atomMap)
126  _setMaterialDiffuseColor(atom.first, atomColorMap[0]);
127  break;
128  case ProteinColorScheme::atoms:
129  _setAtomColorScheme();
130  break;
131  case ProteinColorScheme::chains:
132  _setChainColorScheme(palette);
133  break;
134  case ProteinColorScheme::residues:
135  _setResiduesColorScheme(palette);
136  break;
137  case ProteinColorScheme::amino_acid_sequence:
138  _setAminoAcidSequenceColorScheme(palette);
139  break;
140  case ProteinColorScheme::glycosylation_site:
141  _setGlycosylationSiteColorScheme(palette);
142  break;
143  case ProteinColorScheme::region:
144  _setRegionColorScheme(palette, chainIds);
145  break;
146  default:
147  PLUGIN_THROW("Unknown colorscheme");
148  }

151 void Protein::_setRegionColorScheme(const Palette& palette,
152  const size_ts& chainIds)
153 {
154  size_t atomCount = 0;
155  for (auto& atom : _atomMap)
156  {
157  bool applyColor{true};
158  if (!chainIds.empty())
159  {
160  const size_t chainId =
161  static_cast<size_t>(atom.second.chainId[0] - 64);
162  applyColor = (std::find(chainIds.begin(), chainIds.end(),
163  chainId) != chainIds.end());
164  }
165  if (applyColor)
166  _setMaterialDiffuseColor(atom.first, palette[atom.second.reqSeq]);
167  }
168 
169  PLUGIN_INFO(3, "Applying Amino Acid Sequence color scheme ("
170  << (atomCount > 0 ? "2" : "1") << ")");

173 void Protein::_setGlycosylationSiteColorScheme(const Palette& palette)
174 {
175  if (palette.size() != 2)
176  PLUGIN_THROW("Invalid palette size. 2 colors are expected");
177 
178  // Initialize atom colors
179  for (const auto& atom : _atomMap)
180  {
181  const size_t index = static_cast<size_t>(atom.second.chainId[0]) - 63;
182  _setMaterialDiffuseColor(atom.first, palette[0]);
183  }
184 
185  const auto sites = getGlycosylationSites({});
186 
187  for (const auto chain : sites)
188  for (const auto site : chain.second)
189  for (const auto& atom : _atomMap)
190  if (atom.second.chainId == chain.first &&
191  atom.second.reqSeq == site)
192  _setMaterialDiffuseColor(atom.first, palette[1]);
193 
194  PLUGIN_INFO(3, "Applying Glycosylation Site color scheme ("
195  << (sites.size() > 0 ? "2" : "1") << ")");

198 const std::map<std::string, size_ts> Protein::getGlycosylationSites(
199  const size_ts& siteIndices) const
200 {
201  std::map<std::string, size_ts> sites;
202  for (const auto& sequence : _residueSequenceMap)
203  {
204  std::string shortSequence;
205  for (const auto& resName : sequence.second.resNames)
206  shortSequence += aminoAcidMap[resName].shortName;
207 
208  for (size_t i = 0; i < shortSequence.length(); ++i)
209  {
210  const auto offsetIndex = i + sequence.second.offset;
211  bool acceptSite{true};
212  if (!siteIndices.empty())
213  {
214  const auto it = find(siteIndices.begin(), siteIndices.end(), i);
215  acceptSite = (it != siteIndices.end());
216  }
217 
218  const char aminoAcid = shortSequence[offsetIndex];
219  if (aminoAcid == 'N' && acceptSite)
220  {
221  if (i < shortSequence.length() - 2)
222  {
223  const auto aminAcid1 = shortSequence[offsetIndex + 1];
224  const auto aminAcid2 = shortSequence[offsetIndex + 2];
225  if ((aminAcid2 == 'T' || aminAcid2 == 'S') &&
226  aminAcid1 != 'P')
227  sites[sequence.first].push_back(i);
228  }
229  }
230  }
231  }
232 
233  for (const auto& site : sites)
234  {
235  std::string indices = "[";
236  for (const auto& index : site.second)
237  {
238  if (indices.length() > 1)
239  indices += ",";
240  indices += std::to_string(index + 1); // Indices start at 1, not 0
241  }
242  indices += "]";
243  PLUGIN_INFO(3, "Found " << site.second.size() << " glycosylation sites "
244  << indices << " on sequence " << site.first);
245  }
246  return sites;
247 }

263 void Protein::_getSitesTransformations(
264  Vector3ds& positions, Quaternions& rotations,
265  const std::map<std::string, size_ts>& sitesPerChain) const
266 {
267  for (const auto& chain : sitesPerChain)
268  {
269  const auto itAminoAcids = _aminoAcidBounds.find(chain.first);
270  if (itAminoAcids == _aminoAcidBounds.end())
271  PLUGIN_THROW("Invalid chain");
272 
273  const auto aminoAcidsPerChain = (*itAminoAcids).second;
274  for (const auto site : chain.second)
275  {
276  // Protein center
277  const Vector3d proteinCenter = _bounds.getCenter();
278  Boxf siteBounds;
279  Vector3d siteCenter;
280 
281  const auto offsetSite = site;
282 
283  // Site center
284  const auto it = aminoAcidsPerChain.find(offsetSite);
285  if (it != aminoAcidsPerChain.end())
286  {
287  siteBounds = (*it).second;
288  siteCenter = siteBounds.getCenter();
289 
290  // rotation is determined by the center of the site and the
291  // center of the protein
292  const auto bindrotation = normalize(siteCenter - proteinCenter);
293  positions.push_back(siteCenter);
294  rotations.push_back(safeQuatlookAt(bindrotation));
295  }
296  else
297  PLUGIN_WARN("Chain: "
298  << chain.first << ", Site " << site + 1
299  << " is not available in the protein source");
300  }
301  }
302 }

316 void Protein::getSugarBindingSites(Vector3ds& positions, Quaternions& rotations,
317  const size_ts& siteIndices,
318  const size_ts& chainIds) const
319 {
320  positions.clear();
321  rotations.clear();
322 
323  std::set<std::string> chainIdsAsString;
324  for (const auto& atom : _atomMap)
325  {
326  bool acceptChain{true};
327  const size_t chainId = static_cast<size_t>(atom.second.chainId[0] - 64);
328  if (!chainIds.empty())
329  acceptChain = (std::find(chainIds.begin(), chainIds.end(),
330  chainId) != chainIds.end());
331 
332  if (acceptChain)
333  chainIdsAsString.insert(atom.second.chainId);

337  for (const auto& chainIdAsString : chainIdsAsString)
338  sites[chainIdAsString] = siteIndices;
339 
340  _getSitesTransformations(positions, rotations, sites);

343 void Protein::setAminoAcid(const AminoAcidDetails& details)
344 {
345  for (auto& sequence : _residueSequenceMap)
346  {
347  bool acceptChain = true;
348  if (!details.chainIds.empty())

350  const size_t chainId = static_cast<size_t>(sequence.first[0]) - 64;

352  find(details.chainIds.begin(), details.chainIds.end(), chainId);
353  acceptChain = (it == details.chainIds.end());
354  }
355 
356  if (details.index >= sequence.second.resNames.size())
357  PLUGIN_THROW("Invalid index for the amino acid sequence");
358 
359  if (acceptChain)
360  sequence.second.resNames[details.index] =
361  details.aminoAcidShortName;

365 void Protein::_processInstances(ModelDescriptorPtr md,
366  const Vector3ds& positions,
367  const Quaternions& rotations,
368  const Quaterniond& moleculerotation,
369  const AnimationDetails& randInfo)
370 {
371  size_t count = 0;
372  const auto& proteinInstances = _modelDescriptor->getInstances();
373  for (const auto& proteinInstance : proteinInstances)
374  {
375  const auto& proteinTransformation = proteinInstance.getTransformation();
376  for (size_t i = 0; i < positions.size(); ++i)
377  {
378  const auto& position = positions[i];
379  auto rotation = rotations[i];
380 
381  Transformation glycanTransformation;
382  glycanTransformation.setTranslation(position);
383 
384  if (randInfo.rotationSeed != 0)
385  rotation =
386  Shape::weightedRandomRotation(rotation,
387  randInfo.rotationSeed, i,
388  randInfo.rotationStrength);
389 
390  glycanTransformation.setRotation(moleculerotation * rotation);
391 
392  const Transformation combinedTransformation =
393  proteinTransformation * glycanTransformation;
394 
395  if (count == 0)
396  md->setTransformation(combinedTransformation);
397  const ModelInstance instance(true, false, combinedTransformation);
398  md->addInstance(instance);
399  ++count;
400  }
401  }
402 }

404 void Protein::addGlycan(const SugarDetails& details)
405 {
406  if (_glycans.find(details.name) != _glycans.end())
407  PLUGIN_THROW("A glycan named " + details.name +
408  " already exists in protein " + _details.name +
409  " of assembly " + _details.assemblyName);
410 
411  Vector3ds glycanPositions;
412  Quaternions glycanRotations;
413  getGlycosilationSites(glycanPositions, glycanRotations,
414  details.siteIndices);
415 
416  if (glycanPositions.empty())
417  PLUGIN_THROW("No glycosylation site was found on " +
418  details.proteinName);
419 
420  // Create glycans and attach them to the glycosylation sites of the target
421  // protein
422  GlycansPtr glycans(new Glycans(_scene, details));
423  auto modelDescriptor = glycans->getModelDescriptor();
424  const Quaterniond proteinrotation = doublesToQuaterniond(details.rotation);
425 
426  const auto randInfo = doublesToAnimationDetails(details.animationParams);
427  _processInstances(modelDescriptor, glycanPositions, glycanRotations,
428  proteinrotation, randInfo);
429 
430  _glycans[details.name] = std::move(glycans);
431  _scene.addModel(modelDescriptor);
432 }
