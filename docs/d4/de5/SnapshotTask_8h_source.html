63  SnapshotFunctor(Engine& engine, SnapshotParams&& params, ImageGenerator& imageGenerator)
64  : _params(std::move(params))
65  , _camera(engine.createCamera())
66  , _imageGenerator(imageGenerator)
67  , _engine(engine)
68  {
69  auto& applicationParameters = engine.getParametersManager().getApplicationParameters();
70  const auto& engineName = applicationParameters.getEngine();
71  if (engineName == ENGINE_OPTIX_6)
72  CORE_THROW("Snapshot are currently not supported by the " + engineName + " engine");
73 
74  const auto& parametersManager = engine.getParametersManager();
75  if (_params.animParams == nullptr)
76  {
77  _params.animParams = std::make_unique<AnimationParameters>(parametersManager.getAnimationParameters());
78  }
79 
80  if (_params.geometryParams == nullptr)
81  {
82  _params.geometryParams = std::make_unique<GeometryParameters>(parametersManager.getGeometryParameters());
83  }
84 
85  if (_params.volumeParams == nullptr)
86  {
87  _params.volumeParams = std::make_unique<VolumeParameters>(parametersManager.getVolumeParameters());
88  }
89 
90  _scene = engine.createScene(*_params.animParams, *_params.geometryParams, *_params.volumeParams);
91 
92  if (_params.camera)
93  {
94  *_camera = *_params.camera;
95  _camera->setCurrentType(engine.getCamera().getCurrentType());
96  _camera->clonePropertiesFrom(engine.getCamera());
97  }
98  else
99  *_camera = engine.getCamera();
100 
101  _renderer = engine.createRenderer(*_params.animParams, parametersManager.getRenderingParameters());
102 
103  const auto& renderer = engine.getRenderer();
104  _renderer->setCurrentType(renderer.getCurrentType());
105  _renderer->clonePropertiesFrom(renderer);
106 
107  _scene->copyFrom(engine.getScene());
108  }

110  ImageGenerator::ImageBase64 operator()()
111  {
112  _scene->commit();
113 
114  _camera->updateProperty(CAMERA_PROPERTY_ASPECT_RATIO.name, double(_params.size.x) / _params.size.y);
115  _camera->commit();
116 
117  _renderer->setSamplesPerPixel(1);
118  _renderer->setSubsampling(1);
119  _renderer->setCamera(_camera);
120  _renderer->setEngine(&_engine);
121  _renderer->commit();
122 
123  std::stringstream msg;
124  msg << "Render snapshot";
125  if (!_params.name.empty())
126  msg << " " << string_utils::shortenString(_params.name);
127  msg << " ...";
128 
129  const auto isStereo = _camera->hasProperty(CAMERA_PROPERTY_STEREO.name) &&
130  _camera->getProperty<bool>(CAMERA_PROPERTY_STEREO.name);
131  const auto names = isStereo ? strings{"0L", "0R"} : strings{DEFAULT};
132  std::vector<FrameBufferPtr> frameBuffers;
133  for (const auto& name : names)
134  frameBuffers.push_back(_engine.createFrameBuffer(name, _params.size, FrameBufferFormat::rgba_i8));
135 
136  while (frameBuffers[0]->numAccumFrames() != size_t(_params.samplesPerPixel))
137  {
138  for (auto frameBuffer : frameBuffers)
139  {
140  _camera->setBufferTarget(frameBuffer->getName());
141  _camera->markModified(false);
142  _camera->commit();
143  _camera->resetModified();
144  _renderer->render(frameBuffer);
145  frameBuffer->incrementAccumFrames();
146  }
147 
148  progress(msg.str(), 1.f / frameBuffers[0]->numAccumFrames(),
149  float(frameBuffers[0]->numAccumFrames()) / _params.samplesPerPixel);
150  }
151 
152  if (!_params.filePath.empty() && frameBuffers.size() == 1)
153  {
154  auto& fb = *frameBuffers[0];
155  _writeToDisk(fb);
156 
157  return ImageGenerator::ImageBase64();
158  }
159  else
160  return _imageGenerator.createImage(frameBuffers, _params.format, _params.quality);
161  }
162 
163 private:
164  void _writeToDisk(FrameBuffer& fb)
165  {
166  auto image = fb.getImage();
167  auto fif = _params.format == "jpg" ? FIF_JPEG : FreeImage_GetFIFFromFormat(_params.format.c_str());
168 
169  if (fif == FIF_JPEG)
170  image.reset(FreeImage_ConvertTo24Bits(image.get()));
171  else if (fif == FIF_UNKNOWN)
172  {
173  std::cerr << "Unknown format: " << _params.format << std::endl;
174  return;
175  }
176 
177  int flags = _params.quality;
178  if (fif == FIF_TIFF)
179  flags = TIFF_NONE;
180 
181  core::freeimage::MemoryPtr memory(FreeImage_OpenMemory());
182 
183  FreeImage_SaveToMemory(fif, image.get(), memory.get(), flags);
184 
185  BYTE* pixels = nullptr;
186  DWORD numPixels = 0;
187  FreeImage_AcquireMemory(memory.get(), &pixels, &numPixels);
188 
189  std::ofstream file;
190  const std::string path = _params.filePath + "." + _params.format;
191  file.open(path, std::ios_base::binary);
192  if (!file.is_open())
193  {
194  std::cerr << "Failed to create " << path << std::endl;
195  return;
196  }
197 
198  file.write((char*)pixels, numPixels);
199  file.close();
200  }
