52 bool Material::_loadTexture(const std::string& fileName, const TextureType type)
53 {
54  if (_textures.find(fileName) != _textures.end())
55  return true;
56 
57  auto texture = ImageManager::importTextureFromFile(fileName, type);
58  if (!texture)
59  return false;
60 
61  _textures[fileName] = texture;
62  CORE_DEBUG(fileName << ": " << texture->width << "x" << texture->height << "x" << (int)texture->channels << "x"
63  << (int)texture->depth << " added to the texture cache");
64  return true;
65 }

67 void Material::setTexture(const std::string& fileName, const TextureType type)
68 {
69  auto i = _textureDescriptors.find(type);
70  if (i != _textureDescriptors.end() && i->second->filename == fileName)
71  return;
72 
73  if (_textures.find(fileName) == _textures.end())
74  if (!_loadTexture(fileName, type))
75  throw std::runtime_error("Failed to load texture from " + fileName);
76  _textureDescriptors[type] = _textures[fileName];
77  markModified();
78 }
