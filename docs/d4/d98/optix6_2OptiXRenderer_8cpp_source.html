47 void OptiXRenderer::render(FrameBufferPtr frameBuffer)
48 {
49  if (!frameBuffer->getAccumulation() && frameBuffer->numAccumFrames() > 0)
50  return;
51 
52  // Provide a random seed to the renderer
53  ::optix::float4 jitter = {(float)rand() / (float)RAND_MAX, (float)rand() / (float)RAND_MAX,
54  (float)rand() / (float)RAND_MAX, (float)rand() / (float)RAND_MAX};
55  auto context = OptiXContext::get().getOptixContext();
56  context[CONTEXT_RENDERER_JITTER]->setFloat(jitter);
57  context[CONTEXT_RENDERER_FRAME]->setUint(frameBuffer->numAccumFrames());
58 
59  // Render
60  frameBuffer->map();
61  const auto size = frameBuffer->getSize();
62  context->launch(0, size.x, size.y);
63  frameBuffer->unmap();
64 
65  frameBuffer->markModified();
66 }

68 void OptiXRenderer::commit()
69 {
70  auto& scene = _engine->getScene();
71  if (!_renderingParameters.isModified() && !scene.isModified() && !isModified())
72  return;
73 
74  const bool updateMaterials = isModified() || scene.isModified();
75 
76  // If renderer or scene has changed we have to go through all materials in the scene and update the renderer.
77  if (updateMaterials)
78  {
79  const auto& currentRendererType = _engine->getRenderer().getCurrentType();
80  const auto renderProgram = OptiXContext::get().getRenderer(currentRendererType);
81 
82  scene.visitModels(
83  [&](Model& model)
84  {
85  for (const auto& kv : model.getMaterials())
86  {
87  auto optixMaterial = dynamic_cast<OptiXMaterial*>(kv.second.get());
88  const bool textured = optixMaterial->isTextured();
89  auto material = optixMaterial->getOptixMaterial();
90  if (material)
91  {
92  const auto program =
93  textured ? renderProgram->closest_hit_textured : renderProgram->closest_hit;
94  material->setClosestHitProgram(0, program);
95  material->setAnyHitProgram(1, renderProgram->any_hit);
96  }
97  else
98  CORE_DEBUG("No OptiX material initialized for core material " + kv.second->getName());
99  }
100  });
101  }
102 
103  // Upload common properties
104  auto context = OptiXContext::get().getOptixContext();
105  const auto bounds = scene.getBounds();
106  const auto epsilon = bounds.getSize().x / 1000.f;
107  context[CONTEXT_RENDERER_SCENE_EPSILON]->setFloat(epsilon);
108  context[CONTEXT_RENDERER_RADIANCE_RAY_TYPE]->setUint(0);
109  context[CONTEXT_RENDERER_SHADOW_RAY_TYPE]->setUint(1);
110  context[CONTEXT_RENDERER_AMBIENT_LIGHT_COLOR]->setFloat(_backgroundColor.x, _backgroundColor.y, _backgroundColor.z);
111  context[CONTEXT_RENDERER_BACKGROUND_COLOR]->setFloat(_backgroundColor.x, _backgroundColor.y, _backgroundColor.z);
112  context[CONTEXT_RENDERER_SAMPLES_PER_PIXEL]->setUint(_spp);
113 
114  toOptiXProperties(getPropertyMap());
115 }
