30 inline vec3f RadianceRenderer_shadeRay(const uniform RadianceRenderer* uniform self, varying ScreenSample& sample)
31 {
32  Ray ray = sample.ray;
33  sample.z = inf;
34  sample.alpha = 0.f;
35 
36  traceRay(self->super.model, ray);
37 
38  if (ray.geomID < 0)
39  // No intersection, use environment map or background color
40  return make_vec3f(0.f);
41 
42  DifferentialGeometry dg;
43  postIntersect(self->super.model, dg, ray,
44  DG_NS | DG_NG | DG_NORMALIZE | DG_FACEFORWARD | DG_MATERIALID | DG_COLOR | DG_TEXCOORD);
45 
46  uniform AdvancedMaterial* material = (uniform AdvancedMaterial*)dg.material;
47 
48  vec3f radiance = make_vec3f(0.f);
49  if (material)
50  foreach_unique(mat in material)
51  {
52  radiance = make_vec3f(mat->a);
53  if (valid(mat->map_a))
54  {
55  const vec4f radianceFromMap = get4f(mat->map_a, dg);
56  radiance = make_vec3f(radianceFromMap);
57  }
58  }
59 
60  sample.z = ray.t; // Z-Depth
61  sample.alpha = 1.f; // Transparency is not managed by this renderer
62 
63  return radiance;
64 }

67  varying ScreenSample& sample)
