65 ModelDescriptorPtr NeuronsLoader::importFromStorage(const std::string& storage, const LoaderProgress& callback,
66  const PropertyMap& properties) const
67 {
68  PropertyMap props = _defaults;
69  props.merge(properties);
70 
71  details::NeuronsDetails details;
72  const auto baseName = std::filesystem::path(storage).filename();
73  details.assemblyName = baseName;
74  details.populationName = baseName;
75  details.sqlNodeFilter = props.getProperty<std::string>(LOADER_PROPERTY_DATABASE_SQL_FILTER.name);
76  details.radiusMultiplier = props.getProperty<double>(LOADER_PROPERTY_RADIUS_MULTIPLIER.name);
77  details.populationColorScheme = stringToEnum<morphology::PopulationColorScheme>(
78  props.getProperty<std::string>(LOADER_PROPERTY_POPULATION_COLOR_SCHEME.name));
79  details.morphologyColorScheme = stringToEnum<morphology::MorphologyColorScheme>(
80  props.getProperty<std::string>(LOADER_PROPERTY_MORPHOLOGY_COLOR_SCHEME.name));
81  details.realismLevel = static_cast<int64_t>(morphology::MorphologyRealismLevel::none);
82  details.realismLevel += (props.getProperty<bool>(LOADER_PROPERTY_MORPHOLOGY_REALISM_LEVEL_SOMA.name)
83  ? static_cast<int64_t>(morphology::MorphologyRealismLevel::soma)
84  : 0);
85  details.realismLevel += (props.getProperty<bool>(LOADER_PROPERTY_MORPHOLOGY_REALISM_LEVEL_DENDRITE.name)
86  ? static_cast<int64_t>(morphology::MorphologyRealismLevel::dendrite)
87  : 0);
88  details.realismLevel += (props.getProperty<bool>(LOADER_PROPERTY_NEURONS_REALISM_LEVEL_SPINE.name)
89  ? static_cast<int64_t>(morphology::MorphologyRealismLevel::spine)
90  : 0);
91  details.realismLevel += (props.getProperty<bool>(LOADER_PROPERTY_MORPHOLOGY_REALISM_LEVEL_AXON.name)
92  ? static_cast<int64_t>(morphology::MorphologyRealismLevel::axon)
93  : 0);
94  details.realismLevel += (props.getProperty<bool>(LOADER_PROPERTY_NEURONS_REALISM_LEVEL_EXTERNALS.name)
95  ? static_cast<int64_t>(morphology::MorphologyRealismLevel::externals)
96  : 0);
97  details.realismLevel += (props.getProperty<bool>(LOADER_PROPERTY_MORPHOLOGY_REALISM_LEVEL_INTERNALS.name)
98  ? static_cast<int64_t>(morphology::MorphologyRealismLevel::internals)
99  : 0);
100  details.loadSomas = props.getProperty<bool>(LOADER_PROPERTY_MORPHOLOGY_LOAD_SOMA.name);
101  details.loadAxon = props.getProperty<bool>(LOADER_PROPERTY_NEURONS_LOAD_AXON.name);
102  details.loadBasalDendrites = props.getProperty<bool>(LOADER_PROPERTY_NEURONS_LOAD_BASAL_DENDRITES.name);
103  details.loadApicalDendrites = props.getProperty<bool>(LOADER_PROPERTY_NEURONS_LOAD_APICAL_DENDRITES.name);
104  details.generateInternals = props.getProperty<bool>(LOADER_PROPERTY_MORPHOLOGY_GENERATE_INTERNALS.name);
105  details.generateExternals = props.getProperty<bool>(LOADER_PROPERTY_NEURONS_GENERATE_EXTERNALS.name);
106  details.morphologyRepresentation = stringToEnum<morphology::MorphologyRepresentation>(
107  props.getProperty<std::string>(LOADER_PROPERTY_MORPHOLOGY_REPRESENTATION.name));
108  details.alignToGrid = props.getProperty<double>(LOADER_PROPERTY_ALIGN_TO_GRID.name);
109  details.synapsesType = stringToEnum<morphology::MorphologySynapseType>(
110  props.getProperty<std::string>(LOADER_PROPERTY_NEURONS_SYNAPSE_TYPE.name));
111  const auto position = props.getProperty<std::array<double, 3>>(LOADER_PROPERTY_POSITION.name);
112  const Vector3d pos = core::Vector3d(position[0], position[1], position[2]);
113  const auto rotation = props.getProperty<std::array<double, 4>>(LOADER_PROPERTY_ROTATION.name);
114  const Quaterniond rot = core::Quaterniond(rotation[3], rotation[0], rotation[1], rotation[2]);
115  const auto scale = props.getProperty<std::array<double, 3>>(LOADER_PROPERTY_SCALE.name);
116  details.scale = {scale[0], scale[1], scale[2]};
117  Neurons Neurons(_scene, details, pos, rot, callback);
118  return std::move(Neurons.getModelDescriptor());
119 }
