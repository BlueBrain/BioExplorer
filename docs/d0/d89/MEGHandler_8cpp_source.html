103 ModelMetadata MEGHandler::buildModel(Model& model, const double voxelSize, const double density)
104 {
105  _voxelSize = voxelSize;
106  _density = density;
107 
108  if (density > 1.f || density <= 0.f)
109  PLUGIN_THROW("Density should be higher > 0 and <= 1");
110 
111  _bounds.reset();
112  for (const auto& transformation : _transformations)
113  {
114  const Vector3f position = get_translation(transformation);
115  _bounds.merge(position + DEFAULT_EVENT_VALUE);
116  _bounds.merge(position - DEFAULT_EVENT_VALUE);
117  }
118 
119  // Extend bounds to 200%
120  _bounds.merge(_bounds.getCenter() + _bounds.getSize());
121  _bounds.merge(_bounds.getCenter() - _bounds.getSize());
122 
123  const size_t materialId = 0;
124  auto material = model.createMaterial(materialId, "MEG");
125  TriangleMesh mesh = createBox(_bounds.getMin(), _bounds.getMax());
126  model.getTriangleMeshes()[materialId] = mesh;
127  model.updateBounds();
128 
129  ModelMetadata metadata = {{"Scene AABB",
130  Vector3fToString(_bounds.getMin()) + ", " + Vector3fToString(_bounds.getMax())},
131  {"Scene dimension", Vector3fToString(_bounds.getSize())},
132  {"Element spacing ", Vector3fToString(_spacing)},
133  {"Volume dimensions", Vector3fToString(_dimensions)},
134  {"Element offset", Vector3fToString(_offset)},
135  {"Data size", std::to_string(_frameSize)}};
136  return metadata;
137 }

139 void MEGHandler::_buildOctree()
140 {
141  const brion::Frame frame = _currentFrameFuture.get();
142  if (!frame.data)
143  return;
144 
145  const auto voltages = std::move(*frame.data);
146  if (voltages.size() != _transformations.size())
147  PLUGIN_ERROR("Invalid number of values: " << voltages.size() << " instead of " << _transformations.size());
148  uint64_t index = 0;
149  const uint32_t densityRatio = 1.f / _density;
150  OctreePoints points;
151  for (const auto& transformation : _transformations)
152  {
153  if (index % densityRatio == 0)
154  {
155  ++index;
156  continue;
157  }
158  const Vector3f position = get_translation(transformation);
159  const auto value = voltages[index] - DEFAULT_VOLTAGE_REST_VALUE;
160  OctreePoint point;
161  point.position = position;
162  point.radius = value;
163  point.value = value;
164  points.push_back(point);
165  ++index;
166  }
167 
168  const PointOctree accelerator(points, _voxelSize, _bounds.getMin(), _bounds.getMax());
169  const uint32_t volumeSize = accelerator.getVolumeSize();
170  _offset = _bounds.getMin();
171  _dimensions = accelerator.getVolumeDimensions();
172  _spacing = Vector3f(_bounds.getSize()) / Vector3f(_dimensions);
173 
174  const auto& indices = accelerator.getFlatIndices();
175  const auto& data = accelerator.getFlatData();
176  _frameData.clear();
177  _frameData.push_back(_offset.x);
178  _frameData.push_back(_offset.y);
179  _frameData.push_back(_offset.z);
180  _frameData.push_back(_spacing.x);
181  _frameData.push_back(_spacing.y);
182  _frameData.push_back(_spacing.z);
183  _frameData.push_back(_dimensions.x);
184  _frameData.push_back(_dimensions.y);
185  _frameData.push_back(_dimensions.z);
186  _frameData.push_back(accelerator.getOctreeSize());
187  _frameData.push_back(indices.size());
188  _frameData.insert(_frameData.end(), indices.begin(), indices.end());
189  _startDataIndex = _frameData.size();
190  _frameData.insert(_frameData.end(), data.begin(), data.end());
191  _frameSize = _frameData.size();
192 }

194 void* MEGHandler::getFrameData(const uint32_t frame)
195 {
196  const auto boundedFrame = _startFrame + _getBoundedFrame(frame);
197  if (boundedFrame == _currentFrame)
198  return nullptr;
199 
200  if (!_currentFrameFuture.valid() && _currentFrame != boundedFrame)
201  _triggerLoading(boundedFrame);
202 
203  if (!_makeFrameReady(boundedFrame))
204  return nullptr;
205 
206  return _frameData.data();
207 }

209 void MEGHandler::_triggerLoading(const uint32_t frame)
210 {
211  float timestamp = frame * _dt;
212  timestamp = std::min(static_cast<float>(_nbFrames), timestamp);
213 
214  if (_currentFrameFuture.valid())
215  _currentFrameFuture.wait();
216 
217  _ready = false;
218  _currentFrameFuture = _report->loadFrame(timestamp);
219 }

221 bool MEGHandler::_isFrameLoaded() const
222 {
223  if (!_currentFrameFuture.valid())
224  return false;
225 
226  if (_synchronousMode)
227  {
228  _currentFrameFuture.wait();
229  return true;
230  }
231 
232  return _currentFrameFuture.wait_for(std::chrono::milliseconds(0)) == std::future_status::ready;
233 }

235 bool MEGHandler::_makeFrameReady(const uint32_t frame)
236 {
237  if (_isFrameLoaded())
238  {
239  try
240  {
241  _buildOctree();
242  }
243  catch (const std::exception& e)
244  {
245  PLUGIN_ERROR("Error loading simulation frame " << frame << ": " << e.what());
246  return false;
247  }
248  _currentFrame = frame;
249  _ready = true;
250  }
251  return true;
252 }
