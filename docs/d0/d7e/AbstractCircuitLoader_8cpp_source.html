80 bool AbstractCircuitLoader::isSupported(const std::string &filename, const std::string & /*extension*/) const
81 {
82  const auto ends_with = [](const std::string &value, const std::string &ending)
83  {
84  if (ending.size() > value.size())
85  return false;
86  return std::equal(ending.rbegin(), ending.rend(), value.rbegin());
87  };
88 
89  for (const auto &name : LOADER_EXTENSIONS)
90  if (ends_with(filename, name))
91  return true;
92 
93  const auto contains = [](const std::string &value, const std::string &keyword)
94  {
95  if (value.size() < keyword.size())
96  return false;
97 
98  const auto lastSlash = value.find_last_of("/");
99  std::string compareTo = value;
100  if (lastSlash != std::string::npos)
101  compareTo = value.substr(lastSlash + 1);
102  return compareTo.find(keyword) != std::string::npos;
103  };
104 
105  for (const auto &keyw : LOADER_KEYWORDS)
106  if (contains(filename, keyw))
107  return true;
108 
109  return false;
110 }

122 brain::GIDSet AbstractCircuitLoader::_getGids(const PropertyMap &properties, const brion::BlueConfig &blueConfiguration,
123  const brain::Circuit &circuit, GIDOffsets &targetGIDOffsets) const
124 {
125  const auto circuitTargets = _getTargetsAsStrings(properties.getProperty<std::string>(PROP_TARGETS.name));
126  const auto circuitGids = GIDsAsInts(properties.getProperty<std::string>(PROP_GIDS.name));
127  const auto circuitDensity = properties.getProperty<double>(PROP_DENSITY.name);
128  const auto randomSeed = properties.getProperty<double>(PROP_RANDOM_SEED.name);
129  const auto preSynapticNeuron = properties.getProperty<std::string>(PROP_PRESYNAPTIC_NEURON_GID.name);
130  const auto postSynapticNeuron = properties.getProperty<std::string>(PROP_POSTSYNAPTIC_NEURON_GID.name);
131 
132  // Pre/post synaptic sanity check
133  const bool pre = preSynapticNeuron.empty();
134  const bool post = postSynapticNeuron.empty();
135  bool prePostSynapticUsecase = false;
136  if (!pre || !post)
137  {
138  if (pre || post)
139  PLUGIN_THROW("Pre and post synaptic Gids must be specified");
140  prePostSynapticUsecase = true;
141  }
142 
143  brain::GIDSet gids;
144  targetGIDOffsets.push_back(0);
145 
146  // Pair synapse usecase
147  if (!preSynapticNeuron.empty() && !postSynapticNeuron.empty())
148  {
149  gids.insert(lexical_cast<Gid>(preSynapticNeuron));
150  gids.insert(lexical_cast<Gid>(postSynapticNeuron));
151  return gids;
152  }
153 
154  if (!circuitGids.empty())
155  {
156  for (const auto gid : circuitGids)
157  gids.insert(gid);
158  return gids;
159  }
160 
161  // Determine GIDs according to specified targets
162  strings localTargets;
163  if (circuitTargets.empty())
164  {
165  const auto defaultTarget = blueConfiguration.getCircuitTarget();
166  localTargets.push_back(defaultTarget);
167  PLUGIN_INFO("- No target specified. Loading default one: " << defaultTarget);
168  }
169  else
170  localTargets = circuitTargets;
171 
172  for (const auto &target : localTargets)
173  {
174  const auto targetGids = (circuitDensity == 1.f ? (target.empty() ? circuit.getGIDs() : circuit.getGIDs(target))
175  : circuit.getRandomGIDs(circuitDensity, target, randomSeed));
176  PLUGIN_INFO("- Target " << target << ": " << targetGids.size() << " cells");
177  gids.insert(targetGids.begin(), targetGids.end());
178  targetGIDOffsets.push_back(gids.size());
179  }
180 
181  if (gids.empty())
182  PLUGIN_THROW("Selection is empty");
183  return gids;
184 }

186 CompartmentReportPtr AbstractCircuitLoader::_attachSimulationHandler(const PropertyMap &properties,
187  const brion::BlueConfig &blueConfiguration,
188  Model &model, const ReportType &reportType,
189  brain::GIDSet &gids) const
190 {
191  CompartmentReportPtr compartmentReport{nullptr};
192 
193  // Load simulation information from compartment report
194  const auto report = properties.getProperty<std::string>(PROP_REPORT.name);
195  const auto userDataType = stringToEnum<UserDataType>(properties.getProperty<std::string>(PROP_USER_DATA_TYPE.name));
196 
197  if (report.empty() && (userDataType == UserDataType::simulation_offset || userDataType == UserDataType::undefined))
198  return nullptr;
199 
200  const auto synchronousMode = !properties.getProperty<bool>(PROP_SYNCHRONOUS_MODE.name);
201 
202  AbstractSimulationHandlerPtr simulationHandler{nullptr};
203  switch (reportType)
204  {
205  case ReportType::voltages_from_file:
206  {
207  PLUGIN_INFO("Loading simulation data in " << (synchronousMode ? "a" : "") << "synchronous mode");
208  const auto &voltageReport = blueConfiguration.getReportSource(report);
209  PLUGIN_INFO("Voltage report: " << voltageReport);
210  auto handler = std::make_shared<VoltageSimulationHandler>(voltageReport.getPath(), gids, synchronousMode);
211  compartmentReport = handler->getReport();
212 
213  // Only keep simulated GIDs
214  if (!compartmentReport)
215  PLUGIN_THROW("No voltage report was found");
216  gids = compartmentReport->getGIDs();
217 
218  model.setSimulationHandler(handler);
219 
220  simulationHandler = handler;
221  setSimulationTransferFunction(model.getTransferFunction());
222  break;
223  }
224  case ReportType::spikes:
225  {
226  const auto &spikeReport = blueConfiguration.getSpikeSource();
227  PLUGIN_INFO("Spike report: " << spikeReport);
228  auto handler = std::make_shared<SpikeSimulationHandler>(spikeReport.getPath(), gids);
229  model.setSimulationHandler(handler);
230  simulationHandler = handler;
231  setSimulationTransferFunction(model.getTransferFunction());
232  break;
233  }
234  default:
235  if (userDataType == UserDataType::distance_to_soma)
236  {
237  auto handler = std::make_shared<CellGrowthHandler>(100);
238  model.setSimulationHandler(handler);
239  setSimulationTransferFunction(model.getTransferFunction(), 0.f);
240  }
241  else
242  PLUGIN_THROW("Unknown report type. Simulation ignored");
243  }
244  return compartmentReport;
245 }

247 bool AbstractCircuitLoader::_isClipped(const Vector3f &position) const
248 {
249  auto &clipPlanes = _scene.getClipPlanes();
250  if (clipPlanes.empty())
251  return false;
252 
253  bool visible = true;
254  for (auto clipPlane : clipPlanes)
255  {
256  const auto &plane = clipPlane->getPlane();
257  const Vector3f normal = {plane[0], plane[1], plane[2]};
258  const float d = plane[3];
259  const float distance = dot(normal, position) + d;
260  visible &= (distance > 0.f);
261  }
262  return !visible;
263 }

265 void AbstractCircuitLoader::_filterGIDsWithClippingPlanes(brain::GIDSet &gids, Matrix4fs &transformations) const
266 {
267  // Filter our guids according to clipping planes
268  Matrix4fs clippedTransformations;
269  brain::GIDSet clippedGids;
270  uint64_t i = 0;
271  for (const auto gid : gids)
272  {
273  const auto &transformation = transformations[i];
274  if (!_isClipped(get_translation(transformation)))
275  {
276  clippedTransformations.push_back(transformation);
277  clippedGids.insert(gid);
278  }
279  ++i;
280  }
281  gids = clippedGids;
282  transformations = clippedTransformations;
283  PLUGIN_INFO("Clipped circuit: " << gids.size() << " cells");
284 }

286 void AbstractCircuitLoader::_filterGIDsWithAreasOfInterest(const uint16_t areasOfInterest, brain::GIDSet &gids,
287  Matrix4fs &transformations) const
288 {
289  brain::GIDSet rearrangedGids;
290  Matrix4fs rearrangedTransformations;
291  Boxf aabb;
292  for (const auto &transformation : transformations)
293  aabb.merge(get_translation(transformation));
294 
295  const auto bbMin = aabb.getMin();
296  const auto bbMax = aabb.getMax();
297  const float bbSize = aabb.getSize().x / areasOfInterest;
298  for (int i = 0; i < areasOfInterest; ++i)
299  {
300  Boxf area;
301  area.merge({bbMin.x + i * bbSize, bbMin.y, bbMin.z});
302  area.merge({bbMin.x + (i + 1) * bbSize, bbMax.y, bbMax.z});
303 
304  uint64_t j = 0;
305  for (const auto &gid : gids)
306  {
307  if (inBox(get_translation(transformations[j]), area))
308  {
309  rearrangedGids.insert(gid);
310  rearrangedTransformations.push_back(transformations[j]);
311  break;
312  }
313  ++j;
314  }
315  }
316  gids = rearrangedGids;
317  transformations = rearrangedTransformations;
318  PLUGIN_INFO("Areas of interest: " << gids.size() << " cells");
319 }

321 ModelDescriptorPtr AbstractCircuitLoader::importCircuit(const std::string &circuitConfiguration,
322  const PropertyMap &properties,
323  const LoaderProgress &callback) const
324 {
325  const auto colorScheme =
326  stringToEnum<CircuitColorScheme>(properties.getProperty<std::string>(PROP_CIRCUIT_COLOR_SCHEME.name));
327  const auto morphologyQuality =
328  stringToEnum<AssetQuality>(properties.getProperty<std::string>(PROP_ASSET_QUALITY.name));
329  const auto meshFolder = properties.getProperty<std::string>(PROP_MESH_FOLDER.name);
330  const auto meshFilenamePattern = properties.getProperty<std::string>(PROP_MESH_FILENAME_PATTERN.name);
331  const auto reportType = stringToEnum<ReportType>(properties.getProperty<std::string>(PROP_REPORT_TYPE.name));
332  const auto userDataType = stringToEnum<UserDataType>(properties.getProperty<std::string>(PROP_USER_DATA_TYPE.name));
333  const auto cellClipping = properties.getProperty<bool>(PROP_CELL_CLIPPING.name);
334  const auto areasOfInterest = properties.getProperty<int>(PROP_AREAS_OF_INTEREST.name);
335 
336  // Model (one for the whole circuit)
337  auto model = _scene.createModel();
338  if (!model)
339  PLUGIN_THROW("Failed to create model");
340 
341  // Open Circuit and select GIDs according to specified target
342  callback.updateProgress("Open Brion circuit ...", 0);
343  const brion::BlueConfig blueConfiguration(circuitConfiguration);
344  callback.updateProgress("Open Brain circuit ...", 0);
345  const brain::Circuit circuit(blueConfiguration);
346 
347  callback.updateProgress("Getting GIDs...", 0);
348  GIDOffsets targetGIDOffsets;
349  brain::GIDSet allGids = _getGids(properties, blueConfiguration, circuit, targetGIDOffsets);
350 
351  callback.updateProgress("Attaching to simulation data...", 0);
352 
353  PLUGIN_INFO("- Attaching simulation handler");
354  const auto compartmentReport = _attachSimulationHandler(properties, blueConfiguration, *model, reportType, allGids);
355 
356  PLUGIN_INFO("- Applying cell transformations");
357  Matrix4fs allTransformations;
358  for (const auto &transformation : circuit.getTransforms(allGids))
359  allTransformations.push_back(transformation);
360 
361  PLUGIN_INFO("- Filtering out guids according to clipping planes");
362  if (cellClipping)
363  _filterGIDsWithClippingPlanes(allGids, allTransformations);
364 
365  PLUGIN_INFO("- Filtering out guids according to aeras of interest");
366  if (areasOfInterest != 0)
367  _filterGIDsWithAreasOfInterest(areasOfInterest, allGids, allTransformations);
368 
369  PLUGIN_INFO("- Identifying layer ids");
370  callback.updateProgress("Identifying layer ids...", 0);
371  const auto layerIds = _populateLayerIds(properties, blueConfiguration,
372  colorScheme == CircuitColorScheme::by_target ? allGids : brain::GIDSet());
373 
374  callback.updateProgress("Identifying electro-physiology types...", 0);
375  PLUGIN_INFO("- Identifying electro-physiologycal types");
376  const auto &electrophysiologyTypes =
377  circuit.getElectrophysiologyTypes(colorScheme == CircuitColorScheme::by_etype ? allGids : brain::GIDSet());
378  callback.updateProgress("Getting cell types...", 0);
379  PLUGIN_INFO("- Getting cell types");
380  size_ts morphologyTypes;
381  if (colorScheme == CircuitColorScheme::by_mtype)
382  morphologyTypes = circuit.getMorphologyTypes(allGids);
383 
384  callback.updateProgress("Importing cells...", 0);
385  PLUGIN_INFO("- Importing cells");
386  float maxMorphologyLength = 0.f;
387  if (meshFolder.empty())
388  maxMorphologyLength =
389  _importMorphologies(properties, circuit, *model, allGids, allTransformations, targetGIDOffsets,
390  compartmentReport, layerIds, morphologyTypes, electrophysiologyTypes, callback);
391  else
392  {
393  PLUGIN_INFO("- Importing meshes");
394  _importMeshes(properties, *model, allGids, allTransformations, targetGIDOffsets, layerIds, morphologyTypes,
395  electrophysiologyTypes, callback);
396 
397  if (compartmentReport != nullptr)
398  // If meshes are loaded, and simulation is enabled, a secondary
399  // model is created to store the simulation data in the 3D scene
400  maxMorphologyLength = _importMorphologies(properties, circuit, *model, allGids, allTransformations,
401  targetGIDOffsets, compartmentReport, layerIds, morphologyTypes,
402  electrophysiologyTypes, callback, SECONDARY_MODEL_MATERIAL_ID);
403  }
404 
405  if (userDataType == UserDataType::distance_to_soma)
406  {
407  PLUGIN_INFO("- Updating cell growth information");
408  model->getTransferFunction().setValuesRange({0.f, maxMorphologyLength});
409  const auto frameSize = uint64_t(maxMorphologyLength) + 1;
410  model->getSimulationHandler()->setFrameSize(frameSize);
411  model->getSimulationHandler()->setNbFrames(frameSize);
412  }
413 
414  PLUGIN_INFO("- Creating custom materials");
415  const bool castUserData =
416  (compartmentReport || userDataType == UserDataType::distance_to_soma || reportType == ReportType::spikes);
417  MorphologyLoader::createMissingMaterials(*model, castUserData);
418 
419  PLUGIN_INFO("- Applying default colormap");
420  _setDefaultCircuitColorMap(*model);
421 
422  PLUGIN_INFO("- Computing circuit center according to soma positions");
423  callback.updateProgress("Computing circuit center...", 1);
424  Boxf circuitCenter;
425  for (const auto &transformation : allTransformations)
426  circuitCenter.merge(get_translation(transformation));
427 
428  PLUGIN_INFO("- Creating final 3D model");
429  ModelMetadata metadata = {{"Report", properties.getProperty<std::string>(PROP_REPORT.name)},
430  {"Report type", properties.getProperty<std::string>(PROP_REPORT_TYPE.name)},
431  {"Targets", properties.getProperty<std::string>(PROP_TARGETS.name)},
432  {"GIDs", properties.getProperty<std::string>(PROP_GIDS.name)},
433  {"Color scheme", enumToString<CircuitColorScheme>(colorScheme)},
434  {"Use simulation model", enumToString<bool>(!meshFolder.empty())},
435  {"Mesh filename pattern", meshFilenamePattern},
436  {"Mesh folder", meshFolder},
437  {"Morphology quality", enumToString<AssetQuality>(morphologyQuality)},
438  {"Number of neurons", std::to_string(allGids.size())},
439  {"Density", std::to_string(properties.getProperty<double>(PROP_DENSITY.name))},
440  {"RandomSeed", std::to_string(properties.getProperty<double>(PROP_RANDOM_SEED.name))},
441  {"CircuitPath", circuitConfiguration}};
442 
443  ModelDescriptorPtr modelDescriptor;
444  Transformation transformation;
445  transformation.setRotationCenter(circuitCenter.getCenter());
446  modelDescriptor = std::make_shared<ModelDescriptor>(std::move(model), "Circuit", circuitConfiguration, metadata);
447  modelDescriptor->setTransformation(transformation);
448  return modelDescriptor;
449 }

451 size_t AbstractCircuitLoader::_getMaterialFromCircuitAttributes(const PropertyMap &properties, const uint64_t index,
452  const size_t material,
453  const GIDOffsets &targetGIDOffsets,
454  const size_ts &layerIds, const size_ts &morphologyTypes,
455  const size_ts &electrophysiologyTypes,
456  const bool forSimulationModel) const
457 {
458  if (material != NO_MATERIAL)
459  return material;
460 
461  if (forSimulationModel)
462  return 0;
463 
464  const auto colorScheme =
465  stringToEnum<CircuitColorScheme>(properties.getProperty<std::string>(PROP_CIRCUIT_COLOR_SCHEME.name));
466 
467  size_t materialId;
468  switch (colorScheme)
469  {
470  case CircuitColorScheme::by_id:
471  materialId = NB_MATERIALS_PER_INSTANCE * index;
472  break;
473  case CircuitColorScheme::by_target:
474  for (size_t i = 0; i < targetGIDOffsets.size() - 1; ++i)
475  if (index >= targetGIDOffsets[i] && index < targetGIDOffsets[i + 1])
476  {
477  materialId = NB_MATERIALS_PER_INSTANCE * i;
478  break;
479  }
480  break;
481  case CircuitColorScheme::by_etype:
482  if (index < electrophysiologyTypes.size())
483  materialId = NB_MATERIALS_PER_INSTANCE * electrophysiologyTypes[index];
484  else
485  PLUGIN_DEBUG("Failed to get neuron E-type")
486  break;
487  case CircuitColorScheme::by_mtype:
488  if (index < morphologyTypes.size())
489  materialId = NB_MATERIALS_PER_INSTANCE * morphologyTypes[index];
490  else
491  PLUGIN_DEBUG("Failed to get neuron M-type")
492  break;
493  case CircuitColorScheme::by_layer:
494  if (index < layerIds.size())
495  materialId = NB_MATERIALS_PER_INSTANCE * layerIds[index];
496  else
497  PLUGIN_DEBUG("Failed to get neuron layer");
498  break;
499  default:
500  materialId = 0;
501  }
502  return materialId;
503 }

505 size_ts AbstractCircuitLoader::_populateLayerIds(const PropertyMap &properties, const brion::BlueConfig &blueConfig,
506  const brain::GIDSet &gids) const
507 {
508  size_ts layerIds;
509  const auto circuitSource = blueConfig.getCircuitSource();
510  try
511  {
512  PLUGIN_INFO(" - Opening Brion circuit from " << circuitSource);
513  const brion::Circuit circuit(circuitSource);
514  for (const auto &values : circuit.get(gids, brion::NEURON_LAYER))
515  layerIds.push_back(values.size() > 0 ? std::stoi(values[0]) : 0);
516  }
517  catch (...)
518  {
519  PLUGIN_INFO(" - Failed to open Brion circuit from " << circuitSource);
520  const auto colorScheme =
521  stringToEnum<CircuitColorScheme>(properties.getProperty<std::string>(PROP_CIRCUIT_COLOR_SCHEME.name));
522  if (colorScheme == CircuitColorScheme::by_layer)
523  {
524  PLUGIN_ERROR(
525  " - Only MVD2 format is currently supported by Brion "
526  "circuits. Color scheme by layer not available for this "
527  "circuit");
528  }
529  else
530  PLUGIN_WARN(" - No layer Id could be identified");
531  }
532  return layerIds;
533 }

775 float AbstractCircuitLoader::_importMorphologies(const PropertyMap &properties, const brain::Circuit &circuit,
776  Model &model, const brain::GIDSet &gids,
777  const Matrix4fs &transformations, const GIDOffsets &targetGIDOffsets,
778  CompartmentReportPtr compartmentReport, const size_ts &layerIds,
779  const size_ts &morphologyTypes, const size_ts &electrophysiologyTypes,
780  const LoaderProgress &callback, const size_t materialId) const
781 {
782  const auto preSynapticNeuron = properties.getProperty<std::string>(PROP_PRESYNAPTIC_NEURON_GID.name);
783  const auto postSynapticNeuron = properties.getProperty<std::string>(PROP_POSTSYNAPTIC_NEURON_GID.name);
784  const bool prePostSynapticUsecase = (!preSynapticNeuron.empty() && !postSynapticNeuron.empty());
785  const bool loadAfferentSynapses = properties.getProperty<bool>(PROP_LOAD_AFFERENT_SYNAPSES.name);
786  const bool loadEfferentSynapses = properties.getProperty<bool>(PROP_LOAD_EFFERENT_SYNAPSES.name);
787  const auto position = properties.getProperty<std::array<double, 3>>(PROP_POSITION.name);
788  const auto rotation = properties.getProperty<std::array<double, 4>>(PROP_ROTATION.name);
789  Transformation transformation;
790  transformation.setTranslation(Vector3d(position[0], position[1], position[2]));
791  transformation.setRotation(Quaterniond(rotation[0], rotation[1], rotation[2], rotation[3]));
792 
793  Timer chrono;
794  const auto sectionTypes = MorphologyLoader::getSectionTypesFromProperties(_defaults);
795  const bool somasOnly = (sectionTypes.size() == 1 && sectionTypes[0] == brain::neuron::SectionType::soma);
796 
797  brain::URIs uris;
798  if (!somasOnly)
799  {
800  PLUGIN_INFO("- Getting cell URIs");
801  uris = circuit.getMorphologyURIs(gids);
802  }
803 
804  PropertyMap morphologyProps(properties);
805 
806  std::vector<Gid> localGids;
807  for (const auto gid : gids)
808  localGids.push_back(gid);
809 
810  std::vector<ParallelModelContainer> containers;
811  uint64_t morphologyId;
812 #pragma omp parallel for private(morphologyId)
813  for (morphologyId = 0; morphologyId < localGids.size(); ++morphologyId)
814  {
815  const auto uri = somasOnly ? std::string() : std::string(uris[morphologyId].getPath());
816  try
817  {
818  const auto baseMaterialId =
819  _getMaterialFromCircuitAttributes(properties, morphologyId, materialId, targetGIDOffsets, layerIds,
820  morphologyTypes, electrophysiologyTypes, false);
821  MorphologyLoader loader(_scene, std::move(morphologyProps), transformation);
822  loader.setBaseMaterialId(baseMaterialId);
823 
824  SynapsesInfo synapsesInfo;
825  synapsesInfo.prePostSynapticUsecase = prePostSynapticUsecase;
826  const auto gid = localGids[morphologyId];
827  if (prePostSynapticUsecase)
828  {
829  synapsesInfo.preGid = lexical_cast<Gid>(preSynapticNeuron);
830  synapsesInfo.postGid = lexical_cast<Gid>(postSynapticNeuron);
831  }
832  if (loadAfferentSynapses)
833  synapsesInfo.afferentSynapses =
834  std::unique_ptr<brain::Synapses>(new brain::Synapses(circuit.getAfferentSynapses({gid})));
835  if (loadEfferentSynapses)
836  synapsesInfo.efferentSynapses =
837  std::unique_ptr<brain::Synapses>(new brain::Synapses(circuit.getEfferentSynapses({gid})));
838 
839  const auto layerId = layerIds.empty() ? 0 : layerIds[morphologyId];
840  const auto mitochondriaDensity =
841  (layerId < MITOCHONDRIA_DENSITY.size() ? MITOCHONDRIA_DENSITY[layerId] : 0.f);
842  ParallelModelContainer modelContainer =
843  loader.importMorphology(gid, morphologyProps, uri, morphologyId, synapsesInfo,
844  transformations[morphologyId], compartmentReport, mitochondriaDensity);
845 #pragma omp critical
846  containers.push_back(modelContainer);
847  }
848  catch (const std::runtime_error &e)
849  {
850  PLUGIN_ERROR("Failed to load cell " << morphologyId << " (" << uri << "): " << e.what());
851  }
852  if (omp_get_thread_num() == 0)
853  {
854  const float progress = (1 + morphologyId) * omp_get_num_threads();
855  PLUGIN_PROGRESS("- Loading cells", progress, localGids.size());
856 
857  callback.updateProgress("Loading cells...", 0.5f * (progress / localGids.size()));
858  }
859  }
860  PLUGIN_INFO("");
861 
862  float maxDistanceToSoma = 0.f;
863  for (size_t i = 0; i < containers.size(); ++i)
864  {
865  PLUGIN_PROGRESS("- Compiling 3D geometry...", i + 1, containers.size());
866  callback.updateProgress("Compiling 3D geometry...", 0.5f + 0.5f * (float)(1 + i) / (float)containers.size());
867  auto &container = containers[i];
868  maxDistanceToSoma = std::max(container.getMorphologyInfo().maxDistanceToSoma, maxDistanceToSoma);
869  container.moveGeometryToModel(model);
870  }
871  PLUGIN_INFO("");
872 
873  PLUGIN_TIMER(chrono.elapsed(), "- " << gids.size() << " cells loaded");
874  return maxDistanceToSoma;
875 }
