38 ParametricMembrane::ParametricMembrane(Scene &scene,
39  const Vector3f &assemblyPosition,
40  const Quaterniond &assemblyRotation,
41  const Vector4fs &clippingPlanes,
42  const ParametricMembraneDetails &details)
43  : Membrane(scene, assemblyPosition, assemblyRotation, clippingPlanes)
44  , _details(details)
45 {
46  if (_details.representation == ProteinRepresentation::contour)
47  {
48  switch (_details.shape)
49  {
50  case AssemblyShape::spherical:
51  const std::string name{"ParametricMembrane"};
52  const size_t materialId = 0;
53  auto model = _scene.createModel();
54 
55  auto material = model->createMaterial(materialId, name);
56  brayns::PropertyMap props;
57  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
58  static_cast<int>(MaterialShadingMode::basic)});
59  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
60  props.setProperty(
61  {MATERIAL_PROPERTY_CHAMELEON_MODE,
62  static_cast<int>(
63  MaterialChameleonMode::undefined_chameleon_mode)});
64  material->updateProperties(props);
65 
66  model->addSphere(materialId,
67  {_assemblyPosition, _details.assemblyParams[0]});
68  auto modelDescriptor =
69  std::make_shared<ModelDescriptor>(std::move(model), name);
70  _scene.addModel(modelDescriptor);
71  return;
72  }
73  }
74 
75  std::vector<std::string> proteinContents;
76  proteinContents.push_back(details.content1);
77  if (!details.content2.empty())
78  proteinContents.push_back(details.content2);
79  if (!details.content3.empty())
80  proteinContents.push_back(details.content3);
81  if (!details.content4.empty())
82  proteinContents.push_back(details.content4);
83 
84  // Load proteins
85  size_t i = 0;
86  for (const auto &content : proteinContents)
87  {
88  ProteinDetails pd;
89  pd.assemblyName = details.assemblyName;
90  pd.name = _getElementNameFromId(i);
91  pd.contents = content;
92  pd.chainIds = details.chainIds;
93  pd.recenter = details.recenter;
94  pd.loadBonds = details.loadBonds;
95  pd.randomSeed = details.randomSeed;
96  pd.occurrences = 1;
97  pd.rotation = details.rotation;
98  pd.assemblyParams = details.assemblyParams;
99  pd.atomRadiusMultiplier = details.atomRadiusMultiplier;
100  pd.representation = details.representation;
101  pd.loadNonPolymerChemicals = details.loadNonPolymerChemicals;
102  pd.positionRandomizationType = details.positionRandomizationType;
103  pd.position = {0.f, 0.f, 0.f};
104 
105  ProteinPtr protein(new Protein(_scene, pd));
106  auto modelDescriptor = protein->getModelDescriptor();
107  _proteins[pd.name] = std::move(protein);
108  ++i;
109  }
110 
111  // Assemble proteins
112  _processInstances();
113 
114  // Add proteins to the scene
115  for (size_t i = 0; i < proteinContents.size(); ++i)
116  _scene.addModel(
117  _proteins[_getElementNameFromId(i)]->getModelDescriptor());
118 }

126 void ParametricMembrane::_processInstances()
127 {
128  // Randomization
129  srand(_details.randomSeed);
130 
131  const auto rotation = floatsToQuaterniond(_details.rotation);
132  std::map<size_t, size_t> instanceCounts;
133  for (size_t i = 0; i < _proteins.size(); ++i)
134  instanceCounts[i] = 0;
135 
136  // Shape parameters
137  const auto &params = _details.assemblyParams;
138  const float size = (params.size() > 0 ? params[0] : 0.f);
139  const float extraParameter = (params.size() > 5 ? params[5] : 0.f);
140  auto randInfo =
141  floatsToRandomizationDetails(params, _details.randomSeed,
142  _details.positionRandomizationType);
143 
144  // Shape instances
145  const float offset = 2.f / _details.occurrences;
146  const float increment = M_PI * (3.f - sqrt(5.f));
147 
148  for (uint64_t occurence = 0; occurence < _details.occurrences; ++occurence)
149  {
150  const size_t id = rand() % _proteins.size();
151  auto protein = _proteins[_getElementNameFromId(id)];
152  auto md = protein->getModelDescriptor();
153 
154  const auto &model = md->getModel();
155  const auto &bounds = model.getBounds();
156  const Vector3f &center = bounds.getCenter();
157 
158  randInfo.positionSeed =
159  (params.size() >= 2 ? (params[1] == 0 ? 0 : params[1] + occurence)
160  : 0);
161  randInfo.rotationSeed =
162  (params.size() >= 4 ? (params[3] == 0 ? 0 : params[3] + occurence)
163  : 0);
164 
165  Transformation transformation;
166  switch (_details.shape)
167  {
168  case AssemblyShape::spherical:
169  {
170  transformation =
171  getSphericalPosition(Vector3f(), size, occurence,
172  _details.occurrences, randInfo);
173  break;
174  }
175  case AssemblyShape::sinusoidal:
176  {
177  transformation =
178  getSinosoidalPosition(Vector3f(), size, extraParameter,
179  occurence, randInfo);
180  break;
181  }
182  case AssemblyShape::cubic:
183  {
184  transformation = getCubicPosition(Vector3f(), size, randInfo);
185  break;
186  }
187  case AssemblyShape::fan:
188  {
189  transformation = getFanPosition(Vector3f(), size, occurence,
190  _details.occurrences, randInfo);
191  break;
192  }
193  case AssemblyShape::bezier:
194  {
195  if ((params.size() - 5) % 3 != 0)
196  PLUGIN_THROW(
197  "Invalid number of floats in assembly extra parameters");
198  Vector3fs points;
199  for (uint32_t i = 5; i < params.size(); i += 3)
200  points.push_back(
201  Vector3f(params[i], params[i + 1], params[i + 2]));
202  transformation = getBezierPosition(points, size,
203  float(occurence) /
204  float(_details.occurrences));
205  break;
206  }
207  case AssemblyShape::spherical_to_planar:
208  {
209  transformation =
210  getSphericalToPlanarPosition(Vector3f(), size, occurence,
211  _details.occurrences, randInfo,
212  extraParameter);
213  break;
214  }
215  default:
216  transformation = getPlanarPosition(Vector3f(), size, randInfo);
217  break;
218  }
219 
220  Vector3f translation =
221  Vector3f(_assemblyRotation *
222  (transformation.getTranslation() - Vector3d(center)));
223 
224  // Clipping planes
225  if (isClipped(translation, _clippingPlanes))
226  continue;
227 
228  // Final transformation
229  translation += _assemblyPosition;
230 
231  Transformation finalTransformation;
232  finalTransformation.setTranslation(translation);
233  finalTransformation.setRotation(
234  _assemblyRotation * transformation.getRotation() * rotation);
235 
236  if (instanceCounts[id] == 0)
237  md->setTransformation(finalTransformation);
238  const ModelInstance instance(true, false, finalTransformation);
239  md->addInstance(instance);
240 
241  instanceCounts[id] = instanceCounts[id] + 1;
242  }
243 }
