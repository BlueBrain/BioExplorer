45 void VectorFieldBuilder::buildOctree(core::Engine& engine, core::Model& model, const double voxelSize,
46  const double density, const uint32_ts& modelIds)
47 {
48  PLUGIN_INFO(3, "Building Vector Octree");
49 
50  auto& scene = engine.getScene();
51  const auto& clipPlanes = getClippingPlanes(scene);
52 
53  OctreeVectors vectors;
54  uint32_t count{0};
55  const uint32_t densityRatio = 1.f / density;
56 
57  Boxd bounds;
58  const auto& modelDescriptors = scene.getModelDescriptors();
59  for (const auto modelDescriptor : modelDescriptors)
60  {
61  if (!modelIds.empty())
62  {
63  const auto modelId = modelDescriptor->getModelID();
64  const auto it = std::find(modelIds.begin(), modelIds.end(), modelId);
65  if (it == modelIds.end())
66  continue;
67  }
68 
69  const auto& instances = modelDescriptor->getInstances();
70  for (const auto& instance : instances)
71  {
72  const auto& tf = instance.getTransformation();
73  const auto& model = modelDescriptor->getModel();
74  const auto& conesMap = model.getCones();
75  for (const auto& cones : conesMap)
76  {
77  if (cones.first != BOUNDINGBOX_MATERIAL_ID && cones.first != SECONDARY_MODEL_MATERIAL_ID)
78  for (const auto& cone : cones.second)
79  {
80  const Vector3f center =
81  tf.getTranslation() + tf.getRotation() * (Vector3d(cone.center) - tf.getRotationCenter());
82  const Vector3f up =
83  tf.getTranslation() + tf.getRotation() * (Vector3d(cone.up) - tf.getRotationCenter());
84 
85  const Vector3d c = center;
86  const Vector3d u = center;
87  if (isClipped(c, clipPlanes) || isClipped(u, clipPlanes))
88  {
89  ++count;
90  continue;
91  }
92 
93  if (count % densityRatio == 0)
94  {
95  bounds.merge(center + cone.centerRadius);
96  bounds.merge(center - cone.centerRadius);
97  bounds.merge(up + cone.upRadius);
98  bounds.merge(up - cone.upRadius);
99 
100  OctreeVector vector;
101  vector.position = center;
102  vector.direction = up - center;
103  vectors.push_back(vector);
104  }
105  ++count;
106  }
107  }
108  }
109  }
110 
111  // Events bounds
112  Vector3f sceneSize = bounds.getSize();
113  Vector3f center = bounds.getCenter();
114  Vector3f extendedHalfSize = sceneSize * 0.5f;
115 
116  // Expand volume by 25%
117  const float boundsExpansion = 1.25f;
118  bounds.merge(center + extendedHalfSize * boundsExpansion);
119  bounds.merge(center - extendedHalfSize * boundsExpansion);
120  sceneSize = bounds.getSize();
121  center = bounds.getCenter();
122  extendedHalfSize = sceneSize * 0.5f;
123 
124  // Compute volume information
125  const Vector3f minAABB = center - extendedHalfSize;
126  const Vector3f maxAABB = center + extendedHalfSize;
127 
128  // Build acceleration structure
129  const VectorOctree accelerator(vectors, voxelSize, minAABB, maxAABB);
130  const auto& indices = accelerator.getFlatIndices();
131  const auto& data = accelerator.getFlatData();
132 
133  const uint32_t volumeSize = accelerator.getVolumeSize();
134  const auto offset = center - extendedHalfSize;
135  const auto dimensions = accelerator.getVolumeDimensions();
136  const auto spacing = sceneSize / Vector3f(dimensions);
137 
138  const auto& params = engine.getParametersManager().getApplicationParameters();
139  const auto& engineName = params.getEngine();
140  auto field = model.createField(dimensions, spacing, offset, indices, data, OctreeDataType::odt_vectors);
141  const size_t materialId = FIELD_MATERIAL_ID;
142  model.addField(materialId, field);
143  model.createMaterial(materialId, std::to_string(materialId));
144 
145  PLUGIN_INFO(1, "--------------------------------------------");
146  PLUGIN_INFO(1, "Vector Octree information (" << vectors.size() << " vectors)");
147  PLUGIN_INFO(1, "--------------------------------------------");
148  PLUGIN_INFO(1, "Scene AABB : " << bounds);
149  PLUGIN_INFO(1, "Scene dimension : " << sceneSize);
150  PLUGIN_INFO(1, "Element spacing : " << spacing);
151  PLUGIN_INFO(1, "Volume dimensions : " << dimensions);
152  PLUGIN_INFO(1, "Element offset : " << offset);
153  PLUGIN_INFO(1, "Volume size : " << volumeSize << " bytes");
154  PLUGIN_INFO(1, "Indices size : " << indices.size());
155  PLUGIN_INFO(1, "Octree size : " << accelerator.getOctreeSize());
156  PLUGIN_INFO(1, "Octree depth : " << accelerator.getOctreeDepth());
157  PLUGIN_INFO(1, "--------------------------------------------");
158 }
