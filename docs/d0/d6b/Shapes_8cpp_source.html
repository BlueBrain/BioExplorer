72 {
73  return randoms[index % randoms.size()] - 0.5f;

76 float rnd3(const size_t index)
77 {
78  return cos(index * M_PI / 180.f) + sin(index * M_PI / 45.f) +
79  cos(index * M_PI / 72.f);

82 Quaterniond weightedRandomRotation(const size_t seed, const size_t index,
83  const Quaterniond& q, const float s)
84 {
85  const Quaterniond qPitch =
86  glm::angleAxis(s * rnd2(seed + index * 2), Vector3f(1.f, 0.f, 0.f));
87  const Quaterniond qYaw =
88  glm::angleAxis(s * rnd2(seed + index * 3), Vector3f(0.f, 1.f, 0.f));
89  const Quaterniond qRoll =
90  glm::angleAxis(s * rnd2(seed + index * 5), Vector3f(0.f, 0.f, 1.f));
91  return q * qPitch * qYaw * qRoll;
92 }

94 Quaterniond randomQuaternion(const size_t seed)
95 {
96  double x, y, z, u, v, w, s;
97  do
98  {
99  x = rnd2(seed);
100  y = rnd2(seed + 1);
101  z = x * x + y * y;
102  } while (z > 1.0);
103  do
104  {
105  u = rnd2(seed + 2);
106  v = rnd2(seed + 3);
107  w = u * u + v * v;
108  } while (w > 1.0);
109  s = sqrt((1.0 - z) / w);
110  return Quaterniond(x, y, s * u, s * v);
111 }

113 Transformation getSphericalPosition(const Vector3f& position,
114  const float radius, const size_t occurence,
115  const size_t occurences,
116  const RandomizationDetails& randInfo)
117 {
118  size_t rnd = occurence;
119  if (occurences != 0 && randInfo.seed != 0 &&
120  randInfo.randomizationType == PositionRandomizationType::circular)
121  rnd = rand() % occurences;
122 
123  const double offset = 2.0 / occurences;
124  const double increment = M_PI * (3.0 - sqrt(5.0));
125 
126  // Position randomizer
127  double R = radius;
128  if (randInfo.positionSeed != 0 &&
129  randInfo.randomizationType == PositionRandomizationType::radial)
130  R += randInfo.positionStrength * rnd3(randInfo.positionSeed + rnd);
131 
132  // Sphere filling
133  const double y = ((occurence * offset) - 1.0) + offset / 2.0;
134  const double r = sqrt(1.0 - pow(y, 2.0));
135  const double phi = rnd * increment;
136  const double x = cos(phi) * r;
137  const double z = sin(phi) * r;
138 
139  Vector3d d{x, y, z};
140  Vector3d pos;
141  if (randInfo.randomizationType == PositionRandomizationType::radial)
142  pos = (R + position.y) * d;
143  else
144  pos = Vector3d(position) + R * d;
145 
146  // Rotation
147  Quaterniond rotation = quatLookAt(d, Vector3d(UP_VECTOR));
148  if (randInfo.rotationSeed != 0)
149  rotation = weightedRandomRotation(randInfo.rotationSeed, rnd, rotation,
150  randInfo.rotationStrength);
151 
152  Transformation transformation;
153  transformation.setTranslation(pos);
154  transformation.setRotation(rotation);
155  return transformation;
156 }

158 Transformation getFanPosition(const Vector3f& position, const float radius,
159  const size_t occurence, const size_t occurences,
160  const RandomizationDetails& randInfo)
161 {
162  size_t rnd = occurence;
163  if (occurences != 0 && randInfo.seed != 0 &&
164  randInfo.randomizationType == PositionRandomizationType::circular)
165  rnd = rand() % occurences;
166 
167  const float offset = 2.f / occurences;
168  const float increment = 0.1f * M_PI * (3.f - sqrt(5.f));
169 
170  // Randomizer
171  float R = radius;
172  if (randInfo.seed != 0 &&
173  randInfo.randomizationType == PositionRandomizationType::radial)
174  R *= 1.f + rnd1() / 30.f;
175 
176  // Sphere filling
177  const float y = ((occurence * offset) - 1.f) + offset / 2.f;
178  const float r = sqrt(1.f - pow(y, 2.f));
179  const float phi = rnd * increment;
180  const float x = cos(phi) * r;
181  const float z = sin(phi) * r;
182  const Vector3f d{x, y, z};
183 
184  Transformation transformation;
185  transformation.setTranslation(position + R * d);
186  transformation.setRotation(quatLookAt(d, UP_VECTOR));
187  return transformation;

231 float sinusoide(const float x, const float z)
232 {
233  return 0.2f * cos(x) * sin(z) + 0.05f * cos(x * 2.3f) * sin(z * 4.6f);

236 Transformation getSinosoidalPosition(const Vector3f& position, const float size,
237  const float amplitude,
238  const size_t occurence,
239  const RandomizationDetails& randInfo)
240 {
241  const float step = 0.01f;
242  const float angle = 0.01f;
243  float upOffset = 0.f;
244  if (randInfo.positionSeed != 0 &&
245  randInfo.randomizationType == PositionRandomizationType::radial)
246  upOffset = randInfo.positionStrength *
247  rnd3((randInfo.positionSeed + occurence) * 10);
248 
249  const float x = rnd1() * size;
250  const float z = rnd1() * size;
251  const float y = upOffset + amplitude * sinusoide(x * angle, z * angle);
252 
253  Vector3f pos = Vector3f(x, y, z);
254 
255  const Vector3f v1 =
256  Vector3f(x + step,
257  upOffset +
258  amplitude * sinusoide((x + step) * angle, z * angle),
259  z) -
260  pos;
261  const Vector3f v2 =
262  Vector3f(x,
263  upOffset +
264  amplitude * sinusoide(x * angle, (z + step) * angle),
265  z + step) -
266  pos;
267 
268  pos += position;
269 
270  // Rotation
271  Vector3f d = cross(normalize(v1), normalize(v2));
272  Quaterniond rotation = quatLookAt(normalize(d), UP_VECTOR);
273  if (randInfo.rotationSeed != 0)
274  rotation = weightedRandomRotation(randInfo.rotationSeed, occurence,
275  rotation, randInfo.rotationStrength);
276 
277  Transformation transformation;
278  transformation.setTranslation(pos);
279  transformation.setRotation(rotation);
280  return transformation;
281 }

283 Transformation getBezierPosition(const Vector3fs& points, const float scale,
284  const float t)
285 {
286  Vector3fs bezierPoints = points;
287  for (auto& bezierPoint : bezierPoints)
288  bezierPoint *= scale;
289 
290  size_t i = bezierPoints.size() - 1;
291  while (i > 0)
292  {
293  for (size_t k = 0; k < i; ++k)
294  bezierPoints[k] =
295  bezierPoints[k] + t * (bezierPoints[k + 1] - bezierPoints[k]);
296  --i;
297  }
298 
299  Transformation transformation;
300  transformation.setTranslation(bezierPoints[0]);
301  transformation.setRotation(quatLookAt(
302  normalize(cross({0.f, 0.f, 1.f}, bezierPoints[1] - bezierPoints[0])),
303  UP_VECTOR));
304  return transformation;
305 }

307 Transformation getSphericalToPlanarPosition(
308  const Vector3f& center, const float radius, const size_t occurence,
309  const size_t occurences, const RandomizationDetails& randInfo,
310  const float morphingStep)
311 {
312  size_t rnd = occurence;
313  if (occurences != 0 && randInfo.seed != 0 &&
314  randInfo.randomizationType == PositionRandomizationType::circular)
315  rnd = rand() % occurences;
316 
317  const double offset = 2.0 / occurences;
318  const double increment = M_PI * (3.0 - sqrt(5.0));
319 
320  // Position randomizer
321  double R = radius;
322  if (randInfo.positionSeed != 0 &&
323  randInfo.randomizationType == PositionRandomizationType::radial)
324  R += randInfo.positionStrength * rnd3(randInfo.positionSeed + rnd);
325 
326  // Sphere filling
327  const double y = ((rnd * offset) - 1.0) + (offset / 2.0);
328  const double r = sqrt(1.f - pow(y, 2.0));
329  const double phi = rnd * increment;
330  const double x = cos(phi) * r;
331  const double z = sin(phi) * r;
332 
333  Vector3d startPos;
334  Vector3d endPos = startPos;
335 
336  Vector3d startDir{x, y, z};
337  if (randInfo.randomizationType == PositionRandomizationType::radial)
338  startPos = (R + center.y) * startDir;
339  else
340  startPos = Vector3d(center) + R * startDir;
341 
342  Quaterniond startRotation = quatLookAt(startDir, Vector3d(UP_VECTOR));
343  if (randInfo.rotationSeed != 0)
344  startRotation =
345  weightedRandomRotation(randInfo.rotationSeed, rnd, startRotation,
346  randInfo.rotationStrength);
347 
348  R = radius;
349  const double endRadius = R * 2.0;
350 
351  endPos.y = -R;
352  endPos = endPos + (1.0 - (startPos.y + R) / endRadius) *
353  Vector3d(endRadius, 0.0, endRadius) *
354  normalize(Vector3d(startDir.x, 0.0, startDir.z));
355 
356  const Quaterniond endRotation =
357  quatLookAt({0.0, 1.0, 0.0}, Vector3d(UP_VECTOR));
358  const Quaterniond finalRotation =
359  glm::lerp(startRotation, endRotation, double(morphingStep));
360 
361  // Final transformation
362  Transformation transformation;
363  const Vector3d finalTranslation =
364  endPos * double(morphingStep) + startPos * (1.0 - morphingStep);
365  transformation.setTranslation(finalTranslation);
366  transformation.setRotation(finalRotation);
367  return transformation;
368 }
