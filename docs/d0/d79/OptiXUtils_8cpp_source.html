35 void toOptiXProperties(const PropertyMap& object)
36 {
37  try
38  {
39  auto context = OptiXContext::get().getOptixContext();
40  for (const auto& prop : object.getProperties())
41  {
42  switch (prop->type)
43  {
44  case Property::Type::Double:
45  context[prop->name]->setFloat(static_cast<float>(prop->get<double>()));
46  break;
47  case Property::Type::Int:
48  context[prop->name]->setInt(prop->get<int32_t>());
49  break;
50  case Property::Type::Bool:
51  // Special case, no bool in OptiX
52  context[prop->name]->setUint(prop->get<bool>());
53  break;
54  case Property::Type::String:
55  CORE_WARN("Cannot upload string property to OptiX '" << prop->name << "'");
56  break;
57  case Property::Type::Vec2d:
58  {
59  auto v = prop->get<std::array<double, 2>>();
60  context[prop->name]->setFloat(static_cast<float>(v[0]), static_cast<float>(v[1]));
61  break;
62  }
63  case Property::Type::Vec2i:
64  {
65  auto v = prop->get<std::array<int32_t, 2>>();
66  context[prop->name]->setInt(v[0], v[1]);
67  break;
68  }
69  case Property::Type::Vec3d:
70  {
71  auto v = prop->get<std::array<double, 3>>();
72  context[prop->name]->setFloat(static_cast<float>(v[0]), static_cast<float>(v[1]),
73  static_cast<float>(v[2]));
74  break;
75  }
76  case Property::Type::Vec3i:
77  {
78  auto v = prop->get<std::array<int32_t, 3>>();
79  context[prop->name]->setInt(v[0], v[1], v[2]);
80  break;
81  }
82  case Property::Type::Vec4d:
83  {
84  auto v = prop->get<std::array<double, 4>>();
85  context[prop->name]->setFloat(static_cast<float>(v[0]), static_cast<float>(v[1]),
86  static_cast<float>(v[2]), static_cast<float>(v[3]));
87  break;
88  }
89  }
90  }
91  }
92  catch (const std::exception& e)
93  {
94  CORE_ERROR("Failed to apply properties for OptiX object" << e.what());
95  }
96 }
97 } // namespace optix
98 } // namespace engine
