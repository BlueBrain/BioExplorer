26 void Timeout::set(const std::function<void()>& func, const int64_t wait)
27 {
28  if (_timeout.valid())
29  throw std::logic_error("Timeout cannot be set() while it is still active");
30 
31  _cleared = false;
32  _timeout =
33  std::async(std::launch::async,
34  [&mutex = _mutex, &condition = _condition, &cleared = _cleared, wait, func]
35  {
36  std::unique_lock<std::mutex> lock(mutex);
37  while (!cleared) // deals with spurious wakeups
38  {
39  if (condition.wait_for(lock, std::chrono::milliseconds(wait)) == std::cv_status::timeout)
40  {
41  func();
42  break;
43  }
44  }
45  });
46 };
