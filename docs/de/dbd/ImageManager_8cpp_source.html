81 Texture2DPtr ImageManager::importTextureFromFile(const std::string &filename, const TextureType type)
82 {
83  auto in = ImageInput::open(filename);
84  if (!in)
85  {
86  std::cerr << "Failed to load image: " << filename << std::endl;
87  return {};
88  }
89 
90  const ImageSpec &spec = in->spec();
91  unsigned int width = spec.width;
92  unsigned int height = spec.height;
93  int channels = spec.nchannels;
94  int depth = spec.format.basesize();
95 
96  std::vector<unsigned char> pixels(width * height * channels * depth);
97  if (!in->read_image(TypeDesc::UINT8, &pixels[0]))
98  {
99  std::cerr << "Failed to read image data: " << filename << std::endl;
100  return {};
101  }
102  in->close();
103 
104  Texture2D::Type textureType = Texture2D::Type::default_;
105  const bool isCubeMap = type == TextureType::irradiance || type == TextureType::radiance;
106  if (isCubeMap)
107  {
108  textureType = Texture2D::Type::cubemap;
109  width /= 6;
110  }
111  else if (type == TextureType::normals)
112  {
113  textureType = Texture2D::Type::normal_roughness;
114  }
115  else if (type == TextureType::specular)
116  {
117  textureType = Texture2D::Type::aoe;
118  }
119 
120  auto texture = std::make_shared<Texture2D>(textureType, filename, channels, depth, width, height);
121  if (isCubeMap || type == TextureType::brdf_lut)
122  {
123  texture->setWrapMode(TextureWrapMode::clamp_to_edge);
124  }
125 
126  texture->setRawData(pixels.data(), width * height, channels);
127 
128  const auto path = std::filesystem::path(filename).parent_path().string();
129  const auto basename = path + "/" + std::filesystem::path(filename).stem().string();
130  const auto ext = std::filesystem::path(filename).extension().string();
131 
132  uint8_t mipLevels = 1;
133  while (std::filesystem::exists(basename + std::to_string((int)mipLevels) + ext))
134  {
135  ++mipLevels;
136  }
137 
138  texture->setMipLevels(mipLevels);
139 
140  for (uint8_t mip = 1; mip < mipLevels; ++mip)
141  {
142  auto mipFilename = basename + std::to_string((int)mip) + ext;
143  auto mipIn = ImageInput::open(mipFilename);
144  if (!mipIn)
145  {
146  std::cerr << "Failed to load mip level image: " << mipFilename << std::endl;
147  continue;
148  }
149 
150  const ImageSpec &mipSpec = mipIn->spec();
151  std::vector<unsigned char> mipPixels(mipSpec.width * mipSpec.height * mipSpec.nchannels *
152  mipSpec.format.basesize());
153  if (!mipIn->read_image(TypeDesc::UINT8, &mipPixels[0]))
154  {
155  std::cerr << "Failed to read mip level image data: " << mipFilename << std::endl;
156  mipIn->close();
157  continue;
158  }
159  mipIn->close();
160 
161  texture->setRawData(mipPixels.data(), mipSpec.width * mipSpec.height, mipSpec.nchannels, mip);
162  }
163 
164  return texture;
165 }
