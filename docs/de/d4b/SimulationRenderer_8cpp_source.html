38  int SimulationRenderer_getBytesPerPrimitive(const void* geometry)
39  {
40  const ospray::Geometry* base =
41  static_cast<const ospray::Geometry*>(geometry);
42  if (dynamic_cast<const ospray::Spheres*>(base))
43  return sizeof(brayns::Sphere);
44  else if (dynamic_cast<const ospray::Cylinders*>(base))
45  return sizeof(brayns::Cylinder);
46  else if (dynamic_cast<const ospray::Cones*>(base))
47  return sizeof(brayns::Cone);
48  else if (dynamic_cast<const ospray::SDFGeometries*>(base))
49  return sizeof(brayns::SDFGeometry);
50  return 0;
51  }
52 }
