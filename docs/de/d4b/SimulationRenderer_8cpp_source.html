42  int SimulationRenderer_getBytesPerPrimitive(const void* geometry)
43  {
44  const ::ospray::Geometry* base = static_cast<const ::ospray::Geometry*>(geometry);
45  if (dynamic_cast<const ::ospray::Spheres*>(base))
46  return sizeof(core::Sphere);
47  else if (dynamic_cast<const ::ospray::Cylinders*>(base))
48  return sizeof(core::Cylinder);
49  else if (dynamic_cast<const core::engine::ospray::Cones*>(base))
50  return sizeof(core::Cone);
51  else if (dynamic_cast<const core::engine::ospray::SDFGeometries*>(base))
52  return sizeof(core::SDFGeometry);
53  return 0;
54  }
55 }
