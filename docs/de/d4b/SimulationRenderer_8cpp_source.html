41  int SimulationRenderer_getBytesPerPrimitive(const void* geometry)
42  {
43  const ospray::Geometry* base = static_cast<const ospray::Geometry*>(geometry);
44  if (dynamic_cast<const ospray::Spheres*>(base))
45  return sizeof(core::Sphere);
46  else if (dynamic_cast<const ospray::Cylinders*>(base))
47  return sizeof(core::Cylinder);
48  else if (dynamic_cast<const ospray::Cones*>(base))
49  return sizeof(core::Cone);
50  else if (dynamic_cast<const ospray::SDFGeometries*>(base))
51  return sizeof(core::SDFGeometry);
52  return 0;
53  }
54 }
