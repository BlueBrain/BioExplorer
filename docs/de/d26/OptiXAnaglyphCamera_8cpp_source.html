45 void OptiXAnaglyphCamera::commit(const OptiXCamera& camera, ::optix::Context context)
46 {
47  auto position = camera.getPosition();
48  const auto stereo = camera.getPropertyOrValue<bool>(CAMERA_PROPERTY_STEREO.name.c_str(), DEFAULT_CAMERA_STEREO);
49  const auto interpupillaryDistance =
50  camera.getPropertyOrValue<double>(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str(),
51  DEFAULT_CAMERA_INTERPUPILLARY_DISTANCE);
52  auto aspect =
53  camera.getPropertyOrValue<double>(CAMERA_PROPERTY_ASPECT_RATIO.name.c_str(), DEFAULT_CAMERA_ASPECT_RATIO);
54 
55  if (stereo)
56  aspect *= 2.f;
57 
58  const auto up = glm::rotate(camera.getOrientation(), UP_VECTOR);
59 
60  Vector3d u, v, w;
61  float ulen, vlen, wlen;
62  w = camera.getTarget() - position;
63 
64  wlen = glm::length(w);
65  u = normalize(glm::cross(w, up));
66  v = normalize(glm::cross(u, w));
67 
68  vlen = wlen * tanf(0.5f *
69  camera.getPropertyOrValue<double>(CAMERA_PROPERTY_FIELD_OF_VIEW.name.c_str(),
70  DEFAULT_CAMERA_FIELD_OF_VIEW) *
71  M_PI / 180.f);
72  v *= vlen;
73  ulen = vlen * aspect;
74  u *= ulen;
75  const Vector3f ipd_offset = 0.5f * interpupillaryDistance * u;
76 
77  context[CONTEXT_CAMERA_U]->setFloat(u.x, u.y, u.z);
78  context[CONTEXT_CAMERA_V]->setFloat(v.x, v.y, v.z);
79  context[CONTEXT_CAMERA_W]->setFloat(w.x, w.y, w.z);
80 
81  context[CONTEXT_CAMERA_EYE]->setFloat(position.x, position.y, position.z);
82  context[CONTEXT_CAMERA_APERTURE_RADIUS]->setFloat(
83  camera.getPropertyOrValue<double>(CAMERA_PROPERTY_APERTURE_RADIUS.name.c_str(),
84  DEFAULT_CAMERA_APERTURE_RADIUS));
85  context[CONTEXT_CAMERA_FOCAL_DISTANCE]->setFloat(
86  camera.getPropertyOrValue<double>(CAMERA_PROPERTY_FOCAL_DISTANCE.name.c_str(), DEFAULT_CAMERA_FOCAL_DISTANCE));
87  context[CONTEXT_CAMERA_OFFSET]->setFloat(0, 0);
88 
89  context[CONTEXT_CAMERA_STEREO]->setUint(stereo);
90  context[CONTEXT_CAMERA_IPD_OFFSET]->setFloat(ipd_offset.x, ipd_offset.y, ipd_offset.z);
91 }
