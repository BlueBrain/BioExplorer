58 RNASequence::RNASequence(Scene& scene, const RNASequenceDetails& rd)
59  : Node()
60 {
61  const auto& sequence = rd.contents;
62  const size_t nbElements = sequence.length();
63 
64  auto model = scene.createModel();
65 
66  size_t materialId = 0;
67  for (const auto& nucleotid : nucleotidMap)
68  {
69  auto material =
70  model->createMaterial(materialId, nucleotid.second.name);
71  brayns::PropertyMap props;
72  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
73  static_cast<int>(MaterialShadingMode::basic)});
74  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
75  props.setProperty(
76  {MATERIAL_PROPERTY_CHAMELEON_MODE,
77  static_cast<int>(
78  MaterialChameleonMode::undefined_chameleon_mode)});
79  material->setDiffuseColor(nucleotid.second.color);
80  material->updateProperties(props);
81  ++materialId;
82  }
83 
84  const size_t nbMaterials = materialId;
85 
86  PLUGIN_INFO << "Sequence total length: " << nbElements << std::endl;
87  PLUGIN_INFO << "Created " << nbMaterials << " materials" << std::endl;
88 
89  Vector3f U{rd.range[0], rd.range[1], nbElements};
90  Vector3f V{rd.range[0], rd.range[1], nbElements};
91 
92  switch (rd.shape)
93  {
94  case RNAShape::moebius:
95  U = {2.f * M_PI, 4.f * M_PI, nbElements};
96  V = {-0.4f, 0.4f, 1.f};
97  break;
98  case RNAShape::heart:
99  U = {0.f, 2.f * M_PI, nbElements};
100  V = {0.f, 1.f, 1.f};
101  break;
102  default:
103  break;
104  }
105 
106  const float uStep = (U.y - U.x) / U.z;
107  const float vStep = (V.y - V.x) / V.z;
108 
109  size_t elementId = 0;
110  for (float v(V.x); v < V.y; v += vStep)
111  {
112  for (float u(U.x); u < U.y; u += uStep)
113  {
114  Vector3f src;
115  Vector3f dst;
116  switch (rd.shape)
117  {
118  case RNAShape::moebius:
119  {
120  src = _moebius(rd.assemblyParams[0], u, v);
121  dst = _moebius(rd.assemblyParams[0], u + uStep, v);
122  break;
123  }
124  case RNAShape::torus:
125  {
126  src = _torus(rd.assemblyParams[0], u,
127  {rd.params[0], rd.params[1], rd.params[2]});
128  dst = _torus(rd.assemblyParams[0], u + uStep,
129  {rd.params[0], rd.params[1], rd.params[2]});
130  break;
131  }
132  case RNAShape::star:
133  {
134  src = _star(rd.assemblyParams[0], u);
135  dst = _star(rd.assemblyParams[0], u + uStep);
136  break;
137  }
138  case RNAShape::spring:
139  {
140  src = _spring(rd.assemblyParams[0], u);
141  dst = _spring(rd.assemblyParams[0], u + uStep);
142  break;
143  }
144  case RNAShape::trefoilKnot:
145  {
146  src = _trefoilKnot(rd.assemblyParams[0], u,
147  {rd.params[0], rd.params[1], rd.params[2]});
148  dst = _trefoilKnot(rd.assemblyParams[0], u + uStep,
149  {rd.params[0], rd.params[1], rd.params[2]});
150  break;
151  }
152  case RNAShape::heart:
153  {
154  src = _heart(rd.assemblyParams[0], u);
155  dst = _heart(rd.assemblyParams[0], u + uStep);
156  break;
157  }
158  case RNAShape::thing:
159  {
160  src = _thing(rd.assemblyParams[0], u,
161  {rd.params[0], rd.params[1], rd.params[2]});
162  dst = _thing(rd.assemblyParams[0], u + uStep,
163  {rd.params[0], rd.params[1], rd.params[2]});
164  break;
165  }
166  default:
167  PLUGIN_THROW(std::runtime_error("Undefined shape"));
168  break;
169  }
170 
171  const char letter = sequence[elementId];
172  if (nucleotidMap.find(letter) != nucleotidMap.end())
173  {
174  const auto& codon = nucleotidMap[letter];
175  const auto materialId = codon.index;
176  const auto radius = rd.assemblyParams[1];
177  const Vector3f position = {rd.position[0], rd.position[1],
178  rd.position[2]};
179  model->addCylinder(materialId,
180  {position + src, position + dst, radius});
181  if (elementId == 0)
182  model->addSphere(materialId, {position + src, radius});
183  if (elementId == nbElements - 1)
184  model->addSphere(materialId, {position + dst, radius});
185  }
186  if (elementId == nbElements)
187  break;
188  ++elementId;
189  }
190  }
191 
192  // Metadata
193  ModelMetadata metadata;
194  metadata[METADATA_ASSEMBLY] = rd.assemblyName;
195  metadata["RNA sequence"] = sequence;
196  _modelDescriptor =
197  std::make_shared<ModelDescriptor>(std::move(model), rd.name, metadata);
198  if (_modelDescriptor &&
199  !GeneralSettings::getInstance()->getModelVisibilityOnCreation())
200  _modelDescriptor->setVisible(false);
201 }

203 Vector3f RNASequence::_trefoilKnot(const float radius, const float t,
204  const Vector3f& params)
205 {
206  return {radius * (sin(t) + 2.f * sin(params.x * t)),
207  radius * (cos(t) - 2.f * cos(params.y * t)),
208  radius * (-sin(params.z * t))};
209 }

211 Vector3f RNASequence::_torus(const float radius, const float t,
212  const Vector3f& params)
213 {
214  return {radius * (cos(t) + params.x * cos(params.y * t) * cos(t)),
215  radius * (sin(t) + params.x * cos(params.y * t) * sin(t)),
216  radius * params.x * sin(params.y * t)};
217 }

219 Vector3f RNASequence::_star(const float radius, const float t)
220 {
221  return {radius * (2.f * sin(3.f * t) * cos(t)),
222  radius * (2.f * sin(3.f * t) * sin(t)), radius * sin(3.f * t)};
223 }
224 
225 Vector3f RNASequence::_spring(const float radius, const float t)
226 {
227  return {radius * cos(t), radius * sin(t), radius * cos(t)};
228 }

230 Vector3f RNASequence::_heart(const float radius, const float u)
231 {
232  return {radius * 4.f * pow(sin(u), 3.f),
233  radius * 0.25f *
234  (13.f * cos(u) - 5.f * cos(2.f * u) - 2.f * cos(3.f * u) -
235  cos(4.f * u)),
236  0.f};
237 }

239 Vector3f RNASequence::_thing(const float radius, const float t,
240  const Vector3f& params)
241 {
242  return {radius * (sin(t) + params.x * sin(params.y * t)),
243  radius * (cos(t) - params.x * cos(params.y * t)),
244  radius * (-sin(params.z * t))};
245 }

247 Vector3f RNASequence::_moebius(const float radius, const float u, const float v)
248 {
249  return {4.f * radius * (cos(u) + v * cos(u / 2.f) * cos(u)),
250  4.f * radius * (sin(u) + v * cos(u / 2.f) * sin(u)),
251  8.f * radius * (v * sin(u / 2.f))};
252 }
