88 void OOCManager::_loadBricks()
89 {
90  std::set<int32_t> loadedBricks;
91  std::set<int32_t> bricksToLoad;
92  std::vector<ModelDescriptorPtr> modelsToAddToScene;
93  std::vector<ModelDescriptorPtr> modelsToRemoveFromScene;
94  std::vector<ModelDescriptorPtr> modelsToShow;
95  int32_t previousBrickId{std::numeric_limits<int32_t>::max()};
96  CacheLoader loader(_scene);
97 
98  uint32_t nbLoads = 0;
99  double totalLoadingTime = 0.f;
100 
101  while (true)
102  {
103  const Vector3d& cameraPosition = _camera.getPosition();
104  const Vector3i brick = (cameraPosition - _sceneConfiguration.brickSize / 2.0) / _sceneConfiguration.brickSize;
105  const int32_t brickId = brick.z + brick.y * _sceneConfiguration.nbBricks +
106  brick.x * _sceneConfiguration.nbBricks * _sceneConfiguration.nbBricks;
107 
108  if (_frameBuffer && _frameBuffer->numAccumFrames() > 1)
109  {
110  bricksToLoad.clear();
111 
112  // Identify visible bricks (the ones surrounding the camera)
113  std::set<int32_t> visibleBricks;
114  for (int32_t x = 0; x < _nbVisibleBricks; ++x)
115  for (int32_t y = 0; y < _nbVisibleBricks; ++y)
116  for (int32_t z = 0; z < _nbVisibleBricks; ++z)
117  {
118  visibleBricks.insert((z + brick.z) + (y + brick.y) * _sceneConfiguration.nbBricks +
119  (x + brick.x) * _sceneConfiguration.nbBricks *
120  _sceneConfiguration.nbBricks);
121  visibleBricks.insert((-z + brick.z) + (-y + brick.y) * _sceneConfiguration.nbBricks +
122  (-x + brick.x) * _sceneConfiguration.nbBricks *
123  _sceneConfiguration.nbBricks);
124  }
125 
126  // Identify bricks to load
127  for (const int32_t visibleBrick : visibleBricks)
128  if (std::find(loadedBricks.begin(), loadedBricks.end(), visibleBrick) == loadedBricks.end())
129  {
130  bricksToLoad.insert(visibleBrick);
131  if (bricksToLoad.size() >= _nbBricksPerCycle)
132  break;
133  }
134 
135  if (!bricksToLoad.empty())
136  PLUGIN_INFO(3, "Loading bricks " << int32_set_to_string(bricksToLoad));
137 
138  _progress = double(bricksToLoad.size()) / double(_nbBricksPerCycle);
139  // Loading bricks
140  if (!bricksToLoad.empty())
141  {
142  const auto brickToLoad = (*bricksToLoad.begin());
143  loadedBricks.insert(brickToLoad);
144  const auto start = std::chrono::steady_clock::now();
145  try
146  {
147  modelsToAddToScene = loader.importBrickFromDB(brickToLoad);
148  }
149  catch (std::runtime_error& e)
150  {
151  PLUGIN_DEBUG(e.what());
152  }
153  const auto duration =
154  std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now() - start);
155 
156  totalLoadingTime += duration.count();
157  ++nbLoads;
158 
159  bricksToLoad.erase(bricksToLoad.begin());
160  }
161 
162  if (bricksToLoad.size())
163  {
164  PLUGIN_DEBUG("Current brick Id: " << brickId);
165  PLUGIN_DEBUG("Loaded bricks : " << int32_set_to_string(loadedBricks));
166  PLUGIN_DEBUG("Visible bricks : " << int32_set_to_string(visibleBricks));
167 
168  bool visibilityModified = false;
169 
170  // Make visible models visible and remove invisible models
171  auto modelDescriptors = _scene.getModelDescriptors();
172  for (auto modelDescriptor : modelDescriptors)
173  {
174  const auto metadata = modelDescriptor->getMetadata();
175  const auto it = metadata.find(METADATA_BRICK_ID);
176  if (it != metadata.end())
177  {
178  const int32_t id = atoi(it->second.c_str());
179  const bool visible =
180  std::find(visibleBricks.begin(), visibleBricks.end(), id) != visibleBricks.end();
181  if (visible)
182  {
183  if (!modelDescriptor->getVisible())
184  modelsToShow.push_back(modelDescriptor);
185  }
186  else
187  modelsToRemoveFromScene.push_back(modelDescriptor);
188  }
189  }
190 
191  if (_unloadBricks)
192  {
193  // Prevent invisible bricks from being loaded
194  auto i = loadedBricks.begin();
195  while (i != loadedBricks.end())
196  {
197  const auto loadedBrick = (*i);
198  const auto it = std::find(visibleBricks.begin(), visibleBricks.end(), loadedBrick);
199  if (it == visibleBricks.end())
200  loadedBricks.erase(i++);
201  else
202  ++i;
203  }
204  }
205  }
206 
207  bool sceneModified = false;
208  if (_unloadBricks)
209  {
210  for (auto md : modelsToRemoveFromScene)
211  {
212  PLUGIN_DEBUG("Removing model: " << md->getModelID());
213  _scene.removeModel(md->getModelID());
214  sceneModified = true;
215  }
216  modelsToRemoveFromScene.clear();
217  }
218 
219  for (auto md : modelsToAddToScene)
220  {
221  md->setVisible(true);
222  _scene.addModel(md);
223  PLUGIN_DEBUG("Adding model: " << md->getModelID());
224  sleep(_updateFrequency);
225  sceneModified = true;
226  }
227  modelsToAddToScene.clear();
228 
229  for (auto md : modelsToShow)
230  {
231  PLUGIN_DEBUG("Making model visible: " << md->getModelID());
232  md->setVisible(true);
233  sceneModified = true;
234  }
235  modelsToShow.clear();
236  if (sceneModified)
237  _scene.markModified(false);
238  }
239 
240  sleep(_updateFrequency);
241  _averageLoadingTime = totalLoadingTime / nbLoads;
242  PLUGIN_DEBUG("Average loading time (ms): " << _averageLoadingTime);
243  previousBrickId = brickId;
244  }
245 }

247 void OOCManager::_parseArguments(const CommandLineArguments& arguments)
248 {
249  std::string dbHost, dbPort, dbUser, dbPassword, dbName;
250  for (const auto& argument : arguments)
251  {
252  if (argument.first == ARG_OOC_UPDATE_FREQUENCY)
253  _updateFrequency = atof(argument.second.c_str());
254  if (argument.first == ARG_OOC_VISIBLE_BRICKS)
255  _nbVisibleBricks = atoi(argument.second.c_str());
256  if (argument.first == ARG_OOC_UNLOAD_BRICKS)
257  _unloadBricks = true;
258  if (argument.first == ARG_OOC_SHOW_GRID)
259  _showGrid = true;
260  if (argument.first == ARG_OOC_NB_BRICKS_PER_CYCLE)
261  _nbBricksPerCycle = atoi(argument.second.c_str());
262  }
263 
264  // Configuration
265  auto& connector = DBConnector::getInstance();
266  _sceneConfiguration = connector.getSceneConfiguration();
267 
268  const bool disableBroadcasting = std::getenv(ENV_ROCKETS_DISABLE_SCENE_BROADCASTING.c_str()) != nullptr;
269  if (!disableBroadcasting)
270  PLUGIN_THROW(ENV_ROCKETS_DISABLE_SCENE_BROADCASTING +
271  " environment variable must be set when out-of-core is enabled");
272 }
