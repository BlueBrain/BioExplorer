38std::string int32_set_to_string(const std::set<int32_t>& s)
39{
40 std::string str = "[";
41 uint32_t count = 0;
42 for (const auto v : s)
43 {
44 if (count > 0)
45 str += ", ";
46 str += std::to_string(v);
47 ++count;
48 }
49 str += "]";
50 return str;

92void OOCManager::_loadBricks()
93{
94 std::set<int32_t> loadedBricks;
95 std::set<int32_t> bricksToLoad;
96 std::vector<ModelDescriptorPtr> modelsToAddToScene;
97 std::vector<ModelDescriptorPtr> modelsToRemoveFromScene;
98 std::vector<ModelDescriptorPtr> modelsToShow;
99 int32_t previousBrickId{std::numeric_limits<int32_t>::max()};
100 CacheLoader loader(_scene);
101#ifdef USE_PQXX
102 DBConnector connector(_dbConnectionString, _dbSchema);
103#endif
104
105 uint32_t nbLoads = 0;
106 float totalLoadingTime = 0.f;
107
108 while (true)
109 {
110 const Vector3f& cameraPosition = _camera.getPosition();
111 const Vector3i brick =
112 (cameraPosition - _sceneConfiguration.brickSize / 2.f) /
113 _sceneConfiguration.brickSize;
114 const int32_t brickId = brick.z +
115 brick.y * _sceneConfiguration.nbBricks +
116 brick.x * _sceneConfiguration.nbBricks *
117 _sceneConfiguration.nbBricks;
118
119 if (_frameBuffer && _frameBuffer->numAccumFrames() > 1)
120 {
121 bricksToLoad.clear();
122
123 // Identify visible bricks (the ones surrounding the camera)
124 std::set<int32_t> visibleBricks;
125 for (int32_t x = 0; x < _nbVisibleBricks; ++x)
126 for (int32_t y = 0; y < _nbVisibleBricks; ++y)
127 for (int32_t z = 0; z < _nbVisibleBricks; ++z)
128 {
129 visibleBricks.insert(
130 (z + brick.z) +
131 (y + brick.y) * _sceneConfiguration.nbBricks +
132 (x + brick.x) * _sceneConfiguration.nbBricks *
133 _sceneConfiguration.nbBricks);
134 visibleBricks.insert(
135 (-z + brick.z) +
136 (-y + brick.y) * _sceneConfiguration.nbBricks +
137 (-x + brick.x) * _sceneConfiguration.nbBricks *
138 _sceneConfiguration.nbBricks);
139 }
140
141 // Identify bricks to load
142 for (const int32_t visibleBrick : visibleBricks)
143 if (std::find(loadedBricks.begin(), loadedBricks.end(),
144 visibleBrick) == loadedBricks.end())
145 {
146 bricksToLoad.insert(visibleBrick);
147 if (bricksToLoad.size() >= _nbBricksPerCycle)
148 break;
149 }
150
151 if (!bricksToLoad.empty())
152 PLUGIN_INFO("Loading bricks "
153 << int32_set_to_string(bricksToLoad));
154
155 _progress = float(bricksToLoad.size()) / float(_nbBricksPerCycle);
156 // Loading bricks
157 if (!bricksToLoad.empty())
158 {
159 const auto brickToLoad = (*bricksToLoad.begin());
160 loadedBricks.insert(brickToLoad);
161 const auto start = std::chrono::steady_clock::now();
162 try
163 {
164#ifdef USE_PQXX
165 modelsToAddToScene =
166 loader.importBrickFromDB(connector, brickToLoad);
167#else
168 char idStr[7];
169 sprintf(idStr, "%06d", brickToLoad);
170 const std::string filename =
171 _bricksFolder + "/brick" + idStr + ".bioexplorer";
172 modelsToAddToScene =
173 loader.importModelsFromFile(filename, brickToLoad);
174#endif
175 }
176 catch (std::runtime_error& e)
177 {
178 PLUGIN_DEBUG(e.what());
179 }
180 const auto duration =
181 std::chrono::duration_cast<std::chrono::milliseconds>(
182 std::chrono::steady_clock::now() - start);
183
184 totalLoadingTime += duration.count();
185 ++nbLoads;
186
187 bricksToLoad.erase(bricksToLoad.begin());
188 }
189
190 if (bricksToLoad.size())
191 {
192 PLUGIN_DEBUG("Current brick Id: " << brickId);
193 PLUGIN_DEBUG(
194 "Loaded bricks : " << int32_set_to_string(loadedBricks));
195 PLUGIN_DEBUG(
196 "Visible bricks : " << int32_set_to_string(visibleBricks));
197
198 bool visibilityModified = false;
199
200 // Make visible models visible and remove invisible models
201 auto modelDescriptors = _scene.getModelDescriptors();
202 for (auto modelDescriptor : modelDescriptors)
203 {
204 const auto metadata = modelDescriptor->getMetadata();
205 const auto it = metadata.find(METADATA_BRICK_ID);
206 if (it != metadata.end())
207 {
208 const int32_t id = atoi(it->second.c_str());
209 const bool visible =
210 std::find(visibleBricks.begin(),
211 visibleBricks.end(),
212 id) != visibleBricks.end();
213 if (visible)
214 {
215 if (!modelDescriptor->getVisible())
216 modelsToShow.push_back(modelDescriptor);
217 }
218 else
219 modelsToRemoveFromScene.push_back(modelDescriptor);
220 }
221 }
222
223 if (_unloadBricks)
224 {
225 // Prevent invisible bricks from being loaded
226 auto i = loadedBricks.begin();
227 while (i != loadedBricks.end())
228 {
229 const auto loadedBrick = (*i);
230 const auto it =
231 std::find(visibleBricks.begin(),
232 visibleBricks.end(), loadedBrick);
233 if (it == visibleBricks.end())
234 loadedBricks.erase(i++);
235 else
236 ++i;
237 }
238 }
239 }
240
241 bool sceneModified = false;
242 if (_unloadBricks)
243 {
244 for (auto md : modelsToRemoveFromScene)
245 {
246 PLUGIN_DEBUG("Removing model: " << md->getModelID());
247 _scene.removeModel(md->getModelID());
248 sceneModified = true;
249 }
250 modelsToRemoveFromScene.clear();
251 }
252
253 for (auto md : modelsToAddToScene)
254 {
255 md->setVisible(true);
256 _scene.addModel(md);
257 PLUGIN_DEBUG("Adding model: " << md->getModelID());
258 sleep(_updateFrequency);
259 sceneModified = true;
260 }
261 modelsToAddToScene.clear();
262
263 for (auto md : modelsToShow)
264 {
265 PLUGIN_DEBUG("Making model visible: " << md->getModelID());
266 md->setVisible(true);
267 sceneModified = true;
268 }
269 modelsToShow.clear();
270 if (sceneModified)
271 _scene.markModified(false);
272 }
273
274 sleep(_updateFrequency);
275 _averageLoadingTime = totalLoadingTime / nbLoads;
276 PLUGIN_DEBUG("Average loading time (ms): " << _averageLoadingTime);
277 previousBrickId = brickId;
278 }
279}

281void OOCManager::_parseArguments(const CommandLineArguments& arguments)
282{
283 std::string dbHost, dbPort, dbUser, dbPassword, dbName;
284 for (const auto& argument : arguments)
285 {
286#ifdef USE_PQXX
287 if (argument.first == ARG_OOC_DB_HOST)
288 dbHost = argument.second;
289 if (argument.first == ARG_OOC_DB_PORT)
290 dbPort = argument.second;
291 if (argument.first == ARG_OOC_DB_USER)
292 dbUser = argument.second;
293 if (argument.first == ARG_OOC_DB_PASSWORD)
294 dbPassword = argument.second;
295 if (argument.first == ARG_OOC_DB_NAME)
296 dbName = argument.second;
297 if (argument.first == ARG_OOC_DB_SCHEMA)
298 _dbSchema = argument.second;
299#else
300 if (argument.first == ARG_OOC_BRICKS_FOLDER)
301 _bricksFolder = argument.second;
302#endif
303 if (argument.first == ARG_OOC_UPDATE_FREQUENCY)
304 _updateFrequency = atof(argument.second.c_str());
305 if (argument.first == ARG_OOC_VISIBLE_BRICKS)
306 _nbVisibleBricks = atoi(argument.second.c_str());
307 if (argument.first == ARG_OOC_UNLOAD_BRICKS)
308 _unloadBricks = true;
309 if (argument.first == ARG_OOC_SHOW_GRID)
310 _showGrid = true;
311 if (argument.first == ARG_OOC_NB_BRICKS_PER_CYCLE)
312 _nbBricksPerCycle = atoi(argument.second.c_str());
313 }
314
315#ifdef USE_PQXX
316 // Sanity checks
317 _dbConnectionString = "host=" + dbHost + " port=" + dbPort +
318 " dbname=" + dbName + " user=" + dbUser +
319 " password=" + dbPassword;
320
321 // Configuration
322 DBConnector connector(_dbConnectionString, _dbSchema);
323 _sceneConfiguration = connector.getSceneConfiguration();
324#endif
325
326 const bool disableBroadcasting =
327 std::getenv(ENV_ROCKETS_DISABLE_SCENE_BROADCASTING.c_str()) != nullptr;
328 if (!disableBroadcasting)
329 PLUGIN_THROW(
330 ENV_ROCKETS_DISABLE_SCENE_BROADCASTING +
331 " environment variable must be set when out-of-core is enabled");
332}
