45 std::shared_ptr<T> _find(const std::vector<std::shared_ptr<T>>& list, const size_t id,
46  size_t (U::*getID)() const = &T::getID)
47 {
48  auto i = std::find_if(list.begin(), list.end(), [id, getID](auto x) { return id == ((*x).*getID)(); });
49  return i == list.end() ? std::shared_ptr<T>{} : *i;
50 }

52 template <typename T, typename U = T>
53 std::shared_ptr<T> _remove(std::vector<std::shared_ptr<T>>& list, const size_t id,
54  size_t (U::*getID)() const = &T::getID)
55 {
56  auto i = std::find_if(list.begin(), list.end(), [id, getID](auto x) { return id == ((*x).*getID)(); });
57  if (i == list.end())
58  return std::shared_ptr<T>{};
59  auto result = *i;
60  list.erase(i);
61  return result;
62 }

117 size_t Scene::addModel(ModelDescriptorPtr modelDescriptor)
118 {
119  auto& model = modelDescriptor->getModel();
120  if (model.empty())
121  throw std::runtime_error("Empty models not supported.");
122 
123  const auto defaultBVHFlags = _geometryParameters.getDefaultBVHFlags();
124 
125  model.setBVHFlags(defaultBVHFlags);
126  model.buildBoundingBox();
127 
128  // Since models can be added concurrently we check if that is supported
129  if (supportsConcurrentSceneUpdates())
130  model.commitGeometry();
131 
132  {
133  std::unique_lock<std::shared_timed_mutex> lock(_modelMutex);
134  modelDescriptor->setModelID(_modelID++);
135  _modelDescriptors.push_back(modelDescriptor);
136 
137  // add default instance of this model to render something
138  if (modelDescriptor->getInstances().empty())
139  modelDescriptor->addInstance({true, true, modelDescriptor->getTransformation()});
140  }
141 
142  computeBounds();
143  markModified();
144 
145  return modelDescriptor->getModelID();
146 }

148 bool Scene::removeModel(const size_t id)
149 {
150  ModelDescriptorPtr model = nullptr;
151 
152  if (supportsConcurrentSceneUpdates())
153  {
154  {
155  std::unique_lock<std::shared_timed_mutex> lock(_modelMutex);
156  model = _remove(_modelDescriptors, id, &ModelDescriptor::getModelID);
157  }
158  if (model)
159  model->callOnRemoved();
160  }
161  else
162  {
163  {
164  std::unique_lock<std::shared_timed_mutex> lock(_modelMutex);
165  model = _find(_modelDescriptors, id, &ModelDescriptor::getModelID);
166  }
167  if (model)
168  model->markForRemoval();
169  }
170 
171  if (model)
172  {
173  markModified();
174  return true;
175  }
176 
177  return false;
178 }

435 void Scene::computeBounds()
436 {
437  std::unique_lock<std::shared_timed_mutex> lock(_modelMutex);
438  _bounds.reset();
439  for (const auto& modelDescriptor : _modelDescriptors)
440  {
441  const auto& modelBounds = modelDescriptor->getModel().getBounds();
442  Transformation modelTransformation;
443  modelTransformation.setTranslation(modelBounds.getCenter());
444 
445  const auto modelHalfSize = modelBounds.getSize() / 2.0;
446 
447  Transformation finalTransformation = modelTransformation * modelDescriptor->getTransformation();
448  _bounds.merge(finalTransformation.getTranslation() - modelHalfSize);
449  _bounds.merge(finalTransformation.getTranslation() + modelHalfSize);
450  for (const auto& instance : modelDescriptor->getInstances())
451  {
452  finalTransformation = modelTransformation * instance.getTransformation();
453  _bounds.merge(finalTransformation.getTranslation() - modelHalfSize);
454  _bounds.merge(finalTransformation.getTranslation() + modelHalfSize);
455  }
456  }
457 
458  if (_bounds.isEmpty())
459  // If no model is enabled. return empty bounding box
460  _bounds.merge({0, 0, 0});
461 }
