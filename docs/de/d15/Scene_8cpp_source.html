43 std::shared_ptr<T> _find(const std::vector<std::shared_ptr<T>>& list, const size_t id,
44  size_t (U::*getID)() const = &T::getID)
45 {
46  auto i = std::find_if(list.begin(), list.end(), [id, getID](auto x) { return id == ((*x).*getID)(); });
47  return i == list.end() ? std::shared_ptr<T>{} : *i;
48 }

50 template <typename T, typename U = T>
51 std::shared_ptr<T> _remove(std::vector<std::shared_ptr<T>>& list, const size_t id,
52  size_t (U::*getID)() const = &T::getID)
53 {
54  auto i = std::find_if(list.begin(), list.end(), [id, getID](auto x) { return id == ((*x).*getID)(); });
55  if (i == list.end())
56  return std::shared_ptr<T>{};
57  auto result = *i;
58  list.erase(i);
59  return result;
60 }

115 size_t Scene::addModel(ModelDescriptorPtr modelDescriptor)
116 {
117  auto& model = modelDescriptor->getModel();
118  if (model.empty())
119  throw std::runtime_error("Empty models not supported.");
120 
121  const auto defaultBVHFlags = _geometryParameters.getDefaultBVHFlags();
122 
123  model.setBVHFlags(defaultBVHFlags);
124  model.buildBoundingBox();
125 
126  // Since models can be added concurrently we check if that is supported
127  if (supportsConcurrentSceneUpdates())
128  model.commitGeometry();
129 
130  {
131  std::unique_lock<std::shared_timed_mutex> lock(_modelMutex);
132  modelDescriptor->setModelID(_modelID++);
133  _modelDescriptors.push_back(modelDescriptor);
134 
135  // add default instance of this model to render something
136  if (modelDescriptor->getInstances().empty())
137  modelDescriptor->addInstance({true, true, modelDescriptor->getTransformation()});
138  }
139 
140  computeBounds();
141  markModified();
142 
143  return modelDescriptor->getModelID();
144 }

146 bool Scene::removeModel(const size_t id)
147 {
148  ModelDescriptorPtr model = nullptr;
149 
150  if (supportsConcurrentSceneUpdates())
151  {
152  {
153  std::unique_lock<std::shared_timed_mutex> lock(_modelMutex);
154  model = _remove(_modelDescriptors, id, &ModelDescriptor::getModelID);
155  }
156  if (model)
157  model->callOnRemoved();
158  }
159  else
160  {
161  {
162  std::unique_lock<std::shared_timed_mutex> lock(_modelMutex);
163  model = _find(_modelDescriptors, id, &ModelDescriptor::getModelID);
164  }
165  if (model)
166  model->markForRemoval();
167  }
168 
169  if (model)
170  {
171  markModified();
172  return true;
173  }
174 
175  return false;
176 }

426 void Scene::computeBounds()
427 {
428  std::unique_lock<std::shared_timed_mutex> lock(_modelMutex);
429  _bounds.reset();
430  for (const auto& modelDescriptor : _modelDescriptors)
431  {
432  const auto& modelBounds = modelDescriptor->getModel().getBounds();
433  Transformation modelTransformation;
434  modelTransformation.setTranslation(modelBounds.getCenter());
435 
436  const auto modelHalfSize = modelBounds.getSize() / 2.0;
437 
438  Transformation finalTransformation = modelTransformation * modelDescriptor->getTransformation();
439  _bounds.merge(finalTransformation.getTranslation() - modelHalfSize);
440  _bounds.merge(finalTransformation.getTranslation() + modelHalfSize);
441  for (const auto& instance : modelDescriptor->getInstances())
442  {
443  finalTransformation = modelTransformation * instance.getTransformation();
444  _bounds.merge(finalTransformation.getTranslation() - modelHalfSize);
445  _bounds.merge(finalTransformation.getTranslation() + modelHalfSize);
446  }
447  }
448 
449  if (_bounds.isEmpty())
450  // If no model is enabled. return empty bounding box
451  _bounds.merge({0, 0, 0});
452 }
