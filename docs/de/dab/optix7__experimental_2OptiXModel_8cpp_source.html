40 inline OptixAabb sphereBounds(const Vector3f& center, const float radius)
41 {
42  const Vector3f r{radius, radius, radius};
43  const Vector3f m = center - r;
44  const Vector3f M = center + r;
45  return {m.x, m.y, m.z, M.x, M.y, M.z};
46 }

48 inline OptixAabb cylinderBounds(const Vector3f& center, const Vector3f& up, const float radius)
49 {
50  const Vector3f m = {std::min(center.x, up.x) - radius, std::min(center.y, up.y) - radius,
51  std::min(center.z, up.z) - radius};
52  const Vector3f M = {std::max(center.x, up.x) + radius, std::max(center.y, up.y) + radius,
53  std::max(center.z, up.z) + radius};
54  return {m.x, m.y, m.z, M.x, M.y, M.z};
55 }

57 inline OptixAabb coneBounds(const Vector3f& center, const Vector3f& up, const float centerRadius, const float upRadius)
58 {
59  const float radius = std::max(centerRadius, upRadius);
60  const Vector3f m = {std::min(center.x, up.x) - radius, std::min(center.y, up.y) - radius,
61  std::min(center.z, up.z) - radius};
62  const Vector3f M = {std::max(center.x, up.x) + radius, std::max(center.y, up.y) + radius,
63  std::max(center.z, up.z) + radius};
64  return {m.x, m.y, m.z, M.x, M.y, M.z};
65 }

113 void OptiXModel::_buildGAS(const OptixBuildInput& buildInput)
114 {
115  PLUGIN_INFO("Registering OptiX Geometry Acceleration Structures (GAS)");
116  auto& state = OptiXContext::getInstance().getState();
117 
118  CUDA_CHECK(cudaFree(reinterpret_cast<void*>(state.d_gas_output_buffer)));
119 
120  OptixAccelBuildOptions accelOptions = {OPTIX_BUILD_FLAG_ALLOW_COMPACTION, OPTIX_BUILD_OPERATION_BUILD};
121  OptixAccelBufferSizes gasBufferSizes;
122  OPTIX_CHECK(optixAccelComputeMemoryUsage(state.context, &accelOptions, &buildInput, 1, &gasBufferSizes));
123  CUdeviceptr dTempBufferGas;
124  CUDA_CHECK(cudaMalloc(reinterpret_cast<void**>(&dTempBufferGas), gasBufferSizes.tempSizeInBytes));
125 
126  // non-compacted output
127  CUdeviceptr dBufferTempOutputGasAndCompactedSize;
128  size_t compactedSizeOffset = roundUp<size_t>(gasBufferSizes.outputSizeInBytes, 8ull);
129  CUDA_CHECK(cudaMalloc(reinterpret_cast<void**>(&dBufferTempOutputGasAndCompactedSize), compactedSizeOffset + 8));
130 
131  OptixAccelEmitDesc emitProperty = {};
132  emitProperty.type = OPTIX_PROPERTY_TYPE_COMPACTED_SIZE;
133  emitProperty.result = (CUdeviceptr)((char*)dBufferTempOutputGasAndCompactedSize + compactedSizeOffset);
134 
135  OPTIX_CHECK(optixAccelBuild(state.context,
136  0, // CUDA stream
137  &accelOptions, &buildInput,
138  1, // num build inputs
139  dTempBufferGas, gasBufferSizes.tempSizeInBytes, dBufferTempOutputGasAndCompactedSize,
140  gasBufferSizes.outputSizeInBytes, &state.gas_handle,
141  &emitProperty, // emitted property list
142  1 // num emitted properties
143  ));
144 
145  size_t compacted_gas_size;
146  CUDA_CHECK(cudaMemcpy(&compacted_gas_size, (void*)emitProperty.result, sizeof(size_t), cudaMemcpyDeviceToHost));
147 
148  if (compacted_gas_size < gasBufferSizes.outputSizeInBytes)
149  {
150  CUDA_CHECK(cudaMalloc(reinterpret_cast<void**>(&state.d_gas_output_buffer), compacted_gas_size));
151  // use handle as input and output
152  OPTIX_CHECK(optixAccelCompact(state.context, 0, state.gas_handle, state.d_gas_output_buffer, compacted_gas_size,
153  &state.gas_handle));
154 
155  CUDA_CHECK(cudaFree((void*)dBufferTempOutputGasAndCompactedSize));
156  }
157  else
158  state.d_gas_output_buffer = dBufferTempOutputGasAndCompactedSize;
159 }

161 void OptiXModel::_commitGeometry()
162 {
163  std::vector<OptixAabb> optixAABBs;
164 
165  for (const auto& geometries : _geometries->_spheres)
166  for (const auto& geometry : geometries.second)
167  optixAABBs.push_back(sphereBounds(geometry.center, geometry.radius));
168 
169  for (const auto& geometries : _geometries->_cylinders)
170  for (const auto& geometry : geometries.second)
171  optixAABBs.push_back(cylinderBounds(geometry.center, geometry.up, geometry.radius));
172 
173  for (const auto& geometries : _geometries->_cones)
174  for (const auto& geometry : geometries.second)
175  optixAABBs.push_back(coneBounds(geometry.center, geometry.up, geometry.centerRadius, geometry.upRadius));
176 
177  const auto nbAABBs = optixAABBs.size();
178  PLUGIN_INFO("Number of AABB: " << nbAABBs);
179 
180  uint32_ts sbtIndices;
181  uint32_ts aabbInputFlags;
182  sbtIndices.resize(nbAABBs);
183  aabbInputFlags.resize(nbAABBs);
184  for (uint32_t i = 0; i < nbAABBs; ++i)
185  {
186  sbtIndices[i] = i;
187  aabbInputFlags[i] = OPTIX_GEOMETRY_FLAG_DISABLE_ANYHIT;
188  }
189 
190  // AABB build input
191  CUdeviceptr dAABBs = 0;
192  uint64_t size = nbAABBs * sizeof(OptixAabb);
193  CUDA_CHECK(cudaMalloc(reinterpret_cast<void**>(&dAABBs), size));
194  CUDA_CHECK(cudaMemcpy(reinterpret_cast<void*>(dAABBs), optixAABBs.data(), size, cudaMemcpyHostToDevice));
195 
196  // SBT indices
197  CUdeviceptr dSBTIndices = 0;
198  size = nbAABBs * sizeof(uint32_t);
199  CUDA_CHECK(cudaMalloc(reinterpret_cast<void**>(&dSBTIndices), size));
200  CUDA_CHECK(cudaMemcpy(reinterpret_cast<void*>(dSBTIndices), sbtIndices.data(), size, cudaMemcpyHostToDevice));
201 
202  // AABB description
203  OptixBuildInput aabbInput = {};
204  aabbInput.type = OPTIX_BUILD_INPUT_TYPE_CUSTOM_PRIMITIVES;
205  aabbInput.customPrimitiveArray.aabbBuffers = &dAABBs;
206  aabbInput.customPrimitiveArray.flags = aabbInputFlags.data();
207  aabbInput.customPrimitiveArray.numSbtRecords = nbAABBs;
208  aabbInput.customPrimitiveArray.numPrimitives = nbAABBs;
209  aabbInput.customPrimitiveArray.sbtIndexOffsetBuffer = dSBTIndices;
210  aabbInput.customPrimitiveArray.sbtIndexOffsetSizeInBytes = sizeof(uint32_t);
211  aabbInput.customPrimitiveArray.primitiveIndexOffset = 0;
212 
213  _buildGAS(aabbInput);
214 
215  CUDA_CHECK(cudaFree(reinterpret_cast<void*>(dAABBs)));
216  CUDA_CHECK(cudaFree(reinterpret_cast<void*>(dSBTIndices)));
217 }

219 void OptiXModel::_createSBT()
220 {
221  auto& state = OptiXContext::getInstance().getState();
222  PLUGIN_DEBUG("Creating OptiX SBT Group records");
223  std::vector<HitGroupRecord> hitGroupRecords;
224 
225  // Spheres
226  for (const auto& geometries : _geometries->_spheres)
227  {
228  const auto material = _materials[geometries.first];
229  const Vector3f kd = material->getDiffuseColor();
230  const Vector3f ks = material->getSpecularColor();
231  const float r = material->getReflectionIndex();
232  const float phongExp = material->getSpecularExponent();
233  for (const auto& geometry : geometries.second)
234  {
235  const auto& center = geometry.center;
236  const GeometryData::Sphere optixGeometry{{center.x, center.y, center.z}, geometry.radius};
237  {
238  // Radiance
239  HitGroupRecord hitGroupRecord;
240  OPTIX_CHECK(optixSbtRecordPackHeader(state.sphere_radiance_prog_group, &hitGroupRecord));
241  hitGroupRecord.data.geometry.sphere = optixGeometry;
242  hitGroupRecord.data.shading.phong = {
243  {0.f, 0.f, 0.f}, // Ka
244  {kd.x, kd.y, kd.z}, // Kd
245  {ks.x, ks.y, ks.z}, // Ks
246  {r, r, r}, // Kr
247  phongExp, // phong_exp
248  };
249  hitGroupRecords.push_back(hitGroupRecord);
250  }
251 
252  {
253  // Occlusion
254  HitGroupRecord hitGroupRecord;
255  OPTIX_CHECK(optixSbtRecordPackHeader(state.sphere_occlusion_prog_group, &hitGroupRecord));
256  hitGroupRecord.data.geometry.sphere = optixGeometry;
257  hitGroupRecords.push_back(hitGroupRecord);
258  }
259  }
260  }
261 
262  // Cylinders
263  for (const auto& geometries : _geometries->_cylinders)
264  {
265  const auto material = _materials[geometries.first];
266  const Vector3f kd = material->getDiffuseColor();
267  const Vector3f ks = material->getSpecularColor();
268  const float r = material->getReflectionIndex();
269  const float phongExp = material->getSpecularExponent();
270  for (const auto& geometry : geometries.second)
271  {
272  const auto& center = geometry.center;
273  const auto& up = geometry.up;
274  const GeometryData::Cylinder optixGeometry{{center.x, center.y, center.z},
275  {up.x, up.y, up.z},
276  geometry.radius};
277  {
278  // Radiance
279  HitGroupRecord hitGroupRecord;
280  OPTIX_CHECK(optixSbtRecordPackHeader(state.cylinder_radiance_prog_group, &hitGroupRecord));
281  hitGroupRecord.data.geometry.cylinder = optixGeometry;
282  hitGroupRecord.data.shading.phong = {
283  {0.f, 0.f, 0.f}, // Ka
284  {kd.x, kd.y, kd.z}, // Kd
285  {ks.x, ks.y, ks.z}, // Ks
286  {r, r, r}, // Kr
287  phongExp, // phong_exp
288  };
289  hitGroupRecords.push_back(hitGroupRecord);
290  }
291 
292  {
293  // Occlusion
294  HitGroupRecord hitGroupRecord;
295  OPTIX_CHECK(optixSbtRecordPackHeader(state.cylinder_occlusion_prog_group, &hitGroupRecord));
296  hitGroupRecord.data.geometry.cylinder = optixGeometry;
297  hitGroupRecords.push_back(hitGroupRecord);
298  }
299  }
300  }
301 
302  // Cones
303  for (const auto& geometries : _geometries->_cones)
304  {
305  const auto material = _materials[geometries.first];
306  const Vector3f kd = material->getDiffuseColor();
307  const Vector3f ks = material->getSpecularColor();
308  const float r = material->getReflectionIndex();
309  const float phongExp = material->getSpecularExponent();
310  for (const auto& geometry : geometries.second)
311  {
312  const auto& center = geometry.center;
313  const auto& up = geometry.up;
314  const auto centerRadius = geometry.centerRadius;
315  const auto upRadius = geometry.upRadius;
316  const GeometryData::Cone optixGeometry{{center.x, center.y, center.z},
317  {up.x, up.y, up.z},
318  centerRadius,
319  upRadius};
320  {
321  // Radiance
322  HitGroupRecord hitGroupRecord;
323  OPTIX_CHECK(optixSbtRecordPackHeader(state.cone_radiance_prog_group, &hitGroupRecord));
324  hitGroupRecord.data.geometry.cone = optixGeometry;
325  hitGroupRecord.data.shading.phong = {
326  {0.f, 0.f, 0.f}, // Ka
327  {kd.x, kd.y, kd.z}, // Kd
328  {ks.x, ks.y, ks.z}, // Ks
329  {r, r, r}, // Kr
330  phongExp, // phong_exp
331  };
332  hitGroupRecords.push_back(hitGroupRecord);
333  }
334 
335  {
336  // Occlusion
337  HitGroupRecord hitGroupRecord;
338  OPTIX_CHECK(optixSbtRecordPackHeader(state.cone_occlusion_prog_group, &hitGroupRecord));
339  hitGroupRecord.data.geometry.cone = optixGeometry;
340  hitGroupRecords.push_back(hitGroupRecord);
341  }
342  }
343  }
344 
345  PLUGIN_INFO("Registering " << hitGroupRecords.size() << " Hit Group Records");
346  CUdeviceptr dHitGroupRecords = 0;
347  const uint32_t hitGroupRecordSize = sizeof(HitGroupRecord);
348  PLUGIN_DEBUG("HitGroupData : " << sizeof(HitGroupData));
349  PLUGIN_DEBUG("HitGroupData(geometry): " << sizeof(HitGroupData::geometry));
350  PLUGIN_DEBUG("HitGroupData(shading) : " << sizeof(HitGroupData::shading));
351  PLUGIN_DEBUG("HitGroupData(total) : " << sizeof(HitGroupData) + OPTIX_SBT_RECORD_HEADER_SIZE);
352  PLUGIN_DEBUG("HitGroupRecord : " << hitGroupRecordSize);
353 
354  const size_t bufferSize = hitGroupRecordSize * hitGroupRecords.size();
355  CUDA_CHECK(cudaMalloc(reinterpret_cast<void**>(&dHitGroupRecords), bufferSize));
356 
357  CUDA_CHECK(cudaMemcpy(reinterpret_cast<void*>(dHitGroupRecords), hitGroupRecords.data(), bufferSize,
358  cudaMemcpyHostToDevice));
359 
360  // CUDA_CHECK(cudaFree(reinterpret_cast<void*>(state.sbt.hitgroupRecordBase)));
361 
362  state.sbt.hitgroupRecordBase = dHitGroupRecords;
363  state.sbt.hitgroupRecordCount = hitGroupRecords.size();
364  state.sbt.hitgroupRecordStrideInBytes = hitGroupRecordSize;
365 }

383 void OptiXModel::buildBoundingBox()
384 {
385  if (_boundingBoxBuilt)
386  return;
387 
388  _boundingBoxBuilt = true;
389 
390  auto material = createMaterial(BOUNDINGBOX_MATERIAL_ID, "bounding_box");
391  material->setDiffuseColor({1, 1, 1});
392  material->setEmission(1.f);
393 
394  const Vector3f s(0.5f);
395  const Vector3f c(0.5f);
396  const float radius = 0.005f;
397  const Vector3f positions[8] = {
398  {c.x - s.x, c.y - s.y, c.z - s.z}, {c.x + s.x, c.y - s.y, c.z - s.z}, // 6--------7
399  {c.x - s.x, c.y + s.y, c.z - s.z}, // /| /|
400  {c.x + s.x, c.y + s.y, c.z - s.z}, // 2--------3 |
401  {c.x - s.x, c.y - s.y, c.z + s.z}, // | | | |
402  {c.x + s.x, c.y - s.y, c.z + s.z}, // | 4------|-5
403  {c.x - s.x, c.y + s.y, c.z + s.z}, // |/ |/
404  {c.x + s.x, c.y + s.y, c.z + s.z} // 0--------1
405  };
406 
407  for (size_t i = 0; i < 8; ++i)
408  addSphere(BOUNDINGBOX_MATERIAL_ID, Sphere(positions[i], radius));
409 
410  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[0], positions[1], radius});
411  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[2], positions[3], radius});
412  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[4], positions[5], radius});
413  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[6], positions[7], radius});
414 
415  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[0], positions[2], radius});
416  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[1], positions[3], radius});
417  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[4], positions[6], radius});
418  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[5], positions[7], radius});
419 
420  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[0], positions[4], radius});
421  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[1], positions[5], radius});
422  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[2], positions[6], radius});
423  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[3], positions[7], radius});
424 }
