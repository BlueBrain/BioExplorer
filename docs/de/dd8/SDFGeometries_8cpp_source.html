41 {
42  size_t materialId = 0;
43  const bool useSdf = true;
44  const Vector3f displacement{0.1f, 10.f, 0.f};

51  neighbours.insert(

53  materialId, -1, neighbours, displacement));
54  neighbours.insert(modelContainer.addSphere(Vector3d(-0.5, 0, 0), 0.25,
55  materialId, -1, neighbours,
56  displacement));
57  neighbours.insert(modelContainer.addSphere(Vector3d(0.5, 0, 0), 0.25,
58  materialId, -1, neighbours,
59  displacement));
60  neighbours.insert(modelContainer.addCone(Vector3d(0, 0.25, 0), 0.5,
61  Vector3d(0, 1, 0), 0.0, materialId,
62  -1, neighbours, displacement));

64  modelContainer.commitToModel();
