33Glycans::Glycans(Scene& scene, const SugarsDetails& details)
34 : Molecule(scene, {})
35 , _details(details)
36{
37 size_t lineIndex{0};
38
39 std::stringstream lines{_details.contents};
40 std::string line;
41 std::string title{details.name};
42 std::string header{details.name};
43
44 while (getline(lines, line, '\n'))
45 {
46 if (line.find(KEY_ATOM) == 0 || line.find(KEY_HETATM) == 0)
47 _readAtom(line, true);
48 else if (details.loadBonds && line.find(KEY_CONECT) == 0)
49 _readConnect(line);
50 else if (line.find(KEY_HEADER) == 0)
51 header = _readHeader(line);
52 else if (line.find(KEY_TITLE) == 0)
53 title = _readTitle(line);
54 }
55 auto model = scene.createModel();
56
57 // Build 3d models according to atoms positions (re-centered to origin)
58 Boxf bounds;
59
60 // Recenter
61 if (_details.recenter)
62 {
63 // Get current center
64 for (const auto& atom : _atomMap)
65 bounds.merge(atom.second.position);
66 const auto& center = bounds.getCenter();
67
68 const auto firstAtomPosition = _atomMap.begin()->second.position;
69 const auto translation = center - firstAtomPosition;
70 // Translate according to position of first atom
71 for (auto& atom : _atomMap)
72 atom.second.position -= translation;
73 }
74
75 _buildModel(_details.assemblyName, _details.name, title, header,
76 _details.representation, _details.atomRadiusMultiplier,
77 _details.loadBonds);
78}
