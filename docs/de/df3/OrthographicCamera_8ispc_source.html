22 /*! \brief A plain orthographic camera; no depth of field */
23 struct OrthographicCamera
24 {
25  Camera super;
26 
27  vec3f dir; //!< \brief direction of all rays generated by this camera
28  vec3f pos_00; //!< \brief lower left position of the camera image plane,
29  //! i.e. origin of a ray with screenSample=(0,0)
30  vec3f pos_du; //!< \brief delta of ray origin between two pixels in X
31  vec3f pos_dv; //!< \brief delta of ray origin between two pixels in Y
32 
33  // Clip planes
34  const uniform vec4f* clipPlanes;
35  unsigned int numClipPlanes;
36 };
37 
38 void OrthographicCamera_initRay(uniform Camera* uniform _self, varying Ray& ray, const varying CameraSample& sample)
39 {
40  uniform OrthographicCamera* uniform self = (uniform OrthographicCamera * uniform) _self;
41 
42  const vec2f screen = Camera_subRegion(_self, sample.screen);
43 
44  const vec3f dir = self->dir;
45  const vec3f org = self->pos_00 + screen.x * self->pos_du + screen.y * self->pos_dv;
46 
47  float nearClip = self->super.nearClip;
48  float farClip = inf;
49  const varying vec3f direction = normalize(dir);
50 
51  clipRay(self->clipPlanes, self->numClipPlanes, org, direction, nearClip, farClip);
52 
53  const float time = Camera_shutterTime(_self, sample.time);
54  setRay(ray, org, dir, nearClip, farClip, time);
55 }

57 /*! create a new ispc-side version of a orthographiccamera - with given
58  pointer to the c-class version - and return both class pointer and
59  pointer to internal data back via the supplied reference
60  parameters */
61 export void* uniform OrthographicCamera_create(void* uniform cppE)
62 {
63  uniform OrthographicCamera* uniform cam = uniform new uniform OrthographicCamera;
64  cam->super.cppEquivalent = cppE;
65  cam->super.initRay = OrthographicCamera_initRay;
66  cam->super.doesDOF = false;
67  cam->numClipPlanes = 0;
68  return cam;
69 }
