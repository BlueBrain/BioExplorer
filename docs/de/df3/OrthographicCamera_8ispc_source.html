24 /*! \brief A plain orthographic camera; no depth of field */
25 struct OrthographicCamera
26 {
27  Camera super;
28 
29  vec3f dir; //!< \brief direction of all rays generated by this camera
30  vec3f pos_00; //!< \brief lower left position of the camera image plane,
31  //! i.e. origin of a ray with screenSample=(0,0)
32  vec3f pos_du; //!< \brief delta of ray origin between two pixels in X
33  vec3f pos_dv; //!< \brief delta of ray origin between two pixels in Y
34 
35  // Clip planes
36  const uniform vec4f* clipPlanes;
37  unsigned int numClipPlanes;
38 };

40 void OrthographicCamera_initRay(uniform Camera* uniform _self, varying Ray& ray, const varying CameraSample& sample)
41 {
42  uniform OrthographicCamera* uniform self = (uniform OrthographicCamera * uniform) _self;
43 
44  const vec2f screen = Camera_subRegion(_self, sample.screen);
45 
46  const vec3f dir = self->dir;
47  const vec3f org = self->pos_00 + screen.x * self->pos_du + screen.y * self->pos_dv;
48 
49  float nearClip = self->super.nearClip;
50  float farClip = inf;
51  const varying vec3f direction = normalize(dir);
52 
53  clipRay(self->clipPlanes, self->numClipPlanes, org, direction, nearClip, farClip);
54 
55  const float time = Camera_shutterTime(_self, sample.time);
56  ray.flags = RAY_FLAG_PRIMARY;
57  setRay(ray, org, dir, nearClip, farClip, time);
58 }

60 /*! create a new ::ispc-side version of a orthographiccamera - with given
61  pointer to the c-class version - and return both class pointer and
62  pointer to internal data back via the supplied reference
63  parameters */
64 export void* uniform OrthographicCamera_create(void* uniform cppE)
65 {
66  uniform OrthographicCamera* uniform cam = uniform new uniform OrthographicCamera;
67  cam->super.cppEquivalent = cppE;
68  cam->super.initRay = OrthographicCamera_initRay;
69  cam->super.doesDOF = false;
70  cam->numClipPlanes = 0;
71  return cam;
