24 void clipRay(const uniform vec4f* clipPlanes, const unsigned int numClipPlanes, const varying vec3f& position,
25  const varying vec3f& direction, varying float& near, varying float& far)
26 {
27  for (int i = 0; i < numClipPlanes; ++i)
28  {
29  const vec3f normal = {clipPlanes[i].x, clipPlanes[i].y, clipPlanes[i].z};
30  const float d = clipPlanes[i].w;
31 
32  const float distance = dot(normal, position) + d;
33  const float cosAngle = dot(direction, normal);
34 
35  if (cosAngle == 0.f)
36  {
37  // This ray is parallel to the plane, checking if position is
38  // on the visible side
39  if (distance < 0)
40  {
41  near = far = 0.0;
42  return;
43  }
44  }
45  else
46  {
47  const float t = -distance / cosAngle;
48  if (cosAngle > 0.f) // same direction plane
49  near = max(near, t);
50  else
51  far = min(far, t);
52  }
53  }
54 }
