46 void PerspectiveParallaxCamera_initRay(uniform Camera* uniform _self, varying Ray& ray,
47  const varying CameraSample& sample)
48 {
49  uniform PerspectiveParallaxCamera* uniform self = (uniform PerspectiveParallaxCamera * uniform) _self;
50 
51  vec2f screen = sample.screen;
52  screen = Camera_subRegion(_self, screen);
53  vec3f pixelPos;
54  vec3f org = make_vec3f(0.f);
55 
56  pixelPos.x = (screen.x - 0.5f) * self->imgPlane_size_x;
57  pixelPos.y = (screen.y - 0.5f) * self->imgPlane_size_y;
58  pixelPos.z = -self->distanceToPlane;
59  org.x = self->idpOffset;
60 
61  vec3f dir = normalize(pixelPos - org);
62 
63  const LinearSpace3f cameraSpace = make_LinearSpace3f(self->dir_du, self->dir_dv, self->dir_cam);
64  dir = normalize(cameraSpace * dir);
65  org = cameraSpace * org;
66 
67  const float time = Camera_shutterTime(_self, sample.time);
68  org = self->org + org;
69  ray.flags = RAY_FLAG_PRIMARY;
70  setRay(ray, org, normalize(dir), self->super.nearClip, inf, time);
71 }
