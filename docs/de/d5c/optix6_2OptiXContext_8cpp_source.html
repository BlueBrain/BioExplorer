65 template <typename T>
66 void textureToOptix(T* ptr_dst, const core::Texture2D& texture, const uint8_t face, const uint8_t mipLevel,
67  const bool hasAlpha)
68 {
69  uint16_t width = texture.width;
70  uint16_t height = texture.height;
71  for (uint8_t i = 0; i < mipLevel; ++i)
72  {
73  width /= 2;
74  height /= 2;
75  }
76  size_t idx_src = 0;
77  size_t idx_dst = 0;
78  const auto rawData = texture.getRawData<T>(face, mipLevel);
79  for (uint16_t y = 0; y < height; ++y)
80  {
81  for (uint16_t x = 0; x < width; ++x)
82  {
83  ptr_dst[idx_dst] = rawData[idx_src];
84  ptr_dst[idx_dst + 1u] = rawData[idx_src + 1u];
85  ptr_dst[idx_dst + 2u] = rawData[idx_src + 2u];
86  ptr_dst[idx_dst + 3u] = hasAlpha ? rawData[idx_src + 3u] : white<T>();
87  idx_dst += 4u;
88  idx_src += hasAlpha ? 4u : 3u;
89  }
90  }
91 }

105  return RT_WRAP_REPEAT;

190 ::optix::TextureSampler OptiXContext::createTextureSampler(Texture2DPtr texture)
191 {
192  uint16_t nx = texture->width;
193  uint16_t ny = texture->height;
194  const uint16_t channels = texture->channels;
195  const uint16_t optixChannels = 4;
196  const bool hasAlpha = optixChannels == channels;
197 
198  const bool useFloat = texture->depth == 4;
199  const bool useByte = texture->depth == 1;
200 
201  if (!useFloat && !useByte)
202  throw std::runtime_error("Only byte or float textures are supported");
203 
204  const bool createMipmaps = texture->getMipLevels() == 1 && useByte && !texture->isCubeMap();
205  uint16_t mipMapLevels = texture->getMipLevels();
206  if (createMipmaps)
207  mipMapLevels = texture->getPossibleMipMapsLevels();
208 
209  if (createMipmaps && !useByte)
210  throw std::runtime_error(
211  "Non 8-bits textures are not supported for automatic mipmaps "
212  "generation");
213 
214  RTformat optixFormat = useByte ? RT_FORMAT_UNSIGNED_BYTE4 : RT_FORMAT_FLOAT4;
215 
216  // Create texture sampler
217  ::optix::TextureSampler sampler = _optixContext->createTextureSampler();
218  const auto wrapMode = wrapModeToOptix(texture->getWrapMode());
219  sampler->setWrapMode(0, wrapMode);
220  sampler->setWrapMode(1, wrapMode);
221  sampler->setWrapMode(2, wrapMode);
222  sampler->setIndexingMode(RT_TEXTURE_INDEX_NORMALIZED_COORDINATES);
223  sampler->setReadMode(RT_TEXTURE_READ_NORMALIZED_FLOAT);
224  sampler->setMaxAnisotropy(8.0f);
225 
226  // Create buffer and populate with texture data
227  ::optix::Buffer buffer;
228  if (texture->isCubeMap())
229  buffer = _optixContext->createCubeBuffer(RT_BUFFER_INPUT, optixFormat, nx, ny, mipMapLevels);
230  else
231  buffer = _optixContext->createMipmappedBuffer(RT_BUFFER_INPUT, optixFormat, nx, ny, mipMapLevels);
232 
233  std::vector<void*> mipMapBuffers(mipMapLevels);
234  for (uint8_t currentLevel = 0u; currentLevel < mipMapLevels; ++currentLevel)
235  mipMapBuffers[currentLevel] = buffer->map(currentLevel);
236 
237  if (createMipmaps)
238  {
239  uint8_t* ptr_dst = (uint8_t*)mipMapBuffers[0];
240  size_t idx_src = 0;
241  size_t idx_dst = 0;
242  const auto rawData = texture->getRawData<unsigned char>();
243  for (uint16_t y = 0; y < ny; ++y)
244  {
245  for (uint16_t x = 0; x < nx; ++x)
246  {
247  ptr_dst[idx_dst] = rawData[idx_src];
248  ptr_dst[idx_dst + 1u] = rawData[idx_src + 1u];
249  ptr_dst[idx_dst + 2u] = rawData[idx_src + 2u];
250  ptr_dst[idx_dst + 3u] = hasAlpha ? rawData[idx_src + 3u] : 255u;
251  idx_dst += 4u;
252  idx_src += hasAlpha ? 4u : 3u;
253  }
254  }
255  ny /= 2u;
256  nx /= 2u;
257 
258  for (uint8_t currentLevel = 1u; currentLevel < mipMapLevels; ++currentLevel)
259  {
260  ptr_dst = (uint8_t*)mipMapBuffers[currentLevel];
261  uint8_t* ptr_src = (uint8_t*)mipMapBuffers[currentLevel - 1u];
262  for (uint16_t y = 0u; y < ny; ++y)
263  {
264  for (uint16_t x = 0u; x < nx; ++x)
265  {
266  ptr_dst[(y * nx + x) * 4u] =
267  (ptr_src[(y * 2u * nx + x) * 8u] + ptr_src[((y * 2u * nx + x) * 2u + 1u) * 4u] +
268  ptr_src[((y * 2u + 1u) * nx + x) * 8u] + ptr_src[(((y * 2u + 1u) * nx + x) * 2u + 1u) * 4u]) /
269  4.0f;
270  ptr_dst[(y * nx + x) * 4u + 1u] =
271  (ptr_src[(y * 2u * nx + x) * 8u + 1u] + ptr_src[((y * 2u * nx + x) * 2u + 1u) * 4u + 1u] +
272  ptr_src[((y * 2u + 1u) * nx + x) * 8u + 1u] +
273  ptr_src[(((y * 2u + 1u) * nx + x) * 2u + 1u) * 4u + 1u]) /
274  4.0f;
275  ptr_dst[(y * nx + x) * 4u + 2u] =
276  (ptr_src[(y * 2u * nx + x) * 8u + 2u] + ptr_src[((y * 2u * nx + x) * 2u + 1u) * 4u + 2u] +
277  ptr_src[((y * 2u + 1u) * nx + x) * 8u + 2u] +
278  ptr_src[(((y * 2u + 1u) * nx + x) * 2u + 1u) * 4u + 2u]) /
279  4.0f;
280  ptr_dst[(y * nx + x) * 4u + 3u] =
281  (ptr_src[(y * 2u * nx + x) * 8u + 3u] + ptr_src[((y * 2u * nx + x) * 2u + 1u) * 4u + 3u] +
282  ptr_src[((y * 2u + 1u) * nx + x) * 8u + 3u] +
283  ptr_src[(((y * 2u + 1u) * nx + x) * 2u + 1u) * 4u + 3u]) /
284  4.0f;
285 
286  if (texture->isNormalMap())
287  {
288  glm::vec3 normalized =
289  glm::normalize(glm::vec3(2.0f * (float)ptr_dst[(y * nx + x) * 4u] / 255.0f - 1.0f,
290  2.0f * (float)ptr_dst[(y * nx + x) * 4u + 1u] / 255.0f - 1.0f,
291  2.0f * (float)ptr_dst[(y * nx + x) * 4u + 2u] / 255.0f - 1.0f));
292  ptr_dst[(y * nx + x) * 4u] = 255.0f * (0.5f * normalized.x + 0.5f);
293  ptr_dst[(y * nx + x) * 4u + 1u] = 255.0f * (0.5f * normalized.y + 0.5f);
294  ptr_dst[(y * nx + x) * 4u + 2u] = 255.0f * (0.5f * normalized.z + 0.5f);
295  }
296  }
297  }
298  ny /= 2u;
299  nx /= 2u;
300  }
301  }
302  else
303  {
304  for (uint8_t face = 0; face < texture->getNumFaces(); ++face)
305  {
306  auto mipWidth = nx;
307  auto mipHeight = ny;
308  for (uint16_t mip = 0; mip < mipMapLevels; ++mip)
309  {
310  if (useByte)
311  {
312  auto dst = (uint8_t*)mipMapBuffers[mip];
313  dst += face * mipWidth * mipHeight * 4;
314  textureToOptix<uint8_t>(dst, *texture, face, mip, hasAlpha);
315  }
316  else if (useFloat)
317  {
318  auto dst = (float*)mipMapBuffers[mip];
319  dst += face * mipWidth * mipHeight * 4;
320  textureToOptix<float>(dst, *texture, face, mip, hasAlpha);
321  }
322  mipWidth /= 2;
323  mipHeight /= 2;
324  }
325  }
326  }
327 
328  for (uint8_t currentLevel = 0u; currentLevel < mipMapLevels; ++currentLevel)
329  buffer->unmap(currentLevel);
330 
331  // Assign buffer to sampler
332  sampler->setBuffer(buffer);
333  sampler->setFilteringModes(RT_FILTER_LINEAR, RT_FILTER_LINEAR,
334  mipMapLevels > 1 ? RT_FILTER_LINEAR : RT_FILTER_NONE);
335  sampler->validate();
336  return sampler;
337 }

403 void OptiXContext::_printSystemInformation() const
404 {
405  unsigned int optixVersion;
406  RT_CHECK_ERROR_NO_CONTEXT(rtGetVersion(&optixVersion));
407 
408  unsigned int major = optixVersion / 1000; // Check major with old formula.
409  unsigned int minor;
410  unsigned int micro;
411  if (3 < major) // New encoding since OptiX 4.0.0 to get two digits micro
412  // numbers?
413  {
414  major = optixVersion / 10000;
415  minor = (optixVersion % 10000) / 100;
416  micro = optixVersion % 100;
417  }
418  else // Old encoding with only one digit for the micro number.
419  {
420  minor = (optixVersion % 1000) / 10;
421  micro = optixVersion % 10;
422  }
423  PLUGIN_INFO("OptiX " << major << "." << minor << "." << micro);
424 
425  unsigned int numberOfDevices = 0;
426  RT_CHECK_ERROR_NO_CONTEXT(rtDeviceGetDeviceCount(&numberOfDevices));
427  PLUGIN_INFO("Number of Devices = " << numberOfDevices);
428 
429  for (unsigned int i = 0; i < numberOfDevices; ++i)
430  {
431  char name[256];
432  RT_CHECK_ERROR_NO_CONTEXT(rtDeviceGetAttribute(i, RT_DEVICE_ATTRIBUTE_NAME, sizeof(name), name));
433  PLUGIN_INFO("Device " << i << ": " << name);
434 
435  int computeCapability[2] = {0, 0};
436  RT_CHECK_ERROR_NO_CONTEXT(rtDeviceGetAttribute(i, RT_DEVICE_ATTRIBUTE_COMPUTE_CAPABILITY,
437  sizeof(computeCapability), &computeCapability));
438  PLUGIN_INFO(" Compute Support: " << computeCapability[0] << "." << computeCapability[1]);
439 
440  RTsize totalMemory = 0;
441  RT_CHECK_ERROR_NO_CONTEXT(
442  rtDeviceGetAttribute(i, RT_DEVICE_ATTRIBUTE_TOTAL_MEMORY, sizeof(totalMemory), &totalMemory));
443  PLUGIN_INFO(" Total Memory: " << (unsigned long long)(totalMemory / 1024 / 1024) << " MB");
444 
445  int clockRate = 0;
446  RT_CHECK_ERROR_NO_CONTEXT(
447  rtDeviceGetAttribute(i, RT_DEVICE_ATTRIBUTE_CLOCK_RATE, sizeof(clockRate), &clockRate));
448  PLUGIN_INFO(" Clock Rate: " << (clockRate / 1000) << " MHz");
449 
450  int maxThreadsPerBlock = 0;
451  RT_CHECK_ERROR_NO_CONTEXT(rtDeviceGetAttribute(i, RT_DEVICE_ATTRIBUTE_MAX_THREADS_PER_BLOCK,
452  sizeof(maxThreadsPerBlock), &maxThreadsPerBlock));
453  PLUGIN_INFO(" Max. Threads per Block: " << maxThreadsPerBlock);
454 
455  int smCount = 0;
456  RT_CHECK_ERROR_NO_CONTEXT(
457  rtDeviceGetAttribute(i, RT_DEVICE_ATTRIBUTE_MULTIPROCESSOR_COUNT, sizeof(smCount), &smCount));
458  PLUGIN_INFO(" Streaming Multiprocessor Count: " << smCount);
459 
460  int executionTimeoutEnabled = 0;
461  RT_CHECK_ERROR_NO_CONTEXT(rtDeviceGetAttribute(i, RT_DEVICE_ATTRIBUTE_EXECUTION_TIMEOUT_ENABLED,
462  sizeof(executionTimeoutEnabled), &executionTimeoutEnabled));
463  PLUGIN_INFO(" Execution Timeout Enabled: " << executionTimeoutEnabled);
464 
465  int maxHardwareTextureCount = 0;
466  RT_CHECK_ERROR_NO_CONTEXT(rtDeviceGetAttribute(i, RT_DEVICE_ATTRIBUTE_MAX_HARDWARE_TEXTURE_COUNT,
467  sizeof(maxHardwareTextureCount), &maxHardwareTextureCount));
468  PLUGIN_INFO(" Max. Hardware Texture Count: " << maxHardwareTextureCount);
469 
470  int tccDriver = 0;
471  RT_CHECK_ERROR_NO_CONTEXT(
472  rtDeviceGetAttribute(i, RT_DEVICE_ATTRIBUTE_TCC_DRIVER, sizeof(tccDriver), &tccDriver));
473  PLUGIN_INFO(" TCC Driver enabled: " << tccDriver);
474 
475  int cudaDeviceOrdinal = 0;
