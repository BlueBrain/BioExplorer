60 Transformation SphereShape::_getEmptySphereTransformation(
61  const uint64_t occurrence, const uint64_t nbOccurrences,
62  const MolecularSystemAnimationDetails& MolecularSystemAnimationDetails, const double offset) const
63 {
64  uint64_t rnd = occurrence;
65  if (nbOccurrences != 0 && MolecularSystemAnimationDetails.seed != 0)
66  {
67  if (GeneralSettings::getInstance()->getV1Compatibility())
68  rnd = rand() % nbOccurrences;
69  else
70  rnd = rand() % std::numeric_limits<uint64_t>::max();
71  }
72 
73  const double radius = _radius + (MolecularSystemAnimationDetails.positionSeed == 0
74  ? MolecularSystemAnimationDetails.positionStrength
75  : MolecularSystemAnimationDetails.positionStrength *
76  rnd3(MolecularSystemAnimationDetails.positionSeed + rnd));
77 
78  Vector3d pos;
79  Quaterniond rot;
80  sphereFilling(radius + offset, occurrence, nbOccurrences, rnd, pos, rot);
81 
82  if (isClipped(pos, _clippingPlanes))
83  throw std::runtime_error("Instance is clipped");
84 
85  if (MolecularSystemAnimationDetails.rotationSeed != 0)
86  rot = weightedRandomRotation(rot, MolecularSystemAnimationDetails.rotationSeed, rnd,
87  MolecularSystemAnimationDetails.rotationStrength);
88 
89  Transformation transformation;
90  transformation.setTranslation(pos);
91  transformation.setRotation(rot);
92  return transformation;
93 }

95 Transformation SphereShape::_getEmptySphereMorphedTransformation(
96  const uint64_t occurrence, const uint64_t nbOccurrences,
97  const MolecularSystemAnimationDetails& MolecularSystemAnimationDetails, const double offset) const
98 {
99  uint64_t rnd = occurrence;
100  if (nbOccurrences != 0 && MolecularSystemAnimationDetails.seed != 0)
101  rnd = rand() % nbOccurrences;
102 
103  const double radius = _radius + (MolecularSystemAnimationDetails.positionSeed == 0
104  ? MolecularSystemAnimationDetails.positionStrength
105  : MolecularSystemAnimationDetails.positionStrength *
106  rnd3(MolecularSystemAnimationDetails.positionSeed + rnd));
107 
108  Vector3d startPos;
109  Quaterniond startRot;
110  const Vector3d startDir = sphereFilling(radius, occurrence, nbOccurrences, rnd, startPos, startRot, offset);
111 
112  if (MolecularSystemAnimationDetails.rotationSeed != 0)
113  startRot = weightedRandomRotation(startRot, MolecularSystemAnimationDetails.rotationSeed, rnd,
114  MolecularSystemAnimationDetails.rotationStrength);
115 
116  const double endRadius = radius * 2.0;
117  const double morphingStep = MolecularSystemAnimationDetails.morphingStep;
118 
119  Vector3d endPos = startPos;
120  endPos.y = -radius;
121  endPos = endPos + (1.0 - (startPos.y + _radius) / endRadius) * Vector3d(endRadius, 0.0, endRadius) *
122  normalize(Vector3d(startDir.x, 0.0, startDir.z));
123 
124  Quaterniond endRot{0.0, 0.0, -0.707, 0.707};
125  if (MolecularSystemAnimationDetails.rotationSeed != 0)
126  endRot = weightedRandomRotation(endRot, MolecularSystemAnimationDetails.rotationSeed, rnd,
127  MolecularSystemAnimationDetails.rotationStrength);
128 
129  const Quaterniond finalRotation = slerp(startRot, endRot, morphingStep);
130 
131  const auto finalTranslation = endPos * morphingStep + startPos * (1.0 - morphingStep);
132  if (isClipped(finalTranslation, _clippingPlanes))
133  throw std::runtime_error("Instance is clipped");
134 
135  // Final transformation
136  Transformation transformation;
137  transformation.setTranslation(finalTranslation);
138  transformation.setRotation(finalRotation);
139  return transformation;
140 }

147 Transformation SphereShape::_getFilledSphereTransformation(
148  const uint64_t occurrence, const uint64_t nbOccurrences,
149  const MolecularSystemAnimationDetails& MolecularSystemAnimationDetails, const double offset) const
150 {
151  Vector3d pos;
152  const double diameter = _radius * 2.0;
153  do
154  {
155  pos = Vector3d(rnd1() * diameter, rnd1() * diameter, rnd1() * diameter);
156  } while (length(pos) > _radius);
157 
158  if (MolecularSystemAnimationDetails.positionSeed != 0)
159  {
160  const Vector3d posOffset = MolecularSystemAnimationDetails.positionStrength *
161  Vector3d(rnd2(occurrence + MolecularSystemAnimationDetails.positionSeed),
162  rnd2(occurrence + MolecularSystemAnimationDetails.positionSeed + 1),
163  rnd2(occurrence + MolecularSystemAnimationDetails.positionSeed + 2));
164 
165  pos += posOffset;
166  }
167  if (isClipped(pos, _clippingPlanes))
168  throw std::runtime_error("Instance is clipped");
169 
170  Quaterniond rot = safeQuatlookAt(normalize(pos));
171  if (MolecularSystemAnimationDetails.rotationSeed != 0)
172  rot = weightedRandomRotation(rot, MolecularSystemAnimationDetails.rotationSeed, occurrence,
173  MolecularSystemAnimationDetails.rotationStrength);
174 
175  Transformation transformation;
176  transformation.setTranslation(pos);
177  transformation.setRotation(rot);
178  return transformation;
179 }
