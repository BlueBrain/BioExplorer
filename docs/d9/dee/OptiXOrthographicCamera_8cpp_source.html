49 void OptiXOrthographicCamera::commit(const OptiXCamera& camera, ::optix::Context context)
50 {
51  const auto position = camera.getPosition();
52  const auto target = camera.getTarget();
53  const auto orientation = camera.getOrientation();
54 
55  const auto height = camera.getPropertyOrValue<double>(CAMERA_PROPERTY_HEIGHT.name.c_str(), DEFAULT_CAMERA_HEIGHT);
56  const auto aspect =
57  camera.getPropertyOrValue<double>(CAMERA_PROPERTY_ASPECT_RATIO.name.c_str(), DEFAULT_CAMERA_ASPECT_RATIO);
58 
59  const Vector3d dir = normalize(target - position);
60 
61  Vector3d u = normalize(cross(dir, orientation * UP_VECTOR));
62  Vector3d v = cross(u, dir);
63 
64  u *= height * aspect;
65  v *= height;
66 
67  context[CONTEXT_CAMERA_U]->setFloat(u.x, u.y, u.z);
68  context[CONTEXT_CAMERA_V]->setFloat(v.x, v.y, v.z);
69  context[CONTEXT_CAMERA_W]->setFloat(position.x, position.y, position.z);
70  context[CONTEXT_CAMERA_DIR]->setFloat(dir.x, dir.y, dir.z);
71 }
