38 class Service

41  Service(int argc, const char** argv)
42  : _renderingDone{_mainLoop->resource<uvw::AsyncHandle>()}
43  , _eventRendering{_mainLoop->resource<uvw::IdleHandle>()}
44  , _accumRendering{_mainLoop->resource<uvw::IdleHandle>()}

46  , _sigintHandle{_mainLoop->resource<uvw::SignalHandle>()}

48  , _stopRenderThread{_renderLoop->resource<uvw::AsyncHandle>()}
49  {
50  _checkIdleRendering->start();
51 
52  _setupMainThread();
53  _setupRenderThread();
54 
55  _core = std::make_unique<Core>(argc, argv);
56 
57  // events from rockets, trigger rendering
58  _core->getEngine().triggerRender = [&eventRendering = _eventRendering] { eventRendering->start(); };
59

62

68  void run()
69  {
70  // Start render & main loop
71  std::thread renderThread([&_renderLoop = _renderLoop] { _renderLoop->run(); });
72 #if 1
73  while (_core->getEngine().getKeepRunning())
74  {
75  _core->commit();
76  _mainLoop->run<uvw::Loop::Mode::NOWAIT>();

78 #else

86 private:
87  void _setupMainThread()
88  {
89  // triggered after rendering, send events to rockets from the main thread
90  _renderingDone->on<uvw::AsyncEvent>([&Core = _core](const auto&, auto&) { Core->postRender(); });
91 
92  // render or data load trigger from events
93  _eventRendering->on<uvw::IdleEvent>(
94  [&](const auto&, auto&)
95  {
96  _eventRendering->stop();
97  _accumRendering->stop();
98  _timeSinceLastEvent.start();
99 
100  // stop event loop(s) and exit application
101  if (!_core->getEngine().getKeepRunning())
102  {
103  this->_stopMainLoop();
104  return;
105  }
106 
107  // rendering
108  if (_core->commit())
109  _triggerRendering->send();
110  });
111 
112  // start accum rendering when we have no more other events
113  _checkIdleRendering->on<uvw::CheckEvent>([&accumRendering = _accumRendering](const auto&, auto&)
114  { accumRendering->start(); });
115 
116  // accumulation rendering on idle; re-triggered by _checkIdleRendering
117  _accumRendering->on<uvw::IdleEvent>(
118  [&](const auto&, auto&)
119  {
120  if (_timeSinceLastEvent.elapsed() < _idleRenderingDelay)
121  return;
122 
123  if (_core->getEngine().continueRendering() && _core->commit())
124  _triggerRendering->send();
125 
126  _accumRendering->stop();
127  });
128  }
129 
130  void _setupRenderThread()
131  {
132  // rendering, triggered from main thread
133  _triggerRendering->on<uvw::AsyncEvent>(
134  [&](const auto&, auto&)
135  {
136  _core->render();

138 
139  if (_core->getParametersManager().getApplicationParameters().isBenchmarking())

141  });
142 
143  // stop render loop, triggered from main thread
144  _stopRenderThread->once<uvw::AsyncEvent>([&](const auto&, auto&) { this->_stopRenderLoop(); });
145  }

148  {
