37 SomaSimulationHandler::SomaSimulationHandler(const std::string& populationName, const uint64_t simulationReportId)

71 void* SomaSimulationHandler::getFrameData(const uint32_t frame)
72 {
73  const auto& connector = DBConnector::getInstance();
74  const auto boundedFrame = _getBoundedFrame(frame);
75  if (_currentFrame != boundedFrame)
76  {
77  _currentFrame = boundedFrame;
78  if (_simulationReport.debugMode)
79  {
80  for (uint64_t i = 0; i < _frameData.size(); ++i)
81  _frameData[i] = -70.f + 100 * cos((frame + i) * M_PI / 180.f);
82  }
83  else
84  connector.getNeuronSomaReportValues(_populationName, _simulationReportId, _currentFrame, _frameData);
85  _frameSize = _frameData.size();
86  }
87 
88  return _frameData.data();
89 }
