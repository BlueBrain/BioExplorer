78 bool OptiXScene::commitLights()
79 {
80  if (!_lightManager.isModified())
81  return false;
82 
83  if (_lightManager.getLights().empty())
84  {
85  CORE_ERROR("No lights are currently defined");
86  return false;
87  }
88 
89  _optixLights.clear();
90 
91  for (const auto& kv : _lightManager.getLights())
92  {
93  auto baseLight = kv.second;
94 
95  switch (baseLight->_type)
96  {
97  case LightType::SPHERE:
98  {
99  const auto light = static_cast<SphereLight*>(baseLight.get());
100  const Vector3f position = light->_position;
101  const Vector3f color = light->_color;
102  BasicLight optixLight = {{position.x, position.y, position.z},
103  {color.x, color.y, color.z},
104  1, // Casts shadowIntensity
105  BASIC_LIGHT_TYPE_POINT};
106  _optixLights.push_back(optixLight);
107  break;
108  }
109  case LightType::DIRECTIONAL:
110  {
111  const auto light = static_cast<DirectionalLight*>(baseLight.get());
112  const Vector3f direction = light->_direction;
113  const Vector3f color = light->_color;
114  BasicLight optixLight = {{direction.x, direction.y, direction.z},
115  {color.x, color.y, color.z},
116  1, // Casts shadowIntensity
117  BASIC_LIGHT_TYPE_DIRECTIONAL};
118  _optixLights.push_back(optixLight);
119  break;
120  }
121  default:
122  {
123  CORE_WARN("Unsupported light type");
124  break;
125  }
126  }
127  }
128 
129  if (_lightBuffer)
130  _lightBuffer->destroy();
131 
132  auto context = OptiXContext::get().getOptixContext();
133  _lightBuffer = context->createBuffer(RT_BUFFER_INPUT, RT_FORMAT_USER, _optixLights.size());
134  _lightBuffer->setElementSize(sizeof(BasicLight));
135  memcpy(_lightBuffer->map(), _optixLights.data(), _optixLights.size() * sizeof(_optixLights[0]));
136  _lightBuffer->unmap();
137  context[CONTEXT_LIGHTS]->set(_lightBuffer);
138 
139  return true;
140 }

178 void OptiXScene::_commitClippingPlanes()
179 {
180  if (!isModified())
181  return;
182 
183  auto context = OptiXContext::get().getOptixContext();
184 
185  const size_t numClipPlanes = _clipPlanes.size();
186  const auto size = numClipPlanes * sizeof(Vector4f);
187 
188  RT_DESTROY(_clipPlanesBuffer);
189  _clipPlanesBuffer = context->createBuffer(RT_BUFFER_INPUT, RT_FORMAT_FLOAT4, numClipPlanes);
190 
191  if (numClipPlanes > 0)
192  {
193  Vector4fs buffer;
194  buffer.reserve(numClipPlanes);
195  for (const auto clipPlane : _clipPlanes)
196  {
197  const auto& p = clipPlane->getPlane();
198  buffer.push_back({static_cast<float>(p[0]), static_cast<float>(p[1]), static_cast<float>(p[2]),
199  static_cast<float>(p[3])});
200  }
201 
202  memcpy(_clipPlanesBuffer->map(), buffer.data(), size);
203  _clipPlanesBuffer->unmap();
204  }
205 
206  context[CONTEXT_CLIPPING_PLANES]->setBuffer(_clipPlanesBuffer);
207  context[CONTEXT_NB_CLIPPING_PLANES]->setUint(numClipPlanes);
208 }

210 void OptiXScene::commit()
211 {
212  // Always upload transfer function and simulation data if changed
213  for (size_t i = 0; i < _modelDescriptors.size(); ++i)
214  {
215  auto& model = _modelDescriptors[i]->getModel();
216  model.commitSimulationData();
217  if (model.commitTransferFunction())
218  markModified();
219  _commitVolumeParameters();
220  }
221 
222  commitLights();
223  _commitGeometryParameters();
224  _commitClippingPlanes();
225 
226  if (!isModified())
227  return;
228 
229  // Remove all models marked for removal
230  for (auto& model : _modelDescriptors)
231  if (model->isMarkedForRemoval())
232  model->callOnRemoved();
233 
234  _modelDescriptors.erase(std::remove_if(_modelDescriptors.begin(), _modelDescriptors.end(),
235  [](const auto& m) { return m->isMarkedForRemoval(); }),
236  _modelDescriptors.end());
237 
238  auto context = OptiXContext::get().getOptixContext();
239 
240  auto values = std::map<TextureType, std::string>{{TextureType::diffuse, "envmap"},
241  {TextureType::radiance, "envmap_radiance"},
242  {TextureType::irradiance, "envmap_irradiance"},
243  {TextureType::brdf_lut, "envmap_brdf_lut"}};
244  if (hasEnvironmentMap())
245  _backgroundMaterial->commit();
246 
247  auto optixMat = std::static_pointer_cast<OptiXMaterial>(_backgroundMaterial);
248  for (const auto& i : values)
249  {
250  auto sampler = _dummyTextureSampler;
251  if (hasEnvironmentMap() && optixMat->hasTexture(i.first))
252  sampler = optixMat->getTextureSampler(i.first);
253  context[i.second]->setInt(sampler->getId());
254  if (i.first == TextureType::radiance && _backgroundMaterial->hasTexture(TextureType::radiance))
255  {
256  const auto& radianceTex = _backgroundMaterial->getTexture(TextureType::radiance);
257  context[CONTEXT_MATERIAL_RADIANCE_LODS]->setUint(radianceTex->getMipLevels() - 1);
258  }
259  }
260 
261  context[CONTEXT_USE_ENVIRONMENT_MAP]->setUint(hasEnvironmentMap() ? 1 : 0);
262 
263  // Geometry
264  if (_rootGroup)
265  _rootGroup->destroy();
266 
267  _rootGroup = OptiXContext::get().createGroup();
268 
269  for (size_t i = 0; i < _modelDescriptors.size(); ++i)
270  {
271  auto& modelDescriptor = _modelDescriptors[i];
272  if (!modelDescriptor->getEnabled())
273  continue;
274 
275  auto& impl = static_cast<OptiXModel&>(modelDescriptor->getModel());
276 
277  CORE_DEBUG("Committing " << modelDescriptor->getName());
278 
279  impl.commitGeometry();
280  impl.logInformation();
281 
282  if (modelDescriptor->getVisible())
283  {
284  const auto geometryGroup = impl.getGeometryGroup();
285  const auto& instances = modelDescriptor->getInstances();
286  size_t count{0};
287  for (const auto& instance : instances)
288  {
289  auto modelTransformation = instance.getTransformation();
290  if (count == 0)
291  modelTransformation = modelDescriptor->getTransformation();
292  const ::glm::mat4 matrix = modelTransformation.toMatrix(true);
293  ::optix::Matrix4x4 optixMatrix(glm::value_ptr(matrix));
294  ::optix::Transform instanceTransformation = context->createTransform();
295  instanceTransformation->setChild(geometryGroup);
296  instanceTransformation->setMatrix(true, optixMatrix.getData(), optixMatrix.inverse().getData());
297  _rootGroup->addChild(instanceTransformation);
298  ++count;
299  }
300  CORE_DEBUG("Group has " << geometryGroup->getChildCount() << " children");
301  }
302 
303  if (modelDescriptor->getBoundingBox())
304  {
305  // scale and move the unit-sized bounding box geometry to the model size/scale first, then apply the
306  // instance transform
307  const auto boundingBoxGroup = impl.getBoundingBoxGroup();
308  ::optix::Transform transformation = context->createTransform();
309 
310  const auto& modelBounds = modelDescriptor->getModel().getBounds();
311  Transformation modelTransformation;
312  modelTransformation.setTranslation(modelBounds.getCenter() / modelBounds.getSize() - Vector3d(0.5));
313  modelTransformation.setScale(modelBounds.getSize());
314 
315  Matrix4f modelMatrix = modelTransformation.toMatrix(true);
316  modelMatrix = glm::transpose(modelMatrix);
317  const auto trf = glm::value_ptr(modelMatrix);
318 
319  transformation->setMatrix(false, trf, 0);
320  transformation->setChild(boundingBoxGroup);
321  _rootGroup->addChild(transformation);
322  }
323  }
324  computeBounds();
325 
326  CORE_DEBUG("Root has " << _rootGroup->getChildCount() << " children");
327 
328  context[CONTEXT_SCENE_TOP_OBJECT]->set(_rootGroup);
329  context[CONTEXT_SCENE_TOP_SHADOWER]->set(_rootGroup);
330 
331  // TODO: triggers the change callback to re-broadcast the scene if the clip planes have changed. Provide an RPC
332  // to update/set clip planes.
333  markModified();
334 }
