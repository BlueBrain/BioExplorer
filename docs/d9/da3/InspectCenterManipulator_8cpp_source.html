66 void InspectCenterManipulator::dragLeft(const Vector2i& to, const Vector2i& from)
67 {
68  const float du = (to.x - from.x) * DEFAULT_MOUSE_MOTION_SPEED_MULTIPLIER * getRotationSpeed();
69  const float dv = (to.y - from.y) * DEFAULT_MOUSE_MOTION_SPEED_MULTIPLIER * getRotationSpeed();
70  rotate(_camera.getTarget(), du, dv, AxisMode::localY);
71 }

73 void InspectCenterManipulator::dragRight(const Vector2i& to, const Vector2i& from)
74 {
75  const float distance = -(to.y - from.y) * DEFAULT_MOUSE_MOTION_SPEED_MULTIPLIER * getMotionSpeed();
76  if (distance < glm::length(_camera.getTarget() - _camera.getPosition()))
77  translate(Vector3f(0, 0, -1) * distance);
78 }

80 void InspectCenterManipulator::dragMiddle(const Vector2i& to, const Vector2i& from)
81 {
82  const float x = (to.x - from.x) * DEFAULT_MOUSE_MOTION_SPEED_MULTIPLIER * getMotionSpeed();
83  const float y = (to.y - from.y) * DEFAULT_MOUSE_MOTION_SPEED_MULTIPLIER * getMotionSpeed();
84  const Vector3d translation(-x, y, 0);
85  translate(translation);
86  _camera.setTarget(_camera.getTarget() + glm::rotate(_camera.getOrientation(), translation));
87 }
