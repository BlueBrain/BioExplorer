33 // https://en.wikibooks.org/wiki/Algorithm_Implementation/Strings/Levenshtein_distance#C++
34 unsigned int levenshtein_distance(const std::string& s1, const std::string& s2)
35 {
36  const std::size_t len1 = s1.size(), len2 = s2.size();
37  std::vector<unsigned int> col(len2 + 1), prevCol(len2 + 1);
38 
39  for (unsigned int i = 0; i < prevCol.size(); i++)
40  prevCol[i] = i;
41  for (unsigned int i = 0; i < len1; i++)
42  {
43  col[0] = i + 1;
44  for (unsigned int j = 0; j < len2; j++)
45  col[j + 1] = std::min({prevCol[1 + j] + 1, col[j] + 1, prevCol[j] + (s1[i] == s2[j] ? 0 : 1)});
46  col.swap(prevCol);
47  }
48  return prevCol[len2];
49 }

51 std::vector<std::string> findSimilarOptions(const std::string& name, const std::vector<std::string>& options)
52 {
53  constexpr size_t MAX_SUGGESTIONS = 7;
54 
55  std::vector<std::string> subStringOptions;
56  std::vector<std::string> levenshteinOptions;
57 
58  // Collect substring options
59  {
60  // Strip dashes
61  auto nameStrip = name;
62  nameStrip.erase(std::remove(nameStrip.begin(), nameStrip.end(), '-'), nameStrip.end());
63 
64  // Suggest options containing the substring
65  for (const auto& optionName : options)
66  {
67  if (subStringOptions.size() >= MAX_SUGGESTIONS)
68  break;
69 
70  if (optionName.find(nameStrip) != std::string::npos)
71  subStringOptions.push_back(optionName);
72  }
73  }
74 
75  // Collect best levenshtein distance options
76  {
77  size_t bestDist = UINT_MAX;
78 
79  for (const auto& optionName : options)
80  {
81  if (levenshteinOptions.size() >= MAX_SUGGESTIONS)
82  break;
83 
84  const auto dist = levenshtein_distance(name, optionName);
85  if (dist < bestDist)
86  {
87  levenshteinOptions.clear();
88  bestDist = dist;
89  }
90 
91  if (dist == bestDist &&
92  std::find(subStringOptions.begin(), subStringOptions.end(), optionName) == subStringOptions.end())
93  levenshteinOptions.push_back(optionName);
94  }
95  }
96 
97  // Merge suggestions giving precedence to substrings
98  auto output = subStringOptions;
99  for (const auto& option : levenshteinOptions)
100  output.push_back(option);
101  output.resize(std::min(output.size(), MAX_SUGGESTIONS));
102 
103  return output;
104 }

133 void ParametersManager::_parse(int argc, const char** argv)
134 {
135  try
136  {
137  po::options_description generalOptions("General options");
138  generalOptions.add_options()("help", "Print this help")("version",
139  "Print the Core version")("verbose",
140  "Print parsed parameters");
141 
142  _allOptions.add(generalOptions);
143 
144  po::variables_map vm;
145  po::parsed_options parsedOptions =
146  po::command_line_parser(argc, argv)
147  .options(_allOptions)
148  .allow_unregistered()
149  .positional(_applicationParameters.posArgs())
150  .style(po::command_line_style::unix_style & ~po::command_line_style::allow_short &
151  ~po::command_line_style::allow_guessing)
152  .run();
153 
154  const auto unrecognizedOptions = po::collect_unrecognized(parsedOptions.options, po::exclude_positional);
155 
156  _processUnrecognizedOptions(unrecognizedOptions);
157 
158  po::store(parsedOptions, vm);
159  po::notify(vm);
160 
161  if (vm.count("help"))
162  {
163  usage();
164  exit(EXIT_SUCCESS);
165  }
166 
167  _printVersion();
168  if (vm.count("version"))
169  exit(EXIT_SUCCESS);
170 
171  for (auto parameters : _parameterSets)
172  parameters->parse(vm);
173 
174  if (vm.count("verbose"))
175  print();
176  }
177  catch (const po::error& e)
178  {
179  CORE_ERROR(e.what());
180  exit(EXIT_FAILURE);
181  }
182 }

261 void ParametersManager::_processUnrecognizedOptions(const std::vector<std::string>& unrecognizedOptions) const
262 {
263  if (unrecognizedOptions.empty())
264  return;
265 
266  const auto& unrecognized = unrecognizedOptions.front();
267 
268  std::vector<std::string> availableOptions;
269  for (auto option : _allOptions.options())
270  availableOptions.push_back(option->format_name());
271 
272  const auto suggestions = findSimilarOptions(unrecognized, availableOptions);
273 
274  std::string errorMessage = "Unrecognized option '" + unrecognized + "'.\n\nMost similar options are:";
275 
276  for (const auto& suggestion : suggestions)
277  errorMessage += "\n\t" + suggestion;
278 
279  throw po::error(errorMessage);
280 }
