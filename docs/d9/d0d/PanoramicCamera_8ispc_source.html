50 void PanoramicCamera_initRay(uniform Camera* uniform _self, varying Ray& ray, const varying CameraSample& sample)
51 {
52  uniform PanoramicCamera* uniform self = (uniform PanoramicCamera * uniform) _self;
53 
54  vec2f screen = sample.screen;
55  const vec3f org = self->org;
56 
57  screen = Camera_subRegion(_self, screen);
58 
59  const float phi = self->half ? M_PI / 2.f - two_pi * screen.x * 0.5f : two_pi * screen.x;
60  const float theta = M_PI * screen.y;
61 
62  float sinTheta, cosTheta;
63  sincos(theta, &sinTheta, &cosTheta);
64  const vec3f localDir = cartesian(phi, sinTheta, cosTheta);
65 
66  // trnasform to camera- and then to world-space
67  vec3f dir = make_vec3f(-localDir.y, -localDir.z, localDir.x);
68 
69  const LinearSpace3f cameraSpace = make_LinearSpace3f(self->dirU, self->dirV, self->dirCam);
70 
71  dir = normalize(cameraSpace * dir);
72 
73  float nearClip = self->super.nearClip;
74  float farClip = inf;
75 
76  clipRay(self->clipPlanes, self->numClipPlanes, org, dir, nearClip, farClip);
77 
78  const float time = Camera_shutterTime(_self, sample.time);
79  ray.flags = RAY_FLAG_PRIMARY;
80  setRay(ray, org, dir, nearClip, farClip, time);
81 }

83 /*! create a new ::ispc-side version of a PerspectiveCamera - with given
84  pointer to the c-class version - and return both class pointer and
85  pointer to internal data back via the supplied reference
86  parameters */
87 export void* uniform PanoramicCamera_create(void* uniform cppE)
88 {
89  uniform PanoramicCamera* uniform self = uniform new uniform PanoramicCamera;
90  self->super.cppEquivalent = cppE;
91  self->super.initRay = PanoramicCamera_initRay;
92  self->super.doesDOF = false;
93  self->numClipPlanes = 0;
