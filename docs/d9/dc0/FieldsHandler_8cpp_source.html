60 void FieldsHandler::_buildOctree(const Scene& scene, const float voxelSize,
61  const float density)
62 {
63  PLUGIN_INFO("Building Octree");
64 
65  if (density > 1.f || density <= 0.f)
66  PLUGIN_THROW("Density should be higher > 0 and <= 1");
67  const size_t densityRatio = 1.f / density;
68 
69  const auto clipPlanes = getClippingPlanes(scene);
70 
71  floats events;
72  uint16_t count;
73 
74  const auto& modelDescriptors = scene.getModelDescriptors();
75  for (const auto modelDescriptor : modelDescriptors)
76  {
77  const auto& instances = modelDescriptor->getInstances();
78  for (const auto& instance : instances)
79  {
80  const auto& tf = instance.getTransformation();
81  const auto& model = modelDescriptor->getModel();
82  const auto& spheresMap = model.getSpheres();
83  for (const auto& spheres : spheresMap)
84  {
85  for (const auto& sphere : spheres.second)
86  {
87  const Vector3d center =
88  tf.getTranslation() +
89  tf.getRotation() *
90  (Vector3d(sphere.center) - tf.getRotationCenter());
91 
92  const Vector3f c = center;
93  if (isClipped(c, clipPlanes))
94  continue;
95 
96  if (count % densityRatio == 0)
97  {
98  events.push_back(c.x);
99  events.push_back(c.y);
100  events.push_back(c.z);
101  events.push_back(sphere.radius);
102  events.push_back(sphere.radius);
103  }
104  ++count;
105  }
106  }
107  }
108  }
109 
110  // Determine model bounding box
111  glm::vec3 minAABB(std::numeric_limits<float>::max(),
112  std::numeric_limits<float>::max(),
113  std::numeric_limits<float>::max());
114  glm::vec3 maxAABB(-std::numeric_limits<float>::max(),
115  -std::numeric_limits<float>::max(),
116  -std::numeric_limits<float>::max());
117  for (uint64_t i = 0; i < events.size(); i += 5)
118  {
119  if (events[i + 4] != 0.f)
120  {
121  minAABB = {std::min(minAABB.x, events[i]),
122  std::min(minAABB.y, events[i + 1]),
123  std::min(minAABB.z, events[i + 2])};
124  maxAABB = {std::max(maxAABB.x, events[i]),
125  std::max(maxAABB.y, events[i + 1]),
126  std::max(maxAABB.z, events[i + 2])};
127  }
128  }
129 
130  // Compute volume information
131  const glm::vec3 sceneSize = maxAABB - minAABB;
132 
133  // Double AABB size
134  glm::vec3 center = (minAABB + maxAABB) / 2.f;
135  minAABB = center - sceneSize * 0.5f;
136  maxAABB = center + sceneSize * 0.5f;
137  _offset = minAABB;
138 
139  // Build acceleration structure
140  Octree morphoOctree(events, voxelSize, minAABB, maxAABB);
141  uint64_t volumeSize = morphoOctree.getVolumeSize();
142  _offset = minAABB;
143  _dimensions = morphoOctree.getVolumeDim();
144  _spacing = sceneSize / glm::vec3(_dimensions);
145 
146  PLUGIN_INFO("--------------------------------------------");
147  PLUGIN_INFO("Octree information");
148  PLUGIN_INFO("--------------------------------------------");
149  PLUGIN_INFO("Scene AABB : ["
150  << minAABB.x << "," << minAABB.y << "," << minAABB.z << "] ["
151  << maxAABB.x << "," << maxAABB.y << "," << maxAABB.z << "]");
152  PLUGIN_INFO("Scene dimension : [" << sceneSize.x << "," << sceneSize.y
153  << "," << sceneSize.z << "]");
154  PLUGIN_INFO("Element spacing : [" << _spacing.x << ", " << _spacing.y
155  << ", " << _spacing.z << "] ");
156  PLUGIN_INFO("Volume dimensions : ["
157  << _dimensions.x << ", " << _dimensions.y << ", "
158  << _dimensions.z << "] = " << volumeSize << " bytes");
159  PLUGIN_INFO("Element offset : [" << _offset.x << ", " << _offset.y
160  << ", " << _offset.z << "] ");
161 
162  const auto& indices = morphoOctree.getFlatIndexes();
163  PLUGIN_INFO("Indices size : " << indices.size());
164  const auto& data = morphoOctree.getFlatData();
165  _frameData.push_back(_offset.x);
166  _frameData.push_back(_offset.y);
167  _frameData.push_back(_offset.z);
168  _frameData.push_back(_spacing.x);
169  _frameData.push_back(_spacing.y);
170  _frameData.push_back(_spacing.z);
171  _frameData.push_back(_dimensions.x);
172  _frameData.push_back(_dimensions.y);
173  _frameData.push_back(_dimensions.z);
174  _frameData.push_back(morphoOctree.getOctreeSize());
175  _frameData.push_back(indices.size());
176  _frameData.insert(_frameData.end(), indices.begin(), indices.end());
177  _frameData.insert(_frameData.end(), data.begin(), data.end());
178  _frameSize = _frameData.size();
179  PLUGIN_INFO("Data size : " << _frameSize);
180  PLUGIN_INFO("--------------------------------------------");
181 }

198 const void FieldsHandler::exportToFile(const std::string& filename) const
199 {
200  PLUGIN_INFO("Saving octree to file: " << filename);
201  std::ofstream file(filename, std::ios::out | std::ios::binary);
202  if (!file.good())
203  PLUGIN_THROW("Could not export octree to " + filename);
204 
205  file.write((char*)&_frameSize, sizeof(uint64_t));
206  file.write((char*)_frameData.data(), _frameData.size() * sizeof(float));
207 
208  file.close();
209 }

211 void FieldsHandler::importFromFile(const std::string& filename)
212 {
213  PLUGIN_INFO("Loading octree from file: " << filename);
214  std::ifstream file(filename, std::ios::in | std::ios::binary);
215  if (!file.good())
216  PLUGIN_THROW("Could not import octree from " + filename);
217 
218  file.read((char*)&_frameSize, sizeof(uint64_t));
219  _frameData.resize(_frameSize);
220  file.read((char*)_frameData.data(), _frameData.size() * sizeof(float));
221 
222  _offset = {_frameData[0], _frameData[1], _frameData[2]};
223  _spacing = {_frameData[3], _frameData[4], _frameData[5]};
224  _dimensions = {_frameData[6], _frameData[7], _frameData[8]};
225 
226  PLUGIN_INFO("Octree: dimensions=" << _dimensions << ", offset=" << _offset
227  << ", spacing=" << _spacing);
228 
229  file.close();
230 }
