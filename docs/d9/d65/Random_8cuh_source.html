65 static __host__ __inline__ unsigned int mwc()
66 {
67  static unsigned long long r[4];
68  static unsigned long long carry;
69  static bool init = false;
70  if (!init)
71  {
72  init = true;
73  unsigned int seed = 7654321u, seed0, seed1, seed2, seed3;
74  r[0] = seed0 = lcg2(seed);
75  r[1] = seed1 = lcg2(seed0);
76  r[2] = seed2 = lcg2(seed1);
77  r[3] = seed3 = lcg2(seed2);
78  carry = lcg2(seed3);
79  }
80 
81  unsigned long long sum = 2111111111ull * r[3] + 1492ull * r[2] +
82  1776ull * r[1] + 5115ull * r[0] + 1ull * carry;
83  r[3] = r[2];
84  r[2] = r[1];
