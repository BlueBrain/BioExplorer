40 MeshBasedMembrane::MeshBasedMembrane(Scene& scene,
41  const MeshBasedMembraneDetails& details)
42  : Node()
43 {
44  const auto loader = MeshLoader(scene);
45 
46  // Load protein
47  const Vector3f position = {details.position[0], details.position[1],
48  details.position[2]};
49  const Quaterniond rotation = {details.rotation[0], details.rotation[1],
50  details.rotation[2], details.rotation[3]};
51  const Vector3f scale = {details.scale[0], details.scale[1],
52  details.scale[2]};
53 
54  ProteinDetails pd;
55  pd.assemblyName = details.assemblyName;
56  pd.name = details.name;
57  pd.contents = details.proteinContents;
58  pd.recenter = true;
59  pd.atomRadiusMultiplier = details.atomRadiusMultiplier;
60  pd.representation = details.representation;
61  pd.position = details.position;
62  pd.rotation = details.rotation;
63 
64  // Random seed
65  srand(details.randomSeed);
66 
67  // Create model
68  _protein = ProteinPtr(new Protein(scene, pd));
69  _modelDescriptor = _protein->getModelDescriptor();
70  auto model = &_modelDescriptor->getModel();
71  model->updateBounds();
72  const auto proteinSize = model->getBounds().getSize();
73 
74  // Clipping planes
75  const auto clipPlanes = getClippingPlanes(scene);
76 
77  // Load MeshBasedMembrane
78  Assimp::Importer importer;
79  const aiScene* aiScene =
80  importer.ReadFileFromMemory(details.meshContents.c_str(),
81  details.meshContents.length(),
82  aiProcess_GenSmoothNormals |
83  aiProcess_Triangulate);
84 
85  if (!aiScene)
86  PLUGIN_THROW(importer.GetErrorString());
87 
88  if (!aiScene->HasMeshes())
89  PLUGIN_THROW("No MeshBasedMembranees found");
90 
91  const auto trfm = aiScene->mRootNode->mTransformation;
92  const Matrix4f matrix{trfm.a1, trfm.b1, trfm.c1, trfm.d1, trfm.a2, trfm.b2,
93  trfm.c2, trfm.d2, trfm.a3, trfm.b3, trfm.c3, trfm.d3,
94  trfm.a4, trfm.b4, trfm.c4, trfm.d4};
95 
96  // Add protein instances according to MeshBasedMembrane topology
97  size_t instanceCount = 0;
98  float meshCoveringProgress = 0.f;
99  float instanceCoveringProgress = 0.f;
100  for (size_t m = 0; m < aiScene->mNumMeshes; ++m)
101  {
102  const auto& mesh = aiScene->mMeshes[m];
103 
104  // MeshBasedMembrane scaling
105  Vector3f meshCenter{0.f, 0.f, 0.f};
106  for (size_t i = 0; i < mesh->mNumVertices; ++i)
107  {
108  const auto& v = mesh->mVertices[i];
109  meshCenter += _toVector3f(v);
110  }
111  meshCenter /= mesh->mNumVertices;
112 
113  // Compute full MeshBasedMembrane area
114  std::vector<Vector3ui> faces;
115  for (size_t f = 0; f < mesh->mNumFaces; ++f)
116  if (mesh->mFaces[f].mNumIndices == 3)
117  {
118  const auto i1 = mesh->mVertices[mesh->mFaces[f].mIndices[0]];
119  const auto v1 =
120  position +
121  Vector3f(matrix * Vector4f(_toVector3f(i1, meshCenter,
122  scale, rotation),
123  1.f));
124  const auto i2 = mesh->mVertices[mesh->mFaces[f].mIndices[1]];
125  const auto v2 =
126  position +
127  Vector3f(matrix * Vector4f(_toVector3f(i2, meshCenter,
128  scale, rotation),
129  1.f));
130 
131  const auto i3 = mesh->mVertices[mesh->mFaces[f].mIndices[2]];
132  const auto v3 =
133  position +
134  Vector3f(matrix * Vector4f(_toVector3f(i3, meshCenter,
135  scale, rotation),
136  1.f));
137 
138  if (!isClipped(v1, clipPlanes) || !isClipped(v2, clipPlanes) ||
139  !isClipped(v3, clipPlanes))
140  faces.push_back(Vector3ui(mesh->mFaces[f].mIndices[0],
141  mesh->mFaces[f].mIndices[1],
142  mesh->mFaces[f].mIndices[2]));
143  }
144 
145  float meshSurface = 0.f;
146  for (const auto& face : faces)
147  meshSurface +=
148  _getSurfaceArea(_toVector3f(mesh->mVertices[face.x], meshCenter,
149  scale, rotation),
150  _toVector3f(mesh->mVertices[face.y], meshCenter,
151  scale, rotation),
152  _toVector3f(mesh->mVertices[face.z], meshCenter,
153  scale, rotation));
154 
155  const float proteinSurface = proteinSize.x * proteinSize.x;
156 
157  // Total number of instance needed to fill the MeshBasedMembrane surface
158  const size_t nbInstances =
159  details.density * meshSurface / proteinSurface;
160  const float instanceSurface = meshSurface / nbInstances;
161 
162  PLUGIN_INFO("----=== MeshBasedMembrane ===----");
163  PLUGIN_INFO("Position : " << position);
164  PLUGIN_INFO("rotation : " << rotation);
165  PLUGIN_INFO("Scale : " << scale);
166  PLUGIN_INFO("Number of faces : " << faces.size());
167  PLUGIN_INFO("Mesh surface area : " << meshSurface);
168  PLUGIN_INFO("Protein size : " << proteinSize);
169  PLUGIN_INFO("Protein surface area : " << proteinSurface);
170  PLUGIN_INFO("Instance surface area: " << instanceSurface);
171  PLUGIN_INFO("Number of instances : " << nbInstances);
172 
173 #pragma omp parallel for
174  for (const auto& face : faces)
175  {
176  const auto P0 = _toVector3f(mesh->mVertices[face.x], meshCenter,
177  scale, rotation);
178  const auto P1 = _toVector3f(mesh->mVertices[face.y], meshCenter,
179  scale, rotation);
180  const auto P2 = _toVector3f(mesh->mVertices[face.z], meshCenter,
181  scale, rotation);
182 
183  const auto V0 = P1 - P0;
184  const auto V1 = P2 - P0;
185 
186  const Vector3f defaultNormal = glm::cross(V0, V1);
187 
188  // Compute face surface
189  const float faceSurface = _getSurfaceArea(P0, P1, P2);
190 
191  // Estimate number of proteins for current face
192  meshCoveringProgress += faceSurface;
193  const size_t nbProteins =
194  size_t((meshCoveringProgress - instanceCoveringProgress) /
195  instanceSurface);
196 
197  // compute protein positions and rotations
198  for (size_t i = 0; i < nbProteins; ++i)
199  {
200  instanceCoveringProgress += instanceSurface;
201  Vector2f coordinates{1.f, 1.f};
202  while (coordinates.x + coordinates.y > 1.f)
203  {
204  coordinates.x = float(rand() % nbProteins) / nbProteins;
205  coordinates.y = float(rand() % nbProteins) / nbProteins;
206  }
207 
208  Transformation tf;
209  const Vector3f P = P0 + V0 * coordinates.x + V1 * coordinates.y;
210  const Vector3f transformedVertex =
211  matrix * Vector4f(P.x, P.y, P.z, 1.f);
212 
213  const float variableOffset = details.surfaceVariableOffset *
214  (rand() % 1000 / 1000.f - 0.5f);
215 
216  auto translation = position + transformedVertex +
217  defaultNormal * details.surfaceFixedOffset +
218  defaultNormal * variableOffset;
219 
220  if (mesh->HasNormals())
221  {
222  const auto v0 = P0 - P;
223  const auto v1 = P1 - P;
224  const auto v2 = P2 - P;
225 
226  const Vector3f areas{0.5f * length(glm::cross(v1, v2)),
227  0.5f * length(glm::cross(v0, v2)),
228  0.5f * length(glm::cross(v0, v1))};
229 
230  const auto N0 = _toVector3f(mesh->mNormals[face.x]);
231  const auto N1 = _toVector3f(mesh->mNormals[face.y]);
232  const auto N2 = _toVector3f(mesh->mNormals[face.z]);
233 
234  const Vector3f normal = glm::normalize(
235  matrix *
236  Vector4f(glm::normalize((N0 * areas.x + N1 * areas.y +
237  N2 * areas.z) /
238  (areas.x + areas.y + areas.z)),
239  1.f));
240 
241  if (normal != UP_VECTOR)
242  {
243  const Quaterniond rotation =
244  glm::quatLookAt(normal, UP_VECTOR);
245  tf.setRotation(rotation * rotation);
246  }
247  translation = position + transformedVertex +
248  normal * details.surfaceFixedOffset +
249  normal * variableOffset;
250  }
251 
252  if (isClipped(translation, clipPlanes))
253  continue;
254 
255  tf.setTranslation(translation);
256 
257  if (instanceCount == 0)
258  _modelDescriptor->setTransformation(tf);
259 
260  const ModelInstance instance(true, false, tf);
261 #pragma omp critical
262  _modelDescriptor->addInstance(instance);
263 
264  ++instanceCount;
265  }
266  }
267  }
268 }

270 float MeshBasedMembrane::_getSurfaceArea(const Vector3f& v0, const Vector3f& v1,
271  const Vector3f& v2) const
272 {
273  // Compute triangle area
274  const float a = length(v1 - v0);
275  const float b = length(v2 - v1);
276  const float c = length(v0 - v2);
277  const float p = (a + b + c) / 2.f;
278  const float e = p * (p - a) * (p - b) * (p - c);
279  if (e < 0)
280  return 0.f;
281 
282  return sqrt(e);
283 }

290 Vector3f MeshBasedMembrane::_toVector3f(const aiVector3D& v,
291  const Vector3f& center,
292  const Vector3f& scale,
293  const Quaterniond& rotation) const
294 {
295  const Vector3f p{v.x, v.y, v.z};
296  const Vector3f a = p - center;
297  const Vector3f b = Vector3f(rotation * Vector3d(p + a)) * scale;
298  return b;
299 }
