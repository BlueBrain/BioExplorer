52 void FishEyeCamera_initRay(uniform Camera* uniform _self, varying Ray& ray, const varying CameraSample& sample)
53 {
54  uniform FishEyeCamera* uniform self = (uniform FishEyeCamera * uniform) _self;
55 
56  vec2f screen = sample.screen;
57  vec3f org = self->org;
58 
59  screen = Camera_subRegion(_self, screen);
60 
61  float xp = screen.x * 2.f - 1.f;
62  xp = (xp == 0.f ? 0.0001f : xp);
63 
64  const float yp = screen.y * 2.f - 1.f;
65  const float radius = sqrt(xp * xp + yp * yp);
66  vec3f dir = make_vec3f(0.f);
67  if (radius > 1.05f) // Render 5% larger than required for safety reason :)
68  {
69  setRay(ray, org, dir, 0.f, 0.f, 0.f);
70  return;
71  }
72 
73  const float phi = (radius - 1.f) * PI * 0.5f;
74  const float theta = atan2(yp, xp);
75  const vec3f localDir = {cos(phi) * cos(theta), cos(phi) * sin(theta), sin(phi)};
76 
77  // transform to camera and then to world-space
78  dir = normalize(localDir);
79 
80  const LinearSpace3f cameraSpace = make_LinearSpace3f(self->dirU, self->dirV, self->dirCam);
81 
82  dir = normalize(cameraSpace * dir);
83 
84  if (true)
85  {
86  const float scaledAperture = 0.1f;
87  float random = getRandomValue(self->useHardwareRandomizer, make_vec3ui(screen.x, screen.y, 0.f));
88  random = (random == 0.f ? 0.001f : random);
89  const vec3f llp = uniformSampleDisk(random * self->aperture, sample.lens);
90  // transform local lens point to focal plane (dir_XX are prescaled in this case)
91  const vec3f lp = (llp.x * self->dirU) + (llp.y * self->dirV);
92  org = org + lp;
93  dir = dir - lp;
94  }
95 
96  float nearClip = self->super.nearClip;
97  float farClip = inf;
98 
99  clipRay(self->clipPlanes, self->numClipPlanes, org, dir, nearClip, farClip);
100 
101  const float time = Camera_shutterTime(_self, sample.time);
102  ray.flags = RAY_FLAG_PRIMARY;
103  setRay(ray, org, dir, nearClip, farClip, time);
104 }
