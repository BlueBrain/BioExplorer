33 Membrane::Membrane(Scene &scene, const MembraneDetails &descriptor,
34  const Vector3f &position, const Quaterniond &rotation,
35  const Vector4fs &clippingPlanes)
36  : _scene(scene)
37  , _position(position)
38  , _rotation(rotation)
39  , _details(descriptor)
40  , _clippingPlanes(clippingPlanes)
41 {
42  if (_details.representation == ProteinRepresentation::contour)
43  {
44  switch (_details.shape)
45  {
46  case AssemblyShape::spherical:
47  const std::string name{"Membrane"};
48  const size_t materialId = 0;
49  auto model = _scene.createModel();
50 
51  auto material = model->createMaterial(materialId, name);
52  brayns::PropertyMap props;
53  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
54  static_cast<int>(MaterialShadingMode::basic)});
55  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
56  props.setProperty(
57  {MATERIAL_PROPERTY_CHAMELEON_MODE,
58  static_cast<int>(
59  MaterialChameleonMode::undefined_chameleon_mode)});
60  material->updateProperties(props);
61 
62  model->addSphere(materialId,
63  {_position, _details.assemblyParams[0]});
64  auto modelDescriptor =
65  std::make_shared<ModelDescriptor>(std::move(model), name);
66  _scene.addModel(modelDescriptor);
67  return;
68  }
69  }
70 
71  std::vector<std::string> proteinContents;
72  proteinContents.push_back(descriptor.content1);
73  if (!descriptor.content2.empty())
74  proteinContents.push_back(descriptor.content2);
75  if (!descriptor.content3.empty())
76  proteinContents.push_back(descriptor.content3);
77  if (!descriptor.content4.empty())
78  proteinContents.push_back(descriptor.content4);
79 
80  // Load proteins
81  size_t i = 0;
82  for (const auto &content : proteinContents)
83  {
84  ProteinDetails pd;
85  pd.assemblyName = descriptor.assemblyName;
86  pd.name = _getElementNameFromId(i);
87  pd.contents = content;
88  pd.chainIds = descriptor.chainIds;
89  pd.recenter = descriptor.recenter;
90  pd.loadBonds = descriptor.loadBonds;
91  pd.randomSeed = descriptor.randomSeed;
92  pd.occurrences = 1;
93  pd.rotation = descriptor.rotation;
94  pd.assemblyParams = descriptor.assemblyParams;
95  pd.atomRadiusMultiplier = descriptor.atomRadiusMultiplier;
96  pd.representation = descriptor.representation;
97  pd.loadNonPolymerChemicals = descriptor.loadNonPolymerChemicals;
98  pd.positionRandomizationType = descriptor.positionRandomizationType;
99  pd.position = {0.f, 0.f, 0.f};
100 
101  ProteinPtr protein(new Protein(_scene, pd));
102  auto modelDescriptor = protein->getModelDescriptor();
103  _proteins[pd.name] = std::move(protein);
104  ++i;
105  }
106 
107  // Assemble proteins
108  _processInstances();
109 
110  // Add proteins to the scene
111  for (size_t i = 0; i < proteinContents.size(); ++i)
112  _scene.addModel(
113  _proteins[_getElementNameFromId(i)]->getModelDescriptor());
114 }

122 void Membrane::_processInstances()
123 {
124  // Randomization
125  srand(_details.randomSeed);
126  size_t rnd{1};
127  if (_details.randomSeed != 0 && _details.positionRandomizationType ==
128  PositionRandomizationType::circular)
129  rnd = rand() % _details.occurrences;
130 
131  const Quaterniond rotation = {_details.rotation[0], _details.rotation[1],
132  _details.rotation[2], _details.rotation[3]};
133  std::map<size_t, size_t> instanceCounts;
134  for (size_t i = 0; i < _proteins.size(); ++i)
135  instanceCounts[i] = 0;
136 
137  // Shape parameters
138  const auto &params = _details.assemblyParams;
139  const float size = (params.size() > 0 ? params[0] : 0.f);
140 
141  RandomizationInformation randInfo;
142  randInfo.seed = _details.randomSeed;
143  randInfo.randomizationType = _details.positionRandomizationType;
144  randInfo.positionStrength = (params.size() > 2 ? params[2] : 0.f);
145  randInfo.rotationStrength = (params.size() > 4 ? params[4] : 0.f);
146  const float extraParameter = (params.size() > 5 ? params[5] : 0.f);
147 
148  // Shape instances
149  const float offset = 2.f / _details.occurrences;
150  const float increment = M_PI * (3.f - sqrt(5.f));
151 
152  for (uint64_t occurence = 0; occurence < _details.occurrences; ++occurence)
153  {
154  const size_t id = rand() % _proteins.size();
155  const auto name = _getElementNameFromId(id);
156  if (_proteins.find(name) == _proteins.end())
157  {
158  PLUGIN_ERROR << "Protein " << name << " is not registered"
159  << std::endl;
160  continue;
161  }
162  auto protein = _proteins[name];
163  auto md = protein->getModelDescriptor();
164 
165  const auto &model = md->getModel();
166  const auto &bounds = model.getBounds();
167  const Vector3f &center = bounds.getCenter();
168 
169  randInfo.positionSeed =
170  (params.size() >= 2 ? (params[1] == 0 ? 0 : params[1] + occurence)
171  : 0);
172  randInfo.rotationSeed =
173  (params.size() >= 4 ? (params[3] == 0 ? 0 : params[3] + occurence)
174  : 0);
175 
176  Transformation transformation;
177  switch (_details.shape)
178  {
179  case AssemblyShape::spherical:
180  {
181  transformation =
182  getSphericalPosition(Vector3f(), size, occurence,
183  _details.occurrences, randInfo);
184  break;
185  }
186  case AssemblyShape::sinusoidal:
187  {
188  transformation =
189  getSinosoidalPosition(Vector3f(), size, extraParameter,
190  occurence, randInfo);
191  break;
192  }
193  case AssemblyShape::cubic:
194  {
195  transformation = getCubicPosition(Vector3f(), size, randInfo);
196  break;
197  }
198  case AssemblyShape::fan:
199  {
200  transformation = getFanPosition(Vector3f(), size, occurence,
201  _details.occurrences, randInfo);
202  break;
203  }
204  case AssemblyShape::bezier:
205  {
206  if ((params.size() - 5) % 3 != 0)
207  PLUGIN_THROW(std::runtime_error(
208  "Invalid number of floats in assembly extra parameters"));
209  Vector3fs points;
210  for (uint32_t i = 5; i < params.size(); i += 3)
211  points.push_back(
212  Vector3f(params[i], params[i + 1], params[i + 2]));
213  transformation = getBezierPosition(points, size,
214  float(occurence) /
215  float(_details.occurrences));
216  break;
217  }
218  case AssemblyShape::spherical_to_planar:
219  {
220  transformation =
221  getSphericalToPlanarPosition(Vector3f(), size, occurence,
222  _details.occurrences, randInfo,
223  extraParameter);
224  break;
225  }
226  default:
227  transformation = getPlanarPosition(Vector3f(), size, randInfo);
228  break;
229  }
230 
231  // Final transformation
232  const Vector3f translation =
233  _position + Vector3f(_rotation * (transformation.getTranslation() -
234  Vector3d(center)));
235 
236  if (isClipped(translation, _clippingPlanes))
237  continue;
238 
239  Transformation finalTransformation;
240  finalTransformation.setTranslation(translation);
241  finalTransformation.setRotation(
242  _rotation * transformation.getRotation() * rotation);
243 
244  if (instanceCounts[id] == 0)
245  md->setTransformation(finalTransformation);
246  const ModelInstance instance(true, false, finalTransformation);
247  md->addInstance(instance);
248 
249  instanceCounts[id] = instanceCounts[id] + 1;
250  }
251 }
