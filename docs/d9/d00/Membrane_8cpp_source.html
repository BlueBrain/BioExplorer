38 Membrane::Membrane(const MembraneDetails& details, Scene& scene,
39  const Vector3d& assemblyPosition,
40  const Quaterniond& assemblyRotation, const ShapePtr shape,
41  const ProteinMap& transmembraneProteins)
42  : _scene(scene)
43  , _details(details)
44  , _nbOccurrences{0}
45  , _transmembraneProteins(transmembraneProteins)
46  , _assemblyPosition(assemblyPosition)
47  , _assemblyRotation(assemblyRotation)
48  , _shape(shape)
49 {
50  // Lipid models
51  strings lipidPDBIds = split(_details.lipidPDBIds, CONTENTS_DELIMITER);
52  strings lipidContents = split(_details.lipidContents, CONTENTS_DELIMITER);
53 
54  double lipidAverageSize = 0.0;
55  size_t i = 0;
56  for (const auto& lipidContent : lipidContents)
57  {
58  ProteinDetails pd;
59  pd.assemblyName = _details.assemblyName;
60  pd.name = _getElementNameFromId(i);
61  pd.pdbId = lipidPDBIds[i];
62  pd.contents = lipidContent;
63  pd.recenter = true;
64  pd.atomRadiusMultiplier = _details.atomRadiusMultiplier;
65  pd.representation = _details.representation;
66  pd.loadBonds = _details.loadBonds;
67  pd.loadNonPolymerChemicals = _details.loadNonPolymerChemicals;
68 
69  // Create model
70  ProteinPtr lipid(new Protein(_scene, pd));
71  const auto& lipidSize = lipid->getBounds().getSize();
72  lipidAverageSize +=
73  std::min(lipidSize.x, std::min(lipidSize.y, lipidSize.z));
74  _lipids[pd.name] = std::move(lipid);
75  ++i;
76  }
77  lipidAverageSize /= lipidContents.size();
78  lipidAverageSize /= _details.lipidDensity;
79 
80  _nbOccurrences = _shape->getSurface() / lipidAverageSize *
81  2.0; // WHY DO I HAVE TO DOUBLE IT!
82 
83  _processInstances();
84 
85  // Add models to the scene
86  for (size_t i = 0; i < lipidContents.size(); ++i)
87  _scene.addModel(
88  _lipids[_getElementNameFromId(i)]->getModelDescriptor());
89 }

97 void Membrane::_processInstances()
98 {
99  const auto rotation = doublesToQuaterniond(_details.lipidRotation);
100  const auto animationDetails =
101  doublesToAnimationDetails(_details.animationParams);
102  srand(animationDetails.seed);
103 
104  std::map<size_t, size_t> instanceCounts;
105  for (size_t i = 0; i < _lipids.size(); ++i)
106  instanceCounts[i] = 0;
107 
108  for (uint64_t occurrence = 0; occurrence < _nbOccurrences; ++occurrence)
109  {
110  try
111  {
112  const size_t id = occurrence % _lipids.size();
113  auto lipid = _lipids[_getElementNameFromId(id)];
114  auto md = lipid->getModelDescriptor();
115 
116  const auto& model = md->getModel();
117  const auto& bounds = model.getBounds();
118  const Vector3d& center = bounds.getCenter();
119 
120  Transformations transformations;
121 
122  Transformation assemblyTransformation;
123  assemblyTransformation.setTranslation(_assemblyPosition);
124  assemblyTransformation.setRotation(_assemblyRotation);
125  transformations.push_back(assemblyTransformation);
126 
127  const auto shapeTransformation =
128  _shape->getTransformation(occurrence, _nbOccurrences,
129  animationDetails);
130  transformations.push_back(shapeTransformation);
131 
132  Transformation lipidTransformation;
133  lipidTransformation.setRotation(rotation);
134  transformations.push_back(lipidTransformation);
135 
136  const Transformation finalTransformation =
137  combineTransformations(transformations);
138  const auto& finalTranslation = finalTransformation.getTranslation();
139  const auto& finalRotation = finalTransformation.getRotation();
140 
141  // Collision with trans-membrane proteins
142  bool collision = false;
143  for (const auto& protein : _transmembraneProteins)
144  {
145  const auto transMembraneRadius =
146  protein.second->getTransMembraneRadius();
147  const auto transMembraneOffset =
148  protein.second->getTransMembraneOffset();
149  auto modelDescriptor = protein.second->getModelDescriptor();
150  const auto& instances = modelDescriptor->getInstances();
151  const auto& instanceSize =
152  modelDescriptor->getModel().getBounds().getSize();
153  for (const auto& instance : instances)
154  {
155  const auto& tf = instance.getTransformation();
156  const Vector3d proteinBase =
157  finalTranslation +
158  transMembraneOffset *
159  normalize(finalRotation * UP_VECTOR);
160  if (length(finalTranslation - tf.getTranslation()) <
161  transMembraneRadius)
162  {
163  collision = true;
164  break;
165  }
166  }
167  }
168  if (collision)
169  continue;
170 
171  if (instanceCounts[id] == 0)
172  md->setTransformation(finalTransformation);
173  const ModelInstance instance(true, false, finalTransformation);
174  md->addInstance(instance);
175 
176  instanceCounts[id] = instanceCounts[id] + 1;
177  }
178  catch (const std::runtime_error&)
179  {
180  // Instance is clipped
181  }
182  }
183 }
