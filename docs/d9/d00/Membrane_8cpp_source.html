32 Membrane::Membrane(Scene &scene, const MembraneDescriptor &descriptor,
33  const Vector3f &position, const Vector4fs &clippingPlanes,
34  const OccupiedDirections &occupiedDirections)
35  : _scene(scene)
36  , _position(position)
37  , _descriptor(descriptor)
38  , _clippingPlanes(clippingPlanes)
39  , _occupiedDirections(occupiedDirections)
40 {
41  if (_descriptor.representation == ProteinRepresentation::contour)
42  {
43  switch (_descriptor.shape)
44  {
45  case AssemblyShape::spherical:
46  const std::string name{"Membrane"};
47  const size_t materialId = 0;
48  auto model = _scene.createModel();
49 
50  auto material = model->createMaterial(materialId, name);
51  brayns::PropertyMap props;
52  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
53  static_cast<int>(MaterialShadingMode::basic)});
54  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
55  props.setProperty(
56  {MATERIAL_PROPERTY_CHAMELEON_MODE,
57  static_cast<int>(
58  MaterialChameleonMode::undefined_chameleon_mode)});
59  material->updateProperties(props);
60 
61  model->addSphere(materialId,
62  {_position, _descriptor.assemblyParams[0]});
63  auto modelDescriptor =
64  std::make_shared<ModelDescriptor>(std::move(model), name);
65  _scene.addModel(modelDescriptor);
66  return;
67  }
68  }
69 
70  std::vector<std::string> proteinContents;
71  proteinContents.push_back(descriptor.content1);
72  if (!descriptor.content2.empty())
73  proteinContents.push_back(descriptor.content2);
74  if (!descriptor.content3.empty())
75  proteinContents.push_back(descriptor.content3);
76  if (!descriptor.content4.empty())
77  proteinContents.push_back(descriptor.content4);
78 
79  // Load proteins
80  size_t i = 0;
81  for (const auto &content : proteinContents)
82  {
83  ProteinDescriptor pd;
84  pd.assemblyName = descriptor.assemblyName;
85  pd.name = _getElementNameFromId(i);
86  pd.contents = content;
87  pd.chainIds = descriptor.chainIds;
88  pd.recenter = descriptor.recenter;
89  pd.loadBonds = descriptor.loadBonds;
90  pd.randomSeed = descriptor.randomSeed;
91  pd.occurrences = 1;
92  pd.orientation = descriptor.orientation;
93  pd.assemblyParams = descriptor.assemblyParams;
94  pd.atomRadiusMultiplier = descriptor.atomRadiusMultiplier;
95  pd.representation = descriptor.representation;
96  pd.locationCutoffAngle = descriptor.locationCutoffAngle;
97  pd.loadNonPolymerChemicals = descriptor.loadNonPolymerChemicals;
98  pd.positionRandomizationType = descriptor.positionRandomizationType;
99  pd.position = {0.f, 0.f, 0.f};
100 
101  ProteinPtr protein(new Protein(_scene, pd));
102  auto modelDescriptor = protein->getModelDescriptor();
103  _proteins[pd.name] = std::move(protein);
104  ++i;
105  }
106 
107  // Assemble proteins
108  _processInstances();
109 
110  // Add proteins to the scene
111  for (size_t i = 0; i < proteinContents.size(); ++i)
112  _scene.addModel(
113  _proteins[_getElementNameFromId(i)]->getModelDescriptor());
114 }

122 void Membrane::_processInstances()
123 {
124  const float offset = 2.f / _descriptor.occurrences;
125  const float increment = M_PI * (3.f - sqrt(5.f));
126 
127  srand(_descriptor.randomSeed);
128  size_t rnd{1};
129  if (_descriptor.randomSeed != 0 && _descriptor.positionRandomizationType ==
130  PositionRandomizationType::circular)
131  rnd = rand() % _descriptor.occurrences;
132 
133  const Quaterniond orientation = {_descriptor.orientation[0],
134  _descriptor.orientation[1],
135  _descriptor.orientation[2],
136  _descriptor.orientation[3]};
137  std::map<size_t, size_t> instanceCounts;
138  for (size_t i = 0; i < _proteins.size(); ++i)
139  instanceCounts[i] = 0;
140 
141  for (uint64_t i = 0; i < _descriptor.occurrences; ++i)
142  {
143  const size_t id = rand() % _proteins.size();
144  const auto name = _getElementNameFromId(id);
145  if (_proteins.find(name) == _proteins.end())
146  {
147  PLUGIN_ERROR << "Protein " << name << " is not registered"
148  << std::endl;
149  continue;
150  }
151  auto protein = _proteins[name];
152  auto md = protein->getModelDescriptor();
153 
154  const auto &model = md->getModel();
155  const auto &bounds = model.getBounds();
156  const Vector3f &center = bounds.getCenter();
157 
158  Vector3f pos;
159  Vector3f dir{0.f, 1.f, 0.f};
160  switch (_descriptor.shape)
161  {
162  case AssemblyShape::spherical:
163  {
164  getSphericalPosition(rnd, _descriptor.assemblyParams[0],
165  _descriptor.assemblyParams[1],
166  _descriptor.positionRandomizationType,
167  _descriptor.randomSeed, i,
168  _descriptor.occurrences, {0, 0, 0}, pos, dir);
169  break;
170  }
171  case AssemblyShape::sinusoidal:
172  {
173  const auto assemblySize = _descriptor.assemblyParams[0];
174  const auto assemblyHeight = _descriptor.assemblyParams[1];
175  getSinosoidalPosition(assemblySize, assemblyHeight,
176  _descriptor.positionRandomizationType,
177  _descriptor.randomSeed, {0, 0, 0}, pos, dir);
178  break;
179  }
180  case AssemblyShape::cubic:
181  {
182  const auto assemblySize = _descriptor.assemblyParams[0];
183  getCubicPosition(assemblySize, {0, 0, 0}, pos, dir);
184  break;
185  }
186  case AssemblyShape::fan:
187  {
188  const auto assemblyRadius = _descriptor.assemblyParams[0];
189  getFanPosition(rnd, assemblyRadius,
190  _descriptor.positionRandomizationType,
191  _descriptor.randomSeed, i, _descriptor.occurrences,
192  {0, 0, 0}, pos, dir);
193  break;
194  }
195  case AssemblyShape::bezier:
196  {
197  const Vector3fs points = {
198  {1, 391, 0}, {25, 411, 0}, {48, 446, 0}, {58, 468, 0},
199  {70, 495, 0}, {83, 523, 0}, {110, 535, 0}, {157, 517, 0},
200  {181, 506, 0}, {214, 501, 0}, {216, 473, 0}, {204, 456, 0},
201  {223, 411, 0}, {241, 382, 0}, {261, 372, 0}, {297, 402, 0},
202  {308, 433, 0}, {327, 454, 0}, {355, 454, 0}, {389, 446, 0},
203  {406, 433, 0}, {431, 426, 0}, {458, 443, 0}, {478, 466, 0},
204  {518, 463, 0}, {559, 464, 0}, {584, 478, 0}, {582, 503, 0},
205  {550, 533, 0}, {540, 550, 0}, {540, 574, 0}, {560, 572, 0},
206  {599, 575, 0}, {629, 550, 0}, {666, 548, 0}, {696, 548, 0},
207  {701, 582, 0}, {701, 614, 0}, {683, 639, 0}, {653, 647, 0},
208  {632, 651, 0}, {597, 666, 0}, {570, 701, 0}, {564, 731, 0},
209  {559, 770, 0}, {565, 799, 0}, {577, 819, 0}, {611, 820, 0},
210  {661, 809, 0}, {683, 787, 0}, {700, 768, 0}, {735, 758, 0},
211  {763, 768, 0}, {788, 792, 0}, {780, 820, 0}, {770, 859, 0},
212  {740, 882, 0}, {705, 911, 0}, {688, 931, 0}, {646, 973, 0},
213  {611, 992, 0}, {585, 1022, 0}};
214  const auto assemblySize = _descriptor.assemblyParams[0];
215  getBezierPosition(points, assemblySize,
216  float(i) / float(_descriptor.occurrences), pos,
217  dir);
218  break;
219  }
220  default:
221  const auto assemblySize = _descriptor.assemblyParams[0];
222  getPlanarPosition(assemblySize,
223  _descriptor.positionRandomizationType,
224  _descriptor.randomSeed, {0, 0, 0}, pos, dir);
225  break;
226  }
227 
228  // Clipping planes
229  if (isClipped(pos, _clippingPlanes))
230  continue;
231 
232  // Remove membrane where proteins are. This is currently done
233  // according to the vector orientation
234  bool occupied{false};
235  if (_descriptor.locationCutoffAngle != 0.f)
236  for (const auto &occupiedDirection : _occupiedDirections)
237  if (dot(dir, occupiedDirection.first) >
238  occupiedDirection.second)
239  {
240  occupied = true;
241  break;
242  }
243  if (occupied)
244  continue;
245 
246  // Final transformation
247  Transformation tf;
248  tf.setTranslation(_position + pos - center);
249 
250  Quaterniond assemblyOrientation = glm::quatLookAt(dir, UP_VECTOR);
251 
252  if (_descriptor.randomSeed == 0)
253  tf.setRotation(assemblyOrientation * orientation);
254  else
255  {
256  // Add a bit of randomness in the orientation of the proteins
257  Vector3f eulerAngles(0.3 * (rand() % 100 / 100.0 - 0.5),
258  0.3 * (rand() % 100 / 100.0 - 0.5),
259  0.3 * (rand() % 100 / 100.0 - 0.5));
260  Quaterniond randomOrientation = glm::quat(eulerAngles);
261 
262  tf.setRotation(assemblyOrientation * orientation *
263  randomOrientation);
264  }
265 
266  if (instanceCounts[id] == 0)
267  md->setTransformation(tf);
268  const ModelInstance instance(true, false, tf);
269  md->addInstance(instance);
270 
271  // Save initial transformation for later use
272  // _transformations[_descriptor.name].push_back(tf);
273 
274  instanceCounts[id] = instanceCounts[id] + 1;
275  }
276 
277  for (size_t i = 0; i < _proteins.size(); ++i)
278  PLUGIN_INFO << "Instances for " << i << " : " << instanceCounts[i]
279  << std::endl;
280 }
