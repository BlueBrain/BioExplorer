37 Membrane::Membrane(Scene &scene, const MembraneDetails &details,
38  const Vector3f &position, const Quaterniond &rotation,
39  const Vector4fs &clippingPlanes)
40  : _scene(scene)
41  , _position(position)
42  , _rotation(rotation)
43  , _details(details)
44  , _clippingPlanes(clippingPlanes)
45 {
46  if (_details.representation == ProteinRepresentation::contour)
47  {
48  switch (_details.shape)
49  {
50  case AssemblyShape::spherical:
51  const std::string name{"Membrane"};
52  const size_t materialId = 0;
53  auto model = _scene.createModel();
54 
55  auto material = model->createMaterial(materialId, name);
56  brayns::PropertyMap props;
57  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
58  static_cast<int>(MaterialShadingMode::basic)});
59  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
60  props.setProperty(
61  {MATERIAL_PROPERTY_CHAMELEON_MODE,
62  static_cast<int>(
63  MaterialChameleonMode::undefined_chameleon_mode)});
64  material->updateProperties(props);
65 
66  model->addSphere(materialId,
67  {_position, _details.assemblyParams[0]});
68  auto modelDescriptor =
69  std::make_shared<ModelDescriptor>(std::move(model), name);
70  _scene.addModel(modelDescriptor);
71  return;
72  }
73  }
74 
75  std::vector<std::string> proteinContents;
76  proteinContents.push_back(details.content1);
77  if (!details.content2.empty())
78  proteinContents.push_back(details.content2);
79  if (!details.content3.empty())
80  proteinContents.push_back(details.content3);
81  if (!details.content4.empty())
82  proteinContents.push_back(details.content4);
83 
84  // Load proteins
85  size_t i = 0;
86  for (const auto &content : proteinContents)
87  {
88  ProteinDetails pd;
89  pd.assemblyName = details.assemblyName;
90  pd.name = _getElementNameFromId(i);
91  pd.contents = content;
92  pd.chainIds = details.chainIds;
93  pd.recenter = details.recenter;
94  pd.loadBonds = details.loadBonds;
95  pd.randomSeed = details.randomSeed;
96  pd.occurrences = 1;
97  pd.rotation = details.rotation;
98  pd.assemblyParams = details.assemblyParams;
99  pd.atomRadiusMultiplier = details.atomRadiusMultiplier;
100  pd.representation = details.representation;
101  pd.loadNonPolymerChemicals = details.loadNonPolymerChemicals;
102  pd.positionRandomizationType = details.positionRandomizationType;
103  pd.position = {0.f, 0.f, 0.f};
104 
105  ProteinPtr protein(new Protein(_scene, pd));
106  auto modelDescriptor = protein->getModelDescriptor();
107  _proteins[pd.name] = std::move(protein);
108  ++i;
109  }
110 
111  // Assemble proteins
112  _processInstances();
113 
114  // Add proteins to the scene
115  for (size_t i = 0; i < proteinContents.size(); ++i)
116  _scene.addModel(
117  _proteins[_getElementNameFromId(i)]->getModelDescriptor());
118 }

126 void Membrane::_processInstances()
127 {
128  // Randomization
129  srand(_details.randomSeed);
130  size_t rnd{1};
131  if (_details.randomSeed != 0 && _details.positionRandomizationType ==
132  PositionRandomizationType::circular)
133  rnd = rand() % _details.occurrences;
134 
135  const Quaterniond rotation = {_details.rotation[0], _details.rotation[1],
136  _details.rotation[2], _details.rotation[3]};
137  std::map<size_t, size_t> instanceCounts;
138  for (size_t i = 0; i < _proteins.size(); ++i)
139  instanceCounts[i] = 0;
140 
141  // Shape parameters
142  const auto &params = _details.assemblyParams;
143  const float size = (params.size() > 0 ? params[0] : 0.f);
144 
145  RandomizationDetails randInfo;
146  randInfo.seed = _details.randomSeed;
147  randInfo.randomizationType = _details.positionRandomizationType;
148  randInfo.positionStrength = (params.size() > 2 ? params[2] : 0.f);
149  randInfo.rotationStrength = (params.size() > 4 ? params[4] : 0.f);
150  const float extraParameter = (params.size() > 5 ? params[5] : 0.f);
151 
152  // Shape instances
153  const float offset = 2.f / _details.occurrences;
154  const float increment = M_PI * (3.f - sqrt(5.f));
155 
156  for (uint64_t occurence = 0; occurence < _details.occurrences; ++occurence)
157  {
158  const size_t id = rand() % _proteins.size();
159  const auto name = _getElementNameFromId(id);
160  if (_proteins.find(name) == _proteins.end())
161  {
162  PLUGIN_ERROR("Protein " << name << " is not registered");
163  continue;
164  }
165  auto protein = _proteins[name];
166  auto md = protein->getModelDescriptor();
167 
168  const auto &model = md->getModel();
169  const auto &bounds = model.getBounds();
170  const Vector3f &center = bounds.getCenter();
171 
172  randInfo.positionSeed =
173  (params.size() >= 2 ? (params[1] == 0 ? 0 : params[1] + occurence)
174  : 0);
175  randInfo.rotationSeed =
176  (params.size() >= 4 ? (params[3] == 0 ? 0 : params[3] + occurence)
177  : 0);
178 
179  Transformation transformation;
180  switch (_details.shape)
181  {
182  case AssemblyShape::spherical:
183  {
184  transformation =
185  getSphericalPosition(Vector3f(), size, occurence,
186  _details.occurrences, randInfo);
187  break;
188  }
189  case AssemblyShape::sinusoidal:
190  {
191  transformation =
192  getSinosoidalPosition(Vector3f(), size, extraParameter,
193  occurence, randInfo);
194  break;
195  }
196  case AssemblyShape::cubic:
197  {
198  transformation = getCubicPosition(Vector3f(), size, randInfo);
199  break;
200  }
201  case AssemblyShape::fan:
202  {
203  transformation = getFanPosition(Vector3f(), size, occurence,
204  _details.occurrences, randInfo);
205  break;
206  }
207  case AssemblyShape::bezier:
208  {
209  if ((params.size() - 5) % 3 != 0)
210  PLUGIN_THROW(
211  "Invalid number of floats in assembly extra parameters");
212  Vector3fs points;
213  for (uint32_t i = 5; i < params.size(); i += 3)
214  points.push_back(
215  Vector3f(params[i], params[i + 1], params[i + 2]));
216  transformation = getBezierPosition(points, size,
217  float(occurence) /
218  float(_details.occurrences));
219  break;
220  }
221  case AssemblyShape::spherical_to_planar:
222  {
223  transformation =
224  getSphericalToPlanarPosition(Vector3f(), size, occurence,
225  _details.occurrences, randInfo,
226  extraParameter);
227  break;
228  }
229  default:
230  transformation = getPlanarPosition(Vector3f(), size, randInfo);
231  break;
232  }
233 
234  // Final transformation
235  const Vector3f translation =
236  _position + Vector3f(_rotation * (transformation.getTranslation() -
237  Vector3d(center)));
238 
239  if (isClipped(translation, _clippingPlanes))
240  continue;
241 
242  Transformation finalTransformation;
243  finalTransformation.setTranslation(translation);
244  finalTransformation.setRotation(
245  _rotation * transformation.getRotation() * rotation);
246 
247  if (instanceCounts[id] == 0)
248  md->setTransformation(finalTransformation);
249  const ModelInstance instance(true, false, finalTransformation);
250  md->addInstance(instance);
251 
252  instanceCounts[id] = instanceCounts[id] + 1;
253  }
254 }
