48 void DBConnector::init(const CommandLineArguments& arguments)
49 {
50  try
51  {
52  std::string dbHost, dbPort, dbUser, dbPassword, dbName;
53  for (const auto& argument : arguments)
54  {
55  if (argument.first == ARG_DB_HOST)
56  dbHost = argument.second;
57  if (argument.first == ARG_DB_PORT)
58  dbPort = argument.second;
59  if (argument.first == ARG_DB_USER)
60  dbUser = argument.second;
61  if (argument.first == ARG_DB_PASSWORD)
62  dbPassword = argument.second;
63  if (argument.first == ARG_DB_NAME)
64  dbName = argument.second;
65  }
66 
67  const std::string connectionString =
68  "host=" + dbHost + " port=" + dbPort + " dbname=" + dbName +
69  " user=" + dbUser + " password=" + dbPassword;
70  PLUGIN_ERROR(connectionString);
71  _connection = ConnectionPtr(new pqxx::connection(connectionString));
72  }
73  catch (const pqxx::pqxx_exception& e)
74  {
75  PLUGIN_THROW(
76  "Failed to connect to database, check command line parameters. " +
77  std::string(e.base().what()));
78  }
79 }

98 const OOCSceneConfigurationDetails DBConnector::getSceneConfiguration()
99 {
100  OOCSceneConfigurationDetails sceneConfiguration;
101  pqxx::read_transaction transaction(*_connection);
102  try
103  {
104  const auto sql =
105  "SELECT scene_size_x, scene_size_y, scene_size_z, nb_bricks, "
106  "description FROM " +
107  DB_SCHEMA_OUT_OF_CORE + ".configuration";
108  PLUGIN_DEBUG(sql);
109  auto res = transaction.exec(sql);
110  for (auto c = res.begin(); c != res.end(); ++c)
111  {
112  sceneConfiguration.sceneSize =
113  Vector3d(c[0].as<double>(), c[1].as<double>(),
114  c[2].as<double>());
115  sceneConfiguration.nbBricks = c[3].as<uint32_t>();
116  sceneConfiguration.description = c[4].as<std::string>();
117  if (sceneConfiguration.nbBricks == 0)
118  PLUGIN_THROW("Invalid number of bricks)");
119  sceneConfiguration.brickSize =
120  sceneConfiguration.sceneSize /
121  static_cast<double>(sceneConfiguration.nbBricks);
122  }
123  }
124  catch (pqxx::sql_error& e)
125  {
126  PLUGIN_THROW(e.what());
127  }
128  transaction.abort();
129  return sceneConfiguration;
130 }

212 GeometryNodes DBConnector::getVasculatureNodes(
213  const std::string& populationName, const std::string& filter) const
214 {
215  const auto populationId = getVasculaturePopulationId(populationName);
216 
217  GeometryNodes nodes;
218  pqxx::read_transaction transaction(*_connection);
219  try
220  {
221  std::string sql =
222  "SELECT n.guid, n.x, n.y, n.z, n.radius, "
223  "v.section_guid, v.type_guid, v.subgraph_guid, v.pair_guid, "
224  "v.entry_node_guid FROM " +
225  DB_SCHEMA_VASCULATURE + ".node as n, " + DB_SCHEMA_VASCULATURE +
226  ".vasculature as v WHERE n.node_guid=v.node_guid AND "
227  "n.population_guid=v.population_guid AND "
228  "n.population_guid=" +
229  std::to_string(populationId);
230  if (!filter.empty())
231  sql += "AND " + filter;
232  PLUGIN_DEBUG(sql);
233  auto res = transaction.exec(sql);
234  for (auto c = res.begin(); c != res.end(); ++c)
235  {
236  GeometryNode node;
237  node.position = Vector3d(c[1].as<double>(), c[2].as<double>(),
238  c[3].as<double>());
239  node.radius = c[4].as<double>();
240  node.sectionId = c[5].as<uint64_t>();
241  node.type = c[6].as<uint64_t>();
242  node.graphId = c[7].as<uint64_t>();
243  node.pairId = c[8].as<uint64_t>();
244  node.entryNodeId = c[9].as<uint64_t>();
245  nodes[c[0].as<uint64_t>()] = node;
246  }
247  }
248  catch (const pqxx::sql_error& e)
249  {
250  transaction.abort();
251  PLUGIN_THROW(e.what());

362 {
363  SomaMap somas;
364 
365  pqxx::read_transaction transaction(*_connection);
366  try
367  {
368  std::string sql = "SELECT guid, x, y, z, radius FROM " +
369  DB_SCHEMA_ASTROCYTES + ".node";
370  strings conditions;
371  if (!astrocyteIds.empty())
372  {
373  std::string condition = "guid IN (";
374  for (size_t i = 0; i < astrocyteIds.size(); ++i)
375  {
376  if (i != 0)
377  condition += ",";
378  condition += std::to_string(astrocyteIds[i]);
379  }
380  condition += ") ";
381  conditions.push_back(condition);
382  }
383 
384  if (!sqlCondition.empty())
385  conditions.push_back(sqlCondition);
386 
387  for (size_t i = 0; i < conditions.size(); ++i)
388  {
389  if (i == 0)
390  sql += " WHERE ";
391  else
392  sql += " AND ";
393  sql += conditions[i];
394  }
395 
396  sql += " ORDER BY guid";
397 
398  PLUGIN_DEBUG(sql);
399  auto res = transaction.exec(sql);
400  for (auto c = res.begin(); c != res.end(); ++c)
401  {
402  Soma soma;
403  soma.center =
404  Vector3d(c[1].as<float>(), c[2].as<float>(), c[3].as<float>());
405  soma.radius = c[4].as<float>() * 0.5;
406  somas[c[0].as<uint64_t>()] = soma;
407  }
408  }
409  catch (const pqxx::sql_error& e)
410  {
411  transaction.abort();
