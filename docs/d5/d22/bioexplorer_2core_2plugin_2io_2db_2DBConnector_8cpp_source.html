52 void DBConnector::init(const CommandLineArguments& arguments)
53 {
54  std::string dbHost, dbPort, dbUser, dbPassword, dbName;
55  for (const auto& argument : arguments)
56  {
57  if (argument.first == ARG_DB_HOST)
58  dbHost = argument.second;
59  if (argument.first == ARG_DB_PORT)
60  dbPort = argument.second;
61  if (argument.first == ARG_DB_USER)
62  dbUser = argument.second;
63  if (argument.first == ARG_DB_PASSWORD)
64  dbPassword = argument.second;
65  if (argument.first == ARG_DB_NAME)
66  dbName = argument.second;
67  }
68 
69  _connectionString = "host=" + dbHost + " port=" + dbPort +
70  " dbname=" + dbName + " user=" + dbUser +
71  " password=" + dbPassword;
72 
73  PLUGIN_DEBUG(_connectionString);
74 
75  const auto nbConnections = omp_get_max_threads();
76  for (size_t i = 0; i < nbConnections; ++i)
77  {
78  try
79  {
80  _connections.push_back(
81  ConnectionPtr(new pqxx::connection(_connectionString)));
82  }
83  catch (const pqxx::pqxx_exception& e)
84  {
85  PLUGIN_THROW(
86  "Failed to connect to database, check command line "
87  "parameters. " +
88  std::string(e.base().what()));
89  }
90  }
91  PLUGIN_INFO(1,
92  "Initialized " << nbConnections << " connections to database");
93 }

200 GeometryNodes DBConnector::getVasculatureNodes(
201  const std::string& populationName, const std::string& filter,
202  const std::string& limits) const
203 {
204  GeometryNodes nodes;
205  const auto connection = _connections[omp_get_thread_num()];
206  pqxx::read_transaction transaction(*connection);
207  try
208  {
209  std::string sql =
210  "SELECT guid, x, y, z, radius, section_guid, sub_graph_guid, "
211  "pair_guid, entry_node_guid FROM " +
212  populationName + ".node";
213  if (!filter.empty())
214  sql += " WHERE " + filter;
215  if (!limits.empty())
216  {
217  if (filter.empty())
218  sql += " WHERE ";
219  else
220  sql += " AND ";
221  sql += limits;
222  }
223  sql += " ORDER BY guid ";
224 
225  PLUGIN_DEBUG(sql);
226  auto res = transaction.exec(sql);
227  for (auto c = res.begin(); c != res.end(); ++c)
228  {
229  GeometryNode node;
230  const uint64_t guid = c[0].as<uint64_t>();
231  node.position = Vector3d(c[1].as<double>(), c[2].as<double>(),
232  c[3].as<double>());
233  node.radius = c[4].as<double>();
234  node.sectionId = c[5].as<uint64_t>();
235  node.graphId = c[6].as<uint64_t>();
236  node.pairId = c[7].as<uint64_t>();
237  node.entryNodeId = c[8].as<uint64_t>();
238  nodes[guid] = node;
239  }
240  }
241  catch (const pqxx::sql_error& e)
242  {
243  PLUGIN_THROW(e.what());
244  }
245 
246  return nodes;

249 uint64_ts DBConnector::getVasculatureSections(const std::string& populationName,
250  const std::string& filter)
251 {
252  uint64_ts sectionIds;
253  auto connection = _connections[omp_get_thread_num()];
254  pqxx::read_transaction transaction(*connection);
255  try
256  {
257  pqxx::result res;
258  std::string sql =
259  "SELECT distinct(section_guid) FROM " + populationName + ".node";
260 
261  if (!filter.empty())
262  sql += " WHERE " + filter;
263 
264  const std::string name = "allVasculatureSections" + filter;
265  auto preparedStatements = _preparedStatements[populationName];
266  if (std::find(preparedStatements.begin(), preparedStatements.end(),
267  name) == preparedStatements.end())
268  {
269  PLUGIN_DEBUG("Preparing statement " << name);
270  connection->prepare(name, sql);
271  preparedStatements.push_back(name);
272  }
273  PLUGIN_DEBUG("Executing preparted statement: " << sql);
274  res = transaction.exec_prepared(name);
275  for (auto c = res.begin(); c != res.end(); ++c)
276  sectionIds.push_back(c[0].as<uint64_t>());
277  }
278  catch (const pqxx::sql_error& e)
279  {
280  PLUGIN_THROW(e.what());
281  }
282 
283  return sectionIds;
284 }

295  ".metadata WHERE name='nb_sections'";

301  nbSections.y = c[0].as<uint64_t>();
302  }
303 
304  if (!filter.empty())
305  {
306  const std::string sql =
307  "SELECT count(distinct(section_guid)), min(section_guid), "
308  "max(section_guid) FROM " +
309  populationName + ".node WHERE " + filter;
310  const std::string name = "vasculatureNbSections" + filter;
311  auto preparedStatements = _preparedStatements[populationName];
312  if (std::find(preparedStatements.begin(), preparedStatements.end(),

315  PLUGIN_DEBUG("Preparing statement " << name);
316  connection->prepare(name, sql);
317  preparedStatements.push_back(name);
318  }
319  PLUGIN_DEBUG("Executing preparted statement: " << sql);
320  res = transaction.exec_prepared(name);
321  for (auto c = res.begin(); c != res.end(); ++c)
322  nbSections.x = c[0].as<uint64_t>();

324  }
