44 void DBConnector::init(const CommandLineArguments& arguments)
45 {
46  try
47  {
48  std::string dbHost, dbPort, dbUser, dbPassword, dbName;
49  for (const auto& argument : arguments)
50  {
51  if (argument.first == ARG_DB_HOST)
52  dbHost = argument.second;
53  if (argument.first == ARG_DB_PORT)
54  dbPort = argument.second;
55  if (argument.first == ARG_DB_USER)
56  dbUser = argument.second;
57  if (argument.first == ARG_DB_PASSWORD)
58  dbPassword = argument.second;
59  if (argument.first == ARG_DB_NAME)
60  dbName = argument.second;
61  }
62 
63  const std::string connectionString =
64  "host=" + dbHost + " port=" + dbPort + " dbname=" + dbName +
65  " user=" + dbUser + " password=" + dbPassword;
66  PLUGIN_ERROR(connectionString);
67  _connection = ConnectionPtr(new pqxx::connection(connectionString));
68  }
69  catch (const pqxx::pqxx_exception& e)
70  {
71  PLUGIN_THROW(
72  "Failed to connect to database, check command line parameters. " +
73  std::string(e.base().what()));
74  }
75 }

94 const OOCSceneConfigurationDetails DBConnector::getSceneConfiguration()
95 {
96  OOCSceneConfigurationDetails sceneConfiguration;
97  pqxx::read_transaction transaction(*_connection);
98  try
99  {
100  const auto sql =
101  "SELECT scene_size_x, scene_size_y, scene_size_z, nb_bricks, "
102  "description FROM " +
103  DB_SCHEMA_OUT_OF_CORE + ".configuration";
104  PLUGIN_DEBUG(sql);
105  auto res = transaction.exec(sql);
106  for (auto c = res.begin(); c != res.end(); ++c)
107  {
108  sceneConfiguration.sceneSize =
109  Vector3d(c[0].as<double>(), c[1].as<double>(),
110  c[2].as<double>());
111  sceneConfiguration.nbBricks = c[3].as<uint32_t>();
112  sceneConfiguration.description = c[4].as<std::string>();
113  if (sceneConfiguration.nbBricks == 0)
114  PLUGIN_THROW("Invalid number of bricks)");
115  sceneConfiguration.brickSize =
116  sceneConfiguration.sceneSize /
117  static_cast<double>(sceneConfiguration.nbBricks);
118  }
119  }
120  catch (pqxx::sql_error& e)
121  {
122  PLUGIN_THROW(e.what());
123  }
124  transaction.abort();
125  return sceneConfiguration;
126 }

128 void DBConnector::insertBrick(const int32_t brickId, const uint32_t version,
129  const uint32_t nbModels,
130  const std::stringstream& buffer)
131 {
132  pqxx::work transaction(*_connection);
133  try
134  {
135  const pqxx::binarystring tmp((void*)buffer.str().c_str(),
136  buffer.str().size() * sizeof(char));
137  transaction.exec_params("INSERT INTO " + DB_SCHEMA_OUT_OF_CORE +
138  ".brick VALUES ($1, $2, $3, $4)",
139  brickId, version, nbModels, tmp);
140  transaction.commit();
141  PLUGIN_DEBUG("Brick ID " << brickId << " successfully inserted");
142  }
143  catch (pqxx::sql_error& e)
144  {
145  transaction.abort();
146  PLUGIN_THROW(e.what());
147  }

150 std::stringstream DBConnector::getBrick(const int32_t brickId,
151  const uint32_t& version,
152  uint32_t& nbModels)
153 {
154  std::stringstream s;
155  pqxx::read_transaction transaction(*_connection);
156  try
157  {
158  const auto sql = "SELECT nb_models, buffer FROM " +
159  DB_SCHEMA_OUT_OF_CORE +
160  ".brick WHERE guid=" + std::to_string(brickId) +
161  " AND version=" + std::to_string(version);
162  PLUGIN_DEBUG(sql);
163  auto res = transaction.exec(sql);
164  for (auto c = res.begin(); c != res.end(); ++c)
165  {
166  nbModels = c[0].as<uint32_t>();
167  if (nbModels > 0)
168  {
169  const pqxx::binarystring buffer(c[1]);
170  std::copy(buffer.begin(), buffer.end(),
171  std::ostream_iterator<char>(s));
172  }
173  }
174  }
175  catch (pqxx::sql_error& e)
176  {
177  PLUGIN_THROW(e.what());
178  }
179  transaction.abort();
180 
181  return s;

207 GeometryNodes DBConnector::getVasculatureNodes(
208  const std::string& populationName, const std::string& filter) const
209 {
210  const auto populationId = getVasculaturePopulationId(populationName);
211 
212  GeometryNodes nodes;
213  pqxx::read_transaction transaction(*_connection);
214  try
215  {
216  std::string sql =
217  "SELECT n.guid, n.x, n.y, n.z, n.radius, "
218  "v.section_guid, v.type_guid, v.subgraph_guid, v.pair_guid, "
219  "v.entry_node_guid FROM " +
220  DB_SCHEMA_VASCULATURE + ".node as n, " + DB_SCHEMA_VASCULATURE +
221  ".vasculature as v WHERE n.node_guid=v.node_guid AND "
222  "n.population_guid=v.population_guid AND "
223  "n.population_guid=" +
224  std::to_string(populationId);
225  if (!filter.empty())
226  sql += "AND " + filter;
227  PLUGIN_DEBUG(sql);
228  auto res = transaction.exec(sql);
229  for (auto c = res.begin(); c != res.end(); ++c)
230  {
231  GeometryNode node;
232  node.position = Vector3d(c[1].as<double>(), c[2].as<double>(),
233  c[3].as<double>());
234  node.radius = c[4].as<double>();
235  node.sectionId = c[5].as<uint64_t>();
236  node.type = c[6].as<uint64_t>();
237  node.graphId = c[7].as<uint64_t>();
238  node.pairId = c[8].as<uint64_t>();
239  node.entryNodeId = c[9].as<uint64_t>();
240  nodes[c[0].as<uint64_t>()] = node;
241  }
242  }
243  catch (const pqxx::sql_error& e)
244  {
245  transaction.abort();
246  PLUGIN_THROW(e.what());
247  }
248  transaction.abort();
249 
250  return nodes;
251 }
