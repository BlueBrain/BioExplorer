27 // Plane intersection -- used for refining triangle hit points. Note
28 // that this skips zero denom check (for rays perpendicular to plane normal)
29 // since we know that the ray intersects the plane.
30 static __device__ __inline__ float intersectPlane(const float3& origin,
31  const float3& direction,
32  const float3& normal,
33  const float3& point)
34 {
35  // Skipping checks for non-zero denominator since we know that ray
36  // intersects this plane
37  return -(dot(normal, origin - point)) / dot(normal, direction);
38 }

40 // Offset the hit point using integer arithmetic
41 static __device__ __inline__ float3 offset(const float3& hit_point,
42  const float3& normal)
43 {
44  const float epsilon = 1.0e-4f;
45  const float offset = 4096.0f * 2.0f;
46 
47  float3 offset_point = hit_point;
48  if ((__float_as_int(hit_point.x) & 0x7fffffff) < __float_as_int(epsilon))
49  {
50  offset_point.x += epsilon * normal.x;
51  }
52  else
53  {
54  offset_point.x =
55  __int_as_float(__float_as_int(offset_point.x) +
56  int(copysign(offset, hit_point.x) * normal.x));
57  }
58 
59  if ((__float_as_int(hit_point.y) & 0x7fffffff) < __float_as_int(epsilon))
60  {
61  offset_point.y += epsilon * normal.y;
62  }
63  else
64  {
65  offset_point.y =
66  __int_as_float(__float_as_int(offset_point.y) +
67  int(copysign(offset, hit_point.y) * normal.y));
68  }
69 
70  if ((__float_as_int(hit_point.z) & 0x7fffffff) < __float_as_int(epsilon))
71  {
72  offset_point.z += epsilon * normal.z;
73  }
74  else
75  {
76  offset_point.z =
77  __int_as_float(__float_as_int(offset_point.z) +
78  int(copysign(offset, hit_point.z) * normal.z));
79  }
80 
81  return offset_point;
82 }

84 // Refine the hit point to be more accurate and offset it for reflection and
85 // refraction ray starting points.
86 static __device__ __inline__ void refine_and_offset_hitpoint(
87  const float3& original_hit_point, const float3& direction,
88  const float3& normal, const float3& p, float3& back_hit_point,
89  float3& front_hit_point)
90 {
91  // Refine hit point
92  float refined_t = intersectPlane(original_hit_point, direction, normal, p);
93  float3 refined_hit_point = original_hit_point + refined_t * direction;
94 
95  // Offset hit point
96  if (dot(direction, normal) > 0.0f)
97  {
98  back_hit_point = offset(refined_hit_point, normal);
99  front_hit_point = offset(refined_hit_point, -normal);
100  }
101  else
102  {
103  back_hit_point = offset(refined_hit_point, -normal);
104  front_hit_point = offset(refined_hit_point, normal);
105  }
