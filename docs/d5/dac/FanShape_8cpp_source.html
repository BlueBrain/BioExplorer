43 Transformation FanShape::getTransformation(
44  const uint64_t occurrence, const uint64_t nbOccurrences,
45  const AnimationDetails& animationDetails, const double offset) const
46 {
47  uint64_t rnd = occurrence;
48  if (nbOccurrences != 0 && animationDetails.seed != 0)
49  if (GeneralSettings::getInstance()->getV1Compatibility())
50  rnd = rand() % nbOccurrences;
51  else
52  rnd = rand() % std::numeric_limits<uint64_t>::max();
53 
54  const double radius =
55  _radius + (animationDetails.positionSeed == 0
56  ? animationDetails.positionStrength
57  : animationDetails.positionStrength *
58  rnd3(animationDetails.positionSeed + rnd));
59 
60  Vector3d pos;
61  Quaterniond rot;
62  sphereFilling(radius + offset, occurrence, nbOccurrences, rnd, pos, rot,
63  0.1);
64 
65  if (isClipped(pos, _clippingPlanes))
66  throw std::runtime_error("Instance is clipped");
67 
68  if (animationDetails.rotationSeed != 0)
69  rot = weightedRandomRotation(rot, animationDetails.rotationSeed, rnd,
70  animationDetails.rotationStrength);
71 
72  Transformation transformation;
73  transformation.setTranslation(pos);
74  transformation.setRotation(rot);
75  return transformation;
76 }
