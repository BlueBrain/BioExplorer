43bool inBounds(const Vector3f& point, const Boxd& bounds)
44{
45 const auto mi = bounds.getMin();
46 const auto ma = bounds.getMax();
47 return point.x >= mi.x && point.x < ma.x && point.y >= mi.y &&
48 point.y < ma.y && point.z >= mi.z && point.z < ma.z;
49}

93ModelDescriptorPtr CacheLoader::_importModel(std::stringstream& buffer,
94 const int32_t brickId) const
95{
96 auto model = _scene.createModel();
97
98 // Geometry
99 size_t nbSpheres = 0;
100 size_t nbCylinders = 0;
101 size_t nbCones = 0;
102 size_t nbMeshes = 0;
103 size_t nbVertices = 0;
104 size_t nbIndices = 0;
105 size_t nbNormals = 0;
106 size_t nbTexCoords = 0;
107
108 // Name
109 const auto name = _readString(buffer);
110
111 // Path
112 const auto path = _readString(buffer);
113
114 // Metadata
115 size_t nbElements;
116 ModelMetadata metadata;
117 buffer.read((char*)&nbElements, sizeof(size_t));
118 for (size_t i = 0; i < nbElements; ++i)
119 metadata[_readString(buffer)] = _readString(buffer);
120
121 if (brickId != UNDEFINED_BOX_ID)
122 {
123 metadata.clear();
124 metadata[METADATA_BRICK_ID] = std::to_string(brickId);
125 }
126
127 // Instances
128 std::vector<Transformation> transformations;
129 buffer.read((char*)&nbElements, sizeof(size_t));
130 for (size_t i = 0; i < nbElements; ++i)
131 {
132 Transformation tf;
133 Vector3d t;
134 Vector3d rc;
135 Vector3d s;
136 Quaterniond q;
137
138 buffer.read((char*)&t, sizeof(Vector3d));
139 tf.setTranslation(t);
140 buffer.read((char*)&rc, sizeof(Vector3d));
141 tf.setRotationCenter(rc);
142 buffer.read((char*)&q, sizeof(Quaterniond));
143 tf.setRotation(q);
144 buffer.read((char*)&s, sizeof(Vector3d));
145 tf.setScale(s);
146 transformations.push_back(tf);
147 }
148
149 size_t nbMaterials;
150 buffer.read((char*)&nbMaterials, sizeof(size_t));
151
152 // Materials
153 size_t materialId;
154 for (size_t i = 0; i < nbMaterials; ++i)
155 {
156 buffer.read((char*)&materialId, sizeof(size_t));
157
158 auto name = _readString(buffer);
159 auto material = model->createMaterial(materialId, name);
160
161 Vector3f value3f;
162 buffer.read((char*)&value3f, sizeof(Vector3f));
163 material->setDiffuseColor(value3f);
164 buffer.read((char*)&value3f, sizeof(Vector3f));
165 material->setSpecularColor(value3f);
166 float value;
167 buffer.read((char*)&value, sizeof(float));
168 material->setSpecularExponent(value);
169 buffer.read((char*)&value, sizeof(float));
170 material->setReflectionIndex(value);
171 buffer.read((char*)&value, sizeof(float));
172 material->setOpacity(value);
173 buffer.read((char*)&value, sizeof(float));
174 material->setRefractionIndex(value);
175 buffer.read((char*)&value, sizeof(float));
176 material->setEmission(value);
177 buffer.read((char*)&value, sizeof(float));
178 material->setGlossiness(value);
179
180 brayns::PropertyMap props;
181 double userParameter;
182 buffer.read((char*)&userParameter, sizeof(double));
183 props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, userParameter});
184
185 int32_t shadingMode;
186 buffer.read((char*)&shadingMode, sizeof(int32_t));
187 // props.setProperty({MATERIAL_PROPERTY_SHADING_MODE, shadingMode});
188 props.setProperty({MATERIAL_PROPERTY_SHADING_MODE, 1});
189
190 int32_t chameleonMode;
191 buffer.read((char*)&chameleonMode, sizeof(int32_t));
192 props.setProperty({MATERIAL_PROPERTY_CHAMELEON_MODE, chameleonMode});
193
194 int32_t nodeId;
195 buffer.read((char*)&nodeId, sizeof(int32_t));
196 props.setProperty({MATERIAL_PROPERTY_NODE_ID, nodeId});
197 material->updateProperties(props);
198 }
199
200 uint64_t bufferSize{0};
201
202 // Spheres
203 buffer.read((char*)&nbSpheres, sizeof(size_t));
204 for (size_t i = 0; i < nbSpheres; ++i)
205 {
206 buffer.read((char*)&materialId, sizeof(size_t));
207 buffer.read((char*)&nbElements, sizeof(size_t));
208 auto& spheres = model->getSpheres()[materialId];
209 spheres.resize(nbElements);
210 bufferSize = nbElements * sizeof(Sphere);
211 buffer.read((char*)spheres.data(), bufferSize);
212 }
213
214 // Cylinders
215 buffer.read((char*)&nbCylinders, sizeof(size_t));
216 for (size_t i = 0; i < nbCylinders; ++i)
217 {
218 buffer.read((char*)&materialId, sizeof(size_t));
219 buffer.read((char*)&nbElements, sizeof(size_t));
220 auto& cylinders = model->getCylinders()[materialId];
221 cylinders.resize(nbElements);
222 bufferSize = nbElements * sizeof(Cylinder);
223 buffer.read((char*)cylinders.data(), bufferSize);
224 }
225
226 // Cones
227 buffer.read((char*)&nbCones, sizeof(size_t));
228 for (size_t i = 0; i < nbCones; ++i)
229 {
230 buffer.read((char*)&materialId, sizeof(size_t));
231 buffer.read((char*)&nbElements, sizeof(size_t));
232 auto& cones = model->getCones()[materialId];
233 cones.resize(nbElements);
234 bufferSize = nbElements * sizeof(Cone);
235 buffer.read((char*)cones.data(), bufferSize);
236 }
237
238 // Meshes
239 buffer.read((char*)&nbMeshes, sizeof(size_t));
240 for (size_t i = 0; i < nbMeshes; ++i)
241 {
242 buffer.read((char*)&materialId, sizeof(size_t));
243 auto& meshes = model->getTriangleMeshes()[materialId];
244 // Vertices
245 buffer.read((char*)&nbVertices, sizeof(size_t));
246 if (nbVertices != 0)
247 {
248 bufferSize = nbVertices * sizeof(Vector3f);
249 meshes.vertices.resize(nbVertices);
250 buffer.read((char*)meshes.vertices.data(), bufferSize);
251 }
252
253 // Indices
254 buffer.read((char*)&nbIndices, sizeof(size_t));
255 if (nbIndices != 0)
256 {
257 bufferSize = nbIndices * sizeof(Vector3ui);
258 meshes.indices.resize(nbIndices);
259 buffer.read((char*)meshes.indices.data(), bufferSize);
260 }
261
262 // Normals
263 buffer.read((char*)&nbNormals, sizeof(size_t));
264 if (nbNormals != 0)
265 {
266 bufferSize = nbNormals * sizeof(Vector3f);
267 meshes.normals.resize(nbNormals);
268 buffer.read((char*)meshes.normals.data(), bufferSize);
269 }
270
271 // Texture coordinates
272 buffer.read((char*)&nbTexCoords, sizeof(size_t));
273 if (nbTexCoords != 0)
274 {
275 bufferSize = nbTexCoords * sizeof(Vector2f);
276 meshes.textureCoordinates.resize(nbTexCoords);
277 buffer.read((char*)meshes.textureCoordinates.data(), bufferSize);
278 }
279 }
280
281 // Streamlines
282 size_t nbStreamlines;
283 auto& streamlines = model->getStreamlines();
284 buffer.read((char*)&nbStreamlines, sizeof(size_t));
285 for (size_t i = 0; i < nbStreamlines; ++i)
286 {
287 StreamlinesData streamlineData;
288 // Id
289 size_t id;
290 buffer.read((char*)&id, sizeof(size_t));
291
292 // Vertex
293 buffer.read((char*)&nbElements, sizeof(size_t));
294 bufferSize = nbElements * sizeof(Vector4f);
295 streamlineData.vertex.resize(nbElements);
296 buffer.read((char*)streamlineData.vertex.data(), bufferSize);
297
298 // Vertex Color
299 buffer.read((char*)&nbElements, sizeof(size_t));
300 bufferSize = nbElements * sizeof(Vector4f);
301 streamlineData.vertexColor.resize(nbElements);
302 buffer.read((char*)streamlineData.vertexColor.data(), bufferSize);
303
304 // Indices
305 buffer.read((char*)&nbElements, sizeof(size_t));
306 bufferSize = nbElements * sizeof(int32_t);
307 streamlineData.indices.resize(nbElements);
308 buffer.read((char*)streamlineData.indices.data(), bufferSize);
309
310 streamlines[id] = streamlineData;
311 }
312
313 // SDF geometry
314 auto& sdfData = model->getSDFGeometryData();
315 buffer.read((char*)&nbElements, sizeof(size_t));
316
317 if (nbElements > 0)
318 {
319 // Geometries
320 sdfData.geometries.resize(nbElements);
321 bufferSize = nbElements * sizeof(SDFGeometry);
322 buffer.read((char*)sdfData.geometries.data(), bufferSize);
323
324 // SDF Indices
325 buffer.read((char*)&nbElements, sizeof(size_t));
326 for (size_t i = 0; i < nbElements; ++i)
327 {
328 buffer.read((char*)&materialId, sizeof(size_t));
329 size_t size;
330 buffer.read((char*)&size, sizeof(size_t));
331 bufferSize = size * sizeof(uint64_t);
332 sdfData.geometryIndices[materialId].resize(size);
333 buffer.read((char*)sdfData.geometryIndices[materialId].data(),
334 bufferSize);
335 }
336
337 // Neighbours
338 buffer.read((char*)&nbElements, sizeof(size_t));
339 sdfData.neighbours.resize(nbElements);
340
341 for (size_t i = 0; i < nbElements; ++i)
342 {
343 size_t size;
344 buffer.read((char*)&size, sizeof(size_t));
345 bufferSize = size * sizeof(uint64_t);
346 sdfData.neighbours[i].resize(size);
347 buffer.read((char*)sdfData.neighbours[i].data(), bufferSize);
348 }
349
350 // Neighbours flat
351 buffer.read((char*)&nbElements, sizeof(size_t));
352 bufferSize = nbElements * sizeof(uint64_t);
353 sdfData.neighboursFlat.resize(nbElements);
354 buffer.read((char*)sdfData.neighboursFlat.data(), bufferSize);
355 }
356
357 if (!transformations.empty())
358 {
359 auto modelDescriptor =
360 std::make_shared<ModelDescriptor>(std::move(model), name, path,
361 metadata);
362
363 bool first{true};
364 for (const auto& tf : transformations)
365 {
366 if (first)
367 {
368 modelDescriptor->setTransformation(tf);
369 first = false;
370 }
371
372 const ModelInstance instance(true, false, tf);
373 modelDescriptor->addInstance(instance);
374 }
375
376 const auto visible =
377 GeneralSettings::getInstance()->getModelVisibilityOnCreation();
378 modelDescriptor->setVisible(visible);
379 return modelDescriptor;
380 }
381 return nullptr;
382}

384std::vector<ModelDescriptorPtr> CacheLoader::importModelsFromFile(
385 const std::string& filename, const int32_t brickId,
386 const LoaderProgress& callback, const PropertyMap& properties) const
387{
388 std::vector<ModelDescriptorPtr> modelDescriptors;
389 PropertyMap props = _defaults;
390 props.merge(properties);
391
392 callback.updateProgress("Loading BioExplorer scene...", 0);
393 PLUGIN_DEBUG("Loading models from cache file: " << filename);
394 std::ifstream file(filename, std::ios::in | std::ios::binary);
395 if (!file.good())
396 PLUGIN_THROW("Could not open cache file " + filename);
397
398 std::stringstream buffer;
399 buffer << file.rdbuf();
400 file.close();
401
402 // File version
403 size_t version;
404 buffer.read((char*)&version, sizeof(size_t));
405 PLUGIN_DEBUG("Version: " << version);
406
407 // Models
408 size_t nbModels;
409 buffer.read((char*)&nbModels, sizeof(size_t));
410 PLUGIN_DEBUG("Models : " << nbModels);
411 for (size_t i = 0; i < nbModels; ++i)
412 {
413 auto modelDescriptor = _importModel(buffer, brickId);
414 if (modelDescriptor)
415 modelDescriptors.push_back(modelDescriptor);
416
417 callback.updateProgress("Loading models", float(i) / float(nbModels));
418 }
419
420 return modelDescriptors;
421}
422
423ModelDescriptorPtr CacheLoader::importFromFile(
424 const std::string& filename, const LoaderProgress& callback,
425 const PropertyMap& properties) const
426{
427 const auto modelDescriptors =
428 importModelsFromFile(filename, UNDEFINED_BOX_ID, callback, properties);
429 for (const auto modelDescriptor : modelDescriptors)
430 _scene.addModel(modelDescriptor);
431
432 return (!modelDescriptors.empty() ? modelDescriptors[0] : nullptr);
433}

435std::string CacheLoader::_readString(std::stringstream& buffer) const
436{
437 size_t size;
438 buffer.read((char*)&size, sizeof(size_t));
439 char* str = new char[size + 1];
440 buffer.read(str, size);
441 str[size] = 0;
442 std::string s{str};
443 delete[] str;
444 return s;
445}

447bool CacheLoader::_exportModel(const ModelDescriptorPtr modelDescriptor,
448 std::stringstream& buffer,
449 const Boxd& bounds) const
450{
451 uint64_t bufferSize{0};
452 const auto& model = modelDescriptor->getModel();
453
454 // Instances
455 std::vector<Transformation> transformations;
456 const auto& instances = modelDescriptor->getInstances();
457 bool first{true};
458 for (const auto& instance : instances)
459 {
460 if (first)
461 {
462 first = false;
463 const auto& tf = modelDescriptor->getTransformation();
464 if (!inBounds(tf.getTranslation(), bounds))
465 continue;
466 transformations.push_back(tf);
467 }
468 else
469 {
470 const auto& tf = instance.getTransformation();
471 if (!inBounds(tf.getTranslation(), bounds))
472 continue;
473 transformations.push_back(tf);
474 }
475 }
476
477 if (transformations.empty())
478 return false;
479
480 // Name
481 const auto& name = modelDescriptor->getName();
482 size_t size = name.length();
483 buffer.write((char*)&size, sizeof(size_t));
484 buffer.write((char*)name.c_str(), size);
485
486 // Path
487 const auto& path = modelDescriptor->getPath();
488 size = path.length();
489 buffer.write((char*)&size, sizeof(size_t));
490 buffer.write((char*)path.c_str(), size);
491
492 // Metadata
493 auto metadata = modelDescriptor->getMetadata();
494 size_t nbElements = metadata.size();
495 buffer.write((char*)&nbElements, sizeof(size_t));
496 for (const auto& data : metadata)
497 {
498 size = data.first.length();
499 buffer.write((char*)&size, sizeof(size_t));
500 buffer.write((char*)data.first.c_str(), size);
501 size = data.second.length();
502 buffer.write((char*)&size, sizeof(size_t));
503 buffer.write((char*)data.second.c_str(), size);
504 }
505
506 // Instances
507 nbElements = transformations.size();
508 buffer.write((char*)&nbElements, sizeof(size_t));
509 for (const auto& tf : transformations)
510 {
511 const auto& t = tf.getTranslation();
512 buffer.write((char*)&t, sizeof(brayns::Vector3d));
513 const auto& rc = tf.getRotationCenter();
514 buffer.write((char*)&rc, sizeof(brayns::Vector3d));
515 const auto& q = tf.getRotation();
516 buffer.write((char*)&q, sizeof(brayns::Quaterniond));
517 const auto& s = tf.getScale();
518 buffer.write((char*)&s, sizeof(brayns::Vector3d));
519 }
520
521 // Materials
522 const auto& materials = model.getMaterials();
523 nbElements = materials.size();
524 buffer.write((char*)&nbElements, sizeof(size_t));
525 for (const auto& material : materials)
526 {
527 buffer.write((char*)&material.first, sizeof(size_t));
528
529 auto name = material.second->getName();
530 size_t size = name.length();
531 buffer.write((char*)&size, sizeof(size_t));
532 buffer.write((char*)name.c_str(), size);
533
534 brayns::Vector3f value3f;
535 value3f = material.second->getDiffuseColor();
536 buffer.write((char*)&value3f, sizeof(brayns::Vector3f));
537 value3f = material.second->getSpecularColor();
538 buffer.write((char*)&value3f, sizeof(brayns::Vector3f));
539 float value = material.second->getSpecularExponent();
540 buffer.write((char*)&value, sizeof(float));
541 value = material.second->getReflectionIndex();
542 buffer.write((char*)&value, sizeof(float));
543 value = material.second->getOpacity();
544 buffer.write((char*)&value, sizeof(float));
545 value = material.second->getRefractionIndex();
546 buffer.write((char*)&value, sizeof(float));
547 value = material.second->getEmission();
548 buffer.write((char*)&value, sizeof(float));
549 value = material.second->getGlossiness();
550 buffer.write((char*)&value, sizeof(float));
551 double v = 1.0;
552 try
553 {
554 v = material.second->getProperty<double>(
555 MATERIAL_PROPERTY_USER_PARAMETER);
556 }
557 catch (const std::runtime_error&)
558 {
559 }
560 buffer.write((char*)&v, sizeof(double));
561
562 int32_t shadingMode = MaterialShadingMode::undefined_shading_mode;
563 try
564 {
565 shadingMode = material.second->getProperty<int32_t>(
566 MATERIAL_PROPERTY_SHADING_MODE);
567 }
568 catch (const std::runtime_error&)
569 {
570 }
571 buffer.write((char*)&shadingMode, sizeof(int32_t));
572
573 int32_t chameleonMode = MaterialChameleonMode::undefined_chameleon_mode;
574 try
575 {
576 shadingMode = material.second->getProperty<int32_t>(
577 MATERIAL_PROPERTY_CHAMELEON_MODE);
578 }
579 catch (const std::runtime_error&)
580 {
581 }
582 buffer.write((char*)&chameleonMode, sizeof(int32_t));
583
584 int32_t nodeId = 0;
585 try
586 {
587 shadingMode = material.second->getProperty<int32_t>(
588 MATERIAL_PROPERTY_NODE_ID);
589 }
590 catch (const std::runtime_error&)
591 {
592 }
593 buffer.write((char*)&nodeId, sizeof(int32_t));
594 }
595
596 // Spheres
597 const auto& spheresMap = model.getSpheres();
598 nbElements = spheresMap.size();
599 buffer.write((char*)&nbElements, sizeof(size_t));
600 for (const auto& spheres : spheresMap)
601 {
602 const auto materialId = spheres.first;
603 buffer.write((char*)&materialId, sizeof(size_t));
604
605 const auto& data = spheres.second;
606 nbElements = data.size();
607 buffer.write((char*)&nbElements, sizeof(size_t));
608 bufferSize = nbElements * sizeof(brayns::Sphere);
609 buffer.write((char*)data.data(), bufferSize);
610 }
611
612 // Cylinders
613 const auto& cylindersMap = model.getCylinders();
614 nbElements = cylindersMap.size();
615 buffer.write((char*)&nbElements, sizeof(size_t));
616 for (const auto& cylinders : cylindersMap)
617 {
618 const auto materialId = cylinders.first;
619 buffer.write((char*)&materialId, sizeof(size_t));
620
621 const auto& data = cylinders.second;
622 nbElements = data.size();
623 buffer.write((char*)&nbElements, sizeof(size_t));
624 bufferSize = nbElements * sizeof(brayns::Cylinder);
625 buffer.write((char*)data.data(), bufferSize);
626 }
627
628 // Cones
629 const auto& conesMap = model.getCones();
630 nbElements = conesMap.size();
631 buffer.write((char*)&nbElements, sizeof(size_t));
632 for (const auto& cones : conesMap)
633 {
634 const auto materialId = cones.first;
635 buffer.write((char*)&materialId, sizeof(size_t));
636
637 const auto& data = cones.second;
638 nbElements = data.size();
639 buffer.write((char*)&nbElements, sizeof(size_t));
640 bufferSize = nbElements * sizeof(brayns::Cone);
641 buffer.write((char*)data.data(), bufferSize);
642 }
643
644 // Meshes
645 const auto& trianglesMap = model.getTriangleMeshes();
646 nbElements = trianglesMap.size();
647 buffer.write((char*)&nbElements, sizeof(size_t));
648 for (const auto& meshes : trianglesMap)
649 {
650 const auto materialId = meshes.first;
651 buffer.write((char*)&materialId, sizeof(size_t));
652
653 const auto& data = meshes.second;
654
655 // Vertices
656 nbElements = data.vertices.size();
657 buffer.write((char*)&nbElements, sizeof(size_t));
658 bufferSize = nbElements * sizeof(brayns::Vector3f);
659 buffer.write((char*)data.vertices.data(), bufferSize);
660
661 // Indices
662 nbElements = data.indices.size();
663 buffer.write((char*)&nbElements, sizeof(size_t));
664 bufferSize = nbElements * sizeof(brayns::Vector3ui);
665 buffer.write((char*)data.indices.data(), bufferSize);
666
667 // Normals
668 nbElements = data.normals.size();
669 buffer.write((char*)&nbElements, sizeof(size_t));
670 bufferSize = nbElements * sizeof(brayns::Vector3f);
671 buffer.write((char*)data.normals.data(), bufferSize);
672
673 // Texture coordinates
674 nbElements = data.textureCoordinates.size();
675 buffer.write((char*)&nbElements, sizeof(size_t));
676 bufferSize = nbElements * sizeof(brayns::Vector2f);
677 buffer.write((char*)data.textureCoordinates.data(), bufferSize);
678 }
679
680 // Streamlines
681 const auto& streamlines = model.getStreamlines();
682 nbElements = streamlines.size();
683 buffer.write((char*)&nbElements, sizeof(size_t));
684 for (const auto& streamline : streamlines)
685 {
686 const auto& streamlineData = streamline.second;
687 // Id
688 size_t id = streamline.first;
689 buffer.write((char*)&id, sizeof(size_t));
690
691 // Vertex
692 nbElements = streamlineData.vertex.size();
693 buffer.write((char*)&nbElements, sizeof(size_t));
694 bufferSize = nbElements * sizeof(brayns::Vector4f);
695 buffer.write((char*)streamlineData.vertex.data(), bufferSize);
696
697 // Vertex Color
698 nbElements = streamlineData.vertexColor.size();
699 buffer.write((char*)&nbElements, sizeof(size_t));
700 bufferSize = nbElements * sizeof(brayns::Vector4f);
701 buffer.write((char*)streamlineData.vertexColor.data(), bufferSize);
702
703 // Indices
704 nbElements = streamlineData.indices.size();
705 buffer.write((char*)&nbElements, sizeof(size_t));
706 bufferSize = nbElements * sizeof(int32_t);
707 buffer.write((char*)streamlineData.indices.data(), bufferSize);
708 }
709
710 // SDF geometry
711 const auto& sdfData = model.getSDFGeometryData();
712 nbElements = sdfData.geometries.size();
713 buffer.write((char*)&nbElements, sizeof(size_t));
714
715 if (nbElements > 0)
716 {
717 // Geometries
718 bufferSize = nbElements * sizeof(brayns::SDFGeometry);
719 buffer.write((char*)sdfData.geometries.data(), bufferSize);
720
721 // SDF indices
722 nbElements = sdfData.geometryIndices.size();
723 buffer.write((char*)&nbElements, sizeof(size_t));
724 for (const auto& geometryIndex : sdfData.geometryIndices)
725 {
726 size_t materialId = geometryIndex.first;
727 buffer.write((char*)&materialId, sizeof(size_t));
728 nbElements = geometryIndex.second.size();
729 buffer.write((char*)&nbElements, sizeof(size_t));
730 bufferSize = nbElements * sizeof(uint64_t);
731 buffer.write((char*)geometryIndex.second.data(), bufferSize);
732 }
733
734 // Neighbours
735 nbElements = sdfData.neighbours.size();
736 buffer.write((char*)&nbElements, sizeof(size_t));
737 for (const auto& neighbour : sdfData.neighbours)
738 {
739 nbElements = neighbour.size();
740 buffer.write((char*)&nbElements, sizeof(size_t));
741 bufferSize = nbElements * sizeof(size_t);
742 buffer.write((char*)neighbour.data(), bufferSize);
743 }
744
745 // Neighbours flat
746 nbElements = sdfData.neighboursFlat.size();
747 buffer.write((char*)&nbElements, sizeof(size_t));
748 bufferSize = nbElements * sizeof(uint64_t);
749 buffer.write((char*)sdfData.neighboursFlat.data(), bufferSize);
750 }
751 return true;
752}

775 if (!file.good())
776 PLUGIN_THROW("Could not create BioExplorer file " + filename);
777
778 file.write((char*)buffer.str().c_str(), buffer.str().size());
779 file.close();

820void CacheLoader::exportToXYZ(const std::string& filename,
821 const XYZFileFormat fileFormat) const
822{
823 PLUGIN_INFO("Saving scene to XYZ file: " << filename);
824 std::ios_base::openmode flags = std::ios::out;
825 if (fileFormat == XYZFileFormat::xyz_binary ||
826 fileFormat == XYZFileFormat::xyzr_binary)
827 flags |= std::ios::binary;
828
829 std::ofstream file(filename, flags);
830 if (!file.good())
831 PLUGIN_THROW("Could not create XYZ file " + filename);
832
833 const auto clipPlanes = getClippingPlanes(_scene);
834
835 const auto& modelDescriptors = _scene.getModelDescriptors();
836 for (const auto modelDescriptor : modelDescriptors)
837 {
838 const auto& instances = modelDescriptor->getInstances();
839 for (const auto& instance : instances)
840 {
841 const auto& tf = instance.getTransformation();
842 const auto& model = modelDescriptor->getModel();
843 const auto& spheresMap = model.getSpheres();
844 for (const auto& spheres : spheresMap)
845 {
846 for (const auto& sphere : spheres.second)
847 {
848 const Vector3d center =
849 tf.getTranslation() +
850 tf.getRotation() *
851 (Vector3d(sphere.center) - tf.getRotationCenter());
852
853 const Vector3f c = center;
854 if (isClipped(c, clipPlanes))
855 continue;
856
857 switch (fileFormat)
858 {
859 case XYZFileFormat::xyz_binary:
860 case XYZFileFormat::xyzr_binary:
861 file.write((char*)&c.x, sizeof(float));
862 file.write((char*)&c.y, sizeof(float));
863 file.write((char*)&c.z, sizeof(float));
864 if (fileFormat == XYZFileFormat::xyzr_binary ||
865 fileFormat == XYZFileFormat::xyzrv_binary)
866 {
867 file.write((char*)&sphere.radius, sizeof(float));
868 if (fileFormat == XYZFileFormat::xyzrv_binary)
869 file.write((char*)&sphere.radius,
870 sizeof(float));
871 }
872 break;
873 case XYZFileFormat::xyz_ascii:
874 case XYZFileFormat::xyzr_ascii:
875 file << c.x << " " << c.y << " " << c.z;
876 if (fileFormat == XYZFileFormat::xyzr_ascii ||
877 fileFormat == XYZFileFormat::xyzrv_ascii)
878 {
879 file << " " << sphere.radius;
880 if (fileFormat == XYZFileFormat::xyzrv_ascii)
881 file << " " << sphere.radius;
882 }
883 file << std::endl;
884 break;
885 }
886 }
887 }
888 }
889 }
890 file.close();
891}
