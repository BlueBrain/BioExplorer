43 bool inBounds(const Vector3d& point, const Boxd& bounds)
44 {
45  const auto mi = bounds.getMin();
46  const auto ma = bounds.getMax();
47  return point.x >= mi.x && point.x < ma.x && point.y >= mi.y &&
48  point.y < ma.y && point.z >= mi.z && point.z < ma.z;
49 }

94 ModelDescriptorPtr CacheLoader::_importModel(std::stringstream& buffer,
95  const int32_t brickId) const
96 {
97  auto model = _scene.createModel();
98 
99  // Geometry
100  size_t nbSpheres = 0;
101  size_t nbCylinders = 0;
102  size_t nbCones = 0;
103  size_t nbMeshes = 0;
104  size_t nbVertices = 0;
105  size_t nbIndices = 0;
106  size_t nbNormals = 0;
107  size_t nbTexCoords = 0;
108 
109  // Name
110  const auto name = _readString(buffer);
111 
112  // Path
113  const auto path = _readString(buffer);
114 
115  // Metadata
116  size_t nbElements;
117  ModelMetadata metadata;
118  buffer.read((char*)&nbElements, sizeof(size_t));
119  for (size_t i = 0; i < nbElements; ++i)
120  metadata[_readString(buffer)] = _readString(buffer);
121 
122  if (brickId != UNDEFINED_BOX_ID)
123  {
124  metadata.clear();
125  metadata[METADATA_BRICK_ID] = std::to_string(brickId);
126  }
127 
128  // Instances
129  std::vector<Transformation> transformations;
130  buffer.read((char*)&nbElements, sizeof(size_t));
131  for (size_t i = 0; i < nbElements; ++i)
132  {
133  Transformation tf;
134  Vector3d t;
135  Vector3d rc;
136  Vector3d s;
137  Quaterniond q;
138 
139  buffer.read((char*)&t, sizeof(Vector3d));
140  tf.setTranslation(t);
141  buffer.read((char*)&rc, sizeof(Vector3d));
142  tf.setRotationCenter(rc);
143  buffer.read((char*)&q, sizeof(Quaterniond));
144  tf.setRotation(q);
145  buffer.read((char*)&s, sizeof(Vector3d));
146  tf.setScale(s);
147  transformations.push_back(tf);
148  }
149 
150  size_t nbMaterials;
151  buffer.read((char*)&nbMaterials, sizeof(size_t));
152 
153  // Materials
154  size_t materialId;
155  for (size_t i = 0; i < nbMaterials; ++i)
156  {
157  buffer.read((char*)&materialId, sizeof(size_t));
158 
159  auto name = _readString(buffer);
160  auto material = model->createMaterial(materialId, name);
161 
162  Vector3d value3f;
163  buffer.read((char*)&value3f, sizeof(Vector3d));
164  material->setDiffuseColor(value3f);
165  buffer.read((char*)&value3f, sizeof(Vector3d));
166  material->setSpecularColor(value3f);
167  double value;
168  buffer.read((char*)&value, sizeof(double));
169  material->setSpecularExponent(value);
170  buffer.read((char*)&value, sizeof(double));
171  material->setReflectionIndex(value);
172  buffer.read((char*)&value, sizeof(double));
173  material->setOpacity(value);
174  buffer.read((char*)&value, sizeof(double));
175  material->setRefractionIndex(value);
176  buffer.read((char*)&value, sizeof(double));
177  material->setEmission(value);
178  buffer.read((char*)&value, sizeof(double));
179  material->setGlossiness(value);
180 
181  brayns::PropertyMap props;
182  double userParameter;
183  buffer.read((char*)&userParameter, sizeof(double));
184  material->setUserParameter(value);
185 
186  int32_t shadingMode;
187  buffer.read((char*)&shadingMode, sizeof(int32_t));
188  material->setShadingMode(static_cast<MaterialShadingMode>(shadingMode));
189 
190  int32_t chameleonMode;
191  buffer.read((char*)&chameleonMode, sizeof(int32_t));
192  props.setProperty({MATERIAL_PROPERTY_CHAMELEON_MODE, chameleonMode});
193 
194  int32_t nodeId;
195  buffer.read((char*)&nodeId, sizeof(int32_t));
196  props.setProperty({MATERIAL_PROPERTY_NODE_ID, nodeId});
197  material->updateProperties(props);
198  }
199 
200  uint64_t bufferSize{0};
201 
202  // Spheres
203  buffer.read((char*)&nbSpheres, sizeof(size_t));
204  for (size_t i = 0; i < nbSpheres; ++i)
205  {
206  buffer.read((char*)&materialId, sizeof(size_t));
207  buffer.read((char*)&nbElements, sizeof(size_t));
208  auto& spheres = model->getSpheres()[materialId];
209  spheres.resize(nbElements);
210  bufferSize = nbElements * sizeof(Sphere);
211  buffer.read((char*)spheres.data(), bufferSize);
212  }
213 
214  // Cylinders
215  buffer.read((char*)&nbCylinders, sizeof(size_t));
216  for (size_t i = 0; i < nbCylinders; ++i)
217  {
218  buffer.read((char*)&materialId, sizeof(size_t));
219  buffer.read((char*)&nbElements, sizeof(size_t));
220  auto& cylinders = model->getCylinders()[materialId];
221  cylinders.resize(nbElements);
222  bufferSize = nbElements * sizeof(Cylinder);
223  buffer.read((char*)cylinders.data(), bufferSize);
224  }
225 
226  // Cones
227  buffer.read((char*)&nbCones, sizeof(size_t));
228  for (size_t i = 0; i < nbCones; ++i)
229  {
230  buffer.read((char*)&materialId, sizeof(size_t));
231  buffer.read((char*)&nbElements, sizeof(size_t));
232  auto& cones = model->getCones()[materialId];
233  cones.resize(nbElements);
234  bufferSize = nbElements * sizeof(Cone);
235  buffer.read((char*)cones.data(), bufferSize);
236  }
237 
238  // Meshes
239  buffer.read((char*)&nbMeshes, sizeof(size_t));
240  for (size_t i = 0; i < nbMeshes; ++i)
241  {
242  buffer.read((char*)&materialId, sizeof(size_t));
243  auto& meshes = model->getTriangleMeshes()[materialId];
244  // Vertices
245  buffer.read((char*)&nbVertices, sizeof(size_t));
246  if (nbVertices != 0)
247  {
248  bufferSize = nbVertices * sizeof(Vector3d);
249  meshes.vertices.resize(nbVertices);
250  buffer.read((char*)meshes.vertices.data(), bufferSize);
251  }
252 
253  // Indices
254  buffer.read((char*)&nbIndices, sizeof(size_t));
255  if (nbIndices != 0)
256  {
257  bufferSize = nbIndices * sizeof(Vector3ui);
258  meshes.indices.resize(nbIndices);
259  buffer.read((char*)meshes.indices.data(), bufferSize);
260  }
261 
262  // Normals
263  buffer.read((char*)&nbNormals, sizeof(size_t));
264  if (nbNormals != 0)
265  {
266  bufferSize = nbNormals * sizeof(Vector3d);
267  meshes.normals.resize(nbNormals);
268  buffer.read((char*)meshes.normals.data(), bufferSize);
269  }
270 
271  // Texture coordinates
272  buffer.read((char*)&nbTexCoords, sizeof(size_t));
273  if (nbTexCoords != 0)
274  {
275  bufferSize = nbTexCoords * sizeof(Vector2f);
276  meshes.textureCoordinates.resize(nbTexCoords);
277  buffer.read((char*)meshes.textureCoordinates.data(), bufferSize);
278  }
279  }
280 
281  // Streamlines
282  size_t nbStreamlines;
283  auto& streamlines = model->getStreamlines();
284  buffer.read((char*)&nbStreamlines, sizeof(size_t));
285  for (size_t i = 0; i < nbStreamlines; ++i)
286  {
287  StreamlinesData streamlineData;
288  // Id
289  size_t id;
290  buffer.read((char*)&id, sizeof(size_t));
291 
292  // Vertex
293  buffer.read((char*)&nbElements, sizeof(size_t));
294  bufferSize = nbElements * sizeof(Vector4f);
295  streamlineData.vertex.resize(nbElements);
296  buffer.read((char*)streamlineData.vertex.data(), bufferSize);
297 
298  // Vertex Color
299  buffer.read((char*)&nbElements, sizeof(size_t));
300  bufferSize = nbElements * sizeof(Vector4f);
301  streamlineData.vertexColor.resize(nbElements);
302  buffer.read((char*)streamlineData.vertexColor.data(), bufferSize);
303 
304  // Indices
305  buffer.read((char*)&nbElements, sizeof(size_t));
306  bufferSize = nbElements * sizeof(int32_t);
307  streamlineData.indices.resize(nbElements);
308  buffer.read((char*)streamlineData.indices.data(), bufferSize);
309 
310  streamlines[id] = streamlineData;
311  }
312 
313  // SDF geometry
314  auto& sdfData = model->getSDFGeometryData();
315  buffer.read((char*)&nbElements, sizeof(size_t));
316 
317  if (nbElements > 0)
318  {
319  // Geometries
320  sdfData.geometries.resize(nbElements);
321  bufferSize = nbElements * sizeof(SDFGeometry);
322  buffer.read((char*)sdfData.geometries.data(), bufferSize);
323 
324  // SDF Indices
325  buffer.read((char*)&nbElements, sizeof(size_t));
326  for (size_t i = 0; i < nbElements; ++i)
327  {
328  buffer.read((char*)&materialId, sizeof(size_t));
329  size_t size;
330  buffer.read((char*)&size, sizeof(size_t));
331  bufferSize = size * sizeof(uint64_t);
332  sdfData.geometryIndices[materialId].resize(size);
333  buffer.read((char*)sdfData.geometryIndices[materialId].data(),
334  bufferSize);
335  }
336 
337  // Neighbours
338  buffer.read((char*)&nbElements, sizeof(size_t));
339  sdfData.neighbours.resize(nbElements);
340 
341  for (size_t i = 0; i < nbElements; ++i)
342  {
343  size_t size;
344  buffer.read((char*)&size, sizeof(size_t));
345  bufferSize = size * sizeof(uint64_t);
346  sdfData.neighbours[i].resize(size);
347  buffer.read((char*)sdfData.neighbours[i].data(), bufferSize);
348  }
349 
350  // Neighbours flat
351  buffer.read((char*)&nbElements, sizeof(size_t));
352  bufferSize = nbElements * sizeof(uint64_t);
353  sdfData.neighboursFlat.resize(nbElements);
354  buffer.read((char*)sdfData.neighboursFlat.data(), bufferSize);
355  }
356 
357  if (!transformations.empty())
358  {
359  auto modelDescriptor =
360  std::make_shared<ModelDescriptor>(std::move(model), name, path,
361  metadata);
362 
363  bool first{true};
364  for (const auto& tf : transformations)
365  {
366  if (first)
367  {
368  modelDescriptor->setTransformation(tf);
369  first = false;
370  }
371 
372  const ModelInstance instance(true, false, tf);
373  modelDescriptor->addInstance(instance);
374  }
375 
376  const auto visible =
377  GeneralSettings::getInstance()->getModelVisibilityOnCreation();
378  modelDescriptor->setVisible(visible);
379  return modelDescriptor;
380  }
381  return nullptr;
382 }

384 std::vector<ModelDescriptorPtr> CacheLoader::importModelsFromFile(
385  const std::string& filename, const int32_t brickId,
386  const LoaderProgress& callback, const PropertyMap& properties) const
387 {
388  std::vector<ModelDescriptorPtr> modelDescriptors;
389  PropertyMap props = _defaults;
390  props.merge(properties);
391 
392  callback.updateProgress("Loading BioExplorer scene...", 0);
393  PLUGIN_DEBUG("Loading models from cache file: " << filename);
394  std::ifstream file(filename, std::ios::in | std::ios::binary);
395  if (!file.good())
396  PLUGIN_THROW("Could not open cache file " + filename);
397 
398  std::stringstream buffer;
399  buffer << file.rdbuf();
400  file.close();
401 
402  // File version
403  size_t version;
404  buffer.read((char*)&version, sizeof(size_t));
405  PLUGIN_DEBUG("Version: " << version);
406 
407  // Models
408  size_t nbModels;
409  buffer.read((char*)&nbModels, sizeof(size_t));
410  PLUGIN_DEBUG("Models : " << nbModels);
411  for (size_t i = 0; i < nbModels; ++i)
412  {
413  auto modelDescriptor = _importModel(buffer, brickId);
414  if (modelDescriptor)
415  modelDescriptors.push_back(modelDescriptor);
416 
417  callback.updateProgress("Loading models", double(i) / double(nbModels));
418  }
419 
420  return modelDescriptors;
421 }

423 ModelDescriptorPtr CacheLoader::importFromFile(
424  const std::string& filename, const LoaderProgress& callback,
425  const PropertyMap& properties) const
426 {
427  const auto modelDescriptors =
428  importModelsFromFile(filename, UNDEFINED_BOX_ID, callback, properties);
429  for (const auto modelDescriptor : modelDescriptors)
430  _scene.addModel(modelDescriptor);
431 
432  return (!modelDescriptors.empty() ? modelDescriptors[0] : nullptr);
433 }

435 std::string CacheLoader::_readString(std::stringstream& buffer) const
436 {
437  size_t size;
438  buffer.read((char*)&size, sizeof(size_t));
439  char* str = new char[size + 1];
440  buffer.read(str, size);
441  str[size] = 0;
442  std::string s{str};
443  delete[] str;
444  return s;
445 }

447 bool CacheLoader::_exportModel(const ModelDescriptorPtr modelDescriptor,
448  std::stringstream& buffer,
449  const Boxd& bounds) const
450 {
451  uint64_t bufferSize{0};
452  const auto& model = modelDescriptor->getModel();
453 
454  // Instances
455  std::vector<Transformation> transformations;
456  const auto& instances = modelDescriptor->getInstances();
457  bool first{true};
458  for (const auto& instance : instances)
459  {
460  if (first)
461  {
462  first = false;
463  const auto& tf = modelDescriptor->getTransformation();
464  if (!inBounds(tf.getTranslation(), bounds))
465  continue;
466  transformations.push_back(tf);
467  }
468  else
469  {
470  const auto& tf = instance.getTransformation();
471  if (!inBounds(tf.getTranslation(), bounds))
472  continue;
473  transformations.push_back(tf);
474  }
475  }
476 
477  if (transformations.empty())
478  return false;
479 
480  // Name
481  const auto& name = modelDescriptor->getName();
482  size_t size = name.length();
483  buffer.write((char*)&size, sizeof(size_t));
484  buffer.write((char*)name.c_str(), size);
485 
486  // Path
487  const auto& path = modelDescriptor->getPath();
488  size = path.length();
489  buffer.write((char*)&size, sizeof(size_t));
490  buffer.write((char*)path.c_str(), size);
491 
492  // Metadata
493  auto metadata = modelDescriptor->getMetadata();
494  size_t nbElements = metadata.size();
495  buffer.write((char*)&nbElements, sizeof(size_t));
496  for (const auto& data : metadata)
497  {
498  size = data.first.length();
499  buffer.write((char*)&size, sizeof(size_t));
500  buffer.write((char*)data.first.c_str(), size);
501  size = data.second.length();
502  buffer.write((char*)&size, sizeof(size_t));
503  buffer.write((char*)data.second.c_str(), size);
504  }
505 
506  // Instances
507  nbElements = transformations.size();
508  buffer.write((char*)&nbElements, sizeof(size_t));
509  for (const auto& tf : transformations)
510  {
511  const auto& t = tf.getTranslation();
512  buffer.write((char*)&t, sizeof(brayns::Vector3d));
513  const auto& rc = tf.getRotationCenter();
514  buffer.write((char*)&rc, sizeof(brayns::Vector3d));
515  const auto& q = tf.getRotation();
516  buffer.write((char*)&q, sizeof(brayns::Quaterniond));
517  const auto& s = tf.getScale();
518  buffer.write((char*)&s, sizeof(brayns::Vector3d));
519  }
520 
521  // Materials
522  const auto& materials = model.getMaterials();
523  nbElements = materials.size();
524  buffer.write((char*)&nbElements, sizeof(size_t));
525  for (const auto& material : materials)
526  {
527  buffer.write((char*)&material.first, sizeof(size_t));
528 
529  auto name = material.second->getName();
530  size_t size = name.length();
531  buffer.write((char*)&size, sizeof(size_t));
532  buffer.write((char*)name.c_str(), size);
533 
534  brayns::Vector3d value3f;
535  value3f = material.second->getDiffuseColor();
536  buffer.write((char*)&value3f, sizeof(brayns::Vector3d));
537  value3f = material.second->getSpecularColor();
538  buffer.write((char*)&value3f, sizeof(brayns::Vector3d));
539  double value = material.second->getSpecularExponent();
540  buffer.write((char*)&value, sizeof(double));
541  value = material.second->getReflectionIndex();
542  buffer.write((char*)&value, sizeof(double));
543  value = material.second->getOpacity();
544  buffer.write((char*)&value, sizeof(double));
545  value = material.second->getRefractionIndex();
546  buffer.write((char*)&value, sizeof(double));
547  value = material.second->getEmission();
548  buffer.write((char*)&value, sizeof(double));
549  value = material.second->getGlossiness();
550  buffer.write((char*)&value, sizeof(double));
551  value = material.second->getUserParameter();
552  buffer.write((char*)&value, sizeof(double));
553  int32_t shadingMode = material.second->getShadingMode();
554  buffer.write((char*)&shadingMode, sizeof(int32_t));
555 
556  int32_t chameleonMode = MaterialChameleonMode::undefined_chameleon_mode;
557  try
558  {
559  shadingMode = material.second->getProperty<int32_t>(
560  MATERIAL_PROPERTY_CHAMELEON_MODE);
561  }
562  catch (const std::runtime_error&)
563  {
564  }
565  buffer.write((char*)&chameleonMode, sizeof(int32_t));
566 
567  int32_t nodeId = 0;
568  try
569  {
570  shadingMode = material.second->getProperty<int32_t>(
571  MATERIAL_PROPERTY_NODE_ID);
572  }
573  catch (const std::runtime_error&)
574  {
575  }
576  buffer.write((char*)&nodeId, sizeof(int32_t));
577  }
578 
579  // Spheres
580  const auto& spheresMap = model.getSpheres();
581  nbElements = spheresMap.size();
582  buffer.write((char*)&nbElements, sizeof(size_t));
583  for (const auto& spheres : spheresMap)
584  {
585  const auto materialId = spheres.first;
586  buffer.write((char*)&materialId, sizeof(size_t));
587 
588  const auto& data = spheres.second;
589  nbElements = data.size();
590  buffer.write((char*)&nbElements, sizeof(size_t));
591  bufferSize = nbElements * sizeof(brayns::Sphere);
592  buffer.write((char*)data.data(), bufferSize);
593  }
594 
595  // Cylinders
596  const auto& cylindersMap = model.getCylinders();
597  nbElements = cylindersMap.size();
598  buffer.write((char*)&nbElements, sizeof(size_t));
599  for (const auto& cylinders : cylindersMap)
600  {
601  const auto materialId = cylinders.first;
602  buffer.write((char*)&materialId, sizeof(size_t));
603 
604  const auto& data = cylinders.second;
605  nbElements = data.size();
606  buffer.write((char*)&nbElements, sizeof(size_t));
607  bufferSize = nbElements * sizeof(brayns::Cylinder);
608  buffer.write((char*)data.data(), bufferSize);
609  }
610 
611  // Cones
612  const auto& conesMap = model.getCones();
613  nbElements = conesMap.size();
614  buffer.write((char*)&nbElements, sizeof(size_t));
615  for (const auto& cones : conesMap)
616  {
617  const auto materialId = cones.first;
618  buffer.write((char*)&materialId, sizeof(size_t));
619 
620  const auto& data = cones.second;
621  nbElements = data.size();
622  buffer.write((char*)&nbElements, sizeof(size_t));
623  bufferSize = nbElements * sizeof(brayns::Cone);
624  buffer.write((char*)data.data(), bufferSize);
625  }
626 
627  // Meshes
628  const auto& trianglesMap = model.getTriangleMeshes();
629  nbElements = trianglesMap.size();
630  buffer.write((char*)&nbElements, sizeof(size_t));
631  for (const auto& meshes : trianglesMap)
632  {
633  const auto materialId = meshes.first;
634  buffer.write((char*)&materialId, sizeof(size_t));
635 
636  const auto& data = meshes.second;
637 
638  // Vertices
639  nbElements = data.vertices.size();
640  buffer.write((char*)&nbElements, sizeof(size_t));
641  bufferSize = nbElements * sizeof(brayns::Vector3d);
642  buffer.write((char*)data.vertices.data(), bufferSize);
643 
644  // Indices
645  nbElements = data.indices.size();
646  buffer.write((char*)&nbElements, sizeof(size_t));
647  bufferSize = nbElements * sizeof(brayns::Vector3ui);
648  buffer.write((char*)data.indices.data(), bufferSize);
649 
650  // Normals
651  nbElements = data.normals.size();
652  buffer.write((char*)&nbElements, sizeof(size_t));
653  bufferSize = nbElements * sizeof(brayns::Vector3d);
654  buffer.write((char*)data.normals.data(), bufferSize);
655 
656  // Texture coordinates
657  nbElements = data.textureCoordinates.size();
658  buffer.write((char*)&nbElements, sizeof(size_t));
659  bufferSize = nbElements * sizeof(brayns::Vector2f);
660  buffer.write((char*)data.textureCoordinates.data(), bufferSize);
661  }
662 
663  // Streamlines
664  const auto& streamlines = model.getStreamlines();
665  nbElements = streamlines.size();
666  buffer.write((char*)&nbElements, sizeof(size_t));
667  for (const auto& streamline : streamlines)
668  {
669  const auto& streamlineData = streamline.second;
670  // Id
671  size_t id = streamline.first;
672  buffer.write((char*)&id, sizeof(size_t));
673 
674  // Vertex
675  nbElements = streamlineData.vertex.size();
676  buffer.write((char*)&nbElements, sizeof(size_t));
677  bufferSize = nbElements * sizeof(brayns::Vector4f);
678  buffer.write((char*)streamlineData.vertex.data(), bufferSize);
679 
680  // Vertex Color
681  nbElements = streamlineData.vertexColor.size();
682  buffer.write((char*)&nbElements, sizeof(size_t));
683  bufferSize = nbElements * sizeof(brayns::Vector4f);
684  buffer.write((char*)streamlineData.vertexColor.data(), bufferSize);
685 
686  // Indices
687  nbElements = streamlineData.indices.size();
688  buffer.write((char*)&nbElements, sizeof(size_t));
689  bufferSize = nbElements * sizeof(int32_t);
690  buffer.write((char*)streamlineData.indices.data(), bufferSize);
691  }
692 
693  // SDF geometry
694  const auto& sdfData = model.getSDFGeometryData();
695  nbElements = sdfData.geometries.size();
696  buffer.write((char*)&nbElements, sizeof(size_t));
697 
698  if (nbElements > 0)
699  {
700  // Geometries
701  bufferSize = nbElements * sizeof(brayns::SDFGeometry);
702  buffer.write((char*)sdfData.geometries.data(), bufferSize);
703 
704  // SDF indices
705  nbElements = sdfData.geometryIndices.size();
706  buffer.write((char*)&nbElements, sizeof(size_t));
707  for (const auto& geometryIndex : sdfData.geometryIndices)
708  {
709  size_t materialId = geometryIndex.first;
710  buffer.write((char*)&materialId, sizeof(size_t));
711  nbElements = geometryIndex.second.size();
712  buffer.write((char*)&nbElements, sizeof(size_t));
713  bufferSize = nbElements * sizeof(uint64_t);
714  buffer.write((char*)geometryIndex.second.data(), bufferSize);
715  }
716 
717  // Neighbours
718  nbElements = sdfData.neighbours.size();
719  buffer.write((char*)&nbElements, sizeof(size_t));
720  for (const auto& neighbour : sdfData.neighbours)
721  {
722  nbElements = neighbour.size();
723  buffer.write((char*)&nbElements, sizeof(size_t));
724  bufferSize = nbElements * sizeof(size_t);
725  buffer.write((char*)neighbour.data(), bufferSize);
726  }
727 
728  // Neighbours flat
729  nbElements = sdfData.neighboursFlat.size();
730  buffer.write((char*)&nbElements, sizeof(size_t));
731  bufferSize = nbElements * sizeof(uint64_t);
732  buffer.write((char*)sdfData.neighboursFlat.data(), bufferSize);
733  }
734  return true;

802 void CacheLoader::exportToXYZ(const std::string& filename,
803  const XYZFileFormat fileFormat) const
804 {
805  PLUGIN_INFO(3, "Saving scene to XYZ file: " << filename);
806  std::ios_base::openmode flags = std::ios::out;
807  if (fileFormat == XYZFileFormat::xyz_binary ||
808  fileFormat == XYZFileFormat::xyzr_binary)
809  flags |= std::ios::binary;
810 
811  std::ofstream file(filename, flags);
812  if (!file.good())
813  PLUGIN_THROW("Could not create XYZ file " + filename);
814 
815  const auto clipPlanes = getClippingPlanes(_scene);
816 
817  const auto& modelDescriptors = _scene.getModelDescriptors();
818  for (const auto modelDescriptor : modelDescriptors)
819  {
820  const auto& instances = modelDescriptor->getInstances();
821  for (const auto& instance : instances)
822  {
823  const auto& tf = instance.getTransformation();
824  const auto& model = modelDescriptor->getModel();
825  const auto& spheresMap = model.getSpheres();
826  for (const auto& spheres : spheresMap)
827  {
828  for (const auto& sphere : spheres.second)
829  {
830  const Vector3d center =
831  tf.getTranslation() +
832  tf.getRotation() *
833  (Vector3d(sphere.center) - tf.getRotationCenter());
834 
835  const Vector3d c = center;
836  if (isClipped(c, clipPlanes))
837  continue;
838 
839  switch (fileFormat)
840  {
841  case XYZFileFormat::xyz_binary:
842  case XYZFileFormat::xyzr_binary:
843  case XYZFileFormat::xyzrv_binary:
844  file.write((char*)&c.x, sizeof(double));
845  file.write((char*)&c.y, sizeof(double));
846  file.write((char*)&c.z, sizeof(double));
847  if (fileFormat == XYZFileFormat::xyzr_binary ||
848  fileFormat == XYZFileFormat::xyzrv_binary)
849  {
850  file.write((char*)&sphere.radius, sizeof(double));
851  if (fileFormat == XYZFileFormat::xyzrv_binary)
852  file.write((char*)&sphere.radius,
853  sizeof(double));
854  }
855  break;
856  case XYZFileFormat::xyz_ascii:
857  case XYZFileFormat::xyzr_ascii:
858  case XYZFileFormat::xyzrv_ascii:
859  file << c.x << " " << c.y << " " << c.z;
860  if (fileFormat == XYZFileFormat::xyzr_ascii ||
861  fileFormat == XYZFileFormat::xyzrv_ascii)
862  {
863  file << " " << sphere.radius;
864  if (fileFormat == XYZFileFormat::xyzrv_ascii)
865  file << " " << sphere.radius;
866  }
867  file << std::endl;
868  break;
869  case XYZFileFormat::unspecified:
870  PLUGIN_THROW("A file format must be specified");
871  break;
872  }
873  }
874  }
875  }
876  }
877  file.close();
878 }
