84 void OptiXEngine::_createCameras()
85 {
86  _camera = createCamera();
87 
88  const bool isStereo = _parametersManager.getApplicationParameters().isStereo();
89  Property stereoProperty{CAMERA_PROPERTY_STEREO, isStereo, {CAMERA_PROPERTY_STEREO}};
90  stereoProperty.markReadOnly();
91  Property fovy{CAMERA_PROPERTY_FOVY, 45., .1, 360., {"Field of view"}};
92  Property aspect{CAMERA_PROPERTY_ASPECT, 1., {"Aspect ratio"}};
93  aspect.markReadOnly();
94  Property eyeSeparation{CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE, 0.0635, {"Eye separation"}};
95 
96  auto camera = std::make_shared<OptiXPerspectiveCamera>();
97 
98  PropertyMap properties;
99  properties.setProperty(fovy);
100  properties.setProperty(aspect);
101  properties.setProperty({CAMERA_PROPERTY_APERTURE_RADIUS, 0., {"Aperture radius"}});
102  properties.setProperty({CAMERA_PROPERTY_FOCUS_DISTANCE, 1., {"Focus Distance"}});
103  properties.setProperty({CAMERA_PROPERTY_HEIGHT, 1., {CAMERA_PROPERTY_HEIGHT}});
104  if (isStereo)
105  {
106  properties.setProperty(stereoProperty);
107  properties.setProperty(eyeSeparation);
108  properties.setProperty({CAMERA_PROPERTY_ZERO_PARALLAX_PLANE, 1., {"Zero parallax plane"}});
109  }
110 
111  OptiXContext::getInstance().addCamera("perspective", camera);
112  addCameraType("perspective", properties);
113
