30 vec3f treeWalker(const uniform VectorFieldsRenderer* uniform self, const vec3f& point, const varying float distance,
31  const varying float cutoff, const varying int32 index = 0)
32 {
33  const uint64 begin = self->userData[self->startIndices + index * 2];
34  const uint64 end = self->userData[self->startIndices + index * 2 + 1];
35  const uint64 idxData = self->startData + index * FIELD_VECTOR_DATA_SIZE;
36 
37  if (idxData >= self->userDataSize)
38  return make_vec3f(0.f);
39 
40  if (begin == 0 && end == 0)
41  {
42  // Leaf
43  const vec3f vectorDirection = make_vec3f(self->userData[idxData + FIELD_VECTOR_OFFSET_DIRECTION_X],
44  self->userData[idxData + FIELD_VECTOR_OFFSET_DIRECTION_Y],
45  self->userData[idxData + FIELD_VECTOR_OFFSET_DIRECTION_Z]);
46  return vectorDirection / (distance * distance);
47  }
48 
49  vec3f voxelValue = make_vec3f(0.f);
50  for (unsigned int64 childIndex = begin; childIndex <= end; ++childIndex)
51  {
52  const uint64 idx = self->startData + childIndex * FIELD_VECTOR_DATA_SIZE;
53  const vec3f childPosition = make_vec3f(self->userData[idx + FIELD_VECTOR_OFFSET_POSITION_X],
54  self->userData[idx + FIELD_VECTOR_OFFSET_POSITION_Y],
55  self->userData[idx + FIELD_VECTOR_OFFSET_POSITION_Z]);
56  const float d = length(point - childPosition);
57 
58  if (d >= cutoff)
59  {
60  // Child is further than the cutoff distance, no need to evaluate events in the child node, we take the
61  // precomputed value of node instead
62  const vec3f vectorDirection = make_vec3f(self->userData[idx + FIELD_VECTOR_OFFSET_DIRECTION_X],
63  self->userData[idx + FIELD_VECTOR_OFFSET_DIRECTION_Y],
64  self->userData[idx + FIELD_VECTOR_OFFSET_DIRECTION_Z]);
65  voxelValue = voxelValue + vectorDirection / (d * d);
66  }
67  else
68  // Dive into the child node and compute its contents
69  voxelValue = voxelValue + treeWalker(self, point, d, cutoff / 2.f, childIndex);
70  }
71  return voxelValue;
72 }

74 inline vec3f VectorFieldsRenderer_shadeRay(const uniform VectorFieldsRenderer* uniform self,
75  varying ScreenSample& sample)
76 {
77  vec4f finalColor = make_vec4f(0.f);
78  const Ray ray = sample.ray;
79 
80  if (sample.sampleID.z == 0)
81  {
82  sample.alpha = 0.f;
83  sample.z = inf;
84  }
85 
86  const uniform box3f box = make_box3f(self->offset, self->offset + self->size * self->spacing);
87 
88  const float diag = max(max(self->size.x, self->size.y), self->size.z);
89  const float t_step = max(self->minRayStep, diag / (float)self->nbRaySteps);
90 
91  float t0, t1;
92  intersectBox(ray, box, t0, t1);
93 
94  const float random =
95  t_step * (float)(sample.sampleID.z % self->nbRayRefinementSteps) / (float)(self->nbRayRefinementSteps);
96 
97  float t = max(0.f, t0) + random;
98  while (t < t1 && finalColor.w < 1.f)
99  {
100  const vec3f p = ray.org + t * ray.dir;
101  const vec3f point = (p - self->offset) / self->spacing;
102  const vec3f sampleValue = treeWalker(self, point, self->distance, self->cutoff);
103  const float vectorLength =
104  clamp(length(sampleValue), self->transferFunction->valueRange.x, self->transferFunction->valueRange.y);
105  const float mapOpacity = self->transferFunction->getOpacityForValue(self->transferFunction, vectorLength);
106  if (mapOpacity > 0.f)
107  {
108  const vec3f mapColor = self->transferFunction->getColorForValue(self->transferFunction, vectorLength);
109  if (self->showVectorDirections)
110  {
111  const vec3f v = normalize(sampleValue);
112  const vec3f vectorColor = make_vec3f(0.5f + v.x * 0.5f, 0.5f + v.y * 0.5f, 0.5f + v.z * 0.5f);
113  compose(make_vec4f(vectorColor, mapOpacity), finalColor, self->alphaCorrection);
114  }
115  else
116  compose(make_vec4f(mapColor, mapOpacity), finalColor, self->alphaCorrection);
117  }
118 
119  t += t_step;
120  }
121 
122  sample.alpha = finalColor.w;
123  return clamp(make_vec3f(finalColor) * self->exposure, make_vec3f(0.f), make_vec3f(1.f));
124 }

126 void VectorFieldsRenderer_renderSample(uniform Renderer* uniform _self, void* uniform perFrameData,
127  varying ScreenSample& sample)
128 {
129  uniform VectorFieldsRenderer* uniform self = (uniform VectorFieldsRenderer * uniform) _self;
130  sample.ray.time = self->super.timestamp;
131  if (self->super.anaglyphEnabled)
132  {
133  ScreenSample s = sample;
134  s.ray.org = s.ray.org - self->super.anaglyphIpdOffset;
135  const vec3f colorLeft = VectorFieldsRenderer_shadeRay(self, s);
136  const vec3f leftAnaglyphColor = getAnaglyphLeftColor(colorLeft);
137 
138  s = sample;
139  s.ray.org = s.ray.org + self->super.anaglyphIpdOffset;
140  const vec3f colorRight = VectorFieldsRenderer_shadeRay(self, s);
141  const vec3f rightAnaglyphColor = getAnaglyphRightColor(colorRight);
142 
143  sample.rgb = leftAnaglyphColor + rightAnaglyphColor;
144  }
145  else
146  sample.rgb = VectorFieldsRenderer_shadeRay(self, sample);
147 }
