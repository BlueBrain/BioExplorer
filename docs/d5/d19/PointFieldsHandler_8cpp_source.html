62 void PointFieldsHandler::_buildOctree()
63 {
64  PLUGIN_INFO(3, "Building Point Octree");
65 
66  auto& scene = _engine.getScene();
67  const auto& clipPlanes = getClippingPlanes(scene);
68 
69  OctreePoints points;
70  uint32_t count{0};
71  const uint32_t densityRatio = 1.f / _density;
72 
73  Boxd bounds;
74  const auto& modelDescriptors = scene.getModelDescriptors();
75  for (const auto modelDescriptor : modelDescriptors)
76  {
77  if (!_modelIds.empty())
78  {
79  const auto modelId = modelDescriptor->getModelID();
80  const auto it = std::find(_modelIds.begin(), _modelIds.end(), modelId);
81  if (it == _modelIds.end())
82  continue;
83  }
84 
85  const auto& instances = modelDescriptor->getInstances();
86  for (const auto& instance : instances)
87  {
88  const auto& tf = instance.getTransformation();
89  const auto& model = modelDescriptor->getModel();
90  const auto& spheresMap = model.getSpheres();
91  for (const auto& spheres : spheresMap)
92  {
93  if (spheres.first != BOUNDINGBOX_MATERIAL_ID && spheres.first != SECONDARY_MODEL_MATERIAL_ID)
94  for (const auto& sphere : spheres.second)
95  {
96  const Vector3f center =
97  tf.getTranslation() + tf.getRotation() * (Vector3d(sphere.center) - tf.getRotationCenter());
98 
99  const Vector3d c = center;
100  if (isClipped(c, clipPlanes))
101  {
102  ++count;
103  continue;
104  }
105 
106  if (count % densityRatio == 0)
107  {
108  bounds.merge(center + sphere.radius);
109  bounds.merge(center - sphere.radius);
110 
111  OctreePoint point;
112  point.position = center;
113  point.radius = sphere.radius;
114  point.value = sphere.radius;
115  points.push_back(point);
116  }
117  ++count;
118  }
119  }
120  }
121  }
122 
123  // Events bounds
124  Vector3f sceneSize = bounds.getSize();
125  Vector3f center = bounds.getCenter();
126  Vector3f extendedHalfSize = sceneSize * 0.5f;
127 
128  // Expand volume by 25%
129  const float boundsExpansion = 1.25f;
130  bounds.merge(center + extendedHalfSize * boundsExpansion);
131  bounds.merge(center - extendedHalfSize * boundsExpansion);
132  sceneSize = bounds.getSize();
133  center = bounds.getCenter();
134  extendedHalfSize = sceneSize * 0.5f;
135 
136  // Compute volume information
137  const Vector3f minAABB = center - extendedHalfSize;
138  const Vector3f maxAABB = center + extendedHalfSize;
139 
140  // Build acceleration structure
141  const PointOctree accelerator(points, _voxelSize, minAABB, maxAABB);
142  const auto& indices = accelerator.getFlatIndices();
143  const auto& data = accelerator.getFlatData();
144 
145  const uint32_t volumeSize = accelerator.getVolumeSize();
146  _offset = center - extendedHalfSize;
147  _dimensions = accelerator.getVolumeDimensions();
148  _spacing = sceneSize / Vector3f(_dimensions);
149 
150  const auto& params = _engine.getParametersManager().getApplicationParameters();
151  const auto& engineName = params.getEngine();
152 #ifdef USE_OSPRAY
153  if (engineName == ENGINE_OSPRAY)
154  {
155  _frameData.push_back(_offset.x);
156  _frameData.push_back(_offset.y);
157  _frameData.push_back(_offset.z);
158  _frameData.push_back(_spacing.x);
159  _frameData.push_back(_spacing.y);
160  _frameData.push_back(_spacing.z);
161  _frameData.push_back(_dimensions.x);
162  _frameData.push_back(_dimensions.y);
163  _frameData.push_back(_dimensions.z);
164  _frameData.push_back(accelerator.getOctreeSize());
165  _frameData.push_back(indices.size());
166  _frameData.insert(_frameData.end(), indices.begin(), indices.end());
167  _frameData.insert(_frameData.end(), data.begin(), data.end());
168  _frameSize = _frameData.size();
169 
170  const size_t materialId = 0;
171  auto material = _model->createMaterial(0, "Octree");
172  const TriangleMesh mesh = createBox(bounds.getMin(), bounds.getMax());
173  _model->getTriangleMeshes()[materialId] = mesh;
174  _model->markInstancesDirty();
175  }
176 #endif
177 
178 #ifdef USE_OPTIX6
179  if (engineName == ENGINE_OPTIX_6)
180  {
181  auto octreeVolume = _model->createOctreeVolume(_dimensions, _spacing, DataType::FLOAT);
182  auto optixOctreeVolume = dynamic_cast<core::engine::optix::OptiXOctreeVolume*>(octreeVolume.get());
183  optixOctreeVolume->setOctree(_offset, indices, data, OctreeDataType::point);
184  _frameData.clear();
185  _frameSize = 0;
186  }
187 #endif
188 
189  PLUGIN_INFO(1, "--------------------------------------------");
190  PLUGIN_INFO(1, "Point Octree information (" << points.size() << " points)");
191  PLUGIN_INFO(1, "--------------------------------------------");
192  PLUGIN_INFO(1, "Scene AABB : " << bounds);
193  PLUGIN_INFO(1, "Scene dimension : " << sceneSize);
194  PLUGIN_INFO(1, "Element spacing : " << _spacing);
195  PLUGIN_INFO(1, "Volume dimensions : " << _dimensions);
196  PLUGIN_INFO(1, "Element offset : " << _offset);
197  PLUGIN_INFO(1, "Volume size : " << volumeSize << " bytes");
198  PLUGIN_INFO(1, "Indices size : " << indices.size());
199  PLUGIN_INFO(1, "Data size : " << _frameSize);
200  PLUGIN_INFO(1, "PointOctree depth : " << accelerator.getOctreeDepth());
201  PLUGIN_INFO(1, "--------------------------------------------");
202  _octreeInitialized = true;
203 }
