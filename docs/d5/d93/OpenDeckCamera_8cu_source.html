66 __device__ void getClippingValues(const float3& ray_origin,
67  const float3& ray_direction, float& near,
68  float& far)
69 {
70  for (int i = 0; i < nb_clip_planes; ++i)
71  {
72  float4 clipPlane = clip_planes[i];
73  const float3 planeNormal = {clipPlane.x, clipPlane.y, clipPlane.z};
74  float rn = dot(ray_direction, planeNormal);
75  if (rn == 0.f)
76  rn = scene_epsilon;
77  float d = clipPlane.w;
78  float t = -(dot(planeNormal, ray_origin) + d) / rn;
79  if (rn > 0.f) // opposite direction plane
80  near = max(near, t);
81  else
82  far = min(far, t);
83  }
84 }

86 // Pass 'seed' by reference to keep randomness state
87 __device__ float3 launch(unsigned int& seed, const float2 screen,
88  const bool use_randomness)
89 {
90  float eyeDelta = 0.0f;
91  float alpha = 0.0f;
92  float3 dPx, dPy;
93 
94  float2 sample = make_float2(launch_index) / screen;
95 
96  if (segmentID <= 13 && segmentID % 2 == 0)
97  {
98  eyeDelta = HALF_IPD;
99  unsigned int angularOffset = segmentID / 2;
100 
101  if (segmentID == 0)
102  alpha = sample.x * FULL_ANGLE;
103  else if (segmentID == 12)
104  alpha = PI - FULL_ANGLE + sample.x * FULL_ANGLE;
105  else
106  alpha = angularOffset * (FULL_ANGLE - OPENDECK_BEZEL_ANGLE) +
107  sample.x * FULL_ANGLE;
108  }
109  else if (segmentID <= 13 && segmentID % 2 == 1)
110  {
111  eyeDelta = -HALF_IPD;
112  unsigned int angularOffset = segmentID / 2;
113  if (segmentID == 1)
114  alpha = sample.x * FULL_ANGLE;
115  else if (segmentID == 13)
116  alpha = PI - FULL_ANGLE + sample.x * FULL_ANGLE;
117  else
118  alpha = angularOffset * (FULL_ANGLE - OPENDECK_BEZEL_ANGLE) +
119  sample.x * FULL_ANGLE;
120  }
121  else if (segmentID == 14)
122  {
123  eyeDelta = HALF_IPD;
124  }
125  else if (segmentID == 15)
126  {
127  eyeDelta = -HALF_IPD;
128  }
129 
130  float3 pixelPos;
131  if (segmentID <= 13)
132  {
133  pixelPos.x = OPENDECK_RADIUS * -cosf(alpha);
134  pixelPos.y = OPENDECK_METALSTRIPE_HEIGHT + OPENDECK_HEIGHT * sample.y;
135  pixelPos.z = OPENDECK_RADIUS * -sinf(alpha);
136 
137  dPx = make_float3(FULL_ANGLE * OPENDECK_RADIUS * sinf(alpha), 0.0f,
138  FULL_ANGLE * OPENDECK_RADIUS * -cosf(alpha));
139  dPy = make_float3(0.0f, OPENDECK_HEIGHT, 0.0f);
140  }
141  else if (segmentID > 13)
142  {
143  pixelPos.x = 2.0f * OPENDECK_RADIUS * (sample.x - 0.5f);
144  pixelPos.y = 0.0f;
145  pixelPos.z = -OPENDECK_RADIUS * sample.y;
146 
147  dPx = make_float3(2.0f * OPENDECK_RADIUS, 0.0f, 0.0f);
148  dPy = make_float3(0.0f, 0.0f, -OPENDECK_RADIUS);
149  }
150 
151  // The tracking model of the 3d glasses is inversed
152  // so we need to negate CamDu here.
153  const float3 eyeDeltaPos = -headUVec * eyeDelta;
154 
155  optix::Matrix3x3 transform;
156  transform.setCol(0, U);
157  transform.setCol(1, V);
158  transform.setCol(2, W);
159 
160  const float3 d = pixelPos - headPos + eyeDeltaPos;
161  const float dotD = dot(d, d);
162  const float denom = pow(dotD, 1.5f);
163  float3 dir = normalize(d);
164 
165  float3 dirDx = (dotD * dPx - dot(d, dPx) * d) / (denom * screen.x);
166  float3 dirDy = (dotD * dPy - dot(d, dPy) * d) / (denom * screen.y);
167 
168  PerRayData_radiance prd;
169  prd.importance = 1.f;
170  prd.depth = 0;
171  prd.rayDdx = transform * dirDx;
172  prd.rayDdy = transform * dirDy;
173  dir = transform * dir;
174 
175  const float3 org = eye + headPos - eyeDeltaPos;
176  float near = scene_epsilon;
177  float far = INFINITY;
178 
179  getClippingValues(org, dir, near, far);
180  optix::Ray ray(org, dir, radiance_ray_type, near, far);
181 
182  rtTrace(top_object, ray, prd);
183 
184  return prd.result;
185 }

187 RT_PROGRAM void openDeckCamera()
188 {
189  const size_t2 screen = output_buffer.size();
190  const float2 screen_f = make_float2(screen);
191 
192  unsigned int seed =
193  tea<16>(screen.x * launch_index.y + launch_index.x, frame);
194 
195  const int num_samples = max(1, samples_per_pixel);
196  // We enable randomness if we are using subpixel sampling or accumulation
197  const bool use_randomness = frame > 0 || num_samples > 1;
198 
199  float3 result = make_float3(0, 0, 0);
200  for (int i = 0; i < num_samples; i++)
201  result += launch(seed, screen_f, use_randomness);
202  result /= num_samples;
203 
204  float4 acc_val;
205  if (frame > 0)
206  {
207  acc_val = accum_buffer[launch_index];
208  acc_val = lerp(acc_val, make_float4(result, 0.f),
209  1.0f / static_cast<float>(frame + 1));
210  }
211  else
212  acc_val = make_float4(result, 1.f);
213 
214  output_buffer[launch_index] = make_color(make_float3(acc_val));
215 
216  if (accum_buffer.size().x > 1 && accum_buffer.size().y > 1)
217  accum_buffer[launch_index] = acc_val;
218 }
