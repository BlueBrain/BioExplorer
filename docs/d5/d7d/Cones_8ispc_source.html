53 unmasked void Cones_bounds(const RTCBoundsFunctionArguments* uniform args)
54 {
55  const uniform Cones* uniform self = (uniform Cones * uniform) args->geometryUserPtr;
56  const uniform Cone* uniform conePtr = safeIncrement(self->useSafeIncrement, self->data, args->primID);
57 
58  uniform float extent = conePtr->centerRadius;
59  const uniform float upRadius = conePtr->upRadius;
60 
61  if (upRadius > extent)
62  extent = upRadius;
63 
64  const uniform vec3f v0 = conePtr->center;
65  const uniform vec3f v1 = conePtr->up;
66 
67  box3fa* uniform bbox = (box3fa * uniform) args->bounds_o;
68  *bbox = make_box3fa(min(v0, v1) - make_vec3f(extent), max(v0, v1) + make_vec3f(extent));
69 }

71 unmasked void Cones_intersect(const RTCIntersectFunctionNArguments* uniform args)
72 {
73  const uniform Cones* uniform self = (uniform Cones * uniform) args->geometryUserPtr;
74  const uniform int primID = args->primID;
75 
76  const uniform Cone* uniform conePtr = safeIncrement(self->useSafeIncrement, self->data, primID);
77 
78  uniform float radius0 = conePtr->centerRadius;
79  uniform float radius1 = conePtr->upRadius;
80 
81  uniform vec3f v0 = conePtr->center;
82  uniform vec3f v1 = conePtr->up;
83 
84  if (radius0 < radius1)
85  {
86  // swap radii and positions, so radius0 and v0 are always at the bottom
87  uniform float tmpRadius = radius1;
88  radius1 = radius0;
89  radius0 = tmpRadius;
90 
91  uniform vec3f tmpPos = v1;
92  v1 = v0;
93  v0 = tmpPos;
94  }
95 
96  const vec3f upVector = v1 - v0;
97  const float upLength = length(upVector);
98 
99  // Compute the height of the full cone, in order to obtain its vertex
100  const float deltaRadius = radius0 - radius1;
101  const float tanA = deltaRadius / upLength;
102  const float coneHeight = radius0 / tanA;
103  const float squareTanA = tanA * tanA;
104  const float div = sqrtf(1.f + squareTanA);
105  if (div == 0.f)
106  return;
107  const float cosA = 1.f / div;
108 
109  const vec3f V = v0 + normalize(upVector) * coneHeight;
110  const vec3f v = normalize(v0 - V);
111 
112  // Normal of the plane P determined by V and ray
113  varying Ray* uniform ray = (varying Ray * uniform) args->rayhit;
114  vec3f n = normalize(cross(ray->dir, V - ray->org));
115  const float dotNV = dot(n, v);
116  if (dotNV > 0.f)
117  n = neg(n);
118 
119  const float squareCosTheta = 1.f - dotNV * dotNV;
120  const float cosTheta = sqrtf(squareCosTheta);
121  if (cosTheta < cosA)
122  return; // no intersection
123 
124  if (squareCosTheta == 0.f)
125  return;
126 
127  const float squareTanTheta = (1.f - squareCosTheta) / squareCosTheta;
128  const float tanTheta = sqrtf(squareTanTheta);
129 
130  // Compute u-v-w coordinate system
131  const vec3f u = normalize(cross(v, n));
132  const vec3f w = normalize(cross(u, v));
133 
134  // Circle intersection of cone with plane P
135  const vec3f uComponent = sqrtf(squareTanA - squareTanTheta) * u;
136  const vec3f vwComponent = v + tanTheta * w;
137  const vec3f delta1 = vwComponent + uComponent;
138  const vec3f delta2 = vwComponent - uComponent;
139  const vec3f rayApex = V - ray->org;
140 
141  const vec3f normal1 = cross(ray->dir, delta1);
142  const float length1 = length(normal1);
143 
144  if (length1 == 0.f)
145  return;
146 
147  const float r1 = dot(cross(rayApex, delta1), normal1) / (length1 * length1);
148 
149  const vec3f normal2 = cross(ray->dir, delta2);
150  const float length2 = length(normal2);
151 
152  if (length2 == 0.f)
153  return;
154 
155  const float r2 = dot(cross(rayApex, delta2), normal2) / (length2 * length2);
156 
157  float t_in = r1;
158  float t_out = r2;
159  if (r2 > 0.f)
160  {
161  if (r1 > 0.f)
162  {
163  if (r1 > r2)
164  {
165  t_in = r2;
166  t_out = r1;
167  }
168  }
169  else
170  t_in = r2;
171  }
172 
173  if (t_in > ray->t0 && t_in < ray->t)
174  {
175  const vec3f p1 = ray->org + t_in * ray->dir;
176  // consider only the parts within the extents of the truncated cone
177  if (dot(p1 - v1, v) > 0.f && dot(p1 - v0, v) < 0.f)
178  {
179  ray->primID = primID;
180  ray->geomID = self->super.geomID;
181  ray->instID = args->context->instID[0];
182  ray->t = t_in;
183  const vec3f surfaceVec = normalize(p1 - V);
184  ray->Ng = cross(cross(v, surfaceVec), surfaceVec);
185  return;
186  }
187  }
188  if (t_out > ray->t0 && t_out < ray->t)
189  {
190  const vec3f p2 = ray->org + t_out * ray->dir;
191  // consider only the parts within the extents of the truncated cone
192  if (dot(p2 - v1, v) > 0.f && dot(p2 - v0, v) < 0.f)
193  {
194  ray->primID = primID;
195  ray->geomID = self->super.geomID;
196  ray->instID = args->context->instID[0];
197  ray->t = t_out;
198  const vec3f surfaceVec = normalize(p2 - V);
199  ray->Ng = cross(cross(v, surfaceVec), surfaceVec);
200  }
201  }
202  return;
203 }

205 static void Cones_postIntersect(uniform Geometry* uniform geometry, uniform Model* uniform model,
206  varying DifferentialGeometry& dg, const varying Ray& ray, uniform int64 flags)
207 {
208  uniform Cones* uniform this = (uniform Cones * uniform) geometry;
209  dg.geometry = geometry;
210  vec3f Ng = ray.Ng;
211  vec3f Ns = Ng;
212 
213  if (flags & DG_NORMALIZE)
214  {
215  Ng = normalize(Ng);
216  Ns = normalize(Ns);
217  }
218  if (flags & DG_FACEFORWARD)
219  {
220  if (dot(ray.dir, Ng) >= 0.f)
221  Ng = neg(Ng);
222  if (dot(ray.dir, Ns) >= 0.f)
223  Ns = neg(Ns);
224  }
225  dg.Ng = Ng;
226  dg.Ns = Ns;
227 }

236 export void ConesGeometry_set(void* uniform _self, void* uniform _model, void* uniform data, int uniform numPrimitives)
237 {
238  uniform Cones* uniform self = (uniform Cones * uniform) _self;
239  uniform Model* uniform model = (uniform Model * uniform) _model;
240 
241  RTCGeometry geom = rtcNewGeometry(ispc_embreeDevice(), RTC_GEOMETRY_TYPE_USER);
242  uniform uint32 geomID = rtcAttachGeometry(model->embreeSceneHandle, geom);
243 
244  self->super.model = model;
245  self->super.geomID = geomID;
246  self->super.numPrimitives = numPrimitives;
247  self->data = (uniform Cone * uniform) data;
248  self->useSafeIncrement = needsSafeIncrement(self->data, numPrimitives);
249 
250  rtcSetGeometryUserData(geom, self);
251  rtcSetGeometryUserPrimitiveCount(geom, numPrimitives);
252  rtcSetGeometryBoundsFunction(geom, (uniform RTCBoundsFunction)&Cones_bounds, self);
253  rtcSetGeometryIntersectFunction(geom, (uniform RTCIntersectFunctionN)&Cones_intersect);
254  rtcSetGeometryOccludedFunction(geom, (uniform RTCOccludedFunctionN)&Cones_intersect);
255  rtcCommitGeometry(geom);
256  rtcReleaseGeometry(geom);
257 }
