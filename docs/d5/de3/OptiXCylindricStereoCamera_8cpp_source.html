55 void OptiXCylindricStereoCamera::commit(const OptiXCamera& camera, ::optix::Context context)
56 {
57  const auto pos = camera.getPosition();
58 
59  const Vector3d u = normalize(glm::rotate(camera.getOrientation(), Vector3d(1, 0, 0)));
60  const Vector3d v = normalize(glm::rotate(camera.getOrientation(), Vector3d(0, 1, 0)));
61  const Vector3d w = normalize(glm::rotate(camera.getOrientation(), Vector3d(0, 0, 1)));
62 
63  context[CONTEXT_CAMERA_EYE]->setFloat(pos.x, pos.y, pos.z);
64  context[CONTEXT_CAMERA_U]->setFloat(u.x, u.y, u.z);
65  context[CONTEXT_CAMERA_V]->setFloat(v.x, v.y, v.z);
66  context[CONTEXT_CAMERA_W]->setFloat(w.x, w.y, w.z);
67 
68  const auto headPos = camera.getPropertyOrValue<std::array<double, 3>>(CONTEXT_CAMERA_HEAD_POSITION, {{0., 0., 0.}});
69  const auto headRotation =
70  camera.getPropertyOrValue<std::array<double, 4>>(CONTEXT_CAMERA_HEAD_ROTATION, {{0., 0., 0., 1.}});
71  const auto headUVec = glm::rotate(Quaterniond(headRotation[3], headRotation[0], headRotation[1], headRotation[2]),
72  Vector3d(1., 0., 0.));
73 
74  context[CONTEXT_CAMERA_SEGMENT_ID]->setUint(camera.getPropertyOrValue<int>(CONTEXT_CAMERA_SEGMENT_ID, 0));
75  context[CONTEXT_CAMERA_HALF_IPD]->setFloat(camera.getPropertyOrValue<double>(CONTEXT_CAMERA_IPD, 0.065) / 2.0);
76  context[CONTEXT_CAMERA_HEAD_POSITION]->setFloat(headPos[0], headPos[1], headPos[2]);
77  context[CONTEXT_CAMERA_HEAD_UVEC]->setFloat(headUVec.x, headUVec.y, headUVec.z);
78  context[CONTEXT_CAMERA_APERTURE_RADIUS]->setFloat(
79  camera.getPropertyOrValue<double>(CONTEXT_CAMERA_APERTURE_RADIUS, 0.0));
80  context[CONTEXT_CAMERA_FOCAL_DISTANCE]->setFloat(
81  camera.getPropertyOrValue<double>(CONTEXT_CAMERA_FOCUS_DISTANCE, 1.0));
82 }
