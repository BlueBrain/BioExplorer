54 template <class T>
55 typename std::enable_if<!std::numeric_limits<T>::is_integer, bool>::type almost_equal(T x, T y, int ulp = 2)
56 {
57  // the machine epsilon has to be scaled to the magnitude of the values used
58  // and multiplied by the desired precision in ULPs (units in the last place)
59  return std::abs(x - y) <= std::numeric_limits<T>::epsilon() * std::abs(x + y) * ulp
60  // unless the result is subnormal
61  || std::abs(x - y) < std::numeric_limits<T>::min();
62 }

191 void MorphologyLoader::_connectSDFBifurcations(SDFMorphologyData& sdfMorphologyData,
192  const MorphologyTreeStructure& mts) const
193 {
194  const size_t nbSections = mts.sectionChildren.size();
195 
196  for (uint section = 0; section < nbSections; ++section)
197  {
198  // Find the bifurction geometry id for this section
199  size_t bifurcationId = 0;
200  bool bifurcationIdFound = false;
201  for (size_t bifId : sdfMorphologyData.bifurcationIndices)
202  {
203  const int bifSection = sdfMorphologyData.geometrySection.at(bifId);
204 
205  if (bifSection == static_cast<int>(section))
206  {
207  bifurcationId = bifId;
208  bifurcationIdFound = true;
209  break;
210  }
211  }
212 
213  if (!bifurcationIdFound)
214  continue;
215 
216  // Function for connecting overlapping geometries with current
217  // bifurcation
218  const auto connectGeometriesToBifurcation = [&](const std::vector<size_t>& geometries)
219  {
220  const auto& bifGeom = sdfMorphologyData.geometries[bifurcationId];
221 
222  for (size_t geomIdx : geometries)
223  {
224  // Do not blend yourself
225  if (geomIdx == bifurcationId)
226  continue;
227 
228  const auto& geom = sdfMorphologyData.geometries[geomIdx];
229  const float dist0 = glm::distance2(geom.p0, bifGeom.p0);
230  const float dist1 = glm::distance2(geom.p1, bifGeom.p0);
231  const float radiusSum = geom.r0 + bifGeom.r0;
232  const float radiusSumSq = radiusSum * radiusSum;
233 
234  if (dist0 < radiusSumSq || dist1 < radiusSumSq)
235  {
236  sdfMorphologyData.neighbours[bifurcationId].insert(geomIdx);
237  sdfMorphologyData.neighbours[geomIdx].insert(bifurcationId);
238  }
239  }
240  };
241 
242  // Connect all child sections
243  for (const size_t sectionChild : mts.sectionChildren[section])
244  {
245  if (sdfMorphologyData.sectionGeometries.find(sectionChild) == sdfMorphologyData.sectionGeometries.end())
246  PLUGIN_THROW("Invalid SDF connected children section");
247  connectGeometriesToBifurcation(sdfMorphologyData.sectionGeometries.at(sectionChild));
248  }
249 
250  // Connect with own section
251  if (sdfMorphologyData.sectionGeometries.find(section) == sdfMorphologyData.sectionGeometries.end())
252  PLUGIN_THROW("Invalid SDF own section");
253  connectGeometriesToBifurcation(sdfMorphologyData.sectionGeometries.at(section));
254  }
255 }

257 void MorphologyLoader::_finalizeSDFGeometries(ParallelModelContainer& modelContainer,
258  SDFMorphologyData& sdfMorphologyData) const
259 {
260  const size_t numGeoms = sdfMorphologyData.geometries.size();
261  sdfMorphologyData.localToGlobalIdx.resize(numGeoms, 0);
262 
263  // Extend neighbours to make sure smoothing is applied on all
264  // closely connected geometries
265  for (size_t rep = 0; rep < 4; rep++)
266  {
267  const size_t numNeighs = sdfMorphologyData.neighbours.size();
268  auto neighsCopy = sdfMorphologyData.neighbours;
269  for (size_t i = 0; i < numNeighs; i++)
270  {
271  for (size_t j : sdfMorphologyData.neighbours[i])
272  {
273  for (size_t newNei : sdfMorphologyData.neighbours[j])
274  {
275  neighsCopy[i].insert(newNei);
276  neighsCopy[newNei].insert(i);
277  }
278  }
279  }
280  sdfMorphologyData.neighbours = neighsCopy;
281  }
282 
283  for (size_t i = 0; i < numGeoms; i++)
284  {
285  // Convert neighbours from set to vector and erase itself from its
286  // neighbours
287  std::vector<size_t> neighbours;
288  const auto& neighSet = sdfMorphologyData.neighbours[i];
289  std::copy(neighSet.begin(), neighSet.end(), std::back_inserter(neighbours));
290  neighbours.erase(std::remove_if(neighbours.begin(), neighbours.end(), [i](size_t elem) { return elem == i; }),
291  neighbours.end());
292 
293  modelContainer.addSDFGeometry(sdfMorphologyData.materials[i], sdfMorphologyData.geometries[i], neighbours);
294  }
295 }

297 MorphologyTreeStructure MorphologyLoader::_calculateMorphologyTreeStructure(
298  const PropertyMap& properties, const brain::neuron::Sections& sections) const
299 {
300  const size_t nbSections = sections.size();
301 
302  const auto dampenBranchThicknessChangerate =
303  properties.getProperty<bool>(PROP_DAMPEN_BRANCH_THICKNESS_CHANGERATE.name);
304  if (!dampenBranchThicknessChangerate)
305  {
306  MorphologyTreeStructure mts;
307  mts.sectionTraverseOrder.resize(nbSections);
308  mts.sectionParent.resize(nbSections, -1);
309  std::iota(mts.sectionTraverseOrder.begin(), mts.sectionTraverseOrder.end(), 0);
310  return mts;
311  }
312 
313  std::vector<std::pair<float, Vector3f>> bifurcationPosition(nbSections,
314  std::make_pair<float, Vector3f>(0.0f, {0.f, 0.f, 0.f}));
315 
316  std::vector<std::pair<float, Vector3f>> sectionEndPosition(nbSections,
317  std::make_pair<float, Vector3f>(0.0f, {0.f, 0.f, 0.f}));
318 
319  std::vector<std::vector<size_t>> sectionChildren(nbSections, std::vector<size_t>());
320 
321  std::vector<int> sectionParent(nbSections, -1);
322  std::vector<bool> skipSection(nbSections, true);
323  std::vector<bool> addedSection(nbSections, false);
324 
325  // Find section bifurcations and end positions
326  for (size_t sectionI = 0; sectionI < nbSections; sectionI++)
327  {
328  const auto& section = sections[sectionI];
329 
330  if (section.getType() == brain::neuron::SectionType::soma)
331  continue;
332 
333  const auto& samples = section.getSamples();
334  if (samples.empty())
335  continue;
336 
337  skipSection[sectionI] = false;
338 
339  { // Branch beginning
340  const auto& sample = samples[0];
341 
342  const auto radius = _getCorrectedRadius(properties, sample.w);
343 
344  const Vector3f position(sample.x, sample.y, sample.z);
345 
346  bifurcationPosition[sectionI].first = radius;
347  bifurcationPosition[sectionI].second = position;
348  }
349 
350  { // Branch end
351  const auto& sample = samples.back();
352  const auto radius = _getCorrectedRadius(properties, sample.w);
353  const Vector3f position(sample.x, sample.y, sample.z);
354  sectionEndPosition[sectionI].first = radius;
355  sectionEndPosition[sectionI].second = position;
356  }
357  }
358 
359  const auto overlaps = [](const std::pair<float, Vector3f>& p0, const std::pair<float, Vector3f>& p1)
360  {
361  const float d = (p0.second - p1.second).length();
362  const float r = p0.first + p1.first;
363 
364  return (d < r);
365  };
366 
367  // Find overlapping section bifurcations and end positions
368  for (size_t sectionI = 0; sectionI < nbSections; sectionI++)
369  {
370  if (skipSection[sectionI])
371  continue;
372 
373  for (size_t sectionJ = sectionI + 1; sectionJ < nbSections; sectionJ++)
374  {
375  if (skipSection[sectionJ])
376  continue;
377 
378  if (overlaps(bifurcationPosition[sectionJ], sectionEndPosition[sectionI]))
379  {
380  if (sectionParent[sectionJ] == -1)
381  {
382  sectionChildren[sectionI].push_back(sectionJ);
383  sectionParent[sectionJ] = static_cast<size_t>(sectionI);
384  }
385  }
386  else if (overlaps(bifurcationPosition[sectionI], sectionEndPosition[sectionJ]))
387  {
388  if (sectionParent[sectionI] == -1)
389  {
390  sectionChildren[sectionJ].push_back(sectionI);
391  sectionParent[sectionI] = static_cast<size_t>(sectionJ);
392  }
393  }
394  }
395  }
396 
397  // Fill stack with root sections
398  std::vector<size_t> sectionStack;
399  for (size_t sectionI = 0; sectionI < nbSections; sectionI++)
400  {
401  if (skipSection[sectionI])
402  continue;
403  else if (sectionParent[sectionI] == -1)
404  sectionStack.push_back(sectionI);
405  }
406 
407  // Starting from the roots fill the tree traversal order
408  std::vector<size_t> sectionOrder;
409  while (!sectionStack.empty())
410  {
411  const size_t sectionI = sectionStack.back();
412  sectionStack.pop_back();
413  assert(!addedSection[sectionI]);
414  addedSection[sectionI] = true;
415 
416  sectionOrder.push_back(sectionI);
417  for (const size_t childI : sectionChildren[sectionI])
418  sectionStack.push_back(childI);
419  }
420 
421  MorphologyTreeStructure mts;
422  mts.sectionTraverseOrder = std::move(sectionOrder);
423  mts.sectionParent = std::move(sectionParent);
424  mts.sectionChildren = std::move(sectionChildren);
425  return mts;
426 }

429  const brain::neuron::Soma& soma, uint64_t offset, ParallelModelContainer& model,
430  SDFMorphologyData& sdfMorphologyData, const bool /*useSimulationModel*/,
431  const bool generateInternals, const float mitochondriaDensity,
432  uint32_t& sdfGroupId) const
433 {

435 
436  model.getMorphologyInfo().somaPosition = soma.getCentroid();
437 
438  const float somaRadius = _getCorrectedRadius(properties, soma.getMeanRadius());
439  const bool useSDFSoma = properties.getProperty<bool>(PROP_USE_SDF_SOMA.name);
440 
441  const auto& children = soma.getChildren();
442 
443  if (useSDFSoma)
444  _connectSDFSomaChildren(properties, model.getMorphologyInfo().somaPosition, somaRadius, materialId, offset,
445  children, sdfMorphologyData);
446  else
447  {
448  model.addSphere(materialId, {model.getMorphologyInfo().somaPosition, static_cast<float>(somaRadius), offset});
449  // When using a simulation model, parametric geometries
450  // must occupy as much space as possible in the mesh.
451  // This code inserts a Cone between the soma and the
452  // beginning of each branch.
453  for (const auto& child : children)
454  {
455  const auto& samples = child.getSamples();
456  const Vector3f sample{samples[0].x, samples[0].y, samples[0].z};
457 
458  model.getMorphologyInfo().bounds.merge(sample);

460 
461  model.addCone(materialId, {model.getMorphologyInfo().somaPosition, sample, static_cast<float>(somaRadius),
462  static_cast<float>(sampleRadius), offset});
463  }
464  }
465 
466  if (generateInternals && mitochondriaDensity > 0.f)
467  {
468  materialId = _getMaterialIdFromColorScheme(properties, brain::neuron::SectionType::undefined);
469  _addSomaInternals(properties, index, model, materialId, somaRadius, mitochondriaDensity, sdfMorphologyData,

500 void MorphologyLoader::_addStepConeGeometry(const bool useSDFGeometry, const Vector3f& source, const float sourceRadius,
501  const Vector3f& target, const float targetRadius, const size_t materialId,
502  const uint64_t& userDataOffset, ParallelModelContainer& model,
503  SDFMorphologyData& sdfMorphologyData, const uint32_t sdfGroupId,
504  const float displacementRatio) const
505 {
506  if (target == source || sourceRadius < 0.f || targetRadius < 0.f)
507  return;
508 
509  model.getMorphologyInfo().bounds.merge(source);
510  model.getMorphologyInfo().bounds.merge(target);
511  if (useSDFGeometry)
512  {
513  const Vector3f displacementParams = {std::min(sourceRadius, 0.05f), displacementRatio * 1.2f, 2.f};
514  const auto geom =
515  (almost_equal(sourceRadius, targetRadius))
516  ? createSDFPill(source, target, sourceRadius, userDataOffset, displacementParams)
517  : createSDFConePill(source, target, sourceRadius, targetRadius, userDataOffset, displacementParams);
518  _addSDFGeometry(sdfMorphologyData, geom, {}, materialId, sdfGroupId);
519  }
520  else
521  {
522  if (almost_equal(sourceRadius, targetRadius))
523  model.addCylinder(materialId, {source, target, sourceRadius, userDataOffset});
524  else
525  model.addCone(materialId, {source, target, sourceRadius, targetRadius, userDataOffset});

529 float MorphologyLoader::_distanceToSoma(const brain::neuron::Section& section, const size_t sampleId) const
530 {
531  float distance = 0.0;
532  if (sampleId > 0)
533  {
534  const auto& samples = section.getSamples();
535  for (size_t i = 0; i < std::min(samples.size(), sampleId) - 1; ++i)
536  {
537  const auto& a = samples[i];
538  const auto& b = samples[i + 1];
539  distance = distance + (b - a).length();
540  }
541  }
542 
543  auto s = section;
544  while (s.hasParent())
545  {
546  s = s.getParent();
547 
548  const auto& samples = s.getSamples();
549  for (size_t i = 0; i < samples.size() - 1; ++i)
550  {
551  const auto& a = samples[i];
552  const auto& b = samples[i + 1];
553  distance = distance + (b - a).length();
554  }
555  }
556 
557  return distance * 10.f; // Since user data is uint64_t, we multiply by
558  // 10 to increase the precision of the growth
559  // (first decimal value will then be considered)
560 }

589  uint64_t userDataOffset = 0;
590  if (compartmentReport)
591  userDataOffset = compartmentReport->getOffsets()[index][0];
592 
593  const brion::URI source(uri);
594  const brain::neuron::Morphology morphology(source);
595  const auto sections = morphology.getSections(sectionTypes);
596 
597  uint32_t sdfGroupId = 0;
598  if (std::find(sectionTypes.begin(), sectionTypes.end(), brain::neuron::SectionType::soma) != sectionTypes.end())
599  {
600  _addSomaGeometry(index, properties, morphology.getSoma(), userDataOffset, model, sdfMorphologyData,
601  compartmentReport != nullptr, generateInternals, mitochondriaDensity, sdfGroupId);
602  }
603 
604  // Only the first one or two axon sections are reported, so find the
605  // last one and use its offset for all the other axon sections
606  uint16_t lastAxon = 0;
607  if (compartmentReport &&
608  std::find(sectionTypes.begin(), sectionTypes.end(), brain::neuron::SectionType::axon) != sectionTypes.end())
609  {
610  const auto& counts = compartmentReport->getCompartmentCounts()[index];
611  const auto& axon = morphology.getSections(brain::neuron::SectionType::axon);
612  for (const auto& section : axon)
613  {
614  if (counts[section.getID()] > 0)
615  {
616  lastAxon = section.getID();
617  continue;
618  }
619  break;
620  }
621  }
622 
623  const auto morphologyTree = _calculateMorphologyTreeStructure(properties, sections);
624 
625  // Dendrites and axon
626  const float branchDisplacementRatio = 1.f;
627  for (const size_t sectionId : morphologyTree.sectionTraverseOrder)
628  {
629  const auto& section = sections[sectionId];
630 
631  if (section.getType() == brain::neuron::SectionType::soma)
632  continue;
633 
634  const auto materialId = _getMaterialIdFromColorScheme(properties, section.getType());
635  const auto& samples = section.getSamples();
636  if (samples.empty())
637  continue;
638 
639  const size_t nbSamples = samples.size();
640 
641  Vector3f dstPosition;
642  switch (morphologyQuality)
643  {
644  case AssetQuality::low:
645  case AssetQuality::medium:
646  dstPosition = Vector3f(samples[0]);
647  break;
648  default:
649  dstPosition = getBezierPoint(samples, 0.f);
650  break;
651  }
652  float dstDiameter = (section.hasParent() ? _getLastSampleDiameter(section.getParent()) : samples[0].w);
653 
654  float segmentStep = 0.f;
655  if (compartmentReport)
656  {
657  const auto& counts = compartmentReport->getCompartmentCounts()[index];
658  // Number of compartments usually differs from number of samples
659  segmentStep = counts[section.getID()] / static_cast<float>(nbSamples);
660  }
661 
662  float sectionVolume = 0.f;
663  float sectionLength = 0.f;
664 
665  // Axon and dendrites
666  for (uint64_t s = 0; s < nbSamples; ++s)
667  {
668  const auto distanceToSoma = _distanceToSoma(section, s);
669  if (distanceToSoma > maxDistanceToSoma)
670  continue;
671 
672  model.getMorphologyInfo().maxDistanceToSoma =
673  std::max(model.getMorphologyInfo().maxDistanceToSoma, distanceToSoma);
674 
675  switch (userDataType)
676  {
677  case UserDataType::distance_to_soma:
678  userDataOffset = distanceToSoma;
679  break;
680  case UserDataType::simulation_offset:
681  if (compartmentReport)
682  {
683  const auto& offsets = compartmentReport->getOffsets()[index];
684  const auto& counts = compartmentReport->getCompartmentCounts()[index];
685 
686  // Update the offset if we have enough compartments aka
687  // a full compartment report. Otherwise we keep the soma
688  // offset which happens for soma reports and use this
689  // for all the sections
690  if (section.getID() < counts.size())
691  {
692  if (counts[section.getID()] > 0)
693  userDataOffset = offsets[section.getID()] + static_cast<float>(s - 1) * segmentStep;
694  else
695  {
696  if (section.getType() == brain::neuron::SectionType::axon)
697  userDataOffset = offsets[lastAxon];
698  else
699  // This should never happen, but just in
700  // case use an invalid value to show an
701  // error color
702  userDataOffset = std::numeric_limits<uint64_t>::max();
703  }
704  }
705  }
706  break;
707  case UserDataType::undefined:
708  userDataOffset = 0;
709  break;
710  }
711 
712  Vector3f srcPosition;
713  switch (morphologyQuality)
714  {
715  case AssetQuality::low:
716  case AssetQuality::medium:
717  srcPosition = samples[s];
718  break;
719  default:
720  srcPosition = getBezierPoint(samples, float(s) / float(nbSamples - 1));
721  break;
722  }
723  model.getMorphologyInfo().bounds.merge(srcPosition);
724 
725  const float srcDiameter = samples[s].w;
726 
727  const float sampleLength = length(dstPosition - srcPosition);
728  sectionLength += sampleLength;
729 
730  float srcRadius = _getCorrectedRadius(properties, srcDiameter);
731  float dstRadius = _getCorrectedRadius(properties, dstDiameter);
732 
733 #if 0
734  const float maxRadiusChange = 0.1f;
735  if (sampleLength > 0.0001f && s != nbSamples - 1 &&
736  dampenBranchThicknessChangerate)
737  {
738  const float radiusChange =
739  std::min(std::abs(dstRadius - srcRadius),
740  sampleLength * maxRadiusChange);
741  if (srcRadius < dstRadius)
742  srcRadius = dstRadius - radiusChange;
743  else
744  srcRadius = dstRadius + radiusChange;
745  }
746 #endif
747 
748  // Add Geometry
749  if (s > 0)
750  {
751  _addStepSphereGeometry(useSdfBranches, (s == nbSamples - 1), dstPosition, dstRadius, materialId,
752  userDataOffset, model, sdfMorphologyData, sectionId + sdfGroupId,
753  branchDisplacementRatio);
754 
755  if (srcPosition != dstPosition && dstRadius > 0.f)
756  _addStepConeGeometry(useSdfBranches, srcPosition, srcRadius, dstPosition, dstRadius, materialId,
757  userDataOffset, model, sdfMorphologyData, sectionId + sdfGroupId,
758  branchDisplacementRatio);
759  sectionVolume += coneVolume(sampleLength, srcRadius, dstRadius);
760  }
761 
762  dstPosition = srcPosition;
763  dstDiameter = srcDiameter;
764  }
765 
766  // Generate axon internals
767  if (section.getType() == brain::neuron::SectionType::axon)
768  {
769  uint32_t groupId = sectionId + sdfGroupId;
770  if (generateInternals)
771  _addAxonInternals(properties, sectionLength, sectionVolume, samples, mitochondriaDensity,
772  _baseMaterialId, sdfMorphologyData, groupId, model);
773 
774  if (generateExternals)
775  _addAxonMyelinSheath(properties, sectionLength, samples, mitochondriaDensity, _baseMaterialId,
776  sdfMorphologyData, groupId, model);
777  sdfGroupId = groupId;
778  }
779  }
780 
781  // Synapses
782  const size_t materialId = _getMaterialIdFromColorScheme(properties, brain::neuron::SectionType::undefined);
783  const auto& somaPosition = morphology.getSoma().getCentroid();
784  const auto somaRadius = _getCorrectedRadius(properties, morphology.getSoma().getMeanRadius());
785 
786  const auto inverseTransformation = inverse(transformation);
787  if (synapsesInfo.afferentSynapses)
788  for (const auto& synapse : *synapsesInfo.afferentSynapses)
789  {
790  if (synapsesInfo.prePostSynapticUsecase && gid != synapsesInfo.postGid)
791  if (synapse.getPresynapticGID() != synapsesInfo.preGid)
792  continue;
793 
794  _addSynapse(properties, synapse, SynapseType::afferent, sections, somaPosition, somaRadius,
795  inverseTransformation, _baseMaterialId, model, sdfMorphologyData, sdfGroupId);
796  }
797  if (synapsesInfo.efferentSynapses && !synapsesInfo.prePostSynapticUsecase)
798  for (const auto& synapse : *synapsesInfo.efferentSynapses)
799  _addSynapse(properties, synapse, SynapseType::efferent, sections, somaPosition, somaRadius,
800  inverseTransformation, _baseMaterialId, model, sdfMorphologyData, sdfGroupId);
801 
802  // Finalization
803  if (useSdfBranches || useSdfSoma || useSdfNucleus || useSdfMitochondria || useSdfSynapses || useSdfMyelinSteath)
804  {
805  _connectSDFBifurcations(sdfMorphologyData, morphologyTree);
806  _finalizeSDFGeometries(model, sdfMorphologyData);
807  }
808 }

810 void MorphologyLoader::_addSynapse(const PropertyMap& properties, const brain::Synapse& synapse,
811  const SynapseType synapseType, const brain::neuron::Sections& sections,
812  const Vector3f& somaPosition, const float somaRadius, const Matrix4f& transformation,
813  const size_t materialId, ParallelModelContainer& model,
814  SDFMorphologyData& sdfMorphologyData, uint32_t& sdfGroupId) const
815 {
816  uint32_t sectionId;
817  uint32_t segmentId;
818  Vector3f origin;
819  Vector3f target;
820  brion::Vector4fs segments;
821  float radius = DEFAULT_SYNAPSE_RADIUS;
822  bool processRadius = false;
823  size_t synapseMaterialId;
824 
825  switch (synapseType)
826  {
827  case SynapseType::afferent:
828  {
829  auto pos = synapse.getPostsynapticSurfacePosition();
830  origin = Vector3f(pos.x, pos.y, pos.z);
831  pos = synapse.getPostsynapticCenterPosition();
832  target = Vector3f(pos.x, pos.y, pos.z);
833  sectionId = synapse.getPostsynapticSectionID();
834  if (sectionId < sections.size())
835  {
836  segments = sections[sectionId].getSamples();
837  segmentId = synapse.getPostsynapticSegmentID();
838  synapseMaterialId = materialId + MATERIAL_OFFSET_AFFERENT_SYNPASE;
839  processRadius = true;
840  }
841  break;
842  }
843  case SynapseType::efferent:
844  {
845  auto pos = synapse.getPresynapticSurfacePosition();
846  origin = Vector3f(pos.x, pos.y, pos.z);
847  pos = synapse.getPresynapticCenterPosition();
848  target = Vector3f(pos.x, pos.y, pos.z);
849  sectionId = synapse.getPresynapticSectionID();
850  if (sectionId < sections.size())
851  {
852  segments = sections[sectionId].getSamples();
853  segmentId = synapse.getPresynapticSegmentID();
854  synapseMaterialId = materialId + MATERIAL_OFFSET_EFFERENT_SYNPASE;
855  processRadius = true;
856  }
857  break;
858  }
859  }
860 
861  // Transformation
862  target = transformVector3d(target, transformation);
863  if (length(target - somaPosition) <= somaRadius)
864  return; // Do not process synapses on the soma
865  origin = transformVector3d(origin, transformation);
866 
867  if (processRadius && segmentId < segments.size())
868  {
869  const auto& segment = segments[segmentId];
870  radius = segment.w * 0.2f;
871  }
872 
873  // Spine geometry
874  const auto useSDFSynapses = properties.getProperty<bool>(PROP_USE_SDF_SYNAPSES.name);
875 
876  const float spineRadiusRatio = 0.75f;
877  const float spineSmallRadius = radius * 0.15f;
878  const float spineBaseRadius = radius * 0.25f;
879  const Vector3f direction = origin - target;
880  const Vector3f surfaceOrigin = origin;
881 
882  const Vector3f surfaceTarget = surfaceOrigin + direction;
883  const float spineLargeRadius = radius * spineRadiusRatio;
884 
885  // Create random shape between origin and target
886  Vector3f middle = (surfaceTarget + surfaceOrigin) / 2.f;
887  const float d = length(surfaceTarget - surfaceOrigin) / 5.f;
888  middle += Vector3f(d * (rand() % 1000 / 1000.f), d * (rand() % 1000 / 1000.f), d * (rand() % 1000 / 1000.f));
889  const float spineMiddleRadius = spineSmallRadius + d * 0.1f * (rand() % 1000 / 1000.f);
890  if (useSDFSynapses)
891  {
892  const float spineDisplacementRatio = 20.f;
893  _addStepSphereGeometry(useSDFSynapses, true, surfaceTarget, spineLargeRadius, synapseMaterialId, -1, model,
894  sdfMorphologyData, sdfGroupId, spineDisplacementRatio);
895  _addStepSphereGeometry(useSDFSynapses, true, middle, spineMiddleRadius, synapseMaterialId, -1, model,
896  sdfMorphologyData, sdfGroupId, spineDisplacementRatio);
897  if (surfaceOrigin != middle)
898  _addStepConeGeometry(useSDFSynapses, surfaceOrigin, spineSmallRadius, middle, spineMiddleRadius,
899  synapseMaterialId, -1, model, sdfMorphologyData, sdfGroupId, spineDisplacementRatio);
900  if (middle != surfaceTarget)
901  _addStepConeGeometry(useSDFSynapses, middle, spineMiddleRadius, surfaceTarget, spineBaseRadius,
902  synapseMaterialId, -1, model, sdfMorphologyData, sdfGroupId, spineDisplacementRatio);
903  ++sdfGroupId;
904  }
905  else
906  {
907  model.addSphere(synapseMaterialId, {origin, radius});
908  model.addSphere(synapseMaterialId, {middle, spineMiddleRadius});
909  if (origin != middle)
910  model.addCone(synapseMaterialId, {origin, middle, spineSmallRadius, spineMiddleRadius});
911  if (middle != surfaceTarget)
912  model.addCone(synapseMaterialId, {middle, surfaceTarget, spineMiddleRadius, spineBaseRadius});
913  }
914 }

916 void MorphologyLoader::_addSomaInternals(const PropertyMap& properties, const uint64_t index,
917  ParallelModelContainer& model, const size_t materialId, const float somaRadius,
918  const float mitochondriaDensity, SDFMorphologyData& sdfMorphologyData,
919  uint32_t& sdfGroupId) const
920 {
921  const bool useSDFNucleus = properties.getProperty<bool>(PROP_USE_SDF_NUCLEUS.name);
922  const bool useSDFMitochondria = properties.getProperty<bool>(PROP_USE_SDF_MITOCHONDRIA.name);
923 
924  const float mitochondrionRadiusRatio = 0.025f;
925  const float mitochondrionDisplacementRatio = 20.f;
926  const float nucleusDisplacementRatio = 2.f;
927  const float nucleusRadius = somaRadius * 0.8f; // 80% of the volume of the soma;
928  const float mitochondrionRadius = somaRadius * mitochondrionRadiusRatio; // 5% of the volume of the soma
929 
930  const float somaInnerRadius = nucleusRadius + mitochondrionRadius;
931  const float availableVolumeForMitochondria = sphereVolume(somaRadius) * mitochondriaDensity;
932 
933  // Soma nucleus
934  const auto somaPosition = Vector3f(model.getMorphologyInfo().somaPosition);
935  const size_t nucleusMaterialId = materialId + MATERIAL_OFFSET_NUCLEUS;
936  if (useSDFNucleus)
937  {
938  _addStepSphereGeometry(useSDFNucleus, true, somaPosition, nucleusRadius, nucleusMaterialId, -1, model,
939  sdfMorphologyData, sdfGroupId, nucleusDisplacementRatio);
940  ++sdfGroupId;
941  }
942  else
943  model.addSphere(nucleusMaterialId, {somaPosition, nucleusRadius});
944 
945  // Mitochondria
946  const size_t mitochondrionMaterialId = materialId + MATERIAL_OFFSET_MITOCHONDRION;
947  float mitochondriaVolume = 0.f;
948  while (mitochondriaVolume < availableVolumeForMitochondria)
949  {
950  const size_t nbSegments = _getNbMitochondrionSegments();
951  const auto pointsInSphere = getPointsInSphere(nbSegments, somaInnerRadius / somaRadius);
952  float previousRadius = mitochondrionRadius;
953  for (size_t i = 0; i < nbSegments; ++i)
954  {
955  // Mitochondrion geometry
956  const float radius = (1.f + (rand() % 500 / 1000.f)) * mitochondrionRadius;
957  const auto p2 = somaPosition + somaRadius * pointsInSphere[i];
958  _addStepSphereGeometry(useSDFMitochondria, true, p2, radius, mitochondrionMaterialId, -1, model,
959  sdfMorphologyData, sdfGroupId, mitochondrionDisplacementRatio);
960 
961  mitochondriaVolume += sphereVolume(radius);
962 
963  if (i > 0)
964  {
965  const auto p1 = somaPosition + somaRadius * pointsInSphere[i - 1];
966  _addStepConeGeometry(useSDFMitochondria, p1, previousRadius, p2, radius, mitochondrionMaterialId, -1,
967  model, sdfMorphologyData, sdfGroupId, mitochondrionDisplacementRatio);
968 
969  mitochondriaVolume += coneVolume(length(p2 - p1), previousRadius, radius);
970  }
971  previousRadius = radius;
972  }
973  if (useSDFMitochondria)
974  ++sdfGroupId;
975  }
976 }

983 void MorphologyLoader::_addAxonInternals(const PropertyMap& properties, const float sectionLength,
984  const float sectionVolume, const brion::Vector4fs& samples,
985  const float mitochondriaDensity, const size_t materialId,
986  SDFMorphologyData& sdfMorphologyData, uint32_t& sdfGroupId,
987  ParallelModelContainer& model) const
988 {
989  const bool useSDFMitochondria = properties.getProperty<bool>(PROP_USE_SDF_MITOCHONDRIA.name);
990 
991  // Add mitochondria (density is per section, not for the full axon)
992  const float mitochondrionSegmentSize = 0.25f;
993  const float mitochondrionRadiusRatio = 0.25f;
994 
995  const size_t nbMaxMitochondrionSegments = sectionLength / mitochondrionSegmentSize;
996  const float indexRatio = float(samples.size()) / float(nbMaxMitochondrionSegments);
997 
998  float mitochondriaVolume = 0.f;
999 
1000  size_t nbSegments = _getNbMitochondrionSegments();
1001  int mitochondrionSegment = -(rand() % (1 + nbMaxMitochondrionSegments / 10));
1002  float previousRadius;
1003  Vector3f previousPosition;
1004 
1005  ++sdfGroupId;
1006  for (size_t step = 0; step < nbMaxMitochondrionSegments; ++step)
1007  {
1008  if (mitochondriaVolume < sectionVolume * mitochondriaDensity && mitochondrionSegment >= 0 &&
1009  mitochondrionSegment < nbSegments)
1010  {
1011  const size_t srcIndex = size_t(step * indexRatio);
1012  const size_t dstIndex = size_t(step * indexRatio) + 1;
1013  if (dstIndex < samples.size())
1014  {
1015  const auto& srcSample = samples[srcIndex];
1016  const auto& dstSample = samples[dstIndex];
1017  const float srcRadius = _getCorrectedRadius(properties, srcSample.w);
1018  const Vector3f srcPosition{srcSample.x + srcRadius * (rand() % 100 - 50) / 500.f,
1019  srcSample.y + srcRadius * (rand() % 100 - 50) / 500.f,
1020  srcSample.z + srcRadius * (rand() % 100 - 50) / 500.f};
1021  const float dstRadius = _getCorrectedRadius(properties, dstSample.w);
1022  const Vector3f dstPosition{dstSample.x + dstRadius * (rand() % 100 - 50) / 500.f,
1023  dstSample.y + dstRadius * (rand() % 100 - 50) / 500.f,
1024  dstSample.z + dstRadius * (rand() % 100 - 50) / 500.f};
1025 
1026  const Vector3f direction = dstPosition - srcPosition;
1027  const Vector3f position = srcPosition + direction * (step * indexRatio - srcIndex);
1028  const float mitocondrionRadius = srcRadius * mitochondrionRadiusRatio;
1029  const float radius = (1.f + ((rand() % 500) / 1000.f)) * mitocondrionRadius;
1030 
1031  const size_t mitochondrionMaterialId = materialId + MATERIAL_OFFSET_MITOCHONDRION;
1032  _addStepSphereGeometry(useSDFMitochondria, true, position, radius, mitochondrionMaterialId, -1, model,
1033  sdfMorphologyData, sdfGroupId, mitochondrionRadiusRatio);
1034  mitochondriaVolume += sphereVolume(radius);
1035 
1036  if (mitochondrionSegment > 0)
1037  {
1038  _addStepConeGeometry(useSDFMitochondria, position, radius, previousPosition, previousRadius,
1039  mitochondrionMaterialId, -1, model, sdfMorphologyData, sdfGroupId,
1040  mitochondrionRadiusRatio);
1041  mitochondriaVolume += coneVolume(length(position - previousPosition), radius, previousRadius);
1042  }
1043 
1044  previousPosition = position;
1045  previousRadius = radius;
1046  }
1047  }
1048  ++mitochondrionSegment;
1049 
1050  if (mitochondrionSegment == nbSegments)
1051  {
1052  mitochondrionSegment = -(rand() % (1 + nbMaxMitochondrionSegments / 10));
1053  nbSegments = _getNbMitochondrionSegments();
1054  ++sdfGroupId;
1055  }
1056  }
