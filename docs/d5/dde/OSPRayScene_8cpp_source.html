70 void OSPRayScene::commit()
71 {
72  Scene::commit();
73  commitLights();
74 
75  // copy the list to avoid locking the mutex
76  ModelDescriptors modelDescriptors;
77  {
78  auto lock = acquireReadAccess();
79  modelDescriptors = _modelDescriptors;
80  }
81 
82  const bool rebuildScene = isModified();
83  const bool addRemoveVolumes = _commitVolumeAndTransferFunction(modelDescriptors);
84 
85  if (!rebuildScene && !addRemoveVolumes)
86  {
87  // check for dirty models aka their geometry has been altered
88  bool doUpdate = false;
89  for (auto& modelDescriptor : modelDescriptors)
90  {
91  auto& model = modelDescriptor->getModel();
92  if (model.isDirty())
93  {
94  model.commitGeometry();
95  // need to continue re-adding the models to update the bounding
96  // box model to reflect the new model size
97  doUpdate = true;
98  }
99  }
100  if (!doUpdate)
101  return;
102  }
103 
104  _activeModels.clear();
105 
106  if (_rootModel)
107  ospRelease(_rootModel);
108  _rootModel = ospNewModel();
109 
110  for (auto modelDescriptor : modelDescriptors)
111  {
112  if (!modelDescriptor->getEnabled())
113  continue;
114 
115  // keep models from being deleted via removeModel() as long as we use
116  // them here
117  _activeModels.push_back(modelDescriptor);
118 
119  auto& impl = static_cast<OSPRayModel&>(modelDescriptor->getModel());
120  const auto& transformation = modelDescriptor->getTransformation();
121 
122  CORE_DEBUG("Committing " << modelDescriptor->getName());
123 
124  impl.commitGeometry();
125  impl.logInformation();
126 
127  // add volumes to root model, because scivis renderer does not consider
128  // volumes from instances
129  if (modelDescriptor->getVisible())
130  {
131  modelDescriptor->getModel().commitGeometry();
132  for (auto volume : modelDescriptor->getModel().getVolumes())
133  {
134  auto ospVolume = std::dynamic_pointer_cast<OSPRayVolume>(volume.second);
135  ospAddVolume(_rootModel, ospVolume->impl());
136  }
137  }
138 
139  const auto& instances = modelDescriptor->getInstances();
140  for (size_t i = 0; i < instances.size(); ++i)
141  {
142  const auto& instance = instances[i];
143 
144  // First instance uses model transformation
145  const auto& instanceTransform = (i == 0 ? transformation : instance.getTransformation());
146 
147  if (modelDescriptor->getBoundingBox() && instance.getBoundingBox())
148  {
149  // scale and move the unit-sized bounding box geometry to the
150  // model size/scale first, then apply the instance transform
151  const auto& modelBounds = modelDescriptor->getModel().getBounds();
152  Transformation modelTransform;
153  modelTransform.setTranslation(modelBounds.getCenter() - 0.5 * modelBounds.getSize());
154  modelTransform.setScale(modelBounds.getSize());
155 
156  addInstance(_rootModel, impl.getBoundingBoxModel(),
157  transformationToAffine3f(instanceTransform) * transformationToAffine3f(modelTransform));
158  }
159 
160  if (modelDescriptor->getVisible() && instance.getVisible())
161  addInstance(_rootModel, impl.getPrimaryModel(), instanceTransform);
162  }
163 
164  impl.markInstancesClean();
165  }
166  CORE_DEBUG("Committing root models");
167 
168  ospCommit(_rootModel);
169 
170  computeBounds();
171 }

173 bool OSPRayScene::commitLights()
174 {
175  if (!_lightManager.isModified())
176  return false;
177 
178  _destroyLights();
179 
180  for (const auto& kv : _lightManager.getLights())
181  {
182  auto baseLight = kv.second;
183  OSPLight ospLight{nullptr};
184 
185  switch (baseLight->_type)
186  {
187  case LightType::DIRECTIONAL:
188  {
189  ospLight = ospNewLight3(OSPRAY_LIGHT_PROPERTY_DISTANT);
190  const auto light = static_cast<DirectionalLight*>(baseLight.get());
191  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_DIRECTION, Vector3f(light->_direction));
192  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_DISTANT_ANGULAR_DIAMETER,
193  static_cast<float>(light->_angularDiameter));
194  break;
195  }
196  case LightType::SPHERE:
197  {
198  ospLight = ospNewLight3(OSPRAY_LIGHT_PROPERTY_POINT);
199  const auto light = static_cast<SphereLight*>(baseLight.get());
200  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_POSITION, Vector3f(light->_position));
201  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_RADIUS, static_cast<float>(light->_radius));
202  break;
203  }
204  case LightType::QUAD:
205  {
206  ospLight = ospNewLight3(OSPRAY_LIGHT_PROPERTY_QUAD);
207  const auto light = static_cast<QuadLight*>(baseLight.get());
208  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_POSITION, Vector3f(light->_position));
209  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_QUAD_EDGE1, Vector3f(light->_edge1));
210  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_QUAD_EDGE2, Vector3f(light->_edge2));
211  break;
212  }
213  case LightType::SPOTLIGHT:
214  {
215  ospLight = ospNewLight3(OSPRAY_LIGHT_PROPERTY_SPOT);
216  const auto light = static_cast<SpotLight*>(baseLight.get());
217  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_POSITION, Vector3f(light->_position));
218  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_DIRECTION, Vector3f(light->_direction));
219  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_SPOT_OPENING_ANGLE,
220  static_cast<float>(light->_openingAngle));
221  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_SPOT_PENUMBRA_ANGLE,
222  static_cast<float>(light->_penumbraAngle));
223  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_RADIUS, static_cast<float>(light->_radius));
224  break;
225  }
226  case LightType::AMBIENT:
227  {
228  ospLight = ospNewLight3(OSPRAY_LIGHT_PROPERTY_AMBIENT);
229  break;
230  }
231  }
232 
233  assert(ospLight);
234 
235  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_COLOR, Vector3f(baseLight->_color));
236  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_INTENSITY, static_cast<float>(baseLight->_intensity));
237  osphelper::set(ospLight, OSPRAY_LIGHT_PROPERTY_IS_VISIBLE, baseLight->_isVisible);
238 
239  _ospLights.push_back(ospLight);
240  ospCommit(ospLight);
241  }
242 
243  // NOTE: since the lights are shared between scene and renderer we let
244  // OSPRay allocate a new buffer to avoid use-after-free issues
245  const size_t memoryFlags = 0;
246  _ospLightData = ospNewData(_ospLights.size(), OSP_OBJECT, _ospLights.data(), memoryFlags);
247  ospCommit(_ospLightData);
248 
249  return true;
250 }

252 bool OSPRayScene::_commitVolumeAndTransferFunction(ModelDescriptors& modelDescriptors)
253 {
254  bool rebuildScene = false;
255  for (auto& modelDescriptor : modelDescriptors)
256  {
257  auto& model = static_cast<OSPRayModel&>(modelDescriptor->getModel());
258  const bool dirtyTransferFunction = model.commitTransferFunction();
259  const bool dirtySimulationData = model.commitSimulationData();
260 
261  if (dirtyTransferFunction || dirtySimulationData)
262  markModified(false);
263  if (model.isVolumesDirty())
264  {
265  rebuildScene = true;
266  model.resetVolumesDirty();
267  }
268  for (auto& volume : model.getVolumes())
269  {
270  if (volume.second->isModified() || rebuildScene || _volumeParameters.isModified())
271  {
272  volume.second->commit();
273  // to reset accumulation if new blocks are added
274  markModified(false);
275  }
276  }
277  }
278  return rebuildScene;
279 }
