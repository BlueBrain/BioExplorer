70 inline vec3f getHWRandomVector(const vec3f& normal)
71 {
72  initializeRandomizer();

75  const float rx = getHWRandomValue();
76  const float ry = getHWRandomValue();
77  const float w = sqrt(1.f - ry);
78  const float cx = cos((2.f * M_PI) * rx) * w;
79  const float cy = sin((2.f * M_PI) * rx) * w;
