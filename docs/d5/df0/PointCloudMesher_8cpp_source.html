49 bool PointCloudMesher::toConvexHull(Model& model, const PointCloud& pointCloud)
50 {
51 #ifdef USE_CGAL
52  bool addModel{false};
53  for (const auto& point : pointCloud)
54  {
55  auto material =
56  model.createMaterial(point.first, std::to_string(point.first));
57  brayns::PropertyMap props;
58  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
59  static_cast<int>(MaterialShadingMode::diffuse)});
60  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
61  props.setProperty({MATERIAL_PROPERTY_CHAMELEON_MODE,
62  static_cast<int>(MaterialChameleonMode::receiver)});
63  material->updateProperties(props);
64 
65  std::vector<Point_3> points;
66  for (const auto& c : point.second)
67  points.push_back({c.x, c.y, c.z});
68 
69  CGAL::Object obj;
70  // compute convex hull of non-collinear points
71  CGAL::convex_hull_3(points.begin(), points.end(), obj);
72  if (const Polyhedron_3* poly = CGAL::object_cast<Polyhedron_3>(&obj))
73  {
74  PLUGIN_INFO("The convex hull contains " << poly->size_of_vertices()
75  << " vertices");
76 
77  for (auto eit = poly->edges_begin(); eit != poly->edges_end();
78  ++eit)
79  {
80  Point_3 a = eit->vertex()->point();
81  Point_3 b = eit->opposite()->vertex()->point();
82  model.addCylinder(point.first, {{a.x(), a.y(), a.z()},
83  {b.x(), b.y(), b.z()},
84  point.second[0].w});
85  addModel = true;
86  }
87  }
88  else
89  PLUGIN_ERROR("something else");
90  }
91  return addModel;
92 #else
93  return false;
94 #endif
95 }
