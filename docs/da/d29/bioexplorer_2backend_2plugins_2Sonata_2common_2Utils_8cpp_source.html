52 core::Vector3fs getPointsInSphere(const size_t nbPoints, const float innerRadius)
53 {
54  const float radius = innerRadius + (rand() % 1000 / 1000.f) * (1.f - innerRadius);
55  float phi = M_PI * ((rand() % 2000 - 1000) / 1000.f);
56  float theta = M_PI * ((rand() % 2000 - 1000) / 1000.f);
57  core::Vector3fs points;
58  for (size_t i = 0; i < nbPoints; ++i)
59  {
60  core::Vector3f point = {radius * sin(phi) * cos(theta), radius * sin(phi) * sin(theta), radius * cos(phi)};
61  points.push_back(point);
62  phi += ((rand() % 1000) / 5000.f);
63  theta += ((rand() % 1000) / 5000.f);
64  }
65  return points;
66 }
