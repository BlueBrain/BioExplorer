30 /**
31  Returns the refracted vector according to the direction of the incident ray,
32  he normal to the surface, and localRefraction indices
33  @param direction Direction of the incident ray
34  @param normal Normal to the surface
35  @param n1 In localRefraction index
36  @param n2 Out localRefraction index
37  @return Refracted vector
38 */
39 inline vec3f refractedVector(const varying vec3f& direction,
40  const varying vec3f& normal,
41  const varying float n1, const varying float n2)
42 {
43  if (n2 == 0.f)
44  return direction;
45  const float eta = n1 / n2;
46  const float cos1 = -dot(direction, normal);
47  const float cos2 = 1.f - eta * eta * (1.f - cos1 * cos1);
48  if (cos2 > 0.f)
49  return eta * direction + (eta * cos1 - sqrt(cos2)) * normal;
50  return direction;
51 }

53 inline vec3f pathTracingContribution(
54  const uniform PathTracingRenderer* uniform self,
55  varying ScreenSample& sample, const varying vec3f& sunDirection)
56 {
57  vec3f Kd;
58  Ray localray = sample.ray;
59  vec3f accucolor = make_vec3f(0.f);
60  vec3f mask = make_vec3f(1.f);
61 
62  // path tracing loop
63  for (int bounces = 0; bounces < 3; bounces++)
64  {
65  traceRay(self->super.super.model, localray);
66 
67  // if ray misses scene (no hit occurs), return background colour
68  if (localray.geomID < 0)
69  {
70  localray.t = inf;
71  const vec3f bgcol =
72  make_vec3f(skyboxMapping((Renderer*)self, localray,
73  self->super.bgMaterial)) *
74  skypower;
75  accucolor = accucolor + mask * bgcol;
76  break;
77  }
78  DifferentialGeometry dg;
79  postIntersect(self->super.super.model, dg, localray,
80  DG_NG | DG_NS | DG_NORMALIZE | DG_FACEFORWARD |
81  DG_MATERIALID | DG_COLOR | DG_TEXCOORD);
82 
83  uniform Material* material = dg.material;
84  uniform AdvancedMaterial* objMaterial =
85  (uniform AdvancedMaterial*)material;
86 
87  if (!objMaterial)
88  Kd = make_vec3f(dg.color);
89  else
90  foreach_unique(mat in objMaterial) Kd =
91  mat->Kd * make_vec3f(dg.color);
92 
93  vec4f simulationColor = make_vec4f(0.f);
94  if (self->super.simulationDataSize > 0)
95  simulationColor =
96  getSimulationValue(&self->super, &dg, localray.primID);
97  Kd = make_vec3f(simulationColor) * simulationColor.w +
98  Kd * (1.f - simulationColor.w);
99 
100  // compute local orthornormal base at hitpoint
101  const vec3f N = dg.Ns;
102 
103  // compute ray direction of uniformly random diffuse ray
104  const vec3f raydir =
105  getRandomVector(self->super.useHardwareRandomizer,
106  self->super.super.fb->size.x, sample, N,
107  self->super.randomNumber);
108 
109  // origin of new ray in path is hitpoint of previous ray in path
110  const vec3f hitpoint = dg.P + dg.epsilon * N;
111  setRay(localray, hitpoint, raydir);
112  localray.t0 = dg.epsilon;
113  localray.t = self->aoDistance;
114 
115  //***************************************
116  // direct lighting
117  //***************************************
118 
119  // cosine weighted contribution
120  const float sunLight = dot(dg.Ns, sunDirection);
121 
122  if (sunLight > 0.f) // if surface is facing light
123  {
124  // create and trace a shadowray towards the sun
125  Ray sunRay;
126  setRay(sunRay, hitpoint, sunDirection);
127  }
128 
129  // take surface color at this bounce's hitpoint into account
130  mask = mask * Kd;
131 
132  } // end bounces
133  return accucolor;
134 }

136 /**
137  Renderer a pixel color according to a given location in the screen space.
138  @param self Pointer to current renderer
139  @param sample Screen sample containing information about the ray, and the
140  location in the screen space.
141 */
142 inline vec3f PathTracingRenderer_shadeRay(
143  const uniform PathTracingRenderer* uniform self,
144  varying ScreenSample& sample)
145 {
146  Ray ray = sample.ray;
147  vec3f color = make_vec3f(0.f);
148 
149  const uniform Light* uniform light = self->lights[0];
150  const vec2f s = make_vec2f(0.5f);
151  DifferentialGeometry dg;
152  const varying Light_SampleRes lightSample = light->sample(light, dg, s);
153  const vec3f sunDirection = lightSample.dir;
154  const vec3f radiance = lightSample.weight;
155 
156  float pathOpacity = 1.f;
157  float oldlocalRefraction = 1.f;
158  int depth = 0;
159 
160  sample.z = inf;
161  sample.alpha = 0.f;
162 
163  bool moreRebounds = true;
164  while (moreRebounds && depth < NB_MAX_REBOUNDS && pathOpacity > 0.f)
165  {
166  float opacity = 1.f;
167  float reflection = 0.f;
168  float refraction = 1.f;
169 
170  traceRay(self->super.super.model, ray);
171 
172  if (ray.geomID < 0)
173  {
174  // No intersection. Return skybox color
175  ray.t = inf;
176  color = self->super.showBackground
177  ? make_vec3f(skyboxMapping((Renderer*)self, ray,
178  self->super.bgMaterial)) *
179  skypower_zerobounce
180  : make_vec3f(0.f);
181 
182  // No Geometry intersection. No need to iterate more
183  moreRebounds = false;
184  }
185  else
186  {
187  // else, we've got a hit!
188  DifferentialGeometry dg;
189  postIntersect(self->super.super.model, dg, ray,
190  DG_NS | DG_NG | DG_NORMALIZE | DG_FACEFORWARD |
191  DG_MATERIALID | DG_COLOR | DG_TEXCOORD);
192  const vec3f intersection = dg.P;
193 
194  uniform Material* material = dg.material;
195  uniform AdvancedMaterial* objMaterial =
196  (uniform AdvancedMaterial*)material;
197 
198  vec3f normal = dg.Ns;
199  vec3f Kd = make_vec3f(0.f);
200  vec3f Ks = make_vec3f(0.f);
201  float Ns = 0.f;
202 
203  if (!objMaterial)
204  {
205  Kd = make_vec3f(dg.color);
206  opacity = dg.color.w;
207  }
208  else
209  foreach_unique(mat in objMaterial)
210  {
211  opacity = mat->d;
212 
213  // Diffuse
214  Kd = mat->Kd;
215  if (valid(mat->map_Kd))
216  {
217  const vec4f value = get4f(mat->map_Kd, dg);
218  Kd = make_vec3f(value);
219  opacity *= value.w;
220  }
221 
222  // Specular
223  Ks = mat->Ks;
224  if (valid(mat->map_Ks))
225  Ks = get3f(mat->map_Ks, dg);
226  Ns = mat->Ns;
227  if (valid(mat->map_Ns))
228  Ns = get1f(mat->map_Ns, dg);
229 
230  // Refraction
231  refraction = mat->refraction;
232 
233  // Reflection
234  reflection = mat->reflection;
235  if (valid(mat->map_reflection))
236  {
237  const vec4f value = get4f(mat->map_reflection, dg);
238  Ns = value.x * 100.f;
239  reflection = value.w;
240  }
241 
242  // Normal mapping
243  if (valid(mat->map_Bump))
244  normal = normalize(normal * get3f(mat->map_Bump, dg));
245 
246  // Emissive mapping
247  if (valid(mat->map_a))
248  Kd = Kd + get3f(mat->map_a, dg);
249  }
250 
251  // Simulation
252  vec4f simulationColor = make_vec4f(0.f);
253  if (self->super.simulationDataSize > 0)
254  simulationColor =
255  getSimulationValue(&self->super, &dg, ray.primID);
256  Kd = make_vec3f(simulationColor) * simulationColor.w +
257  Kd * (1.f - simulationColor.w);
258 
259  const vec3f reflectedNormal =
260  ray.dir - 2.f * dot(ray.dir, normal) * normal;
261 
262  // Phong + Blinn shading
263  float cosNL = max(0.f, dot(sunDirection, normal));
264  const float cosLR = max(0.f, dot(sunDirection, reflectedNormal));
265  const vec3f specularColor = Ks * powf(cosLR, Ns);
266 
267  // Path tracing contribution
268  const float ao = self->aoStrength;
269  const vec3f pathTracingColor =
270  ao > 0.f
271  ? Kd * ao *
272  pathTracingContribution(self, sample, sunDirection)
273  : Kd * radiance * cosNL + specularColor;
274 
275  // Alpha and Z-Depth
276  if (depth == 0)
277  {
278  sample.z = ray.t;
279  sample.alpha = opacity;
280  }
281 
282  color =
283  pathTracingColor * pathOpacity + color * (1.f - pathOpacity);
284 
285  // Prepare ray for next iteration
286  bool doRefraction = (opacity < 1.f);
287  bool doReflection = (reflection > 0.f);
288  if (doRefraction && doReflection)
289  {
290  // Switch between refraction and reflection
291  const bool refract = sample.sampleID.z % 2 == 0;
292  doRefraction = refract;
293  doReflection = !refract;
294  }
295 
296  if (doRefraction)
297  {
298  // Refraction
299  vec3f correctedNormal = normal;
300  const float angle = dot(ray.dir, correctedNormal);
301  if (abs(refraction - oldlocalRefraction) < 0.001f &&
302  angle > 0.f)
303  {
304  // Exiting object, back to air (refractionindex = 1)
305  refraction = 1.f;
306  correctedNormal = correctedNormal * -1.f;
307  }
308 
309  ray.t0 = 0.f;
310  ray.dir = refractedVector(ray.dir, correctedNormal,
311  oldlocalRefraction, refraction);
312  ray.org = intersection + dg.epsilon * ray.dir;
313 
314  pathOpacity -= (1.f - opacity);
315  oldlocalRefraction = refraction;
316  ++depth;
317  }
318  else if (doReflection)
319  {
320  // Reflection
321  ray.t0 = 0.f;
322  ray.dir = reflectedNormal;
323  ray.org = intersection + dg.epsilon * reflectedNormal;
324  pathOpacity -= reflection;
325  ++depth;
326  }
327  else
328  moreRebounds = false;
329 
330  ray.t = inf;
331  ray.primID = -1;
332  ray.geomID = -1;
333  ray.instID = -1;
334  }
335  }
336 
337  return clamp(color * self->super.exposure, make_vec3f(0.f),
338  make_vec3f(1.f));
339 }

341 void PathTracingRenderer_renderSample(uniform Renderer* uniform _self,
342  void* uniform perFrameData,
343  varying ScreenSample& sample)
344 {
345  uniform PathTracingRenderer* uniform self =
346  (uniform PathTracingRenderer * uniform) _self;
347  sample.ray.time = self->super.timestamp;
348  sample.rgb = PathTracingRenderer_shadeRay(self, sample);
349 }
