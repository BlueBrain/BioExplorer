28 inline vec3f AmbientOcclusionRenderer_shadeRay(
29  const uniform AmbientOcclusionRenderer* uniform self,
30  varying ScreenSample& sample)
31 {
32  Ray ray = sample.ray;
33  sample.z = inf;
34  sample.alpha = 0.f;
35 
36  const uniform int sampleCnt = self->samplesPerFrame;
37  const uniform int accumID =
38  reduce_max(sample.sampleID.z) * self->samplesPerFrame;
39 
40  traceRay(self->super.model, ray);
41 
42  if (ray.geomID < 0)
43  // No intersection, use environment map or background color
44  return make_vec3f(0.f);
45 
46  sample.z = ray.t;
47  sample.alpha = 1.f;
48 
49  DifferentialGeometry dg;
50  postIntersect(self->super.model, dg, ray,
51  DG_NS | DG_NG | DG_NORMALIZE | DG_FACEFORWARD);
52 
53  // init TEA RNG //
54  uniform FrameBuffer* uniform fb = self->super.fb;
55  RandomTEA rng_state;
56  varying RandomTEA* const uniform rng = &rng_state;
57  RandomTEA__Constructor(rng, 0x290374,
58  (fb->size.x * sample.sampleID.y) +
59  sample.sampleID.x);
60  const vec2f rot = RandomTEA__getFloats(rng);
61 
62  int hits = 0;
63  const vec3f N = dg.Ns;
64  const linear3f localToWorld = frame(N);
65 
66  for (uniform int i = 0; i < sampleCnt; i++)
67  {
68  const vec2f halton = HaltonSequence_get2D(sampleCnt * accumID + i);
69  const vec2f r = CranleyPattersonRotation(halton, rot);
70  const vec3f local_ao_dir = cosineSampleHemisphere(r);
71  vec3f ao_dir = normalize(localToWorld * local_ao_dir);
72  if (dot(ao_dir, N) < 0.f)
73  ao_dir = neg(ao_dir);
74 
75  Ray ao_ray;
76  setRay(ao_ray, dg.P, ao_dir, dg.epsilon, self->aoRayLength);
77  traceRay(self->super.model, ao_ray);
78  if (ao_ray.geomID >= 0)
79  hits++;
80  }
81 
82  return make_vec3f(1.0f - hits / (float)sampleCnt);
83 }

85 void AmbientOcclusionRenderer_renderSample(uniform Renderer* uniform _self,
86  void* uniform perFrameData,
87  varying ScreenSample& sample)
88 {
89  uniform AmbientOcclusionRenderer* uniform self =
90  (uniform AmbientOcclusionRenderer * uniform) _self;
91  sample.ray.time = inf;
92  sample.rgb = AmbientOcclusionRenderer_shadeRay(self, sample);
93 }
