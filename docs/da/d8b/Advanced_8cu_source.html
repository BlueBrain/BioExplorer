36 static __device__ void phongShadowed(float3 p_Ko)
37 {

39  prd_shadow.attenuation = 1.f - p_Ko;
40  rtTerminateRay();
41 }

43 static __device__ void phongShade(float3 p_Kd, float3 p_Ka, float3 p_Ks, float3 p_Kr, float3 p_Ko,
44  float p_refractionIndex, float p_phong_exp, float p_glossiness, uint p_shadingMode,
45  float p_user_parameter, float3 p_normal)
46 {
47  const float3 hit_point = ray.origin + t_hit * ray.direction;
48  float3 color = make_float3(0.f);
49  float3 opacity = p_Ko;
50  float3 l_Kd = p_Kd;
51 
52  float3 normal = ::optix::normalize(p_normal);
53  if (fmaxf(opacity) > 0.f && prd.depth < maxRayDepth)
54  {
55  // User data
56  const float4 userDataColor = getUserData();
57  l_Kd = l_Kd * (1.f - userDataColor.w) + make_float3(userDataColor) * userDataColor.w;
58 
59  const float userParameter = p_user_parameter;
60 
61  // Randomness
62  optix::size_t2 screen = output_buffer.size();
63  uint seed = tea<16>(screen.x * launch_index.y + launch_index.x, frame);
64 
65  // Glossiness
66  if (p_glossiness < 1.f)
67  normal = optix::normalize(normal + (1.f - p_glossiness) *
68  make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f));
69 
70  // compute direct lighting
71  float3 directLightingColor = make_float3(0.f, 0.f, 0.f);
72  uint num_lights = lights.size();
73  for (int i = 0; i < num_lights; ++i)
74  {
75  // Surface
76  float light_attenuation = 1.f;
77 
78  BasicLight light = lights[i];
79  float3 lightDirection;
80 
81  if (light.type == BASIC_LIGHT_TYPE_POINT)
82  {
83  // Point light
84  float3 pos = light.pos;
85  if (shadowIntensity > 0.f && softShadowStrength > 0.f)
86  // Soft shadowIntensity
87  pos += softShadowStrength * make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f);
88  lightDirection = optix::normalize(pos - hit_point);
89  }
90  else
91  {
92  // Directional light
93  lightDirection = -light.pos;
94  if (shadowIntensity > 0.f && softShadowStrength > 0.f)
95  // Soft shadowIntensity
96  lightDirection +=
97  softShadowStrength * make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f);
98  lightDirection = optix::normalize(lightDirection);
99  }
100  float nDl = optix::dot(normal, lightDirection);
101 
102  // Shadows
103  if (shadowIntensity > 0.f)
104  {
105  if (nDl > 0.f && light.casts_shadow)
106  {
107  PerRayData_shadow shadow_prd;
108  shadow_prd.attenuation = make_float3(1.f);
109  optix::Ray shadow_ray(hit_point, lightDirection, shadowRayType, sceneEpsilon, giRayLength);
110  rtTrace(top_shadower, shadow_ray, shadow_prd);
111 
112  // light_attenuation is zero if completely shadowed
113  light_attenuation -= shadowIntensity * (1.f - ::optix::luminance(shadow_prd.attenuation));
114  }
115  }
116 
117  // If not completely shadowed, light the hit point
118  if (light_attenuation > 0.f)
119  {
120  const float3 Lc = light.color * light_attenuation;
121  switch (p_shadingMode)
122  {
123  case MaterialShadingMode::diffuse:
124  case MaterialShadingMode::diffuse_transparency:
125  case MaterialShadingMode::perlin:
126  {
127  float pDl = 1.f;
128  if (p_shadingMode == MaterialShadingMode::perlin)
129  {
130  const float3 point = userParameter * hit_point;
131  const float n1 = 0.25f + 0.75f * optix::clamp(worleyNoise(point, 2.f), 0.f, 1.f);
132  pDl = 1.f - n1;
133  normal.x += 0.5f * n1;
134  normal.y += 0.5f * (0.5f - n1);
135  normal.z += 0.5f * (0.25f - n1);
136  normal = optix::normalize(normal);
137  }
138 
139  // Diffuse
140  directLightingColor += light_attenuation * l_Kd * nDl * pDl * Lc;
141  const float3 H = optix::normalize(lightDirection - ray.direction);
142  const float nDh = optix::dot(normal, H);
143  if (nDh > 0.f)
144  {
145  // Specular
146  const float power = pow(nDh, p_phong_exp);
147  directLightingColor += p_Ks * power * Lc;
148  }
149  if (p_shadingMode == MaterialShadingMode::diffuse_transparency)
150  opacity *= nDh;
151  break;
152  }
153  case MaterialShadingMode::cartoon:
154  {
155  float cosNL = max(0.f, optix::dot(optix::normalize(eye - hit_point), normal));
156  const uint angleAsInt = cosNL * userParameter;
157  cosNL = (float)angleAsInt / userParameter;
158  directLightingColor += light_attenuation * l_Kd * cosNL * Lc;
159  break;
160  }
161  case MaterialShadingMode::basic:
162  {
163  const float cosNL = optix::max(0.f, optix::dot(optix::normalize(eye - hit_point), normal));
164  directLightingColor += light_attenuation * l_Kd * cosNL * Lc;
165  break;
166  }
167  case MaterialShadingMode::electron:
168  case MaterialShadingMode::electron_transparency:
169  {
170  float cosNL = max(0.f, optix::dot(optix::normalize(eye - hit_point), normal));
171  cosNL = 1.f - pow(cosNL, userParameter);
172  directLightingColor += light_attenuation * l_Kd * cosNL * Lc;
173  if (p_shadingMode == MaterialShadingMode::electron_transparency)
174  opacity *= cosNL;
175  break;
176  }
177  case MaterialShadingMode::checker:
178  {
179  const int3 point = make_int3(userParameter * (hit_point + make_float3(1e2f)));
180  const int3 p = make_int3(point.x % 2, point.y % 2, point.z % 2);
181  if ((p.x == p.y && p.z == 1) || (p.x != p.y && p.z == 0))
182  directLightingColor += light_attenuation * l_Kd;
183  else
184  directLightingColor += light_attenuation * (1.f - l_Kd);
185  break;
186  }
187  case MaterialShadingMode::goodsell:
188  {
189  const float cosNL = max(0.f, optix::dot(optix::normalize(eye - hit_point), normal));
190  directLightingColor += light_attenuation * l_Kd * (cosNL > userParameter ? 1.f : 0.5f);
191  break;
192  }
193  case MaterialShadingMode::surface_normal:
194  {
195  directLightingColor += light_attenuation * (0.5f + 0.5f * normal);
196  break;
197  }
198  default:
199  {
200  directLightingColor += light_attenuation * l_Kd;
201  break;
202  }
203  }
204  }
205  }
206  color += directLightingColor;
207 
208  // Reflection
209  const float reflection = fmaxf(p_Kr);
210  if (reflection > 0.f && prd.depth < maxRayDepth - 1)
211  {
212  PerRayData_radiance reflected_prd;
213  reflected_prd.result = make_float4(0.f);
214  reflected_prd.importance = prd.importance * fmaxf(p_Kr);
215  reflected_prd.depth = prd.depth + 1;
216 
217  const float3 R = optix::reflect(ray.direction, normal);
218  const optix::Ray reflected_ray(hit_point, R, radianceRayType, sceneEpsilon, giRayLength);
219  rtTrace(top_object, reflected_ray, reflected_prd);
220  color = color * (1.f - reflection) + Kr * make_float3(reflected_prd.result);
221  }
222 
223  // Refraction
224  if (fmaxf(opacity) < 1.f && prd.depth < maxRayDepth - 1)
225  {
226  PerRayData_radiance refracted_prd;
227  refracted_prd.result = make_float4(0.f);
228  refracted_prd.importance = prd.importance * (1.f - fmaxf(opacity));
229  refracted_prd.depth = prd.depth + 1;
230 
231  const float3 refractedNormal = refractedVector(ray.direction, normal, p_refractionIndex, 1.f);
232  const optix::Ray refracted_ray(hit_point, refractedNormal, radianceRayType, sceneEpsilon, giRayLength);
233  rtTrace(top_object, refracted_ray, refracted_prd);
234  color = color * opacity + (1.f - opacity) * make_float3(refracted_prd.result);
235  }
236 
237  // Ambient occlusion
238  if (giSamples > 0 && giWeight > 0.f && prd.depth == 0)
239  {
240  float aa_attenuation = 0.f;
241  float3 cb_color = make_float3(0.f);
242  for (uint i = 0; i < giSamples; ++i)
243  {
244  // Ambient occlusion
245  PerRayData_shadow aa_prd;
246  aa_prd.attenuation = make_float3(0.f);
247 
248  float3 aa_normal = optix::normalize(make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f));
249  if (optix::dot(aa_normal, normal) < 0.f)
250  aa_normal = -aa_normal;
251 
252  const optix::Ray aa_ray(hit_point, aa_normal, shadowRayType, sceneEpsilon, giRayLength);
253  rtTrace(top_object, aa_ray, aa_prd);
254  aa_attenuation += giWeight * ::optix::luminance(aa_prd.attenuation);
255 
256  // Color bleeding
257  PerRayData_radiance cb_prd;
258  cb_prd.result = make_float4(0.f);
259  cb_prd.importance = 0.f;
260  cb_prd.depth = prd.depth + 1;
261 
262  float3 cb_normal =
263  ::optix::normalize(make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f));
264  if (optix::dot(cb_normal, normal) < 0.f)
265  cb_normal = -cb_normal;
266 
267  const optix::Ray cb_ray =
268  optix::make_Ray(hit_point, cb_normal, radianceRayType, sceneEpsilon, ray.tmax);
269  rtTrace(top_shadower, cb_ray, cb_prd);
270  cb_color += giWeight * make_float3(cb_prd.result);
271  }
272  aa_attenuation /= (float)giSamples;
273  cb_color /= (float)giSamples;
274  color += cb_color * (1.f - aa_attenuation);
275  }
276  }
277 
278  float4 finalColor = make_float4(color, ::optix::luminance(p_Ko));
279 
280  float3 result = make_float3(finalColor);
281 
282  // Matrix filter :)
283  if (matrixFilter)
284  result = make_float3(result.x * 0.666f, result.y * 0.8f, result.z * 0.666f);
285 
286  // Fog attenuation
287  const float z = optix::length(eye - hit_point);
288  const float fogAttenuation = z > fogStart ? optix::clamp((z - fogStart) / fogThickness, 0.f, 1.f) : 0.f;
289  result = result * (1.f - fogAttenuation) + fogAttenuation * getEnvironmentColor(ray.direction);
290 
291  prd.result = make_float4(result, finalColor.w);
292 }

299 static __device__ inline void shade(bool textured)
300 {
301  const float3 world_shading_normal = normalize(rtTransformNormal(RT_OBJECT_TO_WORLD, shading_normal));
302  const float3 world_geometric_normal = normalize(rtTransformNormal(RT_OBJECT_TO_WORLD, geometric_normal));
303  const float3 normal = faceforward(world_shading_normal, -ray.direction, world_geometric_normal);
304 
305  float3 p_Kd = Kd;
306  float3 p_Ko = Ko;
307  if (textured)
308  {
309  if (volume_map)
310  {
311  const float voxelValue = optix::rtTex3D<float>(volume_map, texcoord3d.x, texcoord3d.y, texcoord3d.z);
312  const float4 voxelColor = calcTransferFunctionColor(transfer_function_map, value_range, voxelValue);
313  p_Kd = make_float3(voxelColor);
314  p_Ko = make_float3(voxelColor.w);
315  }
316  else if (octree_indices_map && octree_values_map)
317  {
318  const float4 voxelColor = calcTransferFunctionColor(transfer_function_map, value_range, texcoord.x);
319  p_Kd = make_float3(voxelColor);
320  p_Ko = make_float3(voxelColor.w);
321  }
322  else
323  p_Kd = (albedoMetallic_map ? make_float3(optix::rtTex2D<float4>(albedoMetallic_map, texcoord.x, texcoord.y))
324  : Kd);
325  }
326 
327  phongShade(p_Kd, Ka, Ks, Kr, p_Ko, refraction_index, phong_exp, glossiness, shading_mode, user_parameter, normal);
328 }
