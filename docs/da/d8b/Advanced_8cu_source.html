36 static __device__ void phongShadowed(float3 p_Ko)
37 {

39  prd_shadow.attenuation = 1.f - p_Ko;
40  rtTerminateRay();

43 static __device__ void phongShade(float3 p_Kd, float3 p_Ka, float3 p_Ks, float3 p_Kr, float3 p_Ko,
44  float p_refractionIndex, float p_phong_exp, float p_glossiness, uint p_shadingMode,
45  float p_user_parameter, float3 p_normal)
46 {
47  const float3 hit_point = ray.origin + t_hit * ray.direction;
48  float3 color = make_float3(0.f);
49  float3 opacity = p_Ko;
50  float3 l_Kd = p_Kd;
51 
52  float3 normal = ::optix::normalize(p_normal);
53  if (fmaxf(opacity) > 0.f && prd.depth < maxRayDepth)
54  {
55  // User data
56  const float4 userDataColor = getUserData();
57  l_Kd = l_Kd * (1.f - userDataColor.w) + make_float3(userDataColor) * userDataColor.w;
58 
59  const float userParameter = p_user_parameter;
60 
61  // Randomness
62  optix::size_t2 screen = output_buffer.size();
63  uint seed = tea<16>(screen.x * launch_index.y + launch_index.x, frame);
64 
65  // Glossiness
66  if (p_glossiness < 1.f)
67  normal = optix::normalize(normal + (1.f - p_glossiness) *
68  make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f));
69 
70  // compute direct lighting
71  float3 directLightingColor = make_float3(0.f, 0.f, 0.f);
72  uint num_lights = lights.size();
73  for (int i = 0; i < num_lights; ++i)
74  {
75  // Surface
76  float light_attenuation = 1.f;
77 
78  BasicLight light = lights[i];
79  float3 lightDirection;
80 
81  if (light.type == BASIC_LIGHT_TYPE_POINT)
82  {
83  // Point light
84  float3 pos = light.pos;
85  if (shadowIntensity > 0.f && softShadowStrength > 0.f)
86  // Soft shadowIntensity
87  pos += softShadowStrength * make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f);
88  lightDirection = optix::normalize(pos - hit_point);
89  }
90  else
91  {
92  // Directional light
93  lightDirection = -light.pos;
94  if (shadowIntensity > 0.f && softShadowStrength > 0.f)
95  // Soft shadowIntensity
96  lightDirection +=
97  softShadowStrength * make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f);
98  lightDirection = optix::normalize(lightDirection);
99  }
100  float nDl = optix::dot(normal, lightDirection);
101 
102  // Shadows
103  if (shadowIntensity > 0.f)
104  {
105  if (nDl > 0.f && light.casts_shadow)
106  {
107  PerRayData_shadow shadow_prd;
108  shadow_prd.attenuation = make_float3(1.f);
109  optix::Ray shadow_ray(hit_point, lightDirection, shadowRayType, sceneEpsilon, giRayLength);
110  rtTrace(top_shadower, shadow_ray, shadow_prd);
111 
112  // light_attenuation is zero if completely shadowed
113  light_attenuation -= shadowIntensity * (1.f - ::optix::luminance(shadow_prd.attenuation));
114  }
115  }
116 
117  // If not completely shadowed, light the hit point
118  if (light_attenuation > 0.f)
119  {
120  const float3 Lc = light.color * light_attenuation;
121  switch (p_shadingMode)
122  {
123  case MaterialShadingMode::diffuse:
124  case MaterialShadingMode::diffuse_transparency:
125  case MaterialShadingMode::perlin:
126  {
127  float pDl = 1.f;
128  if (p_shadingMode == MaterialShadingMode::perlin)
129  {
130  const float3 point = userParameter * hit_point;
131  const float n1 = 0.25f + 0.75f * optix::clamp(worleyNoise(point, 2.f), 0.f, 1.f);
132  pDl = 1.f - n1;
133  normal.x += 0.5f * n1;
134  normal.y += 0.5f * (0.5f - n1);
135  normal.z += 0.5f * (0.25f - n1);
136  normal = optix::normalize(normal);
137  }
138 
139  // Diffuse
140  directLightingColor += light_attenuation * l_Kd * nDl * pDl * Lc;
141  const float3 H = optix::normalize(lightDirection - ray.direction);
142  const float nDh = optix::dot(normal, H);
143  if (nDh > 0.f)
144  {
145  // Specular
146  const float power = pow(nDh, p_phong_exp);
147  directLightingColor += p_Ks * power * Lc;
148  }
149  if (p_shadingMode == MaterialShadingMode::diffuse_transparency)
150  opacity *= nDh;
151  break;
152  }
153  case MaterialShadingMode::cartoon:
154  {
155  float cosNL = max(0.f, optix::dot(optix::normalize(eye - hit_point), normal));
156  const uint angleAsInt = cosNL * userParameter;
157  cosNL = (float)angleAsInt / userParameter;
158  directLightingColor += light_attenuation * l_Kd * cosNL * Lc;
159  break;
160  }
161  case MaterialShadingMode::basic:
162  {
163  const float cosNL = optix::max(0.f, optix::dot(optix::normalize(eye - hit_point), normal));
164  directLightingColor += light_attenuation * l_Kd * cosNL * Lc;
165  break;
166  }
167  case MaterialShadingMode::electron:
168  case MaterialShadingMode::electron_transparency:
169  {
170  float cosNL = max(0.f, optix::dot(optix::normalize(eye - hit_point), normal));
171  cosNL = 1.f - pow(cosNL, userParameter);
172  directLightingColor += light_attenuation * l_Kd * cosNL * Lc;
173  if (p_shadingMode == MaterialShadingMode::electron_transparency)
174  opacity *= cosNL;
175  break;
176  }
177  case MaterialShadingMode::checker:
178  {
179  const int3 point = make_int3(userParameter * (hit_point + make_float3(1e2f)));
180  const int3 p = make_int3(point.x % 2, point.y % 2, point.z % 2);
181  if ((p.x == p.y && p.z == 1) || (p.x != p.y && p.z == 0))
182  directLightingColor += light_attenuation * l_Kd;
183  else
184  directLightingColor += light_attenuation * (1.f - l_Kd);
185  break;
186  }
187  case MaterialShadingMode::goodsell:
188  {
189  const float cosNL = max(0.f, optix::dot(optix::normalize(eye - hit_point), normal));
190  directLightingColor += light_attenuation * l_Kd * (cosNL > userParameter ? 1.f : 0.5f);
191  break;
192  }
193  default:
194  {
195  directLightingColor += light_attenuation * l_Kd;
196  break;
197  }
198  }
199  }
200  }
201  color += directLightingColor;
202 
203  // Reflection
204  const float reflection = fmaxf(p_Kr);
205  if (reflection > 0.f && prd.depth < maxRayDepth - 1)
206  {
207  PerRayData_radiance reflected_prd;
208  reflected_prd.result = make_float4(0.f);
209  reflected_prd.importance = prd.importance * fmaxf(p_Kr);
210  reflected_prd.depth = prd.depth + 1;
211 
212  const float3 R = optix::reflect(ray.direction, normal);
213  const optix::Ray reflected_ray(hit_point, R, radianceRayType, sceneEpsilon, giRayLength);
214  rtTrace(top_object, reflected_ray, reflected_prd);
215  color = color * (1.f - reflection) + Kr * make_float3(reflected_prd.result);
216  }
217 
218  // Refraction
219  if (fmaxf(opacity) < 1.f && prd.depth < maxRayDepth - 1)
220  {
221  PerRayData_radiance refracted_prd;
222  refracted_prd.result = make_float4(0.f);
223  refracted_prd.importance = prd.importance * (1.f - fmaxf(opacity));
224  refracted_prd.depth = prd.depth + 1;
225 
226  const float3 refractedNormal = refractedVector(ray.direction, normal, p_refractionIndex, 1.f);
227  const optix::Ray refracted_ray(hit_point, refractedNormal, radianceRayType, sceneEpsilon, giRayLength);
228  rtTrace(top_object, refracted_ray, refracted_prd);
229  color = color * opacity + (1.f - opacity) * make_float3(refracted_prd.result);
230  }
231 
232  // Ambient occlusion
233  if (giSamples > 0 && giWeight > 0.f && prd.depth == 0)
234  {
235  float aa_attenuation = 0.f;
236  float3 cb_color = make_float3(0.f);
237  for (uint i = 0; i < giSamples; ++i)
238  {
239  // Ambient occlusion
240  PerRayData_shadow aa_prd;
241  aa_prd.attenuation = make_float3(0.f);
242 
243  float3 aa_normal = optix::normalize(make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f));
244  if (optix::dot(aa_normal, normal) < 0.f)
245  aa_normal = -aa_normal;
246 
247  const optix::Ray aa_ray(hit_point, aa_normal, shadowRayType, sceneEpsilon, giRayLength);
248  rtTrace(top_object, aa_ray, aa_prd);
249  aa_attenuation += giWeight * ::optix::luminance(aa_prd.attenuation);
250 
251  // Color bleeding
252  PerRayData_radiance cb_prd;
253  cb_prd.result = make_float4(0.f);
254  cb_prd.importance = 0.f;
255  cb_prd.depth = prd.depth + 1;
256 
257  float3 cb_normal =
258  ::optix::normalize(make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f));
259  if (optix::dot(cb_normal, normal) < 0.f)
260  cb_normal = -cb_normal;
261 
262  const optix::Ray cb_ray =
263  optix::make_Ray(hit_point, cb_normal, radianceRayType, sceneEpsilon, ray.tmax);
264  rtTrace(top_shadower, cb_ray, cb_prd);
265  cb_color += giWeight * make_float3(cb_prd.result);
266  }
267  aa_attenuation /= (float)giSamples;
268  cb_color /= (float)giSamples;
269  color += cb_color * (1.f - aa_attenuation);
270  }
271  }
272 
273  float4 finalColor = make_float4(color, ::optix::luminance(p_Ko));
274 
275  float3 result = make_float3(finalColor);
276 
277  // Matrix filter :)
278  if (matrixFilter)
279  result = make_float3(result.x * 0.666f, result.y * 0.8f, result.z * 0.666f);
280 
281  // Fog attenuation
282  const float z = optix::length(eye - hit_point);
283  const float fogAttenuation = z > fogStart ? optix::clamp((z - fogStart) / fogThickness, 0.f, 1.f) : 0.f;
284  result = result * (1.f - fogAttenuation) + fogAttenuation * getEnvironmentColor(ray.direction);
285 
286  prd.result = make_float4(result, finalColor.w);
287 }

294 static __device__ inline void shade(bool textured)
295 {
296  const float3 world_shading_normal = normalize(rtTransformNormal(RT_OBJECT_TO_WORLD, shading_normal));
297  const float3 world_geometric_normal = normalize(rtTransformNormal(RT_OBJECT_TO_WORLD, geometric_normal));
298  const float3 normal = faceforward(world_shading_normal, -ray.direction, world_geometric_normal);
299 
300  float3 p_Kd = Kd;
301  float3 p_Ko = Ko;
302  if (textured)
303  {
304  if (volume_map != 0)
305  {
306  const float voxelValue = optix::rtTex3D<float>(volume_map, texcoord3d.x, texcoord3d.y, texcoord3d.z);
307  const float4 voxelColor = calcTransferFunctionColor(transfer_function_map, value_range, voxelValue);
308  p_Kd = make_float3(voxelColor);
309  p_Ko = make_float3(voxelColor.w);
310  }
311  else if (octree_indices_map != 0 && octree_values_map != 0)
312  {
313  const float4 voxelColor = calcTransferFunctionColor(transfer_function_map, value_range, texcoord.x);
314  p_Kd = make_float3(voxelColor);
315  p_Ko = make_float3(voxelColor.w);
316  }
317  else
318  p_Kd = make_float3(optix::rtTex2D<float4>(albedoMetallic_map, texcoord.x, texcoord.y));
319  }
320 
321  phongShade(p_Kd, Ka, Ks, Kr, p_Ko, refraction_index, phong_exp, glossiness, shading_mode, user_parameter, normal);
322 }
