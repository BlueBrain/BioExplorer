43 double SinusoidShape::_sinusoide(const double x, const double z) const
44 {
45  return 0.2f * cos(x) * sin(z) + 0.05f * cos(x * 2.3f) * sin(z * 4.6f);
46 }

48 Transformation SinusoidShape::getTransformation(
49  const uint64_t occurrence, const uint64_t nbOccurrences,
50  const AnimationDetails& animationDetails, const double offset) const
51 {
52  const double step = 0.01f;
53  const double angle = 0.01f;
54  double upOffset = 0.f;
55  if (animationDetails.positionSeed != 0)
56  upOffset = animationDetails.positionStrength *
57  rnd3((animationDetails.positionSeed + occurrence) * 10);
58 
59  const double x = rnd1() * _size.x;
60  const double z = rnd1() * _size.z;
61  const double y = upOffset + _size.y * _sinusoide(x * angle, z * angle);
62 
63  Vector3d pos = Vector3d(x, y, z);
64 
65  if (isClipped(pos, _clippingPlanes))
66  throw std::runtime_error("Instance is clipped");
67 
68  const Vector3d v1 =
69  Vector3d(x + step,
70  upOffset + _size.y * _sinusoide((x + step) * angle, z * angle),
71  z) -
72  pos;
73  const Vector3d v2 =
74  Vector3d(x,
75  upOffset + _size.y * _sinusoide(x * angle, (z + step) * angle),
76  z + step) -
77  pos;
78 
79  // Rotation
80  const Vector3d normal = normalize(cross(normalize(v1), normalize(v2)));
81  Quaterniond rot = safeQuatlookAt(normal);
82  if (animationDetails.rotationSeed != 0)
83  rot = weightedRandomRotation(rot, animationDetails.rotationSeed,
84  occurrence,
85  animationDetails.rotationStrength);
86 
87  pos += normal * offset;
88 
89  Transformation transformation;
90  transformation.setTranslation(pos);
91  transformation.setRotation(rot);
92  return transformation;
93 }
