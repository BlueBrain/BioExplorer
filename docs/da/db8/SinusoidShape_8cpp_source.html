45 double SinusoidShape::_sinusoide(const double x, const double z) const
46 {
47  return 0.2f * cos(x) * sin(z) + 0.05f * cos(x * 2.3f) * sin(z * 4.6f);
48 }

50 Transformation SinusoidShape::getTransformation(const uint64_t occurrence, const uint64_t nbOccurrences,
51  const MolecularSystemAnimationDetails& MolecularSystemAnimationDetails,
52  const double offset) const
53 {
54  const double step = 0.01f;
55  const double angle = 0.01f;
56  double upOffset = 0.f;
57  if (MolecularSystemAnimationDetails.positionSeed != 0)
58  upOffset = MolecularSystemAnimationDetails.positionStrength *
59  rnd3((MolecularSystemAnimationDetails.positionSeed + occurrence) * 10);
60 
61  const double x = rnd1() * _size.x;
62  const double z = rnd1() * _size.z;
63  const double y = upOffset + _size.y * _sinusoide(x * angle, z * angle);
64 
65  Vector3d pos = Vector3d(x, y, z);
66 
67  if (isClipped(pos, _clippingPlanes))
68  throw std::runtime_error("Instance is clipped");
69 
70  const Vector3d v1 = Vector3d(x + step, upOffset + _size.y * _sinusoide((x + step) * angle, z * angle), z) - pos;
71  const Vector3d v2 = Vector3d(x, upOffset + _size.y * _sinusoide(x * angle, (z + step) * angle), z + step) - pos;
72 
73  // Rotation
74  const Vector3d normal = normalize(cross(normalize(v1), normalize(v2)));
75  Quaterniond rot = safeQuatlookAt(normal);
76  if (MolecularSystemAnimationDetails.rotationSeed != 0)
77  rot = weightedRandomRotation(rot, MolecularSystemAnimationDetails.rotationSeed, occurrence,
78  MolecularSystemAnimationDetails.rotationStrength);
79 
80  pos += normal * offset;
81 
82  Transformation transformation;
83  transformation.setTranslation(pos);
84  transformation.setRotation(rot);
85  return transformation;
86 }
