104 void RNASequence::_buildRNAAsCurve(const Quaterniond& rotation)
105 {
106  const auto& sequence = _details.contents;
107  const auto MolecularSystemAnimationDetails = doublesToMolecularSystemAnimationDetails(_details.animationParams);
108  const auto shapeParams = doublesToVector2d(_details.shapeParams);
109  const auto radius = shapeParams.y;
110 
111  auto model = _scene.createModel();
112 
113  size_t materialId = 0;
114  for (const auto& nucleotid : nucleotidMap)
115  {
116  auto material = model->createMaterial(materialId, nucleotid.second.name);
117  material->setDiffuseColor(nucleotid.second.color);
118  ++materialId;
119  }
120  PLUGIN_INFO(3, "Created " << materialId << " materials");
121 
122  const auto occurrences = _nbElements;
123  for (uint64_t occurrence = 0; occurrence < occurrences - 1; ++occurrence)
124  {
125  const char letter = sequence[occurrence];
126  if (nucleotidMap.find(letter) != nucleotidMap.end())
127  {
128  const auto& codon = nucleotidMap[letter];
129  const auto materialId = codon.index;
130 
131  const auto src = _shape->getTransformation(occurrence, occurrences, MolecularSystemAnimationDetails, 0.f);
132  const auto dst =
133  _shape->getTransformation(occurrence + 1, occurrences, MolecularSystemAnimationDetails, 0.f);
134 
135  model->addCylinder(materialId, {src.getTranslation(), dst.getTranslation(), static_cast<float>(radius)});
136  }
137  }
138 
139  // Metadata
140  ModelMetadata metadata;
141  metadata[METADATA_ASSEMBLY] = _details.assemblyName;
142  metadata["RNA sequence"] = sequence;
143  _modelDescriptor = std::make_shared<ModelDescriptor>(std::move(model), _details.name, metadata);
144  if (_modelDescriptor && !GeneralSettings::getInstance()->getModelVisibilityOnCreation())
145  _modelDescriptor->setVisible(false);
146 }

148 void RNASequence::_buildRNAAsProteinInstances(const Quaterniond& rotation)
149 {
150  const auto& sequence = _details.contents;
151  const auto MolecularSystemAnimationDetails = doublesToMolecularSystemAnimationDetails(_details.animationParams);
152  const size_t nbElements = sequence.length();
153  Vector3d position = Vector3d(0.f);
154 
155  // Load protein
156  ModelPtr model{nullptr};
157  const std::string proteinName = _details.assemblyName + "_RNA sequence";
158  ProteinDetails pd;
159  pd.assemblyName = _details.assemblyName;
160  pd.name = proteinName;
161  pd.pdbId = _details.pdbId;
162  pd.contents = _details.proteinContents;
163  pd.recenter = true;
164  pd.atomRadiusMultiplier = _details.atomRadiusMultiplier;
165  pd.representation = _details.representation;
166 
167  _protein = ProteinPtr(new Protein(_scene, pd));
168  _modelDescriptor = _protein->getModelDescriptor();
169 
170  const auto proteinBounds = _protein->getBounds().getSize();
171  const double proteinSize = std::min(proteinBounds.x, std::min(proteinBounds.y, proteinBounds.z));
172  double proteinSpacing = 0.f;
173 
174  Vector3d previousTranslation;
175 
176  const auto occurrences = _nbElements;
177  uint64_t nbInstances = 0;
178  for (uint64_t occurrence = 0; occurrence < occurrences; ++occurrence)
179  {
180  try
181  {
182  Transformations transformations;
183 
184  Transformation assemblyTransformation;
185  assemblyTransformation.setTranslation(_assemblyPosition);
186  assemblyTransformation.setRotation(_assemblyRotation);
187  transformations.push_back(assemblyTransformation);
188 
189  const auto shapeTransformation =
190  _shape->getTransformation(occurrence, occurrences, MolecularSystemAnimationDetails, 0.f);
191  transformations.push_back(shapeTransformation);
192 
193  const Transformation finalTransformation = combineTransformations(transformations);
194 
195  const Vector3d translation = finalTransformation.getTranslation();
196 
197  if (nbInstances == 0)
198  _modelDescriptor->setTransformation(finalTransformation);
199  else
200  {
201  proteinSpacing += length(previousTranslation - translation);
202  if (proteinSpacing < proteinSize)
203  continue;
204  }
205  previousTranslation = translation;
206  ++nbInstances;
207 
208  const ModelInstance instance(true, false, finalTransformation);
209  _modelDescriptor->addInstance(instance);
210  proteinSpacing = 0.f;
211  }
212  catch (const std::runtime_error&)
213  {
214  // Instance is clipped
215  }
216  }
217 }
