64 static __device__ float3 refractedVector(const float3 direction,
65  const float3 normal, const float n1,
66  const float n2)
67 {
68  if (n2 == 0.f)
69  return direction;
70  const float eta = n1 / n2;
71  const float cos1 = -optix::dot(direction, normal);
72  const float cos2 = 1.f - eta * eta * (1.f - cos1 * cos1);
73  if (cos2 > 0.f)
74  return ::optix::normalize(eta * direction +
75  (eta * cos1 - sqrt(cos2)) * normal);
76  return direction;
77 }

86 static __device__ void phongShade(float3 p_Kd, float3 p_Ka, float3 p_Ks,
87  float3 p_Kr, float3 p_Ko,
88  float p_refractionIndex, float p_phong_exp,
89  float p_glossiness, float3 p_normal)
90 {
91  float3 color = make_float3(0.f);
92 
93  // Randomness
94  optix::size_t2 screen = output_buffer.size();
95  unsigned int seed =
96  tea<16>(screen.x * launch_index.y + launch_index.x, frame);
97 
98  // Glossiness
99  if (p_glossiness < 1.f)
100  p_normal = optix::normalize(
101  p_normal + (1.f - p_glossiness) * make_float3(rnd(seed) - 0.5f,
102  rnd(seed) - 0.5f,
103  rnd(seed) - 0.5f));
104  float light_attenuation = 1.f;
105  const float3 hit_point = ray.origin + t_hit * ray.direction;
106 
107  // Ambient occlusion
108  if (ambientOcclusionStrength > 0.f)
109  {
110  PerRayData_shadow aa_prd;
111  aa_prd.attenuation = make_float3(1.f);
112 
113  float3 aa_normal = optix::normalize(
114  make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f, rnd(seed) - 0.5f));
115  if (optix::dot(aa_normal, p_normal) < 0.f)
116  aa_normal = -aa_normal;
117 
118  const optix::Ray aa_ray(hit_point, aa_normal, shadowRayType,
119  sceneEpsilon);
120  rtTrace(top_shadower, aa_ray, aa_prd);
121  light_attenuation -=
122  ambientOcclusionStrength * ::optix::luminance(aa_prd.attenuation);
123  }
124 
125  // Surface
126  if (electronShadingEnabled == 1)
127  color = p_Kd * (1.f - abs(optix::dot(optix::normalize(hit_point - eye),
128  p_normal)));
129  else if (shadingEnabled == 0)
130  color = p_Kd;
131  else
132  {
133  // compute direct lighting
134  unsigned int num_lights = lights.size();
135  for (int i = 0; i < num_lights; ++i)
136  {
137  BasicLight light = lights[i];
138 
139  float3 lightDirection;
140  float lightDistance = ray.tmax;
141  if (light.type == BASIC_LIGHT_TYPE_POINT)
142  {
143  // Point light
144  float3 pos = light.pos;
145  if (shadows > 0.f && softShadows > 0.f)
146  // Soft shadows
147  pos += softShadows * make_float3(rnd(seed) - 0.5f,
148  rnd(seed) - 0.5f,
149  rnd(seed) - 0.5f);
150  lightDirection = optix::normalize(pos - hit_point);
151  lightDistance = optix::length(pos - hit_point);
152  }
153  else
154  {
155  // Directional light
156  lightDirection = -light.pos;
157  if (shadows > 0.f && softShadows > 0.f)
158  // Soft shadows
159  lightDirection +=
160  softShadows * make_float3(rnd(seed) - 0.5f,
161  rnd(seed) - 0.5f,
162  rnd(seed) - 0.5f);
163  lightDirection = optix::normalize(lightDirection);
164  }
165  float nDl = optix::dot(p_normal, lightDirection);
166 
167  // Shadows
168  if (shadows > 0.f)
169  {
170  if (nDl > 0.f && light.casts_shadow)
171  {
172  PerRayData_shadow shadow_prd;
173  shadow_prd.attenuation = make_float3(1.f);
174  optix::Ray shadow_ray(hit_point, lightDirection,
175  shadowRayType, sceneEpsilon,
176  lightDistance);
177  rtTrace(top_shadower, shadow_ray, shadow_prd);
178  // light_attenuation is zero if completely shadowed
179  light_attenuation -=
180  1.f - ::optix::luminance(shadow_prd.attenuation);
181  }
182  }
183 
184  // If not completely shadowed, light the hit point
185  if (light_attenuation > 0.f)
186  {
187  // Diffuse
188  const float3 Lc = light.color * light_attenuation;
189  color += light_attenuation * p_Kd * nDl * Lc;
190 
191  float3 H = optix::normalize(lightDirection - ray.direction);
192  float nDh = optix::dot(p_normal, H);
193  if (nDh > 0.f)
194  {
195  // Specular
196  float power = pow(nDh, p_phong_exp);
197  color += p_Ks * power * Lc;
198  }
199  }
200  }
201  }
202 
203  // Reflection
204  const float importance = prd_radiance.importance * optix::luminance(p_Kr);
205  if (importance > 0.f && prd_radiance.depth < maxDepth)
206  {
207  PerRayData_radiance reflected_prd;
208  reflected_prd.depth = prd_radiance.depth + 1;
209  reflected_prd.importance = importance;
210 
211  const float3 R = optix::reflect(ray.direction, p_normal);
212  const optix::Ray reflected_ray(hit_point, R, radianceRayType,
213  sceneEpsilon);
214  rtTrace(top_object, reflected_ray, reflected_prd);
215  color += p_Kr * reflected_prd.result;
216  }
217 
218  // Refraction
219  if (fmaxf(p_Ko) < 1.f && prd_radiance.depth < maxDepth)
220  {
221  PerRayData_radiance new_prd;
222  new_prd.depth = prd_radiance.depth + 1;
223 
224  const float3 R =
225  refractedVector(ray.direction, p_normal, p_refractionIndex, 1.f);
226  const optix::Ray refracted_ray(hit_point, R, radianceRayType,
227  sceneEpsilon);
228  rtTrace(top_object, refracted_ray, new_prd);
229  color += (1.f - p_Ko) * new_prd.result;
230  }
231 
232  // Only opaque surfaces are affected by Global Illumination
233  if (fmaxf(p_Ko) == 1.f && prd_radiance.depth < maxDepth)
234  {
235  // Color bleeding
236  if (ambientOcclusionStrength > 0.f && prd_radiance.depth == 0)
237  {
238  PerRayData_radiance new_prd;
239  new_prd.depth = prd_radiance.depth + 1;
240 
241  float3 ra_normal = ::optix::normalize(
242  make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f,
243  rnd(seed) - 0.5f));
244  if (optix::dot(ra_normal, p_normal) < 0.f)
245  ra_normal = -ra_normal;
246 
247  const float3 origin = hit_point + sceneEpsilon * ra_normal;
248  const optix::Ray ra_ray =
249  optix::make_Ray(origin, ra_normal, radianceRayType,
250  sceneEpsilon, ray.tmax);
251  rtTrace(top_shadower, ra_ray, new_prd);
252  color += ambientOcclusionStrength * new_prd.result;
253  }
254  }
255 
256  prd_radiance.result = color;
257 }
