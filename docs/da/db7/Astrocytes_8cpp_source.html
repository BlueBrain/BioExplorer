53 void Astrocytes::_buildModel(const doubles& radii)
54 {
55  if (_modelDescriptor)
56  _scene.removeModel(_modelDescriptor->getModelID());
57 
58  auto& connector = DBConnector::getInstance();
59 
60  auto model = _scene.createModel();
61  std::set<uint64_t> materialIds;
62  const auto useSdf = _details.useSdf;
63  const auto somas = connector.getAstrocytes(_details.sqlFilter);
64 
65  PLUGIN_INFO(1, "Building " << somas.size() << " astrocytes");
66 
67  // Astrocytes
68  size_t baseMaterialId = 0;
69  const uint64_t userData = 0;
70 
71  ThreadSafeContainers containers;
72  uint64_t index;
73 #pragma omp parallel for private(index)
74  for (index = 0; index < somas.size(); ++index)
75  {
76  if (omp_get_thread_num() == 0)
77  PLUGIN_PROGRESS("Loading astrocytes", index,
78  somas.size() / omp_get_max_threads());
79 
80  auto it = somas.begin();
81  std::advance(it, index);
82  const auto& soma = it->second;
83  const auto somaId = it->first;
84 
85  ThreadSafeContainer container(*model, useSdf, _scale);
86 
87  const auto& somaPosition = soma.center;
88 
89  // Load data from DB
90  double somaRadius = 0.0;
91  SectionMap sections;
92  if (_details.loadSomas || _details.loadDendrites)
93  sections = connector.getAstrocyteSections(somaId);
94 
95  EndFootMap endFeet;
96  if (_details.loadEndFeet)
97  endFeet = connector.getAstrocyteEndFeet(somaId);
98 
99  // Soma radius
100  uint64_t count = 1;
101  for (const auto& section : sections)
102  if (section.second.parentId == SOMA_AS_PARENT)
103  {
104  const auto& point = section.second.points[0];
105  somaRadius += 0.75 * length(Vector3d(point));
106  ++count;
107  }
108  somaRadius = _radiusMultiplier * somaRadius / count;
109 
110  // Color scheme
111  switch (_details.populationColorScheme)
112  {
113  case PopulationColorScheme::id:
114  baseMaterialId = somaId * NB_MATERIALS_PER_MORPHOLOGY;
115  break;
116  default:
117  baseMaterialId = 0;
118  }
119 #pragma omp critical
120  materialIds.insert(baseMaterialId);
121 
122  const auto somaMaterialId =
123  baseMaterialId +
124  (_details.morphologyColorScheme == MorphologyColorScheme::section
125  ? MATERIAL_OFFSET_SOMA
126  : 0);
127 #pragma omp critical
128  materialIds.insert(somaMaterialId);
129 
130  uint64_t somaGeometryIndex = 0;
131  if (_details.loadSomas)
132  {
133  somaGeometryIndex =
134  container.addSphere(somaPosition, somaRadius, somaMaterialId,
135  NO_USER_DATA, {},
136  DEFAULT_SOMA_DISPLACEMENT);
137  if (_details.generateInternals)
138  {
139  _addSomaInternals(somaId, container, baseMaterialId,
140  somaPosition, somaRadius,
141  DEFAULT_MITOCHONDRIA_DENSITY);
142  materialIds.insert(baseMaterialId + MATERIAL_OFFSET_NUCLEUS);
143  materialIds.insert(baseMaterialId +
144  MATERIAL_OFFSET_MITOCHONDRION);
145  }
146  }
147 
148  Neighbours neighbours;
149  neighbours.insert(somaGeometryIndex);
150  for (const auto& section : sections)
151  {
152  uint64_t geometryIndex = 0;
153  const auto& points = section.second.points;
154 
155  size_t sectionMaterialId = baseMaterialId;
156  const auto sectionId = section.first;
157  switch (_details.morphologyColorScheme)
158  {
159  case MorphologyColorScheme::section:
160  sectionMaterialId = baseMaterialId + section.second.type;
161  break;
162  default:
163  break;
164  }
165 #pragma omp critical
166  materialIds.insert(sectionMaterialId);
167 
168  size_t step = 1;
169  switch (_details.geometryQuality)
170  {
171  case GeometryQuality::low:
172  step = points.size() - 2;
173  break;
174  default:
175  break;
176  }
177 
178  if (_details.loadDendrites)
179  {
180  uint64_t geometryIndex = 0;
181  if (section.second.parentId == SOMA_AS_PARENT)
182  {
183  // Section connected to the soma
184  const auto& point = points[0];
185  geometryIndex =
186  container.addCone(somaPosition,
187  somaRadius * 0.75 * _radiusMultiplier,
188  somaPosition + Vector3d(point),
189  point.w * 0.5 * _radiusMultiplier,
190  somaMaterialId, userData, neighbours,
191  DEFAULT_SOMA_DISPLACEMENT);
192  neighbours.insert(geometryIndex);
193  }
194 
195  for (uint64_t i = 0; i < points.size() - 1; i += step)
196  {
197  const auto srcPoint = points[i];
198  const auto src = somaPosition + Vector3d(srcPoint);
199  const float srcRadius =
200  srcPoint.w * 0.5 * _radiusMultiplier;
201 
202  // Ignore points that are too close the previous one
203  // (according to respective radii)
204  Vector4f dstPoint;
205  float dstRadius;
206  do
207  {
208  dstPoint = points[i + step];
209  dstRadius = dstPoint.w * 0.5 * _radiusMultiplier;
210  ++i;
211  } while (length(Vector3f(dstPoint) - Vector3f(srcPoint)) <
212  (srcRadius + dstRadius) &&
213  (i + step) < points.size() - 1);
214  --i;
215 
216  const auto dst = somaPosition + Vector3d(dstPoint);
217  if (!useSdf)
218  geometryIndex = container.addSphere(dst, dstRadius,
219  sectionMaterialId,
220  NO_USER_DATA, {});
221 
222  geometryIndex =
223  container.addCone(src, srcRadius, dst, dstRadius,
224  sectionMaterialId, userData,
225  {geometryIndex},
226  DEFAULT_SECTION_DISPLACEMENT);
227 
228  _bounds.merge(srcPoint);
229  }
230  }
231  }
232 
233  if (_details.loadEndFeet && !_details.vasculaturePopulationName.empty())
234  _addEndFoot(container, endFeet, radii, somaMaterialId);
235 #pragma omp critical
236  containers.push_back(container);
237  }
238 
239  for (size_t i = 0; i < containers.size(); ++i)
240  {
241  const float progress = 1.f + i;
242  PLUGIN_PROGRESS("- Compiling 3D geometry...", progress,
243  containers.size());
244  auto& container = containers[i];
245  container.commitToModel();
246  }
247 
248  ModelMetadata metadata = {
249  {"Number of astrocytes", std::to_string(somas.size())}};
250 
251  _modelDescriptor.reset(new brayns::ModelDescriptor(std::move(model),
252  _details.assemblyName,
253  metadata));
254  if (_modelDescriptor)
255  _scene.addModel(_modelDescriptor);
256  else
257  PLUGIN_THROW("Astrocytes model could not be created");
258 }

260 void Astrocytes::_addEndFoot(ThreadSafeContainer& container,
261  const EndFootMap& endFeet, const doubles& radii,
262  const size_t materialId)
263 {
264  const double DEFAULT_ENDFOOT_RADIUS_RATIO = 1.2;
265  const auto radiusMultiplier = _details.radiusMultiplier;
266  for (const auto& endFoot : endFeet)
267  {
268  for (const auto& node : endFoot.second.nodes)
269  {
270  const auto& connector = DBConnector::getInstance();
271  const auto vasculatureNodes = connector.getVasculatureNodes(
272  _details.vasculaturePopulationName,
273  "section_guid=" +
274  std::to_string(endFoot.second.vasculatureSectionId));
275 
276  uint64_t startIndex = 0;
277  uint64_t endIndex = 1;
278  const auto halfLength = endFoot.second.length / 2.0;
279  auto it = vasculatureNodes.begin();
280  std::advance(it, endFoot.second.vasculatureSegmentId);
281  const auto centerPosition = it->second.position;
282 
283  double length = 0.0;
284  int64_t i = -1;
285  // Find start segment making the assumption that the segment Id is
286  // in the middle of the end-foot
287  while (length < halfLength &&
288  endFoot.second.vasculatureSegmentId + i >= 0)
289  {
290  const int64_t segmentId =
291  endFoot.second.vasculatureSegmentId + i;
292  if (segmentId < 0)
293  break;
294  auto it = vasculatureNodes.begin();
295  std::advance(it, segmentId);
296  length = glm::length(centerPosition - it->second.position);
297  startIndex = segmentId;
298  --i;
299  }
300 
301  length = 0.0;
302  i = 1;
303  // Now find the end segment
304  while (length < halfLength &&
305  endFoot.second.vasculatureSegmentId + i <
306  vasculatureNodes.size())
307  {
308  const int64_t segmentId =
309  endFoot.second.vasculatureSegmentId + i;
310  auto it = vasculatureNodes.begin();
311  std::advance(it, segmentId);
312  length = glm::length(centerPosition - it->second.position);
313  endIndex = segmentId;
314  ++i;
315  }
316 
317  // Build the segment using spheres
318  for (uint64_t i = startIndex; i < endIndex - 1; ++i)
319  {
320  auto it = vasculatureNodes.begin();
321  std::advance(it, i);
322  const auto& startNode = it->second;
323  const auto startRadius =
324  (it->first < radii.size() ? radii[it->first]
325  : startNode.radius) *
326  DEFAULT_ENDFOOT_RADIUS_RATIO * radiusMultiplier;
327 
328  std::advance(it, 1);
329  const auto& endNode = it->second;
330  const auto endRadius =
331  (it->first < radii.size() ? radii[it->first]
332  : startNode.radius) *
333  DEFAULT_ENDFOOT_RADIUS_RATIO * radiusMultiplier;
334 
335  if (!_details.useSdf)
336  container.addSphere(
337  startNode.position, startRadius, materialId,
338  NO_USER_DATA, {},
339  vasculature::DEFAULT_VASCULATURE_DISPLACEMENT);
340  container.addCone(
341  startNode.position, startRadius, endNode.position,
342  endRadius, materialId, NO_USER_DATA, {},
343  vasculature::DEFAULT_VASCULATURE_DISPLACEMENT);
344  }
345  }
346  }

349 void Astrocytes::setVasculatureRadiusReport(
350  const VasculatureRadiusReportDetails& details)
351 {
352  auto& connector = DBConnector::getInstance();
353  const auto simulationReport =
354  connector.getVasculatureSimulationReport(details.populationName,
355  details.simulationReportId);
356 
357  const size_t nbFrames =
358  (simulationReport.endTime - simulationReport.startTime) /
359  simulationReport.timeStep;
360  if (nbFrames == 0)
361  PLUGIN_THROW("Report does not contain any simulation data: " +
362  simulationReport.description);
363 
364  if (details.frame >= nbFrames)
365  PLUGIN_THROW("Invalid frame specified for report: " +
366  simulationReport.description);
367  const floats radii =
368  connector.getVasculatureSimulationTimeSeries(details.simulationReportId,
369  details.frame);
370  doubles series;
371  for (const double radius : radii)
372  series.push_back(details.amplitude * radius);
373  _buildModel(series);
374 }
