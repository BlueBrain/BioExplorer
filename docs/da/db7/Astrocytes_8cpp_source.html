48 size_t Astrocytes::_addSDFGeometry(SDFMorphologyData& sdfMorphologyData,
49  const SDFGeometry& geometry,
50  const std::set<size_t>& neighbours,
51  const size_t materialId, const int section)
52 {
53  const size_t idx = sdfMorphologyData.geometries.size();
54  sdfMorphologyData.geometries.push_back(geometry);
55  sdfMorphologyData.neighbours.push_back(neighbours);
56  sdfMorphologyData.materials.push_back(materialId);
57  sdfMorphologyData.geometrySection[idx] = section;
58  sdfMorphologyData.sectionGeometries[section].push_back(idx);
59  return idx;
60 }

62 void Astrocytes::_addStepConeGeometry(
63  const bool useSDF, const Vector3d& position, const double radius,
64  const Vector3d& target, const double previousRadius,
65  const size_t materialId, const uint64_t& userDataOffset, Model& model,
66  SDFMorphologyData& sdfMorphologyData, const uint32_t sdfGroupId)
67 {
68  if (useSDF)
69  {
70  const Vector3f displacementParams(std::min(radius, 0.05), 1.2, 2.0);
71  const auto geom =
72  (abs(radius - previousRadius) < 1e-6)
73  ? createSDFPill(position, target, radius, userDataOffset,
74  displacementParams)
75  : createSDFConePill(position, target, radius, previousRadius,
76  userDataOffset, displacementParams);
77  _addSDFGeometry(sdfMorphologyData, geom, {}, materialId, sdfGroupId);
78  }
79  else if (abs(radius - previousRadius) < 1e-6)
80  model.addCylinder(materialId,
81  {position, target, static_cast<float>(radius),
82  userDataOffset});
83  else
84  model.addCone(materialId,
85  {position, target, static_cast<float>(radius),
86  static_cast<float>(previousRadius), userDataOffset});
87 }

89 void Astrocytes::_finalizeSDFGeometries(Model& model,
90  SDFMorphologyData& sdfMorphologyData)
91 {
92  const size_t numGeoms = sdfMorphologyData.geometries.size();
93  sdfMorphologyData.localToGlobalIdx.resize(numGeoms, 0);
94 
95  // Extend neighbours to make sure smoothing is applied on all
96  // closely connected geometries
97  for (size_t rep = 0; rep < 4; rep++)
98  {
99  const size_t numNeighs = sdfMorphologyData.neighbours.size();
100  auto neighsCopy = sdfMorphologyData.neighbours;
101  for (size_t i = 0; i < numNeighs; i++)
102  {
103  for (size_t j : sdfMorphologyData.neighbours[i])
104  {
105  for (size_t newNei : sdfMorphologyData.neighbours[j])
106  {
107  neighsCopy[i].insert(newNei);
108  neighsCopy[newNei].insert(i);
109  }
110  }
111  }
112  sdfMorphologyData.neighbours = neighsCopy;
113  }
114 
115  for (size_t i = 0; i < numGeoms; i++)
116  {
117  // Convert neighbours from set to vector and erase itself from its
118  // neighbours
119  std::vector<size_t> neighbours;
120  const auto& neighSet = sdfMorphologyData.neighbours[i];
121  std::copy(neighSet.begin(), neighSet.end(),
122  std::back_inserter(neighbours));
123  neighbours.erase(std::remove_if(neighbours.begin(), neighbours.end(),
124  [i](size_t elem) { return elem == i; }),
125  neighbours.end());
126 
127  model.addSDFGeometry(sdfMorphologyData.materials[i],
128  sdfMorphologyData.geometries[i], neighbours);
129  }
130 }

132 void Astrocytes::_buildModel()
133 {
134  auto& connector = DBConnector::getInstance();
135 
136  auto model = _scene.createModel();
137  std::set<uint64_t> materialIds;
138  SDFMorphologyData sdfMorphologyData;
139  const auto useSdf = _details.useSdf;
140  const auto somas =
141  connector.getNodes(_details.astrocyteIds, _details.sqlFilter);
142 
143  // Astrocytes
144  size_t previousMaterialId = std::numeric_limits<size_t>::max();
145  size_t materialId = 0;
146  const uint64_t userData = 0;
147  Vector3ui indexOffset;
148 
149  for (const auto& soma : somas)
150  {
151  const auto somaId = soma.first;
152  const auto& somaCenter = soma.second.center;
153  const auto somaRadius = soma.second.radius;
154 
155  PLUGIN_PROGRESS("Loading astrocytes", soma.first, somas.size());
156  switch (_details.populationColorScheme)
157  {
158  case PopulationColorScheme::id:
159  materialId = somaId * NB_MATERIALS_PER_ASTROCYTE;
160  break;
161  default:
162  materialId = 0;
163  }
164  materialIds.insert(materialId);
165 
166  if (_details.loadSomas)
167  {
168  if (useSdf)
169  _addSDFGeometry(sdfMorphologyData,
170  createSDFSphere(somaCenter, somaRadius,
171  userData),
172  {}, materialId, -1);
173  else
174  model->addSphere(materialId,
175  {somaCenter, static_cast<float>(somaRadius),
176  userData});
177  }
178 
179  if (!_details.loadDendrites && !_details.loadEndFeet)
180  continue;
181 
182  const auto sections = connector.getAstrocyteSections(somaId);
183  // End feet
184  const auto endFeet =
185  (_details.loadEndFeet ? connector.getAstrocyteEndFeetAreas(somaId)
186  : EndFootMap());
187 
188  for (const auto& section : sections)
189  {
190  const auto sectionId = section.first;
191  switch (_details.morphologyColorScheme)
192  {
193  case MorphologyColorScheme::section:
194  materialId += section.second.type;
195  break;
196  default:
197  break;
198  }
199  materialIds.insert(materialId);
200 
201  const auto& points = section.second.points;
202  size_t step = 1;
203  switch (_details.geometryQuality)
204  {
205  case GeometryQuality::low:
206  step = points.size() - 1;
207  break;
208  default:
209  break;
210  }
211 
212  if (_details.loadDendrites)
213  {
214  if (section.second.parentId == -1)
215  {
216  const auto& point = points[0];
217  _addStepConeGeometry(useSdf, somaCenter, somaRadius,
218  somaCenter + Vector3d(point), point.w,
219  materialId, userData, *model,
220  sdfMorphologyData, sectionId);
221  }
222 
223  for (uint64_t i = 0; i < points.size() - 2; i += step)
224  {
225  const auto& srcPoint = points[i];
226  const auto src = somaCenter + Vector3d(srcPoint);
227  const float srcRadius = srcPoint.w * 0.5;
228 
229  _bounds.merge(srcPoint);
230 
231  const auto& dstPoint = points[i + step];
232  const auto dst = somaCenter + Vector3d(dstPoint);
233  const float dstRadius = dstPoint.w * 0.5;
234 
235  if (useSdf)
236  _addSDFGeometry(sdfMorphologyData,
237  createSDFSphere(dst, dstRadius,
238  userData),
239  {}, materialId, sectionId);
240  else
241  model->addSphere(materialId,
242  {dst, dstRadius, userData});
243 
244  _addStepConeGeometry(useSdf, src, srcRadius, dst, dstRadius,
245  materialId, userData, *model,
246  sdfMorphologyData, sectionId);
247  }
248  }
249 
250  if (_details.loadEndFeet)
251  {
252  const auto it = endFeet.find(sectionId);
253  if (it != endFeet.end())
254  {
255  const auto endFoot = (*it).second;
256  auto& tm = model->getTriangleMeshes()[materialId];
257  for (const auto& vertex : endFoot.vertices)
258  {
259  _bounds.merge(vertex);
260  tm.vertices.push_back(vertex);
261  }
262  for (const auto& index : endFoot.indices)
263  tm.indices.push_back(index + indexOffset);
264 
265  const auto nbVertices = endFoot.vertices.size();
266  indexOffset +=
267  Vector3ui(nbVertices, nbVertices, nbVertices);
268  }
269  }
270  if (materialId != previousMaterialId)
271  indexOffset = Vector3ui();
272  previousMaterialId = materialId;
273  }
274  }
275 
276  for (const auto materialId : materialIds)
277  {
278  Vector3f color{1.f, 1.f, 1.f};
279  auto nodeMaterial =
280  model->createMaterial(materialId, std::to_string(materialId));
281  nodeMaterial->setDiffuseColor(color);
282  nodeMaterial->setSpecularColor(color);
283  nodeMaterial->setSpecularExponent(100.f);
284  PropertyMap props;
285  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE, 0});
286  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
287  props.setProperty({MATERIAL_PROPERTY_CHAMELEON_MODE, 0});
288  props.setProperty({MATERIAL_PROPERTY_CAST_USER_DATA, false});
289  props.setProperty({MATERIAL_PROPERTY_NODE_ID, 0});
290  nodeMaterial->updateProperties(props);
291  }
292 
293  if (useSdf)
294  _finalizeSDFGeometries(*model, sdfMorphologyData);
295 
296  ModelMetadata metadata = {
297  {"Number of astrocytes", std::to_string(_details.astrocyteIds.size())}};
298 
299  _modelDescriptor.reset(new brayns::ModelDescriptor(std::move(model),
300  _details.assemblyName,
301  metadata));
302  if (_modelDescriptor)
303  _scene.addModel(_modelDescriptor);
304  else
305  PLUGIN_THROW("Astrocytes model could not be created");
306 }
