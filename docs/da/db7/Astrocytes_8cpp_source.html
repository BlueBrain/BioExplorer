78 void Astrocytes::_buildModel(const doubles& radii)
79 {
80  const auto animationParams =
81  doublesToMolecularSystemAnimationDetails(_details.animationParams);
82  srand(animationParams.seed);
83 
84  if (_modelDescriptor)
85  _scene.removeModel(_modelDescriptor->getModelID());
86 
87  auto& connector = DBConnector::getInstance();
88 
89  auto model = _scene.createModel();
90  const auto realismLevel = _details.realismLevel;
91  const auto somas =
92  connector.getAstrocytes(_details.populationName, _details.sqlFilter);
93  const auto loadEndFeet = !_details.vasculaturePopulationName.empty();
94 
95  PLUGIN_INFO(1, "Building " << somas.size() << " astrocytes");
96  _logRealismParams();
97 
98  // Astrocytes
99  size_t baseMaterialId = 0;
100  const uint64_t userData = 0;
101 
102  ThreadSafeContainers containers;
103  const auto nbDBConnections = DBConnector::getInstance().getNbConnections();
104  uint64_t index;
105 #pragma omp parallel for num_threads(nbDBConnections)
106  for (index = 0; index < somas.size(); ++index)
107  {
108  if (omp_get_thread_num() == 0)
109  PLUGIN_PROGRESS("Loading astrocytes", index,
110  somas.size() / nbDBConnections);
111 
112  auto it = somas.begin();
113  std::advance(it, index);
114  const auto& soma = it->second;
115  const auto somaId = it->first;
116 
117  ThreadSafeContainer container(*model, _scale);
118 
119  // Load data from DB
120  double somaRadius = 0.0;
121  SectionMap sections;
122  if (_details.loadSomas || _details.loadDendrites)
123  sections =
124  connector.getAstrocyteSections(_details.populationName, somaId);
125 
126  EndFootMap endFeet;
127  if (loadEndFeet)
128  endFeet = connector.getAstrocyteEndFeet(
129  _details.vasculaturePopulationName, somaId);
130 
131  // Soma radius
132  uint64_t count = 1;
133  for (const auto& section : sections)
134  if (section.second.parentId == SOMA_AS_PARENT)
135  {
136  const auto& point = section.second.points[0];
137  somaRadius += 0.75 * length(Vector3d(point));
138  ++count;
139  }
140  somaRadius = _radiusMultiplier * somaRadius / count;
141  const auto somaPosition =
142  _animatedPosition(Vector4d(soma.center, somaRadius), somaId);
143 
144  // Color scheme
145  switch (_details.populationColorScheme)
146  {
147  case PopulationColorScheme::id:
148  baseMaterialId = somaId * NB_MATERIALS_PER_MORPHOLOGY;
149  break;
150  default:
151  baseMaterialId = 0;
152  }
153 
154  const auto somaMaterialId =
155  baseMaterialId + (_details.morphologyColorScheme ==
156  MorphologyColorScheme::section_type
157  ? MATERIAL_OFFSET_SOMA
158  : 0);
159 
160  uint64_t somaGeometryIndex = 0;
161  if (_details.loadSomas)
162  {
163  const bool useSdf =
164  andCheck(static_cast<uint32_t>(_details.realismLevel),
165  static_cast<uint32_t>(MorphologyRealismLevel::soma));
166  somaGeometryIndex = container.addSphere(
167  somaPosition, somaRadius, somaMaterialId, NO_USER_DATA, useSdf,
168  {},
169  Vector3f(somaRadius * astrocyteSomaDisplacementStrength,
170  somaRadius * astrocyteSomaDisplacementFrequency, 0.f));
171  if (_details.generateInternals)
172  {
173  const auto useSdf =
174  andCheck(static_cast<uint32_t>(_details.realismLevel),
175  static_cast<uint32_t>(
176  MorphologyRealismLevel::internals));
177  _addSomaInternals(container, baseMaterialId, somaPosition,
178  somaRadius, DEFAULT_MITOCHONDRIA_DENSITY,
179  useSdf);
180  }
181  }
182 
183  Neighbours neighbours;
184  neighbours.insert(somaGeometryIndex);
185  for (const auto& section : sections)
186  {
187  uint64_t geometryIndex = 0;
188  const auto& points = section.second.points;
189 
190  size_t sectionMaterialId = baseMaterialId;
191  const auto sectionId = section.first;
192  switch (_details.morphologyColorScheme)
193  {
194  case MorphologyColorScheme::section_type:
195  sectionMaterialId = baseMaterialId + section.second.type;
196  break;
197  case MorphologyColorScheme::section_orientation:
198  {
199  sectionMaterialId = getMaterialIdFromOrientation(
200  Vector3d(points[points.size() - 1]) - Vector3d(points[0]));
201  break;
202  }
203  default:
204  break;
205  }
206  size_t step = 1;
207  switch (_details.morphologyRepresentation)
208  {
209  case MorphologyRepresentation::section:
210  step = points.size() - 2;
211  break;
212  default:
213  break;
214  }
215 
216  if (_details.loadDendrites)
217  {
218  const bool useSdf =
219  andCheck(static_cast<uint32_t>(_details.realismLevel),
220  static_cast<uint32_t>(
221  MorphologyRealismLevel::dendrite));
222  uint64_t geometryIndex = 0;
223  if (section.second.parentId == SOMA_AS_PARENT)
224  {
225  // Section connected to the soma
226  const auto& point = points[0];
227  const float srcRadius =
228  somaRadius * 0.75f * _radiusMultiplier;
229  const float dstRadius = point.w * 0.5f * _radiusMultiplier;
230  const auto dstPosition = _animatedPosition(
231  Vector4d(somaPosition + Vector3d(point), dstRadius),
232  somaId);
233  geometryIndex = container.addCone(
234  somaPosition, srcRadius, dstPosition, dstRadius,
235  somaMaterialId, useSdf, userData, neighbours,
236  Vector3f(srcRadius * astrocyteSomaDisplacementStrength,
237  srcRadius * astrocyteSomaDisplacementFrequency,
238  0.f));
239  neighbours.insert(geometryIndex);
240  }
241 
242  for (uint64_t i = 0; i < points.size() - 1; i += step)
243  {
244  const auto srcPoint = points[i];
245  const float srcRadius =
246  srcPoint.w * 0.5 * _radiusMultiplier;
247  const auto src = _animatedPosition(
248  Vector4d(somaPosition + Vector3d(srcPoint), srcRadius),
249  somaId);
250 
251  // Ignore points that are too close the previous one
252  // (according to respective radii)
253  Vector4f dstPoint;
254  float dstRadius;
255  do
256  {
257  dstPoint = points[i + step];
258  dstRadius = dstPoint.w * 0.5 * _radiusMultiplier;
259  ++i;
260  } while (length(Vector3f(dstPoint) - Vector3f(srcPoint)) <
261  (srcRadius + dstRadius) &&
262  (i + step) < points.size() - 1);
263  --i;
264 
265  const auto dst = _animatedPosition(
266  Vector4d(somaPosition + Vector3d(dstPoint), dstRadius),
267  somaId);
268  if (!useSdf)
269  geometryIndex =
270  container.addSphere(dst, dstRadius,
271  sectionMaterialId, useSdf,
272  NO_USER_DATA);
273 
274  geometryIndex = container.addCone(
275  src, srcRadius, dst, dstRadius, sectionMaterialId,
276  useSdf, userData, {geometryIndex},
277  Vector3f(srcRadius * sectionDisplacementStrength,
278  sectionDisplacementFrequency, 0.f));
279 
280  _bounds.merge(srcPoint);
281  }
282  }
283  }
284 
285  if (loadEndFeet)
286  _addEndFoot(container, endFeet, radii, somaMaterialId);
287 #pragma omp critical
288  containers.push_back(container);
289  }
290 
291  for (uint64_t i = 0; i < containers.size(); ++i)
292  {
293  PLUGIN_PROGRESS("- Compiling 3D geometry...", i, containers.size());
294  auto& container = containers[i];
295  container.commitToModel();
296  }
297 
298  ModelMetadata metadata = {
299  {"Number of astrocytes", std::to_string(somas.size())}};
300 
301  _modelDescriptor.reset(new brayns::ModelDescriptor(std::move(model),
302  _details.assemblyName,
303  metadata));
304  if (_modelDescriptor)
305  _scene.addModel(_modelDescriptor);
306  else
307  PLUGIN_THROW("Astrocytes model could not be created");
308 }

310 void Astrocytes::_addEndFoot(ThreadSafeContainer& container,
311  const EndFootMap& endFeet, const doubles& radii,
312  const size_t materialId)
313 {
314  const auto radiusMultiplier = _details.radiusMultiplier;
315  const Vector3d displacement{vasculatureSegmentDisplacementStrength,
316  vasculatureSegmentDisplacementFrequency, 0.0};
317  const auto useSdf =
318  andCheck(static_cast<uint32_t>(_details.realismLevel),
319  static_cast<uint32_t>(VasculatureRealismLevel::section));
320 
321  for (const auto& endFoot : endFeet)
322  {
323  for (const auto& node : endFoot.second.nodes)
324  {
325  const auto& connector = DBConnector::getInstance();
326  const auto vasculatureNodes = connector.getVasculatureNodes(
327  _details.vasculaturePopulationName,
328  "section_guid=" +
329  std::to_string(endFoot.second.vasculatureSectionId));
330 
331  uint64_t startIndex = 0;
332  uint64_t endIndex = 1;
333  const auto halfLength = endFoot.second.length / 2.0;
334  auto it = vasculatureNodes.begin();
335  std::advance(it, endFoot.second.vasculatureSegmentId);
336  const auto centerPosition = it->second.position;
337 
338  double length = 0.0;
339  int64_t i = -1;
340  // Find start segment making the assumption that the segment Id is
341  // in the middle of the end-foot
342  while (length < halfLength &&
343  endFoot.second.vasculatureSegmentId + i >= 0)
344  {
345  const int64_t segmentId =
346  endFoot.second.vasculatureSegmentId + i;
347  if (segmentId < 0)
348  break;
349  auto it = vasculatureNodes.begin();
350  std::advance(it, segmentId);
351  length = glm::length(centerPosition - it->second.position);
352  startIndex = segmentId;
353  --i;
354  }
355 
356  length = 0.0;
357  i = 1;
358  // Now find the end segment
359  while (length < halfLength &&
360  endFoot.second.vasculatureSegmentId + i <
361  vasculatureNodes.size())
362  {
363  const int64_t segmentId =
364  endFoot.second.vasculatureSegmentId + i;
365  auto it = vasculatureNodes.begin();
366  std::advance(it, segmentId);
367  length = glm::length(centerPosition - it->second.position);
368  endIndex = segmentId;
369  ++i;
370  }
371 
372  // Build the segment using spheres and cones
373  uint64_t geometryIndex = 0;
374  Neighbours neighbours;
375  for (uint64_t i = startIndex; i < endIndex - 1; ++i)
376  {
377  auto it = vasculatureNodes.begin();
378  std::advance(it, i);
379  const auto& srcNode = it->second;
380  const auto srcUserData = it->first;
381  const auto srcVasculatureRadius =
382  (srcUserData < radii.size() ? radii[srcUserData]
383  : srcNode.radius) *
384  radiusMultiplier;
385  const float srcEndFootRadius =
386  DEFAULT_ENDFOOT_RADIUS_RATIO * srcVasculatureRadius;
387 
388  ++it;
389  const auto& dstNode = it->second;
390  const auto dstUserData = it->first;
391  const auto dstVasculatureRadius =
392  (dstUserData < radii.size() ? radii[dstUserData]
393  : dstNode.radius) *
394  radiusMultiplier;
395  const float dstEndFootRadius =
396  DEFAULT_ENDFOOT_RADIUS_RATIO * dstVasculatureRadius;
397 
398  const auto srcPosition = _animatedPosition(
399  Vector4d(srcNode.position, srcVasculatureRadius));
400  const auto dstPosition = _animatedPosition(
401  Vector4d(dstNode.position, dstVasculatureRadius));
402 
403  if (!useSdf)
404  container.addSphere(srcPosition, srcEndFootRadius,
405  materialId, useSdf, srcUserData);
406  geometryIndex =
407  container.addCone(srcPosition, srcEndFootRadius,
408  dstPosition, dstEndFootRadius, materialId,
409  useSdf, srcUserData, neighbours,
410  displacement);
411  neighbours = {geometryIndex};
412  }
413  }
414  }

417 void Astrocytes::setVasculatureRadiusReport(
418  const VasculatureRadiusReportDetails& details)
419 {
420  auto& connector = DBConnector::getInstance();
421  const auto simulationReport =
422  connector.getSimulationReport(details.populationName,
423  details.simulationReportId);
424 
425  const size_t nbFrames =
426  (simulationReport.endTime - simulationReport.startTime) /
427  simulationReport.timeStep;
428  if (nbFrames == 0)
429  PLUGIN_THROW("Report does not contain any simulation data: " +
430  simulationReport.description);
431 
432  if (details.frame >= nbFrames)
433  PLUGIN_THROW("Invalid frame specified for report: " +
434  simulationReport.description);
435  const floats radii =
436  connector.getVasculatureSimulationTimeSeries(details.populationName,
437  details.simulationReportId,
438  details.frame);
439  doubles series;
440  for (const double radius : radii)
441  series.push_back(details.amplitude * radius);
442  _buildModel(series);
443 }
