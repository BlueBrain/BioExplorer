26 inline vec3f DepthRenderer_shadeRay(const uniform DepthRenderer* uniform self, varying ScreenSample& sample)
27 {
28  Ray ray = sample.ray;
29  sample.z = inf;
30  sample.alpha = 1.f;
31 
32  traceRay(self->super.model, ray);
33 
34  if (ray.geomID < 0)
35  // No intersection, use environment map or background color
36  return make_vec3f(1.f);
37 
38  // Z-Depth
39  sample.z = ray.t;
40 
41  return make_vec3f(ray.t / self->infinity);
42 }
