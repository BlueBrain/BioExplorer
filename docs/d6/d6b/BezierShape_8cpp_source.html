44 Transformation BezierShape::getTransformation(
45  const uint64_t occurrence, const uint64_t nbOccurrences,
46  const AnimationDetails& animationDetails, const double offset) const
47 {
48  Vector3ds bezierPoints = _points;
49  size_t i = bezierPoints.size() - 1;
50  while (i > 0)
51  {
52  for (size_t k = 0; k < i; ++k)
53  bezierPoints[k] =
54  bezierPoints[k] + static_cast<double>(occurrence) *
55  (bezierPoints[k + 1] - bezierPoints[k]);
56  --i;
57  }
58  const Vector3d normal =
59  cross({0.0, 0.0, 1.0}, normalize(bezierPoints[1] - bezierPoints[0]));
60 
61  Vector3d pos = bezierPoints[0];
62 
63  if (isClipped(pos, _clippingPlanes))
64  throw std::runtime_error("Instance is clipped");
65 
66  const Quaterniond rot = safeQuatlookAt(normal);
67 
68  pos += normal * offset;
69 
70  Transformation transformation;
71  transformation.setTranslation(pos);
72  transformation.setRotation(rot);
73  return transformation;
74 }
