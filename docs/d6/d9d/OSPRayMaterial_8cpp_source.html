61 void OSPRayMaterial::commit()
62 {
63  // Do nothing until this material is instanced for a specific renderer
64  if (!_ospMaterial || !isModified())
65  return;
66 
67  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_DIFFUSE_COLOR, Vector3f(_diffuseColor));
68  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_SPECULAR_COLOR, Vector3f(_specularColor));
69  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_SPECULAR_INDEX, static_cast<float>(_specularExponent));
70 #if 0
71  // For some unknown reason, this simply does not work!?!
72  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_OPACITY, static_cast<float>(_opacity));
73 #else
74  osphelper::set(_ospMaterial, "opacity", static_cast<float>(_opacity));
75 #endif
76  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_REFRACTION, static_cast<float>(_refractionIndex));
77  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_REFLECTION, static_cast<float>(_reflectionIndex));
78  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_EMISSION, static_cast<float>(_emission));
79  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_GLOSSINESS, static_cast<float>(_glossiness));
80  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_SHADING_MODE, static_cast<MaterialShadingMode>(_shadingMode));
81  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_USER_PARAMETER, static_cast<float>(_userParameter));
82  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_CAST_USER_DATA, static_cast<int32_t>(_castUserData));
83  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_CLIPPING_MODE, static_cast<MaterialClippingMode>(_clippingMode));
84  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_CHAMELEON_MODE, static_cast<MaterialChameleonMode>(_chameleonMode));
85  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_NODE_ID, static_cast<int32_t>(_nodeId));
86 
87  // Properties
88  toOSPRayProperties(*this, _ospMaterial);
89 
90  // Textures
91  for (const auto& textureType : textureTypeMaterialAttribute)
92  ospSetObject(_ospMaterial, textureType.attribute.c_str(), nullptr);
93 
94  for (const auto& textureDescriptor : _textureDescriptors)
95  {
96  const auto texType = textureDescriptor.first;
97  auto texture = getTexture(texType);
98  if (texture)
99  {
100  auto ospTexture = _createOSPTexture2D(texture);
101  const auto str = textureTypeMaterialAttribute[int(texType)].attribute.c_str();
102  ospSetObject(_ospMaterial, str, ospTexture);
103  ospRelease(ospTexture);
104  }
105  }
106 
107  ospCommit(_ospMaterial);
108  resetModified();
109 }

111 void OSPRayMaterial::commit(const std::string& renderer)
112 {
113  if (!isModified() && renderer == _renderer)
114  return;
115 
116  ospRelease(_ospMaterial);
117  _ospMaterial = ospNewMaterial2(renderer.c_str(), DEFAULT);
118  if (!_ospMaterial)
119  throw std::runtime_error("Could not create material for renderer '" + renderer + "'");
120  _renderer = renderer;
121  markModified(false); // Ensure commit recreates the ISPC object
122  commit();
123 }

125 OSPTexture OSPRayMaterial::_createOSPTexture2D(Texture2DPtr texture)
126 {
127  OSPTextureFormat type = OSP_TEXTURE_R8; // smallest valid type as default
128  if (texture->depth == 1)
129  {
130  if (texture->channels == 1)
131  type = OSP_TEXTURE_R8;
132  if (texture->channels == 3)
133  type = OSP_TEXTURE_RGB8;
134  if (texture->channels == 4)
135  type = OSP_TEXTURE_RGBA8;
136  }
137  else if (texture->depth == 4)
138  {
139  if (texture->channels == 1)
140  type = OSP_TEXTURE_R32F;
141  if (texture->channels == 3)
142  type = OSP_TEXTURE_RGB32F;
143  if (texture->channels == 4)
144  type = OSP_TEXTURE_RGBA32F;
145  }
146 
147  CORE_DEBUG("Creating OSPRay texture from " << texture->filename << ": " << texture->width << "x" << texture->height
148  << "x" << (int)type);
149 
150  OSPTexture ospTexture = ospNewTexture(OSPRAY_MATERIAL_TEXTURE_2D);
151 
152  const Vector2i size{int(texture->width), int(texture->height)};
153 
154  osphelper::set(ospTexture, OSPRAY_MATERIAL_PROPERTY_TEXTURE_TYPE, static_cast<int>(type));
155  osphelper::set(ospTexture, OSPRAY_MATERIAL_PROPERTY_TEXTURE_SIZE, size);
156  auto textureData =
157  ospNewData(texture->getSizeInBytes(), OSP_RAW, texture->getRawData<unsigned char>(), OSP_DATA_SHARED_BUFFER);
158  ospSetObject(ospTexture, OSPRAY_MATERIAL_PROPERTY_TEXTURE_DATA, textureData);
159  ospRelease(textureData);
160  ospCommit(ospTexture);
161 
162  return ospTexture;
163 }
