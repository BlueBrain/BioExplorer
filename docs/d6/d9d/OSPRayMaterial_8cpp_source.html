56 void OSPRayMaterial::commit()
57 {
58  // Do nothing until this material is instanced for a specific renderer
59  if (!_ospMaterial || !isModified())
60  return;
61 
62  if (getCurrentType() == "simulation")
63  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_APPLY_SIMULATION, true);
64  else
65  ospRemoveParam(_ospMaterial, MATERIAL_PROPERTY_APPLY_SIMULATION);
66 
67  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_DIFFUSE_COLOR, Vector3f(_diffuseColor));
68  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_SPECULAR_COLOR, Vector3f(_specularColor));
69  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_SPECULAR_INDEX, static_cast<float>(_specularExponent));
70  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_OPACITY, static_cast<float>(_opacity));
71  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_REFRACTION, static_cast<float>(_refractionIndex));
72  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_REFLECTION, static_cast<float>(_reflectionIndex));
73  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_EMISSION, static_cast<float>(_emission));
74  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_GLOSSINESS, static_cast<float>(_glossiness));
75  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_SHADING_MODE, static_cast<MaterialShadingMode>(_shadingMode));
76  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_USER_PARAMETER, static_cast<float>(_userParameter));
77  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_CAST_USER_DATA, static_cast<bool>(_castUserData));
78  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_CLIPPING_MODE, static_cast<MaterialClippingMode>(_clippingMode));
79  osphelper::set(_ospMaterial, MATERIAL_PROPERTY_SKYBOX, _isBackGroundMaterial);
80 
81  // Properties
82  toOSPRayProperties(*this, _ospMaterial);
83 
84  // Textures
85  for (const auto& textureType : textureTypeMaterialAttribute)
86  ospSetObject(_ospMaterial, textureType.attribute.c_str(), nullptr);
87 
88  for (const auto& textureDescriptor : _textureDescriptors)
89  {
90  const auto texType = textureDescriptor.first;
91  auto texture = getTexture(texType);
92  if (texture)
93  {
94  auto ospTexture = _createOSPTexture2D(texture);
95  const auto str = textureTypeMaterialAttribute[int(texType)].attribute.c_str();
96  ospSetObject(_ospMaterial, str, ospTexture);
97  ospRelease(ospTexture);
98  }
99  }
100 
101  ospCommit(_ospMaterial);
102  resetModified();
103 }

105 void OSPRayMaterial::commit(const std::string& renderer)
106 {
107  if (!isModified() && renderer == _renderer)
108  return;
109 
110  ospRelease(_ospMaterial);
111  _ospMaterial = ospNewMaterial2(renderer.c_str(), "default");
112  if (!_ospMaterial)
113  throw std::runtime_error("Could not create material for renderer '" + renderer + "'");
114  _renderer = renderer;
115  markModified(false); // Ensure commit recreates the ISPC object
116  commit();
117 }

119 OSPTexture OSPRayMaterial::_createOSPTexture2D(Texture2DPtr texture)
120 {
121  OSPTextureFormat type = OSP_TEXTURE_R8; // smallest valid type as default
122  if (texture->depth == 1)
123  {
124  if (texture->channels == 1)
125  type = OSP_TEXTURE_R8;
126  if (texture->channels == 3)
127  type = OSP_TEXTURE_RGB8;
128  if (texture->channels == 4)
129  type = OSP_TEXTURE_RGBA8;
130  }
131  else if (texture->depth == 4)
132  {
133  if (texture->channels == 1)
134  type = OSP_TEXTURE_R32F;
135  if (texture->channels == 3)
136  type = OSP_TEXTURE_RGB32F;
137  if (texture->channels == 4)
138  type = OSP_TEXTURE_RGBA32F;
139  }
140 
141  CORE_DEBUG("Creating OSPRay texture from " << texture->filename << ": " << texture->width << "x" << texture->height
142  << "x" << (int)type);
143 
144  OSPTexture ospTexture = ospNewTexture("texture2d");
145 
146  const Vector2i size{int(texture->width), int(texture->height)};
147 
148  osphelper::set(ospTexture, "type", static_cast<int>(type));
149  osphelper::set(ospTexture, "size", size);
150  auto textureData =
151  ospNewData(texture->getSizeInBytes(), OSP_RAW, texture->getRawData<unsigned char>(), OSP_DATA_SHARED_BUFFER);
152  ospSetObject(ospTexture, "data", textureData);
153  ospRelease(textureData);
154  ospCommit(ospTexture);
155 
156  return ospTexture;
157 }
