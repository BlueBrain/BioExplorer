30void AdvancedMaterial::commit()
31{
32 if (ispcEquivalent == nullptr)
33 ispcEquivalent = ispc::AdvancedMaterial_create(this);
34
35 // Opacity
36 map_d = (ospray::Texture2D*)getParamObject("map_d", nullptr);
37 xform_d = getTextureTransform("map_d");
38 d = getParam1f("d", 1.f);
39
40 // Diffuse color
41 Kd = getParam3f("kd", ospray::vec3f(.8f));
42 map_Kd = (ospray::Texture2D*)getParamObject("map_kd", nullptr);
43 xform_Kd = getTextureTransform("map_kd");
44
45 // Specular color
46 Ks = getParam3f("ks", ospray::vec3f(0.f));
47 map_Ks = (ospray::Texture2D*)getParamObject("map_ks", nullptr);
48 xform_Ks = getTextureTransform("map_ks");
49
50 // Specular exponent
51 Ns = getParam1f("ns", 10.f);
52 map_Ns = (ospray::Texture2D*)getParamObject("map_ns", nullptr);
53 xform_Ns = getTextureTransform("map_ns");
54
55 // Bump mapping
56 map_Bump = (ospray::Texture2D*)getParamObject("map_bump", nullptr);
57 xform_Bump = getTextureTransform("map_bump");
58 rot_Bump = xform_Bump.l.orthogonal().transposed();
59
60 // Refraction mapping
61 refraction = getParam1f("refraction", 0.f);
62 xform_Refraction = getTextureTransform("map_refraction");
63 map_Refraction =
64 (ospray::Texture2D*)getParamObject("map_refraction", nullptr);
65
66 // Reflection mapping
67 reflection = getParam1f("reflection", 0.f);
68 xform_Reflection = getTextureTransform("map_reflection");
69 map_Reflection =
70 (ospray::Texture2D*)getParamObject("map_reflection", nullptr);
71
72 // Light emission mapping
73 a = getParam1f("a", 0.f);
74 xform_a = getTextureTransform("map_a");
75 map_a = (ospray::Texture2D*)getParamObject("map_a", nullptr);
76
77 // Glossiness
78 glossiness = getParam1f("glossiness", 1.f);
79
80 // Shading mode
81 shadingMode = static_cast<MaterialShadingMode>(getParam1i(
82 MATERIAL_PROPERTY_SHADING_MODE,
83 static_cast<int>(MaterialShadingMode::undefined_shading_mode)));
84
85 // User parameter
86 userParameter = getParam1f(MATERIAL_PROPERTY_USER_PARAMETER, 1.f);
87
88 // Chameleon mode
89 chameleonMode = static_cast<MaterialChameleonMode>(getParam1i(
90 MATERIAL_PROPERTY_CHAMELEON_MODE,
91 static_cast<int>(MaterialChameleonMode::undefined_chameleon_mode)));
92
93 // Model Id
94 nodeId = getParam1i(MATERIAL_PROPERTY_NODE_ID, 0);
95
96 ispc::AdvancedMaterial_set(
97 getIE(), map_d ? map_d->getIE() : nullptr,
98 (const ispc::AffineSpace2f&)xform_d, d,
99 map_Refraction ? map_Refraction->getIE() : nullptr,
100 (const ispc::AffineSpace2f&)xform_Refraction, refraction,
101 map_Reflection ? map_Reflection->getIE() : nullptr,
102 (const ispc::AffineSpace2f&)xform_Reflection, reflection,
103 map_a ? map_a->getIE() : nullptr, (const ispc::AffineSpace2f&)xform_a,
104 a, glossiness, map_Kd ? map_Kd->getIE() : nullptr,
105 (const ispc::AffineSpace2f&)xform_Kd, (ispc::vec3f&)Kd,
106 map_Ks ? map_Ks->getIE() : nullptr,
107 (const ispc::AffineSpace2f&)xform_Ks, (ispc::vec3f&)Ks,
108 map_Ns ? map_Ns->getIE() : nullptr,
109 (const ispc::AffineSpace2f&)xform_Ns, Ns,
110 map_Bump ? map_Bump->getIE() : nullptr,
111 (const ispc::AffineSpace2f&)xform_Bump,
112 (const ispc::LinearSpace2f&)rot_Bump,
113 (const ispc::MaterialShadingMode&)shadingMode, userParameter,
114 (const ispc::MaterialChameleonMode&)chameleonMode, nodeId);
115}
