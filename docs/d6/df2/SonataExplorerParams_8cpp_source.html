34 #else
35 #define FROM_JSON(PARAM, JSON, NAME) \
36  try \
37  { \
38  PARAM.NAME = JSON[#NAME].get<decltype(PARAM.NAME)>(); \
39  } \
40  catch (...) \
41  { \
42  PLUGIN_ERROR("JSON parsing error for attribute '" << #NAME << "'!"); \
43  throw; \
44  }
45 #endif
46 #define TO_JSON(PARAM, JSON, NAME) JSON[#NAME] = PARAM.NAME
