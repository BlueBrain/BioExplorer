89 void OSPRayModel::buildBoundingBox()
90 {
91  if (_boundingBoxModel)
92  return;
93  _boundingBoxModel = ospNewModel();
94 
95  auto material = createMaterial(BOUNDINGBOX_MATERIAL_ID, "bounding_box");
96  material->setDiffuseColor({1, 1, 1});
97  material->setEmission(1.f);
98  const Vector3f s(0.5f);
99  const Vector3f c(0.5f);
100  const float radius = 0.005f;
101  const Vector3f positions[8] = {
102  {c.x - s.x, c.y - s.y, c.z - s.z}, {c.x + s.x, c.y - s.y, c.z - s.z}, // 6--------7
103  {c.x - s.x, c.y + s.y, c.z - s.z}, // /| /|
104  {c.x + s.x, c.y + s.y, c.z - s.z}, // 2--------3 |
105  {c.x - s.x, c.y - s.y, c.z + s.z}, // | | | |
106  {c.x + s.x, c.y - s.y, c.z + s.z}, // | 4------|-5
107  {c.x - s.x, c.y + s.y, c.z + s.z}, // |/ |/
108  {c.x + s.x, c.y + s.y, c.z + s.z} // 0--------1
109  };
110 
111  for (size_t i = 0; i < 8; ++i)
112  addSphere(BOUNDINGBOX_MATERIAL_ID, Sphere(positions[i], radius));
113 
114  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[0], positions[1], radius});
115  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[2], positions[3], radius});
116  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[4], positions[5], radius});
117  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[6], positions[7], radius});
118 
119  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[0], positions[2], radius});
120  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[1], positions[3], radius});
121  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[4], positions[6], radius});
122  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[5], positions[7], radius});
123 
124  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[0], positions[4], radius});
125  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[1], positions[5], radius});
126  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[2], positions[6], radius});
127  addCylinder(BOUNDINGBOX_MATERIAL_ID, {positions[3], positions[7], radius});
128 }

130 void OSPRayModel::_addGeometryToModel(const OSPGeometry geometry, const size_t materialId)
131 {

149 OSPGeometry& OSPRayModel::_createGeometry(GeometryMap& map, const size_t materialId, const char* name)
150 {
151  auto& geometry = map[materialId];
152  if (geometry)
153  {
154  ospRemoveGeometry(_primaryModel, geometry);
155  ospRelease(geometry);
156  }
157  geometry = ospNewGeometry(name);
158 
159  auto matIt = _materials.find(materialId);
160  if (matIt != _materials.end())
161  {
162  auto material = std::static_pointer_cast<OSPRayMaterial>(matIt->second);
163  if (material->getOSPMaterial())
164  ospSetMaterial(geometry, material->getOSPMaterial());
165  }
166 
167  return geometry;
168 }

312 void OSPRayModel::_commitSDFGeometries()
313 {
314  if (_geometries->_sdf.geometries.empty())
315  return;
316 
317  auto globalData = allocateVectorData(_geometries->_sdf.geometries, OSP_CHAR, _memoryManagementFlags);
318 
319  // Create and upload flat list of neighbours
320  const size_t numGeoms = _geometries->_sdf.geometries.size();
321  _geometries->_sdf.neighboursFlat.clear();
322 
323  for (size_t geomI = 0; geomI < numGeoms; geomI++)
324  {
325  const size_t currOffset = _geometries->_sdf.neighboursFlat.size();
326  const auto& neighsI = _geometries->_sdf.neighbours[geomI];
327  if (!neighsI.empty())
328  {
329  _geometries->_sdf.geometries[geomI].numNeighbours = neighsI.size();
330  _geometries->_sdf.geometries[geomI].neighboursIndex = currOffset;
331  _geometries->_sdf.neighboursFlat.insert(std::end(_geometries->_sdf.neighboursFlat), std::begin(neighsI),
332  std::end(neighsI));
333  }
334  }
335 
336  // Make sure we don't create an empty buffer in the case of no neighbours
337  if (_geometries->_sdf.neighboursFlat.empty())
338  _geometries->_sdf.neighboursFlat.resize(1, 0);
339 
340  auto neighbourData = allocateVectorData(_geometries->_sdf.neighboursFlat, OSP_ULONG, _memoryManagementFlags);
341 
342  for (const auto& mat : _materials)
343  {
344  const size_t materialId = mat.first;
345 
346  if (_geometries->_sdf.geometryIndices.find(materialId) == _geometries->_sdf.geometryIndices.end())
347  continue;
348 
349  auto& geometry = _createGeometry(_ospSDFGeometries, materialId, OSPRAY_GEOMETRY_PROPERTY_SDF);
350 
351  auto data =
352  allocateVectorData(_geometries->_sdf.geometryIndices[materialId], OSP_ULONG, _memoryManagementFlags);
353  ospSetObject(geometry, OSPRAY_GEOMETRY_PROPERTY_SDF, data);
354  ospRelease(data);
355 
356  ospSetData(geometry, OSPRAY_GEOMETRY_PROPERTY_SDF_NEIGHBOURS, neighbourData);
357  ospSetData(geometry, OSPRAY_GEOMETRY_PROPERTY_SDF_GEOMETRIES, globalData);
358  osphelper::set(geometry, OSPRAY_GEOMETRY_PROPERTY_SDF_EPSILON, _geometryParameters.getSdfEpsilon());
359  osphelper::set(geometry, OSPRAY_GEOMETRY_PROPERTY_SDF_NB_MARCH_ITERATIONS,
360  static_cast<int>(_geometryParameters.getSdfNbMarchIterations()));
361  osphelper::set(geometry, OSPRAY_GEOMETRY_PROPERTY_SDF_BLEND_FACTOR, _geometryParameters.getSdfBlendFactor());
362  osphelper::set(geometry, OSPRAY_GEOMETRY_PROPERTY_SDF_BLEND_LERP_FACTOR,
363  _geometryParameters.getSdfBlendLerpFactor());
364  osphelper::set(geometry, OSPRAY_GEOMETRY_PROPERTY_SDF_OMEGA, _geometryParameters.getSdfOmega());
365  osphelper::set(geometry, OSPRAY_GEOMETRY_PROPERTY_SDF_DISTANCE, _geometryParameters.getSdfDistance());
366 
367  ospCommit(geometry);
368 
369  ospAddGeometry(_primaryModel, geometry);
370  }
371 
372  ospRelease(globalData);
373  ospRelease(neighbourData);
374 }

423 void OSPRayModel::commitGeometry()
424 {
425  for (auto volume : _geometries->_volumes)
426  {
427  auto ospVolume = std::dynamic_pointer_cast<OSPRayVolume>(volume.second);
428  ospVolume->commit();
429  }
430 
431  if (!isDirty())
432  return;
433 
434  if (!_primaryModel)
435  _primaryModel = ospNewModel();
436 
437  // Materials
438  for (auto material : _materials)
439  material.second->commit();
440 
441  // Group geometry
442  if (_spheresDirty)
443  for (const auto& spheres : _geometries->_spheres)
444  _commitSpheres(spheres.first);
445 
446  if (_cylindersDirty)
447  for (const auto& cylinders : _geometries->_cylinders)
448  _commitCylinders(cylinders.first);
449 
450  if (_conesDirty)
451  for (const auto& cones : _geometries->_cones)
452  _commitCones(cones.first);
453 
454  if (_sdfBeziersDirty)
455  for (const auto& sdfBeziers : _geometries->_sdfBeziers)
456  _commitSDFBeziers(sdfBeziers.first);
457 
458  if (_triangleMeshesDirty)
459  for (const auto& meshes : _geometries->_triangleMeshes)
460  _commitMeshes(meshes.first);
461 
462  if (_streamlinesDirty)
463  for (const auto& streamlines : _geometries->_streamlines)
464  _commitStreamlines(streamlines.first);
465 
466  if (_sdfGeometriesDirty)
467  _commitSDFGeometries();
468 
469  if (_curvesDirty)
470  for (const auto& curve : _geometries->_curves)
471  _commitCurves(curve.first);
472 
473  updateBounds();
474  _markGeometriesClean();
475  _setBVHFlags();
476 
477  // handled by the scene
478  _instancesDirty = false;
479 
480  // Commit models
481  ospCommit(_primaryModel);
482  if (_secondaryModel)
483  ospCommit(_secondaryModel);
484  if (_boundingBoxModel)
485  ospCommit(_boundingBoxModel);
486 }

488 void OSPRayModel::commitMaterials(const std::string& renderer)
489 {
490  if (renderer.empty())
491  throw std::runtime_error("Materials cannot be instanced with an empty renderer name");
492  if (_renderer != renderer)
493  {
494  for (auto kv : _materials)
495  {
496  auto& material = *kv.second;
497  static_cast<OSPRayMaterial&>(material).commit(renderer);
498  }
499 
500  _renderer = renderer;
501 
502  for (auto& map : {_ospSpheres, _ospCylinders, _ospCones, _ospMeshes, _ospStreamlines, _ospSDFGeometries})
503  {
504  auto matIt = _materials.begin();
505  auto geomIt = map.begin();
506  while (matIt != _materials.end() && geomIt != map.end())
507  {
508  while (matIt->first < geomIt->first && matIt != _materials.end())
509  ++matIt;
510  if (matIt->first != geomIt->first)
511  {
512  CORE_ERROR("Material for geometry missing");
513  ++geomIt;
514  continue;
515  }
516  auto& material = static_cast<OSPRayMaterial&>(*matIt->second);
517  ospSetMaterial(geomIt->second, material.getOSPMaterial());
518  ospCommit(geomIt->second);
519  ++geomIt;
520  }
521  }
522  }
523  else
524  {
525  for (auto kv : _materials)
526  {
527  auto& material = *kv.second;
528  static_cast<OSPRayMaterial&>(material).commit();
529  }
530  }
531 }
