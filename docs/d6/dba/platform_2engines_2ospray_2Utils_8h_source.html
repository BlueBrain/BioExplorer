52 template <size_t S>
53 std::vector<std::array<float, S>> convertVectorToFloat(const std::vector<std::array<double, S>>& input)
54 {
55  std::vector<std::array<float, S>> output;
56  output.reserve(input.size());
57  for (const auto& value : input)
58  {
59  std::array<float, S> converted;
60  std::copy(value.data(), value.data() + S, converted.data());
61  output.push_back(converted);
62  }
63  return output;
64 }
