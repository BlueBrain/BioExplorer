49 void PerspectiveStereoCamera::commit()
50 {
51  Camera::commit();
52 
53  fieldOfView = getParamf(CAMERA_PROPERTY_FIELD_OF_VIEW.name.c_str(), DEFAULT_CAMERA_FIELD_OF_VIEW);
54  aspect = getParamf(CAMERA_PROPERTY_ASPECT_RATIO.name.c_str(), DEFAULT_CAMERA_ASPECT_RATIO);
55  apertureRadius = getParamf(CAMERA_PROPERTY_APERTURE_RADIUS.name.c_str(), DEFAULT_CAMERA_APERTURE_RADIUS);
56  focalDistance = getParamf(CAMERA_PROPERTY_FOCAL_DISTANCE.name.c_str(), DEFAULT_CAMERA_FOCAL_DISTANCE);
57  nearClip = getParamf(CAMERA_PROPERTY_NEAR_CLIP.name.c_str(), DEFAULT_CAMERA_NEAR_CLIP);
58  stereoMode = getParam(CAMERA_PROPERTY_STEREO.name.c_str(), static_cast<int>(DEFAULT_CAMERA_STEREO))
59  ? CameraStereoMode::side_by_side
60  : CameraStereoMode::mono;
61  interpupillaryDistance =
62  getParamf(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str(), DEFAULT_CAMERA_INTERPUPILLARY_DISTANCE);
63  enableClippingPlanes =
64  getParam(CAMERA_PROPERTY_ENABLE_CLIPPING_PLANES.name.c_str(), DEFAULT_CAMERA_ENABLE_CLIPPING_PLANES);
65  clipPlanes = enableClippingPlanes ? getParamData(CAMERA_PROPERTY_CLIPPING_PLANES, nullptr) : nullptr;
66  useHardwareRandomizer = getParam(COMMON_PROPERTY_USE_HARDWARE_RANDOMIZER.name.c_str(),
67  static_cast<int>(DEFAULT_COMMON_USE_HARDWARE_RANDOMIZER));
68 
69  dir = normalize(dir);
70  ::ospray::vec3f dir_du = normalize(cross(dir, up));
71  ::ospray::vec3f dir_dv = cross(dir_du, dir);
72 
73  ::ospray::vec3f org = pos;
74  const ::ospray::vec3f ipd_offset = 0.5f * interpupillaryDistance * dir_du;
75 
76  switch (stereoMode)
77  {
78  case CameraStereoMode::left:
79  org -= ipd_offset;
80  break;
81  case CameraStereoMode::right:
82  org += ipd_offset;
83  break;
84  case CameraStereoMode::side_by_side:
85  aspect *= 0.5f;
86  break;
87  case CameraStereoMode::mono:
88  break;
89  }
90 
91  double imgPlane_size_y = 2.f * tanf(::ospray::deg2rad(0.5f * fieldOfView));
92  double imgPlane_size_x = imgPlane_size_y * aspect;
93 
94  dir_du *= imgPlane_size_x;
95  dir_dv *= imgPlane_size_y;
96 
97  ::ospray::vec3f dir_00 = dir - 0.5f * dir_du - 0.5f * dir_dv;
98 
99  double scaledAperture = 0.f;
100  if (apertureRadius > 0.f)
101  {
102  dir_du *= focalDistance;
103  dir_dv *= focalDistance;
104  dir_00 *= focalDistance;
105  scaledAperture = apertureRadius / imgPlane_size_x;
106  }
107 
108  const auto clipPlaneData = clipPlanes ? clipPlanes->data : nullptr;
109  const size_t numClipPlanes = clipPlanes ? clipPlanes->numItems : 0;
110 
111  ::ispc::PerspectiveStereoCamera_set(getIE(), (const ::ispc::vec3f&)org, (const ::ispc::vec3f&)dir_00,
112  (const ::ispc::vec3f&)dir_du, (const ::ispc::vec3f&)dir_dv, scaledAperture,
113  aspect, (const ::ispc::vec3f&)ipd_offset, stereoMode,
114  (const ::ispc::vec4f*)clipPlaneData, numClipPlanes, nearClip,
115  useHardwareRandomizer);
116 }
