42 void PerspectiveStereoCamera::commit()
43 {
44  Camera::commit();
45 
46  fieldOfView = getParamf(CAMERA_PROPERTY_FIELD_OF_VIEW.name.c_str(), DEFAULT_CAMERA_FIELD_OF_VIEW);
47  aspect = getParamf(CAMERA_PROPERTY_ASPECT_RATIO.name.c_str(), DEFAULT_CAMERA_ASPECT_RATIO);
48  apertureRadius = getParamf(CAMERA_PROPERTY_APERTURE_RADIUS.name.c_str(), DEFAULT_CAMERA_APERTURE_RADIUS);
49  focalDistance = getParamf(CAMERA_PROPERTY_FOCAL_DISTANCE.name.c_str(), DEFAULT_CAMERA_FOCAL_DISTANCE);
50  nearClip = getParamf(CAMERA_PROPERTY_NEAR_CLIP.name.c_str(), DEFAULT_CAMERA_NEAR_CLIP);
51  stereoMode = getParam(CAMERA_PROPERTY_STEREO.name.c_str(), static_cast<int>(DEFAULT_CAMERA_STEREO))
52  ? CameraStereoMode::side_by_side
53  : CameraStereoMode::mono;
54  interpupillaryDistance =
55  getParamf(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str(), DEFAULT_CAMERA_INTERPUPILLARY_DISTANCE);
56  enableClippingPlanes =
57  getParam(CAMERA_PROPERTY_ENABLE_CLIPPING_PLANES.name.c_str(), DEFAULT_CAMERA_ENABLE_CLIPPING_PLANES);
58  clipPlanes = enableClippingPlanes ? getParamData(CAMERA_PROPERTY_CLIPPING_PLANES, nullptr) : nullptr;
59  useHardwareRandomizer = getParam(COMMON_PROPERTY_USE_HARDWARE_RANDOMIZER.name.c_str(),
60  static_cast<int>(DEFAULT_COMMON_USE_HARDWARE_RANDOMIZER));
61 
62  dir = normalize(dir);
63  ::ospray::vec3f dir_du = normalize(cross(dir, up));
64  ::ospray::vec3f dir_dv = cross(dir_du, dir);
65 
66  ::ospray::vec3f org = pos;
67  const ::ospray::vec3f ipd_offset = 0.5f * interpupillaryDistance * dir_du;
68 
69  switch (stereoMode)
70  {
71  case CameraStereoMode::left:
72  org -= ipd_offset;
73  break;
74  case CameraStereoMode::right:
75  org += ipd_offset;
76  break;
77  case CameraStereoMode::side_by_side:
78  aspect *= 0.5f;
79  break;
80  case CameraStereoMode::mono:
81  break;
82  }
83 
84  double imgPlane_size_y = 2.f * tanf(::ospray::deg2rad(0.5f * fieldOfView));
85  double imgPlane_size_x = imgPlane_size_y * aspect;
86 
87  dir_du *= imgPlane_size_x;
88  dir_dv *= imgPlane_size_y;
89 
90  ::ospray::vec3f dir_00 = dir - 0.5f * dir_du - 0.5f * dir_dv;
91 
92  double scaledAperture = 0.f;
93  if (apertureRadius > 0.f)
94  {
95  dir_du *= focalDistance;
96  dir_dv *= focalDistance;
97  dir_00 *= focalDistance;
98  scaledAperture = apertureRadius / imgPlane_size_x;
99  }
100 
101  const auto clipPlaneData = clipPlanes ? clipPlanes->data : nullptr;
102  const size_t numClipPlanes = clipPlanes ? clipPlanes->numItems : 0;
103 
104  ::ispc::PerspectiveStereoCamera_set(getIE(), (const ::ispc::vec3f&)org, (const ::ispc::vec3f&)dir_00,
105  (const ::ispc::vec3f&)dir_du, (const ::ispc::vec3f&)dir_dv, scaledAperture,
106  aspect, (const ::ispc::vec3f&)ipd_offset, stereoMode,
107  (const ::ispc::vec4f*)clipPlaneData, numClipPlanes, nearClip,
108  useHardwareRandomizer);
109 }
