43 void PerspectiveStereoCamera::commit()
44 {
45  Camera::commit();
46 
47  fovy = getParamf("fovy", 60.f);
48  aspect = getParamf("aspect", 1.f);
49  apertureRadius = getParamf("apertureRadius", 0.f);
50  focusDistance = getParamf("focusDistance", 1.f);
51  nearClip = getParamf("nearClip", 0.f);
52  stereoMode = getParam("stereo", 0) ? CameraStereoMode::side_by_side : CameraStereoMode::mono;
53  interpupillaryDistance = getParamf("interpupillaryDistance", 0.0635f);
54  enableClippingPlanes = getParam("enableClippingPlanes", 0);
55  clipPlanes = enableClippingPlanes ? getParamData("clipPlanes", nullptr) : nullptr;
56 
57  dir = normalize(dir);
58  vec3f dir_du = normalize(cross(dir, up));
59  vec3f dir_dv = cross(dir_du, dir);
60 
61  vec3f org = pos;
62  const vec3f ipd_offset = 0.5f * interpupillaryDistance * dir_du;
63 
64  switch (stereoMode)
65  {
66  case CameraStereoMode::left:
67  org -= ipd_offset;
68  break;
69  case CameraStereoMode::right:
70  org += ipd_offset;
71  break;
72  case CameraStereoMode::side_by_side:
73  aspect *= 0.5f;
74  break;
75  case CameraStereoMode::mono:
76  break;
77  }
78 
79  double imgPlane_size_y = 2.f * tanf(deg2rad(0.5f * fovy));
80  double imgPlane_size_x = imgPlane_size_y * aspect;
81 
82  dir_du *= imgPlane_size_x;
83  dir_dv *= imgPlane_size_y;
84 
85  vec3f dir_00 = dir - 0.5f * dir_du - 0.5f * dir_dv;
86 
87  double scaledAperture = 0.f;
88  if (apertureRadius > 0.f)
89  {
90  dir_du *= focusDistance;
91  dir_dv *= focusDistance;
92  dir_00 *= focusDistance;
93  scaledAperture = apertureRadius / imgPlane_size_x;
94  }
95 
96  const auto clipPlaneData = clipPlanes ? clipPlanes->data : nullptr;
97  const size_t numClipPlanes = clipPlanes ? clipPlanes->numItems : 0;
98 
99  ispc::PerspectiveStereoCamera_set(getIE(), (const ispc::vec3f&)org, (const ispc::vec3f&)dir_00,
100  (const ispc::vec3f&)dir_du, (const ispc::vec3f&)dir_dv, scaledAperture, aspect,
101  (const ispc::vec3f&)ipd_offset, stereoMode, (const ispc::vec4f*)clipPlaneData,
102  numClipPlanes, nearClip);
103 }
