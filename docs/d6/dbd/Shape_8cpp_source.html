90 Quaterniond Shape::weightedRandomRotation(const Quaterniond& q,
91  const uint64_t seed,
92  const uint64_t index,
93  const double weight)
94 {
95  const Quaterniond qPitch =
96  angleAxis(weight * rnd2(seed + index * 2), Vector3d(1.0, 0.0, 0.0));
97  const Quaterniond qYaw =
98  angleAxis(weight * rnd2(seed + index * 3), Vector3d(0.0, 1.0, 0.0));
99  const Quaterniond qRoll =
100  angleAxis(weight * rnd2(seed + index * 5), Vector3d(0.0, 0.0, 1.0));
101  return q * qPitch * qYaw * qRoll;
102 }

104 Quaterniond Shape::randomQuaternion(const uint64_t seed) const
105 {
106  double x, y, z, u, v, w, s;
107  do
108  {
109  x = rnd2(seed);
110  y = rnd2(seed + 1);
111  z = x * x + y * y;
112  } while (z > 1.0);
113  do
114  {
115  u = rnd2(seed + 2);
116  v = rnd2(seed + 3);
117  w = u * u + v * v;
118  } while (w > 1.0);
119  s = sqrt((1.0 - z) / w);
120  return Quaterniond(x, y, s * u, s * v);
121 }
