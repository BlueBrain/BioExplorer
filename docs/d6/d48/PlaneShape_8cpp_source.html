42 Transformation PlaneShape::getTransformation(
43  const uint64_t occurrence, const uint64_t nbOccurrences,
44  const MolecularSystemAnimationDetails& MolecularSystemAnimationDetails,
45  const double offset) const
46 {
47  double up = 0.f;
48  if (MolecularSystemAnimationDetails.seed != 0)
49  up = rnd1() * MolecularSystemAnimationDetails.positionStrength;
50 
51  Vector3d pos{rnd1() * _size.x, up, rnd1() * _size.y};
52  const Quaterniond rot{0.0, 0.0, 0.707, 0.707};
53 
54  pos += offset * UP_VECTOR;
55 
56  if (isClipped(pos, _clippingPlanes))
57  throw std::runtime_error("Instance is clipped");
58 
59  Transformation transformation;
60  transformation.setTranslation(pos);
61  transformation.setRotation(rot);
62  return transformation;
63 }
