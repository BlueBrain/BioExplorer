31 strings parseFolder(const std::string& folder, const strings& filters)
32 {
33  strings files;
34  fs::directory_iterator endIter;
35  if (fs::is_directory(folder))
36  {
37  for (fs::directory_iterator dirIter(folder); dirIter != endIter; ++dirIter)
38  {
39  if (fs::is_regular_file(dirIter->status()))
40  {
41  const auto filename = dirIter->path().c_str();
42  if (filters.empty())
43  files.push_back(filename);
44  else
45  {
46  const auto& fileExtension = dirIter->path().extension();
47  const auto found = std::find(filters.begin(), filters.end(), fileExtension);
48  if (found != filters.end())
49  files.push_back(filename);
50  }
51  }
52  }
53  }
54  std::sort(files.begin(), files.end());
55  return files;
56 }

67 Vector4f getBezierPoint(const Vector4fs& controlPoints, const float t)
68 {
69  if (t < 0.f || t > 1.f)
70  CORE_THROW("Invalid value with t=" + std::to_string(t) + ". Must be between 0 and 1");
71 
72  const size_t n = controlPoints.size();
73  Vector4fs tempPoints = controlPoints;
74 
75  for (uint64_t k = 1; k < n; ++k)
76  {
77  for (uint64_t i = 0; i < n - k; ++i)
78  {
79  tempPoints[i].x = (1 - t) * tempPoints[i].x + t * tempPoints[i + 1].x;
80  tempPoints[i].y = (1 - t) * tempPoints[i].y + t * tempPoints[i + 1].y;
81  tempPoints[i].z = (1 - t) * tempPoints[i].z + t * tempPoints[i + 1].z;
82  tempPoints[i].w = (1 - t) * tempPoints[i].w + t * tempPoints[i + 1].w;
83  }
84  }
85  return tempPoints[0];
86 }

93 Vector3f hsvToRgb(float h, float s, float v)
94 {
95  int i = h * 6;
96  float f = h * 6 - i;
97  float p = v * (1 - s);
98  float q = v * (1 - f * s);
99  float t = v * (1 - (1 - f) * s);
100 
101  switch (i % 6)
102  {
103  case 0:
104  return {v, t, p};
105  case 1:
106  return {q, v, p};
107  case 2:
108  return {p, v, t};
109  case 3:
110  return {p, q, v};
111  case 4:
112  return {t, p, v};
113  case 5:
114  return {v, p, q};
115  default:
116  return {0, 0, 0}; // Should not reach here
117  }
118 }
