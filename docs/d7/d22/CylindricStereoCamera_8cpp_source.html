46 void CylindricStereoCamera::commit()
47 {
48  Camera::commit();
49 
50  const auto stereoMode = getStereoMode();
51  const auto ipd = getInterpupillaryDistance(stereoMode);
52  const auto sideBySide = stereoMode == StereoMode::OSP_STEREO_SIDE_BY_SIDE;
53 
54  dir = normalize(dir);
55  // The tracking model of the 3d glasses is inversed
56  // so we need to negate dir_du here.
57  const auto dir_du = -normalize(cross(dir, up));
58  const auto dir_dv = normalize(up);
59  dir = -dir;
60 
61  const auto imgPlane_size_y = 2.0f * tanf(deg2rad(0.5f * OPENDECK_FOV_Y));
62 
63  ::ispc::CylindricStereoCamera_set(getIE(), (const ::ispc::vec3f&)pos, (const ::ispc::vec3f&)dir,
64  (const ::ispc::vec3f&)dir_du, (const ::ispc::vec3f&)dir_dv, imgPlane_size_y, ipd,
65  sideBySide);
66 }

73 float CylindricStereoCamera::getInterpupillaryDistance(const StereoMode stereoMode)
74 {
75  const auto interpupillaryDistance =
76  getParamf(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str(), DEFAULT_CAMERA_INTERPUPILLARY_DISTANCE);
77 
78  switch (stereoMode)
79  {
80  case StereoMode::OSP_STEREO_LEFT:
81  return -interpupillaryDistance;
82  case StereoMode::OSP_STEREO_RIGHT:
83  return interpupillaryDistance;
84  case StereoMode::OSP_STEREO_SIDE_BY_SIDE:
85  return interpupillaryDistance;
86  case StereoMode::OSP_STEREO_NONE:
87  default:
88  return 0.0f;
89  }
