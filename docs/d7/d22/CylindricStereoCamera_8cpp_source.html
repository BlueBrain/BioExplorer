43 void CylindricStereoCamera::commit()
44 {
45  Camera::commit();
46 
47  const auto stereoMode = getStereoMode();
48  const auto ipd = getInterpupillaryDistance(stereoMode);
49  const auto sideBySide = stereoMode == StereoMode::OSP_STEREO_SIDE_BY_SIDE;
50 
51  dir = normalize(dir);
52  // The tracking model of the 3d glasses is inversed
53  // so we need to negate dir_du here.
54  const auto dir_du = -normalize(cross(dir, up));
55  const auto dir_dv = normalize(up);
56  dir = -dir;
57 
58  const auto imgPlane_size_y = 2.0f * tanf(deg2rad(0.5f * OPENDECK_FOV_Y));
59 
60  ispc::CylindricStereoCamera_set(getIE(), (const ispc::vec3f&)pos, (const ispc::vec3f&)dir,
61  (const ispc::vec3f&)dir_du, (const ispc::vec3f&)dir_dv, imgPlane_size_y, ipd,
62  sideBySide);
63 }

70 float CylindricStereoCamera::getInterpupillaryDistance(const StereoMode stereoMode)
71 {
72  const auto interpupillaryDistance = getParamf("interpupillaryDistance", DEFAULT_INTERPUPILLARY_DISTANCE);
73 
74  switch (stereoMode)
75  {
76  case StereoMode::OSP_STEREO_LEFT:
77  return -interpupillaryDistance;
78  case StereoMode::OSP_STEREO_RIGHT:
79  return interpupillaryDistance;
80  case StereoMode::OSP_STEREO_SIDE_BY_SIDE:
81  return interpupillaryDistance;
82  case StereoMode::OSP_STEREO_NONE:
83  default:
84  return 0.0f;
85  }
