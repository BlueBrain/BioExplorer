41 void CylindricStereoCamera::commit()
42 {
43  Camera::commit();
44 
45  const auto stereoMode = getStereoMode();
46  const auto ipd = getInterpupillaryDistance(stereoMode);
47  const auto sideBySide = stereoMode == StereoMode::OSP_STEREO_SIDE_BY_SIDE;
48 
49  dir = normalize(dir);
50  // The tracking model of the 3d glasses is inversed
51  // so we need to negate dir_du here.
52  const auto dir_du = -normalize(cross(dir, up));
53  const auto dir_dv = normalize(up);
54  dir = -dir;
55 
56  const auto imgPlane_size_y = 2.0f * tanf(deg2rad(0.5f * OPENDECK_FOV_Y));
57 
58  ::ispc::CylindricStereoCamera_set(getIE(), (const ::ispc::vec3f&)pos, (const ::ispc::vec3f&)dir,
59  (const ::ispc::vec3f&)dir_du, (const ::ispc::vec3f&)dir_dv, imgPlane_size_y, ipd,
60  sideBySide);
61 }

68 float CylindricStereoCamera::getInterpupillaryDistance(const StereoMode stereoMode)
69 {
70  const auto interpupillaryDistance =
71  getParamf(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str(), DEFAULT_CAMERA_INTERPUPILLARY_DISTANCE);
72 
73  switch (stereoMode)
74  {
75  case StereoMode::OSP_STEREO_LEFT:
76  return -interpupillaryDistance;
77  case StereoMode::OSP_STEREO_RIGHT:
78  return interpupillaryDistance;
79  case StereoMode::OSP_STEREO_SIDE_BY_SIDE:
80  return interpupillaryDistance;
81  case StereoMode::OSP_STEREO_NONE:
82  default:
83  return 0.0f;
84  }
