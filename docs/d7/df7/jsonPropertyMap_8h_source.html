117 // Create JSON schema for given property and add it to the given properties
118 // parent.
119 template <typename T>
120 void _addPropertySchema(const Property& prop, rapidjson::Value& properties,
121  rapidjson::Document::AllocatorType& allocator)
122 {
123  using namespace rapidjson;
124 
125  Document jsonSchema;
126  if (prop.enums.empty())
127  {
128  auto value = prop.get<T>();
129  jsonSchema = staticjson::export_json_schema(&value, &allocator);
130  jsonSchema.AddMember(StringRef("title"), StringRef(prop.metaData.label.c_str()), allocator);
131  addDefaultValue(jsonSchema, allocator, value);
132  jsonSchema.AddMember(StringRef("readOnly"), prop.readOnly(), allocator);
133  const auto minValue = prop.min<T>();
134  const auto maxValue = prop.max<T>();
135  const bool hasLimits = maxValue - minValue != 0;
136  if (hasLimits)
137  {
138  if (jsonSchema.HasMember("minimum"))
139  jsonSchema["minimum"] = minValue;
140  else
141  jsonSchema.AddMember(StringRef("minimum"), minValue, allocator);
142 
143  if (jsonSchema.HasMember("maximum"))
144  jsonSchema["maximum"] = maxValue;
145  else
146  jsonSchema.AddMember(StringRef("maximum"), maxValue, allocator);
147  }
148  else
149  {
150  if (jsonSchema.HasMember("minimum"))
151  jsonSchema.RemoveMember("minimum");
152  if (jsonSchema.HasMember("maximum"))
153  jsonSchema.RemoveMember("maximum");
154  }
155  }
156  else
157  {
158  jsonSchema.SetObject();
159  jsonSchema.AddMember(StringRef("type"), StringRef("string"), allocator);
160  jsonSchema.AddMember(StringRef("title"), StringRef(prop.metaData.label.c_str()), allocator);
161  jsonSchema.AddMember(StringRef("readOnly"), prop.readOnly(), allocator);
162 
163  // Specialized enum code
164  auto value = prop.get<int32_t>();
165  const auto valueStr = matchEnum(prop.enums, value);
166  addDefaultValue(jsonSchema, allocator, valueStr);
167 
168  Value enumerations(kArrayType);
169  for (const auto& name : prop.enums)
170  enumerations.PushBack(StringRef(name.data(), name.size()), allocator);
171  jsonSchema.AddMember(StringRef("enum"), enumerations, allocator);
172  }
173  properties.AddMember(make_json_string(prop.name, allocator).Move(), jsonSchema, allocator);
174 }

251 // Create JSON schema for a property map and add it to the given propSchema
252 // parent.
253 void _addPropertyMapSchema(const PropertyMap& propertyMap, const std::string& title,
254  rapidjson::Document::AllocatorType& allocator, rapidjson::Value& propSchema)
255 {
256  using namespace rapidjson;
257  propSchema.AddMember(StringRef("title"), StringRef(title.c_str()), allocator);
258  propSchema.AddMember(StringRef("type"), StringRef("object"), allocator);
259 
260  Value properties(kObjectType);
261  for (auto prop : propertyMap.getProperties())
262  {
263  switch (prop->type)
264  {
265  case Property::Type::Double:
266  _addPropertySchema<double>(*prop, properties, allocator);
267  break;
268  case Property::Type::Int:
269  _addPropertySchema<int32_t>(*prop, properties, allocator);
270  break;
271  case Property::Type::String:
272  _addPropertySchema<std::string>(*prop, properties, allocator);
273  break;
274  case Property::Type::Bool:
275  _addPropertySchema<bool>(*prop, properties, allocator);
276  break;
277  case Property::Type::Vec2d:
278  _addArrayPropertySchema<double, 2>(*prop, properties, allocator);
279  break;
280  case Property::Type::Vec2i:
281  _addArrayPropertySchema<int32_t, 2>(*prop, properties, allocator);
282  break;
283  case Property::Type::Vec3d:
284  _addArrayPropertySchema<double, 3>(*prop, properties, allocator);
285  break;
286  case Property::Type::Vec3i:
287  _addArrayPropertySchema<int32_t, 3>(*prop, properties, allocator);
288  break;
289  case Property::Type::Vec4d:
290  _addArrayPropertySchema<double, 4>(*prop, properties, allocator);
291  break;
292  }
293  }
294 
295  propSchema.AddMember(StringRef("properties"), properties, allocator);
296 }

470 template <>
471 inline std::string to_json(const core::PropertyMap& obj)
472 {
473  using namespace rapidjson;
474  using core::Property;
475  Document json(kObjectType);
476  auto& allocator = json.GetAllocator();
477 
478  for (auto prop : obj.getProperties())
479  {
480  switch (prop->type)
481  {
482  case Property::Type::Double:
483  json.AddMember(core::make_json_string(prop->name, allocator).Move(), prop->get<double>(), allocator);
484  break;
485  case Property::Type::Int:
486  if (prop->enums.empty())
487  {
488  json.AddMember(core::make_json_string(prop->name, allocator).Move(), prop->get<int32_t>(), allocator);
489  }
490  else
491  {
492  auto enumStr = matchEnum(prop->enums, prop->get<int32_t>());
493  json.AddMember(core::make_json_string(prop->name, allocator).Move(),
494  core::make_value_string(enumStr, allocator).Move(), allocator);
495  }
496  break;
497  case Property::Type::String:
498  json.AddMember(core::make_json_string(prop->name, allocator).Move(),
499  core::make_value_string(prop->get<std::string>(), allocator).Move(), allocator);
500  break;
501  case Property::Type::Bool:
502  json.AddMember(core::make_json_string(prop->name, allocator).Move(), prop->get<bool>(), allocator);
503  break;
504  case Property::Type::Vec2d:
505  core::_arrayPropertyToJson<std::array<double, 2>>(json, *prop);
506  break;
507  case Property::Type::Vec2i:
508  core::_arrayPropertyToJson<std::array<int32_t, 2>>(json, *prop);
509  break;
510  case Property::Type::Vec3d:
511  core::_arrayPropertyToJson<std::array<double, 3>>(json, *prop);
512  break;
513  case Property::Type::Vec3i:
514  core::_arrayPropertyToJson<std::array<int32_t, 3>>(json, *prop);
515  break;
516  case Property::Type::Vec4d:
517  core::_arrayPropertyToJson<std::array<double, 4>>(json, *prop);
518  break;
519  }
520  }
521 
522  StringBuffer buffer;
523  Writer<StringBuffer> writer(buffer);
524  json.Accept(writer);
525  return buffer.GetString();

580 template <typename T, size_t S>
581 bool get_array(const rapidjson::Value& v, std::array<T, S>& val)
582 {
583  if (!(v.IsArray() && v.Size() == S))
584  return false;
585 
586  int j = 0;
587  for (const auto& i : v.GetArray())
588  {
589  if (!isValue<T>(i))
590  return false;
591  val[j++] = getValue<T>(i);
592  }
593  return true;
594 }

601 
602  return true;
603 }

612 core::PropertyMap jsonToPropertyMap(const std::string& json)
613 {
614  using namespace rapidjson;
615  using core::Property;
616  Document document;
617  document.Parse(json.c_str());
618 
619  core::PropertyMap map;
620 
621  if (!document.IsObject())
622  return map;
623 
624  for (const auto& m : document.GetObject())
625  {
626  const auto propName = core::snakeCaseToCamelCase(m.name.GetString());
627 
628  const auto trySetProperty = [&](auto val)
629  {
630  if (get_property(m.value, val))
631  {
632  map.setProperty({propName, val, {propName}});
633  return true;
634  }
635  return false;
636  };
637 
638  const auto trySetArray = [&](auto val)
639  {
640  if (get_array(m.value, val))
641  {
642  map.setProperty({propName, val, {propName}});
643  return true;
644  }
645  return false;
646  };
647 
648  if (trySetProperty(double(0)))
649  continue;
650  if (trySetProperty(int32_t(0)))
651  continue;
652  if (trySetProperty(std::string("")))
653  continue;
654  if (trySetProperty(bool(false)))
655  continue;
656  if (trySetArray(std::array<double, 2>{{0, 0}}))
657  continue;
658  if (trySetArray(std::array<int32_t, 2>{{0, 0}}))
659  continue;
660  if (trySetArray(std::array<double, 3>{{0, 0, 0}}))
661  continue;
662  if (trySetArray(std::array<int32_t, 3>{{0, 0, 0}}))
663  continue;
664  if (trySetArray(std::array<double, 4>{{0, 0, 0, 0}}))
665  continue;
666  }
667 
668  return map;
669 }
