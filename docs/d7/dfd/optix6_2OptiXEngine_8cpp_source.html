85 void OptiXEngine::_createCameras()
86 {
87  _camera = createCamera();
88 
89  const bool isStereo = _parametersManager.getApplicationParameters().isStereo();
90  Property stereoProperty{CAMERA_PROPERTY_STEREO.name, isStereo, CAMERA_PROPERTY_STEREO.metaData};
91  Property aspect = CAMERA_PROPERTY_ASPECT_RATIO;
92  aspect.markReadOnly();
93 
94  OptiXContext& context = OptiXContext::get();
95 
96  {
97  PLUGIN_INFO("Registering '" << CAMERA_PROPERTY_TYPE_PERSPECTIVE << "' camera");
98  PropertyMap properties;
99  properties.setProperty(COMMON_PROPERTY_EXPOSURE);
100  properties.setProperty(CAMERA_PROPERTY_FIELD_OF_VIEW);
101  properties.setProperty(aspect);
102  properties.setProperty(CAMERA_PROPERTY_APERTURE_RADIUS);
103  properties.setProperty(CAMERA_PROPERTY_FOCAL_DISTANCE);
104  properties.setProperty(CAMERA_PROPERTY_NEAR_CLIP);
105  properties.setProperty(CAMERA_PROPERTY_ENABLE_CLIPPING_PLANES);
106  properties.setProperty(stereoProperty);
107  properties.setProperty(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE);
108 
109  auto camera = std::make_shared<OptiXPerspectiveCamera>();
110  context.addCamera(CAMERA_PROPERTY_TYPE_PERSPECTIVE, camera);
111  addCameraType(CAMERA_PROPERTY_TYPE_PERSPECTIVE, properties);
112  }
113 
114  {
115  PLUGIN_INFO("Registering '" << CAMERA_PROPERTY_TYPE_ORTHOGRAPHIC << "' camera");
116  PropertyMap properties;
117  properties.setProperty(COMMON_PROPERTY_EXPOSURE);
118  properties.setProperty(CAMERA_PROPERTY_HEIGHT);
119  properties.setProperty(aspect);
120  properties.setProperty(CAMERA_PROPERTY_ENABLE_CLIPPING_PLANES);
121 
122  auto camera = std::make_shared<OptiXOrthographicCamera>();
123  context.addCamera(CAMERA_PROPERTY_TYPE_ORTHOGRAPHIC, camera);

145 void OptiXEngine::_createRenderers()
146 {
147  _renderer = std::make_shared<OptiXRenderer>(_parametersManager.getAnimationParameters(),
148  _parametersManager.getRenderingParameters());
149  _renderer->setEngine(this);
150  OptiXContext& context = OptiXContext::get();
151 
152  { // Advanced renderer
153  PLUGIN_INFO("Registering '" << RENDERER_PROPERTY_TYPE_ADVANCED << "' renderer");
154  const std::string CUDA_ADVANCED_SIMULATION_RENDERER = OptiX6Engine_generated_Advanced_cu_ptx;
155 
156  auto renderer = std::make_shared<OptixShaderProgram>();
157  renderer->closest_hit =
158  context.getOptixContext()->createProgramFromPTXString(CUDA_ADVANCED_SIMULATION_RENDERER,
159  OPTIX_CUDA_FUNCTION_CLOSEST_HIT_RADIANCE);
160  renderer->closest_hit_textured =
161  context.getOptixContext()->createProgramFromPTXString(CUDA_ADVANCED_SIMULATION_RENDERER,
162  OPTIX_CUDA_FUNCTION_CLOSEST_HIT_RADIANCE_TEXTURED);
163  renderer->any_hit = context.getOptixContext()->createProgramFromPTXString(CUDA_ADVANCED_SIMULATION_RENDERER,
164  OPTIX_CUDA_FUNCTION_ANY_HIT_SHADOW);
165  renderer->exception_program =
166  context.getOptixContext()->createProgramFromPTXString(CUDA_ADVANCED_SIMULATION_RENDERER,
167  OPTIX_CUDA_FUNCTION_EXCEPTION);
168  context.getOptixContext()->setExceptionProgram(0, renderer->exception_program);
169  context.addRenderer(RENDERER_PROPERTY_TYPE_ADVANCED, renderer);
170 
171  PropertyMap properties;
172  properties.setProperty(RENDERER_PROPERTY_ALPHA_CORRECTION);
173  properties.setProperty(RENDERER_PROPERTY_MAX_DISTANCE_TO_SECONDARY_MODEL);
174  properties.setProperty(RENDERER_PROPERTY_GLOBAL_ILLUMINATION_RAY_LENGTH);
175  properties.setProperty(RENDERER_PROPERTY_GLOBAL_ILLUMINATION_STRENGTH);
176  properties.setProperty(RENDERER_PROPERTY_GLOBAL_ILLUMINATION_SAMPLES);
177  properties.setProperty(RENDERER_PROPERTY_SHADOW_INTENSITY);
178  properties.setProperty(RENDERER_PROPERTY_SOFT_SHADOW_STRENGTH);
179  properties.setProperty(RENDERER_PROPERTY_SHADOW_SAMPLES);
180  properties.setProperty(COMMON_PROPERTY_EXPOSURE);
181  properties.setProperty(RENDERER_PROPERTY_EPSILON_MULTIPLIER);
182  properties.setProperty(RENDERER_PROPERTY_FOG_START);
183  properties.setProperty(RENDERER_PROPERTY_FOG_THICKNESS);
184  properties.setProperty(RENDERER_PROPERTY_MAX_RAY_DEPTH);
185  properties.setProperty(RENDERER_PROPERTY_SHOW_BACKGROUND);
186  properties.setProperty(RENDERER_PROPERTY_MATRIX_FILTER);
187  addRendererType(RENDERER_PROPERTY_TYPE_ADVANCED, properties);
188  }
189 
190  { // Basic simulation / Basic renderer
191  PLUGIN_INFO("Registering '" << RENDERER_PROPERTY_TYPE_BASIC << "' renderer");
192  const std::string CUDA_BASIC_SIMULATION_RENDERER = OptiX6Engine_generated_Basic_cu_ptx;
193 
194  auto renderer = std::make_shared<OptixShaderProgram>();
195  renderer->closest_hit =
196  context.getOptixContext()->createProgramFromPTXString(CUDA_BASIC_SIMULATION_RENDERER,
197  OPTIX_CUDA_FUNCTION_CLOSEST_HIT_RADIANCE);
198  renderer->closest_hit_textured =
199  context.getOptixContext()->createProgramFromPTXString(CUDA_BASIC_SIMULATION_RENDERER,
200  OPTIX_CUDA_FUNCTION_CLOSEST_HIT_RADIANCE_TEXTURED);
201  renderer->any_hit = context.getOptixContext()->createProgramFromPTXString(CUDA_BASIC_SIMULATION_RENDERER,
202  OPTIX_CUDA_FUNCTION_ANY_HIT_SHADOW);
203  renderer->exception_program =
204  context.getOptixContext()->createProgramFromPTXString(CUDA_BASIC_SIMULATION_RENDERER,
205  OPTIX_CUDA_FUNCTION_EXCEPTION);
206 
207  context.addRenderer(RENDERER_PROPERTY_TYPE_BASIC, renderer);
208 
209  PropertyMap properties;
210  properties.setProperty(COMMON_PROPERTY_EXPOSURE);
211  properties.setProperty(RENDERER_PROPERTY_SHOW_BACKGROUND);
212  addRendererType(RENDERER_PROPERTY_TYPE_BASIC, properties);
213  }
