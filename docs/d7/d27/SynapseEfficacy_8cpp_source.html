52 void SynapseEfficacy::_buildModel()
53 {
54  if (_modelDescriptor)
55  _scene.removeModel(_modelDescriptor->getModelID());
56 
57  auto model = _scene.createModel();
58  ThreadSafeContainer container(*model);
59 
60  const auto synapsePositions =
61  DBConnector::getInstance().getSynapseEfficacyPositions(
62  _details.populationName, _details.sqlFilter);
63 
64  const auto nbSynapses = synapsePositions.size();
65  const size_t materialId = 0;
66  const bool useSdf = false;
67  uint64_t progressStep = synapsePositions.size() / 100 + 1;
68  uint64_t i = 0;
69  for (const auto& position : synapsePositions)
70  {
71  const auto src = getAlignmentToGrid(_details.alignToGrid, position);
72  container.addSphere(src, _details.radius, materialId, useSdf, i);
73  if (i % progressStep == 0)
74  PLUGIN_PROGRESS("Loading " << i << "/" << nbSynapses << " synapses",
75  i, nbSynapses);
76  ++i;
77  }
78 
79  container.commitToModel();
80  PLUGIN_INFO(1, "");
81 
82  const ModelMetadata metadata = {{"Number of synapses",
83  std::to_string(nbSynapses)},
84  {"SQL filter", _details.sqlFilter}};
85 
86  _modelDescriptor.reset(new brayns::ModelDescriptor(std::move(model),
87  _details.assemblyName,
88  metadata));
89  if (_modelDescriptor)
90  _scene.addModel(_modelDescriptor);
91  else
92  PLUGIN_THROW(
93  "Synapse efficacy model could not be created for "
94  "population " +
95  _details.populationName);
96 }
