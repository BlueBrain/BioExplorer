46 void Cones::finalize(::ospray::Model* model)
47 {
48  data = getParamData(OSPRAY_GEOMETRY_PROPERTY_CONES, nullptr);
49  constexpr size_t bytesPerCone = sizeof(Cone);
50 
51  if (data.ptr == nullptr || bytesPerCone == 0)
52  throw std::runtime_error("#ospray:geometry/cones: no 'cones' data specified");
53 
54  const size_t numCones = data->numBytes / bytesPerCone;
55 
56  bounds = ::ospray::empty;
57  const auto geoms = static_cast<Cone*>(data->data);
58  for (size_t i = 0; i < numCones; i++)
59  {
60  const core::Cone& geom = geoms[i];
61  const auto center = ::ospray::vec3f(geom.center[0], geom.center[1], geom.center[2]);
62  const auto up = ::ospray::vec3f(geom.up[0], geom.up[1], geom.up[2]);
63 
64  bounds.extend(center - geom.centerRadius);
65  bounds.extend(center + geom.centerRadius);
66  bounds.extend(up - geom.upRadius);
67  bounds.extend(up + geom.upRadius);
68  }
69 
70  ::ispc::ConesGeometry_set(getIE(), model->getIE(), data->data, numCones);
71 }
