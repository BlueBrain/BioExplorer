26 /**
27  Composes source and destination colors according to specified alpha
28  correction
29  @param src Source RGBA color
30  @param dst Returned destination RGBA color
31  @param alphaCorrection Alpha correction used to blend colors
32 */
33 inline void composite(const varying vec4f& src, varying vec4f& dst,
34  const varying float alphaCorrection)
35 {
36  const float alpha =
37  1.f - pow(1.f - min(src.w, 1.f - 1.f / 256.f), alphaCorrection);
38  const float a = alpha * (1.f - dst.w);
39  dst.x = dst.x + src.x * a;
40  dst.y = dst.y + src.y * a;
41  dst.z = dst.z + src.z * a;
42  dst.w += (alpha * (1.f - dst.w));
43 }

45 /**
46  Returns the refracted vector according to the direction of the incident ray,
47  he normal to the surface, and localRefraction indices
48  @param direction Direction of the incident ray
49  @param normal Normal to the surface
50  @param n1 In localRefraction index
51  @param n2 Out localRefraction index
52  @return Refracted vector
53 */
54 inline vec3f refractedVector(const varying vec3f& direction,
55  const varying vec3f& normal,
56  const varying float n1, const varying float n2)
57 {
58  if (n2 == 0.f)
59  return direction;
60  const float eta = n1 / n2;
61  const float cos1 = -dot(direction, normal);
62  const float cos2 = 1.f - eta * eta * (1.f - cos1 * cos1);
63  if (cos2 > 0.f)
64  return eta * direction + (eta * cos1 - sqrt(cos2)) * normal;
65  return direction;
66 }

144 inline void setGeometryShadingAttributes(
145  const uniform AdvancedRenderer* uniform self, DifferentialGeometry& dg,
146  varying ScreenSample& sample, const varying Ray& ray,
147  ShadingAttributes& attributes)
148 {
149  attributes.dg = &dg;
150 
151  // Intersection point with geometry
152  attributes.origin = dg.P;
153 
154  // Shading normal
155  attributes.normal = dg.Ng;
156 
157  // Shading mode
158  attributes.shadingMode = undefined_shading_mode;
159 
160  // Reflected shading normal
161  attributes.reflectedNormal =
162  ray.dir - 2.f * dot(ray.dir, attributes.normal) * attributes.normal;
163 
164  // Chameleon
165  attributes.chameleonMode = undefined_chameleon_mode;
166 
167  // Model Id
168  attributes.nodeId = 0;
169 
170  // Process material attributes
171  const uniform Material* material = dg.material;
172  const uniform AdvancedMaterial* objMaterial =
173  (const uniform AdvancedMaterial*)material;
174  if (!objMaterial)
175  {
176  // Diffuse color
177  attributes.diffuseColor = make_vec3f(dg.color);
178 
179  // if no material is defined, localOpacity is stored in the
180  // alpha
181  // component of the RGBA color
182  attributes.opacity = dg.color.w;
183  return;
184  }
185 
186  foreach_unique(mat in objMaterial)
187  {
188  // Glossiness
189  if (mat->glossiness < 1.f)
190  {
191  attributes.glossiness = mat->glossiness;
192  const vec3f randomNormal =
193  (1.f - mat->glossiness) *
194  getRandomVector(self->useHardwareRandomizer,
195  self->super.fb->size.x, sample,
196  attributes.normal, self->randomNumber);
197  attributes.normal = normalize(attributes.normal + randomNormal);
198  }
199 
200  // Refraction index
201  attributes.refraction = mat->refraction;
202 
203  // Opacity
204  if (valid(mat->map_d))
205  attributes.opacity = mat->d * get1f(mat->map_d, dg);
206  else
207  attributes.opacity = mat->d;
208 
209  // Diffuse color
210  attributes.diffuseColor = mat->Kd * make_vec3f(dg.color);
211  if (valid(mat->map_Kd))
212  {
213  const vec3f diffuseColorFromMap = get3f(mat->map_Kd, dg);
214  attributes.diffuseColor =
215  attributes.diffuseColor * make_vec3f(diffuseColorFromMap);
216  // attributes.opacity *= diffuseColorFromMap.w;
217  }
218 
219  // Specular color
220  if (valid(mat->map_Ks))
221  attributes.specularColor =
222  max(attributes.specularColor, mat->Ks * get3f(mat->map_Ks, dg));
223  else
224  attributes.specularColor = max(attributes.specularColor, mat->Ks);
225 
226  // Specular exponent
227  if (valid(mat->map_Ns))
228  attributes.specularExponent = max(attributes.specularExponent,
229  mat->Ns * get1f(mat->map_Ns, dg));
230  else
231  attributes.specularExponent =
232  max(attributes.specularExponent, mat->Ns);
233 
234  // Light emission intensity
235  attributes.lightEmissionIntensity += mat->a;
236  attributes.lightEmissionColor = attributes.diffuseColor;
237 
238  // Normal mapping
239  if (valid(mat->map_Bump))
240  {
241  const vec3f n = get3f(mat->map_Bump, dg, make_vec3f(0.f)) - 0.5f;
242  attributes.normal.x -= n.x;
243  attributes.normal.y -= n.y;
244  }
245 
246  // Reflection index
247  attributes.reflection = mat->reflection;
248  if (valid(mat->map_reflection))
249  {
250  const vec3f value = get3f(mat->map_reflection, dg);
251  attributes.reflection *= (value.x + value.y + value.z) / 3.f;
252  }
253 
254  // Shading mode
255  attributes.shadingMode = mat->shadingMode;
256 
257  // User parameter
258  attributes.userParameter = mat->userParameter;
259 
260  // Chameleon mode
261  attributes.chameleonMode = mat->chameleonMode;
262 
263  // Cast user data
264  attributes.castUserData = mat->castUserData;
265 
266  // Model Id
267  attributes.nodeId = mat->nodeId;
268  }
269 
270  // Update reflected shading normal
271  attributes.reflectedNormal =
272  ray.dir - 2.f * dot(ray.dir, attributes.normal) * attributes.normal;
273 }

275 inline bool launchRandomRay(
276  const uniform AdvancedRenderer* uniform self, varying ScreenSample& sample,
277  const varying vec3f& intersection, const varying vec3f& normal,
278  varying vec3f& backgroundColor, varying float& distanceToIntersection,
279  varying vec3f& randomDirection, varying vec3f& color,
280  varying float& emission, const uint32 iteration, const float epsilon)
281 {
282  randomDirection =
283  getRandomVector(self->useHardwareRandomizer, self->super.fb->size.x,
284  sample, normal, iteration + self->randomNumber);
285  backgroundColor = make_vec3f(0.f);
286 
287  if (dot(randomDirection, normal) < 0.f)
288  // Invert direction of random ray direction is opposite to surface
289  // normal
290  randomDirection = neg(randomDirection);
291 
292  Ray randomRay;
293  randomRay.org = intersection;
294  randomRay.dir = randomDirection;
295  randomRay.t0 = epsilon;
296  randomRay.t = self->giDistance;
297  randomRay.time = inf;
298  randomRay.primID = -1;
299  randomRay.geomID = -1;
300  randomRay.instID = -1;
301 
302  bool moreRebounds = true;
303  bool hit = false;
304  while (moreRebounds)
305  {
306  traceRay(self->super.model, randomRay);
307 
308  if (randomRay.geomID < 0)
309  {
310  // No intersection
311  backgroundColor =
312  make_vec3f(skyboxMapping((Renderer*)self, randomRay,
313  (AdvancedMaterial*)self->bgMaterial));
314  return false;
315  }
316 
317  DifferentialGeometry dg;
318  postIntersect(self->super.model, dg, randomRay,
319  DG_NG | DG_NS | DG_NORMALIZE | DG_TANGENTS |
320  DG_FACEFORWARD | DG_MATERIALID | DG_COLOR |
321  DG_TEXCOORD);
322 
323  uniform AdvancedMaterial* objMaterial =
324  (uniform AdvancedMaterial*)dg.material;
325  if (objMaterial)
326  foreach_unique(mat in objMaterial)
327  {
328  if (mat->chameleonMode != emitter)
329  {
330  hit = true;
331  moreRebounds = false;
332 
333  // Random ray hits a primitive
334  distanceToIntersection = randomRay.t * randomRay.t;
335 
336  // Kd
337  if (valid(mat->map_Kd))
338  color = color + get3f(mat->map_Kd, dg);
339  else
340  color = color + mat->Kd;
341 
342  // Emission
343  if (valid(mat->map_a))
344  {
345  vec3f Ka = get3f(mat->map_a, dg);
346  emission += reduce_max(Ka);
347  }
348  else
349  emission += mat->a;
350  }
351  }
352  else
353  moreRebounds = false;
354 
355  randomRay.t0 = randomRay.t + self->epsilonFactor * dg.epsilon;
356  randomRay.t = inf;
357  randomRay.primID = -1;
358  randomRay.geomID = -1;
359  randomRay.instID = -1;
360  }
361 
362  return hit;
363 }

365 inline void indirectShading(const uniform AdvancedRenderer* uniform self,
366  varying ScreenSample& sample,
367  const varying vec3f& intersection,
368  const varying vec3f& normal,
369  varying vec3f& indirectColor,
370  varying float& indirectIntensity,
371  const float epsilon)
372 {
373  if (self->giSamples == 0)
374  return;
375 
376  vec3f backgroundColor = make_vec3f(0.f);
377  float distanceToIntersection = inf;
378 
379  float shadingPower = 0.f;
380  vec3f shadingColor = make_vec3f(0.f);
381 
382  for (uint32 i = 0; i < self->giSamples; ++i)
383  {
384  // Launch a random ray
385  vec3f randomDirection;
386  vec3f color = make_vec3f(0.f);
387  float emission = 0.f;
388  if (launchRandomRay(self, sample, intersection, normal, backgroundColor,
389  distanceToIntersection, randomDirection, color,
390  emission, i, epsilon))
391  {
392  // Kd
393  shadingColor = shadingColor + color;
394 
395  // Emission
396  shadingPower += 1.f + emission;
397  }
398  else
399  {
400  shadingColor = shadingColor + backgroundColor;
401  shadingPower = shadingPower + DEFAULT_SKY_POWER;
402  }
403  }
404 
405  indirectColor = indirectColor + shadingColor / self->giSamples;
406  indirectIntensity = indirectIntensity + shadingPower / self->giSamples;
407 }

409 inline float shadedLightIntensity(varying ScreenSample& sample,
410  const varying Ray& ray,
411  const ShadingAttributes& attributes,
412  const varying vec3f& lightDirection,
413  DifferentialGeometry& dg)
414 {
415  float intensity = 0.f;
416 
417  const uint32 nbSamples = (attributes.self->softShadows > 0.f
418  ? attributes.self->softShadowsSamples
419  : 1);
420 
421  for (uint32 s = 0; s < nbSamples; ++s)
422  {
423  float shadowIntensity = 0.f;
424 
425  vec3f ld = lightDirection;
426  if (attributes.self->softShadows > 0.f)
427  // Slightly alter light direction for Soft shadows
428  ld = normalize(
429  ld + attributes.self->softShadows *
430  getRandomVector(attributes.self->useHardwareRandomizer,
431  attributes.self->super.fb->size.x,
432  sample, attributes.normal,
433  s + attributes.self->randomNumber));
434 
435  if (dot(ld, lightDirection) < 0.f)
436  ld = neg(ld);
437 
438  Ray shadowRay;
439  shadowRay.org = dg.P;
440  shadowRay.dir = ld;
441  shadowRay.t0 = max(0.f, attributes.self->epsilonFactor * dg.epsilon);
442  shadowRay.t = attributes.self->giDistance;
443  shadowRay.time = inf;
444  shadowRay.geomID = -1;
445  shadowRay.primID = -1;
446  shadowRay.instID = -1;
447 
448  while (shadowIntensity < 1.f)
449  {
450  traceRay(attributes.self->super.model, shadowRay);
451 
452  if (shadowRay.geomID == -1)
453  break;
454 
455  DifferentialGeometry shadowingGeometry;
456  postIntersect(attributes.self->super.model, shadowingGeometry,
457  shadowRay,
458  DG_NG | DG_NS | DG_NORMALIZE | DG_TANGENTS |
459  DG_FACEFORWARD | DG_MATERIALID | DG_COLOR |
460  DG_TEXCOORD);
461 
462  uniform AdvancedMaterial* objMaterial =
463  (uniform AdvancedMaterial*)shadowingGeometry.material;
464 
465  float opacity = 1.f;
466 
467  if (objMaterial == 0)
468  opacity = 1.f - shadowingGeometry.color.w;
469  else
470  foreach_unique(mat in objMaterial)
471  {
472  if (mat->chameleonMode == emitter)
473  opacity = 0.f;
474  else
475  {
476  opacity = mat->d;
477  if (valid(mat->map_d))
478  {
479  vec4f d_map = get4f(mat->map_d, shadowingGeometry);
480  opacity = d_map.x;
481  }
482  else if (valid(mat->map_Kd))
483  {
484  vec4f localDiffuseColor_map =
485  get4f(mat->map_Kd, shadowingGeometry);
486  opacity = 1.f - localDiffuseColor_map.w;
487  }
488  }
489  }
490 
491  shadowIntensity += opacity;
492 
493  shadowRay.t0 = max(0.f, shadowRay.t + shadowingGeometry.epsilon);
494  shadowRay.t = inf;
495  shadowRay.primID = -1;
496  shadowRay.geomID = -1;
497  shadowRay.instID = -1;
498  }
499  intensity += shadowIntensity;
500  }
501  intensity /= (float)nbSamples;
502 
503  return clamp(intensity * attributes.self->shadows, 0.f, 1.f);
504 }

527 inline void processShadows(DifferentialGeometry& dg,
528  varying ScreenSample& sample, const varying Ray& ray,
529  ShadingAttributes& attributes)
530 {
531  for (uniform uint32 i = 0; i < attributes.self->numLights; ++i)
532  {
533  const uniform Light* uniform light = attributes.self->lights[i];
534  const vec2f samplingLocation = make_vec2f(0.5f);
535  const varying Light_SampleRes lightSample =
536  light->sample(light, dg, samplingLocation);
537  const vec3f lightColor = lightSample.weight;
538  const float radiance = reduce_max(lightColor);
539  const vec3f lightDirection = lightSample.dir;
540 
541  if (radiance <= 0.f)
542  continue;
543 
544  attributes.shadowIntensity +=
545  shadedLightIntensity(sample, ray, attributes, lightDirection, dg);
546  }
547 
548  if (attributes.shadowIntensity > 0.f)
549  // Remove specular color if surface is in the shades
550  attributes.specularColor = make_vec3f(0.f);
551 }

553 inline vec3f getChameleonColor(const uniform AdvancedRenderer* uniform self,
554  DifferentialGeometry& dg,
555  varying ScreenSample& sample,
556  ShadingAttributes& attributes)
557 {
558  vec3f shadingColor = make_vec3f(0.f);
559  size_t hits = 0;
560  for (uint32 i = 0; i < 20; ++i)
561  {
562  const vec3f randomDirection =
563  getRandomVector(self->useHardwareRandomizer, self->super.fb->size.x,
564  sample, neg(attributes.normal),
565  i + self->randomNumber);
566  Ray ray;
567  ray.org = attributes.origin +
568  randomDirection * (self->epsilonFactor + dg.epsilon);
569  ray.dir = randomDirection;
570  ray.t0 = max(0.f, self->epsilonFactor * dg.epsilon);
571  ray.t = self->giDistance;
572  ray.time = inf;
573  ray.primID = -1;
574  ray.geomID = -1;
575  ray.instID = -1;
576 
577  traceRay(self->super.model, ray);
578 
579  if (ray.geomID >= 0)
580  {
581  // Get intersection information
582  DifferentialGeometry geometry;
583  postIntersect(self->super.model, geometry, ray,
584  DG_NG | DG_NS | DG_NORMALIZE | DG_FACEFORWARD |
585  DG_TANGENTS | DG_MATERIALID | DG_COLOR |
586  DG_TEXCOORD);
587 
588  const uniform Material* material = geometry.material;
589  const uniform AdvancedMaterial* objMaterial =
590  (const uniform AdvancedMaterial*)material;
591  if (objMaterial)
592  {
593  foreach_unique(mat in objMaterial)
594  {
595  if (mat->chameleonMode == emitter &&
596  mat->nodeId == attributes.nodeId)
597  {
598  if (valid(mat->map_Kd))
599  shadingColor =
600  shadingColor + get3f(mat->map_Kd, geometry);
601  else
602  shadingColor = shadingColor + mat->Kd;
603  ++hits;
604  }
605  }
606  }
607  }
608  }
609  if (hits == 0)
610  return shadingColor;
611  return shadingColor / hits;
612 }

614 inline void processLightShading(const uniform AdvancedRenderer* uniform self,
615  DifferentialGeometry& dg,
616  varying ScreenSample& sample,
617  const varying Ray& ray,
618  ShadingAttributes& attributes)
619 {
620  // Shading for all lights defined in the scene
621  attributes.shadingContribution = make_vec3f(0.f);
622 
623  vec3f accDiffuse = make_vec3f(0.f);
624  vec3f accSpecular = make_vec3f(0.f);
625 
626  for (uniform uint32 i = 0; i < attributes.self->numLights; ++i)
627  {
628  const uniform Light* uniform light = attributes.self->lights[i];
629  const vec2f samplingLocation = make_vec2f(0.5f);
630  const varying Light_SampleRes lightSample =
631  light->sample(light, dg, samplingLocation);
632  const vec3f lightColor = lightSample.weight;
633  const float radiance = reduce_max(lightColor);
634 
635  if (radiance <= 0.f)
636  continue;
637 
638  const vec3f lightDirection = lightSample.dir;
639  float cosNL = 1.f;
640 
641  if (attributes.shadingMode != undefined_shading_mode)
642  {
643  // Phong + Blinn shading
644  if (attributes.shadingMode != perlin)
645  cosNL = max(0.f, dot(lightDirection, attributes.normal));
646 
647  // Specular color
648  const float cosLR = dot(lightDirection, attributes.reflectedNormal);
649  accSpecular = accSpecular +
650  lightColor * powf(cosLR, attributes.specularExponent);
651  }
652 
653  accDiffuse =
654  accDiffuse + (lightColor * cosNL * attributes.diffuseColor);
655 
656  if (attributes.shadingMode == diffuse_transparency)
657  attributes.opacity *= sqrt(cosNL);
658  }
659 
660  // Process light contribution
661  attributes.specularColor = accSpecular;
662  attributes.shadingContribution = accDiffuse;
663 }

712 inline void processElectronTransparencyShading(
713  const uniform AdvancedRenderer* uniform self, DifferentialGeometry& dg,
714  varying ScreenSample& sample, ShadingAttributes& attributes)
715 {
716  const vec3f viewer = normalize(sample.ray.org - attributes.origin);
717  const float el = 1.f - max(0.f, dot(viewer, attributes.normal));
718  const float cosNL = 1.f - pow(el, attributes.userParameter);
719 
720  if (attributes.chameleonMode == receiver)
721  attributes.diffuseColor =
722  getChameleonColor(self, dg, sample, attributes);
723

747 inline void processNoiseShading(const uniform AdvancedRenderer* uniform self,
748  DifferentialGeometry& dg,
749  varying ScreenSample& sample,
750  const varying Ray& ray,
751  ShadingAttributes& attributes)
752 {
753  vec3f point = attributes.userParameter * dg.P;
754 
755  const float n1 = 0.25f + 0.75f * clamp(worleyNoise(point, 2.f), 0.f, 1.f);
756 
757  if (attributes.chameleonMode == receiver)
758  attributes.diffuseColor =
759  getChameleonColor(self, dg, sample, attributes);
760 
761  attributes.diffuseColor = (1.f - n1) * attributes.diffuseColor;
762  attributes.specularColor = (1.f - n1) * attributes.specularColor;
763 
764  attributes.normal.x += 0.5f * n1;
765  attributes.normal.y += 0.5f * (0.5f - n1);
766  attributes.normal.z += 0.5f * (0.25f - n1);
767  attributes.normal = normalize(attributes.normal);
768 
769  attributes.reflectedNormal =
770  ray.dir - 2.f * dot(ray.dir, attributes.normal) * attributes.normal;
771 
772  processLightShading(self, dg, sample, ray, attributes);
773 }

775 inline void processCheckerShading(const uniform AdvancedRenderer* uniform self,
776  DifferentialGeometry& dg,
777  varying ScreenSample& sample,
778  const varying Ray& ray,
779  ShadingAttributes& attributes)
780 {
781  if (attributes.chameleonMode == receiver)
782  attributes.diffuseColor =
783  getChameleonColor(self, dg, sample, attributes);
784

786  make_vec3i(attributes.userParameter * (dg.P + make_vec3f(1e2f)));
787  const vec3i p = make_vec3i(point.x % 2, point.y % 2, point.z % 2);

883 inline void processFinalContribution(ShadingAttributes& attributes)
884 {
885  // Surface shading
886  vec3f totalContributions =
887  attributes.simulationColor * attributes.simulationIntensity +
888  attributes.shadingContribution *
889  (1.f - attributes.simulationIntensity) +
890  attributes.specularColor;
891 
892  // Shadow contribution
893  totalContributions =
894  totalContributions * make_vec3f(1.f - attributes.shadowIntensity);
895 
896  // Indirect light contribution
897  totalContributions =
898  totalContributions +
899  attributes.self->giStrength *
900  (attributes.indirectColor * attributes.indirectIntensity);
901 
902  // Light emission contribution
903  totalContributions =
904  totalContributions +
905  attributes.lightEmissionIntensity * attributes.lightEmissionColor;
906 
907  // Store final contribution
908  attributes.finalContribution =
909  clamp(totalContributions, make_vec3f(0.f), make_vec3f(1.f));
910 }

912 inline vec3f AdvancedRenderer_shadeRay(
913  const uniform AdvancedRenderer* uniform self, varying ScreenSample& sample)
914 {
915  Ray ray = sample.ray;
916  float maxt = self->fogStart + self->fogThickness;
917 
918  vec4f color = make_vec4f(0.f);
919  vec3f bgColor = make_vec3f(0.f);
920  if (self->showBackground)
921  bgColor =
922  make_vec3f(skyboxMapping((Renderer*)self, ray,
923  (AdvancedMaterial*)self->bgMaterial));
924 
925  uint32 depth = 0;
926  float oldlocalRefraction = 1.f;
927 
928  sample.z = maxt;
929  ray.t = maxt;
930  bool moreRebounds = true;
931 
932  while (moreRebounds && depth < self->maxBounces && color.w < 1.f)
933  {
934  float totalOpacity = 0.f;
935 
936  // Shading attributes store all color contributions for the
937  // current ray
938  ShadingAttributes attributes;
939  initializeShadingAttributes(self, attributes);
940 
941  // Trace ray
942  traceRay(self->super.model, ray);
943 
944  if (ray.geomID < 0)
945  {
946  if (self->showBackground)
947  {
948  processBackgroundColor(sample, ray, attributes);
949  bgColor = make_vec3f(attributes.bgColor);
950  }
951  moreRebounds = false;
952  }
953  else
954  {
955  DifferentialGeometry dg;
956  postIntersect(self->super.model, dg, ray,
957  DG_NG | DG_NS | DG_NORMALIZE | DG_TANGENTS |
958  DG_MATERIALID | DG_COLOR | DG_FACEFORWARD |
959  DG_TEXCOORD);
960 
961  // Get intersection information
962  // Initialize geometry shading attributes
963  setGeometryShadingAttributes(self, dg, sample, ray, attributes);
964 
965  if (attributes.chameleonMode == emitter)
966  ray.t0 = ray.t + self->epsilonFactor * dg.epsilon;
967  else
968  {
969  // Compute simulation contribution
970  processSimulationContribution(sample, attributes, dg.materialID,
971  ray.primID);
972 
973  if (sample.sampleID.z > 0)
974  // Compute indirect lighting contribution
975  computeIndirectShading(dg, sample, attributes);
976 
977  // Compute surface shading
978  if (attributes.shadingMode == basic)
979  processBasicShading(self, dg, sample, attributes);
980  else if (attributes.shadingMode == electron)
981  processElectronShading(self, dg, sample, attributes);
982  else if (attributes.shadingMode == cartoon)
983  processCartoonShading(self, dg, sample, attributes);
984  else if (attributes.shadingMode == electron_transparency)
985  processElectronTransparencyShading(self, dg, sample,
986  attributes);
987  else if (attributes.shadingMode == perlin)
988  processNoiseShading(self, dg, sample, ray, attributes);
989  else if (attributes.shadingMode == checker)
990  processCheckerShading(self, dg, sample, ray, attributes);
991  else if (attributes.shadingMode == goodsell)
992  processGoodsellShading(self, dg, sample, attributes);
993  else
994  processDiffuseShading(self, dg, sample, ray, attributes);
995 
996  // Compute shadows
997  if (sample.sampleID.z > 0 && self->shadows > 0.f)
998  processShadows(dg, sample, ray, attributes);
999 
1000  // Z-Depth
1001  if (depth == 0)
1002  sample.z = ray.t;
1003 
1004  // Alpha
1005  totalOpacity +=
1006  (1.f - attributes.reflection) * attributes.opacity;
1007 
1008  // Prepare ray for next iteration
1009  bool doRefraction = (attributes.opacity < 1.f);
1010  bool doReflection = (attributes.reflection > 0.f);
1011  if (doRefraction && doReflection)
1012  {
1013  // Switch between refraction and reflection
1014  const bool reflect = sample.sampleID.z % 4 == 1;
1015  doRefraction = !reflect;
1016  doReflection = reflect;
1017  }
1018 
1019  if (doRefraction)
1020  {
1021  // Refraction
1022  ray.dir = refractedVector(ray.dir, attributes.normal,
1023  oldlocalRefraction,
1024  attributes.refraction);
1025  oldlocalRefraction = attributes.refraction;
1026  }
1027  else if (doReflection)
1028  // Reflection
1029  ray.dir = attributes.reflectedNormal;
1030 
1031  // Prepare next ray
1032  ray.org = dg.P + ray.dir * (self->epsilonFactor * dg.epsilon);
1033  ray.t0 = max(0.f, self->epsilonFactor * dg.epsilon);
1034  ++depth;
1035  }
1036 
1037  maxt -= ray.t;
1038  ray.t = maxt;
1039  ray.primID = -1;
1040  ray.geomID = -1;
1041  ray.instID = -1;
1042  }
1043 
1044  // Combine all contributions
1045  processFinalContribution(attributes);
1046 
1047  composite(make_vec4f(attributes.finalContribution, totalOpacity), color,
1048  1.f);
1049  }
1050 
1051  color.w = clamp(color.w, 0.f, 1.f);
1052 
1053  // Fog attenuation
1054  const float fogAttenuation =
1055  sample.z > self->fogStart
1056  ? clamp((sample.z - self->fogStart) / self->fogThickness, 0.f, 1.f)
1057  : 0.f;
1058  const float fog = 1.f - color.w * (1.f - fogAttenuation);
1059 
1060  // Final color
1061  sample.alpha = min(1.f, color.w);
1062  vec3f finalColor =
1063  (make_vec3f(color) * (1.f - fog) + fog * bgColor) * self->exposure;
1064 
1065  if (self->matrixFilter)
1066  finalColor = make_vec3f(finalColor.x * 0.666f, finalColor.y * 0.8f,
1067  finalColor.z * 0.666f);
1068 
1069  finalColor = clamp(finalColor, make_vec3f(0.f), make_vec3f(1.f));
1070  return finalColor;
1071 }
