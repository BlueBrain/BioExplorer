120 bool isClipped(const uniform AdvancedRenderer* uniform self, const varying vec3f& position,
121  const MaterialClippingMode clippingMode)
122 {
123  if (clippingMode == no_clipping)
124  return false;
125 
126  bool visible = true;
127  for (uint32 i = 0; i < self->numClipPlanes && visible; ++i)
128  {
129  if (clippingMode == plane)
130  {
131  const vec3f normal = {self->clipPlanes[i].x, self->clipPlanes[i].y, self->clipPlanes[i].z};
132  const float d = self->clipPlanes[i].w;
133  const float distance = dot(normal, position) + d;
134  visible &= (distance > 0.f);
135  }
136  else if (clippingMode == sphere)
137  {
138  const vec3f center = {self->clipPlanes[i].x, self->clipPlanes[i].y, self->clipPlanes[i].z};
139  const float radius = self->clipPlanes[i].w;
140  visible &= length(center - position) < radius;
141  }
142  }
143  return !visible;
144 }

146 inline void setGeometryShadingAttributes(const uniform AdvancedRenderer* uniform self, DifferentialGeometry& dg,
147  varying ScreenSample& sample, const varying Ray& ray,
148  ShadingAttributes& attributes)
149 {
150  attributes.dg = &dg;
151 
152  // Intersection point with geometry
153  attributes.origin = dg.P;
154 
155  // Shading normal
156  attributes.normal = dg.Ng;
157 
158  // Shading mode
159  attributes.shadingMode = undefined_shading_mode;
160 
161  // Reflected shading normal
162  attributes.reflectedNormal = ray.dir - 2.f * dot(ray.dir, attributes.normal) * attributes.normal;
163 
164  // Chameleon
165  attributes.chameleonMode = undefined_chameleon_mode;
166 
167  // Model Id
168  attributes.nodeId = 0;
169 
170  // Update reflected shading normal
171  attributes.reflectedNormal = ray.dir - 2.f * dot(ray.dir, attributes.normal) * attributes.normal;
172 
173  // Process material attributes
174  const uniform Material* material = dg.material;
175  const uniform AdvancedMaterial* objMaterial = (const uniform AdvancedMaterial*)material;
176  if (!objMaterial)
177  {
178  // Diffuse color
179  attributes.diffuseColor = make_vec3f(dg.color);
180 
181  // if no material is defined, localOpacity is stored in the alpha component of the RGBA color
182  attributes.opacity = dg.color.w;
183  return;
184  }
185 
186  foreach_unique(mat in objMaterial)
187  {
188  // Glossiness
189  if (mat->glossiness < 1.f)
190  {
191  attributes.glossiness = mat->glossiness;
192  const vec3f randomNormal =
193  (1.f - mat->glossiness) * getRandomVector(self->super.useHardwareRandomizer,
194  self->super.super.super.fb->size.x, sample, attributes.normal,
195  self->super.randomNumber);
196  attributes.normal = normalize(attributes.normal + randomNormal);
197  }
198 
199  // Refraction index
200  attributes.refraction = mat->refraction;
201 
202  // Opacity
203  if (valid(mat->map_d))
204  attributes.opacity = mat->d * get1f(mat->map_d, dg);
205  else
206  attributes.opacity = mat->d;
207 
208  // Diffuse color
209  attributes.diffuseColor = mat->Kd * make_vec3f(dg.color);
210  if (valid(mat->map_Kd))
211  {
212  const vec3f diffuseColorFromMap = get3f(mat->map_Kd, dg);
213  attributes.diffuseColor = attributes.diffuseColor * make_vec3f(diffuseColorFromMap);
214  // attributes.opacity *= diffuseColorFromMap.w;
215  }
216 
217  // Specular color
218  if (valid(mat->map_Ks))
219  attributes.specularColor = max(attributes.specularColor, mat->Ks * get3f(mat->map_Ks, dg));
220  else
221  attributes.specularColor = max(attributes.specularColor, mat->Ks);
222 
223  // Specular exponent
224  if (valid(mat->map_Ns))
225  attributes.specularExponent = max(attributes.specularExponent, mat->Ns * get1f(mat->map_Ns, dg));
226  else
227  attributes.specularExponent = max(attributes.specularExponent, mat->Ns);
228 
229  // Light emission intensity
230  attributes.lightEmissionIntensity += mat->a;
231  attributes.lightEmissionColor = attributes.diffuseColor;
232 
233  // Normal mapping
234  if (valid(mat->map_Bump))
235  {
236  const vec3f n = get3f(mat->map_Bump, dg, make_vec3f(0.f)) - 0.5f;
237  attributes.normal.x -= n.x;
238  attributes.normal.y -= n.y;
239  }
240 
241  // Reflection index
242  attributes.reflection = mat->reflection;
243  if (valid(mat->map_reflection))
244  {
245  const vec3f value = get3f(mat->map_reflection, dg);
246  attributes.reflection *= (value.x + value.y + value.z) / 3.f;
247  }
248 
249  // Shading mode
250  attributes.shadingMode = mat->shadingMode;
251 
252  // User parameter
253  attributes.userParameter = mat->userParameter;
254 
255  // Chameleon mode
256  attributes.chameleonMode = mat->chameleonMode;
257 
258  // Cast user data
259  attributes.castUserData = mat->castUserData;
260 
261  // Model Id
262  attributes.nodeId = mat->nodeId;
263 
264  // Clipping planes
265  attributes.clippingMode = mat->clippingMode;
266  }
267 }

269 inline bool launchRandomRay(const uniform AdvancedRenderer* uniform self, varying ScreenSample& sample,
270  const varying vec3f& intersection, const varying vec3f& normal,
271  varying vec3f& backgroundColor, varying float& distanceToIntersection,
272  varying vec3f& randomDirection, varying vec3f& color, varying float& emission,
273  const uint32 iteration, const float epsilon)
274 {
275  randomDirection = getRandomVector(self->super.useHardwareRandomizer, self->super.super.super.fb->size.x, sample,
276  normal, iteration + self->super.randomNumber);
277  backgroundColor = make_vec3f(0.f);
278 
279  if (dot(randomDirection, normal) < 0.f)
280  // Invert direction of random ray direction is opposite to surface normal
281  randomDirection = neg(randomDirection);
282 
283  float nearClip = epsilon;
284  float farClip = self->giRayLength;
285 
286  Ray randomRay;
287  randomRay.org = intersection;
288  randomRay.dir = randomDirection;
289  randomRay.time = inf;
290  randomRay.primID = -1;
291  randomRay.geomID = -1;
292  randomRay.instID = -1;
293 
294  bool moreRebounds = true;
295  bool hit = false;
296  while (moreRebounds)
297  {
298  clipRay(self->clipPlanes, self->numClipPlanes, intersection, randomDirection, nearClip, farClip);
299  randomRay.t0 = nearClip;
300  randomRay.t = farClip;
301  traceRay(self->super.super.super.model, randomRay);
302 
303  if (randomRay.geomID < 0)
304  {
305  // No intersection
306  backgroundColor =
307  make_vec3f(skyboxMapping((Renderer*)self, randomRay, (AdvancedMaterial*)self->super.super.bgMaterial));
308  return false;
309  }
310 
311  DifferentialGeometry dg;
312  postIntersect(self->super.super.super.model, dg, randomRay,
313  DG_NG | DG_NS | DG_NORMALIZE | DG_TANGENTS | DG_FACEFORWARD | DG_MATERIALID | DG_COLOR |
314  DG_TEXCOORD);
315 
316  uniform AdvancedMaterial* objMaterial = (uniform AdvancedMaterial*)dg.material;
317  if (objMaterial)
318  foreach_unique(mat in objMaterial)
319  {
320  if (mat->chameleonMode != emitter)
321  {
322  hit = true;
323  moreRebounds = false;
324 
325  // Random ray hits a primitive
326  distanceToIntersection = randomRay.t * randomRay.t;
327 
328  // Kd
329  if (valid(mat->map_Kd))
330  color = color + get3f(mat->map_Kd, dg);
331  else
332  color = color + mat->Kd;
333 
334  // Emission
335  if (valid(mat->map_a))
336  {
337  vec3f Ka = get3f(mat->map_a, dg);
338  emission += reduce_max(Ka);
339  }
340  else
341  emission += mat->a;
342  }
343  }
344  else
345  moreRebounds = false;
346 
347  nearClip = randomRay.t + self->super.epsilonMultiplier * dg.epsilon;
348  farClip = self->giRayLength - randomRay.t;
349  randomRay.primID = -1;
350  randomRay.geomID = -1;
351  randomRay.instID = -1;
352  }
353 
354  return hit;
355 }

357 inline void indirectShading(const uniform AdvancedRenderer* uniform self, varying ScreenSample& sample,
358  const varying vec3f& intersection, const varying vec3f& normal,
359  varying vec3f& indirectColor, varying float& indirectIntensity, const float epsilon)
360 {
361  if (self->giSamples == 0)
362  return;
363 
364  vec3f backgroundColor = make_vec3f(0.f);
365  float distanceToIntersection = inf;
366 
367  float shadingPower = 0.f;
368  vec3f shadingColor = make_vec3f(0.f);
369 
370  for (uint32 i = 0; i < self->giSamples; ++i)
371  {
372  // Launch a random ray
373  vec3f randomDirection;
374  vec3f color = make_vec3f(0.f);
375  float emission = 0.f;
376  if (launchRandomRay(self, sample, intersection, normal, backgroundColor, distanceToIntersection,
377  randomDirection, color, emission, i, epsilon))
378  {
379  // Kd
380  shadingColor = shadingColor + color;
381 
382  // Emission
383  shadingPower += 1.f + emission;
384  }
385  else
386  {
387  shadingColor = shadingColor + backgroundColor;
388  shadingPower = shadingPower + DEFAULT_SKY_POWER;
389  }
390  }
391 
392  indirectColor = indirectColor + shadingColor / self->giSamples;
393  indirectIntensity = indirectIntensity + shadingPower / self->giSamples;
394 }

396 inline float getVolumeShadowContribution(Volume* uniform volume, const uniform AdvancedRenderer* uniform self,
397  const varying Ray& ray, varying ScreenSample& sample)
398 {
399  // Find volume intersections
400  float t0, t1;
401  intersectBox(ray, volume->boundingBox, t0, t1);
402 
403  // Ray marching from light source to voxel
404  float shadowIntensity = 0.f;
405  const float epsilon = volume->samplingStep / volume->samplingRate;
406  for (float t = t1; t > epsilon && shadowIntensity < 1.f; t -= epsilon)
407  {
408  const vec3f point = ray.org + ray.dir * t;
409  if (isClipped(self, point, plane))
410  continue;
411  const float sample = volume->sample(volume, point);
412 
413  // Look up the opacity associated with the volume sample.
414  shadowIntensity += volume->transferFunction->getOpacityForValue(volume->transferFunction, sample);
415  }
416  return shadowIntensity;
417 }

419 inline float getVolumeShadowContributions(Volume* uniform volume, const uniform uint32 uniform lightIndex,
420  const uniform AdvancedRenderer* uniform self, const varying Ray& ray,
421  varying ScreenSample& sample, const vec3f& point, const float epsilon)
422 {
423  float shadowIntensity = 0.f;
424  const uniform Light* uniform light = self->super.super.lights[lightIndex];
425  const varying vec2f s = make_vec2f(0.5f);
426  DifferentialGeometry dg;
427  dg.P = point;
428  const varying Light_SampleRes lightSample = light->sample(light, dg, s);
429 
430  Ray lightRay = ray;
431  lightRay.t = inf;
432  lightRay.t0 = max(0.f, epsilon);
433  lightRay.org = point;
434  lightRay.geomID = -1;
435  lightRay.flags = RAY_FLAG_SECONDARY;
436 
437  if (self->softShadowStrength > 0.f)
438  lightRay.dir = normalize(lightSample.dir +
439  self->softShadowStrength * getRandomVector(self->super.useHardwareRandomizer,
440  self->super.super.super.fb->size.x, sample,
441  lightSample.dir, self->super.randomNumber));
442  else
443  lightRay.dir = lightSample.dir;
444 
445  // Intersection with Geometry
446  traceRay(self->super.super.super.model, lightRay);
447  if (lightRay.geomID != -1)
448  {
449  shadowIntensity += 1.f;
450  return shadowIntensity * self->shadowIntensity;
451  }
452 
453  // Intersection with volume
454  shadowIntensity += getVolumeShadowContribution(volume, self, lightRay, sample);
455  return shadowIntensity * self->shadowIntensity;
456 }

458 inline vec4f getVolumeContribution(Volume* uniform volume, const uniform AdvancedRenderer* uniform self,
459  varying Ray& ray, varying ScreenSample& sample, float& firstIntersection)
460 {
461  // Find volume intersections
462  float t0, t1;
463  intersectBox(ray, volume->boundingBox, t0, t1);
464  t1 = min(ray.t, t1);
465 
466  vec4f pathColor = make_vec4f(0.f);
467  float epsilon = volume->samplingStep;
468  float shadowIntensity = 0.f;
469 
470  // Ray marching
471  uint32 shadingOccurrence = 0;
472  for (float t = t0 + epsilon /** (sample.sampleID.z % 100)*/; t < t1 && pathColor.w < 1.f; t += epsilon)
473  {
474  const vec3f point = ray.org + t * ray.dir;
475  const float volumeSample = volume->sample(volume, point);
476 
477  // Look up the opacity associated with the volume sample
478  const float sampleOpacity =
479  volume->transferFunction->getOpacityForValue(volume->transferFunction, volumeSample);
480 
481  if (sampleOpacity <= self->volumeSamplingThreshold)
482  // Continue walking for as long as voxel opacity is below
483  // sampling threshold
484  continue;
485 
486  if (shadingOccurrence == 0)
487  {
488  firstIntersection = t;
489  ray.t = t;
490  // Introduce a bit of randomness to smooth the shading
491  t += getRandomValue(self->super.useHardwareRandomizer, make_vec3ui(sample.sampleID)) * ((t1 - t0) * 0.01f);
492  }
493 
494  // Adapt sampling rate to shading occurrence
495  epsilon = (volume->samplingStep + shadingOccurrence) / volume->samplingRate;
496 
497  // Look up the color associated with the volume sample
498  vec3f volumeSampleColor = volume->transferFunction->getColorForValue(volume->transferFunction, volumeSample);
499 
500  // Voxel shading
501  const bool firstShadingOccurrence = shadingOccurrence == 0;
502  const bool shade = volume->singleShade ? firstShadingOccurrence : true;
503 
504  vec3f gradient;
505  DifferentialGeometry dg;
506  const bool shadingEnabled = (shade && volume->gradientShadingEnabled);
507 
508  const bool aoEnabled = (firstShadingOccurrence && self->giStrength > 0.f);
509 
510  const bool shadowsEnabled = (firstShadingOccurrence && self->shadowIntensity > 0.f);
511 
512  if (shadingEnabled || aoEnabled)
513  {
514  gradient = safe_normalize(volume->computeGradient(volume, point));
515  // Setup differential geometry for the volume sample point.
516  dg.P = point;
517  dg.epsilon = self->super.epsilonMultiplier * epsilon;
518  dg.Ns = gradient;
519  }
520 
521  if (shadingEnabled)
522  {
523  // Compute light contributions
524  vec3f shadedColor = make_vec3f(0.f);
525  for (uniform uint32 i = 0; i < self->super.super.numLights; ++i)
526  {
527  const vec2f samplingLocation = make_vec2f(0.5f);
528  const Light_SampleRes light =
529  self->super.super.lights[i]->sample(self->super.super.lights[i], dg, samplingLocation);
530  // Diffuse
531  const float cosNL = (gradient.x == 0.f && gradient.y == 0.f && gradient.z == 0.f)
532  ? 1.f
533  : abs(dot(safe_normalize(light.dir), gradient));
534  shadedColor =
535  clamp(shadedColor + volumeSampleColor * cosNL * light.weight, make_vec3f(0.f), make_vec3f(1.f));
536 
537  // Specular
538  const vec3f reflectedNormal = normalize(ray.dir - 2.f * dot(ray.dir, gradient) * gradient);
539  const float cosNS = powf(max(0.f, dot(light.dir, reflectedNormal)), self->volumeSpecularExponent);
540  shadedColor = shadedColor + volume->specular * cosNS;
541 
542  // Shadow
543  if (shadowsEnabled)
544  // Compute shadow contribution
545  shadowIntensity = getVolumeShadowContributions(volume, i, self, ray, sample, point, epsilon);
546  shadedColor = shadedColor * (1.f - shadowIntensity);
547  }
548  volumeSampleColor = shadedColor;
549  }
550 
551  if (aoEnabled)
552  {
553  // Compute ambient occlusion contribution
554  vec3f indirectColor;
555  float indirectIntensity = 1.f;
556  indirectShading(self, sample, point, gradient, indirectColor, indirectIntensity, dg.epsilon);
557  volumeSampleColor = volumeSampleColor + sampleOpacity * indirectColor * indirectIntensity;
558  }
559 
560  // Compose color with according alpha correction
561  compose(make_vec4f(volumeSampleColor, sampleOpacity), pathColor, self->volumeAlphaCorrection * epsilon);
562 
563  ++shadingOccurrence;
564  }
565 
566  // Apply shadow to RGB values only
567  return pathColor * make_vec4f(make_vec3f(1.f - shadowIntensity), 1.f);
568 }

570 inline float shadedLightIntensity(varying ScreenSample& sample, const varying Ray& ray,
571  const ShadingAttributes& attributes, const varying vec3f& lightDirection,
572  DifferentialGeometry& dg)
573 {
574  float intensity = 0.f;
575 
576  const uint32 nbSamples = (attributes.self->softShadowStrength > 0.f ? attributes.self->shadowSamples : 1);
577 
578  for (uint32 s = 0; s < nbSamples; ++s)
579  {
580  float shadowIntensity = 0.f;
581  vec3f ld = lightDirection;
582  if (attributes.self->softShadowStrength > 0.f)
583  // Slightly alter light direction for Soft shadowIntensity
584  ld = normalize(ld + attributes.self->softShadowStrength *
585  getRandomVector(attributes.self->super.useHardwareRandomizer,
586  attributes.self->super.super.super.fb->size.x, sample,
587  attributes.normal, s + attributes.self->super.randomNumber));
588 
589  if (dot(ld, lightDirection) < 0.f)
590  ld = neg(ld);
591 
592  float nearClip = attributes.self->super.epsilonMultiplier * dg.epsilon;
593  float farClip = attributes.self->giRayLength;
594 
595  Ray shadowRay;
596  shadowRay.org = dg.P;
597  shadowRay.dir = ld;
598  shadowRay.time = ray.time;
599  shadowRay.geomID = -1;
600  shadowRay.primID = -1;
601  shadowRay.instID = -1;
602  shadowRay.flags = RAY_FLAG_SECONDARY;
603 
604  while (shadowIntensity < 1.f)
605  {
606  clipRay(attributes.self->clipPlanes, attributes.self->numClipPlanes, shadowRay.org, shadowRay.dir, nearClip,
607  farClip);
608 
609  shadowRay.t0 = nearClip;
610  shadowRay.t = farClip;
611  traceRay(attributes.self->super.super.super.model, shadowRay);
612 
613  if (shadowRay.geomID < 0)
614  break;
615 
616  DifferentialGeometry shadowingGeometry;
617  postIntersect(attributes.self->super.super.super.model, shadowingGeometry, shadowRay, DG_MATERIALID);
618 
619  if (!isClipped(attributes.self, shadowingGeometry.P, attributes.clippingMode))
620  {
621  uniform AdvancedMaterial* objMaterial = (uniform AdvancedMaterial*)shadowingGeometry.material;
622 
623  float opacity = 1.f;
624  if (objMaterial == 0)
625  opacity = 1.f - shadowingGeometry.color.w;
626  else
627  foreach_unique(mat in objMaterial)
628  {
629  if (mat->chameleonMode == emitter)
630  opacity = 0.f;
631  else
632  {
633  opacity = mat->d;
634  if (valid(mat->map_d))
635  {
636  vec4f d_map = get4f(mat->map_d, shadowingGeometry);
637  opacity = d_map.x;
638  }
639  else if (valid(mat->map_Kd))
640  {
641  vec4f localDiffuseColor_map = get4f(mat->map_Kd, shadowingGeometry);
642  opacity = 1.f - localDiffuseColor_map.w;
643  }
644  }
645  }
646  shadowIntensity += opacity;
647  }
648 
649  // Prepare next ray
650  nearClip = shadowRay.t + shadowingGeometry.epsilon;
651  // farClip = attributes.self->giRayLength - shadowRay.t0;
652 
653  shadowRay.geomID = -1;
654  shadowRay.primID = -1;
655  shadowRay.instID = -1;
656  }
657  intensity += shadowIntensity;
658  }
659  intensity /= (float)nbSamples;
660 
661  return clamp(intensity * attributes.self->shadowIntensity, 0.f, 1.f);
662 }

703 inline vec3f getChameleonColor(const uniform AdvancedRenderer* uniform self, DifferentialGeometry& dg,
704  varying ScreenSample& sample, ShadingAttributes& attributes)
705 {

707  size_t hits = 0;
708  for (uint32 i = 0; i < 20; ++i)
709  {
710  const vec3f randomDirection =
711  getRandomVector(self->super.useHardwareRandomizer, self->super.super.super.fb->size.x, sample,
712  neg(attributes.normal), i + self->super.randomNumber);

714  ray.org = attributes.origin + randomDirection * (self->super.epsilonMultiplier + dg.epsilon);
715  ray.dir = randomDirection;
716  ray.t0 = max(0.f, self->super.epsilonMultiplier * dg.epsilon);
717  ray.t = self->giRayLength;
718  ray.time = inf;
719  ray.primID = -1;
720  ray.geomID = -1;
721  ray.instID = -1;

723  traceRay(self->super.super.super.model, ray);

726  {

728  DifferentialGeometry geometry;
729  postIntersect(self->super.super.super.model, geometry, ray,
730  DG_NG | DG_NS | DG_NORMALIZE | DG_FACEFORWARD | DG_TANGENTS | DG_MATERIALID | DG_COLOR |
731  DG_TEXCOORD);
732 
733  const uniform Material* material = geometry.material;

735  if (objMaterial)

748  }

756 inline void processLightShading(const uniform AdvancedRenderer* uniform self, DifferentialGeometry& dg,
757  varying ScreenSample& sample, const varying Ray& ray, ShadingAttributes& attributes)
758 {
759  // Shading for all lights defined in the scene
760  attributes.shadingContribution = make_vec3f(0.f);
761 
762  vec3f accDiffuse = make_vec3f(0.f);
763  vec3f accSpecular = make_vec3f(0.f);
764 
765  for (uniform uint32 i = 0; i < attributes.self->super.super.numLights; ++i)
766  {
767  const uniform Light* uniform light = attributes.self->super.super.lights[i];
768  const vec2f samplingLocation = make_vec2f(0.5f);
769  const varying Light_SampleRes lightSample = light->sample(light, dg, samplingLocation);
770  const vec3f lightColor = lightSample.weight;
771  const float radiance = reduce_max(lightColor);
772 
773  if (radiance <= 0.f)
774  continue;
775 
776  const vec3f lightDirection = lightSample.dir;
777  float cosNL = 1.f;
778 
779  if (attributes.shadingMode != undefined_shading_mode)
780  {
781  // Phong + Blinn shading
782  if (attributes.shadingMode != perlin)
783  cosNL = max(0.f, dot(lightDirection, attributes.normal));
784 
785  // Specular color
786  const float cosLR = dot(lightDirection, attributes.reflectedNormal);
787  accSpecular = accSpecular + lightColor * powf(cosLR, attributes.specularExponent);
788  }
789 
790  accDiffuse = accDiffuse + (lightColor * cosNL * attributes.diffuseColor);
791 
792  if (attributes.shadingMode == diffuse_transparency)
793  attributes.opacity *= sqrt(cosNL);
794  }
795 
796  // Process light contribution
797  attributes.specularColor = accSpecular;
798  attributes.shadingContribution = accDiffuse;
799 }

980 inline vec3f AdvancedRenderer_shadeRay(const uniform AdvancedRenderer* uniform self, varying ScreenSample& sample)
981 {
982  Ray ray = sample.ray;
983 
984  vec4f color = make_vec4f(0.f);
985  vec3f bgColor = make_vec3f(0.f);
986  if (self->super.super.showBackground)
987  bgColor = make_vec3f(skyboxMapping((Renderer*)self, ray, (AdvancedMaterial*)self->super.super.bgMaterial));
988 
989  sample.z = inf;
990 
991  uint32 depth = 0;
992  float oldRefraction = 1.f;
993  float nearClip = ray.t0;
994  float farClip = ray.t;
995  bool moreRebounds = true;
996 
997  while (moreRebounds && depth < self->super.maxRayDepth && color.w < 1.f)
998  {
999  float totalOpacity = 0.f;
1000 
1001  // Shading attributes store all color contributions for the current ray
1002  ShadingAttributes attributes;
1003  initializeShadingAttributes(self, attributes);
1004 
1005  float firstIntersection = inf;
1006 
1007  // Trace ray
1008  clipRay(attributes.self->clipPlanes, attributes.self->numClipPlanes, ray.org, ray.dir, nearClip, farClip);
1009  ray.t0 = nearClip;
1010  ray.t = farClip;
1011  traceRay(self->super.super.super.model, ray);
1012 
1013  if (ray.geomID < 0)
1014  {
1015  // Volume contribution
1016  processVolumeContribution(sample, ray, attributes, firstIntersection);
1017  sample.z = min(sample.z, firstIntersection);
1018 
1019  // Alpha
1020  totalOpacity += attributes.volumeIntensity;
1021 
1022  // Background
1023  if (self->super.super.showBackground)
1024  {
1025  processBackgroundColor(sample, ray, attributes);
1026  bgColor = make_vec3f(attributes.bgColor);
1027  }
1028  moreRebounds = false;
1029  }
1030  else
1031  {
1032  DifferentialGeometry dg;
1033  postIntersect(self->super.super.super.model, dg, ray,
1034  DG_NG | DG_NS | DG_NORMALIZE | DG_TANGENTS | DG_MATERIALID | DG_COLOR | DG_FACEFORWARD |
1035  DG_TEXCOORD);
1036 
1037  // Initialize geometry shading attributes
1038  setGeometryShadingAttributes(self, dg, sample, ray, attributes);
1039 
1040  const bool discardIntersection =
1041  isClipped(self, dg.P, attributes.clippingMode) || attributes.chameleonMode == emitter;
1042 
1043  if (discardIntersection)
1044  nearClip = ray.t + self->super.epsilonMultiplier * dg.epsilon;
1045  else
1046  {
1047  // Compute user data contribution
1048  processUserDataContribution(sample, attributes, dg.materialID, ray.primID);
1049 
1050  const bool processIndirectLighting = (self->fastPreview ? (sample.sampleID.z > 0) : true);
1051  if (processIndirectLighting)
1052  // Compute indirect lighting contribution
1053  computeIndirectShading(dg, sample, attributes);
1054 
1055  // Compute surface shading
1056  if (attributes.shadingMode == basic)
1057  processBasicShading(self, dg, sample, attributes);
1058  else if (attributes.shadingMode == electron)
1059  processElectronShading(self, dg, sample, attributes);
1060  else if (attributes.shadingMode == cartoon)
1061  processCartoonShading(self, dg, sample, attributes);
1062  else if (attributes.shadingMode == electron_transparency)
1063  processElectronTransparencyShading(self, dg, sample, attributes);
1064  else if (attributes.shadingMode == perlin)
1065  processNoiseShading(self, dg, sample, ray, attributes);
1066  else if (attributes.shadingMode == checker)
1067  processCheckerShading(self, dg, sample, ray, attributes);
1068  else if (attributes.shadingMode == goodsell)
1069  processGoodsellShading(self, dg, sample, attributes);
1070  else if (attributes.shadingMode == surface_normal)
1071  processNormalShading(self, attributes);
1072  else
1073  processDiffuseShading(self, dg, sample, ray, attributes);
1074 
1075  // Compute shadowIntensity
1076  if (processIndirectLighting && self->shadowIntensity > 0.f)
1077  processShadows(dg, sample, ray, attributes);
1078 
1079  // Compute volume contribution
1080  processVolumeContribution(sample, ray, attributes, firstIntersection);
1081 
1082  // Z-Depth
1083  if (depth == 0)
1084  sample.z = ray.t;
1085 
1086  // Opacity
1087  totalOpacity += (1.f - attributes.reflection) * attributes.opacity;
1088 
1089  // Prepare ray for next iteration
1090  bool doRefraction = (attributes.opacity < 1.f);
1091  bool doReflection = (attributes.reflection > 0.f);
1092  if (doRefraction && doReflection)
1093  {
1094  // Switch between refraction and reflection
1095  const bool reflect = sample.sampleID.z % 4 == 1;
1096  doRefraction = !reflect;
1097  doReflection = reflect;
1098  }
1099 
1100  if (doRefraction)
1101  {
1102  // Refraction
1103  ray.dir = refractedVector(ray.dir, attributes.normal, oldRefraction, attributes.refraction);
1104  oldRefraction = attributes.refraction;
1105  }
1106  else if (doReflection)
1107  // Reflection
1108  ray.dir = attributes.reflectedNormal;
1109  else
1110  moreRebounds = false;
1111 
1112  // Prepare next ray
1113  ray.org = dg.P + ray.dir * self->super.epsilonMultiplier * dg.epsilon;
1114  nearClip = 0.f;
1115  farClip = self->fogStart + self->fogThickness;
1116  ++depth;
1117  }
1118  ray.primID = -1;
1119  ray.geomID = -1;
1120  ray.instID = -1;
1121  }
1122 
1123  // Combine all contributions
1124  processFinalContribution(attributes);
1125 
1126  adaptiveCompose(make_vec4f(attributes.finalContribution, totalOpacity), color, 1.f);
1127  }
1128 
1129  color.w = clamp(color.w, 0.f, 1.f);
1130 
1131  // Fog attenuation
1132  const float fogAttenuation =
1133  sample.z > self->fogStart ? clamp((sample.z - self->fogStart) / self->fogThickness, 0.f, 1.f) : 0.f;
1134  const float fog = 1.f - color.w * (1.f - fogAttenuation);
1135 
1136  // Final color
1137  sample.alpha = min(1.f, color.w);
1138  vec3f finalColor = (make_vec3f(color) * (1.f - fog) + fog * bgColor) * self->super.super.exposure;
1139 
1140  if (self->matrixFilter)
1141  finalColor = make_vec3f(finalColor.x * 0.666f, finalColor.y * 0.8f, finalColor.z * 0.666f);
1142 
1143  finalColor = clamp(finalColor, make_vec3f(0.f), make_vec3f(1.f));
1144  return finalColor;
1145 }
