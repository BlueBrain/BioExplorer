60 void OptiXFrameBuffer::_recreate()
61 {
62  CORE_DEBUG("Creating frame buffer...");
63  auto lock = getScopeLock();
64  auto& state = OptiXContext::getInstance().getState();
65 
66  state.params.width = _frameSize.x;
67  state.params.height = _frameSize.y;
68 
69  if (_frameBuffer)
70  {
71  _unmapUnsafe();
72  _frameBuffer->resize(state.params.width, state.params.height);
73  }
74  else
75  {
76  _frameBuffer = new sutil::CUDAOutputBuffer<uchar4>(sutil::CUDAOutputBufferType::CUDA_DEVICE, state.params.width,
77  state.params.height);
78  _frameBuffer->setStream(state.stream);
79  }
80 
81  CUDA_CHECK(cudaFree(state.params.accum_buffer));
82  CUDA_CHECK(cudaMalloc(reinterpret_cast<void**>(&state.params.accum_buffer),
83  state.params.width * state.params.height * sizeof(float4)));
84  state.params.frame_buffer = nullptr; // Will be set when output buffer is mapped
85 
86  clear();
87  CORE_DEBUG("Frame buffer created");
88 }

96 void OptiXFrameBuffer::_mapUnsafe()
97 {
98  // Launch
99  auto& state = OptiXContext::getInstance().getState();
100  state.params.frame_buffer = _frameBuffer->map();
101  CUDA_CHECK(cudaMemcpyAsync(reinterpret_cast<void*>(state.d_params), &state.params, sizeof(Params),
102  cudaMemcpyHostToDevice, state.stream));
103 
104  OPTIX_CHECK(optixLaunch(state.pipeline, state.stream, reinterpret_cast<CUdeviceptr>(state.d_params), sizeof(Params),
105  &state.sbt, state.params.width, state.params.height, 1));
106  _frameBuffer->unmap();
107  CUDA_SYNC_CHECK();
108 
109  sutil::ImageBuffer buffer;
110  buffer.data = _frameBuffer->getHostPointer();
111  buffer.width = state.params.width;
112  buffer.height = state.params.height;
113 
114  switch (_frameBufferFormat)
115  {
116  case FrameBufferFormat::rgba_i8:
117  case FrameBufferFormat::bgra_i8:
118  buffer.pixel_format = sutil::BufferImageFormat::UNSIGNED_BYTE4;
119  break;
120  case FrameBufferFormat::rgb_f32:
121  buffer.pixel_format = sutil::BufferImageFormat::FLOAT4;
122  break;
123  default:
124  CORE_THROW(std::runtime_error("Unsupported frame buffer format"));
125  }
126 
127  _imageData = buffer.data;
128 
129  switch (_frameBufferFormat)
130  {
131  case FrameBufferFormat::rgba_i8:
132  case FrameBufferFormat::bgra_i8:
133  _colorBuffer = (uint8_t*)(_imageData);
134  break;
135  case FrameBufferFormat::rgb_f32:
136  _floatBuffer = (float*)_imageData;
137  break;
138  default:
139  CORE_THROW(std::runtime_error("Unsupported frame buffer format"));
140  }
141 }
