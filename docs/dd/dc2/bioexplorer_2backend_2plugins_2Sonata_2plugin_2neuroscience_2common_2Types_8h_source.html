49 /* Returns a mapping from a name to an enum type. */
50 template <typename EnumT>
51 std::vector<std::pair<std::string, EnumT>> enumerateMap();
52 
53 /* Returns all names for given enum type 'EnumT' */
54 template <typename EnumT>
55 inline std::vector<std::string> enumerateNames()
56 {
57  std::vector<std::string> v;
58  for (const auto& p : enumerateMap<EnumT>())
59  v.push_back(p.first);
60  return v;
61 }
62 
63 /* Convert a string to an enum. */
64 template <typename EnumT>
65 inline EnumT stringToEnum(const std::string& v)
66 {
67  for (const auto& p : enumerateMap<EnumT>())
68  if (p.first == v)
69  return p.second;
70 
71  throw std::runtime_error("Could not match enum '" + v + "'");
72  return static_cast<EnumT>(0);
73 }
74 
75 /* Convert an enum to a string. */
76 template <typename EnumT>
77 inline std::string enumToString(const EnumT v)
78 {
79  for (const auto& p : enumerateMap<EnumT>())
80  if (p.second == v)
81  return p.first;
82 
83  throw std::runtime_error("Could not match enum");
84  return "Invalid";
85 }
