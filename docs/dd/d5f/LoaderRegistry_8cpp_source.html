59 const Loader& LoaderRegistry::getSuitableLoader(const std::string& filename, const std::string& filetype,
60  const std::string& loaderName) const
61 {
62  if (fs::is_directory(filename))
63  throw std::runtime_error("'" + filename + "' is a directory");
64 
65  const auto extension = filetype.empty() ? extractExtension(filename) : filetype;
66 
67  // If we have an archive we always use the archive loader even if a specific
68  // loader is specified
69  if (_archiveSupported(filename, extension))
70  return *_archiveLoader;
71 
72  // Find specific loader
73  if (!loaderName.empty())
74  {
75  for (const auto& loader : _loaders)
76  if (loader->getName() == loaderName)
77  return *loader.get();
78 
79  throw std::runtime_error("No loader found with name '" + loaderName + "'");
80  }
81 
82  for (const auto& loader : _loaders)
83  if (loader->isSupported(filename, extension))
84  return *loader;
85 
86  throw std::runtime_error("No loader found for filename '" + filename + "' and filetype '" + filetype + "'");
87 }
