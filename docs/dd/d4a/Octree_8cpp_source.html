31typedef std::map<uint64_t, OctreeNode> OctreeLevelMap;

33Octree::Octree(const floats &events, float voxelSize, const glm::vec3 &minAABB,
34 const glm::vec3 &maxAABB)
35 : _volumeDim(glm::uvec3(0u, 0u, 0u))
36 , _volumeSize(0u)
37 , _offsetPerLevel(nullptr)
38{
39 PLUGIN_INFO("Nb of events : " << events.size() / 5);
40
41 // **************** Octree creations *******************
42 // *****************************************************
43 glm::uvec3 octreeSize(
44 _pow2roundup(std::ceil((maxAABB.x - minAABB.x) / voxelSize)),
45 _pow2roundup(std::ceil((maxAABB.y - minAABB.y) / voxelSize)),
46 _pow2roundup(std::ceil((maxAABB.z - minAABB.z) / voxelSize)));
47
48 // This octree is always cubic
49 _octreeSize = std::max(std::max(octreeSize.x, octreeSize.y), octreeSize.z);
50
51 PLUGIN_INFO("Octree size : " << _octreeSize);
52
53 uint32_t octreeDepth = std::log2(_octreeSize) + 1u;
54 std::vector<OctreeLevelMap> octree(octreeDepth);
55
56 PLUGIN_INFO("Octree depth : " << octreeDepth << " " << octree.size());
57
58 if (octreeDepth == 0)
59 return;
60
61 for (uint32_t i = 0; i < events.size(); i += 5)
62 {
63 const uint64_t xpos = std::floor((events[i] - minAABB.x) / voxelSize);
64 const uint64_t ypos =
65 std::floor((events[i + 1] - minAABB.y) / voxelSize);
66 const uint64_t zpos =
67 std::floor((events[i + 2] - minAABB.z) / voxelSize);
68 const float value = events[i + 4];
69
70 const uint64_t indexX = xpos;
71 const uint64_t indexY = ypos * (uint64_t)_octreeSize;
72 const uint64_t indexZ =
73 zpos * (uint64_t)_octreeSize * (uint64_t)_octreeSize;
74
75 auto it = octree[0].find(indexX + indexY + indexZ);
76 if (it == octree[0].end())
77 {
78 OctreeNode *child = nullptr;
79 for (uint32_t level = 0; level < octreeDepth; ++level)
80 {
81 bool newNode = false;
82 const uint64_t divisor = std::pow(2, level);
83 const glm::vec3 center(divisor * (xpos / divisor + 0.5f),
84 divisor * (ypos / divisor + 0.5f),
85 divisor * (zpos / divisor + 0.5f));
86
87 const uint64_t nBlock = _octreeSize / divisor;
88 const uint64_t index =
89 std::floor(xpos / divisor) +
90 nBlock * std::floor(ypos / divisor) +
91 nBlock * nBlock * std::floor(zpos / divisor);
92
93 const float size = voxelSize * (level + 1u);
94
95 if (octree[level].find(index) == octree[level].end())
96 {
97 octree[level].insert(
98 OctreeLevelMap::value_type(index,
99 OctreeNode(center, size)));
100 newNode = true;
101 }
102
103 octree[level].at(index).addValue(value);
104
105 if ((level > 0) && (child != nullptr))
106 octree[level].at(index).setChild(child);
107
108 if (newNode)
109 child = &(octree[level].at(index));
110 else
111 child = nullptr;
112 }
113 }
114 else
115 {
116 for (uint32_t level = 0; level < octreeDepth; ++level)
117 {
118 const uint64_t divisor = std::pow(2, level);
119 const uint64_t nBlock = _octreeSize / divisor;
120 const uint64_t index =
121 std::floor(xpos / divisor) +
122 nBlock * std::floor(ypos / divisor) +
123 nBlock * nBlock * std::floor(zpos / divisor);
124 octree[level].at(index).addValue(value);
125 }
126 }
127 }
128 for (uint32_t i = 0; i < octree.size(); ++i)
129 PLUGIN_DEBUG("Number of leaves [" << i << "]: " << octree[i].size());
130
131 // **************** Octree flattening *******************
132 // ******************************************************
133
134 _offsetPerLevel = new uint32_t[octreeDepth];
135 _offsetPerLevel[octreeDepth - 1u] = 0;
136 uint32_t previousOffset = 0u;
137 for (uint32_t i = octreeDepth - 1u; i > 0u; --i)
138 {
139 _offsetPerLevel[i - 1u] = previousOffset + octree[i].size();
140 previousOffset = _offsetPerLevel[i - 1u];
141 }
142
143 uint32_t totalNodeNumber = 0;
144
145 for (uint32_t i = 0; i < octree.size(); ++i)
146 totalNodeNumber += octree[i].size();
147
148 // need to be initialized with zeros
149 _flatIndexes.resize(totalNodeNumber * 2u, 0);
150 _flatData.resize(totalNodeNumber * 4);
151
152 // The root node
153 _flattenChildren(&(octree[octreeDepth - 1u].at(0)), octreeDepth - 1u);
154
155 // **************** Octree flattening end *****************
156 // ********************************************************
157
158 _volumeDim = glm::uvec3(std::ceil((maxAABB.x - minAABB.x) / voxelSize),
159 std::ceil((maxAABB.y - minAABB.y) / voxelSize),
160 std::ceil((maxAABB.z - minAABB.z) / voxelSize));
161 _volumeSize = (uint64_t)_volumeDim.x * (uint64_t)_volumeDim.y *
162 (uint64_t)_volumeDim.z;
163}

170void Octree::_flattenChildren(const OctreeNode *node, uint32_t level)
171{
172 const std::vector<OctreeNode *> children = node->getChildren();
173
174 if ((children.empty()) || (level == 0))
175 {
176 _flatData[_offsetPerLevel[level] * 4u] = node->getCenter().x;
177 _flatData[_offsetPerLevel[level] * 4u + 1] = node->getCenter().y;
178 _flatData[_offsetPerLevel[level] * 4u + 2] = node->getCenter().z;
179 _flatData[_offsetPerLevel[level] * 4u + 3] = node->getValue();
180
181 _offsetPerLevel[level] += 1u;
182 return;
183 }
184 _flatData[_offsetPerLevel[level] * 4u] = node->getCenter().x;
185 _flatData[_offsetPerLevel[level] * 4u + 1] = node->getCenter().y;
186 _flatData[_offsetPerLevel[level] * 4u + 2] = node->getCenter().z;
187 _flatData[_offsetPerLevel[level] * 4u + 3] = node->getValue();
188
189 _flatIndexes[_offsetPerLevel[level] * 2u] = _offsetPerLevel[level - 1];
190 _flatIndexes[_offsetPerLevel[level] * 2u + 1] =
191 _offsetPerLevel[level - 1] + children.size() - 1u;
192 _offsetPerLevel[level] += 1u;
193
194 for (const OctreeNode *child : children)
195 _flattenChildren(child, level - 1u);
196}
