238 void Model::addStreamline(const size_t materialId, const Streamline& streamline)
239 {
240  if (streamline.position.size() < 2)
241  throw std::runtime_error(
242  "Number of vertices is less than two which is minimum needed for a "
243  "streamline.");
244 
245  if (streamline.position.size() != streamline.color.size())
246  throw std::runtime_error("Number of vertices and colors do not match.");
247 
248  if (streamline.position.size() != streamline.radius.size())
249  throw std::runtime_error("Number of vertices and radii do not match.");
250 
251  auto& streamlinesData = _geometries->_streamlines[materialId];
252 
253  const size_t startIndex = streamlinesData.vertex.size();
254  const size_t endIndex = startIndex + streamline.position.size() - 1;
255 
256  for (size_t index = startIndex; index < endIndex; ++index)
257  streamlinesData.indices.push_back(index);
258 
259  for (size_t i = 0; i < streamline.position.size(); i++)
260  streamlinesData.vertex.push_back(Vector4f(streamline.position[i], streamline.radius[i]));
261 
262  for (const auto& color : streamline.color)
263  streamlinesData.vertexColor.push_back(color);
264 
265  _streamlinesDirty = true;
266 }

325 void Model::setMaterialsColorMap(const MaterialsColorMap colorMap)
326 {
327  size_t index = 0;
328  for (auto material : _materials)
329  {
330  material.second->setSpecularColor(Vector3f(0.f));
331  material.second->setOpacity(1.f);
332  material.second->setReflectionIndex(0.f);
333  material.second->setEmission(0.f);
334 
335  switch (colorMap)
336  {
337  case MaterialsColorMap::gradient:
338  {
339  const float a = float(index) / float(_materials.size() - 1);
340  material.second->setDiffuseColor(Vector3f(a * a, std::sqrt(a), 1.f - a));
341  break;
342  }
343  case MaterialsColorMap::pastel:
344  material.second->setDiffuseColor(Vector3f(0.5f + float(std::rand() % 127) / 255.f,
345  0.5f + float(std::rand() % 127) / 255.f,
346  0.5f + float(std::rand() % 127) / 255.f));
347  break;
348  case MaterialsColorMap::random:
349  material.second->setDiffuseColor(
350  Vector3f(float(rand() % 255) / 255.f, float(rand() % 255) / 255.f, float(rand() % 255) / 255.f));
351  switch (rand() % 10)
352  {
353  case 0:
354  // Transparency only
355  material.second->setOpacity(float(std::rand() % 100) / 100.f);
356  material.second->setRefractionIndex(1.2f);
357  material.second->setSpecularColor(Vector3f(1.f));
358  material.second->setSpecularExponent(10.f);
359  break;
360  case 1:
361  // Light emission
362  material.second->setEmission(std::rand() % 20);
363  break;
364  case 2:
365  // Reflection only
366  material.second->setReflectionIndex(float(std::rand() % 100) / 100.f);
367  material.second->setSpecularColor(Vector3f(1.f));
368  material.second->setSpecularExponent(10.f);
369  break;
370  case 3:
371  // Reflection and refraction
372  material.second->setReflectionIndex(float(std::rand() % 100) / 100.f);
373  material.second->setOpacity(float(std::rand() % 100) / 100.f);
374  material.second->setRefractionIndex(1.2f);
375  material.second->setSpecularColor(Vector3f(1.f));
376  material.second->setSpecularExponent(10.f);
377  break;
378  case 4:
379  // Reflection and glossiness
380  material.second->setReflectionIndex(float(std::rand() % 100) / 100.f);
381  material.second->setSpecularColor(Vector3f(1.f));
382  material.second->setSpecularExponent(10.f);
383  material.second->setGlossiness(float(std::rand() % 100) / 100.f);
384  break;
385  case 5:
386  // Transparency and glossiness
387  material.second->setOpacity(float(std::rand() % 100) / 100.f);
388  material.second->setRefractionIndex(1.2f);
389  material.second->setSpecularColor(Vector3f(1.f));
390  material.second->setSpecularExponent(10.f);
391  material.second->setGlossiness(float(std::rand() % 100) / 100.f);
392  break;
393  }
394  break;
395  case MaterialsColorMap::shades_of_grey:
396  float value = float(std::rand() % 255) / 255.f;
397  material.second->setDiffuseColor(Vector3f(value, value, value));
398  break;
399  }
400  material.second->commit();
401  ++index;
402  }

435 void Model::_updateSizeInBytes()
436 {
437  _sizeInBytes = 0;
438  for (const auto& spheres : _geometries->_spheres)
439  _sizeInBytes += spheres.second.size() * sizeof(Sphere);
440  for (const auto& cylinders : _geometries->_cylinders)
441  _sizeInBytes += cylinders.second.size() * sizeof(Cylinder);
442  for (const auto& cones : _geometries->_cones)
443  _sizeInBytes += cones.second.size() * sizeof(Cones);
444  for (const auto& triangleMesh : _geometries->_triangleMeshes)
445  {
446  const auto& mesh = triangleMesh.second;
447  _sizeInBytes += mesh.indices.size() * sizeof(Vector3f);
448  _sizeInBytes += mesh.normals.size() * sizeof(Vector3f);
449  _sizeInBytes += mesh.colors.size() * sizeof(Vector4f);
450  _sizeInBytes += mesh.indices.size() * sizeof(Vector3ui);
451  _sizeInBytes += mesh.textureCoordinates.size() * sizeof(Vector2f);
452  }
453  for (const auto& streamline : _geometries->_streamlines)
454  {
455  _sizeInBytes += streamline.second.indices.size() * sizeof(int32_t);
456  _sizeInBytes += streamline.second.vertex.size() * sizeof(Vector4f);
457  _sizeInBytes += streamline.second.vertexColor.size() * sizeof(Vector4f);
458  }
459 
460  _sizeInBytes += _geometries->_sdf.geometries.size() * sizeof(SDFGeometry);
461  _sizeInBytes += _geometries->_sdf.neighboursFlat.size() * sizeof(uint64_t);
462  for (const auto& sdfIndices : _geometries->_sdf.geometryIndices)
463  _sizeInBytes += sdfIndices.second.size() * sizeof(uint64_t);
464  for (const auto& sdfNeighbours : _geometries->_sdf.neighbours)
465  _sizeInBytes += sdfNeighbours.size() * sizeof(size_t);
466 }

501 void Model::updateBounds()
502 {
503  if (_spheresDirty)
504  {
505  _geometries->_sphereBounds.reset();
506  for (const auto& spheres : _geometries->_spheres)
507  if (spheres.first != BOUNDINGBOX_MATERIAL_ID)
508  for (const auto& sphere : spheres.second)
509  {
510  _geometries->_sphereBounds.merge(sphere.center + sphere.radius);
511  _geometries->_sphereBounds.merge(sphere.center - sphere.radius);
512  }
513  }
514 
515  if (_cylindersDirty)
516  {
517  _geometries->_cylindersBounds.reset();
518  for (const auto& cylinders : _geometries->_cylinders)
519  if (cylinders.first != BOUNDINGBOX_MATERIAL_ID)
520  for (const auto& cylinder : cylinders.second)
521  {
522  _geometries->_cylindersBounds.merge(cylinder.center);
523  _geometries->_cylindersBounds.merge(cylinder.up);
524  }
525  }
526 
527  if (_conesDirty)
528  {
529  _geometries->_conesBounds.reset();
530  for (const auto& cones : _geometries->_cones)
531  if (cones.first != BOUNDINGBOX_MATERIAL_ID)
532  for (const auto& cone : cones.second)
533  {
534  _geometries->_conesBounds.merge(cone.center);
535  _geometries->_conesBounds.merge(cone.up);
536  }
537  }
538 
539  if (_triangleMeshesDirty)
540  {
541  _geometries->_triangleMeshesBounds.reset();
542  for (const auto& mesh : _geometries->_triangleMeshes)
543  if (mesh.first != BOUNDINGBOX_MATERIAL_ID)
544  for (const auto& vertex : mesh.second.vertices)
545  _geometries->_triangleMeshesBounds.merge(vertex);
546  }
547 
548  if (_streamlinesDirty)
549  {
550  _geometries->_streamlinesBounds.reset();
551  for (const auto& streamline : _geometries->_streamlines)
552  for (size_t index = 0; index < streamline.second.vertex.size(); ++index)
553  {
554  const auto& pos = Vector3f(streamline.second.vertex[index]);
555  const float radius = streamline.second.vertex[index][3];
556  const auto radiusVec = Vector3f(radius, radius, radius);
557  _geometries->_streamlinesBounds.merge(pos + radiusVec);
558  _geometries->_streamlinesBounds.merge(pos - radiusVec);
559  }
560  }
561 
562  if (_sdfGeometriesDirty)
563  {
564  _geometries->_sdfGeometriesBounds.reset();
565  for (const auto& geom : _geometries->_sdf.geometries)
566  _geometries->_sdfGeometriesBounds.merge(getSDFBoundingBox(geom));
567  }
568 
569  if (_volumesDirty)
570  {
571  _geometries->_volumesBounds.reset();
572  for (const auto& volume : _geometries->_volumes)
573  _geometries->_volumesBounds.merge(volume.second->getBounds());
574  }
575 
576  if (_curvesDirty)
577  {
578  _geometries->_curvesBounds.reset();
579  for (const auto& curves : _geometries->_curves)
580  for (const auto& curve : curves.second)
581  for (const auto& vertex : curve.vertices)
582  _geometries->_curvesBounds.merge(vertex);
583  }
584 
585  _bounds.reset();
586  _bounds.merge(_geometries->_sphereBounds);
587  _bounds.merge(_geometries->_cylindersBounds);
588  _bounds.merge(_geometries->_conesBounds);
589  _bounds.merge(_geometries->_triangleMeshesBounds);
590  _bounds.merge(_geometries->_streamlinesBounds);
591  _bounds.merge(_geometries->_sdfGeometriesBounds);
592  _bounds.merge(_geometries->_volumesBounds);
593  _bounds.merge(_geometries->_curvesBounds);
594 }
