245 void Model::addStreamline(const size_t materialId, const Streamline& streamline)
246 {
247  if (streamline.position.size() < 2)
248  throw std::runtime_error(
249  "Number of vertices is less than two which is minimum needed for a "
250  "streamline.");
251 
252  if (streamline.position.size() != streamline.color.size())
253  throw std::runtime_error("Number of vertices and colors do not match.");
254 
255  if (streamline.position.size() != streamline.radius.size())
256  throw std::runtime_error("Number of vertices and radii do not match.");
257 
258  auto& streamlinesData = _geometries->_streamlines[materialId];
259 
260  const size_t startIndex = streamlinesData.vertex.size();
261  const size_t endIndex = startIndex + streamline.position.size() - 1;
262 
263  for (size_t index = startIndex; index < endIndex; ++index)
264  streamlinesData.indices.push_back(index);
265 
266  for (size_t i = 0; i < streamline.position.size(); i++)
267  streamlinesData.vertex.push_back(Vector4f(streamline.position[i], streamline.radius[i]));
268 
269  for (const auto& color : streamline.color)
270  streamlinesData.vertexColor.push_back(color);
271 
272  _streamlinesDirty = true;
273 }

332 void Model::setMaterialsColorMap(const MaterialsColorMap colorMap)
333 {
334  size_t index = 0;
335  for (auto material : _materials)
336  {
337  material.second->setSpecularColor(Vector3f(0.f));
338  material.second->setOpacity(1.f);
339  material.second->setReflectionIndex(0.f);
340  material.second->setEmission(0.f);
341 
342  switch (colorMap)
343  {
344  case MaterialsColorMap::gradient:
345  {
346  const float a = float(index) / float(_materials.size() - 1);
347  material.second->setDiffuseColor(Vector3f(a * a, std::sqrt(a), 1.f - a));
348  break;
349  }
350  case MaterialsColorMap::pastel:
351  material.second->setDiffuseColor(Vector3f(0.5f + float(std::rand() % 127) / 255.f,
352  0.5f + float(std::rand() % 127) / 255.f,
353  0.5f + float(std::rand() % 127) / 255.f));
354  break;
355  case MaterialsColorMap::random:
356  material.second->setDiffuseColor(
357  Vector3f(float(rand() % 255) / 255.f, float(rand() % 255) / 255.f, float(rand() % 255) / 255.f));
358  switch (rand() % 10)
359  {
360  case 0:
361  // Transparency only
362  material.second->setOpacity(float(std::rand() % 100) / 100.f);
363  material.second->setRefractionIndex(1.2f);
364  material.second->setSpecularColor(Vector3f(1.f));
365  material.second->setSpecularExponent(10.f);
366  break;
367  case 1:
368  // Light emission
369  material.second->setEmission(std::rand() % 20);
370  break;
371  case 2:
372  // Reflection only
373  material.second->setReflectionIndex(float(std::rand() % 100) / 100.f);
374  material.second->setSpecularColor(Vector3f(1.f));
375  material.second->setSpecularExponent(10.f);
376  break;
377  case 3:
378  // Reflection and refraction
379  material.second->setReflectionIndex(float(std::rand() % 100) / 100.f);
380  material.second->setOpacity(float(std::rand() % 100) / 100.f);
381  material.second->setRefractionIndex(1.2f);
382  material.second->setSpecularColor(Vector3f(1.f));
383  material.second->setSpecularExponent(10.f);
384  break;
385  case 4:
386  // Reflection and glossiness
387  material.second->setReflectionIndex(float(std::rand() % 100) / 100.f);
388  material.second->setSpecularColor(Vector3f(1.f));
389  material.second->setSpecularExponent(10.f);
390  material.second->setGlossiness(float(std::rand() % 100) / 100.f);
391  break;
392  case 5:
393  // Transparency and glossiness
394  material.second->setOpacity(float(std::rand() % 100) / 100.f);
395  material.second->setRefractionIndex(1.2f);
396  material.second->setSpecularColor(Vector3f(1.f));
397  material.second->setSpecularExponent(10.f);
398  material.second->setGlossiness(float(std::rand() % 100) / 100.f);
399  break;
400  }
401  break;
402  case MaterialsColorMap::shades_of_grey:
403  float value = float(std::rand() % 255) / 255.f;
404  material.second->setDiffuseColor(Vector3f(value, value, value));
405  break;
406  }
407  material.second->commit();
408  ++index;
409  }

445 void Model::_updateSizeInBytes()
446 {
447  _sizeInBytes = 0;
448  for (const auto& spheres : _geometries->_spheres)
449  _sizeInBytes += spheres.second.size() * sizeof(Sphere);
450  for (const auto& cylinders : _geometries->_cylinders)
451  _sizeInBytes += cylinders.second.size() * sizeof(Cylinder);
452  for (const auto& cones : _geometries->_cones)
453  _sizeInBytes += cones.second.size() * sizeof(Cones);
454  for (const auto& sdfBeziers : _geometries->_sdfBeziers)
455  _sizeInBytes += sdfBeziers.second.size() * sizeof(SDFBeziers);
456  for (const auto& triangleMesh : _geometries->_triangleMeshes)
457  {
458  const auto& mesh = triangleMesh.second;
459  _sizeInBytes += mesh.indices.size() * sizeof(Vector3f);
460  _sizeInBytes += mesh.normals.size() * sizeof(Vector3f);
461  _sizeInBytes += mesh.colors.size() * sizeof(Vector4f);
462  _sizeInBytes += mesh.indices.size() * sizeof(Vector3ui);
463  _sizeInBytes += mesh.textureCoordinates.size() * sizeof(Vector2f);
464  }
465  for (const auto& streamline : _geometries->_streamlines)
466  {
467  _sizeInBytes += streamline.second.indices.size() * sizeof(int32_t);
468  _sizeInBytes += streamline.second.vertex.size() * sizeof(Vector4f);
469  _sizeInBytes += streamline.second.vertexColor.size() * sizeof(Vector4f);
470  }
471 
472  _sizeInBytes += _geometries->_sdf.geometries.size() * sizeof(SDFGeometry);
473  _sizeInBytes += _geometries->_sdf.neighboursFlat.size() * sizeof(uint64_t);
474  for (const auto& sdfIndices : _geometries->_sdf.geometryIndices)
475  _sizeInBytes += sdfIndices.second.size() * sizeof(uint64_t);
476  for (const auto& sdfNeighbours : _geometries->_sdf.neighbours)
477  _sizeInBytes += sdfNeighbours.size() * sizeof(size_t);
478 }

484

514 void Model::updateBounds()
515 {
516  if (_spheresDirty)
517  {
518  _geometries->_sphereBounds.reset();
519  for (const auto& spheres : _geometries->_spheres)
520  if (spheres.first != BOUNDINGBOX_MATERIAL_ID)
521  for (const auto& sphere : spheres.second)
522  {
523  _geometries->_sphereBounds.merge(sphere.center + sphere.radius);
524  _geometries->_sphereBounds.merge(sphere.center - sphere.radius);
525  }
526  }
527 
528  if (_cylindersDirty)
529  {
530  _geometries->_cylindersBounds.reset();
531  for (const auto& cylinders : _geometries->_cylinders)
532  if (cylinders.first != BOUNDINGBOX_MATERIAL_ID)
533  for (const auto& cylinder : cylinders.second)
534  {
535  _geometries->_cylindersBounds.merge(cylinder.center);
536  _geometries->_cylindersBounds.merge(cylinder.up);
537  }
538  }
539 
540  if (_conesDirty)
541  {
542  _geometries->_conesBounds.reset();
543  for (const auto& cones : _geometries->_cones)
544  if (cones.first != BOUNDINGBOX_MATERIAL_ID)
545  for (const auto& cone : cones.second)
546  {
547  _geometries->_conesBounds.merge(cone.center);
548  _geometries->_conesBounds.merge(cone.up);
549  }
550  }
551 
552  if (_sdfBeziersDirty)
553  {
554  _geometries->_sdfBeziersBounds.reset();
555  for (const auto& sdfBeziers : _geometries->_sdfBeziers)
556  if (sdfBeziers.first != BOUNDINGBOX_MATERIAL_ID)
557  for (const auto& sdfBezier : sdfBeziers.second)
558  _geometries->_sdfBeziersBounds.merge(bezierBounds(sdfBezier));
559  }
560 
561  if (_triangleMeshesDirty)
562  {
563  _geometries->_triangleMeshesBounds.reset();
564  for (const auto& mesh : _geometries->_triangleMeshes)
565  if (mesh.first != BOUNDINGBOX_MATERIAL_ID)
566  for (const auto& vertex : mesh.second.vertices)
567  _geometries->_triangleMeshesBounds.merge(vertex);
568  }
569 
570  if (_streamlinesDirty)
571  {
572  _geometries->_streamlinesBounds.reset();
573  for (const auto& streamline : _geometries->_streamlines)
574  for (size_t index = 0; index < streamline.second.vertex.size(); ++index)
575  {
576  const auto& pos = Vector3f(streamline.second.vertex[index]);
577  const float radius = streamline.second.vertex[index][3];
578  const auto radiusVec = Vector3f(radius, radius, radius);
579  _geometries->_streamlinesBounds.merge(pos + radiusVec);
580  _geometries->_streamlinesBounds.merge(pos - radiusVec);
581  }
582  }
583 
584  if (_sdfGeometriesDirty)
585  {
586  _geometries->_sdfGeometriesBounds.reset();
587  for (const auto& geom : _geometries->_sdf.geometries)
588  _geometries->_sdfGeometriesBounds.merge(getSDFBoundingBox(geom));
589  }
590 
591  if (_volumesDirty)
592  {
593  _geometries->_volumesBounds.reset();
594  for (const auto& volume : _geometries->_volumes)
595  _geometries->_volumesBounds.merge(volume.second->getBounds());
596  }
597 
598  if (_curvesDirty)
599  {
600  _geometries->_curvesBounds.reset();
601  for (const auto& curves : _geometries->_curves)
602  for (const auto& curve : curves.second)
603  for (const auto& vertex : curve.vertices)
604  _geometries->_curvesBounds.merge(vertex);
605  }
606 
607  _bounds.reset();
608  _bounds.merge(_geometries->_sphereBounds);
609  _bounds.merge(_geometries->_cylindersBounds);
610  _bounds.merge(_geometries->_conesBounds);
611  _bounds.merge(_geometries->_sdfBeziersBounds);
612  _bounds.merge(_geometries->_triangleMeshesBounds);
613  _bounds.merge(_geometries->_streamlinesBounds);
614  _bounds.merge(_geometries->_sdfGeometriesBounds);
615  _bounds.merge(_geometries->_volumesBounds);
616  _bounds.merge(_geometries->_curvesBounds);
617 }
