29 /**
30  Renderer a pixel color according to a given location in the screen space.
31  @param self Pointer to current renderer
32  @param sample Screen sample containing information about the ray, and the
33  location in the screen space.
34 */
35 inline vec3f BasicRenderer_shadeRay(const uniform BasicRenderer* uniform self, varying ScreenSample& sample)
36 {
37  Ray ray = sample.ray;
38  varying vec3f color = make_vec3f(0.f);
39 
40  sample.z = inf;
41  sample.alpha = 0.f;
42 
43  traceRay(self->abstract.super.model, ray);
44 
45  if (ray.geomID < 0)
46  {
47  // No intersection, use environment map or background color
48  vec4f bgSample = skyboxMapping((Renderer*)self, ray, self->abstract.bgMaterial);
49  sample.alpha = bgSample.w;
50  return make_vec3f(bgSample);
51  }
52 
53  DifferentialGeometry dg;
54  postIntersect(self->abstract.super.model, dg, ray,
55  DG_NS | DG_NG | DG_NORMALIZE | DG_FACEFORWARD | DG_MATERIALID | DG_COLOR | DG_TEXCOORD);
56 
57  uniform AdvancedMaterial* material = (uniform AdvancedMaterial*)dg.material;
58 
59  vec3f Kd = make_vec3f(0.f);
60  if (!material)
61  Kd = make_vec3f(dg.color);
62  else
63  foreach_unique(mat in material)
64  {
65  Kd = mat->Kd * make_vec3f(dg.color);
66  if (valid(mat->map_Kd))
67  {
68  const vec4f diffuseColorFromMap = get4f(mat->map_Kd, dg);
69  Kd = make_vec3f(dg.color * diffuseColorFromMap);
70  }
71  }
72 
73  // Head-light shading
74  color = Kd * max(0.f, dot(neg(ray.dir), dg.Ns));
75  sample.alpha = 1.f;
76 
77  // Z-Depth
78  sample.z = ray.t;
79  sample.alpha = 1.f;
80 
81  return color;
82 }

84 void BasicRenderer_renderSample(uniform Renderer* uniform _self, void* uniform perFrameData,
85  varying ScreenSample& sample)
86 {
87  uniform BasicRenderer* uniform self = (uniform BasicRenderer * uniform) _self;
88  sample.ray.time = self->abstract.timestamp;
89  if (self->abstract.anaglyphEnabled)
90  {
91  ScreenSample s = sample;
92  s.ray.org = s.ray.org - self->abstract.anaglyphIpdOffset;
93  const vec3f colorLeft = BasicRenderer_shadeRay(self, s);
94  const vec3f leftAnaglyphColor = getAnaglyphLeftColor(colorLeft);
95 
96  s = sample;
97  s.ray.org = s.ray.org + self->abstract.anaglyphIpdOffset;
98  const vec3f colorRight = BasicRenderer_shadeRay(self, s);
99  const vec3f rightAnaglyphColor = getAnaglyphRightColor(colorRight);
100 
101  sample.rgb = leftAnaglyphColor + rightAnaglyphColor;
102  }
103  else
104  sample.rgb = BasicRenderer_shadeRay(self, sample);
105 }
