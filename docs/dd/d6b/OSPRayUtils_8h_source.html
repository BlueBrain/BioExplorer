54 template <size_t S>
55 std::vector<std::array<float, S>> convertVectorToFloat(const std::vector<std::array<double, S>>& input)
56 {
57  std::vector<std::array<float, S>> output;
58  output.reserve(input.size());
59  for (const auto& value : input)
60  {
61  std::array<float, S> converted;
62  std::copy(value.data(), value.data() + S, converted.data());
63  output.push_back(converted);
64  }
65  return output;
66 }
