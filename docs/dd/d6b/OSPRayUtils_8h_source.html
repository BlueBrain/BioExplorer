48 template <size_t S>
49 std::vector<std::array<float, S>> convertVectorToFloat(const std::vector<std::array<double, S>>& input)
50 {
51  std::vector<std::array<float, S>> output;
52  output.reserve(input.size());
53  for (const auto& value : input)
54  {
55  std::array<float, S> converted;
56  std::copy(value.data(), value.data() + S, converted.data());
57  output.push_back(converted);
58  }
59  return output;
60 }
