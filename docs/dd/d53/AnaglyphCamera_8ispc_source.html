52 void AnaglyphCamera_initRay(uniform Camera* uniform _self, varying Ray& ray, const varying CameraSample& sample)
53 {
54  uniform AnaglyphCamera* uniform self = (uniform AnaglyphCamera * uniform) _self;
55 
56  vec2f screen = sample.screen;
57  vec3f org = self->org;
58 
59  screen = Camera_subRegion(_self, screen);
60 
61  vec3f dir = self->dir_00 + screen.x * self->dir_du + screen.y * self->dir_dv;
62 
63  if (self->super.doesDOF)
64  {
65  float random = getRandomValue(self->useHardwareRandomizer, make_vec3ui(screen.x, screen.y, 0.f));
66  random = (random == 0.f ? 0.001f : random);
67  const vec3f llp = uniformSampleDisk(random * self->scaledAperture, sample.lens);
68  // transform local lens point to focal plane (dir_XX are prescaled in this case)
69  const vec3f lp = (llp.x * self->dir_du) + ((llp.y * self->aspect) * self->dir_dv);
70  org = org + lp;
71  dir = dir - lp;
72  }
73 
74  float nearClip = self->super.nearClip;
75  float farClip = inf;
76  const varying vec3f direction = normalize(dir);
77 
78  clipRay(self->clipPlanes, self->numClipPlanes, org, direction, nearClip, farClip);
79 
80  const float time = Camera_shutterTime(_self, sample.time);
81  ray.flags = RAY_FLAG_PRIMARY;
82  setRay(ray, org, direction, nearClip, farClip, time);
83 }
