142 
143  entryPoint = PLUGIN_API_PREFIX + "set-odu-camera";
144  PLUGIN_REGISTER_ENDPOINT(entryPoint);
145  actionInterface->registerNotification<CameraDefinition>(entryPoint,

147 
148  entryPoint = PLUGIN_API_PREFIX + "get-odu-camera";
149  PLUGIN_REGISTER_ENDPOINT(entryPoint);

151  [&]() -> CameraDefinition { return _getCamera(); });
152 
153  entryPoint = PLUGIN_API_PREFIX + "export-frames-to-disk";
154  PLUGIN_REGISTER_ENDPOINT(entryPoint);
155  actionInterface->registerNotification<ExportFramesToDisk>(entryPoint, [&](const ExportFramesToDisk &s)
156  { _exportFramesToDisk(s); });
157 
158  entryPoint = PLUGIN_API_PREFIX + "get-export-frames-progress";
159  PLUGIN_REGISTER_ENDPOINT(entryPoint);
160  actionInterface->registerRequest<FrameExportProgress>(entryPoint,
161  [&](void) -> FrameExportProgress

163  }

165  auto &engine = _api->getEngine();

167  const auto &engineName = params.getEngine();
168 #ifdef USE_OPTIX6
169  if (engineName == ENGINE_OPTIX_6)

173  }
174 #endif
175  if (engineName == ENGINE_OSPRAY)

199  ptx, ::core::engine::optix::OPTIX_CUDA_FUNCTION_CLOSEST_HIT_RADIANCE);

201  ptx, ::core::engine::optix::OPTIX_CUDA_FUNCTION_CLOSEST_HIT_RADIANCE_TEXTURED);

229 void MediaMakerPlugin::preRender()
230 {
231  if (_exportFramesToDiskDirty && _accumulationFrameNumber == 0)
232  {
233  auto &frameBuffer = _api->getEngine().getFrameBuffer();
234  frameBuffer.resize(_frameBufferSize);
235  frameBuffer.clear();
236 
237  auto &camera = _api->getCamera();
238  if (camera.hasProperty(CAMERA_PROPERTY_ASPECT_RATIO.name))
239  camera.updateProperty(CAMERA_PROPERTY_ASPECT_RATIO.name,
240  static_cast<double>(_frameBufferSize.x) / static_cast<double>(_frameBufferSize.y));
241  camera.commit();
242 
243  const uint64_t i = CAMERA_DEFINITION_SIZE * _frameNumber;
244  // Camera position
245  CameraDefinition cd;
246  const auto &ci = _exportFramesToDiskPayload.cameraInformation;
247  cd.origin = {ci[i], ci[i + 1], ci[i + 2]};
248  cd.direction = {ci[i + 3], ci[i + 4], ci[i + 5]};
249  cd.up = {ci[i + 6], ci[i + 7], ci[i + 8]};
250  cd.apertureRadius = ci[i + 9];
251  cd.focalDistance = ci[i + 10];
252  cd.interpupillaryDistance = ci[i + 11];
253  _setCamera(cd);
254 
255  // Animation parameters
256  const auto &ai = _exportFramesToDiskPayload.animationInformation;
257  if (!ai.empty())
258  _api->getParametersManager().getAnimationParameters().setFrame(ai[_frameNumber]);
259  }
260 }

262 void MediaMakerPlugin::postRender()
263 {
264  ++_accumulationFrameNumber;
265 
266  if (_exportFramesToDiskDirty)
267  {
268  try
269  {
270  if (_exportFramesToDiskPayload.exportIntermediateFrames)
271  _exportFrameToDisk();
272 
273  if (_accumulationFrameNumber == _exportFramesToDiskPayload.spp)
274  {
275  ++_frameNumber;
276  _accumulationFrameNumber = 0;
277  _exportFramesToDiskDirty = (_frameNumber < _exportFramesToDiskPayload.endFrame);
278  _exportFrameToDisk();
279  }
280  }
281  catch (const std::runtime_error &e)
282  {
283  PLUGIN_ERROR(e.what());
284  }
285  }

288 void MediaMakerPlugin::_setCamera(const CameraDefinition &payload)
289 {
290  auto &camera = _api->getCamera();
291 
292  // Origin
293  const auto &o = payload.origin;
294  core::Vector3d origin{o[0], o[1], o[2]};
295  camera.setPosition(origin);
296 
297  // Target
298  const auto &d = payload.direction;
299  core::Vector3d direction{d[0], d[1], d[2]};
300  camera.setTarget(origin + direction);
301 
302  // Up
303  const auto &u = payload.up;
304  core::Vector3d up{u[0], u[1], u[2]};
305 
306  // Orientation
307  const auto q = glm::inverse(glm::lookAt(origin, origin + direction,
308  up)); // Not quite sure why this
309  // should be inverted?!?
310  camera.setOrientation(q);
311 
312  // Aperture
313  if (camera.hasProperty(CAMERA_PROPERTY_APERTURE_RADIUS.name))
314  camera.updateProperty(CAMERA_PROPERTY_APERTURE_RADIUS.name, payload.apertureRadius);
315 
316  // Focus distance
317  if (camera.hasProperty(CAMERA_PROPERTY_FOCAL_DISTANCE.name))
318  camera.updateProperty(CAMERA_PROPERTY_FOCAL_DISTANCE.name, payload.focalDistance);
319 
320  // Stereo
321  if (camera.hasProperty(CAMERA_PROPERTY_STEREO.name))
322  camera.updateProperty(CAMERA_PROPERTY_STEREO.name, payload.interpupillaryDistance != 0.0);
323  if (camera.hasProperty(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name))
324  camera.updateProperty(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name, payload.interpupillaryDistance);
325 
326  _api->getCamera().markModified();
327 }

329 CameraDefinition MediaMakerPlugin::_getCamera()
330 {
331  const auto &camera = _api->getCamera();
332 
333  CameraDefinition cd;
334  const auto &p = camera.getPosition();
335  cd.origin = {p.x, p.y, p.z};
336  const auto d = glm::rotate(camera.getOrientation(), core::Vector3d(0., 0., -1.));
337  cd.direction = {d.x, d.y, d.z};
338  const auto u = glm::rotate(camera.getOrientation(), core::Vector3d(0., 1., 0.));
339  cd.up = {u.x, u.y, u.z};
340  if (camera.hasProperty(CAMERA_PROPERTY_APERTURE_RADIUS.name))
341  cd.apertureRadius = camera.getProperty<double>(CAMERA_PROPERTY_APERTURE_RADIUS.name);
342  if (camera.hasProperty(CAMERA_PROPERTY_FOCAL_DISTANCE.name))
343  cd.focalDistance = camera.getProperty<double>(CAMERA_PROPERTY_FOCAL_DISTANCE.name);
344  if (camera.hasProperty(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name))
345  cd.interpupillaryDistance = camera.getProperty<double>(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name);
346  return cd;
347 }

361 void MediaMakerPlugin::_exportColorBuffer() const
362 {
363  auto &frameBuffer = _api->getEngine().getFrameBuffer();
364  auto image = frameBuffer.getImage();
365  auto fif = _exportFramesToDiskPayload.format == "jpg"
366  ? FIF_JPEG
367  : FreeImage_GetFIFFromFormat(_exportFramesToDiskPayload.format.c_str());
368  if (fif == FIF_JPEG)
369  image.reset(FreeImage_ConvertTo24Bits(image.get()));
370  else if (fif == FIF_UNKNOWN)
371  PLUGIN_THROW("Unknown format: " + _exportFramesToDiskPayload.format);
372 
373  int flags = _exportFramesToDiskPayload.quality;
374  if (fif == FIF_TIFF)
375  flags = TIFF_NONE;
376 
377  core::freeimage::MemoryPtr memory(FreeImage_OpenMemory());
378 
379  FreeImage_SaveToMemory(fif, image.get(), memory.get(), flags);
380 
381  BYTE *pixels = nullptr;
382  DWORD numPixels = 0;
383  FreeImage_AcquireMemory(memory.get(), &pixels, &numPixels);
384 
385  const auto filename = _getFileName(_exportFramesToDiskPayload.format);
386  std::ofstream file;
387  file.open(filename, std::ios_base::binary);
388  if (!file.is_open())
389  PLUGIN_THROW("Failed to create " + filename);
390 
391  file.write((char *)pixels, numPixels);
392  file.close();
393  frameBuffer.clear();
394 
395  auto finalImage = Exiv2::ImageFactory::open(filename);
396  if (finalImage.get())
397  {
398  Exiv2::XmpData xmpData;
399  xmpData["Xmp.dc.Source"] = "Blue Brain BioExplorer";
400  xmpData["Xmp.dc.Subject"] = _exportFramesToDiskPayload.keywords;
401  finalImage->setXmpData(xmpData);
402  finalImage->writeMetadata();
403  }
404 
405  PLUGIN_INFO("Color frame saved to " + filename);
406 }

408 void MediaMakerPlugin::_exportDepthBuffer() const
409 {
410  auto &frameBuffer = _api->getEngine().getFrameBuffer();
411  frameBuffer.map();
412  const auto depthBuffer = frameBuffer.getFloatBuffer();
413  const auto &size = frameBuffer.getSize();
414 
415  const auto filename = _getFileName("tiff");
416 
417  TIFF *image = TIFFOpen(filename.c_str(), "w");
418  TIFFSetField(image, TIFFTAG_IMAGEWIDTH, size.x);
419  TIFFSetField(image, TIFFTAG_IMAGELENGTH, size.y);
420  TIFFSetField(image, TIFFTAG_BITSPERSAMPLE, 32);
421  TIFFSetField(image, TIFFTAG_SAMPLESPERPIXEL, 1);
422  TIFFSetField(image, TIFFTAG_ROWSPERSTRIP, 1);
423  TIFFSetField(image, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT);
424  TIFFSetField(image, TIFFTAG_PLANARCONFIG, PLANARCONFIG_CONTIG);
425  TIFFSetField(image, TIFFTAG_PHOTOMETRIC, PHOTOMETRIC_MINISBLACK);
426  TIFFSetField(image, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_IEEEFP);
427  TIFFSetField(image, TIFFTAG_COMPRESSION, COMPRESSION_NONE);
428 
429  float *scan_line = (float *)malloc(1 + size.x * (sizeof(float)));
430 
431  for (uint32_t i = 0; i < size.y; ++i)
432  {
433  memcpy(scan_line, &depthBuffer[i * size.x], size.x * sizeof(float));
434  TIFFWriteScanline(image, scan_line, size.y - 1 - i, 0);
435  }
436 
437  TIFFClose(image);
438  free(scan_line);
439 
440  PLUGIN_INFO("Depth frame saved to " + filename);
441 
442  frameBuffer.unmap();
443 }

488 FrameExportProgress MediaMakerPlugin::_getFrameExportProgress()
489 {
490  FrameExportProgress result;
491  double percentage = 1.f;
492  const size_t nbFrames = _exportFramesToDiskPayload.cameraInformation.size() / CAMERA_DEFINITION_SIZE;
493  const size_t totalNumberOfFrames = nbFrames * _exportFramesToDiskPayload.spp;
494 
495  if (totalNumberOfFrames != 0)
496  {
497  const double currentProgress = _frameNumber * _exportFramesToDiskPayload.spp + _accumulationFrameNumber;
498  percentage = currentProgress / double(totalNumberOfFrames);
499  }
500  result.progress = percentage;
501  result.done = !_exportFramesToDiskDirty;
502  PLUGIN_DEBUG("Percentage = " << result.progress << ", Done = " << (result.done ? "True" : "False"));
503  return result;
504 }
