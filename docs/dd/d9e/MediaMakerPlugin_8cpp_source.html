97 void MediaMakerPlugin::init()
98 {
99  auto actionInterface = _api->getActionInterface();
100  auto &engine = _api->getEngine();
101 
102  if (actionInterface)
103  {
104  std::string entryPoint = PLUGIN_API_PREFIX + "version";
105  PLUGIN_INFO("Registering '" + entryPoint + "' endpoint");
106  actionInterface->registerRequest<Response>(entryPoint, [&]()
107  { return _version(); });
108 
109  entryPoint = PLUGIN_API_PREFIX + "set-odu-camera";
110  PLUGIN_INFO("Registering '" + entryPoint + "' endpoint");
111  actionInterface->registerNotification<CameraDefinition>(
112  entryPoint, [&](const CameraDefinition &s) { _setCamera(s); });
113 
114  entryPoint = PLUGIN_API_PREFIX + "get-odu-camera";
115  PLUGIN_INFO("Registering '" + entryPoint + "' endpoint");
116  actionInterface->registerRequest<CameraDefinition>(
117  entryPoint, [&]() -> CameraDefinition { return _getCamera(); });
118 
119  entryPoint = PLUGIN_API_PREFIX + "export-frames-to-disk";
120  PLUGIN_INFO("Registering '" + entryPoint + "' endpoint");
121  actionInterface->registerNotification<ExportFramesToDisk>(
122  entryPoint,
123  [&](const ExportFramesToDisk &s) { _exportFramesToDisk(s); });
124 
125  entryPoint = PLUGIN_API_PREFIX + "get-export-frames-progress";
126  PLUGIN_INFO("Registering '" + entryPoint + "' endpoint");
127  actionInterface->registerRequest<FrameExportProgress>(
128  entryPoint,
129  [&](void) -> FrameExportProgress
130  { return _getFrameExportProgress(); });
131  }
132 
133  _addDepthRenderer(engine);
134  _addAlbedoRenderer(engine);
135  _addAmbientOcclusionRenderer(engine);
136  _addShadowRenderer(engine);
137 
138  engine.addRendererType("raycast_Ng");
139  engine.addRendererType("raycast_Ns");
140 }

149 void MediaMakerPlugin::preRender()
150 {
151  if (_exportFramesToDiskDirty && _accumulationFrameNumber == 0)
152  {
153  const uint64_t i = CAMERA_DEFINITION_SIZE * _frameNumber;
154  // Camera position
155  CameraDefinition cd;
156  const auto &ci = _exportFramesToDiskPayload.cameraInformation;
157  cd.origin = {ci[i], ci[i + 1], ci[i + 2]};
158  cd.direction = {ci[i + 3], ci[i + 4], ci[i + 5]};
159  cd.up = {ci[i + 6], ci[i + 7], ci[i + 8]};
160  cd.apertureRadius = ci[i + 9];
161  cd.focusDistance = ci[i + 10];
162  cd.interpupillaryDistance = ci[i + 11];
163  _setCamera(cd);
164 
165  // Animation parameters
166  const auto &ai = _exportFramesToDiskPayload.animationInformation;
167  if (!ai.empty())
168  _api->getParametersManager().getAnimationParameters().setFrame(
169  ai[_frameNumber]);
170  }
171 }

173 void MediaMakerPlugin::postRender()
174 {
175  ++_accumulationFrameNumber;
176 
177  if (_exportFramesToDiskDirty)
178  {
179  if (_exportFramesToDiskPayload.exportIntermediateFrames)
180  _exportFrameToDisk();
181 
182  if (_accumulationFrameNumber == _exportFramesToDiskPayload.spp)
183  {
184  _exportFrameToDisk();
185  ++_frameNumber;
186  _accumulationFrameNumber = 0;
187  _exportFramesToDiskDirty =
188  (_frameNumber < _exportFramesToDiskPayload.endFrame);
189  }
190  }
191 }

193 void MediaMakerPlugin::_setCamera(const CameraDefinition &payload)
194 {
195  auto &camera = _api->getCamera();
196 
197  // Origin
198  const auto &o = payload.origin;
199  brayns::Vector3d origin{o[0], o[1], o[2]};
200  camera.setPosition(origin);
201 
202  // Target
203  const auto &d = payload.direction;
204  brayns::Vector3d direction{d[0], d[1], d[2]};
205  camera.setTarget(origin + direction);
206 
207  // Up
208  const auto &u = payload.up;
209  brayns::Vector3d up{u[0], u[1], u[2]};
210 
211  // Orientation
212  const auto q = glm::inverse(glm::lookAt(origin, origin + direction,
213  up)); // Not quite sure why this
214  // should be inverted?!?
215  camera.setOrientation(q);
216 
217  // Aperture
218  camera.updateProperty("apertureRadius", payload.apertureRadius);
219 
220  // Focus distance
221  camera.updateProperty("focusDistance", payload.focusDistance);
222 
223  // Stereo
224  camera.updateProperty("stereo", payload.interpupillaryDistance != 0.0);
225  camera.updateProperty("interpupillaryDistance",
226  payload.interpupillaryDistance);
227 
228  _api->getCamera().markModified();
229 }

231 CameraDefinition MediaMakerPlugin::_getCamera()
232 {
233  const auto &camera = _api->getCamera();
234 
235  CameraDefinition cd;
236  const auto &p = camera.getPosition();
237  cd.origin = {p.x, p.y, p.z};
238  const auto d =
239  glm::rotate(camera.getOrientation(), brayns::Vector3d(0., 0., -1.));
240  cd.direction = {d.x, d.y, d.z};
241  const auto u =
242  glm::rotate(camera.getOrientation(), brayns::Vector3d(0., 1., 0.));
243  cd.up = {u.x, u.y, u.z};
244  return cd;
245 }

260 void MediaMakerPlugin::_exportColorBuffer() const
261 {
262  auto &frameBuffer = _api->getEngine().getFrameBuffer();
263  auto image = frameBuffer.getImage();
264  auto fif = _exportFramesToDiskPayload.format == "jpg"
265  ? FIF_JPEG
266  : FreeImage_GetFIFFromFormat(
267  _exportFramesToDiskPayload.format.c_str());
268  if (fif == FIF_JPEG)
269  image.reset(FreeImage_ConvertTo24Bits(image.get()));
270  else if (fif == FIF_UNKNOWN)
271  PLUGIN_THROW("Unknown format: " + _exportFramesToDiskPayload.format);
272 
273  int flags = _exportFramesToDiskPayload.quality;
274  if (fif == FIF_TIFF)
275  flags = TIFF_NONE;
276 
277  brayns::freeimage::MemoryPtr memory(FreeImage_OpenMemory());
278 
279  FreeImage_SaveToMemory(fif, image.get(), memory.get(), flags);
280 
281  BYTE *pixels = nullptr;
282  DWORD numPixels = 0;
283  FreeImage_AcquireMemory(memory.get(), &pixels, &numPixels);
284 
285  const auto filename = _getFileName(_exportFramesToDiskPayload.format);
286  std::ofstream file;
287  file.open(filename, std::ios_base::binary);
288  if (!file.is_open())
289  PLUGIN_THROW("Failed to create " + filename);
290 
291  file.write((char *)pixels, numPixels);
292  file.close();
293  frameBuffer.clear();
294 
295  PLUGIN_INFO("Color frame saved to " + filename);
296 }

298 void MediaMakerPlugin::_exportDepthBuffer() const
299 {
300  auto &frameBuffer = _api->getEngine().getFrameBuffer();
301  frameBuffer.map();
302  const auto depthBuffer = frameBuffer.getDepthBuffer();
303  const auto &size = frameBuffer.getSize();
304 
305  const auto filename = _getFileName("tiff");
306 
307  TIFF *image = TIFFOpen(filename.c_str(), "w");
308  TIFFSetField(image, TIFFTAG_IMAGEWIDTH, size.x);
309  TIFFSetField(image, TIFFTAG_IMAGELENGTH, size.y);
310  TIFFSetField(image, TIFFTAG_BITSPERSAMPLE, 32);
311  TIFFSetField(image, TIFFTAG_SAMPLESPERPIXEL, 1);
312  TIFFSetField(image, TIFFTAG_ROWSPERSTRIP, 1);
313  TIFFSetField(image, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT);
314  TIFFSetField(image, TIFFTAG_PLANARCONFIG, PLANARCONFIG_CONTIG);
315  TIFFSetField(image, TIFFTAG_PHOTOMETRIC, PHOTOMETRIC_MINISBLACK);
316  TIFFSetField(image, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_IEEEFP);
317  TIFFSetField(image, TIFFTAG_COMPRESSION, COMPRESSION_NONE);
318 
319  double *scan_line = (double *)malloc(size.x * (sizeof(double)));
320 
321  for (uint32_t i = 0; i < size.y; ++i)
322  {
323  memcpy(scan_line, &depthBuffer[i * size.x], size.x * sizeof(double));
324  TIFFWriteScanline(image, scan_line, size.y - 1 - i, 0);
325  }
326 
327  TIFFClose(image);
328  free(scan_line);
329 
330  PLUGIN_INFO("Depth frame saved to " + filename);
331 
332  frameBuffer.unmap();
333 }

335 void MediaMakerPlugin::_exportFrameToDisk() const
336 {
337  switch (_exportFramesToDiskPayload.frameBufferMode)
338  {
339  case FrameBufferMode::color:
340  _exportColorBuffer();
341  break;
342  case FrameBufferMode::depth:
343  _exportDepthBuffer();
344  break;
345  default:
346  PLUGIN_THROW("Undefined frame buffer mode")
347  }

350 void MediaMakerPlugin::_exportFramesToDisk(const ExportFramesToDisk &payload)
351 {
352  _exportFramesToDiskPayload = payload;
353  _exportFramesToDiskDirty = true;
354  _frameNumber = payload.startFrame;
355  _accumulationFrameNumber = 0;
356  _baseName = payload.baseName;
357  auto &frameBuffer = _api->getEngine().getFrameBuffer();
358  const auto size = frameBuffer.getSize();
359  frameBuffer.clear();
360  const size_t nbFrames = _exportFramesToDiskPayload.endFrame -
361  _exportFramesToDiskPayload.startFrame;
362  PLUGIN_INFO(
363  "----------------------------------------------------------------------"
364  "----------");
365  PLUGIN_INFO("Movie settings :");
366  PLUGIN_INFO("- Samples per pixel : " +
367  std::to_string(payload.spp));
368  PLUGIN_INFO("- Frame size : " + std::to_string(size.x) +
369  "x" + std::to_string(size.y));
370  PLUGIN_INFO("- Export folder : " + payload.path);
371  PLUGIN_INFO("- Export intermediate frames : " +
372  std::string(payload.exportIntermediateFrames ? "Yes" : "No"));
373  PLUGIN_INFO("- Start frame : " +
374  std::to_string(payload.startFrame));
375  PLUGIN_INFO(
376  "- End frame : " << std::to_string(payload.endFrame));
377  PLUGIN_INFO("- Frame base name : " << payload.baseName);
378  PLUGIN_INFO("- Number of frames : " << std::to_string(nbFrames));
379  PLUGIN_INFO(
380  "----------------------------------------------------------------------"
381  "----------");
382 }

384 FrameExportProgress MediaMakerPlugin::_getFrameExportProgress()
385 {
386  FrameExportProgress result;
387  double percentage = 1.f;
388  const size_t nbFrames =
389  _exportFramesToDiskPayload.cameraInformation.size() /
390  CAMERA_DEFINITION_SIZE;
391  const size_t totalNumberOfFrames =
392  nbFrames * _exportFramesToDiskPayload.spp;
393 
394  if (totalNumberOfFrames != 0)
395  {
396  const double currentProgress =
397  _frameNumber * _exportFramesToDiskPayload.spp +
398  _accumulationFrameNumber;
399  percentage = currentProgress / double(totalNumberOfFrames);
400  }
401  result.progress = percentage;
402  result.done = !_exportFramesToDiskDirty;
403  PLUGIN_DEBUG("Percentage = " << result.progress << ", Done = "
404  << (result.done ? "True" : "False"));
405  return result;
406 }
