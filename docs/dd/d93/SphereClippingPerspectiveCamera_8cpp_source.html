47 void SphereClippingPerspectiveCamera::commit()
48 {
49  Camera::commit();
50 
51  fovy = getParamf(CAMERA_PROPERTY_FIELD_OF_VIEW.name.c_str(), DEFAULT_CAMERA_FIELD_OF_VIEW);
52  aspect = getParamf(CAMERA_PROPERTY_ASPECT_RATIO.name.c_str(), DEFAULT_CAMERA_ASPECT_RATIO);
53  apertureRadius = getParamf(CAMERA_PROPERTY_APERTURE_RADIUS.name.c_str(), DEFAULT_CAMERA_APERTURE_RADIUS);
54  focalDistance = getParamf(CAMERA_PROPERTY_FOCAL_DISTANCE.name.c_str(), DEFAULT_CAMERA_FOCAL_DISTANCE);
55  stereo = getParam(CAMERA_PROPERTY_STEREO.name.c_str(), static_cast<int>(DEFAULT_CAMERA_STEREO));
56  interpupillaryDistance =
57  getParamf(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str(), DEFAULT_CAMERA_INTERPUPILLARY_DISTANCE);
58  enableClippingPlanes =
59  getParam(CAMERA_PROPERTY_ENABLE_CLIPPING_PLANES.name.c_str(), DEFAULT_CAMERA_ENABLE_CLIPPING_PLANES);
60  clipPlanes = enableClippingPlanes ? getParamData(CAMERA_PROPERTY_CLIPPING_PLANES, nullptr) : nullptr;
61  useHardwareRandomizer = getParam(COMMON_PROPERTY_USE_HARDWARE_RANDOMIZER.name.c_str(),
62  static_cast<int>(DEFAULT_COMMON_USE_HARDWARE_RANDOMIZER));
63 
64  dir = normalize(dir);
65  ::ospray::vec3f dir_du = normalize(cross(dir, up));
66  ::ospray::vec3f dir_dv = cross(dir_du, dir);
67 
68  ::ospray::vec3f org = pos;
69  const ::ospray::vec3f ipd_offset = 0.5f * interpupillaryDistance * dir_du;
70 
71  if (stereo)
72  {
73  auto bufferTarget = getParamString(CAMERA_PROPERTY_BUFFER_TARGET);
74  if (bufferTarget.length() == 2)
75  {
76  if (bufferTarget.at(1) == 'L')
77  org -= ipd_offset;
78  if (bufferTarget.at(1) == 'R')
79  org += ipd_offset;
80  }
81  }
82 
83  float imgPlane_size_y = 2.f * tanf(::ospray::deg2rad(0.5f * fovy));
84  float imgPlane_size_x = imgPlane_size_y * aspect;
85 
86  dir_du *= imgPlane_size_x;
87  dir_dv *= imgPlane_size_y;
88 
89  ::ospray::vec3f dir_00 = dir - 0.5f * dir_du - 0.5f * dir_dv;
90 
91  float scaledAperture = 0.f;
92  // prescale to focal plane
93  if (apertureRadius > 0.f)
94  {
95  dir_du *= focalDistance;
96  dir_dv *= focalDistance;
97  dir_00 *= focalDistance;
98  scaledAperture = apertureRadius / imgPlane_size_x;
99  }
100 
101  const auto clipPlaneData = clipPlanes ? clipPlanes->data : nullptr;
102  const size_t numClipPlanes = clipPlanes ? clipPlanes->numItems : 0;
103 
104  ::ispc::SphereClippingPerspectiveCamera_set(getIE(), (const ::ispc::vec3f&)org, (const ::ispc::vec3f&)dir_00,
105  (const ::ispc::vec3f&)dir_du, (const ::ispc::vec3f&)dir_dv,
106  scaledAperture, aspect, (const ::ispc::vec3f&)ipd_offset,
107  (const ::ispc::vec4f*)clipPlaneData, numClipPlanes,
108  useHardwareRandomizer);
109 }
