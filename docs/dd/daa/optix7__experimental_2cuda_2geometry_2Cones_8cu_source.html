31 namespace core
32 {
33 extern "C" __global__ void __intersection__cone()
34 {
35  const GeometryData::Cone* cone =
36  reinterpret_cast<GeometryData::Cone*>(optixGetSbtDataPointer());
37 
38  float3 v0 = cone->center;
39  float3 v1 = cone->up;
40  float radius0 = cone->centerRadius;
41  float radius1 = cone->upRadius;
42 
43  const float3 ray_orig = optixGetWorldRayOrigin();
44  const float3 ray_dir = optixGetWorldRayDirection();
45  const float ray_tmin = optixGetRayTmin();
46  const float ray_tmax = optixGetRayTmax();
47 
48  if (radius0 < radius1)
49  {
50  // swap radii and positions, so radius0 and v0 are always at the bottom
51  const float tmpRadius = radius1;
52  radius1 = radius0;
53  radius0 = tmpRadius;
54 
55  const float3 tmpPos = v1;
56  v1 = v0;
57  v0 = tmpPos;
58  }
59 
60  const float3 upVector = v1 - v0;
61  const float upLength = length(upVector);
62 
63  // Compute the height of the full cone, in order to obtain its vertex
64  const float deltaRadius = radius0 - radius1;
65  const float tanA = deltaRadius / upLength;
66  const float coneHeight = radius0 / tanA;
67  const float squareTanA = tanA * tanA;
68  const float div = sqrtf(1.f + squareTanA);
69  if (div == 0.f)
70  return;
71  const float cosA = 1.f / div;
72 
73  const float3 V = v0 + normalize(upVector) * coneHeight;
74  const float3 v = normalize(v0 - V);
75 
76  // Normal of the plane P determined by V and ray
77  float3 n = normalize(cross(ray_dir, V - ray_orig));
78  const float dotNV = dot(n, v);
79  if (dotNV > 0.f)
80  n = -n;
81 
82  const float squareCosTheta = 1.f - dotNV * dotNV;
83  const float cosTheta = sqrtf(squareCosTheta);
84  if (cosTheta < cosA)
85  return; // no intersection
86 
87  if (squareCosTheta == 0.f)
88  return;
89 
90  const float squareTanTheta = (1.f - squareCosTheta) / squareCosTheta;
91  const float tanTheta = sqrtf(squareTanTheta);
92 
93  // Compute u-v-w coordinate system
94  const float3 u = normalize(cross(v, n));
95  const float3 w = normalize(cross(u, v));
96 
97  // Circle intersection of cone with plane P
98  const float3 uComponent = sqrtf(squareTanA - squareTanTheta) * u;
99  const float3 vwComponent = v + tanTheta * w;
100  const float3 delta1 = vwComponent + uComponent;
101  const float3 delta2 = vwComponent - uComponent;
102  const float3 rayApex = V - ray_orig;
103 
104  const float3 normal1 = cross(ray_dir, delta1);
105  const float length1 = length(normal1);
106 
107  if (length1 == 0.f)
108  return;
109 
110  const float r1 = dot(cross(rayApex, delta1), normal1) / (length1 * length1);
111 
112  const float3 normal2 = cross(ray_dir, delta2);
113  const float length2 = length(normal2);
114 
115  if (length2 == 0.f)
116  return;
117 
118  const float r2 = dot(cross(rayApex, delta2), normal2) / (length2 * length2);
119 
120  float t_in = r1;
121  float t_out = r2;
122  if (r2 > 0.f)
123  {
124  if (r1 > 0.f)
125  {
126  if (r1 > r2)
127  {
128  t_in = r2;
129  t_out = r1;
130  }
131  }
132  else
133  t_in = r2;
134  }
135 
136  if (t_in > ray_tmin && t_in < ray_tmax)
137  {
138  const float3 p1 = ray_orig + t_in * ray_dir;
139  // consider only the parts within the extents of the truncated cone
140  if (dot(p1 - v1, v) > 0.f && dot(p1 - v0, v) < 0.f)
141  {
142  const float3 surfaceVec = normalize(p1 - V);
143  const float3 normal = cross(cross(v, surfaceVec), surfaceVec);
144  optixReportIntersection(t_in, 0, float3_as_uints(normal),
145  __float_as_uint(r1));
146  return;
147  }
148  }
149  if (t_out > ray_tmin && t_out < ray_tmax)
150  {
151  const float3 p2 = ray_orig + t_out * ray_dir;
152  // consider only the parts within the extents of the truncated cone
153  if (dot(p2 - v1, v) > 0.f && dot(p2 - v0, v) < 0.f)
154  {
155  const float3 surfaceVec = normalize(p2 - V);
156  const float3 normal = cross(cross(v, surfaceVec), surfaceVec);
157  optixReportIntersection(t_out, 0, float3_as_uints(normal),
158  __float_as_uint(r2));
159  return;
160  }
161  }
162 }
163 } // namespace core
