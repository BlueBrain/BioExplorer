78 void* VoltageSimulationHandler::getFrameData(const uint32_t frame)
79 {
80  const auto boundedFrame = _startFrame + _getBoundedFrame(frame);
81 
82  if (!_currentFrameFuture.valid() && _currentFrame != boundedFrame)
83  _triggerLoading(boundedFrame);
84 
85  if (!_makeFrameReady(boundedFrame))
86  return nullptr;
87 
88  return _frameData.data();
89 }

91 void VoltageSimulationHandler::_triggerLoading(const uint32_t frame)
92 {
93  float timestamp = frame * _dt;
94  timestamp = std::min(static_cast<float>(_nbFrames), timestamp);
95 
96  if (_currentFrameFuture.valid())
97  _currentFrameFuture.wait();
98 
99  _ready = false;
100  _currentFrameFuture = _compartmentReport->loadFrame(timestamp);
101 }
102 
103 bool VoltageSimulationHandler::_isFrameLoaded() const
104 {
105  if (!_currentFrameFuture.valid())
106  return false;
107 
108  if (_synchronousMode)
109  {
110  _currentFrameFuture.wait();
111  return true;
112  }
113 
114  return _currentFrameFuture.wait_for(std::chrono::milliseconds(0)) == std::future_status::ready;
115 }
116 
117 bool VoltageSimulationHandler::_makeFrameReady(const uint32_t frame)
118 {
119  if (_isFrameLoaded())
120  {
121  try
122  {
123  _frameData = std::move(*_currentFrameFuture.get().data);
124  }
125  catch (const std::exception& e)
126  {
127  PLUGIN_ERROR("Error loading simulation frame " << frame << ": " << e.what());
128  return false;
129  }
130  _currentFrame = frame;
131  _ready = true;
132  }
133  return true;
134 }
