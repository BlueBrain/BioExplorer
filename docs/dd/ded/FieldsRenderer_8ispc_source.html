27float treeWalker(const uniform FieldsRenderer* uniform self, const vec3f& point,
28 const varying float distance, const varying float cutoff,
29 const varying int32 index = 0)
30{
31 const uint32 begin = self->userData[self->startIndices + index * 2];
32 const uint32 end = self->userData[self->startIndices + index * 2 + 1];
33 const uint32 idxData = self->startData + index * 4;
34
35 if (idxData >= self->userDataSize)
36 return 0.f;
37
38 if (begin == 0 && end == 0)
39 return self->userData[idxData + 3] / (distance * distance);
40
41 float voxelValue = 0.f;
42 for (unsigned int32 childIndex = begin; childIndex <= end; ++childIndex)
43 {
44 const vec3f childCenter =
45 make_vec3f(self->userData[idxData], self->userData[idxData + 1],
46 self->userData[idxData + 2]);
47 const vec3f delta = point - childCenter;
48
49 float d =
50 sqrt(delta.x * delta.x + delta.y * delta.y + delta.z * delta.z);
51
52 // Minimum distance
53 if (d < self->spacing.x / 2.f)
54 d = self->spacing.x / 2.f;
55
56 if (d >= cutoff)
57 {
58 // Child is further than the cutoff distance, no need to evaluate
59 // events in the child node, we take the precomputed value of node
60 // instead
61 voxelValue += self->userData[idxData + 3] / (d * d);
62 }
63 else
64 // Dive into the child node and compute its contents
65 voxelValue += treeWalker(self, point, d, cutoff / 2.f, childIndex);
66 }
67 return voxelValue;
68}

70/**
71 Composes source and destination colors according to specified alpha
72 correction
73 @param src Source RGBA color
74 @param dst Returned destination RGBA color
75 @param alphaCorrection Alpha correction used to blend colors
76*/
77inline void composite(const varying vec4f& src, varying vec4f& dst,
78 const varying float alphaCorrection)
79{
80 const float alpha =
81 1.f - pow(1.f - min(src.w, 1.f - 1.f / 256.f), alphaCorrection);
82 const float a = alpha * (1.f - dst.w);
83 dst.x = dst.x + src.x * a;
84 dst.y = dst.y + src.y * a;
85 dst.z = dst.z + src.z * a;
86 dst.w += (alpha * (1.f - dst.w));
87}

89inline vec3f FieldsRenderer_shadeRay(const uniform FieldsRenderer* uniform self,
90 varying ScreenSample& sample)
91{
92 const Ray ray = sample.ray;
93
94 if (sample.sampleID.z == 0)
95 {
96 sample.alpha = 0.f;
97 sample.z = inf;
98 }
99
100 const uniform box3f box =
101 make_box3f(self->offset, self->offset + self->size * self->spacing);
102
103 float t0, t1;
104 intersectBox(ray, box, t0, t1);
105 t0 = max(self->minRayStep, t0);
106
107 const vec3f s = self->size * self->spacing;
108 const float diag = max(max(s.x, s.y), s.z);
109 const float step = max(self->minRayStep, diag / self->nbRaySteps);
110
111 float t = t0 + (sample.sampleID.z % self->nbRayRefinementSteps) * step /
112 self->nbRayRefinementSteps;
113
114 vec4f finalColor = make_vec4f(0.f);
115 bool firstVisibleVoxel = true;
116
117 while (finalColor.w < 0.9f && t < t1)
118 {
119 const vec3f p = ray.org + t * ray.dir;
120 const vec3f point = (p - self->offset) / self->spacing;
121
122 float value = treeWalker(self, point, self->distance, self->cutoff);
123
124 value = clamp(value, self->transferFunction->valueRange.x,
125 self->transferFunction->valueRange.y);
126
127 const float sampleOpacity =
128 self->transferFunction->getOpacityForValue(self->transferFunction,
129 value);
130 if (sampleOpacity > 0.f)
131 {
132 const vec3f sampleColor =
133 self->transferFunction->getColorForValue(self->transferFunction,
134 value);
135 const float alpha = self->alphaCorrection * sampleOpacity;
136 composite(make_vec4f(sampleColor, alpha), finalColor, 1.0f);
137
138 if (firstVisibleVoxel)
139 sample.z = min(t, sample.z);
140
141 firstVisibleVoxel = false;
142 }
143
144 t += step;
145 }
146
147 sample.alpha = finalColor.w;
148 return clamp(make_vec3f(finalColor) * self->exposure, make_vec3f(0.f),
149 make_vec3f(1.f));
150}

152void FieldsRenderer_renderSample(uniform Renderer* uniform _self,
153 void* uniform perFrameData,
154 varying ScreenSample& sample)
155{
156 uniform FieldsRenderer* uniform self =
157 (uniform FieldsRenderer * uniform) _self;
158 sample.ray.time = self->timestamp;
159 sample.rgb = FieldsRenderer_shadeRay(self, sample);
160}

171export void FieldsRenderer_set(
172 void* uniform _self, void* uniform bgMaterial,
173 uniform float* uniform userData, const uniform uint64 userDataSize,
174 const uniform int& randomNumber, const uniform float& timestamp,
175 const uniform int& spp, void** uniform lights,
176 const uniform int32 numLights, const uniform float& minRayStep,
177 const uniform int32& nbRaySteps, const uniform int32& nbRayRefinementSteps,
178 const uniform float& exposure, const uniform bool& useHardwareRandomizer,
179 const uniform float& cutoff, const uniform float& alphaCorrection)
180{
181 uniform FieldsRenderer* uniform self =
182 (uniform FieldsRenderer * uniform) _self;
183
184 // OSPRay
185 self->super.spp = spp;
186
187 self->lights = (const uniform Light* uniform* uniform)lights;
188 self->numLights = numLights;
189
190 // Bio explorer
191 self->bgMaterial = (uniform AdvancedMaterial * uniform) bgMaterial;
192
193 self->timestamp = timestamp;
194 self->exposure = exposure;
195 self->useHardwareRandomizer = useHardwareRandomizer;
196
197 self->minRayStep = minRayStep;
198 self->nbRaySteps = nbRaySteps;
199 self->nbRayRefinementSteps = nbRayRefinementSteps;
200 self->alphaCorrection = alphaCorrection;
201
202 self->cutoff = cutoff;
203
204 self->userData = userData;
205 self->userDataSize = userDataSize;
206
207 self->offset = (userData ? make_vec3f(userData[0], userData[1], userData[2])
208 : make_vec3f(0.f));
209 self->spacing =
210 (userData ? make_vec3f(userData[3], userData[4], userData[5])
211 : make_vec3f(0.f));
212 self->size = (userData ? make_vec3f(userData[6], userData[7], userData[8])
213 : make_vec3f(0.f));
214 self->distance = (userData ? userData[9] * 5.f : 0.f); // Octree size * 5
215 self->startIndices = 11;
216 self->startData = (userData ? self->startIndices + userData[10] : 0);
217}
