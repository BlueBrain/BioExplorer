30 float treeWalker(const uniform FieldsRenderer* uniform self, const vec3f& point, const varying float distance,
31  const varying float cutoff, const varying int32 index = 0)
32 {
33  const uint64 begin = self->userData[self->startIndices + index * 2];
34  const uint64 end = self->userData[self->startIndices + index * 2 + 1];
35  const uint64 idxData = self->startData + index * 4;
36 
37  if (idxData >= self->userDataSize)
38  return 0.f;
39 
40  if (begin == 0 && end == 0)
41  // Leaf
42  return self->userData[idxData + 3] / (distance * distance);
43 
44  float voxelValue = 0.f;
45  for (unsigned int64 childIndex = begin; childIndex <= end; ++childIndex)
46  {
47  const uint64 idx = self->startData + childIndex * 4;
48  const vec3f childPosition = make_vec3f(self->userData[idx], self->userData[idx + 1], self->userData[idx + 2]);
49  const vec3f delta = point - childPosition;
50 
51  const float d = sqrt(delta.x * delta.x + delta.y * delta.y + delta.z * delta.z);
52 
53  if (d >= cutoff)
54  {
55  // Child is further than the cutoff distance, no need to evaluate
56  // events in the child node, we take the precomputed value of node
57  // instead
58  voxelValue += self->userData[idx + 3] / (d * d);
59  }
60  else
61  // Dive into the child node and compute its contents
62  voxelValue += treeWalker(self, point, d, cutoff / 2.f, childIndex);
63  }
64  return voxelValue;
65 }

67 /**
68  Composes source and destination colors according to specified alpha
69  correction
70  @param src Source RGBA color
71  @param dst Returned destination RGBA color
72  @param alphaCorrection Alpha correction used to blend colors
73 */
74 inline void composite(const varying vec4f& src, varying vec4f& dst, const varying float alphaCorrection)
75 {
76  const float alpha = 1.f - pow(1.f - min(src.w, 1.f - 1.f / 256.f), alphaCorrection);
77  const float a = alpha * (1.f - dst.w);
78  dst.x = dst.x + src.x * a;
79  dst.y = dst.y + src.y * a;
80  dst.z = dst.z + src.z * a;
81  dst.w += (alpha * (1.f - dst.w));
82 }

84 inline vec3f FieldsRenderer_shadeRay(const uniform FieldsRenderer* uniform self, varying ScreenSample& sample)
85 {
86  const Ray ray = sample.ray;
87 
88  if (sample.sampleID.z == 0)
89  {
90  sample.alpha = 0.f;
91  sample.z = inf;
92  }
93 
94  const uniform box3f box = make_box3f(self->offset, self->offset + self->size * self->spacing);
95 
96  float t0, t1;
97  intersectBox(ray, box, t0, t1);
98  t0 = max(self->minRayStep, t0);
99 
100  const vec3f s = self->size * self->spacing;
101  const float diag = min(min(s.x, s.y), s.z);
102  const float step = max(self->minRayStep, diag / self->nbRaySteps);
103  const float sampleStep = sample.sampleID.z % self->nbRayRefinementSteps;
104 
105  float t = t0 + sampleStep * step / self->nbRayRefinementSteps;
106 
107  vec4f finalColor = make_vec4f(0.f);
108  while (finalColor.w < 0.95f && t < t1)
109  {
110  const vec3f p = ray.org + t * ray.dir;
111  const vec3f point = (p - self->offset) / self->spacing;
112 
113  float value = treeWalker(self, point, self->distance, self->cutoff);
114 
115  value = clamp(value, self->transferFunction->valueRange.x, self->transferFunction->valueRange.y);
116 
117  const float sampleOpacity = self->transferFunction->getOpacityForValue(self->transferFunction, value);
118  if (sampleOpacity > 0.f)
119  {
120  const vec3f sampleColor = self->transferFunction->getColorForValue(self->transferFunction, value);
121  finalColor =
122  make_vec4f(make_vec3f(finalColor) * finalColor.w +
123  make_vec3f(sampleColor) * sampleOpacity * self->alphaCorrection * (1.f - finalColor.w),
124  finalColor.w + sampleOpacity * self->alphaCorrection);
125  }
126 
127  t += step;
128  }
129 
130  sample.alpha = finalColor.w;
131  return clamp(make_vec3f(finalColor) * self->exposure, make_vec3f(0.f), make_vec3f(1.f));
132 }
