174 void Molecule::_computeReqSetOffset()
175 {
176  for (auto& sequence : _residueSequenceMap)
177  {
178  std::string physicalReqSeq;
179  size_t firstReqSeq;
180  size_t previousReqSeq;
181  bool initialized{false};
182  for (const auto& atom : _atomMap)
183  {
184  if (atom.second.chainId != sequence.first)
185  continue;
186 
187  if (!initialized)
188  {
189  firstReqSeq = atom.second.reqSeq;
190  previousReqSeq = firstReqSeq - 1;
191  initialized = true;
192  }
193 
194  if (previousReqSeq != atom.second.reqSeq)
195  {
196  if (atom.second.reqSeq != previousReqSeq + 1)
197  break;
198 
199  physicalReqSeq +=
200  aminoAcidMap.find(atom.second.resName)->second.shortName;
201 
202  if (physicalReqSeq.length() > 10)
203  break;
204  }
205 
206  previousReqSeq = atom.second.reqSeq;
207  }
208 
209  std::string theoreticalReqSeq;
210  for (const auto& aa : sequence.second.resNames)
211  theoreticalReqSeq += aminoAcidMap.find(aa)->second.shortName;
212 
213  sequence.second.offset =
214  theoreticalReqSeq.find(physicalReqSeq) - firstReqSeq;
215  PLUGIN_INFO("Sequence " << sequence.first
216  << " offset: " << sequence.second.offset
217  << "Theoretical: " << theoreticalReqSeq
218  << "Physical : " << physicalReqSeq);
219  }
220 }

222 const StringMap Molecule::getSequencesAsString() const
223 {
224  StringMap sequencesAsStrings;
225  for (const auto& sequence : _residueSequenceMap)
226  {
227  std::string shortSequence = std::to_string(_aminoAcidRange.x) + "," +
228  std::to_string(_aminoAcidRange.y) + ",";
229  for (const auto& resName : sequence.second.resNames)
230  shortSequence += aminoAcidMap[resName].shortName;
231 
232  sequencesAsStrings[sequence.first] = shortSequence;
233  PLUGIN_DEBUG(sequence.first << " (" << sequence.second.resNames.size()
234  << "): " << shortSequence);
235  }
236  return sequencesAsStrings;

239 void Molecule::_buildAtomicStruture(const ProteinRepresentation representation,
240  const float atomRadiusMultiplier,
241  const bool surface, const bool loadBonds,
242  Model& model)
243 {
244  // Atoms
245  for (const auto& atom : _atomMap)
246  {
247  // Material
248  auto material =
249  model.createMaterial(atom.first, std::to_string(atom.first));
250 
251  RGBColorDetails rgb{255, 255, 255};
252  const auto it = atomColorMap.find(atom.second.element);
253  if (it != atomColorMap.end())
254  rgb = (*it).second;
255 
256  brayns::PropertyMap props;
257  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
258  static_cast<int>(MaterialShadingMode::basic)});
259  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
260  if (surface)
261  props.setProperty(
262  {MATERIAL_PROPERTY_CHAMELEON_MODE,
263  static_cast<int>(MaterialChameleonMode::emitter)});
264  else
265  props.setProperty(
266  {MATERIAL_PROPERTY_CHAMELEON_MODE,
267  static_cast<int>(
268  MaterialChameleonMode::undefined_chameleon_mode)});
269 
270  props.setProperty({MATERIAL_PROPERTY_NODE_ID, static_cast<int>(_uuid)});
271 
272  material->setDiffuseColor(
273  {rgb.r / 255.f, rgb.g / 255.f, rgb.b / 255.f});
274  material->updateProperties(props);
275 
276  // Geometry
277  model.addSphere(atom.first,
278  {atom.second.position,
279  atom.second.radius * atomRadiusMultiplier});
280  }
281 
282  // Bonds
283  if (loadBonds)
284  {
285  PLUGIN_INFO("Building " << _bondsMap.size() << " bonds...");
286  for (const auto& bond : _bondsMap)
287  {
288  const auto& atomSrc = _atomMap.find(bond.first)->second;
289  for (const auto bondedAtom : bond.second)
290  {
291  const auto& atomDst = _atomMap.find(bondedAtom)->second;
292 
293  const auto center = (atomSrc.position + atomDst.position) / 2.f;
294 
295  model.addCylinder(bond.first,
296  {atomSrc.position, center,
297  atomRadiusMultiplier * BOND_RADIUS});
298 
299  model.addCylinder(bondedAtom,
300  {atomDst.position, center,
301  atomRadiusMultiplier * BOND_RADIUS});
302  }
303  }
304  }
305 
306  // Sticks
307  if (representation == ProteinRepresentation::atoms_and_sticks)
308  {
309  PLUGIN_INFO("Building sticks (" << _atomMap.size() << " atoms)...");
310 #pragma omp parallel for
311  for (const auto& atom1 : _atomMap)
312  for (const auto& atom2 : _atomMap)
313  if (atom1.first != atom2.first &&
314  atom1.second.reqSeq == atom2.second.reqSeq &&
315  atom1.second.chainId == atom2.second.chainId)
316  {
317  const auto stick =
318  atom2.second.position - atom1.second.position;
319 
320  if (length(stick) < DEFAULT_STICK_DISTANCE)
321  {
322  const auto center =
323  (atom2.second.position + atom1.second.position) /
324  2.f;
325 #pragma omp critical
326  model.addCylinder(atom1.first,
327  {atom1.second.position, center,
328  atomRadiusMultiplier * BOND_RADIUS});
329  }
330  }
331  }
332 }

334 void Molecule::_buildModel(const std::string& assemblyName,
335  const std::string& name, const std::string& title,
336  const std::string& header,
337  const ProteinRepresentation& representation,
338  const float atomRadiusMultiplier,
339  const bool loadBonds)
340 {
341  PLUGIN_INFO("Building protein " << name << "...");
342 
343  // Metadata
344  ModelMetadata metadata;
345  metadata[METADATA_ASSEMBLY] = assemblyName;
346  metadata[METADATA_TITLE] = title;
347  metadata[METADATA_HEADER] = header;
348  metadata[METADATA_ATOMS] = std::to_string(_atomMap.size());
349  metadata[METADATA_BONDS] = std::to_string(_bondsMap.size());
350  metadata[METADATA_AA_RANGE] = std::to_string(_aminoAcidRange.x) + ":" +
351  std::to_string(_aminoAcidRange.y);
352 
353  const auto& size = _bounds.getSize();
354  metadata[METADATA_SIZE] = std::to_string(size.x) + ", " +
355  std::to_string(size.y) + ", " +
356  std::to_string(size.z) + " angstroms";
357 
358  for (const auto& sequence : getSequencesAsString())
359  metadata[METADATA_AA_SEQUENCE + sequence.first] =
360  "[" + std::to_string(sequence.second.size()) + "] " +
361  sequence.second;
362 
363  switch (representation)
364  {
365  case ProteinRepresentation::atoms:
366  case ProteinRepresentation::atoms_and_sticks:
367  {
368  auto model = _scene.createModel();
369  _buildAtomicStruture(representation, atomRadiusMultiplier, false,
370  loadBonds, *model);
371  _modelDescriptor =
372  std::make_shared<ModelDescriptor>(std::move(model), name, header,
373  metadata);
374  break;
375  }
376 #ifdef USE_CGAL
377  case ProteinRepresentation::surface:
378  case ProteinRepresentation::union_of_balls:
379  {
380  // Surface
381  Vector4fs pointCloud;
382  const size_t materialId{0};
383  for (const auto& atom : _atomMap)
384  {
385  if (atom.first % std::max(1, int(atomRadiusMultiplier)) != 0)
386  continue;
387  pointCloud.push_back({atom.second.position.x,
388  atom.second.position.y,
389  atom.second.position.z,
390  atom.second.radius * atomRadiusMultiplier});
391  }
392 
393  SurfaceMesher sm(_uuid);
394  if (representation == ProteinRepresentation::union_of_balls)
395  _modelDescriptor =
396  sm.generateUnionOfBalls(_scene, name, pointCloud);
397  else
398  _modelDescriptor = sm.generateSurface(_scene, name, pointCloud);
399  _modelDescriptor->setMetadata(metadata);
400 
401  Model& model = _modelDescriptor->getModel();
402  _buildAtomicStruture(representation, atomRadiusMultiplier * 2.f, true,
403  loadBonds, model);
404  break;
405  }
406  case ProteinRepresentation::contour:
407  {
408  auto model = _scene.createModel();
409  PointCloud pointCloud;
410  for (const auto& atom : _atomMap)
411  pointCloud[0].push_back(
412  {atom.second.position.x, atom.second.position.y,
413  atom.second.position.z,
414  atom.second.radius * atomRadiusMultiplier});
415 
416  PointCloudMesher pcm;
417  pcm.toConvexHull(*model, pointCloud);
418  _modelDescriptor =
419  std::make_shared<ModelDescriptor>(std::move(model), name, header,
420  metadata);
421  break;
422  }
423 #else
424  case ProteinRepresentation::surface:
425  case ProteinRepresentation::union_of_balls:
426  case ProteinRepresentation::contour:
427  PLUGIN_THROW("CGAL is required to create surfaces");
428  break;
429 #endif
430  case ProteinRepresentation::debug:
431  {
432  auto model = _scene.createModel();
433  const size_t materialId = 0;
434  auto material = model->createMaterial(materialId, "Debug");
435  brayns::PropertyMap props;
436  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
437  static_cast<int>(MaterialShadingMode::basic)});
438  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
439  props.setProperty(
440  {MATERIAL_PROPERTY_CHAMELEON_MODE,
441  static_cast<int>(
442  MaterialChameleonMode::undefined_chameleon_mode)});
443  material->setDiffuseColor({1.f, 1.f, 1.f});
444  material->updateProperties(props);
445 
446  brayns::Boxf box;
447  for (const auto& atom : _atomMap)
448  box.merge({atom.second.position.x, atom.second.position.y,
449  atom.second.position.z});
450 
451  const auto halfSize = box.getSize() * 0.5f;
452  const auto center = box.getCenter();
453 
454  const brayns::Vector3f a = {0.f, 0.f, center.z + halfSize.z};
455  const brayns::Vector3f b = {0.f, 0.f, center.z - halfSize.z * 0.5f};
456  const brayns::Vector3f c = {0.f, 0.f, center.z - halfSize.z * 0.51f};
457  const brayns::Vector3f d = {0.f, 0.f, center.z - halfSize.z};
458 
459  model->addSphere(materialId, {a, atomRadiusMultiplier * 0.2f});
460  model->addCylinder(materialId, {a, b, atomRadiusMultiplier * 0.2f});
461  model->addCone(materialId, {b, c, atomRadiusMultiplier * 0.2f,
462  atomRadiusMultiplier});
463  model->addCone(materialId, {c, d, atomRadiusMultiplier, 0.f});
464  _modelDescriptor =
465  std::make_shared<ModelDescriptor>(std::move(model), name, header,
466  metadata);
467  break;
468  }
469  }
470 
471  PLUGIN_INFO("Protein model successfully built");
472 
473  PLUGIN_INFO("---=== Protein ===--- ");
474  PLUGIN_INFO("Assembly name : " << assemblyName);
475  PLUGIN_INFO("Name : " << name);
476  PLUGIN_INFO("Atom Radius multiplier: " << atomRadiusMultiplier);
477  PLUGIN_INFO("Number of atoms : " << _atomMap.size());
478  PLUGIN_INFO("Number of bonds : " << _bondsMap.size());
479 
480  if (_modelDescriptor &&
481  !GeneralSettings::getInstance()->getModelVisibilityOnCreation())
482  _modelDescriptor->setVisible(false);
483 }

485 void Molecule::_readAtom(const std::string& line, const bool loadHydrogen)
486 {
487  // --------------------------------------------------------------------
488  // COLUMNS DATA TYPE FIELD DEFINITION
489  // --------------------------------------------------------------------
490  // 1 - 6 Record name "ATOM "
491  // 7 - 11 Integer serial Atom serial number
492  // 13 - 16 Atom name Atom name
493  // 17 Character altLoc Alternate location indicator
494  // 18 - 20 Residue name resName Residue name
495  // 22 Character chainID Chain identifier
496  // 23 - 26 Integer resSeq Residue sequence number
497  // 27 AChar iCode Code for insertion of residues
498  // 31 - 38 Real(8.3) x Orthogonal coords for X in angstroms
499  // 39 - 46 Real(8.3) y Orthogonal coords for Y in Angstroms
500  // 47 - 54 Real(8.3) z Orthogonal coords for Z in Angstroms
501  // 55 - 60 Real(6.2) occupancy Occupancy
502  // 61 - 66 Real(6.2) tempFactor Temperature factor
503  // 77 - 78 LString(2) element Element symbol, right-justified
504  // 79 - 80 LString(2) charge Charge on the atom
505  // --------------------------------------------------------------------
506 
507  std::string s = line.substr(21, 1);
508  std::string chainId = trim(s);
509  if (!_loadChain(static_cast<size_t>(chainId[0] - 64)))
510  return;
511 
512  const size_t serial = static_cast<size_t>(atoi(line.substr(6, 5).c_str()));
513 
514  Atom atom;
515  atom.chainId = chainId;
516 
517  s = line.substr(12, 4);
518  atom.name = trim(s);
519 
520  s = line.substr(16, 1);
521  atom.altLoc = trim(s);
522 
523  s = line.substr(17, 3);
524  atom.resName = trim(s);
525 
526  _residues.insert(atom.resName);
527 
528  atom.reqSeq = static_cast<size_t>(atoi(line.substr(22, 4).c_str()));
529  _aminoAcidRange.x = std::min(atom.reqSeq, size_t(_aminoAcidRange.x));
530  _aminoAcidRange.y = std::max(atom.reqSeq, size_t(_aminoAcidRange.y));
531 
532  atom.iCode = line.substr(26, 1);
533 
534  atom.position.x = static_cast<float>(atof(line.substr(30, 8).c_str()));
535  atom.position.y = static_cast<float>(atof(line.substr(38, 8).c_str()));
536  atom.position.z = static_cast<float>(atof(line.substr(46, 8).c_str()));
537 
538  if (line.length() >= 60)
539  atom.occupancy = static_cast<float>(atof(line.substr(54, 6).c_str()));
540 
541  if (line.length() >= 66)
542  atom.tempFactor = static_cast<float>(atof(line.substr(60, 6).c_str()));
543 
544  if (line.length() >= 78)
545  {
546  s = line.substr(76, 2);
547  atom.element = trim(s);
548  if (s == "H" && !loadHydrogen)
549  return;
550  }
551 
552  if (line.length() >= 80)
553  {
554  s = line.substr(78, 2);
555  atom.charge = trim(s);
556  }
557 
558  // Convert position from amstrom (10e-10) to nanometers (10e-9)
559  atom.position = 0.1f * atom.position;
560 
561  // Bounds
562  _bounds.merge(atom.position);
563 
564  // Convert radius from picometers (10e-12) to nanometers (10e-9)
565  atom.radius = 0.001f * DEFAULT_ATOM_RADIUS;
566  auto it = atomicRadii.find(atom.element);
567  if (it != atomicRadii.end())
568  atom.radius = 0.001f * (*it).second;
569  else
570  {
571  it = atomicRadii.find(atom.name);
572  if (it != atomicRadii.end())
573  atom.radius = 0.001f * (*it).second;
574  else
575  PLUGIN_WARN("[" << atom.element << "]/[" << atom.name
576  << "] not found");
577  }
578 
579  _atomMap.insert(std::make_pair(serial, atom));
580 }

582 void Molecule::_readSequence(const std::string& line)
583 {
584  // -------------------------------------------------------------------------
585  // COLUMNS TYPE FIELD DEFINITION
586  // -------------------------------------------------------------------------
587  // 1 - 6 Record name "SEQRES"
588  // 8 - 10 Integer serNum Serial number of the SEQRES record for the
589  // current chain. Starts at 1 and increments
590  // by one each line. Reset to 1 for each
591  // chain.
592  // 12 Character chainID Chain identifier. This may be any single
593  // legal
594  // character, including a blank which is is
595  // used if there is only one chain
596  // 14 - 17 Integer numRes Number of residues in the chain. This
597  // value is
598  // repeated on every record.
599  // 20 - 22 String resName Residue name
600  // 24 - 26 ...
601  // -------------------------------------------------------------------------
602 
603  std::string s = line.substr(11, 1);
604 
605  ResidueSequence& sequence = _residueSequenceMap[s];
606  sequence.numRes = static_cast<size_t>(atoi(line.substr(13, 4).c_str()));
607 
608  for (size_t i = 19; i < line.length(); i += 4)
609  {
610  s = line.substr(i, 4);
611  s = trim(s);
612  if (!s.empty())
613  sequence.resNames.push_back(s);
614  }
615 }

617 void Molecule::_readConnect(const std::string& line)
618 {
619  // -------------------------------------------------------------------------
620  // COLUMNS TYPE FIELD DEFINITION
621  // -------------------------------------------------------------------------
622  // 1 - 6 Record name "CONECT"
623  // 7 - 11 Integer serial Atom serial number
624  // 12 - 16 Integer serial Serial number of bonded atom
625  // 17 - 21 Integer serial Serial number of bonded atom
626  // 22 - 26 Integer serial Serial number of bonded atom
627  // 27 - 31 Integer serial Serial number of bonded atom
628  // -------------------------------------------------------------------------
629 
630  const size_t serial = static_cast<size_t>(atoi(line.substr(6, 5).c_str()));
631 
632  if (_atomMap.find(serial) != _atomMap.end())
633  {
634  auto& bond = _bondsMap[serial];
635 
636  for (size_t i = 11; i < line.length(); i += 5)
637  {
638  std::string s = line.substr(i, 5);
639  s = trim(s);
640  if (!s.empty())

736 void Molecule::_setAminoAcidSequenceColorScheme(const Palette& palette)
737 {
738  if (palette.size() != 2)
739  PLUGIN_THROW("Invalid palette size. 2 colors are expected");
740 
741  size_t atomCount = 0;
742  for (const auto& sequence : _residueSequenceMap)
743  {
744  if (_selectedAminoAcidSequence.empty())
745  {
746  // Range based coloring
747  for (auto& atom : _atomMap)
748  {
749  bool selected = false;
750  for (const auto& range : _selectedAminoAcidRanges)
751  {
752  selected = (atom.second.reqSeq >= range.x &&
753  atom.second.reqSeq <= range.y);
754  if (selected)
755  break;
756  }
757  _setMaterialDiffuseColor(atom.first,
758  selected ? palette[1] : palette[0]);
759  }
760  }
761  else
762  {
763  // String based coloring
764  std::string shortSequence;
765  for (const auto& resName : sequence.second.resNames)
766  shortSequence += aminoAcidMap[resName].shortName;
767 
768  const auto sequencePosition =
769  shortSequence.find(_selectedAminoAcidSequence);
770  if (sequencePosition != -1)
771  {
772  PLUGIN_INFO(_selectedAminoAcidSequence
773  << " was found at position " << sequencePosition);
774  size_t minSeq = 1e6;
775  size_t maxSeq = 0;
776  for (auto& atom : _atomMap)
777  {
778  minSeq = std::min(minSeq, atom.second.reqSeq);
779  maxSeq = std::max(maxSeq, atom.second.reqSeq);
780  if (atom.second.reqSeq >= sequencePosition &&
781  atom.second.reqSeq <
782  sequencePosition +
783  _selectedAminoAcidSequence.length())
784  {
785  _setMaterialDiffuseColor(atom.first, palette[1]);
786  ++atomCount;
787  }
788  else
789  _setMaterialDiffuseColor(atom.first, palette[0]);
790  }
791  PLUGIN_DEBUG(atomCount << "[" << minSeq << "," << maxSeq
792  << "] atoms where colored");
793  }
794  else
795  PLUGIN_WARN(_selectedAminoAcidSequence << " was not found in "
796  << shortSequence);
797  }
798  }
799  PLUGIN_INFO("Applying Amino Acid Sequence color scheme ("
800  << (atomCount > 0 ? "2" : "1") << ")");
801 }
