176void Molecule::_computeReqSetOffset()
177{
178 for (auto& sequence : _residueSequenceMap)
179 {
180 std::string physicalReqSeq;
181 size_t firstReqSeq;
182 size_t previousReqSeq;
183 bool initialized{false};
184 for (const auto& atom : _atomMap)
185 {
186 if (atom.second.chainId != sequence.first)
187 continue;
188
189 if (!initialized)
190 {
191 firstReqSeq = atom.second.reqSeq;
192 previousReqSeq = firstReqSeq - 1;
193 initialized = true;
194 }
195
196 if (previousReqSeq != atom.second.reqSeq)
197 {
198 if (atom.second.reqSeq != previousReqSeq + 1)
199 break;
200
201 physicalReqSeq +=
202 aminoAcidMap.find(atom.second.resName)->second.shortName;
203
204 if (physicalReqSeq.length() > 10)
205 break;
206 }
207
208 previousReqSeq = atom.second.reqSeq;
209 }
210
211 std::string theoreticalReqSeq;
212 for (const auto& aa : sequence.second.resNames)
213 theoreticalReqSeq += aminoAcidMap.find(aa)->second.shortName;
214
215 sequence.second.offset =
216 theoreticalReqSeq.find(physicalReqSeq) - firstReqSeq;
217 PLUGIN_INFO("Sequence [" << sequence.first
218 << "], offset: " << sequence.second.offset
219 << ", Theoretical: " << theoreticalReqSeq
220 << ", Physical: " << physicalReqSeq);
221 }
222}

224const StringMap Molecule::getSequencesAsString() const
225{
226 StringMap sequencesAsStrings;
227 for (const auto& sequence : _residueSequenceMap)
228 {
229 std::string shortSequence = std::to_string(_aminoAcidRange.x) + "," +
230 std::to_string(_aminoAcidRange.y) + ",";
231 for (const auto& resName : sequence.second.resNames)
232 shortSequence += aminoAcidMap[resName].shortName;
233
234 sequencesAsStrings[sequence.first] = shortSequence;
235 PLUGIN_DEBUG(sequence.first << " (" << sequence.second.resNames.size()
236 << "): " << shortSequence);
237 }
238 return sequencesAsStrings;
239}

241void Molecule::_buildAtomicStruture(const ProteinRepresentation representation,
242 const float atomRadiusMultiplier,
243 const bool surface, const bool loadBonds,
244 Model& model)
245{
246 // Atoms
247 for (const auto& atom : _atomMap)
248 {
249 // Material
250 auto material =
251 model.createMaterial(atom.first, std::to_string(atom.first));
252
253 RGBColorDetails rgb{255, 255, 255};
254 const auto it = atomColorMap.find(atom.second.element);
255 if (it != atomColorMap.end())
256 rgb = (*it).second;
257
258 brayns::PropertyMap props;
259 props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
260 static_cast<int>(MaterialShadingMode::basic)});
261 props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
262 if (surface)
263 props.setProperty(
264 {MATERIAL_PROPERTY_CHAMELEON_MODE,
265 static_cast<int>(MaterialChameleonMode::emitter)});
266 else
267 props.setProperty(
268 {MATERIAL_PROPERTY_CHAMELEON_MODE,
269 static_cast<int>(
270 MaterialChameleonMode::undefined_chameleon_mode)});
271
272 props.setProperty({MATERIAL_PROPERTY_NODE_ID, static_cast<int>(_uuid)});
273
274 material->setDiffuseColor(
275 {rgb.r / 255.f, rgb.g / 255.f, rgb.b / 255.f});
276 material->updateProperties(props);
277
278 // Geometry
279 model.addSphere(atom.first,
280 {atom.second.position,
281 atom.second.radius * atomRadiusMultiplier});
282 }
283
284 // Bonds
285 if (loadBonds)
286 {
287 PLUGIN_INFO("Building " << _bondsMap.size() << " bonds...");
288 for (const auto& bond : _bondsMap)
289 {
290 const auto& atomSrc = _atomMap.find(bond.first)->second;
291 for (const auto bondedAtom : bond.second)
292 {
293 const auto& atomDst = _atomMap.find(bondedAtom)->second;
294
295 const auto center = (atomSrc.position + atomDst.position) / 2.f;
296
297 model.addCylinder(bond.first,
298 {atomSrc.position, center,
299 atomRadiusMultiplier * BOND_RADIUS});
300
301 model.addCylinder(bondedAtom,
302 {atomDst.position, center,
303 atomRadiusMultiplier * BOND_RADIUS});
304 }
305 }
306 }
307
308 // Sticks
309 if (representation == ProteinRepresentation::atoms_and_sticks)
310 {
311 PLUGIN_INFO("Building sticks (" << _atomMap.size() << " atoms)...");
312 auto it1 = _atomMap.begin();
313 while (it1 != _atomMap.end())
314 {
315 const auto atom1 = (*it1);
316 auto it2 = it1;
317 ++it2;
318 while ((*it2).second.reqSeq == atom1.second.reqSeq)
319 {
320 const auto stick =
321 (*it2).second.position - atom1.second.position;
322 if (length(stick) < DEFAULT_STICK_DISTANCE)
323 {
324 const auto center =
325 ((*it2).second.position + atom1.second.position) / 2.f;
326 model.addCylinder(atom1.first,
327 {atom1.second.position, center,
328 atomRadiusMultiplier * BOND_RADIUS});
329 model.addCylinder((*it2).first,
330 {(*it2).second.position, center,
331 atomRadiusMultiplier * BOND_RADIUS});
332 }
333 ++it2;
334 ++it1;
335 }
336 ++it1;
337 }
338 }
339}

341void Molecule::_buildModel(const std::string& assemblyName,
342 const std::string& name, const std::string& title,
343 const std::string& header,
344 const ProteinRepresentation& representation,
345 const float atomRadiusMultiplier,
346 const bool loadBonds)
347{
348 PLUGIN_INFO("Building protein " << name << "...");
349
350 // Metadata
351 ModelMetadata metadata;
352 metadata[METADATA_ASSEMBLY] = assemblyName;
353 metadata[METADATA_TITLE] = title;
354 metadata[METADATA_HEADER] = header;
355 metadata[METADATA_ATOMS] = std::to_string(_atomMap.size());
356 metadata[METADATA_BONDS] = std::to_string(_bondsMap.size());
357 metadata[METADATA_AA_RANGE] = std::to_string(_aminoAcidRange.x) + ":" +
358 std::to_string(_aminoAcidRange.y);
359
360 const auto& size = _bounds.getSize();
361 metadata[METADATA_SIZE] = std::to_string(size.x) + ", " +
362 std::to_string(size.y) + ", " +
363 std::to_string(size.z) + " angstroms";
364
365 for (const auto& sequence : getSequencesAsString())
366 metadata[METADATA_AA_SEQUENCE + sequence.first] =
367 "[" + std::to_string(sequence.second.size()) + "] " +
368 sequence.second;
369
370 switch (representation)
371 {
372 case ProteinRepresentation::atoms:
373 case ProteinRepresentation::atoms_and_sticks:
374 {
375 auto model = _scene.createModel();
376 _buildAtomicStruture(representation, atomRadiusMultiplier, false,
377 loadBonds, *model);
378 _modelDescriptor =
379 std::make_shared<ModelDescriptor>(std::move(model), name, header,
380 metadata);
381 break;
382 }
383#ifdef USE_CGAL
384 case ProteinRepresentation::surface:
385 case ProteinRepresentation::union_of_balls:
386 {
387 // Surface
388 Vector4fs pointCloud;
389 const size_t materialId{0};
390 for (const auto& atom : _atomMap)
391 {
392 if (atom.first % std::max(1, int(atomRadiusMultiplier)) != 0)
393 continue;
394 pointCloud.push_back({atom.second.position.x,
395 atom.second.position.y,
396 atom.second.position.z,
397 atom.second.radius * atomRadiusMultiplier});
398 }
399
400 SurfaceMesher sm(_uuid);
401 if (representation == ProteinRepresentation::union_of_balls)
402 _modelDescriptor =
403 sm.generateUnionOfBalls(_scene, name, pointCloud);
404 else
405 _modelDescriptor = sm.generateSurface(_scene, name, pointCloud);
406 _modelDescriptor->setMetadata(metadata);
407
408 Model& model = _modelDescriptor->getModel();
409 _buildAtomicStruture(representation, atomRadiusMultiplier * 2.f, true,
410 loadBonds, model);
411 break;
412 }
413 case ProteinRepresentation::contour:
414 {
415 auto model = _scene.createModel();
416 PointCloud pointCloud;
417 for (const auto& atom : _atomMap)
418 pointCloud[0].push_back(
419 {atom.second.position.x, atom.second.position.y,
420 atom.second.position.z,
421 atom.second.radius * atomRadiusMultiplier});
422
423 PointCloudMesher pcm;
424 pcm.toConvexHull(*model, pointCloud);
425 _modelDescriptor =
426 std::make_shared<ModelDescriptor>(std::move(model), name, header,
427 metadata);
428 break;
429 }
430#else
431 case ProteinRepresentation::surface:
432 case ProteinRepresentation::union_of_balls:
433 case ProteinRepresentation::contour:
434 PLUGIN_THROW("CGAL is required to create surfaces");
435 break;
436#endif
437 case ProteinRepresentation::debug:
438 {
439 auto model = _scene.createModel();
440 const size_t materialId = 0;
441 auto material = model->createMaterial(materialId, "Debug");
442 brayns::PropertyMap props;
443 props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
444 static_cast<int>(MaterialShadingMode::basic)});
445 props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
446 props.setProperty(
447 {MATERIAL_PROPERTY_CHAMELEON_MODE,
448 static_cast<int>(
449 MaterialChameleonMode::undefined_chameleon_mode)});
450 material->setDiffuseColor({1.f, 1.f, 1.f});
451 material->updateProperties(props);
452
453 brayns::Boxf box;
454 for (const auto& atom : _atomMap)
455 box.merge({atom.second.position.x, atom.second.position.y,
456 atom.second.position.z});
457
458 const auto halfSize = box.getSize() * 0.5f;
459 const auto center = box.getCenter();
460
461 const brayns::Vector3f a = {0.f, 0.f, center.z + halfSize.z};
462 const brayns::Vector3f b = {0.f, 0.f, center.z - halfSize.z * 0.5f};
463 const brayns::Vector3f c = {0.f, 0.f, center.z - halfSize.z * 0.51f};
464 const brayns::Vector3f d = {0.f, 0.f, center.z - halfSize.z};
465
466 model->addSphere(materialId, {a, atomRadiusMultiplier * 0.2f});
467 model->addCylinder(materialId, {a, b, atomRadiusMultiplier * 0.2f});
468 model->addCone(materialId, {b, c, atomRadiusMultiplier * 0.2f,
469 atomRadiusMultiplier});
470 model->addCone(materialId, {c, d, atomRadiusMultiplier, 0.f});
471 _modelDescriptor =
472 std::make_shared<ModelDescriptor>(std::move(model), name, header,
473 metadata);
474 break;
475 }
476 }
477
478 PLUGIN_INFO("Protein model successfully built");
479
480 PLUGIN_INFO("---=== Protein ===--- ");
481 PLUGIN_INFO("Assembly name : " << assemblyName);
482 PLUGIN_INFO("Name : " << name);
483 PLUGIN_INFO("Atom Radius multiplier: " << atomRadiusMultiplier);
484 PLUGIN_INFO("Number of atoms : " << _atomMap.size());
485 PLUGIN_INFO("Number of bonds : " << _bondsMap.size());
486
487 if (_modelDescriptor &&
488 !GeneralSettings::getInstance()->getModelVisibilityOnCreation())
489 _modelDescriptor->setVisible(false);
490}

492void Molecule::_readAtom(const std::string& line, const bool loadHydrogen)
493{
494 // --------------------------------------------------------------------
495 // COLUMNS DATA TYPE FIELD DEFINITION
496 // --------------------------------------------------------------------
497 // 1 - 6 Record name "ATOM "
498 // 7 - 11 Integer serial Atom serial number
499 // 13 - 16 Atom name Atom name
500 // 17 Character altLoc Alternate location indicator
501 // 18 - 20 Residue name resName Residue name
502 // 22 Character chainID Chain identifier
503 // 23 - 26 Integer resSeq Residue sequence number
504 // 27 AChar iCode Code for insertion of residues
505 // 31 - 38 Real(8.3) x Orthogonal coords for X in angstroms
506 // 39 - 46 Real(8.3) y Orthogonal coords for Y in Angstroms
507 // 47 - 54 Real(8.3) z Orthogonal coords for Z in Angstroms
508 // 55 - 60 Real(6.2) occupancy Occupancy
509 // 61 - 66 Real(6.2) tempFactor Temperature factor
510 // 77 - 78 LString(2) element Element symbol, right-justified
511 // 79 - 80 LString(2) charge Charge on the atom
512 // --------------------------------------------------------------------
513
514 std::string s = line.substr(21, 1);
515 std::string chainId = trim(s);
516 if (!_loadChain(static_cast<size_t>(chainId[0] - 64)))
517 return;
518
519 const size_t serial = static_cast<size_t>(atoi(line.substr(6, 5).c_str()));
520
521 Atom atom;
522 atom.chainId = chainId;
523
524 s = line.substr(12, 4);
525 atom.name = trim(s);
526
527 s = line.substr(16, 1);
528 atom.altLoc = trim(s);
529
530 s = line.substr(17, 3);
531 atom.resName = trim(s);
532
533 _residues.insert(atom.resName);
534
535 atom.reqSeq = static_cast<size_t>(atoi(line.substr(22, 4).c_str()));
536 _aminoAcidRange.x = std::min(atom.reqSeq, size_t(_aminoAcidRange.x));
537 _aminoAcidRange.y = std::max(atom.reqSeq, size_t(_aminoAcidRange.y));
538
539 atom.iCode = line.substr(26, 1);
540
541 atom.position.x = static_cast<float>(atof(line.substr(30, 8).c_str()));
542 atom.position.y = static_cast<float>(atof(line.substr(38, 8).c_str()));
543 atom.position.z = static_cast<float>(atof(line.substr(46, 8).c_str()));
544
545 if (line.length() >= 60)
546 atom.occupancy = static_cast<float>(atof(line.substr(54, 6).c_str()));
547
548 if (line.length() >= 66)
549 atom.tempFactor = static_cast<float>(atof(line.substr(60, 6).c_str()));
550
551 if (line.length() >= 78)
552 {
553 s = line.substr(76, 2);
554 atom.element = trim(s);
555 if (s == "H" && !loadHydrogen)
556 return;
557 }
558
559 if (line.length() >= 80)
560 {
561 s = line.substr(78, 2);
562 atom.charge = trim(s);
563 }
564
565 // Convert position from amstrom (10e-10) to nanometers (10e-9)
566 atom.position = 0.1f * atom.position;
567
568 // Bounds
569 _bounds.merge(atom.position);
570
571 // Convert radius from picometers (10e-12) to nanometers (10e-9)
572 atom.radius = 0.001f * DEFAULT_ATOM_RADIUS;
573 auto it = atomicRadii.find(atom.element);
574 if (it != atomicRadii.end())
575 atom.radius = 0.001f * (*it).second;
576 else
577 {
578 it = atomicRadii.find(atom.name);
579 if (it != atomicRadii.end())
580 atom.radius = 0.001f * (*it).second;
581 else
582 PLUGIN_WARN("[" << atom.element << "]/[" << atom.name
583 << "] not found");
584 }
585
586 _atomMap.insert(std::make_pair(serial, atom));
587}

589void Molecule::_readSequence(const std::string& line)
590{
591 // -------------------------------------------------------------------------
592 // COLUMNS TYPE FIELD DEFINITION
593 // -------------------------------------------------------------------------
594 // 1 - 6 Record name "SEQRES"
595 // 8 - 10 Integer serNum Serial number of the SEQRES record for the
596 // current chain. Starts at 1 and increments
597 // by one each line. Reset to 1 for each
598 // chain.
599 // 12 Character chainID Chain identifier. This may be any single
600 // legal
601 // character, including a blank which is is
602 // used if there is only one chain
603 // 14 - 17 Integer numRes Number of residues in the chain. This
604 // value is
605 // repeated on every record.
606 // 20 - 22 String resName Residue name
607 // 24 - 26 ...
608 // -------------------------------------------------------------------------
609
610 std::string s = line.substr(11, 1);
611
612 ResidueSequence& sequence = _residueSequenceMap[s];
613 sequence.numRes = static_cast<size_t>(atoi(line.substr(13, 4).c_str()));
614
615 for (size_t i = 19; i < line.length(); i += 4)
616 {
617 s = line.substr(i, 4);
618 s = trim(s);
619 if (!s.empty())
620 sequence.resNames.push_back(s);
621 }
622}

624void Molecule::_readConnect(const std::string& line)
625{
626 // -------------------------------------------------------------------------
627 // COLUMNS TYPE FIELD DEFINITION
628 // -------------------------------------------------------------------------
629 // 1 - 6 Record name "CONECT"
630 // 7 - 11 Integer serial Atom serial number
631 // 12 - 16 Integer serial Serial number of bonded atom
632 // 17 - 21 Integer serial Serial number of bonded atom
633 // 22 - 26 Integer serial Serial number of bonded atom
634 // 27 - 31 Integer serial Serial number of bonded atom
635 // -------------------------------------------------------------------------
636
637 const size_t serial = static_cast<size_t>(atoi(line.substr(6, 5).c_str()));
638
639 if (_atomMap.find(serial) != _atomMap.end())
640 {
641 auto& bond = _bondsMap[serial];
642
643 for (size_t i = 11; i < line.length(); i += 5)
644 {
645 std::string s = line.substr(i, 5);
646 s = trim(s);
647 if (!s.empty())
648 {
649 const size_t atomSerial = static_cast<size_t>(atoi(s.c_str()));
650 if (_atomMap.find(atomSerial) != _atomMap.end())
651 bond.push_back(atomSerial);
652 }
653 }
654 }

657void Molecule::_readRemark(const std::string& line)
658{
659 // -------------------------------------------------------------------------
660 // COLUMNS TYPE FIELD DEFINITION
661 // -------------------------------------------------------------------------
662 // 1 - 6 Record name "REMARK"
663 // 8 - 10 Integer remarkNum Remark number. It is not an error for
664 // remark
665 // n to exist in an entry when remark n-1
666 // does not.
667 // 13 - 16 String "ALN"
668 // 17 - 18 String "C"
669 // 19 - 22 String "TRG"
670 // 23 - 81 String Sequence
671 // -------------------------------------------------------------------------
672
673 std::string s = line.substr(9, 1);
674 if (s != "3")
675 return;
676
677 if (line.length() < 23)
678 return;
679
680 s = line.substr(12, 3);
681 if (s != "ALN")
682 return;
683
684 s = line.substr(16, 1);
685 if (s != "C")
686 return;
687
688 s = line.substr(18, 3);
689 if (s != "TRG")
690 return;
691
692 s = line.substr(22, line.length() - 23);
693 ResidueSequence& sequence = _residueSequenceMap[0];
694 if (sequence.resNames.empty())
695 sequence.resNames.push_back(s);
696 else
697 sequence.resNames[0] = sequence.resNames[0] + s;
698}

743void Molecule::_setAminoAcidSequenceColorScheme(const Palette& palette)
744{
745 if (palette.size() != 2)
746 PLUGIN_THROW("Invalid palette size. 2 colors are expected");
747
748 size_t atomCount = 0;
749 for (const auto& sequence : _residueSequenceMap)
750 {
751 if (_selectedAminoAcidSequence.empty())
752 {
753 // Range based coloring
754 for (auto& atom : _atomMap)
755 {
756 bool selected = false;
757 for (const auto& range : _selectedAminoAcidRanges)
758 {
759 selected = (atom.second.reqSeq >= range.x &&
760 atom.second.reqSeq <= range.y);
761 if (selected)
762 break;
763 }
764 _setMaterialDiffuseColor(atom.first,
765 selected ? palette[1] : palette[0]);
766 }
767 }
768 else
769 {
770 // String based coloring
771 std::string shortSequence;
772 for (const auto& resName : sequence.second.resNames)
773 shortSequence += aminoAcidMap[resName].shortName;
774
775 const auto sequencePosition =
776 shortSequence.find(_selectedAminoAcidSequence);
777 if (sequencePosition != -1)
778 {
779 PLUGIN_INFO(_selectedAminoAcidSequence
780 << " was found at position " << sequencePosition);
781 size_t minSeq = 1e6;
782 size_t maxSeq = 0;
783 for (auto& atom : _atomMap)
784 {
785 minSeq = std::min(minSeq, atom.second.reqSeq);
786 maxSeq = std::max(maxSeq, atom.second.reqSeq);
787 if (atom.second.reqSeq >= sequencePosition &&
788 atom.second.reqSeq <
789 sequencePosition +
790 _selectedAminoAcidSequence.length())
791 {
792 _setMaterialDiffuseColor(atom.first, palette[1]);
793 ++atomCount;
794 }
795 else
796 _setMaterialDiffuseColor(atom.first, palette[0]);
797 }
798 PLUGIN_DEBUG(atomCount << "[" << minSeq << "," << maxSeq
799 << "] atoms where colored");
800 }
801 else
802 PLUGIN_WARN(_selectedAminoAcidSequence << " was not found in "
803 << shortSequence);
804 }
805 }
806 PLUGIN_INFO("Applying Amino Acid Sequence color scheme ("
807 << (atomCount > 0 ? "2" : "1") << ")");
808}
