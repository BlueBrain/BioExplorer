101 void Molecule::_computeReqSetOffset()
102 {
103  for (auto& sequence : _residueSequenceMap)
104  {
105  std::string physicalReqSeq;
106  size_t firstReqSeq;
107  size_t previousReqSeq;
108  bool initialized{false};
109  for (const auto& atom : _atomMap)
110  {
111  if (atom.second.chainId != sequence.first)
112  continue;
113 
114  if (!initialized)
115  {
116  firstReqSeq = atom.second.reqSeq;
117  previousReqSeq = firstReqSeq - 1;
118  initialized = true;
119  }
120 
121  if (previousReqSeq != atom.second.reqSeq)
122  {
123  if (atom.second.reqSeq != previousReqSeq + 1)
124  break;
125 
126  const auto it = aminoAcidMap.find(atom.second.resName);
127  if (it != aminoAcidMap.end())
128  physicalReqSeq += it->second.shortName;
129 
130  if (physicalReqSeq.length() > 10)
131  break;
132  }
133 
134  previousReqSeq = atom.second.reqSeq;
135  }
136 
137  std::string theoreticalReqSeq;
138  for (const auto& aa : sequence.second.resNames)
139  theoreticalReqSeq += aminoAcidMap.find(aa)->second.shortName;
140 
141  sequence.second.offset = theoreticalReqSeq.find(physicalReqSeq) - firstReqSeq;
142  PLUGIN_INFO(3, "Sequence [" << sequence.first << "], offset: " << sequence.second.offset
143  << ", Theoretical: " << theoreticalReqSeq << ", Physical: " << physicalReqSeq);
144  }
145 }

162 void Molecule::_buildAtomicStruture(const ProteinRepresentation representation, const double atomRadiusMultiplier,
163  const bool surface, const bool loadBonds, ThreadSafeContainer& container)
164 {
165  const uint64_t userData = NO_USER_DATA;
166  const bool useSdf = DEFAULT_USE_SDF;
167 
168  // Atoms
169  std::map<uint64_t, Neighbours> neighbours;
170  size_t currentReqSeq = 0;
171  for (const auto& atom : _atomMap)
172  {
173  // Geometry
174  const float radius = static_cast<float>(atom.second.radius * atomRadiusMultiplier);
175  neighbours[currentReqSeq].insert(container.addSphere(atom.second.position, radius, atom.first, useSdf, userData,
176  neighbours[currentReqSeq], DEFAULT_SDF_DISPLACEMENT));
177  if (currentReqSeq != atom.second.reqSeq)
178  currentReqSeq = atom.second.reqSeq;
179  }
180 
181  // Bonds
182  if (loadBonds)
183  {
184  PLUGIN_INFO(3, "Building " << _bondsMap.size() << " bonds...");
185  for (const auto& bond : _bondsMap)
186  {
187  const auto& atomSrc = _atomMap.find(bond.first)->second;
188  for (const auto bondedAtom : bond.second)
189  {
190  const auto& atomDst = _atomMap.find(bondedAtom)->second;
191  const float radius = static_cast<float>(atomRadiusMultiplier * DEFAULT_BOND_RADIUS);
192 
193  const auto center = (atomSrc.position + atomDst.position) / 2.0;
194 
195  const auto reqSeq = atomSrc.reqSeq;
196  neighbours[reqSeq].insert(container.addCone(atomSrc.position, radius, center, radius, bond.first,
197  useSdf, userData, neighbours[reqSeq],
198  DEFAULT_SDF_DISPLACEMENT));
199 
200  neighbours[reqSeq].insert(container.addCone(atomDst.position, radius, center, radius, bondedAtom,
201  useSdf, userData, neighbours[reqSeq],
202  DEFAULT_SDF_DISPLACEMENT));
203  }
204  }
205  }
206 
207  // Sticks
208  if (representation == ProteinRepresentation::atoms_and_sticks)
209  {
210  PLUGIN_INFO(3, "Building sticks (" << _atomMap.size() << " atoms)...");
211  auto it1 = _atomMap.begin();
212  while (it1 != _atomMap.end())
213  {
214  const auto atom1 = (*it1);
215  auto it2 = it1;
216  ++it2;
217  const auto reqSeq = atom1.second.reqSeq;
218  while ((*it2).second.reqSeq == reqSeq)
219  {
220  const auto stick = (*it2).second.position - atom1.second.position;
221  if (length(stick) < DEFAULT_STICK_DISTANCE)
222  {
223  const auto center = ((*it2).second.position + atom1.second.position) / 2.0;
224  const float radius = static_cast<float>(atomRadiusMultiplier * DEFAULT_BOND_RADIUS);
225  neighbours[reqSeq].insert(container.addCone(atom1.second.position, radius, center, radius,
226  atom1.first, useSdf, userData, neighbours[reqSeq],
227  DEFAULT_SDF_DISPLACEMENT));
228  neighbours[reqSeq].insert(container.addCone((*it2).second.position, radius, center, radius,
229  (*it2).first, useSdf, userData, neighbours[reqSeq],
230  DEFAULT_SDF_DISPLACEMENT));
231  }
232  ++it2;
233  ++it1;
234  }
235  ++it1;
236  }
237  }
238 }

240 void Molecule::_rescaleMesh(Model& model, const Vector3f& scale)
241 {
242  auto& triangleMeshes = model.getTriangleMeshes();
243  const auto& bounds = model.getBounds();
244  const Vector3f center = bounds.getCenter();
245  for (auto& triangleMesh : triangleMeshes)
246  {
247  auto& vertices = triangleMesh.second.vertices;
248  for (auto& vertex : vertices)
249  vertex = center + (vertex - center) * scale;
250  }
251 }

253 void Molecule::_buildModel(const std::string& assemblyName, const std::string& name, const std::string& pdbId,
254  const std::string& header, const ProteinRepresentation& representation,
255  const double atomRadiusMultiplier, const bool loadBonds)
256 {
257  PLUGIN_INFO(3, "Building protein " << name << " [PDB " << pdbId << "]...");
258 
259  // Metadata
260  ModelMetadata metadata;
261  metadata[METADATA_ASSEMBLY] = assemblyName;
262  metadata[METADATA_PDB_ID] = pdbId;
263  metadata[METADATA_HEADER] = header;
264  metadata[METADATA_ATOMS] = std::to_string(_atomMap.size());
265  metadata[METADATA_BONDS] = std::to_string(_bondsMap.size());
266  metadata[METADATA_AA_RANGE] = std::to_string(_aminoAcidRange.x) + ":" + std::to_string(_aminoAcidRange.y);
267 
268  const auto& size = _bounds.getSize();
269  metadata[METADATA_SIZE] =
270  std::to_string(size.x) + ", " + std::to_string(size.y) + ", " + std::to_string(size.z) + " angstroms";
271 
272  for (const auto& sequence : getSequencesAsString())
273  metadata[METADATA_AA_SEQUENCE + sequence.first] =
274  "[" + std::to_string(sequence.second.size()) + "] " + sequence.second;
275 
276  switch (representation)
277  {
278  case ProteinRepresentation::atoms:
279  case ProteinRepresentation::atoms_and_sticks:
280  {
281  auto model = _scene.createModel();
282  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation);
283 
284  _buildAtomicStruture(representation, atomRadiusMultiplier, false, loadBonds, container);
285  container.commitToModel();
286 
287  // Materials
288  for (const auto atom : _atomMap)
289  {
290  const auto materialId = atom.first;
291  auto material = model->getMaterial(materialId);
292  RGBColorDetails rgb{255, 255, 255};
293  const auto it = atomColorMap.find(atom.second.element);
294  if (it != atomColorMap.end())
295  rgb = (*it).second;
296 
297  material->setNodeId(_uuid);
298  material->setDiffuseColor({rgb.r / 255.0, rgb.g / 255.0, rgb.b / 255.0});
299  }
300 
301  _modelDescriptor = std::make_shared<ModelDescriptor>(std::move(model), name, header, metadata);
302 
303  break;
304  }
305 #ifdef USE_ASSIMP
306  case ProteinRepresentation::mesh:
307  {
308  const std::string filename = GeneralSettings::getInstance()->getMeshFolder() + pdbId + ".obj";
309  MeshLoader meshLoader(_scene);
310  _modelDescriptor = meshLoader.importFromStorage(filename, LoaderProgress(), {});
311  _setMaterialExtraAttributes();
312  _rescaleMesh(_modelDescriptor->getModel(), Vector3d(0.1, 0.1, 0.1));
313  break;
314  }
315 #endif
316 #ifdef USE_CGAL
317  case ProteinRepresentation::surface:
318  case ProteinRepresentation::union_of_balls:
319  {
320  // Surface
321  Vector4ds pointCloud;
322  const size_t materialId{0};
323  for (const auto& atom : _atomMap)
324  {
325  if (atom.first % std::max(1, int(atomRadiusMultiplier)) != 0)
326  continue;
327  pointCloud.push_back({atom.second.position.x, atom.second.position.y, atom.second.position.z,
328  atom.second.radius * atomRadiusMultiplier});
329  }
330 
331  SurfaceMesher sm(_uuid);
332  if (representation == ProteinRepresentation::union_of_balls)
333  _modelDescriptor = sm.generateUnionOfBalls(_scene, pdbId, pointCloud);
334  else
335  _modelDescriptor = sm.generateSurface(_scene, pdbId, pointCloud);
336  _setMaterialExtraAttributes();
337  _modelDescriptor->setMetadata(metadata);
338 
339  Model& model = _modelDescriptor->getModel();
340  ThreadSafeContainer container(model, _alignToGrid, _position, _rotation);
341  _buildAtomicStruture(representation, atomRadiusMultiplier * 2.0, true, loadBonds, container);
342  container.commitToModel();
343  // Materials
344  for (const auto atom : _atomMap)
345  {
346  const auto materialId = atom.first;
347  auto material = model.getMaterial(materialId);
348  RGBColorDetails rgb{255, 255, 255};
349  const auto it = atomColorMap.find(atom.second.element);
350  if (it != atomColorMap.end())
351  rgb = (*it).second;
352 
353  core::PropertyMap props;
354  material->setChameleonMode(MaterialChameleonMode::emitter);
355  material->setNodeId(_uuid);
356  material->setDiffuseColor({rgb.r / 255.0, rgb.g / 255.0, rgb.b / 255.0});
357  }
358  break;
359  }
360  case ProteinRepresentation::contour:
361  {
362  auto model = _scene.createModel();
363  PointCloud pointCloud;
364  for (const auto& atom : _atomMap)
365  pointCloud[0].push_back({atom.second.position.x, atom.second.position.y, atom.second.position.z,
366  atom.second.radius * atomRadiusMultiplier});
367 
368  PointCloudMesher pcm;
369  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation);
370  pcm.toConvexHull(container, pointCloud);
371  container.commitToModel();
372  _modelDescriptor = std::make_shared<ModelDescriptor>(std::move(model), name, header, metadata);
373  _setMaterialExtraAttributes();
374  break;
375  }
376 #else
377  case ProteinRepresentation::surface:
378  case ProteinRepresentation::union_of_balls:
379  case ProteinRepresentation::contour:
380  PLUGIN_THROW("CGAL is required to create surfaces");
381  break;
382 #endif
383  case ProteinRepresentation::debug:
384  {
385  auto model = _scene.createModel();
386  const size_t materialId = 0;
387  core::Boxf box;
388  for (const auto& atom : _atomMap)
389  box.merge({atom.second.position.x, atom.second.position.y, atom.second.position.z});
390 
391  const auto halfSize = box.getSize() * 0.5;
392  const auto center = box.getCenter();
393 
394  const core::Vector3d a = {0.0, 0.0, center.z + halfSize.z};
395  const core::Vector3d b = {0.0, 0.0, center.z - halfSize.z * 0.5};
396  const core::Vector3d c = {0.0, 0.0, center.z - halfSize.z * 0.51};
397  const core::Vector3d d = {0.0, 0.0, center.z - halfSize.z};
398 
399  model->addSphere(materialId, {a, static_cast<float>(atomRadiusMultiplier * 0.2)});
400  model->addCylinder(materialId, {a, b, static_cast<float>(atomRadiusMultiplier * 0.2)});
401  model->addCone(materialId, {b, c, static_cast<float>(atomRadiusMultiplier * 0.2),
402  static_cast<float>(atomRadiusMultiplier)});
403  model->addCone(materialId, {c, d, static_cast<float>(atomRadiusMultiplier), 0.0});
404  _modelDescriptor = std::make_shared<ModelDescriptor>(std::move(model), name, header, metadata);
405  break;
406  }
407  }
408 
409  PLUGIN_INFO(3, "Molecule model successfully built");
410 
411  PLUGIN_INFO(3, "---=== Molecule ===--- ");
412  PLUGIN_INFO(3, "Assembly name : " << assemblyName);
413  PLUGIN_INFO(3, "Name : " << name);
414  PLUGIN_INFO(3, "Atom Radius multiplier: " << atomRadiusMultiplier);
415  PLUGIN_INFO(3, "Number of atoms : " << _atomMap.size());
416  PLUGIN_INFO(3, "Number of bonds : " << _bondsMap.size());
417 
418  if (_modelDescriptor && !GeneralSettings::getInstance()->getModelVisibilityOnCreation())
419  _modelDescriptor->setVisible(false);
420 }

422 void Molecule::_readAtom(const std::string& line, const bool loadHydrogen)
423 {
424  // --------------------------------------------------------------------
425  // COLUMNS DATA TYPE FIELD DEFINITION
426  // --------------------------------------------------------------------
427  // 1 - 6 Record name "ATOM "
428  // 7 - 11 Integer serial Atom serial number
429  // 13 - 16 Atom name Atom name
430  // 17 Character altLoc Alternate location indicator
431  // 18 - 20 Residue name resName Residue name
432  // 22 Character chainID Chain identifier
433  // 23 - 26 Integer resSeq Residue sequence number
434  // 27 AChar iCode Code for insertion of residues
435  // 31 - 38 Real(8.3) x Orthogonal coords for X in angstroms
436  // 39 - 46 Real(8.3) y Orthogonal coords for Y in Angstroms
437  // 47 - 54 Real(8.3) z Orthogonal coords for Z in Angstroms
438  // 55 - 60 Real(6.2) occupancy Occupancy
439  // 61 - 66 Real(6.2) tempFactor Temperature factor
440  // 77 - 78 LString(2) element Element symbol, right-justified
441  // 79 - 80 LString(2) charge Charge on the atom
442  // --------------------------------------------------------------------
443 
444  std::string s = line.substr(21, 1);
445  std::string chainId = trim(s);
446  if (!_loadChain(static_cast<size_t>(chainId[0] - 64)))
447  return;
448 
449  const size_t serial = static_cast<size_t>(atoi(line.substr(6, 5).c_str()));
450 
451  Atom atom;
452  atom.chainId = chainId;
453 
454  s = line.substr(12, 4);
455  atom.name = trim(s);
456 
457  s = line.substr(16, 1);
458  atom.altLoc = trim(s);
459 
460  s = line.substr(17, 3);
461  atom.resName = trim(s);
462 
463  _residues.insert(atom.resName);
464 
465  atom.reqSeq = static_cast<size_t>(atoi(line.substr(22, 4).c_str()));
466  _aminoAcidRange.x = std::min(atom.reqSeq, size_t(_aminoAcidRange.x));
467  _aminoAcidRange.y = std::max(atom.reqSeq, size_t(_aminoAcidRange.y));
468 
469  atom.iCode = line.substr(26, 1);
470 
471  atom.position.x = static_cast<double>(atof(line.substr(30, 8).c_str()));
472  atom.position.y = static_cast<double>(atof(line.substr(38, 8).c_str()));
473  atom.position.z = static_cast<double>(atof(line.substr(46, 8).c_str()));
474 
475  if (line.length() >= 60)
476  atom.occupancy = static_cast<double>(atof(line.substr(54, 6).c_str()));
477 
478  if (line.length() >= 66)
479  atom.tempFactor = static_cast<double>(atof(line.substr(60, 6).c_str()));
480 
481  if (line.length() >= 78)
482  {
483  s = line.substr(76, 2);
484  atom.element = trim(s);
485  if (s == "H" && !loadHydrogen)
486  return;
487  }
488 
489  if (line.length() >= 80)
490  {
491  s = line.substr(78, 2);
492  atom.charge = trim(s);
493  }
494 
495  // Convert position from amstrom (10e-10) to nanometers (10e-9)
496  atom.position = 0.1 * atom.position;
497 
498  // Bounds
499  _bounds.merge(atom.position);
500 
501  // Convert radius from picometers (10e-12) to nanometers (10e-9)
502  atom.radius = 0.001 * DEFAULT_ATOM_RADIUS;
503  auto it = atomicRadii.find(atom.element);
504  if (it != atomicRadii.end())
505  atom.radius = 0.001 * (*it).second;
506  else
507  {
508  it = atomicRadii.find(atom.name);
509  if (it != atomicRadii.end())
510  atom.radius = 0.001 * (*it).second;
511  else
512  PLUGIN_DEBUG("[" << atom.element << "]/[" << atom.name << "] not found");
513  }
514 
515  _atomMap.insert(std::make_pair(serial, atom));
516 }

518 void Molecule::_readSequence(const std::string& line)
519 {
520  // -------------------------------------------------------------------------
521  // COLUMNS TYPE FIELD DEFINITION
522  // -------------------------------------------------------------------------
523  // 1 - 6 Record name "SEQRES"
524  // 8 - 10 Integer serNum Serial number of the SEQRES record for the
525  // current chain. Starts at 1 and increments
526  // by one each line. Reset to 1 for each
527  // chain.
528  // 12 Character chainID Chain identifier. This may be any single
529  // legal
530  // character, including a blank which is is
531  // used if there is only one chain
532  // 14 - 17 Integer numRes Number of residues in the chain. This
533  // value is
534  // repeated on every record.
535  // 20 - 22 String resName Residue name
536  // 24 - 26 ...
537  // -------------------------------------------------------------------------
538 
539  std::string s = line.substr(11, 1);
540 
541  ResidueSequence& sequence = _residueSequenceMap[s];
542  sequence.numRes = static_cast<size_t>(atoi(line.substr(13, 4).c_str()));
543 
544  for (size_t i = 19; i < line.length(); i += 4)

671 void Molecule::_setAminoAcidSequenceColorScheme(const Palette& palette)
672 {
673  if (palette.size() != 2)
674  PLUGIN_THROW("Invalid palette size. 2 colors are expected");
675 
676  size_t atomCount = 0;
677  for (const auto& sequence : _residueSequenceMap)
678  {
679  if (_selectedAminoAcidSequence.empty())
680  {
681  // Range based coloring
682  for (auto& atom : _atomMap)
683  {
684  bool selected = false;
685  for (const auto& range : _selectedAminoAcidRanges)
686  {
687  selected = (atom.second.reqSeq >= range.x && atom.second.reqSeq <= range.y);
688  if (selected)
689  break;
690  }
691  _setMaterialDiffuseColor(atom.first, selected ? palette[1] : palette[0]);
692  }
693  }
694  else
695  {
696  // String based coloring
697  std::string shortSequence;
698  for (const auto& resName : sequence.second.resNames)
699  shortSequence += aminoAcidMap[resName].shortName;
700 
701  const auto sequencePosition = shortSequence.find(_selectedAminoAcidSequence);
702  if (sequencePosition != -1)
703  {
704  PLUGIN_INFO(3, _selectedAminoAcidSequence << " was found at position " << sequencePosition);
705  size_t minSeq = 1e6;
706  size_t maxSeq = 0;
707  for (auto& atom : _atomMap)
708  {
709  minSeq = std::min(minSeq, atom.second.reqSeq);
710  maxSeq = std::max(maxSeq, atom.second.reqSeq);
711  if (atom.second.reqSeq >= sequencePosition &&
712  atom.second.reqSeq < sequencePosition + _selectedAminoAcidSequence.length())
713  {
714  _setMaterialDiffuseColor(atom.first, palette[1]);
715  ++atomCount;
716  }
717  else
718  _setMaterialDiffuseColor(atom.first, palette[0]);
719  }
720  PLUGIN_DEBUG(atomCount << "[" << minSeq << "," << maxSeq << "] atoms where colored");
721  }
722  else
723  PLUGIN_WARN(_selectedAminoAcidSequence << " was not found in " << shortSequence);
724  }
725  }
726  PLUGIN_INFO(3, "Applying Amino Acid Sequence color scheme (" << (atomCount > 0 ? "2" : "1") << ")");
727 }
