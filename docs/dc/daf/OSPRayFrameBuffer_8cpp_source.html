69 void OSPRayFrameBuffer::resize(const Vector2ui& frameSize)
70 {

82 void OSPRayFrameBuffer::_recreate()
83 {
84  auto lock = getScopeLock();
85 
86  _unmapUnsafe();
87  ospRelease(_frameBuffer);
88 
89  const osp::vec2i size = {int(_frameSize.x), int(_frameSize.y)};
90 
91  size_t attributes = OSP_FB_COLOR | OSP_FB_DEPTH;
92  if (_accumulation)
93  attributes |= OSP_FB_ACCUM | OSP_FB_VARIANCE;
94 
95  _frameBuffer = ospNewFrameBuffer(size, toOSPFrameBufferFormat(_frameBufferFormat), attributes);
96  if (_pixelOp)
97  ospSetPixelOp(_frameBuffer, _pixelOp);
98  osphelper::set(_frameBuffer, OSPRAY_FRAME_BUFFER_PROPERTY_NAME, getName());
99  ospCommit(_frameBuffer);
100 
101  _recreateSubsamplingBuffer();
102 
103  clear();
104 }

106 void OSPRayFrameBuffer::_recreateSubsamplingBuffer()
107 {
108  ospRelease(_subsamplingFrameBuffer);
109  _subsamplingFrameBuffer = nullptr;
110  const auto subsamplingSize = _subsamplingSize();
111  if (_frameSize != subsamplingSize)
112  {
113  _subsamplingFrameBuffer =
114  ospNewFrameBuffer({int(subsamplingSize.x), int(subsamplingSize.y)},
115  toOSPFrameBufferFormat(_frameBufferFormat), OSP_FB_COLOR | OSP_FB_DEPTH);
116  if (_pixelOp)
117  ospSetPixelOp(_subsamplingFrameBuffer, _pixelOp);
118  ospCommit(_subsamplingFrameBuffer);
119  }
120 }

205 void OSPRayFrameBuffer::setSubsampling(size_t factor)
206 {
