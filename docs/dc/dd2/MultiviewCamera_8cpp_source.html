39 void MultiviewCamera::commit()
40 {
41  Camera::commit();
42 
43  const float fovy = getParamf("fovy", 60.f);
44  const float aspect = getParamf("aspect", 1.66667f);
45  const float apertureRadius = getParamf("apertureRadius", 0.f);
46  const float focusDistance = getParamf("focusDistance", 1.f);
47  const float height = getParamf("height", 10.f); // imgPlane_size_y
48  const float armLength = getParamf("armLength", 5.f);
49 
50  clipPlanes = getParamData("clipPlanes", nullptr);
51 
52  dir = normalize(dir);
53  vec3f dir_du = normalize(cross(dir, up));
54  vec3f dir_dv = cross(dir_du, dir);
55 
56  vec3f org = pos;
57 
58  float imgPlane_size_y = 2.f * tanf(deg2rad(0.5f * fovy));
59  float imgPlane_size_x = imgPlane_size_y * aspect;
60 
61  dir_du *= imgPlane_size_x;
62  dir_dv *= imgPlane_size_y;
63 
64  vec3f dir_00 = dir - 0.5f * dir_du - 0.5f * dir_dv;
65 
66  float scaledAperture = 0.f;
67  // prescale to focal plane
68  if (apertureRadius > 0.f)
69  {
70  dir_du *= focusDistance;
71  dir_dv *= focusDistance;
72  dir_00 *= focusDistance;
73  scaledAperture = apertureRadius / imgPlane_size_x;
74  }
75 
76  const auto clipPlaneData = clipPlanes ? clipPlanes->data : nullptr;
77  const size_t numClipPlanes = clipPlanes ? clipPlanes->numItems : 0;
78 
79  ispc::MultiviewCamera_set(getIE(), (const ispc::vec3f&)org, (const ispc::vec3f&)dir_00, (const ispc::vec3f&)dir_du,
80  (const ispc::vec3f&)dir_dv, scaledAperture, height, aspect, armLength,
81  (const ispc::vec4f*)clipPlaneData, numClipPlanes);
82 }
