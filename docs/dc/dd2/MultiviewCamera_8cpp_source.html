44 void MultiviewCamera::commit()
45 {
46  Camera::commit();
47 
48  const float fovy = getParamf(CAMERA_PROPERTY_FIELD_OF_VIEW.name.c_str(), DEFAULT_CAMERA_FIELD_OF_VIEW);
49  const float aspect = getParamf(CAMERA_PROPERTY_ASPECT_RATIO.name.c_str(), 1.66667f);
50  const float apertureRadius =
51  getParamf(CAMERA_PROPERTY_APERTURE_RADIUS.name.c_str(), DEFAULT_CAMERA_APERTURE_RADIUS);
52  const float focalDistance = getParamf(CAMERA_PROPERTY_FOCAL_DISTANCE.name.c_str(), DEFAULT_CAMERA_FOCAL_DISTANCE);
53  const float height = getParamf(CAMERA_PROPERTY_HEIGHT.name.c_str(), DEFAULT_CAMERA_HEIGHT);
54  const float armLength = getParamf(PARAM_ARM_LENGTH.c_str(), DEFAULT_PARAM_ARM_LENGTH);
55 
56  clipPlanes = getParamData(CAMERA_PROPERTY_CLIPPING_PLANES, nullptr);
57 
58  dir = normalize(dir);
59  vec3f dir_du = normalize(cross(dir, up));
60  vec3f dir_dv = cross(dir_du, dir);
61 
62  vec3f org = pos;
63 
64  float imgPlane_size_y = 2.f * tanf(deg2rad(0.5f * fovy));
65  float imgPlane_size_x = imgPlane_size_y * aspect;
66 
67  dir_du *= imgPlane_size_x;
68  dir_dv *= imgPlane_size_y;
69 
70  vec3f dir_00 = dir - 0.5f * dir_du - 0.5f * dir_dv;
71 
72  float scaledAperture = 0.f;
73  // prescale to focal plane
74  if (apertureRadius > 0.f)
75  {
76  dir_du *= focalDistance;
77  dir_dv *= focalDistance;
78  dir_00 *= focalDistance;
79  scaledAperture = apertureRadius / imgPlane_size_x;
80  }
81 
82  const auto clipPlaneData = clipPlanes ? clipPlanes->data : nullptr;
83  const size_t numClipPlanes = clipPlanes ? clipPlanes->numItems : 0;
84 
85  ::ispc::MultiviewCamera_set(getIE(), (const ::ispc::vec3f&)org, (const ::ispc::vec3f&)dir_00, (const ::ispc::vec3f&)dir_du,
86  (const ::ispc::vec3f&)dir_dv, scaledAperture, height, aspect, armLength,
87  (const ::ispc::vec4f*)clipPlaneData, numClipPlanes);
88 }
