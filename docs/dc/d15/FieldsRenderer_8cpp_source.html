35void FieldsRenderer::commit()
36{
37 Renderer::commit();
38
39 _lightData = (ospray::Data*)getParamData("lights");
40 _lightArray.clear();
41
42 if (_lightData)
43 for (size_t i = 0; i < _lightData->size(); ++i)
44 _lightArray.push_back(
45 ((ospray::Light**)_lightData->data)[i]->getIE());
46
47 _lightPtr = _lightArray.empty() ? nullptr : &_lightArray[0];
48
49 _bgMaterial = (AdvancedMaterial*)getParamObject("bgMaterial", nullptr);
50
51 _useHardwareRandomizer = getParam("useHardwareRandomizer", 0);
52
53 _exposure = getParam1f("exposure", 1.f);
54 _randomNumber = getParam1i("randomNumber", 0);
55 _timestamp = getParam1f("timestamp", 0.f);
56
57 // Sampling
58 _minRayStep = getParam1f("minRayStep", 0.1f);
59 _nbRaySteps = getParam1i("nbRaySteps", 8);
60 _nbRayRefinementSteps = getParam1i("nbRayRefinementSteps", 8);
61 _alphaCorrection = getParam1f("alphaCorrection", 1.0f);
62
63 // Extra
64 _cutoff = getParam1f("cutoff", 1.f);
65
66 // Octree
67 _userData = getParamData("simulationData");
68 _userDataSize = _userData ? _userData->size() : 0;
69
70 // Transfer function
71 ospray::TransferFunction* transferFunction =
72 (ospray::TransferFunction*)getParamObject("transferFunction", nullptr);
73 if (transferFunction)
74 ispc::FieldsRenderer_setTransferFunction(getIE(),
75 transferFunction->getIE());
76
77 // Renderer
78 ispc::FieldsRenderer_set(getIE(),
79 (_bgMaterial ? _bgMaterial->getIE() : nullptr),
80 (_userData ? (float*)_userData->data : nullptr),
81 _userDataSize, _randomNumber, _timestamp, spp,
82 _lightPtr, _lightArray.size(), _minRayStep,
83 _nbRaySteps, _nbRayRefinementSteps, _exposure,
84 _useHardwareRandomizer, _cutoff, _alphaCorrection);
85}
