195 void Neurons::_buildModel(const LoaderProgress& callback)
196 {
197  const auto& connector = DBConnector::getInstance();
198 
199  auto model = _scene.createModel();
200 
201  // Simulation report
202  std::string sqlNodeFilter = _details.sqlNodeFilter;
203  if (_details.simulationReportId != -1)
204  {
205  _simulationReport = connector.getSimulationReport(_details.populationName, _details.simulationReportId);
206  _attachSimulationReport(*model);
207  }
208 
209  // Neurons
210  const auto somas = connector.getNeurons(_details.populationName, sqlNodeFilter);
211 
212  if (somas.empty())
213  PLUGIN_THROW("Selection returned no nodes");
214 
215  PLUGIN_INFO(1, "Building " << somas.size() << " neurons");
216  _logRealismParams();
217 
218  size_t previousMaterialId = std::numeric_limits<size_t>::max();
219  Vector3ui indexOffset;
220 
221  const bool somasOnly =
222  _details.loadSomas && !_details.loadAxon && !_details.loadApicalDendrites && !_details.loadBasalDendrites;
223 
224  ThreadSafeContainers containers;
225  if (somasOnly || _details.morphologyRepresentation == MorphologyRepresentation::orientation ||
226  _details.morphologyRepresentation == MorphologyRepresentation::contour)
227  {
228  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation, _scale);
229  switch (_details.morphologyRepresentation)
230  {
231  case MorphologyRepresentation::orientation:
232  {
233  _buildOrientations(container, somas);
234  break;
235  }
236  case MorphologyRepresentation::contour:
237  {
238  _buildContours(container, somas);
239  break;
240  }
241  case MorphologyRepresentation::surface:
242  {
243  _buildSurface(somas);
244  return;
245  break;
246  }
247  default:
248  {
249  _buildSomasOnly(container, somas);
250  break;
251  }
252  }
253  containers.push_back(container);
254  }
255  else
256  {
257  const auto nbDBConnections = DBConnector::getInstance().getNbConnections();
258 
259  uint64_t neuronIndex;
260  volatile bool flag = false;
261  std::string flagMessage;
262 #pragma omp parallel for shared(flag, flagMessage) num_threads(nbDBConnections)
263  for (neuronIndex = 0; neuronIndex < somas.size(); ++neuronIndex)
264  {
265  try
266  {
267  if (flag)
268  continue;
269 
270  if (omp_get_thread_num() == 0)
271  {
272  PLUGIN_PROGRESS("Loading neurons...", neuronIndex, somas.size() / nbDBConnections);
273  try
274  {
275  callback.updateProgress("Loading neurons...",
276  0.5f *
277  ((float)neuronIndex / ((float)(somas.size() / nbDBConnections))));
278  }
279  catch (...)
280  {
281 #pragma omp critical
282  flag = true;
283  }
284  }
285 
286  auto it = somas.begin();
287  std::advance(it, neuronIndex);
288  const auto& soma = it->second;
289  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation, _scale);
290  _buildMorphology(container, it->first, soma, neuronIndex);
291 
292 #pragma omp critical
293  containers.push_back(container);
294  }
295  catch (const std::runtime_error& e)
296  {
297 #pragma omp critical
298  flagMessage = e.what();
299 #pragma omp critical
300  flag = true;
301  }
302  catch (...)
303  {
304 #pragma omp critical
305  flagMessage = "Loading was canceled";
306 #pragma omp critical
307  flag = true;
308  }
309  }
310  }
311 
312  for (uint64_t i = 0; i < containers.size(); ++i)
313  {
314  PLUGIN_PROGRESS("- Compiling 3D geometry...", i, containers.size());
315  callback.updateProgress("Compiling 3D geometry...", 0.5f + 0.5f * (float)(1 + i) / (float)containers.size());
316  auto& container = containers[i];
317  container.commitToModel();
318  }
319  model->applyDefaultColormap();
320  if (_details.simulationReportId != -1)
321  setDefaultTransferFunction(*model, DEFAULT_SIMULATION_VALUE_RANGE);
322 
323  ModelMetadata metadata = {{"Number of Neurons", std::to_string(somas.size())},
324  {"Number of Spines", std::to_string(_nbSpines)},
325  {"SQL node filter", _details.sqlNodeFilter},
326  {"SQL section filter", _details.sqlSectionFilter},
327  {"Max distance to soma", std::to_string(_maxDistanceToSoma)},
328  {"Min soma radius", std::to_string(_minMaxSomaRadius.x)},
329  {"Max soma radius", std::to_string(_minMaxSomaRadius.y)}};
330 
331  if (!_simulationReport.description.empty())
332  metadata["Simulation " + reportTypeAsString[_simulationReport.type] + " report"] =
333  _simulationReport.description;
334 
335  _modelDescriptor.reset(new core::ModelDescriptor(std::move(model), _details.assemblyName, metadata));
336  if (!_modelDescriptor)
337  PLUGIN_THROW("Neurons model could not be created");
338 }

340 void Neurons::_buildSomasOnly(ThreadSafeContainer& container, const NeuronSomaMap& somas)
341 {
342  uint64_t progress = 0;
343  uint64_t i = 0;
344  _minMaxSomaRadius = Vector2d(_details.radiusMultiplier, _details.radiusMultiplier);
345  for (const auto soma : somas)
346  {
347  PLUGIN_PROGRESS("Loading somas", progress, somas.size());
348  ++progress;
349 
350  const auto useSdf =
351  andCheck(static_cast<uint32_t>(_details.realismLevel), static_cast<uint32_t>(MorphologyRealismLevel::soma));
352  const auto baseMaterialId =
353  _details.populationColorScheme == PopulationColorScheme::id ? i * NB_MATERIALS_PER_MORPHOLOGY : 0;
354  if (_details.showMembrane)
355  {
356  uint64_t somaUserData = NO_USER_DATA;
357  const auto neuronId = soma.first;
358  switch (_simulationReport.type)
359  {
360  case ReportType::spike:
361  case ReportType::soma:
362  {
363  if (_simulationReport.guids.empty())
364  somaUserData = neuronId;
365  else
366  {
367  const auto it = _simulationReport.guids.find(neuronId);
368  if (it == _simulationReport.guids.end() && !_details.loadNonSimulatedNodes)
369  continue; // Ignore non-simulated nodes
370  somaUserData = (*it).second;
371  }
372  break;
373  }
374  }
375 
376  const Vector3d position = soma.second.position;
377  container.addSphere(position, _details.radiusMultiplier, baseMaterialId, useSdf, somaUserData, {},
378  Vector3f(_getDisplacementValue(DisplacementElement::morphology_soma_strength),
379  _getDisplacementValue(DisplacementElement::morphology_soma_frequency), 0.f));
380  }
381  if (_details.generateInternals)
382  {
383  const double mitochondriaDensity =
384  (soma.second.layer < MITOCHONDRIA_DENSITY.size() ? MITOCHONDRIA_DENSITY[soma.second.layer] : 0.0);
385 
386  const auto useSdf = andCheck(static_cast<uint32_t>(_details.realismLevel),
387  static_cast<uint32_t>(MorphologyRealismLevel::internals));
388  _addSomaInternals(container, baseMaterialId, soma.second.position, _details.radiusMultiplier,
389  mitochondriaDensity, useSdf, _details.radiusMultiplier);
390  }
391  ++i;
392  }

410 SectionSynapseMap Neurons::_buildDebugSynapses(const uint64_t neuronId, const SectionMap& sections)
411 {
412  SectionSynapseMap synapses;
413  for (const auto& section : sections)
414  {
415  if (static_cast<NeuronSectionType>(section.second.type) == NeuronSectionType::axon)
416  continue;
417 
418  // Process points according to representation
419  const auto points = _getProcessedSectionPoints(_details.morphologyRepresentation, section.second.points);
420  double sectionLength = 0.0;
421  doubles segmentEnds;
422  for (size_t i = 0; i < points.size() - 1; ++i)
423  {
424  const double segmentLength = length(points[i + 1] - points[i]);
425  sectionLength += segmentLength;
426  segmentEnds.push_back(sectionLength);
427  }
428  const size_t potentialNumberOfSynapses =
429  DEFAULT_DEBUG_SYNAPSE_DENSITY_RATIO * sectionLength / DEFAULT_SPINE_RADIUS + 1;
430  size_t effectiveNumberOfSynapses = potentialNumberOfSynapses * (0.5 + 0.5 * rnd0());
431 
432  for (size_t i = 0; i < effectiveNumberOfSynapses; ++i)
433  {
434  const double distance = rnd0() * sectionLength;
435  size_t segmentId = 0;
436  while (distance > segmentEnds[segmentId] && segmentId < segmentEnds.size())
437  ++segmentId;
438 
439  const auto preSynapticSectionId = section.first;
440  const auto preSynapticSegmentId = segmentId;
441  Synapse synapse;
442  synapse.type = (rand() % 2 == 0 ? MorphologySynapseType::afferent : MorphologySynapseType::efferent);
443  synapse.preSynapticSegmentDistance = distance - (segmentId > 0 ? segmentEnds[segmentId - 1] : 0.f);
444  synapse.postSynapticNeuronId = neuronId;
445  synapse.postSynapticSectionId = 0;
446  synapse.postSynapticSegmentId = 0;
447  synapse.postSynapticSegmentDistance = 0.0;
448  synapses[preSynapticSectionId][preSynapticSegmentId].push_back(synapse);
449  }
450  }
451  return synapses;
452 }

454 void Neurons::_buildMorphology(ThreadSafeContainer& container, const uint64_t neuronId, const NeuronSoma& soma,
455  const uint64_t neuronIndex)
456 {
457  const auto& connector = DBConnector::getInstance();
458 
459  const auto& somaRotation = soma.rotation;
460  const auto layer = soma.layer;
461  const double mitochondriaDensity = (layer < MITOCHONDRIA_DENSITY.size() ? MITOCHONDRIA_DENSITY[layer] : 0.0);
462 
463  SectionMap sections;
464 
465  // Soma radius
466  double somaRadius = _getCorrectedRadius(1.f, _details.radiusMultiplier);
467  if (_details.loadAxon || _details.loadApicalDendrites || _details.loadBasalDendrites)
468  {
469  sections = connector.getNeuronSections(_details.populationName, neuronId, _details.sqlSectionFilter);
470  double count = 0.0;
471  for (const auto& section : sections)
472  if (section.second.parentId == SOMA_AS_PARENT)
473  {
474  const Vector3d point = section.second.points[0];
475  somaRadius += 0.5 * length(point);
476  count += 1.0;
477  }
478  if (count > 0.0)
479  somaRadius /= count;
480  _minMaxSomaRadius.x = std::min(_minMaxSomaRadius.x, somaRadius);
481  _minMaxSomaRadius.y = std::max(_minMaxSomaRadius.y, somaRadius);
482  somaRadius = _getCorrectedRadius(std::min(somaRadius, MAX_SOMA_RADIUS), _details.radiusMultiplier);
483  }
484  const auto somaPosition = _animatedPosition(Vector4d(soma.position, somaRadius), neuronId);
485 
486  size_t baseMaterialId;
487  switch (_details.populationColorScheme)
488  {
489  case PopulationColorScheme::none:
490  baseMaterialId = 0;
491  break;
492  case PopulationColorScheme::id:
493  baseMaterialId = neuronIndex * NB_MATERIALS_PER_MORPHOLOGY;
494  break;
495  }
496 
497  size_t somaMaterialId;
498  switch (_details.morphologyColorScheme)
499  {
500  case MorphologyColorScheme::none:
501  somaMaterialId = baseMaterialId;
502  break;
503  case MorphologyColorScheme::section_type:
504  somaMaterialId = baseMaterialId + MATERIAL_OFFSET_SOMA;
505  break;
506  case MorphologyColorScheme::section_orientation:
507  somaMaterialId = getMaterialIdFromOrientation({1.0, 1.0, 1.0});
508  break;
509  case MorphologyColorScheme::distance_to_soma:
510  somaMaterialId = 0;
511  break;
512  }
513 
514  // Soma
515  uint64_t somaUserData = NO_USER_DATA;
516  switch (_simulationReport.type)
517  {
518  case ReportType::compartment:
519  {
520  const auto compartments =
521  connector.getNeuronSectionCompartments(_details.populationName, _details.simulationReportId, neuronId, 0);
522  if (!compartments.empty())
523  somaUserData = compartments[0];
524  break;
525  }
526  case ReportType::spike:
527  case ReportType::soma:
528  {
529  if (_simulationReport.guids.empty())
530  somaUserData = neuronId + 1;
531  else
532  {
533  const auto it = _simulationReport.guids.find(neuronId);
534  if (it == _simulationReport.guids.end() && !_details.loadNonSimulatedNodes)
535  return; // Ignore non-simulated nodes
536  somaUserData = (*it).second + 1;
537  }
538  break;
539  }
540  }
541 
542  // Load synapses for all sections
543  SectionSynapseMap synapses;
544  switch (_details.synapsesType)
545  {
546  case morphology::MorphologySynapseType::afferent:
547  {
548  synapses = connector.getNeuronAfferentSynapses(_details.populationName, neuronId);
549  break;
550  }
551  case morphology::MorphologySynapseType::debug:
552  {
553  synapses = _buildDebugSynapses(neuronId, sections);
554  break;
555  }
556  }
557 
558  // Sections (dendrites and axon)
559  Neighbours somaNeighbours;
560  double correctedSomaRadius = 0.f;
561  for (const auto& section : sections)
562  {
563  Neighbours sectionNeighbours;
564  const auto sectionType = static_cast<NeuronSectionType>(section.second.type);
565  const auto& points = section.second.points;
566  bool useSdf = andCheck(static_cast<uint32_t>(_details.realismLevel), static_cast<uint32_t>(sectionType));
567 
568  double distanceToSoma = 0.0;
569  if (_details.maxDistanceToSoma > 0.0)
570  // If maxDistanceToSoma != 0, then compute actual distance from soma
571  distanceToSoma = _getDistanceToSoma(sections, section.second);
572 
573  if (sectionType == NeuronSectionType::axon && !_details.loadAxon)
574  continue;
575  if (sectionType == NeuronSectionType::basal_dendrite && !_details.loadBasalDendrites)
576  continue;
577  if (sectionType == NeuronSectionType::apical_dendrite && !_details.loadApicalDendrites)
578  continue;
579  if (_details.morphologyRepresentation == MorphologyRepresentation::graph)
580  {
581  if (distanceToSoma <= _details.maxDistanceToSoma)
582  _addArrow(container, neuronIndex, somaPosition, somaRotation, section.second.points[0],
583  section.second.points[section.second.points.size() - 1], sectionType, baseMaterialId,
584  distanceToSoma);
585  continue;
586  }
587 
588  uint64_t count = 0.0;
589  // Sections connected to the soma
590  if (_details.showMembrane && _details.loadSomas && section.second.parentId == SOMA_AS_PARENT)
591  {
592  const auto& points = section.second.points;
593  const auto& firstPoint = points[0];
594  const auto& lastPoint = points[points.size() - 1];
