58 inline void sphereClipRay(const uniform vec4f* clipPlanes, const unsigned int numClipPlanes,
59  const varying vec3f& position, const varying vec3f& direction, varying float& near,
60  varying float& far)
61 {
62  if (numClipPlanes == 0)
63  return;
64 
65  near = far = 0.f;
66 
67  for (unsigned int i = 0; i < numClipPlanes; ++i)
68  {
69  const float r = dot(direction, direction);
70  if (r == 0.f)
71  continue;
72 
73  const vec3f center = {clipPlanes[i].x, clipPlanes[i].y, clipPlanes[i].z};
74  const float radius = clipPlanes[i].w;
75 
76  const float rd2 = 1.f / r;
77  const vec3f CO = center - position;
78  const float projCO = dot(CO, direction) * rd2;
79  const vec3f perp = CO - projCO * direction;
80  const float l2 = dot(perp, perp);
81  const float r2 = sqr(radius);
82  if (l2 > r2)
83  continue;
84 
85  const float a = (r2 - l2) * rd2;
86  if (a < 0.f)
87  continue;
88 
89  float td = sqrt(a);
90  const float t1 = projCO - td;
91  const float t2 = projCO + td;
92  near = max(near, min(t1, t2));
93  far = max(far, max(t1, t2));
94  }
95 }

97 void SphereClippingPerspectiveCamera_initRay(uniform Camera* uniform _self, varying Ray& ray,
98  const varying CameraSample& sample)
99 {
100  uniform SphereClippingPerspectiveCamera* uniform self = (uniform SphereClippingPerspectiveCamera * uniform) _self;
101 
102  vec2f screen = sample.screen;
103  vec3f org = self->org;
104 
105  screen = Camera_subRegion(_self, screen);
106 
107  vec3f dir = self->dir_00 + screen.x * self->dir_du + screen.y * self->dir_dv;
108 
109  if (self->super.doesDOF)
110  {
111  float random = 1.f;
112  random = getRandomValue();
113  random = (random == 0.f ? 0.001f : random);
114  const vec3f llp = uniformSampleDisk(random * self->scaledAperture, sample.lens);
115  // transform local lens point to focal plane (dir_XX are prescaled in
116  // this case)
117  const vec3f lp = (llp.x * self->dir_du) + ((llp.y * self->aspect) * self->dir_dv);
118  org = org + lp;
119  dir = dir - lp;
120  }
121 
122  varying float nearClip = self->super.nearClip;
123  varying float farClip = inf;
124  const varying vec3f direction = normalize(dir);
125 
126  sphereClipRay(self->clipPlanes, self->numClipPlanes, org, direction, nearClip, farClip);
127 
128  const float time = Camera_shutterTime(_self, sample.time);
129  setRay(ray, org, direction, nearClip, farClip, time);
130 }
