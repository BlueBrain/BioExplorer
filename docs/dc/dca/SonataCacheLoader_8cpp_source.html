107 ModelDescriptorPtr SonataCacheLoader::importFromStorage(const std::string& path, const LoaderProgress& callback,
108  const PropertyMap& properties) const
109 {
110  PropertyMap props = _defaults;
111  props.merge(properties);
112 
113  callback.updateProgress("Loading cache...", 0);
114  PLUGIN_INFO("Loading model from cache file: " << path);
115  std::ifstream file(path, std::ios::in | std::ios::binary);
116  if (!file.good())
117  PLUGIN_THROW("Could not open cache file " + path);
118 
119  // File version
120  size_t version;
121  file.read((char*)&version, sizeof(size_t));
122 
123  PLUGIN_INFO("Version: " << version);
124 
125  auto model = _scene.createModel();
126 
127  // Geometry
128  size_t nbSpheres = 0;
129  size_t nbCylinders = 0;
130  size_t nbCones = 0;
131  size_t nbMeshes = 0;
132  size_t nbVertices = 0;
133  size_t nbIndices = 0;
134  size_t nbNormals = 0;
135  size_t nbTexCoords = 0;
136 
137  // Metadata
138  size_t nbElements;
139  ModelMetadata metadata;
140  file.read((char*)&nbElements, sizeof(size_t));
141  for (size_t i = 0; i < nbElements; ++i)
142  metadata[_readString(file)] = _readString(file);
143 
144  size_t nbMaterials;
145  file.read((char*)&nbMaterials, sizeof(size_t));
146 
147  // Materials
148  size_t materialId;
149  for (size_t i = 0; i < nbMaterials; ++i)
150  {
151  callback.updateProgress("Materials (" + std::to_string(i + 1) + "/" + std::to_string(nbMaterials) + ")",
152  0.1f * float(i) / float(nbMaterials));
153  file.read((char*)&materialId, sizeof(size_t));
154 
155  auto name = _readString(file);
156  auto material = model->createMaterial(materialId, name);
157 
158  Vector3f value3f;
159  file.read((char*)&value3f, sizeof(Vector3f));
160  material->setDiffuseColor(value3f);
161  file.read((char*)&value3f, sizeof(Vector3f));
162  material->setSpecularColor(value3f);
163  float value;
164  file.read((char*)&value, sizeof(float));
165  material->setSpecularExponent(value);
166  file.read((char*)&value, sizeof(float));
167  material->setReflectionIndex(value);
168  file.read((char*)&value, sizeof(float));
169  material->setOpacity(value);
170  file.read((char*)&value, sizeof(float));
171  material->setRefractionIndex(value);
172  file.read((char*)&value, sizeof(float));
173  material->setEmission(value);
174  file.read((char*)&value, sizeof(float));
175  material->setGlossiness(value);
176 
177  if (version == CACHE_VERSION_1)
178  {
179  bool userData;
180  file.read((char*)&userData, sizeof(bool));
181  material->setCastUserData(userData);
182 
183  size_t shadingMode;
184  file.read((char*)&shadingMode, sizeof(size_t));
185  material->setShadingMode(static_cast<MaterialShadingMode>(shadingMode));
186  }
187 
188  if (version >= CACHE_VERSION_2)
189  {
190  int32_t userData;
191  file.read((char*)&userData, sizeof(int32_t));
192  material->setCastUserData(userData);
193 
194  int32_t shadingMode;
195  file.read((char*)&shadingMode, sizeof(int32_t));
196  material->setShadingMode(static_cast<MaterialShadingMode>(shadingMode));
197  }
198 
199  if (version == CACHE_VERSION_3)
200  {
201  bool clipped;
202  file.read((char*)&clipped, sizeof(bool));
203  material->setClippingMode(clipped ? MaterialClippingMode::plane : MaterialClippingMode::no_clipping);
204  }
205 
206  if (version >= CACHE_VERSION_4)
207  {
208  int32_t clippingMode;
209  file.read((char*)&clippingMode, sizeof(int32_t));
210  material->setClippingMode(static_cast<MaterialClippingMode>(clippingMode));
211  }
212  }
213 
214  uint64_t bufferSize{0};
215 
216  // Spheres
217  callback.updateProgress("Spheres", 0.2f);
218  file.read((char*)&nbSpheres, sizeof(size_t));
219  for (size_t i = 0; i < nbSpheres; ++i)
220  {
221  file.read((char*)&materialId, sizeof(size_t));
222  file.read((char*)&nbElements, sizeof(size_t));
223 
224  struct SphereV1
225  {
226  Vector3f center;
227  float radius;
228  float timestamp;
229  float value;
230  };
231 
232  if (props.getProperty<bool>(PROP_LOAD_SPHERES.name))
233  {
234  callback.updateProgress("Spheres (" + std::to_string(i + 1) + "/" + std::to_string(nbSpheres) + ")",
235  0.2f + 0.1f * float(i) / float(nbSpheres));
236  auto& spheres = model->getSpheres()[materialId];
237  spheres.resize(nbElements);
238 
239  if (version >= CACHE_VERSION_2)
240  {
241  bufferSize = nbElements * sizeof(Sphere);
242  file.read((char*)spheres.data(), bufferSize);
243  }
244  else
245  {
246  std::vector<SphereV1> spheresV1;
247  spheresV1.resize(nbElements);
248  bufferSize = nbElements * sizeof(SphereV1);
249  file.read((char*)spheresV1.data(), bufferSize);
250  for (uint64_t s = 0; s < spheresV1.size(); ++s)
251  spheres[s] = {spheresV1[i].center, spheresV1[i].radius};
252  }
253  }
254  else
255  {
256  if (version >= CACHE_VERSION_2)
257  bufferSize = nbElements * sizeof(Sphere);
258  else
259  bufferSize = nbElements * sizeof(SphereV1);
260  file.ignore(bufferSize);
261  }
262  }
263 
264  // Cylinders
265  file.read((char*)&nbCylinders, sizeof(size_t));
266  for (size_t i = 0; i < nbCylinders; ++i)
267  {
268  file.read((char*)&materialId, sizeof(size_t));
269  file.read((char*)&nbElements, sizeof(size_t));
270 
271  struct CylinderV1
272  {
273  Vector3f center;
274  Vector3f up;
275  float radius;
276  float timestamp;
277  float value;
278  };
279 
280  if (props.getProperty<bool>(PROP_LOAD_CYLINDERS.name))
281  {
282  callback.updateProgress("Cylinders (" + std::to_string(i + 1) + "/" + std::to_string(nbCylinders) + ")",
283  0.3f + 0.1f * float(i) / float(nbCylinders));
284  auto& cylinders = model->getCylinders()[materialId];
285  cylinders.resize(nbElements);
286  if (version >= CACHE_VERSION_2)
287  {
288  bufferSize = nbElements * sizeof(Cylinder);
289  file.read((char*)cylinders.data(), bufferSize);
290  }
291  else
292  {
293  std::vector<CylinderV1> cylindersV1(nbElements);
294  bufferSize = nbElements * sizeof(CylinderV1);
295  file.read((char*)cylindersV1.data(), bufferSize);
296  for (uint64_t s = 0; s < cylindersV1.size(); ++s)
297  cylinders[s] = {cylindersV1[i].center, cylindersV1[i].up, cylindersV1[i].radius};
298  }
299  }
300  else
301  {
302  if (version >= CACHE_VERSION_2)
303  bufferSize = nbElements * sizeof(Cylinder);
304  else
305  bufferSize = nbElements * sizeof(CylinderV1);
306  file.ignore(bufferSize);
307  }
308  }
309 
310  // Cones
311  file.read((char*)&nbCones, sizeof(size_t));
312  for (size_t i = 0; i < nbCones; ++i)
313  {
314  file.read((char*)&materialId, sizeof(size_t));
315  file.read((char*)&nbElements, sizeof(size_t));
316 
317  struct ConeV1
318  {
319  Vector3f center;
320  Vector3f up;
321  float centerRadius;
322  float upRadius;
323  float timestamp;
324  float value;
325  };
326 
327  if (props.getProperty<bool>(PROP_LOAD_CONES.name))
328  {
329  callback.updateProgress("Cones (" + std::to_string(i + 1) + "/" + std::to_string(nbCones) + ")",
330  0.4f + 0.1f * float(i) / float(nbCones));
331  auto& cones = model->getCones()[materialId];
332  cones.resize(nbElements);
333  if (version >= CACHE_VERSION_2)
334  {
335  bufferSize = nbElements * sizeof(Cone);
336  file.read((char*)cones.data(), bufferSize);
337  }
338  else
339  {
340  std::vector<ConeV1> conesV1(nbElements);
341  bufferSize = nbElements * sizeof(ConeV1);
342  file.read((char*)conesV1.data(), bufferSize);
343  for (uint64_t s = 0; s < conesV1.size(); ++s)
344  cones[s] = {conesV1[i].center, conesV1[i].up, conesV1[i].centerRadius, conesV1[i].upRadius};
345  }
346  }
347  else
348  {
349  if (version >= CACHE_VERSION_2)
350  bufferSize = nbElements * sizeof(Cone);
351  else
352  bufferSize = nbElements * sizeof(ConeV1);
353 
354  file.ignore(bufferSize);
355  }
356  }
357 
358  // Meshes
359  bool load = props.getProperty<bool>(PROP_LOAD_MESHES.name);
360  file.read((char*)&nbMeshes, sizeof(size_t));
361  for (size_t i = 0; i < nbMeshes; ++i)
362  {
363  file.read((char*)&materialId, sizeof(size_t));
364  auto& meshes = model->getTriangleMeshes()[materialId];
365  // Vertices
366  file.read((char*)&nbVertices, sizeof(size_t));
367  if (nbVertices != 0)
368  {
369  bufferSize = nbVertices * sizeof(Vector3f);
370  if (load)
371  {
372  callback.updateProgress("Meshes (" + std::to_string(i + 1) + "/" + std::to_string(nbMeshes) + ")",
373  0.5f + 0.1f * float(i) / float(nbMeshes));
374  meshes.vertices.resize(nbVertices);
375  file.read((char*)meshes.vertices.data(), bufferSize);
376  }
377  else
378  file.ignore(bufferSize);
379  }
380 
381  // Indices
382  file.read((char*)&nbIndices, sizeof(size_t));
383  if (nbIndices != 0)
384  {
385  bufferSize = nbIndices * sizeof(Vector3ui);
386  if (load)
387  {
388  meshes.indices.resize(nbIndices);
389  file.read((char*)meshes.indices.data(), bufferSize);
390  }
391  else
392  file.ignore(bufferSize);
393  }
394 
395  // Normals
396  file.read((char*)&nbNormals, sizeof(size_t));
397  if (nbNormals != 0)
398  {
399  bufferSize = nbNormals * sizeof(Vector3f);
400  if (load)
401  {
402  meshes.normals.resize(nbNormals);
403  file.read((char*)meshes.normals.data(), bufferSize);
404  }
405  else
406  file.ignore(bufferSize);
407  }
408 
409  // Texture coordinates
410  file.read((char*)&nbTexCoords, sizeof(size_t));
411  if (nbTexCoords != 0)
412  {
413  bufferSize = nbTexCoords * sizeof(Vector2f);
414  if (load)
415  {
416  meshes.textureCoordinates.resize(nbTexCoords);
417  file.read((char*)meshes.textureCoordinates.data(), bufferSize);
418  }
419  else
420  file.ignore(bufferSize);
421  }
422  }
423 
424  // Streamlines
425  load = props.getProperty<bool>(PROP_LOAD_STREAMLINES.name);
426  size_t nbStreamlines;
427  auto& streamlines = model->getStreamlines();
428  file.read((char*)&nbStreamlines, sizeof(size_t));
429  for (size_t i = 0; i < nbStreamlines; ++i)
430  {
431  StreamlinesData streamlineData;
432  // Id
433  size_t id;
434  file.read((char*)&id, sizeof(size_t));
435 
436  // Vertex
437  file.read((char*)&nbElements, sizeof(size_t));
438  bufferSize = nbElements * sizeof(Vector4f);
439  if (load)
440  {
441  callback.updateProgress("Streamlines (" + std::to_string(i + 1) + "/" + std::to_string(nbStreamlines) + ")",
442  0.6f + 0.1f * float(i) / float(nbStreamlines));
443  streamlineData.vertex.resize(nbElements);
444  file.read((char*)streamlineData.vertex.data(), bufferSize);
445  }
446  else
447  file.ignore(bufferSize);
448 
449  // Vertex Color
450  file.read((char*)&nbElements, sizeof(size_t));
451  bufferSize = nbElements * sizeof(Vector4f);
452  if (load)
453  {
454  streamlineData.vertexColor.resize(nbElements);
455  file.read((char*)streamlineData.vertexColor.data(), bufferSize);
456  }
457  else
458  file.ignore(bufferSize);
459 
460  // Indices
461  file.read((char*)&nbElements, sizeof(size_t));
462  bufferSize = nbElements * sizeof(int32_t);
463  if (load)
464  {
465  streamlineData.indices.resize(nbElements);
466  file.read((char*)streamlineData.indices.data(), bufferSize);
467  }
468  else
469  file.ignore(bufferSize);
470 
471  streamlines[id] = streamlineData;
472  }
473 
474  // SDF geometry
475  load = props.getProperty<bool>(PROP_LOAD_SDF.name);
476  file.read((char*)&nbElements, sizeof(size_t));
477 
478  if (nbElements > 0)
479  {
480  // Geometries
481  auto& sdfData = model->getSDFGeometryData();
482  sdfData.geometries.resize(nbElements);
483  bufferSize = nbElements * sizeof(SDFGeometry);
484  file.read((char*)sdfData.geometries.data(), bufferSize);
485 
486  if (version <= CACHE_VERSION_3)
487  // Update userParams (Displacement parameters)
488  for (uint64_t i = 0; i < nbElements; ++i)
489  {
490  const char* index = (char*)sdfData.geometries.data() + i * sizeof(SDFGeometry) + sizeof(uint64_t);
491  memcpy((char*)index, &neuroscience::common::DISPLACEMENT_PARAMS, sizeof(Vector3f));
492  }
493 
494  // SDF Indices
495  file.read((char*)&nbElements, sizeof(size_t));
496  for (size_t i = 0; i < nbElements; ++i)
497  {
498  file.read((char*)&materialId, sizeof(size_t));
499  size_t size;
500  file.read((char*)&size, sizeof(size_t));
501  bufferSize = size * sizeof(uint64_t);
502  if (load)
503  {
504  callback.updateProgress("SDF geometries indices (" + std::to_string(i + 1) + "/" +
505  std::to_string(nbElements) + ")",
506  0.8f + 0.1f * float(i) / float(nbElements));
507  sdfData.geometryIndices[materialId].resize(size);
508  file.read((char*)sdfData.geometryIndices[materialId].data(), bufferSize);
509  }
510  else
511  file.ignore(bufferSize);
512  }
513 
514  // Neighbours
515  file.read((char*)&nbElements, sizeof(size_t));
516  sdfData.neighbours.resize(nbElements);
517 
518  if (load)
519  callback.updateProgress("SDF geometries neighbours", 0.9f);
520 
521  for (size_t i = 0; i < nbElements; ++i)
522  {
523  size_t size;
524  file.read((char*)&size, sizeof(size_t));
525  bufferSize = size * sizeof(uint64_t);
526  if (load)
527  {
528  sdfData.neighbours[i].resize(size);
529  file.read((char*)sdfData.neighbours[i].data(), bufferSize);
530  }
531  else
532  file.ignore(bufferSize);
533  }
534 
535  // Neighbours flat
536  file.read((char*)&nbElements, sizeof(size_t));
537  bufferSize = nbElements * sizeof(uint64_t);
538  if (load)
539  {
540  sdfData.neighboursFlat.resize(nbElements);
541  file.read((char*)sdfData.neighboursFlat.data(), bufferSize);
542  }
543  else
544  file.ignore(bufferSize);
545  }
546 
547  load = props.getProperty<bool>(PROP_LOAD_SIMULATION.name);
548  if (version >= CACHE_VERSION_3 && load)
549  {
550  // Simulation Handler
551  size_t reportType{0};
552  file.read((char*)&reportType, sizeof(size_t));
553 
554  switch (static_cast<neuroscience::common::ReportType>(reportType))
555  {
556  case neuroscience::common::ReportType::voltages_from_file:
557  {
558  // Report path
559  const auto reportPath = _readString(file);
560 
561  // GIDs
562  file.read((char*)&nbElements, sizeof(size_t));
563  brion::GIDSet gids;
564  for (uint32_t i = 0; i < nbElements; ++i)
565  {
566  uint32_t gid;
567  file.read((char*)&gid, sizeof(uint32_t));
568  gids.insert(gid);
569  }
570 
571  // Synchronization
572  bool synchronized{false};
573  file.read((char*)&synchronized, sizeof(bool));
574 
575  // Handler
576  auto handler = std::make_shared<VoltageSimulationHandler>(reportPath, gids, synchronized);
577  model->setSimulationHandler(handler);
578  break;
579  }
580  case neuroscience::common::ReportType::spikes:
581  {
582  // Report path
583  const auto reportPath = _readString(file);
584 
585  // GIDs
586  file.read((char*)&nbElements, sizeof(size_t));
587  brion::GIDSet gids;
588  for (uint32_t i = 0; i < nbElements; ++i)
589  {
590  uint32_t gid;
591  file.read((char*)&gid, sizeof(uint32_t));
592  gids.insert(gid);
593  }
594 
595  // Handler
596  auto handler = std::make_shared<SpikeSimulationHandler>(reportPath, gids);
597  model->setSimulationHandler(handler);
598  break;
599  }
600  default:
601  {
602  // No report in that brick!
603  }
604  }
605 
606  // Transfer function
607  file.read((char*)&nbElements, sizeof(size_t));
608  if (nbElements == 1)
609  {
610  auto& tf = model->getTransferFunction();
611  // Values range
612  Vector2d valuesRange;
613  file.read((char*)&valuesRange, sizeof(Vector2d));
614  tf.setValuesRange(valuesRange);
615 
616  // Control points
617  file.read((char*)&nbElements, sizeof(size_t));
618  Vector2ds controlPoints(nbElements);
619  file.read((char*)&controlPoints[0], nbElements * sizeof(Vector2d));
620  tf.setControlPoints(controlPoints);
621 
622  // Color map
623  ColorMap colorMap;
624  colorMap.name = _readString(file);
625  file.read((char*)&nbElements, sizeof(size_t));
626  auto& colors = colorMap.colors;
627  colors.resize(nbElements);
628  file.read((char*)&colors[0], nbElements * sizeof(Vector3f));
629  tf.setColorMap(colorMap);
630  }
631  }
632  callback.updateProgress("Done", 1.f);
633 
634  file.close();
635 
636  // Restore original circuit config file from cache metadata, if present
637  std::string circuitPath = path;
638  auto cpIt = metadata.find("CircuitPath");
639  if (cpIt != metadata.end())
640  circuitPath = cpIt->second;
641 
642  auto modelDescriptor = std::make_shared<ModelDescriptor>(std::move(model), "Brick", circuitPath, metadata);
643  return modelDescriptor;
644 }

646 void SonataCacheLoader::exportToFile(const ModelDescriptorPtr modelDescriptor, const std::string& filename)
647 {
648  PLUGIN_INFO("Saving model to cache file: " << filename);
649  std::ofstream file(filename, std::ios::out | std::ios::binary);
650  if (!file.good())
651  {
652  const std::string msg = "Could not open cache file " + filename;
653  PLUGIN_THROW(msg);
654  }
655 
656  const size_t version = CACHE_VERSION_3;
657  file.write((char*)&version, sizeof(size_t));
658 
659  // Save geometry
660  auto& model = modelDescriptor->getModel();
661  uint64_t bufferSize{0};
662 
663  // Metadata
664  auto metadata = modelDescriptor->getMetadata();
665  size_t nbElements = metadata.size();
666  file.write((char*)&nbElements, sizeof(size_t));
667  for (const auto& data : metadata)
668  {
669  size_t size = data.first.length();

910  }

940  nbElements = 0;

942  }
943

945  nbElements = 1;
