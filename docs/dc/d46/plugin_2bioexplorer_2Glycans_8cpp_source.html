33 Glycans::Glycans(Scene& scene, const SugarsDetails& sd)
34  : Molecule(scene, {})
35  , _details(sd)
36 {
37  size_t lineIndex{0};
38 
39  std::stringstream lines{_details.contents};
40  std::string line;
41  std::string title{sd.name};
42  std::string header{sd.name};
43 
44  while (getline(lines, line, '\n'))
45  {
46  if (line.find(KEY_ATOM) == 0 || line.find(KEY_HETATM) == 0)
47  _readAtom(line, true);
48  else if (line.find(KEY_HEADER) == 0)
49  header = _readHeader(line);
50  else if (line.find(KEY_TITLE) == 0)
51  title = _readTitle(line);
52  }
53  auto model = scene.createModel();
54 
55  // Build 3d models according to atoms positions (re-centered to origin)
56  Boxf bounds;
57 
58  // Recenter
59  if (_details.recenter)
60  {
61  // Get current center
62  for (const auto& atom : _atomMap)
63  bounds.merge(atom.second.position);
64  const auto& center = bounds.getCenter();
65 
66  const auto firstAtomPosition = _atomMap.begin()->second.position;
67  const auto translation = center - firstAtomPosition;
68  // Translate according to position of first atom
69  for (auto& atom : _atomMap)
70  atom.second.position -= translation;
71  }
72 
73  _buildModel(_details.assemblyName, _details.name, title, header,
74  _details.representation, _details.atomRadiusMultiplier,
75  _details.loadBonds);
76 }
