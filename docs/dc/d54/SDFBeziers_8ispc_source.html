56 inline uniform float bezier(uniform float t, uniform float p0, uniform float c0, uniform float p1, uniform float c1)
57 {
58  const uniform float s = 1 - t;
59  const uniform float s2 = s * s;
60  const uniform float t2 = t * t;
61  return s * s2 * p0 + 3 * s2 * t * c0 + 3 * s * t2 * c1 + t * t2 * p1;
62 }
63 
64 inline vec3f bezier(float t, const uniform vec3f& p0, const uniform vec3f& c0, const uniform vec3f& p1,
65  const uniform vec3f& c1)
66 {
67  const vec3f s = make_vec3f(1 - t);
68  const vec3f s2 = s * s;
69  const vec3f t2 = make_vec3f(t * t);
70  return s * s2 * p0 + 3 * s2 * t * c0 + 3 * s * t2 * c1 + t * t2 * p1;
71 }

75 inline float sdf_tube(const vec3f& P, const vec3f& P0, const vec3f& P1, float r0_, float r1_, float t, float deltaT,
76  const float epsilon)
77 {
78 #define TANGENTIAL_TUBE
79 #ifndef TANGENTIAL_TUBE
80  const vec3f v = P1 - P0;
81  const vec3f w = P - P0;
82 
83  // distance to P0 along tube axis
84  const float c1 = dot(w, v);
85  if (c1 < 0.0f)
86  return length(P - P0) - lerp(t - deltaT, r0_, r1_);
87 
88  // tube length
89  const float c2 = dot(v, v);
90  if (c2 <= c1)
91  return length(P - P1) - lerp(t, r0_, r1_);
92 
93  const float b = c1 / c2;
94  vec3f Pb = P0 + b * v;
95 
96  const float thickness = lerp(t + (b - 1.0f) * deltaT, r0_, r1_);
97 
98  return length(P - Pb) - thickness;
99 #else
100  // based on http://sci.utah.edu/~will/papers/tubes.pdf
101 
102  // extremities radii
103  const float r0 = lerp(t - deltaT, r0_, r1_);
104  const float r1 = lerp(t, r0_, r1_);
105 
106  // cone starting and ending points
107  vec3f coneP0 = P0;
108  vec3f coneP1 = P1;
109  // cone starting and ending radii
110  float coneR0 = r0;
111  float coneR1 = r1;
112 
113  // find tangential cone to spheres if needed
114  if (abs(r0 - r1) > epsilon)
115  {
116  const vec3f V = P1 - P0;
117  const float len = length(V);
118  const vec3f C = V / len;
119 
120  // distance(P0, A)
121  const float d0 = len * r0 / (r1 - r0);
122  // distance(P2, A)
123  const float d1 = len + d0;
124  // apex point
125  const vec3f A = P0 - d0 * C;
126 
127  // distance (coneP0, A)
128  const float z0 = d0 - r0 * r0 / d0;
129  // distance (coneP1, A)
130  const float z1 = d1 - r1 * r1 / d1;
131 
132  // tangential cone starting and ending points
133  coneP0 = A + z0 * C;
134  coneP1 = A + z1 * C;
135  // tangential cone starting and ending radii
136  coneR0 = sqrt(sqr(r0) - sqr(d0 - z0));
137  coneR1 = sqrt(sqr(r1) - sqr(d1 - z1));
138  }
139 
140  const vec3f v = coneP1 - coneP0;
141  const vec3f w = P - coneP0;
142 
143  // distance to P0 along tube axis
144  const float c1 = dot(w, v);
145  if (c1 <= 0.0f)
146  return length(P - P0) - r0;
147 
148  // tube length
149  const float c2 = dot(v, v);
150  if (c2 <= c1)
151  return length(P - P1) - r1;
152 
153  const float b = c1 / c2;
154  vec3f Pb = coneP0 + b * v;
155 
156  const float thickness = lerp(b, coneR0, coneR1);
157 
158  return length(P - Pb) - thickness;
159 #endif
160 }

185 uniform box3fa bezierBounds(uDataPtr_t geo_, uDataPtr_t bezier_)
186 {
187 #define USE_TIGHT_BOUNDS
188 #ifndef USE_TIGHT_BOUNDS
189  const Bezier_ptr bc = (const Bezier_ptr)bezier_;
190  const uniform vec3f mi = min(min(bc->p0 - bc->r0, bc->p1 - bc->r1), min(bc->c0 - bc->r0, bc->c1 - bc->r1));
191  const uniform vec3f ma = max(max(bc->p0 + bc->r0, bc->p1 + bc->r1), max(bc->c0 + bc->r0, bc->c1 + bc->r1));
192  return make_box3fa(mi, ma);
193 #else
194  const Bezier_ptr bc = (const Bezier_ptr)bezier_;
195 
196  uniform float r = max(bc->r0, bc->r1);
197 
198  // tight bounding box
199  uniform vec3f mi = min(bc->p0, bc->p1);
200  uniform vec3f ma = max(bc->p0, bc->p1);
201 
202  const uniform vec3f c = -1.0f * bc->p0 + bc->c0;
203  const uniform vec3f b = 1.0f * bc->p0 - 2 * bc->c0 + bc->c1;
204  const uniform vec3f a = -1.0f * bc->p0 + 3 * bc->c0 - 3 * bc->c1 + bc->p1;
205 
206  uniform vec3f h = b * b - a * c;
207 
208  if (h.x > 0.0f)
209  {
210  h.x = sqrt(h.x);
211  uniform float t = (-b.x - h.x) / a.x;
212  if (!isnan(t) && t > 0.0f && t < 1.0f)
213  {
214  uniform float q = bezier(t, bc->p0.x, bc->c0.x, bc->p1.x, bc->c1.x);
215  mi.x = min(mi.x, q);
216  ma.x = max(ma.x, q);
217  }
218 
219  t = (-b.x + h.x) / a.x;
220  if (!isnan(t) && t > 0.0f && t < 1.0f)
221  {
222  uniform float q = bezier(t, bc->p0.x, bc->c0.x, bc->p1.x, bc->c1.x);
223  mi.x = min(mi.x, q);
224  ma.x = max(ma.x, q);
225  }
226  }
227 
228  if (h.y > 0.0f)
229  {
230  h.y = sqrt(h.y);
231  uniform float t = (-b.y - h.y) / a.y;
232  if (!isnan(t) && t > 0.0f && t < 1.0f)
233  {
234  uniform float q = bezier(t, bc->p0.y, bc->c0.y, bc->p1.y, bc->c1.y);
235  mi.y = min(mi.y, q);
236  ma.y = max(ma.y, q);
237  }
238  t = (-b.y + h.y) / a.y;
239  if (!isnan(t) && t > 0.0f && t < 1.0f)
240  {
241  uniform float q = bezier(t, bc->p0.y, bc->c0.y, bc->p1.y, bc->c1.y);
242  mi.y = min(mi.y, q);
243  ma.y = max(ma.y, q);
244  }
245  }
246 
247  if (h.z > 0.0f)
248  {
249  h.z = sqrt(h.z);
250  uniform float t = (-b.z - h.z) / a.z;
251  if (!isnan(t) && t > 0.0f && t < 1.0f)
252  {
253  uniform float q = bezier(t, bc->p0.z, bc->c0.z, bc->p1.z, bc->c1.z);
254  mi.z = min(mi.z, q);
255  ma.z = max(ma.z, q);
256  }
257  t = (-b.z + h.z) / a.z;
258  if (!isnan(t) && t > 0.0f && t < 1.0f)
259  {
260  uniform float q = bezier(t, bc->p0.z, bc->c0.z, bc->p1.z, bc->c1.z);
261  mi.z = min(mi.z, q);
262  ma.z = max(ma.z, q);
263  }
264  }
265 
266  return make_box3fa(mi - r, ma + r);
267 #endif
268 }

271 #define MIN_SEGMENT_COUNT 1
272 #define MAX_SEGMENT_COUNT 6
273 
274 int screenAwareSegmentCount(const Bezier_ptr bc, const SDFParams& params)
275 {
276  float costhetaEye = dot(normalize(bc->p0 - params.eye), normalize(bc->p1 - params.eye));
277 
278  float ratio = (1.0f - costhetaEye) / 0.5f; // cos(vfov / 180 * M_PI);
279 
280  return clamp((int)(MIN_SEGMENT_COUNT + (30 * sqrt(ratio)) * (MAX_SEGMENT_COUNT - MIN_SEGMENT_COUNT)),
281  MIN_SEGMENT_COUNT, MAX_SEGMENT_COUNT) *
282  params.segmentCountMultiplier;
283 }

290 float bezierDistance(const vec3f& p, uDataPtr_t geo_, uDataPtr_t bezier_, const SDFParams& params,
291  const bool /*processDisplacement*/)
292 {
293  const Geo_ptr geo = (const Geo_ptr)geo_;
294  const Bezier_ptr bc = (const Bezier_ptr)bezier_;
295 
296  const int segmentCount = screenAwareSegmentCount(bc, params);
297  const float deltaT = 1.0f / segmentCount;
298 
299  float minDist = 1e10f;
300 
301  vec3f prevB = bezier(0.f, bc->p0, bc->c0, bc->p1, bc->c1);
302 
303  for (int i = 1; i <= segmentCount; i++)
304  {
305  const float t = i * deltaT;
306  const vec3f b = bezier(t, bc->p0, bc->c0, bc->p1, bc->c1);
307  const float d = sdf_tube(p, prevB, b, bc->r0, bc->r1, t, deltaT, params.epsilon);
308  minDist = min(minDist, d);
309  prevB = b;
310  }
311 
312  return minDist;
313 }

327 unmasked void SDFBeziers_intersect(const RTCIntersectFunctionNArguments* uniform args)
328 {
329  const Geo_ptr geo = (Geo_ptr)args->geometryUserPtr;
330 
331  const Bezier_ptr bezier = safeIncrement(geo->useSafeIncrement, geo->data, args->primID);
332 
333  varying Ray* uniform ray = (varying Ray * uniform) args->rayhit;
334 
335  SDFParams sdfParams;
336  sdfParams.epsilon = geo->epsilon;
337  sdfParams.nbMarchIterations = geo->nbMarchIterations;
338  sdfParams.eye = ray->org;
339  sdfParams.segmentCountMultiplier = 1.f;
340  sdfParams.blendFactor = geo->blendFactor;
341  sdfParams.blendLerpFactor = geo->blendLerpFactor;
342  sdfParams.omega = geo->omega;
343  sdfParams.distance = geo->distance;
344 
345  bool processDisplacement = true;
346  const float t_in = raymarching(*ray, bezierDistance, bezierBounds, (uDataPtr_t)geo, (uDataPtr_t)bezier, sdfParams,
347  processDisplacement);
348 
349  if (t_in > 0 && t_in > ray->t0 && t_in < ray->t)
350  {
351  const vec3f pos = ray->org + t_in * ray->dir;
352 
353  ray->primID = args->primID;
354  ray->geomID = geo->super.geomID;
355  ray->instID = args->context->instID[0];
356  ray->t = t_in;
357 
358  // increase multiplier for smooth normals
359  sdfParams.segmentCountMultiplier = 5.f;
360 
361  // TODO epsilon based on pixel radius
362  ray->Ng = computeNormal(pos, bezierDistance, calcEpsilon(ray->org, t_in), (uDataPtr_t)geo, (uDataPtr_t)bezier,
363  sdfParams, processDisplacement);
364  }
365 }

369 static void SDFBeziers_postIntersect(uniform Geometry* uniform geometry, uniform Model* uniform model,
370  varying DifferentialGeometry& dg, const varying Ray& ray, uniform int64 flags)
371 {
372  Geo_ptr this = (Geo_ptr)geometry;
373  dg.geometry = geometry;
374  vec3f Ng = ray.Ng;
375  vec3f Ns = Ng;
376 
377  if (flags & DG_NORMALIZE)
378  {
379  Ng = normalize(Ng);
380  Ns = normalize(Ns);
381  }
382  if (flags & DG_FACEFORWARD)
383  {
384  if (dot(ray.dir, Ng) >= 0.f)
385  Ng = neg(Ng);
386  if (dot(ray.dir, Ns) >= 0.f)
387  Ns = neg(Ns);
388  }
389  dg.Ng = Ng;
390  dg.Ns = Ns;
391 }
