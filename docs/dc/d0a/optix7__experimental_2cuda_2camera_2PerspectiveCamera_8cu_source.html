36 static __forceinline__ __device__ void trace(OptixTraversableHandle handle,
37  float3 ray_origin,
38  float3 ray_direction, float tmin,
39  float tmax, float3* prd)
40 {
41  unsigned int p0, p1, p2;
42  p0 = __float_as_uint(prd->x);
43  p1 = __float_as_uint(prd->y);
44  p2 = __float_as_uint(prd->z);
45  optixTrace(handle, ray_origin, ray_direction, tmin, tmax,
46  0.0f, // rayTime
47  OptixVisibilityMask(1), OPTIX_RAY_FLAG_NONE,
48  0, // SBT offset
49  0, // SBT stride
50  0, // missSBTIndex
51  p0, p1, p2);
52  prd->x = __uint_as_float(p0);
53  prd->y = __uint_as_float(p1);
54  prd->z = __uint_as_float(p2);
55 }

71 extern "C" __global__ void __raygen__rg()
72 {
73  const uint3 idx = optixGetLaunchIndex();
74  const uint3 dim = optixGetLaunchDimensions();
75 
76  const RayGenData* rtData = (RayGenData*)optixGetSbtDataPointer();
77  const float3 U = rtData->camera_u;
78  const float3 V = rtData->camera_v;
79  const float3 W = rtData->camera_w;
80  const float2 d =
81  2.0f *
82  make_float2(static_cast<float>(idx.x) / static_cast<float>(dim.x),
83  static_cast<float>(idx.y) / static_cast<float>(dim.y)) -
84  1.0f;
85 
86  const float3 origin = rtData->cam_eye;
87  const float3 direction = normalize(d.x * U + d.y * V + W);
88  float3 payload_rgb = make_float3(0.5f, 0.5f, 0.5f);
89  trace(params.handle, origin, direction,
90  0.00f, // tmin
91  1e16f, // tmax
92  &payload_rgb);
93 
94  params.frame_buffer[idx.y * params.width + idx.x] = make_color(payload_rgb);
95 }

131 extern "C" __global__ void __closesthit__ch()
132 {
133  float t_hit = optixGetRayTmax();
134  // Backface hit not used.
135  // float t_hit2 = __uint_as_float( optixGetAttribute_0() );
136 
137  const float3 ray_orig = optixGetWorldRayOrigin();
138  const float3 ray_dir = optixGetWorldRayDirection();
139 
140  const unsigned int prim_idx = optixGetPrimitiveIndex();
141  const OptixTraversableHandle gas = optixGetGASTraversableHandle();
142  const unsigned int sbtGASIndex = optixGetSbtGASIndex();
143 
144  float4 q;
145  // sphere center (q.x, q.y, q.z), sphere radius q.w
146  optixGetSphereData(gas, prim_idx, sbtGASIndex, 0.f, &q);
147 
148  float3 world_raypos = ray_orig + t_hit * ray_dir;
149  float3 obj_raypos = optixTransformPointFromWorldToObjectSpace(world_raypos);
150  float3 obj_normal = (obj_raypos - make_float3(q)) / q.w;
151  float3 world_normal =
152  normalize(optixTransformNormalFromObjectToWorldSpace(obj_normal));
153 
154  setPayload(world_normal * 0.5f + 0.5f);
155 }
