55  getParamf(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name.c_str(), DEFAULT_CAMERA_INTERPUPILLARY_DISTANCE);

60  dir = normalize(dir);

66  if (stereo)

69  if (bufferTarget.length() == 2)
70  {
71  if (bufferTarget.at(1) == 'L')
72  org -= ipd_offset;
73  if (bufferTarget.at(1) == 'R')
74  org += ipd_offset;
75  }
76  }
