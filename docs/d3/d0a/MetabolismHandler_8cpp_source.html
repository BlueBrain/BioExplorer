75 void* MetabolismHandler::getFrameData(const uint32_t frame)
76 {
77  if (_currentFrame == frame)
78  return _frameData.data();
79  _currentFrame = frame;
80 
81  const auto values = _connector->getConcentrations(frame, _ids);
82 
83  _frameData.clear();
84  _frameData.push_back(frame);
85  size_t j = 0;
86  for (size_t i = 0; i < _locations.size(); ++i)
87  {
88  const auto idx = values.find(_locations[i].guid);
89  if (idx != values.end())
90  {
91  _frameData.push_back((*idx).second);
92  ++j;
93  }
94  else
95  _frameData.push_back(0.f);
96  }
97 
98  _frameSize = _frameData.size();
99 
100 #if DEBUG
101  std::string s;
102  for (uint64_t i = 1; i < _frameData.size(); ++i)
103  {
104  if (!s.empty())
105  s += ",";
106  s += "[" + std::to_string(_locations[i - 1].guid) + "] " +
107  std::to_string(_frameData[i]);
108  }
109  PLUGIN_INFO(s);
110 #endif
111 
112  return _frameData.data();
113 }
