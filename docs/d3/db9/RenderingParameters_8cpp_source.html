77 void RenderingParameters::parse(const po::variables_map& vm)
78 {
79  if (vm.count(PARAM_RENDERER))
80  {
81  const std::string& rendererName = vm[PARAM_RENDERER].as<std::string>();
82  addRenderer(rendererName);
83  _renderer = rendererName;
84  }
85  _accumulation = !vm[PARAM_ACCUMULATION].as<bool>();
86  if (vm.count(PARAM_BACKGROUND_COLOR))
87  {
88  floats values = vm[PARAM_BACKGROUND_COLOR].as<floats>();
89  _backgroundColor = Vector3f(values[0], values[1], values[2]);
90  }
91  if (vm.count(PARAM_CAMERA))
92  {
93  const std::string& cameraName = vm[PARAM_CAMERA].as<std::string>();
94  _camera = cameraName;
95  if (std::find(_cameras.begin(), _cameras.end(), cameraName) == _cameras.end())
96  _cameras.push_front(cameraName);
97  }
98  _headLight = !vm[PARAM_HEAD_LIGHT].as<bool>();
99  markModified();
100 }
