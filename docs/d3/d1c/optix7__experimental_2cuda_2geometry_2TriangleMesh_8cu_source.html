58 static __device__ void computeWPosDerivatives(float3& ddxwpos, float3& ddywpos,
59  float3 p0, float3 p1, float3 p2,
60  float2 betaDerivative,
61  float2 gammaDerivative)
62 {
63  ddxwpos = p1 * betaDerivative.x + p2 * gammaDerivative.x +
64  p0 * (-betaDerivative.x - gammaDerivative.x);
65  ddywpos = p1 * betaDerivative.y + p2 * gammaDerivative.y +
66  p0 * (-betaDerivative.y - gammaDerivative.y);
67 }

69 static __device__ bool intersect_triangle_filtered(
70  const Ray& ray, const float3& p0, const float3& p1, const float3& p2,
71  const float3& rayDdx, const float3& rayDdy, float3& n, float& t,
72  float& beta, float& gamma, float2& betaDerivative, float2& gammaDerivative)
73 {
74  const float3 e0 = p1 - p0;
75  const float3 e1 = p0 - p2;
76  n = cross(e1, e0);
77 
78  const float3 g0 = p0 - ray.origin;
79 
80  const float NdotRay = dot(n, ray.direction);
81 
82  float3 e2 = g0 / NdotRay;
83  float3 i = cross(ray.direction, e2);
84 
85  beta = dot(i, e1);
86  gamma = dot(i, e0);
87  t = dot(n, e2);
88 
89  if (!((t < ray.tmax) & (t > ray.tmin) & (beta >= 0.0f) & (gamma >= 0.0f) &
90  (beta + gamma <= 1)))
91  return false;
92 
93  const float3 differentialX =
94  cross(rayDdx, e2) +
95  cross(ray.direction, -e2 * dot(rayDdx, n) / NdotRay);
96  const float3 differentialY =
97  cross(rayDdy, e2) +
98  cross(ray.direction, -e2 * dot(rayDdy, n) / NdotRay);
99  betaDerivative.x = dot(differentialX, e1);
100  betaDerivative.y = dot(differentialY, e1);
101  gammaDerivative.x = dot(differentialX, e0);
102  gammaDerivative.y = dot(differentialY, e0);
103 
104  return true;
105 }

107 template <bool DO_REFINE>
108 static __device__ void meshIntersect(int primIdx)
109 {
110  const int3 v_idx = indices_buffer[primIdx];
111 
112  const float3 p0 = vertices_buffer[v_idx.x];
113  const float3 p1 = vertices_buffer[v_idx.y];
114  const float3 p2 = vertices_buffer[v_idx.z];
115 
116  // Intersect ray with triangle
117  float3 n;
118  float t;
119  float beta, gamma;
120  float2 betaDerivative, gammaDerivative;
121  if (intersect_triangle_filtered(ray, p0, p1, p2, prd.rayDdx, prd.rayDdy, n,
122  t, beta, gamma, betaDerivative,
123  gammaDerivative))
124  {
125  if (rtPotentialIntersection(t))
126  {
127  v0 = p0;
128  v1 = p1;
129  v2 = p2;
130  geometric_normal = normalize(n);
131  if (normal_buffer.size() == 0)
132  shading_normal = geometric_normal;
133  else
134  {
135  float3 n0 = normal_buffer[v_idx.x];
136  float3 n1 = normal_buffer[v_idx.y];
137  float3 n2 = normal_buffer[v_idx.z];
138  shading_normal = normalize(n1 * beta + n2 * gamma +
139  n0 * (1.f - beta - gamma));
140  }
141 
142  if (texcoord_buffer.size() == 0)
143  {
144  texcoord = make_float2(0.f, 0.f);
145  computeWPosDerivatives(ddxWPos, ddyWPos, p0, p1, p2,
146  betaDerivative, gammaDerivative);
147  }
148  else
149  {
150  t0 = texcoord_buffer[v_idx.x];
151  t1 = texcoord_buffer[v_idx.y];
152  t2 = texcoord_buffer[v_idx.z];
153 
154  texcoord = t1 * beta + t2 * gamma + t0 * (1.f - beta - gamma);
155 
156  ddx = t1 * betaDerivative.x + t2 * gammaDerivative.x +
157  t0 * (-betaDerivative.x - gammaDerivative.x);
158  ddy = t1 * betaDerivative.y + t2 * gammaDerivative.y +
159  t0 * (-betaDerivative.y - gammaDerivative.y);
160 
161  computeWPosDerivatives(ddxWPos, ddyWPos, p0, p1, p2,
162  betaDerivative, gammaDerivative);
163  }
164 
165  if (DO_REFINE)
166  refine_and_offset_hitpoint(ray.origin + t * ray.direction,
167  ray.direction, geometric_normal, p0,
168  back_hit_point, front_hit_point);
169  simulation_idx = 0;
170  rtReportIntersection(0);
171  }
172  }
173 }

185 RT_PROGRAM void bounds(int primIdx, float result[6])
186 {
187  const int3 v_idx = indices_buffer[primIdx];
188 
189  const float3 v0 = vertices_buffer[v_idx.x];
190  const float3 v1 = vertices_buffer[v_idx.y];
191  const float3 v2 = vertices_buffer[v_idx.z];
192  const float area = length(cross(v1 - v0, v2 - v0));
193 
194  optix::Aabb* aabb = (optix::Aabb*)result;
195 
196  if (area > 0.0f && !isinf(area))
197  {
198  aabb->m_min = fminf(fminf(v0, v1), v2);
199  aabb->m_max = fmaxf(fmaxf(v0, v1), v2);
200  }
201  else
202  {
203  aabb->invalidate();
204  }
205 }
