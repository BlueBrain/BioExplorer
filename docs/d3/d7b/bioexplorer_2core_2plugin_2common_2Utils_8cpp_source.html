37 Quaterniond safeQuatlookAt(const Vector3d& v)
38 {
39  const Vector3d vector = normalize(v);
40  auto upVector = UP_VECTOR;
41  if (abs(dot(vector, upVector)) > 0.999)
42  // Gimble lock
43  upVector = Vector3d(0.0, 0.0, 1.0);
44  return quatLookAtRH(vector, upVector);
45 }

68 bool isClipped(const Vector3d& position, const Vector4ds& clippingPlanes)
69 {
70  if (clippingPlanes.empty())
71  return false;
72 
73  bool visible = true;
74  for (auto plane : clippingPlanes)
75  {
76  const Vector3d normal = normalize(Vector3d(plane.x, plane.y, plane.z));
77  const double d = plane.w;
78  const double distance = dot(normal, position) - d;
79  visible &= (distance < 0.0);
80  }
81  return !visible;
82 }

288 std::vector<std::string> split(const std::string& s,
289  const std::string& delimiter)
290 {
291  std::vector<std::string> values;
292  if (s.empty())
293  return values;
294 
295  std::string str = s;
296  size_t pos = 0;
297  std::string token;
298  while ((pos = str.find(delimiter)) != std::string::npos)
299  {
300  token = str.substr(0, pos);
301  values.push_back(token);
302  str.erase(0, pos + delimiter.length());
303  }
304  values.push_back(str);
305  return values;
306 }

308 Transformation combineTransformations(const Transformations& transformations)
309 {
310  glm::mat4 finalMatrix;
311  for (const auto& transformation : transformations)
312  {
313  const glm::mat4 matrix = transformation.toMatrix();
314  finalMatrix *= matrix;
315  }
316 
317  glm::vec3 scale;
318  glm::quat rotation;
319  glm::vec3 translation;
320  glm::vec3 skew;
321  glm::vec4 perspective;
322  glm::decompose(finalMatrix, scale, rotation, translation, skew,
323  perspective);
324 
325  Transformation transformation;
326  transformation.setTranslation(translation);
327  transformation.setRotation(rotation);
328  transformation.setScale(scale);
329  return transformation;
330 }

332 Vector3d sphereFilling(const double radius, const uint64_t occurrence,
333  const uint64_t occurrences, const uint64_t rnd,
334  Vector3d& position, Quaterniond& rotation,
335  const double ratio)
336 {
337  const double off = 2.0 / occurrences;
338  const double increment = ratio * M_PI * (3.0 - sqrt(5.0));
339  const double y = ((occurrence * off) - 1.0) + off / 2.0;
340  const double r = sqrt(1.0 - pow(y, 2.0));
341  const double phi = rnd * increment;
342  const double x = cos(phi) * r;
343  const double z = sin(phi) * r;
344 
345  const Vector3d normal = normalize(Vector3d(x, y, z));
346  position = normal * radius;
347  rotation = safeQuatlookAt(normal);
348 
349  return normal;
350 }

352 bool rayBoxIntersection(const Vector3d& origin, const Vector3d& direction,
353  const Boxd& box, const double t0, const double t1,
354  double& t)
355 {
356  const Vector3d bounds[2]{box.getMin(), box.getMax()};
357  const Vector3d invDir = 1.0 / direction;
358  const Vector3ui sign{invDir.x < 0.0, invDir.y < 0.0, invDir.z < 0.0};
359 
360  double tmin, tmax, tymin, tymax, tzmin, tzmax;
361 
362  tmin = (bounds[sign.x].x - origin.x) * invDir.x;
363  tmax = (bounds[1 - sign.x].x - origin.x) * invDir.x;
364  tymin = (bounds[sign.y].y - origin.y) * invDir.y;
365  tymax = (bounds[1 - sign.y].y - origin.y) * invDir.y;
366 
367  if ((tmin > tymax) || (tymin > tmax))
368  return false;
369 
370  if (tymin > tmin)
371  tmin = tymin;
372  if (tymax < tmax)
373  tmax = tymax;
374 
375  tzmin = (bounds[sign.z].z - origin.z) * invDir.z;
376  tzmax = (bounds[1 - sign.z].z - origin.z) * invDir.z;
377 
378  if ((tmin > tzmax) || (tzmin > tmax))
379  return false;
380 
381  if (tzmin > tmin)
382  tmin = tzmin;
383  if (tzmax < tmax)
384  tmax = tzmax;
385 
386  t = std::min(tmin, tmax);
387 
388  return (tmin < t1 && tmax > t0);
389 }

391 Vector4d getBezierPoint(const Vector4ds& controlPoints, const double t)
392 {
393  if (t < 0.0 || t > 1.0)
394  PLUGIN_THROW("Invalid value with t=" + std::to_string(t) +
395  ". Must be between 0 and 1");
396  const uint64_t nbControlPoints = controlPoints.size();
397  // 3D points
398  Vector3ds points;
399  points.reserve(nbControlPoints);
400  for (const auto& controlPoint : controlPoints)
401  points.push_back({controlPoint.x, controlPoint.y, controlPoint.z});
402  for (int64_t i = nbControlPoints - 1; i >= 0; --i)
403  for (uint64_t j = 0; j < i; ++j)
404  points[j] += t * (points[j + 1] - points[j]);
405 
406  // Radius
407  const double radius = controlPoints[floor(t * double(nbControlPoints))].w;
408  return Vector4d(points[0].x, points[0].y, points[0].z, radius);
409 }

443 Vector3ds getPointsInSphere(const size_t nbPoints, const double innerRadius)
444 {
445  const double radius =
446  innerRadius + (rand() % 1000 / 1000.0) * (1.0 - innerRadius);
447  double phi = M_PI * ((rand() % 2000 - 1000) / 1000.0);
448  double theta = M_PI * ((rand() % 2000 - 1000) / 1000.0);
449  Vector3ds points;
450  for (size_t i = 0; i < nbPoints; ++i)
451  {
452  Vector3d point = {radius * sin(phi) * cos(theta),
453  radius * sin(phi) * sin(theta), radius * cos(phi)};
454  points.push_back(point);
455  phi += ((rand() % 1000) / 5000.0);
456  theta += ((rand() % 1000) / 5000.0);
457  }
458  return points;
459 }
