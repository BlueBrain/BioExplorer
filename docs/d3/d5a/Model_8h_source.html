56  std::map<size_t, uint64_ts> geometryIndices; // A map where the keys are size_t values that represent
57  // the indices of geometries vector. The map values are
58  // vectors of uint64_t values that represent the indices of
59  // vertices that belong to the geometry.
60  std::vector<uint64_ts> neighbours; // A vector of vectors of uint64_t values that represent the indices
61  // of neighbouring vertices for each vertex.
62  std::vector<uint64_t> neighboursFlat; // A vector of uint64_t values that represents the indices of neighbouring
63  // vertices flattened into a single vector.
64

66  {
67  geometries.clear();
68  geometryIndices.clear();
