48 struct SDFGeometryData
49 {
50  std::vector<SDFGeometry> geometries; // A vector of SDFGeometry objects
51  std::map<size_t, uint64_ts> geometryIndices; // A map where the keys are size_t values that represent
52  // the indices of geometries vector. The map values are
53  // vectors of uint64_t values that represent the indices of
54  // vertices that belong to the geometry.
55  std::vector<uint64_ts> neighbours; // A vector of vectors of uint64_t values that represent the indices
56  // of neighbouring vertices for each vertex.
57  std::vector<uint64_t> neighboursFlat; // A vector of uint64_t values that represents the indices of neighbouring
58  // vertices flattened into a single vector.
59 
60  void clear()
61  {
62  geometries.clear();
63  geometryIndices.clear();
64  neighbours.clear();
65  neighboursFlat.clear();
66  }
