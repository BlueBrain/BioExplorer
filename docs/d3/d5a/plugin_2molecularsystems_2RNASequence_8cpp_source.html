88  PLUGIN_INFO(3, "- Shape params : " << shapeParams);

91  PLUGIN_INFO(3, "- Position : " << position);
92  PLUGIN_INFO(3, "- RNA Sequence length : " << _nbElements);

94  _shape =
95  RNAShapePtr(new RNAShape(clippingPlanes, _details.shape, _nbElements,
96  shapeParams, valuesRange, curveParams));

104 void RNASequence::_buildRNAAsCurve(const Quaterniond& rotation)
105 {
106  const auto& sequence = _details.contents;
107  const auto animationDetails =
108  doublesToAnimationDetails(_details.animationParams);
109  const auto shapeParams = doublesToVector2d(_details.shapeParams);
110  const auto radius = shapeParams.y;
111 
112  auto model = _scene.createModel();
113 
114  size_t materialId = 0;
115  for (const auto& nucleotid : nucleotidMap)
116  {
117  auto material =
118  model->createMaterial(materialId, nucleotid.second.name);
119  brayns::PropertyMap props;
120  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
121  static_cast<int>(MaterialShadingMode::basic)});
122  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
123  props.setProperty(
124  {MATERIAL_PROPERTY_CHAMELEON_MODE,
125  static_cast<int>(
126  MaterialChameleonMode::undefined_chameleon_mode)});
127  material->setDiffuseColor(nucleotid.second.color);
128  material->updateProperties(props);
129  ++materialId;
130  }
131  PLUGIN_INFO(3, "Created " << materialId << " materials");
132 
133  const auto occurrences = _nbElements;
134  for (uint64_t occurrence = 0; occurrence < occurrences - 1; ++occurrence)
135  {
136  const char letter = sequence[occurrence];
137  if (nucleotidMap.find(letter) != nucleotidMap.end())
138  {
139  const auto& codon = nucleotidMap[letter];
140  const auto materialId = codon.index;
141 
142  const auto src = _shape->getTransformation(occurrence, occurrences,
143  animationDetails, 0.f);
144  const auto dst =
145  _shape->getTransformation(occurrence + 1, occurrences,
146  animationDetails, 0.f);
147 
148  model->addCylinder(materialId,
149  {src.getTranslation(), dst.getTranslation(),
150  static_cast<float>(radius)});
151  }
152  }
153 
154  // Metadata
155  ModelMetadata metadata;
156  metadata[METADATA_ASSEMBLY] = _details.assemblyName;
157  metadata["RNA sequence"] = sequence;
158  _modelDescriptor =
159  std::make_shared<ModelDescriptor>(std::move(model), _details.name,
160  metadata);
161  if (_modelDescriptor &&
162  !GeneralSettings::getInstance()->getModelVisibilityOnCreation())
163  _modelDescriptor->setVisible(false);
164 }

166 void RNASequence::_buildRNAAsProteinInstances(const Quaterniond& rotation)
167 {
168  const auto& sequence = _details.contents;
169  const auto animationDetails =
170  doublesToAnimationDetails(_details.animationParams);
171  const size_t nbElements = sequence.length();
172  Vector3d position = Vector3d(0.f);
173 
174  // Load protein
175  ModelPtr model{nullptr};
176  const std::string proteinName = _details.assemblyName + "_RNA sequence";
177  ProteinDetails pd;
178  pd.assemblyName = _details.assemblyName;
179  pd.name = proteinName;
180  pd.pdbId = _details.pdbId;
181  pd.contents = _details.proteinContents;
182  pd.recenter = true;
183  pd.atomRadiusMultiplier = _details.atomRadiusMultiplier;
184  pd.representation = _details.representation;
185 
186  _protein = ProteinPtr(new Protein(_scene, pd));
187  _modelDescriptor = _protein->getModelDescriptor();
188 
189  const auto proteinBounds = _protein->getBounds().getSize();
190  const double proteinSize =
191  std::min(proteinBounds.x, std::min(proteinBounds.y, proteinBounds.z));
192  double proteinSpacing = 0.f;
193 
194  Vector3d previousTranslation;
195 
196  const auto occurrences = _nbElements;
197  uint64_t nbInstances = 0;
198  for (uint64_t occurrence = 0; occurrence < occurrences; ++occurrence)
199  {
200  try
201  {
202  Transformations transformations;
203 
204  Transformation assemblyTransformation;
205  assemblyTransformation.setTranslation(_assemblyPosition);
206  assemblyTransformation.setRotation(_assemblyRotation);
207  transformations.push_back(assemblyTransformation);
208 
209  const auto shapeTransformation =
210  _shape->getTransformation(occurrence, occurrences,
211  animationDetails, 0.f);
212  transformations.push_back(shapeTransformation);
213 
214  const Transformation finalTransformation =
215  combineTransformations(transformations);
216 
217  const Vector3d translation = finalTransformation.getTranslation();
218 
219  if (nbInstances == 0)
220  _modelDescriptor->setTransformation(finalTransformation);
221  else
222  {
223  proteinSpacing += length(previousTranslation - translation);
224  if (proteinSpacing < proteinSize)
225  continue;
226  }
227  previousTranslation = translation;
228  ++nbInstances;
229 
230  const ModelInstance instance(true, false, finalTransformation);
231  _modelDescriptor->addInstance(instance);
232  proteinSpacing = 0.f;
233  }
234  catch (const std::runtime_error&)
235  {
236  // Instance is clipped
237  }
238  }
