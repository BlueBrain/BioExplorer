33 double _interpolatedOpacity(const core::Vector2ds& controlPointsSorted, const double x)
34 {
35  const auto& firstPoint = controlPointsSorted.front();
36  if (x <= firstPoint.x)
37  return firstPoint.y;
38 
39  for (size_t i = 1; i < controlPointsSorted.size(); ++i)
40  {
41  const auto& current = controlPointsSorted[i];
42  const auto& previous = controlPointsSorted[i - 1];
43  if (x <= current.x)
44  {
45  const auto t = (x - previous.x) / (current.x - previous.x);
46  return (1.0 - t) * previous.y + t * current.y;
47  }
48  }
49 
50  const auto& lastPoint = controlPointsSorted.back();
51  return lastPoint.y;
52 }

82 floats TransferFunction::calculateInterpolatedOpacities() const
83 {
84  constexpr size_t numSamples = 256;
85  constexpr double dx = 1. / (numSamples - 1);
86 
87  auto tfPoints = getControlPoints();
88  std::sort(tfPoints.begin(), tfPoints.end(), [](auto a, auto b) { return a.x < b.x; });
89 
90  floats opacities;
91  opacities.reserve(numSamples);
92  for (size_t i = 0; i < numSamples; ++i)
93  opacities.push_back(_interpolatedOpacity(tfPoints, i * dx));
94  return opacities;
95 }
