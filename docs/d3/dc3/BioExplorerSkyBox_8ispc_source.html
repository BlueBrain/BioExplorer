26vec4f skyboxMapping(const uniform Renderer* uniform renderer,
27 const varying Ray& ray,
28 const uniform AdvancedMaterial* uniform bgMaterial)
29{
30 varying vec4f result = make_vec4f(bgMaterial->Kd);
31
32 const varying float radius = 1e6;
33 if (!valid(bgMaterial->map_Kd))
34 return result;
35
36 // solve the equation sphere-ray to find the intersections
37 const varying float a = 2.f * dot(ray.dir, ray.dir);
38 const varying float b = 2.f * dot(ray.org, ray.dir);
39 const varying float c = dot(ray.org, ray.org) - radius * radius;
40 const varying float d = b * b - 2.f * a * c;
41
42 if (d <= 0.f || a == 0.f)
43 return result;
44
45 const varying float r = sqrt(d);
46 const varying float t1 = (-b - r) / a;
47 const varying float t2 = (-b + r) / a;
48
49 const varying float epsilon = 0.00001f;
50 if (t1 <= epsilon && t2 <= epsilon)
51 return result; // both intersections are behind the ray origin
52
53 varying float t = 0.f;
54 if (t1 <= epsilon)
55 t = t2;
56 else if (t2 <= epsilon)
57 t = t1;
58 else
59 t = (t1 < t2) ? t1 : t2;
60
61 if (t < epsilon || t > ray.t)
62 return result; // Too close to intersection
63
64 const varying vec3f intersection = normalize(ray.org + t * ray.dir);
65
66 // Only st needs to be set to sample from Texture2D
67 varying DifferentialGeometry dg;
68 dg.st =
69 make_vec2f(((atan2(-intersection.x, intersection.z) / PI) + 1.f) * .5f,
70 (asin(intersection.y) / PI) + .5f);
71
72 return get4f(bgMaterial->map_Kd.map, dg);
73}
