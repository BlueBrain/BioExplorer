60 void Atlas::_buildModel(const LoaderProgress& callback)
61 {
62  if (_modelDescriptor)
63  _scene.removeModel(_modelDescriptor->getModelID());
64 
65  auto model = _scene.createModel();
66 
67  auto& connector = DBConnector::getInstance();
68  const auto regions = connector.getAtlasRegions(_details.regionSqlFilter);
69  const bool useSdf = false;
70 
71  ThreadSafeContainers containers;
72  uint64_t counter = 0;
73  uint64_t nbCells = 0;
74 
75  const auto nbDBConnections = DBConnector::getInstance().getNbConnections();
76  uint64_t index;
77  volatile bool flag = false;
78  std::string flagMessage;
79 #pragma omp parallel for num_threads(nbDBConnections)
80  for (index = 0; index < regions.size(); ++index)
81  {
82  try
83  {
84  if (flag)
85  continue;
86 
87  if (omp_get_thread_num() == 0)
88  {
89  PLUGIN_PROGRESS("Loading regions...", index, regions.size());
90  try
91  {
92  callback.updateProgress("Loading regions...",
93  0.5f * ((float)index / (float)(regions.size() / nbDBConnections)));
94  }
95  catch (...)
96  {
97 #pragma omp critical
98  flag = true;
99  }
100  }
101  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation);
102 
103  const auto region = regions[index];
104  if (_details.loadCells)
105  {
106  const auto cells = connector.getAtlasCells(region, _details.cellSqlFilter);
107  for (const auto& cell : cells)
108  container.addSphere(cell.second.position, _details.cellRadius, cell.second.region, useSdf);
109 #pragma omp critical
110  nbCells += cells.size();
111  }
112 
113  if (_details.loadMeshes)
114  {
115  const Vector3d position = doublesToVector3d(_details.meshPosition);
116  const Quaterniond rotation = doublesToQuaterniond(_details.meshRotation);
117  const Vector3d scale = doublesToVector3d(_details.meshScale);
118  auto mesh = connector.getAtlasMesh(region);
119  for (auto& vertex : mesh.vertices)
120  {
121  const Vector3d p = Vector3d(vertex) + position;
122  const Vector3d r = rotation * p;
123  vertex = scale * r;
124  }
125  container.addMesh(region, mesh);
126  }
127 
128 #pragma omp critical
129  ++counter;
130 
131 #pragma omp critical
132  containers.push_back(container);
133  }
134  catch (const std::runtime_error& e)
135  {
136 #pragma omp critical
137  flagMessage = e.what();
138 #pragma omp critical
139  flag = true;
140  }
141  catch (...)
142  {
143 #pragma omp critical
144  flagMessage = "Loading was canceled";
145 #pragma omp critical
146  flag = true;
147  }
148  }
149 
150  for (uint64_t i = 0; i < containers.size(); ++i)
151  {
152  PLUGIN_PROGRESS("- Compiling 3D geometry...", i + 1, containers.size());
153  callback.updateProgress("Compiling 3D geometry...", 0.5f + 0.5f * (float)(1 + i) / (float)containers.size());
154  auto& container = containers[i];
155  container.commitToModel();
156  }
157  model->applyDefaultColormap();
158 
159  const ModelMetadata metadata = {{"Number of regions", std::to_string(regions.size())},
160  {"Number of cells", std::to_string(nbCells)},
161  {"Cell SQL filter", _details.cellSqlFilter},
162  {"Region SQL filter", _details.regionSqlFilter}};
163 
164  _modelDescriptor.reset(new core::ModelDescriptor(std::move(model), _details.assemblyName, metadata));
165  if (!_modelDescriptor)
166  PLUGIN_THROW("Atlas model could not be created");
167 }
