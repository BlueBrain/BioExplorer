50 void Atlas::_load()
51 {
52  if (_modelDescriptor)
53  _scene.removeModel(_modelDescriptor->getModelID());
54 
55  auto model = _scene.createModel();
56 
57  auto& connector = DBConnector::getInstance();
58  const auto regions = connector.getAtlasRegions(_details.regionSqlFilter);
59 
60  ThreadSafeContainers containers;
61  uint64_t counter = 0;
62  uint64_t nbCells = 0;
63  uint64_t index;
64 #pragma omp parallel for
65  for (index = 0; index < regions.size(); ++index)
66  {
67  ThreadSafeContainer container(*model, false);
68 
69  const auto region = regions[index];
70  if (_details.loadCells)
71  {
72  const auto cells =
73  connector.getAtlasCells(region, _details.cellSqlFilter);
74  for (const auto& cell : cells)
75  container.addSphere(cell.second.position, _details.cellRadius,
76  cell.second.region);
77 #pragma omp critical
78  nbCells += cells.size();
79  }
80 
81  if (_details.loadMeshes)
82  {
83  const Vector3d position = doublesToVector3d(_details.meshPosition);
84  const Quaterniond rotation =
85  doublesToQuaterniond(_details.meshRotation);
86  const Vector3d scale = doublesToVector3d(_details.meshScale);
87  auto mesh = connector.getAtlasMesh(region);
88  for (auto& vertex : mesh.vertices)
89  {
90  const Vector3d p = Vector3d(vertex) + position;
91  const Vector3d r = rotation * p;
92  vertex = scale * r;
93  }
94  container.addMesh(region, mesh);
95  }
96 
97 #pragma omp critical
98  containers.push_back(container);
99 
100 #pragma omp critical
101  ++counter;
102 
103 #pragma omp critical
104  PLUGIN_PROGRESS("Loading " << regions.size() << " regions", counter,
105  regions.size());
106  }
107 
108  for (uint64_t i = 0; i < containers.size(); ++i)
109  {
110  PLUGIN_PROGRESS("- Compiling 3D geometry...", i, containers.size());
111  auto& container = containers[i];
112  container.commitToModel();
113  }
114  ModelMetadata metadata = {{"Number of regions",
115  std::to_string(regions.size())},
116  {"Number of cells", std::to_string(nbCells)}};
117 
118  _modelDescriptor.reset(new brayns::ModelDescriptor(std::move(model),
119  _details.assemblyName,
120  metadata));
121  if (_modelDescriptor)
122  _scene.addModel(_modelDescriptor);
123  else
124  PLUGIN_THROW("Atlas model could not be created");
125 }
