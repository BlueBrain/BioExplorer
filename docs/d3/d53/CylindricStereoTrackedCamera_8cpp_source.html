36 vec3f _rotateVectorByQuat(const vec3f& v, const vec4f& q)
37 {
38  const auto u = vec3f{q[0], q[1], q[2]}; // vector part of the quaternion
39  const auto s = q[3]; // scalar part of the quaternion
40 
41  return 2.0f * dot(u, v) * u + (s * s - dot(u, u)) * v + 2.0f * s * cross(u, v);
42 }
