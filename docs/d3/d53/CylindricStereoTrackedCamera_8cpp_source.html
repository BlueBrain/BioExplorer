40 vec3f _rotateVectorByQuat(const vec3f& v, const vec4f& q)
41 {
42  const auto u = vec3f{q[0], q[1], q[2]}; // vector part of the quaternion
43  const auto s = q[3]; // scalar part of the quaternion
44 
45  return 2.0f * dot(u, v) * u + (s * s - dot(u, u)) * v + 2.0f * s * cross(u, v);
46 }
