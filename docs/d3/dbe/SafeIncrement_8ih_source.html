30 #define DEFINE_SAFE_INCREMENT_INTERNALl(T, UNIFORM_VARYING, CONST) \
31  inline CONST uniform T* UNIFORM_VARYING \
32  safeIncrement(const uniform bool useSafeIncrement, \
33  CONST T* UNIFORM_VARYING const base, \
34  const UNIFORM_VARYING uint64 index) \
35  { \
36  if (useSafeIncrement) \
37  { \
38  const uniform uint64 tSize = sizeof(uniform T); \
39  const UNIFORM_VARYING uint64 offsetBytes = tSize * index; \
40  return (CONST T * UNIFORM_VARYING const)((uint64)base + \
41  offsetBytes); \
42  } \
43  else \
44  { \
45  return base + index; \
46  } \
47  }
