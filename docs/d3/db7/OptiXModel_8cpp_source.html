47 template <typename T>
48 void setBufferRaw(RTbuffertype bufferType, RTformat bufferFormat, ::optix::Handle<::optix::BufferObj>& buffer,
49  ::optix::Handle<::optix::VariableObj> geometry, T* src, const size_t numElements, const size_t bytes)
50 {
51  auto context = OptiXContext::get().getOptixContext();
52  if (!buffer)
53  buffer = context->createBuffer(bufferType, bufferFormat, numElements);
54  else
55  buffer->setSize(numElements);
56  if (src != nullptr && numElements > 0 && bytes > 0)
57  {
58  memcpy(buffer->map(0, RT_BUFFER_MAP_WRITE_DISCARD), src, bytes);
59  buffer->unmap();
60  }
61  geometry->setBuffer(buffer);
62 }

125 void OptiXModel::commitGeometry()
126 {
127  // Materials
128  uint64_t memoryFootPrint = _commitMaterials();
129 
130  if (!isDirty())
131  return;
132 
133  const auto compactBVH = getBVHFlags().count(BVHFlag::compact) > 0;
134  auto& context = OptiXContext::get();
135  // Geometry group
136  if (!_geometryGroup)
137  _geometryGroup = context.createGeometryGroup(compactBVH);
138 
139  // Bounding box group
140  if (!_boundingBoxGroup)
141  _boundingBoxGroup = context.createGeometryGroup(compactBVH);
142 
143  // Geometry
144  updateBounds();
145  if (_spheresDirty)
146  for (const auto& spheres : _geometries->_spheres)
147  memoryFootPrint += _commitSpheres(spheres.first);
148 
149  if (_cylindersDirty)
150  for (const auto& cylinders : _geometries->_cylinders)
151  memoryFootPrint += _commitCylinders(cylinders.first);
152 
153  if (_conesDirty)
154  for (const auto& cones : _geometries->_cones)
155  memoryFootPrint += _commitCones(cones.first);
156 
157  if (_sdfGeometriesDirty)
158  memoryFootPrint += _commitSDFGeometries();
159 
160  if (_triangleMeshesDirty)
161  for (const auto& meshes : _geometries->_triangleMeshes)
162  memoryFootPrint += _commitMeshes(meshes.first);
163 
164  if (_volumesDirty)
165  for (const auto& volume : _geometries->_volumes)
166  memoryFootPrint += _commitVolumes(volume.first);
167 
168  if (_streamlinesDirty)
169  for (const auto& streamlines : _geometries->_streamlines)
170  memoryFootPrint += _commitStreamlines(streamlines.first);
171 
172  _markGeometriesClean();
173  _transferFunction.markModified();
174  _instancesDirty = false;
175  CORE_DEBUG("Geometry group has " << _geometryGroup->getChildCount() << " children instances");
176  CORE_DEBUG("Bounding box group has " << _boundingBoxGroup->getChildCount() << " children instances");
177  CORE_DEBUG("Model memory footprint: " << memoryFootPrint / 1024 / 1024 << " MB");
178 }

190 uint64_t OptiXModel::_commitSpheres(const size_t materialId)
191 {
192  uint64_t memoryFootPrint = 0;
193  if (_geometries->_spheres.find(materialId) == _geometries->_spheres.end())
194  return memoryFootPrint;
195 
196  auto context = OptiXContext::get().getOptixContext();
197  const auto& spheres = _geometries->_spheres[materialId];
198  context[CONTEXT_SPHERE_SIZE]->setUint(sizeof(Sphere) / sizeof(float));
199 
200  // Geometry
201  _optixSpheres[materialId] = OptiXContext::get().createGeometry(OptixGeometryType::sphere);
202  _optixSpheres[materialId]->setPrimitiveCount(spheres.size());
203 
204  const uint64_t bufferSize = sizeof(Sphere) * spheres.size();
205  setBuffer(RT_BUFFER_INPUT, RT_FORMAT_FLOAT, _spheresBuffers[materialId],
206  _optixSpheres[materialId][OPTIX_GEOMETRY_PROPERTY_SPHERES], spheres, bufferSize);
207  memoryFootPrint += bufferSize;
208 
209  // Material
210  auto& mat = static_cast<OptiXMaterial&>(*_materials[materialId]);
211  const auto material = mat.getOptixMaterial();
212  if (!material)
213  CORE_THROW(std::runtime_error("Material is not defined"));
214 
215  // Instance
216  auto instance = context->createGeometryInstance();
217  instance->setGeometry(_optixSpheres[materialId]);
218  instance->setMaterialCount(1);
219  instance->setMaterial(0, material);
220  if (materialId == BOUNDINGBOX_MATERIAL_ID)
221  _boundingBoxGroup->addChild(instance);
222  else
223  _geometryGroup->addChild(instance);
224  CORE_DEBUG("Spheres memory footprint: " << memoryFootPrint / 1024 / 1024 << " MB");
225  return memoryFootPrint;
226 }

300 uint64_t OptiXModel::_commitSDFGeometries()
301 {
302  uint64_t memoryFootPrint = 0;
303  if (_geometries->_sdf.geometries.empty())
304  return memoryFootPrint;
305 
306  auto context = OptiXContext::get().getOptixContext();
307  auto& sdfGeometries = _geometries->_sdf;
308  const uint32_t sdfGeometrySize = sizeof(SDFGeometry);
309  context[CONTEXT_SDF_GEOMETRY_SIZE]->setUint(sdfGeometrySize);
310 
311  const uint64_t nbGeometries = sdfGeometries.geometries.size();
312  const uint64_t geometriesBufferSize = sizeof(SDFGeometry) * nbGeometries;
313 
314  uint64_ts geometryMaterials;
315  for (const auto& geometryIndex : sdfGeometries.geometryIndices)
316  geometryMaterials.push_back(geometryIndex.first);
317  const uint64_t nbMaterials = geometryMaterials.size();
318 
319  uint64_t i = 0;
320 #pragma omp parallel for private(memoryFootPrint)
321  for (i = 0; i < nbMaterials; ++i)
322  {
323  const auto materialId = geometryMaterials[i];
324 
325  // Indices of geometries for current material id
326  const auto& geometryIndices = sdfGeometries.geometryIndices[materialId];
327 
328  // Create a local copy of SDF geometries attached to the material id
329  std::map<uint64_t, SDFGeometry*> localGeometries;
330 
331  for (const auto geometryIndex : geometryIndices)
332  {
333  localGeometries[geometryIndex] = &sdfGeometries.geometries[geometryIndex];
334  const auto& neighbours = sdfGeometries.neighbours[geometryIndex];
335  for (const auto neighbour : neighbours)
336  localGeometries[neighbour] = &sdfGeometries.geometries[neighbour];
337  }
338 
339  uint64_t index = 0;
340  uint64_tm localGeometriesMapping;
341  for (const auto& localGeometry : localGeometries)
342  {
343  localGeometriesMapping[localGeometry.first] = index;
344  ++index;
345  }
346 
347  uint64_ts localIndicesFlat;
348  localIndicesFlat.reserve(geometryIndices.size());
349  for (const auto geometryIndex : geometryIndices)
350  localIndicesFlat.push_back(localGeometriesMapping[geometryIndex]);
351 
352  if (localIndicesFlat.empty())
353  continue;
354 
355  // Create a local flat representation of the SDF geometries neighbours
356  uint64_ts localNeighboursFlat;
357  for (const auto geometryIndex : geometryIndices)
358  {
359  const auto& neighbours = sdfGeometries.neighbours[geometryIndex];
360  localGeometries[geometryIndex]->neighboursIndex = localNeighboursFlat.size();
361  localGeometries[geometryIndex]->numNeighbours = static_cast<uint8_t>(neighbours.size());
362  for (uint64_t i = 0; i < neighbours.size(); ++i)
363  {
364  const auto it = localGeometriesMapping.find(neighbours[i]);
365  if (it == localGeometriesMapping.end())
366  CORE_THROW("Invalid neighbour index");
367  if (geometryIndex != (*it).second)
368  localNeighboursFlat.push_back((*it).second);
369  }
370  }
371 
372  // Convert map to vector in order to send the data to the device
373  std::vector<SDFGeometry> localGeometriesFlat;
374  localGeometriesFlat.reserve(localGeometries.size());
375  for (const auto geometry : localGeometries)
376  localGeometriesFlat.push_back(*geometry.second);
377 
378  // Prepare OptiX geometry and corresponding buffers
379  _optixSdfGeometries[materialId] = OptiXContext::get().createGeometry(OptixGeometryType::sdfGeometry);
380  _optixSdfGeometries[materialId]->setPrimitiveCount(localIndicesFlat.size());
381 
382  uint64_t bufferSize = localIndicesFlat.size();
383 #pragma omp critical
384  setBuffer(RT_BUFFER_INPUT, RT_FORMAT_UNSIGNED_LONG_LONG, _sdfGeometriesBuffers[materialId].indices_buffer,
385  _optixSdfGeometries[materialId][OPTIX_GEOMETRY_PROPERTY_SDF_GEOMETRIES_INDICES], localIndicesFlat,
386  bufferSize);
387 #pragma omp critical
388  memoryFootPrint += bufferSize;
389 
390  bufferSize = localNeighboursFlat.size();
391 #pragma omp critical
392  setBuffer(RT_BUFFER_INPUT, RT_FORMAT_UNSIGNED_LONG_LONG, _sdfGeometriesBuffers[materialId].neighbours_buffer,
393  _optixSdfGeometries[materialId][OPTIX_GEOMETRY_PROPERTY_SDF_GEOMETRIES_NEIGHBOURS],
394  localNeighboursFlat, bufferSize);
395 #pragma omp critical
396  memoryFootPrint += bufferSize;
397 
398  bufferSize = sizeof(SDFGeometry) * localGeometriesFlat.size();
399 #pragma omp critical
400  setBuffer(RT_BUFFER_INPUT, RT_FORMAT_BYTE, _sdfGeometriesBuffers[materialId].geometries_buffer,
401  _optixSdfGeometries[materialId][OPTIX_GEOMETRY_PROPERTY_SDF_GEOMETRIES], localGeometriesFlat,
402  bufferSize);
403 #pragma omp critical
404  memoryFootPrint += bufferSize;
405 
406  // Create material
407  auto& material = static_cast<OptiXMaterial&>(*_materials[materialId]);
408  auto optixMaterial = material.getOptixMaterial();
409  if (!optixMaterial)
410  CORE_THROW(std::runtime_error("OptiX material is not defined"));
411 
412  auto instance = context->createGeometryInstance();
413  instance->setGeometry(_optixSdfGeometries[materialId]);
414  instance->setMaterialCount(1);
415  instance->setMaterial(0, optixMaterial);
416 
417  // Add geometry to the model
418 #pragma omp critical
419  _geometryGroup->addChild(instance);
420  }
421  CORE_DEBUG("SDFGeometries memory footprint: " << memoryFootPrint / 1024 / 1024 << " MB");
422  return memoryFootPrint;
423 }

475 uint64_t OptiXModel::_commitStreamlines(const size_t materialId)
476 {
477  uint64_t memoryFootPrint = 0;
478  if (_geometries->_streamlines.find(materialId) == _geometries->_streamlines.end())
479  return memoryFootPrint;
480 
481  const auto& streamlines = _geometries->_streamlines[materialId];
482 
483  // Identify streamlines according to indices
484  std::vector<Vector2ui> indices;
485  size_t begin = 0;
486  for (size_t i = 0; i < streamlines.indices.size() - 1; ++i)
487  {
488  if (streamlines.indices[i] + 1 != streamlines.indices[i + 1])
489  {
490  indices.push_back({begin, streamlines.indices[i]});
491  begin = streamlines.indices[i + 1];
492  }
493  }
494 
495  Vector4fs vertexCurve;
496  Vector4fs colorCurve = streamlines.vertexColor;
497  uint32_ts indexCurve;
498  const bool processColor = colorCurve.empty();
499  size_t count = 0;
500  for (const auto& index : indices)
501  {
502  const uint32_t begin = index.x;
503  const uint32_t end = index.y;
504 
505  Vector4fs controlPoints;
506  for (uint32_t idx = begin; idx < end; ++idx)
507  controlPoints.push_back(streamlines.vertex[idx]);
508 
509  const auto lengthSegment = length(Vector3f(streamlines.vertex[end]) - Vector3f(streamlines.vertex[begin]));
510  const auto nbSteps = max(2, floor(lengthSegment / 3.f));
511  const float t_step = 1.f / nbSteps;
512  for (float t = 0.f; t < 1.f - t_step; t += t_step)
513  {
514  const auto a = getBezierPoint(controlPoints, t);
515  const auto b = getBezierPoint(controlPoints, t + t_step);
516  const auto src = Vector3f(a);
517  const auto dst = Vector3f(b);
518  const auto srcRadius = a.w;
519  const auto dstRadius = b.w;
520  if (t == 0.f)
521  {
522  vertexCurve.push_back(Vector4f(src, 0.f));
523  vertexCurve.push_back(Vector4f(dst, dstRadius));
524  }
525  else if (t == 1.f - t_step)
526  {
527  vertexCurve.push_back(Vector4f(src, srcRadius));
528  vertexCurve.push_back(Vector4f(dst, 0.f));
529  }
530  else
531  {
532  vertexCurve.push_back(Vector4f(src, srcRadius));
533  vertexCurve.push_back(Vector4f(dst, dstRadius));
534  }
535 
536  if (processColor)
537  colorCurve.push_back(Vector4f(0.5f + 0.5f * normalize(dst - src), 1.f));
538 
539  indexCurve.push_back(count);
540  count += 2;
541  }
542  }
543 
544  _optixStreamlines[materialId] = OptiXContext::get().createGeometry(OptixGeometryType::streamline);
545 
546  auto& optixStreamlines = _optixStreamlines[materialId];
547  optixStreamlines->setPrimitiveCount(indexCurve.size());
548 
549  setBuffer(RT_BUFFER_INPUT, RT_FORMAT_FLOAT4, _streamlinesBuffers[materialId].vertices_buffer,
550  _optixStreamlines[materialId][OPTIX_GEOMETRY_PROPERTY_STREAMLINE_VERTEX], vertexCurve,
551  vertexCurve.size());
552  memoryFootPrint += vertexCurve.size() * sizeof(float) * 4;
553 
554  setBuffer(RT_BUFFER_INPUT, RT_FORMAT_UNSIGNED_INT, _streamlinesBuffers[materialId].indices_buffer,
555  _optixStreamlines[materialId][OPTIX_GEOMETRY_PROPERTY_STREAMLINE_MESH_INDEX], indexCurve,
556  indexCurve.size());
557  memoryFootPrint += indexCurve.size() * sizeof(uint);
558 
559  auto& material = static_cast<OptiXMaterial&>(*_materials[materialId]);
560  auto optixMaterial = material.getOptixMaterial();
561  if (!optixMaterial)
562  CORE_THROW(std::runtime_error("OptiX material is not defined"));
563 
564  auto context = OptiXContext::get().getOptixContext();
565

715 uint64_t OptiXModel::_commitVolumes(const size_t materialId)
716 {
717  uint64_t memoryFootPrint = 0;
718  if (!_volumesDirty)
719  return memoryFootPrint;
720 
721  auto iter = _geometries->_volumes.find(materialId);
722  if (iter == _geometries->_volumes.end())
723  return memoryFootPrint;
724 
725  _optixVolumes[materialId] = OptiXContext::get().createGeometry(OptixGeometryType::volume);
726 
727  auto material = dynamic_cast<OptiXMaterial*>(_materials[materialId].get());
728  auto optixMaterial = material->getOptixMaterial();
729  if (!optixMaterial)
730  CORE_THROW(std::runtime_error("OptiX material is not defined"));
731 
732  auto& optixVolumes = _optixVolumes[materialId];
733  optixVolumes->setPrimitiveCount(1);
734 
735  auto context = OptiXContext::get().getOptixContext();
736  auto instance = context->createGeometryInstance();
737  auto& optixVolume = _optixVolumes[materialId];
738  instance->setGeometry(optixVolume);
739  instance->setMaterialCount(1);
740  instance->setMaterial(0, optixMaterial);
741  _geometryGroup->addChild(instance);
742 
743  const auto volume = dynamic_cast<OptiXVolume*>((*iter).second.get());
744  _volumeGeometries[materialId].offset = volume->getOffset();
745 
746  const auto sharedDataVolume = dynamic_cast<OptiXSharedDataVolume*>(volume);
747  const auto octreeVolume = dynamic_cast<OptiXOctreeVolume*>(volume);
748 
749  if (sharedDataVolume)
750  {
751  const auto& memoryBuffer = sharedDataVolume->getMemoryBuffer();
752  if (!memoryBuffer.empty())
753  {
754  // Volume as 3D texture
755  const auto& dimensions = sharedDataVolume->getDimensions();
756  const auto& valueRange = sharedDataVolume->getValueRange();
757  Buffer buffer = context->createMipmappedBuffer(RT_BUFFER_INPUT, RT_FORMAT_FLOAT, dimensions.x, dimensions.y,
758  dimensions.z, 1u);
759  const size_t size = dimensions.x * dimensions.y * dimensions.z;
760  memcpy(buffer->map(), memoryBuffer.data(), size * sizeof(float));
761  buffer->unmap();
762  _createSampler(materialId, buffer, TextureType::volume, RT_TEXTURE_INDEX_ARRAY_INDEX, valueRange);
763  memoryFootPrint += dimensions.x * dimensions.y * dimensions.z * sizeof(float);
764  }
765  }
766 
767  if (octreeVolume)
768  {
769  _volumeGeometries[materialId].octreeDataType = octreeVolume->getOctreeDataType();
770  const auto& octreeIndices = octreeVolume->getOctreeIndices();
771  if (!octreeIndices.empty())
772  {
773  // Octree indices as texture
774  const size_t size = octreeIndices.size();
775  Buffer buffer = context->createMipmappedBuffer(RT_BUFFER_INPUT, RT_FORMAT_UNSIGNED_INT, size, 1u);
776  memcpy(buffer->map(), octreeIndices.data(), size * sizeof(uint32_t));
777  buffer->unmap();
778  _createSampler(materialId, buffer, TextureType::octree_indices, RT_TEXTURE_INDEX_ARRAY_INDEX);
779  memoryFootPrint += size * sizeof(uint32_t);
780  }
781 
782  const auto& octreeValues = octreeVolume->getOctreeValues();
783  if (!octreeValues.empty())
784  {
785  // Octree values as texture
786  const size_t size = octreeValues.size();
787  Buffer buffer = context->createMipmappedBuffer(RT_BUFFER_INPUT, RT_FORMAT_FLOAT, size, 1u);
788  memcpy(buffer->map(), octreeValues.data(), size * sizeof(float));
789  buffer->unmap();
790  _createSampler(materialId, buffer, TextureType::octree_values, RT_TEXTURE_INDEX_ARRAY_INDEX);
791  memoryFootPrint += size * sizeof(float);
792  }
793  }
794  _commitVolumesBuffers(materialId);
795  _volumesDirty = false;
796  CORE_DEBUG("Volumes memory footprint: " << memoryFootPrint / 1024 / 1024 << " MB");
797  return memoryFootPrint;
798 }

821 void OptiXModel::_commitTransferFunctionImpl(const Vector3fs& colors, const floats& opacities,
822  const Vector2d valueRange)
823 {
824  auto context = OptiXContext::get().getOptixContext();
825  const auto nbColors = colors.size();
826  Vector4fs colormap;
827  for (uint64_t i = 0; i < nbColors; ++i)
828  colormap.push_back({colors[i].x, colors[i].y, colors[i].z, opacities[i * 256 / nbColors]});
829 
830  // Attach transfer function texture to all materials in the model
831  for (auto material : _materials)
832  {
833  const auto materialId = material.first;
834  if (materialId == BOUNDINGBOX_MATERIAL_ID || materialId == SECONDARY_MODEL_MATERIAL_ID)
835  continue;
836 
837  auto optixMaterial = static_cast<OptiXMaterial*>(getMaterial(materialId).get());
838  auto deviceMaterial = optixMaterial->getOptixMaterial();
839  if (!deviceMaterial)
840  continue;
841 
842  Buffer buffer = context->createMipmappedBuffer(RT_BUFFER_INPUT, RT_FORMAT_FLOAT4, nbColors, 1u);
843  memcpy(buffer->map(), colormap.data(), sizeof(Vector4f) * colormap.size());
844  buffer->unmap();
845  const auto samplerId = _createSampler(materialId, buffer, TextureType::transfer_function,
846  RT_TEXTURE_INDEX_NORMALIZED_COORDINATES, valueRange);
847 
848  if (materialId == VOLUME_MATERIAL_ID || materialId == VOLUME_OCTREE_INDICES_MATERIAL_ID ||
849  materialId == VOLUME_OCTREE_VALUES_MATERIAL_ID)
850  {
851  for (auto& volumeGeometry : _volumeGeometries)
852  {
853  volumeGeometry.second.transferFunctionSamplerId = samplerId;
854  volumeGeometry.second.valueRange = valueRange;
855  }
856 
857  // Update volume buffers with transfer function texture sampler Id and range
858  _commitVolumesBuffers(material.first);
859  }
860  }

870 size_t OptiXModel::_createSampler(const size_t materialId, const Buffer& buffer, const TextureType textureType,
871  const RTtextureindexmode textureIndexType, const Vector2f& valueRange)
872 {
873  auto context = OptiXContext::get().getOptixContext();
874  auto material = static_cast<OptiXMaterial*>(getMaterial(materialId).get());
875  auto optixMaterial = material->getOptixMaterial();
876  material->setValueRange(valueRange);
877  auto& textureSamplers = material->getTextureSamplers();
878 
879  // Remove existing sampler (if applicable)
880  const auto it = textureSamplers.find(textureType);
881  if (it != textureSamplers.end())
882  textureSamplers.erase(it);
883 
884  // Create new sample
885  TextureSampler sampler = context->createTextureSampler();
886  const auto samplerId = sampler->getId();
887  auto filteringMode = RT_FILTER_LINEAR;
888  switch (textureType)
889  {
890  case TextureType::volume:
891  _volumeGeometries[materialId].volumeSamplerId = samplerId;
892  break;
893  case TextureType::transfer_function:
894  if (_volumeGeometries.find(materialId) != _volumeGeometries.end())
895  {
896  _volumeGeometries[materialId].transferFunctionSamplerId = samplerId;
897  _volumeGeometries[materialId].valueRange = valueRange;
898  }
899  break;
900  case TextureType::octree_indices:
901  _volumeGeometries[materialId].octreeIndicesSamplerId = samplerId;
902  filteringMode = RT_FILTER_NEAREST;
903  break;
904  case TextureType::octree_values:
905  _volumeGeometries[materialId].octreeValuesSamplerId = samplerId;
906  filteringMode = RT_FILTER_NEAREST;
907  break;
908  }
909 
910  sampler->setWrapMode(0, RT_WRAP_CLAMP_TO_EDGE);
911  sampler->setWrapMode(1, RT_WRAP_CLAMP_TO_EDGE);
912  sampler->setWrapMode(2, RT_WRAP_CLAMP_TO_EDGE);
913  sampler->setIndexingMode(textureIndexType);
914  sampler->setReadMode(RT_TEXTURE_READ_NORMALIZED_FLOAT);
915  sampler->setBuffer(0u, 0u, buffer);
916  sampler->setFilteringModes(filteringMode, filteringMode, RT_FILTER_NONE);
917  sampler->setMaxAnisotropy(8.0f);
918  sampler->validate();
919 
920  textureSamplers.insert(std::make_pair(textureType, sampler));
921  const auto textureName = textureTypeToString[static_cast<uint8_t>(textureType)];
922  optixMaterial[textureName]->setInt(samplerId);
923  material->commit();
924  return samplerId;
925 }
