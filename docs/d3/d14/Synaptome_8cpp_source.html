76 void Synaptome::_buildModel(const LoaderProgress& callback)
77 {
78  if (_modelDescriptor)
79  _scene.removeModel(_modelDescriptor->getModelID());
80 
81  auto model = _scene.createModel();
82 
83  auto& connector = DBConnector::getInstance();
84  const auto nbDBConnections = connector.getNbConnections();
85  const auto somas = connector.getNeurons(_details.populationName, _details.sqlNodeFilter);
86  const auto synaptome = connector.getSynaptome(_details.populationName, _details.sqlEdgeFilter);
87 
88  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation);
89  const size_t nodeMaterialId = 0;
90  const size_t edgeMaterialId = 1;
91  const bool useSdf = false;
92  const float nodeRadius = _details.radius;
93  const float edgeRadius = _details.radius / 5.f;
94 
95  for (const auto& soma : somas)
96  _addNode(soma.first, soma.second.position, 1.f);
97 
98  for (const auto& edge : synaptome)
99  {
100  const auto itSrc = somas.find(edge.first);
101  const auto itDst = somas.find(edge.second);
102  if (itSrc != somas.end() && itDst != somas.end())
103  {
104  const auto direction = _details.force * ((*itDst).second.position - (*itSrc).second.position);
105  _addEdge(edge.first, edge.second, direction);
106  }
107  }
108  for (const auto& node : _nodes)
109  container.addSphere(node.second.position + node.second.direction * node.second.mass, nodeRadius, nodeMaterialId,
110  useSdf);
111  for (const auto& edge : _edges)
112  container.addCone(_nodes[edge.x].position + _nodes[edge.x].direction * _nodes[edge.x].mass, edgeRadius,
113  _nodes[edge.y].position + _nodes[edge.y].direction * _nodes[edge.y].mass, edgeRadius,
114  static_cast<size_t>(_nodes[edge.x].mass + 1), useSdf);
115  container.commitToModel();
116  model->applyDefaultColormap();
117 
118  const ModelMetadata metadata = {{"Number of nodes", std::to_string(_nodes.size())},
119  {"Number of edges", std::to_string(_edges.size())},
120  {"SQL node filter", _details.sqlNodeFilter},
121  {"SQL edge filter", _details.sqlEdgeFilter}};
122  _modelDescriptor.reset(new core::ModelDescriptor(std::move(model), _details.assemblyName, metadata));
123  if (!_modelDescriptor)
124  PLUGIN_THROW(
125  "Synaptome model could not be created for "
126  "population " +
127  _details.populationName);
128 }
