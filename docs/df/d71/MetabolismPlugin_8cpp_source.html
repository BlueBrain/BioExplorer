81 void MetabolismPlugin::init()
82 {
83  auto actionInterface = _api->getActionInterface();
84  auto &engine = _api->getEngine();
85 
86  if (actionInterface)
87  {
88  std::string endPoint = PLUGIN_API_PREFIX + "attach-handler";
89  PLUGIN_REGISTER_ENDPOINT(endPoint);
90  actionInterface->registerRequest<AttachHandlerDetails, Response>(endPoint,
91  [&](const AttachHandlerDetails &payload)
92  { return _attachHandler(payload); });
93  }
94 
95  _addMetabolismRenderer(engine);
96 }

98 void MetabolismPlugin::_parseCommandLineArguments(int argc, char **argv)
99 {
100  for (size_t i = 0; i < argc; ++i)
101  {
102  const std::string argument = argv[i];
103  std::string key;
104  std::string value;
105  const int pos = argument.find("=");
106  if (pos == std::string::npos)
107  key = argument;
108  else
109  {
110  key = argument.substr(0, pos);
111  value = argument.substr(pos + 1);
112  }
113  _commandLineArguments[key] = value;
114  }
115 }

117 Response MetabolismPlugin::_attachHandler(const AttachHandlerDetails &payload)
118 {
119  Response response;
120  try
121  {
122  auto &scene = _api->getScene();
123  auto descriptors = scene.getModelDescriptors();
124  if (descriptors.empty())
125  PLUGIN_THROW("Scene must contain a model");
126 
127  auto descriptor = descriptors[0];
128  auto &model = descriptor->getModel();
129 
130  auto handler = std::make_shared<MetabolismHandler>(payload);
131  model.setSimulationHandler(handler);
132  }
133  catch (const std::runtime_error &e)
134  {
135  response.status = false;
136  response.contents = e.what();
137  PLUGIN_ERROR(e.what());
138  }
139  return response;
140 }
