45 void PointFieldBuilder::buildOctree(core::Engine& engine, core::Model& model, const double voxelSize,
46  const double density, const uint32_ts& modelIds)
47 {
48  PLUGIN_INFO(3, "Building Point Octree");
49 
50  auto& scene = engine.getScene();
51  const auto& clipPlanes = getClippingPlanes(scene);
52 
53  OctreePoints points;
54  uint32_t count{0};
55  const uint32_t densityRatio = 1.f / density;
56 
57  Boxd bounds;
58  const auto& modelDescriptors = scene.getModelDescriptors();
59  for (const auto modelDescriptor : modelDescriptors)
60  {
61  if (!modelIds.empty())
62  {
63  const auto modelId = modelDescriptor->getModelID();
64  const auto it = std::find(modelIds.begin(), modelIds.end(), modelId);
65  if (it == modelIds.end())
66  continue;
67  }
68 
69  const auto& instances = modelDescriptor->getInstances();
70  for (const auto& instance : instances)
71  {
72  const auto& tf = instance.getTransformation();
73  const auto& model = modelDescriptor->getModel();
74  const auto& spheresMap = model.getSpheres();
75  for (const auto& spheres : spheresMap)
76  {
77  if (spheres.first != BOUNDINGBOX_MATERIAL_ID && spheres.first != SECONDARY_MODEL_MATERIAL_ID)
78  for (const auto& sphere : spheres.second)
79  {
80  const Vector3f center =
81  tf.getTranslation() + tf.getRotation() * (Vector3d(sphere.center) - tf.getRotationCenter());
82 
83  const Vector3d c = center;
84  if (isClipped(c, clipPlanes))
85  {
86  ++count;
87  continue;
88  }
89 
90  if (count % densityRatio == 0)
91  {
92  bounds.merge(center + sphere.radius);
93  bounds.merge(center - sphere.radius);
94 
95  OctreePoint point;
96  point.position = center;
97  point.radius = sphere.radius;
98  point.value = sphere.radius;
99  points.push_back(point);
100  }
101  ++count;
102  }
103  }
104  }
105  }
106 
107  // Events bounds
108  Vector3f sceneSize = bounds.getSize();
109  Vector3f center = bounds.getCenter();
110  Vector3f extendedHalfSize = sceneSize * 0.5f;
111 
112  // Expand volume by 25%
113  const float boundsExpansion = 1.25f;
114  bounds.merge(center + extendedHalfSize * boundsExpansion);
115  bounds.merge(center - extendedHalfSize * boundsExpansion);
116  sceneSize = bounds.getSize();
117  center = bounds.getCenter();
118  extendedHalfSize = sceneSize * 0.5f;
119 
120  // Compute volume information
121  Boxd extendedAABB;
122  extendedAABB.merge(center - extendedHalfSize);
123  extendedAABB.merge(center + extendedHalfSize);
124 
125  // Build acceleration structure
126  const PointOctree accelerator(points, voxelSize, extendedAABB.getMin(), extendedAABB.getMax());
127  const auto& indices = accelerator.getFlatIndices();
128  const auto& data = accelerator.getFlatData();
129 
130  const auto offset = extendedAABB.getMin();
131  const auto dimensions = accelerator.getVolumeDimensions();
132  const auto spacing = extendedAABB.getSize() / Vector3d(dimensions);
133 
134  const auto& params = engine.getParametersManager().getApplicationParameters();
135  const auto& engineName = params.getEngine();
136  auto field = model.createField(dimensions, spacing, offset, indices, data, OctreeDataType::odt_points);
137  const size_t materialId = FIELD_MATERIAL_ID;
138  model.addField(materialId, field);
139  model.createMaterial(materialId, std::to_string(materialId));
140 
141  PLUGIN_INFO(1, "--------------------------------------------");
142  PLUGIN_INFO(1, "Point Octree information (" << points.size() << " points)");
143  PLUGIN_INFO(1, "--------------------------------------------");
144  PLUGIN_INFO(1, "Dimensions : " << sceneSize);
145  PLUGIN_INFO(1, "Element spacing : " << spacing);
146  PLUGIN_INFO(1, "Offset : " << offset);
147  PLUGIN_INFO(1, "Bounding box : " << bounds);
148  PLUGIN_INFO(1, "Volume dimensions : " << dimensions);
149  PLUGIN_INFO(1, "Volume size : " << accelerator.getVolumeSize() << " bytes");
150  PLUGIN_INFO(1, "Indices size : " << indices.size());
151  PLUGIN_INFO(1, "PointOctree depth : " << accelerator.getOctreeDepth());
152  PLUGIN_INFO(1, "--------------------------------------------");
153 }
