54 const OOCSceneConfigurationDetails DBConnector::getSceneConfiguration()
55 {
56  OOCSceneConfigurationDetails sceneConfiguration;
57  pqxx::read_transaction transaction(_connection);
58  try
59  {
60  const auto sql =
61  "SELECT scene_size_x, scene_size_y, scene_size_z, nb_bricks, "
62  "description FROM " +
63  _schema + ".configuration";
64  PLUGIN_DEBUG(sql);
65  auto res = transaction.exec(sql);
66  for (auto c = res.begin(); c != res.end(); ++c)
67  {
68  sceneConfiguration.sceneSize =
69  Vector3f(c[0].as<float>(), c[1].as<float>(), c[2].as<float>());
70  sceneConfiguration.nbBricks = c[3].as<uint32_t>();
71  sceneConfiguration.description = c[4].as<std::string>();
72  if (sceneConfiguration.nbBricks == 0)
73  PLUGIN_THROW("Invalid number of bricks)");
74  sceneConfiguration.brickSize =
75  sceneConfiguration.sceneSize / sceneConfiguration.nbBricks;
76  }
77  }
78  catch (pqxx::sql_error& e)
79  {
80  PLUGIN_THROW(e.what());
81  }
82  transaction.abort();
83  return sceneConfiguration;
84 }

87  const uint32_t nbModels,

89 {

91  try
92  {
93  const pqxx::binarystring tmp((void*)buffer.str().c_str(),
94  buffer.str().size() * sizeof(char));
95  transaction.exec_params("INSERT INTO " + _schema +
96  ".brick VALUES ($1, $2, $3, $4)",
97  brickId, version, nbModels, tmp);
98  transaction.commit();
99  PLUGIN_DEBUG("Brick ID " << brickId << " successfully inserted");
100  }
101  catch (pqxx::sql_error& e)

103  transaction.abort();
104  PLUGIN_THROW(e.what());

108 std::stringstream DBConnector::getBrick(const int32_t brickId,
109  const uint32_t& version,
110  uint32_t& nbModels)
111 {
112  std::stringstream s;
113  pqxx::read_transaction transaction(_connection);
114  try
115  {
116  const auto sql = "SELECT nb_models, buffer FROM " + _schema +
117  ".brick WHERE guid=" + std::to_string(brickId) +
118  " AND version=" + std::to_string(version);
119  PLUGIN_DEBUG(sql);
120  auto res = transaction.exec(sql);
121  for (auto c = res.begin(); c != res.end(); ++c)
122  {
123  nbModels = c[0].as<uint32_t>();
124  if (nbModels > 0)
125  {
126  const pqxx::binarystring buffer(c[1]);
127  std::copy(buffer.begin(), buffer.end(),
128  std::ostream_iterator<char>(s));
129  }
130  }
131  }
132  catch (pqxx::sql_error& e)
133  {
134  PLUGIN_THROW(e.what());
135  }
136  transaction.abort();
137 
138  return s;
139 }
