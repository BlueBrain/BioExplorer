58 const OOCSceneConfigurationDetails DBConnector::getSceneConfiguration()
59 {
60  OOCSceneConfigurationDetails sceneConfiguration;
61  pqxx::read_transaction transaction(_connection);
62  try
63  {
64  const auto sql =
65  "SELECT scene_size_x, scene_size_y, scene_size_z, nb_bricks, "
66  "description FROM " +
67  _schema + ".configuration";
68  PLUGIN_DEBUG(sql);
69  auto res = transaction.exec(sql);
70  for (auto c = res.begin(); c != res.end(); ++c)
71  {
72  sceneConfiguration.sceneSize =
73  Vector3d(c[0].as<double>(), c[1].as<double>(),
74  c[2].as<double>());
75  sceneConfiguration.nbBricks = c[3].as<uint32_t>();
76  sceneConfiguration.description = c[4].as<std::string>();
77  if (sceneConfiguration.nbBricks == 0)
78  PLUGIN_THROW("Invalid number of bricks)");
79  sceneConfiguration.brickSize =
80  sceneConfiguration.sceneSize /
81  static_cast<double>(sceneConfiguration.nbBricks);
82  }
83  }
84  catch (pqxx::sql_error& e)
85  {
86  PLUGIN_THROW(e.what());
87  }
88  transaction.abort();
89  return sceneConfiguration;
90 }

92 void DBConnector::insertBrick(const int32_t brickId, const uint32_t version,
93  const uint32_t nbModels,
94  const std::stringstream& buffer)
95 {
96  pqxx::work transaction(_connection);
97  try
98  {
99  const pqxx::binarystring tmp((void*)buffer.str().c_str(),
100  buffer.str().size() * sizeof(char));
101  transaction.exec_params("INSERT INTO " + _schema +
102  ".brick VALUES ($1, $2, $3, $4)",
103  brickId, version, nbModels, tmp);
104  transaction.commit();
105  PLUGIN_DEBUG("Brick ID " << brickId << " successfully inserted");
106  }
107  catch (pqxx::sql_error& e)
108  {
109  transaction.abort();
110  PLUGIN_THROW(e.what());
111  }
112 }

114 std::stringstream DBConnector::getBrick(const int32_t brickId,
115  const uint32_t& version,
116  uint32_t& nbModels)
117 {
118  std::stringstream s;
119  pqxx::read_transaction transaction(_connection);
120  try
121  {
122  const auto sql = "SELECT nb_models, buffer FROM " + _schema +
123  ".brick WHERE guid=" + std::to_string(brickId) +
124  " AND version=" + std::to_string(version);
125  PLUGIN_DEBUG(sql);
126  auto res = transaction.exec(sql);
127  for (auto c = res.begin(); c != res.end(); ++c)
128  {
129  nbModels = c[0].as<uint32_t>();
130  if (nbModels > 0)
131  {
132  const pqxx::binarystring buffer(c[1]);
133  std::copy(buffer.begin(), buffer.end(),
134  std::ostream_iterator<char>(s));
135  }
136  }
137  }
138  catch (pqxx::sql_error& e)
139  {
140  PLUGIN_THROW(e.what());
141  }
142  transaction.abort();
143 
144  return s;
145 }
