57 void getTangentVectors(const vec3f& normal, vec3f& tangent, vec3f& biTangent)
58 {
59  initializeRandomizer();
60 
61  tangent = make_vec3f(1.f, 0.f, 0.f);
62  if (abs(dot(tangent, normal)) > 0.95f)
63  tangent = make_vec3f(0.f, 1.f, 0.f);
64  biTangent = normalize(cross(tangent, normal));
65  tangent = normalize(cross(biTangent, normal));
66 }

68 inline vec3f getHWRandomVector(const vec3f& normal)
69 {
70  initializeRandomizer();
71  vec3f tangent, biTangent;
72  getTangentVectors(normal, tangent, biTangent);
73  const float rx = getHWRandomValue();
74  const float ry = getHWRandomValue();
75  const float w = sqrt(1.f - ry);
76  const float cx = cos((2.f * M_PI) * rx) * w;
77  const float cy = sin((2.f * M_PI) * rx) * w;
78  const float cz = sqrt(ry);
79  return normalize(cx * tangent + cy * biTangent + cz * normal);
80 }
