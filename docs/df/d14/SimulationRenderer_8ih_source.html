71 static inline uint64 getOffset(const uniform Geometry* geometry, const varying int primID)
72 {
73  if (!geometry)
74  return 0;
75 
76  // The data pointer in all "derived" geometries is just after data members of the base Geometry struct. That's why
77  // array index starts at 1
78  const uniform uint8* data = *((const uniform uint8**)&geometry[1]);
79 
80  const int bytesPerPrimitive = getBytesPerPrimitive(geometry->cppEquivalent);
81  const uint64 bytesPerPrimitive64 = (uint64)bytesPerPrimitive;
82  if (primID * bytesPerPrimitive64 > 0x7FFFFFFF)
83  data = (const uniform uint8*)((uint64)data + bytesPerPrimitive64 * primID);
84  else
85  data += bytesPerPrimitive * primID;
86 
87  return *((const uniform uint64*)data);
88 }

90 inline vec4f getSimulationValue(const uniform SimulationRenderer* uniform self, varying DifferentialGeometry* dg,
91  const varying int primID)
92 {
93  const uint64 offset = getOffset(dg->geometry, primID);
94  if (offset != NO_USER_DATA && offset < self->simulationDataSize)
95  {
96  const varying float value = self->simulationData[offset];
97  const uniform TransferFunction* uniform tf = self->transferFunction;
98  return make_vec4f(tf->getColorForValue(tf, value), tf->getOpacityForValue(tf, value));
99  }
100  return make_vec4f(0.f, 1.f, 0.f, 1.f);
101 }
