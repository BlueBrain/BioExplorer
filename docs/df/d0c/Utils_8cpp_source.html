87 Quaterniond quatLookAt(const Vector3f& dir)
88 {
89  const Vector3f d = normalize(dir);
90  Vector3f u = Vector3f(0.f, 0.f, 1.f);
91  if (abs(dot(d, u)) > 0.99f)
92  u = Vector3f(0.f, 1.f, 0.f);
93  return glm::quatLookAt(d, u);
94 }

117 bool isClipped(const Vector3f& position, const Vector4fs& clippingPlanes)
118 {
119  if (clippingPlanes.empty())
120  return false;
121 
122  bool visible = true;
123  for (auto plane : clippingPlanes)
124  {
125  const Vector3f normal = {plane.x, plane.y, plane.z};
126  const float d = plane.w;
127  const float distance = dot(normal, position) + d;
128  visible &= (distance > 0.f);
129  }
130  return !visible;
131 }

133 Transformation getSphericalPosition(const Vector3f& position,
134  const float radius, const size_t occurence,
135  const size_t occurences,
136  const RandomizationDetails& randInfo)
137 {
138  const float offset = 2.f / occurences;
139  const float increment = M_PI * (3.f - sqrt(5.f));
140  const size_t index = (occurence + randInfo.seed) % occurences;
141 
142  // Position randomizer
143  float R = radius;
144  if (randInfo.positionSeed != 0 &&
145  randInfo.randomizationType == PositionRandomizationType::radial)
146  R *= 1.f +
147  randInfo.positionStrength * rnd3(randInfo.positionSeed + index);
148 
149  // Sphere filling
150  const float y = ((occurence * offset) - 1.f) + offset / 2.f;
151  const float r = sqrt(1.f - pow(y, 2.f));
152  const float phi = index * increment;
153  const float x = cos(phi) * r;
154  const float z = sin(phi) * r;
155  Vector3f d{x, y, z};
156  Vector3f pos;
157  if (randInfo.randomizationType == PositionRandomizationType::radial)
158  pos = (R + position.y) * d;
159  else
160  pos = position + R * d;
161 
162  // rotation randomizer
163  if (randInfo.rotationSeed != 0)
164  d = d + randInfo.rotationStrength *
165  Vector3f(rnd2(randInfo.rotationSeed + index * 2),
166  rnd2(randInfo.rotationSeed + index * 3),
167  rnd2(randInfo.rotationSeed + index * 5));
168 
169  Transformation transformation;
170  transformation.setTranslation(pos);
171  transformation.setRotation(quatLookAt(normalize(d), UP_VECTOR));
172  return transformation;
173 }

175 Transformation getFanPosition(const Vector3f& position, const float radius,
176  const size_t occurence, const size_t occurences,
177  const RandomizationDetails& randInfo)
178 {
179  const float offset = 2.f / occurences;
180  const float increment = 0.1f * M_PI * (3.f - sqrt(5.f));
181 
182  // Randomizer
183  float R = radius;
184  if (randInfo.seed != 0 &&
185  randInfo.randomizationType == PositionRandomizationType::radial)
186  R *= 1.f + rnd1() / 30.f;
187 
188  // Sphere filling
189  const float y = ((occurence * offset) - 1.f) + offset / 2.f;
190  const float r = sqrt(1.f - pow(y, 2.f));
191  const float phi = ((occurence + randInfo.seed) % occurences) * increment;
192  const float x = cos(phi) * r;
193  const float z = sin(phi) * r;
194  const Vector3f d{x, y, z};
195 
196  Transformation transformation;
197  transformation.setTranslation(position + R * d);
198  transformation.setRotation(quatLookAt(d, UP_VECTOR));
199  return transformation;
200 }

202 Transformation getPlanarPosition(const Vector3f& position, const float size,
203  const RandomizationDetails& randInfo)
204 {
205  float up = 0.f;
206  if (randInfo.seed != 0 &&
207  randInfo.randomizationType == PositionRandomizationType::radial)
208  up = rnd1() / 20.f;
209 
210  Transformation transformation;
211  transformation.setTranslation(position +
212  Vector3f(rnd1() * size, up, rnd1() * size));
213  transformation.setRotation(quatLookAt({0.f, 1.f, 0.f}, UP_VECTOR));
214  return transformation;

217 Transformation getCubicPosition(const Vector3f& position, const float size,
218  const RandomizationDetails& randInfo)
219 {
220  Vector3f pos =
221  position + Vector3f(rnd1() * size, rnd1() * size, rnd1() * size);
222  Quaterniond dir = quatLookAt({rnd1(), rnd1(), rnd1()}, UP_VECTOR);
223 
224  if (randInfo.positionSeed != 0)
225  {
226  const Vector3f posOffset = randInfo.positionStrength *
227  Vector3f(rnd2(randInfo.positionSeed),
228  rnd2(randInfo.positionSeed + 1),
229  rnd2(randInfo.positionSeed + 2));
230 
231  pos += posOffset;
232  }
233 
234  if (randInfo.rotationSeed != 0)
235  dir = dir + randomQuaternion(randInfo.rotationSeed);
236 
237  Transformation transformation;
238  transformation.setTranslation(pos);
239  transformation.setRotation(dir);
240  return transformation;
241 }

243 float sinusoide(const float x, const float z)
244 {
245  return 0.2f * cos(x) * sin(z) + 0.05f * cos(x * 2.3f) * sin(z * 4.6f);
246 }

248 Transformation getSinosoidalPosition(const Vector3f& position, const float size,
249  const float amplitude,
250  const size_t occurence,
251  const RandomizationDetails& randInfo)
252 {
253  const float step = 0.01f;
254  const float angle = 0.01f;
255  float up = 1.f;
256  if (randInfo.positionSeed != 0 &&
257  randInfo.randomizationType == PositionRandomizationType::radial)
258  up = 1.f + randInfo.positionStrength *
259  rnd3((randInfo.positionSeed + occurence) * 10);
260 
261  const float x = rnd1() * size;
262  const float z = rnd1() * size;
263  const float y = amplitude * up * sinusoide(x * angle, z * angle);
264 
265  Vector3f pos = Vector3f(x, y, z);
266 
267  const Vector3f v1 =
268  Vector3f(x + step,
269  amplitude * up * sinusoide((x + step) * angle, z * angle), z) -
270  pos;
271  const Vector3f v2 =
272  Vector3f(x, amplitude * up * sinusoide(x * angle, (z + step) * angle),
273  z + step) -
274  pos;
275 
276  pos += position;
277 
278  Vector3f d = cross(normalize(v1), normalize(v2));
279  if (randInfo.rotationSeed != 0)
280  d = d + randInfo.rotationStrength *
281  Vector3f(rnd2(randInfo.rotationSeed + occurence + 1),
282  rnd2(randInfo.rotationSeed + occurence + 2),
283  rnd2(randInfo.rotationSeed + occurence + 3));
284 
285  Transformation transformation;
286  transformation.setTranslation(pos);
287  transformation.setRotation(quatLookAt(normalize(d), UP_VECTOR));
288  return transformation;
289 }

291 Transformation getBezierPosition(const Vector3fs& points, const float scale,
292  const float t)
293 {
294  Vector3fs bezierPoints = points;
295  for (auto& bezierPoint : bezierPoints)
296  bezierPoint *= scale;
297 
298  size_t i = bezierPoints.size() - 1;
299  while (i > 0)
300  {
301  for (size_t k = 0; k < i; ++k)
302  bezierPoints[k] =
303  bezierPoints[k] + t * (bezierPoints[k + 1] - bezierPoints[k]);
304  --i;
305  }
306 
307  Transformation transformation;
308  transformation.setTranslation(bezierPoints[0]);
309  transformation.setRotation(quatLookAt(
310  normalize(cross({0.f, 0.f, 1.f}, bezierPoints[1] - bezierPoints[0])),
311  UP_VECTOR));
312  return transformation;
313 }

315 Transformation getSphericalToPlanarPosition(
316  const Vector3f& center, const float radius, const size_t occurence,
317  const size_t occurences, const RandomizationDetails& randInfo,
318  const float morphingStep)
319 {
320  const float offset = 2.f / occurences;
321  const float increment = M_PI * (3.f - sqrt(5.f));
322  const size_t index = (occurence + randInfo.seed) % occurences;
323 
324  // Position randomizer
325  float R = radius;
326  if (randInfo.positionSeed != 0 &&
327  randInfo.randomizationType == PositionRandomizationType::radial)
328  R *= 1.f +
329  randInfo.positionStrength * rnd3(randInfo.positionSeed + index);
330 
331  // Sphere filling
332  const float y = ((occurence * offset) - 1.f) + (offset / 2.f);
333  const float r = sqrt(1.f - pow(y, 2.f));
334  const float phi = index * increment;
335  const float x = cos(phi) * r;
336  const float z = sin(phi) * r;
337 
338  Vector3f startPos;
339  Vector3f startDir = Vector3f(x, y, z);
340  if (randInfo.randomizationType == PositionRandomizationType::radial)
341  startPos = (R + center.y) * startDir;
342  else
343  startPos = center + R * startDir;
344 
345  Vector3f endPos = startPos;
346 
347  // rotation randomizer
348  if (randInfo.rotationSeed != 0)
349  startDir =
350  startDir + randInfo.rotationStrength *
351  Vector3f(rnd2(randInfo.rotationSeed + index * 2),
352  rnd2(randInfo.rotationSeed + index * 3),
353  rnd2(randInfo.rotationSeed + index * 5));
354 
355  R = radius;
356  const float endRadius = R * 1.75f;
357 
358  endPos.y = -R;
359  endPos = endPos + (1.f - (startPos.y + R) / endRadius) *
360  Vector3f(endRadius, 0.f, endRadius) *
361  normalize(Vector3f(startDir.x, 0.f, startDir.z));
362 
363  const Vector3f endDir{0.f, 1.f, 0.f};
364  const Vector3f d = startDir + morphingStep * (endDir - startDir);
365 
366  Transformation transformation;
367  transformation.setTranslation(endPos * morphingStep +
368  startPos * (1.f - morphingStep));
369  transformation.setRotation(quatLookAt(normalize(d), UP_VECTOR));
370  return transformation;
371 }

524 Quaterniond randomQuaternion(const size_t seed)
525 {
526  double x, y, z, u, v, w, s;
527  do
528  {
529  x = rnd2(seed);
530  y = rnd2(seed + 1);
531  z = x * x + y * y;
532  } while (z > 1.0);
533  do
534  {
535  u = rnd2(seed + 2);
536  v = rnd2(seed + 3);
537  w = u * u + v * v;
538  } while (w > 1.0);
539  s = sqrt((1.0 - z) / w);
540  return Quaterniond(x, y, s * u, s * v);
