50 bool isClipped(const Vector3f& position, const Vector4fs& clippingPlanes)
51 {
52  if (clippingPlanes.empty())
53  return false;
54 
55  bool visible = true;
56  for (auto plane : clippingPlanes)
57  {
58  const Vector3f normal = {plane.x, plane.y, plane.z};
59  const float d = plane.w;
60  const float distance = dot(normal, position) + d;
61  visible &= (distance > 0.f);
62  }
63  return !visible;
64 }

66 void getSphericalPosition(const size_t rnd, const float assemblyRadius,
67  const float height,
68  const PositionRandomizationType randomizationType,
69  const size_t randomSeed, const size_t occurence,
70  const size_t occurences, const Vector3f& position,
71  Vector3f& pos, Vector3f& dir)
72 {
73  const float offset = 2.f / occurences;
74  const float increment = M_PI * (3.f - sqrt(5.f));
75 
76  // Randomizer
77  float radius = assemblyRadius;
78  if (randomSeed != 0 &&
79  randomizationType == PositionRandomizationType::radial)
80  radius *= 1.f + height * (float(rand() % 1000 - 500) / 30000.f);
81 
82  // Sphere filling
83  const float y = ((occurence * offset) - 1.f) + (offset / 2.f);
84  const float r = sqrt(1.f - pow(y, 2.f));
85  const float phi = ((occurence + rnd) % occurences) * increment;
86  const float x = cos(phi) * r;
87  const float z = sin(phi) * r;
88  dir = {x, y, z};
89  pos = position + radius * dir;
90 }

92 void getFanPosition(const size_t rnd, const float assemblyRadius,
93  const PositionRandomizationType randomizationType,
94  const size_t randomSeed, const size_t occurence,
95  const size_t occurences, const Vector3f& position,
96  Vector3f& pos, Vector3f& dir)
97 {
98  const float offset = 2.f / occurences;
99  const float increment = 0.1f * M_PI * (3.f - sqrt(5.f));
100 
101  // Randomizer
102  float radius = assemblyRadius;
103  if (randomSeed != 0 &&
104  randomizationType == PositionRandomizationType::radial)
105  radius *= 1.f + (float(rand() % 1000 - 500) / 30000.f);
106 
107  // Sphere filling
108  const float y = ((occurence * offset) - 1.f) + (offset / 2.f);
109  const float r = sqrt(1.f - pow(y, 2.f));
110  const float phi = ((occurence + rnd) % occurences) * increment;
111  const float x = cos(phi) * r;
112  const float z = sin(phi) * r;
113  dir = {x, y, z};
114  pos = position + radius * dir;
115 }

117 void getPlanarPosition(const float assemblyRadius,
118  const PositionRandomizationType randomizationType,
119  const size_t randomSeed, const Vector3f& position,
120  Vector3f& pos, Vector3f& dir)
121 {
122  float up = 0.f;
123  if (randomSeed != 0 &&
124  randomizationType == PositionRandomizationType::radial)
125  up = (float(rand() % 1000 - 500) / 20000.f);
126 
127  pos = position +
128  Vector3f(float(rand() % 1000 - 500) / 1000.f * assemblyRadius, up,
129  float(rand() % 1000 - 500) / 1000.f * assemblyRadius);
130  dir = {0.f, 1.f, 0.f};
131 }

145 float sinusoide(const float x, const float z)
146 {
147  return 0.2f * cos(x) * sin(z) + 0.05f * cos(x * 2.3f) * sin(z * 4.6f);
148 }

150 void getSinosoidalPosition(const float size, const float height,
151  const PositionRandomizationType randomizationType,
152  const size_t randomSeed, const Vector3f& position,
153  Vector3f& pos, Vector3f& dir)
154 {
155  const float step = 0.01f;
156  const float angle = 0.01f;
157  float up = 1.f;
158  if (randomSeed != 0 &&
159  randomizationType == PositionRandomizationType::radial)
160  up = 1.f + (float(rand() % 1000 - 500) / 5000.f);
161 
162  const float x = float(rand() % 1000 - 500) / 1000.f * size;
163  const float z = float(rand() % 1000 - 500) / 1000.f * size;
164  const float y = height * up * sinusoide(x * angle, z * angle);
165 
166  pos = Vector3f(x, y, z);
167 
168  const Vector3f v1 =
169  Vector3f(x + step,
170  height * up * sinusoide((x + step) * angle, z * angle), z) -
171  pos;
172  const Vector3f v2 =
173  Vector3f(x, height * up * sinusoide(x * angle, (z + step) * angle),
174  z + step) -
175  pos;
176 
177  pos += position;
178  dir = normalize(cross(normalize(v1), normalize(v2)));
179 }

181 void getBezierPosition(const Vector3fs& points, const float assemblyRadius,
182  const float t, Vector3f& pos, Vector3f& dir)
183 {
184  Vector3fs bezierPoints = points;
185  for (auto& bezierPoint : bezierPoints)
186  bezierPoint *= assemblyRadius;
187 
188  size_t i = bezierPoints.size() - 1;
189  while (i > 0)
190  {
191  for (size_t k = 0; k < i; ++k)
192  bezierPoints[k] =
193  bezierPoints[k] + t * (bezierPoints[k + 1] - bezierPoints[k]);
194  --i;
195  }
196  dir = normalize(cross({0, 0, 1}, bezierPoints[1] - bezierPoints[0]));
197  pos = bezierPoints[0];
198 }
