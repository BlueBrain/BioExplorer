61 void VectorFieldsHandler::_buildOctree()
62 {
63  PLUGIN_INFO(3, "Building Vector Octree");
64 
65  auto& scene = _engine.getScene();
66  const auto& clipPlanes = getClippingPlanes(scene);
67 
68  OctreeVectors vectors;
69  uint32_t count{0};
70  const uint32_t densityRatio = 1.f / _density;
71 
72  Boxd bounds;
73  const auto& modelDescriptors = scene.getModelDescriptors();
74  for (const auto modelDescriptor : modelDescriptors)
75  {
76  if (!_modelIds.empty())
77  {
78  const auto modelId = modelDescriptor->getModelID();
79  const auto it = std::find(_modelIds.begin(), _modelIds.end(), modelId);
80  if (it == _modelIds.end())
81  continue;
82  }
83 
84  const auto& instances = modelDescriptor->getInstances();
85  for (const auto& instance : instances)
86  {
87  const auto& tf = instance.getTransformation();
88  const auto& model = modelDescriptor->getModel();
89  const auto& conesMap = model.getCones();
90  for (const auto& cones : conesMap)
91  {
92  if (cones.first != BOUNDINGBOX_MATERIAL_ID && cones.first != SECONDARY_MODEL_MATERIAL_ID)
93  for (const auto& cone : cones.second)
94  {
95  const Vector3f center =
96  tf.getTranslation() + tf.getRotation() * (Vector3d(cone.center) - tf.getRotationCenter());
97  const Vector3f up =
98  tf.getTranslation() + tf.getRotation() * (Vector3d(cone.up) - tf.getRotationCenter());
99 
100  const Vector3d c = center;
101  const Vector3d u = center;
102  if (isClipped(c, clipPlanes) || isClipped(u, clipPlanes))
103  {
104  ++count;
105  continue;
106  }
107 
108  if (count % densityRatio == 0)
109  {
110  bounds.merge(center + cone.centerRadius);
111  bounds.merge(center - cone.centerRadius);
112  bounds.merge(up + cone.upRadius);
113  bounds.merge(up - cone.upRadius);
114 
115  OctreeVector vector;
116  vector.position = center;
117  vector.direction = up - center;
118  vectors.push_back(vector);
119  }
120  ++count;
121  }
122  }
123  }
124  }
125 
126  // Events bounds
127  Vector3f sceneSize = bounds.getSize();
128  Vector3f center = bounds.getCenter();
129  Vector3f extendedHalfSize = sceneSize * 0.5f;
130 
131  // Expand volume by 25%
132  const float boundsExpansion = 1.25f;
133  bounds.merge(center + extendedHalfSize * boundsExpansion);
134  bounds.merge(center - extendedHalfSize * boundsExpansion);
135  sceneSize = bounds.getSize();
136  center = bounds.getCenter();
137  extendedHalfSize = sceneSize * 0.5f;
138 
139  // Compute volume information
140  const Vector3f minAABB = center - extendedHalfSize;
141  const Vector3f maxAABB = center + extendedHalfSize;
142 
143  // Build acceleration structure
144  const VectorOctree accelerator(vectors, _voxelSize, minAABB, maxAABB);
145  const auto& indices = accelerator.getFlatIndices();
146  const auto& data = accelerator.getFlatData();
147 
148  const uint32_t volumeSize = accelerator.getVolumeSize();
149  _offset = center - extendedHalfSize;
150  _dimensions = accelerator.getVolumeDimensions();
151  _spacing = sceneSize / Vector3f(_dimensions);
152 
153  const auto& params = _engine.getParametersManager().getApplicationParameters();
154  const auto& engineName = params.getEngine();
155 #ifdef USE_OSPRAY
156  if (engineName == ENGINE_OSPRAY)
157  {
158  _frameData.push_back(_offset.x);
159  _frameData.push_back(_offset.y);
160  _frameData.push_back(_offset.z);
161  _frameData.push_back(_spacing.x);
162  _frameData.push_back(_spacing.y);
163  _frameData.push_back(_spacing.z);
164  _frameData.push_back(_dimensions.x);
165  _frameData.push_back(_dimensions.y);
166  _frameData.push_back(_dimensions.z);
167  _frameData.push_back(accelerator.getOctreeSize());
168  _frameData.push_back(indices.size());
169  _frameData.insert(_frameData.end(), indices.begin(), indices.end());
170  _frameData.insert(_frameData.end(), data.begin(), data.end());
171  _frameSize = _frameData.size();
172 
173  const size_t materialId = 0;
174  auto material = _model->createMaterial(0, "Octree");
175  const TriangleMesh mesh = createBox(bounds.getMin(), bounds.getMax());
176  _model->getTriangleMeshes()[materialId] = mesh;
177  _model->markInstancesDirty();
178  }
179 #endif
180 
181 #ifdef USE_OPTIX6
182  if (engineName == ENGINE_OPTIX_6)
183  {
184  auto volume = _model->createOctreeVolume(_dimensions, _spacing, DataType::FLOAT);
185  auto optixVolume = dynamic_cast<core::engine::optix::OptiXOctreeVolume*>(volume.get());
186  optixVolume->setOctree(_offset, indices, data, OctreeDataType::vector);
187  _frameData.clear();
188  _frameSize = 0;
189  }
190 #endif
191 
192  PLUGIN_INFO(1, "--------------------------------------------");
193  PLUGIN_INFO(1, "Vector Octree information (" << vectors.size() << " vectors)");
194  PLUGIN_INFO(1, "--------------------------------------------");
195  PLUGIN_INFO(1, "Scene AABB : " << bounds);
196  PLUGIN_INFO(1, "Scene dimension : " << sceneSize);
197  PLUGIN_INFO(1, "Element spacing : " << _spacing);
198  PLUGIN_INFO(1, "Volume dimensions : " << _dimensions);
199  PLUGIN_INFO(1, "Element offset : " << _offset);
200  PLUGIN_INFO(1, "Volume size : " << volumeSize << " bytes");
201  PLUGIN_INFO(1, "Indices size : " << indices.size());
202  PLUGIN_INFO(1, "Data size : " << _frameSize);
203  PLUGIN_INFO(1, "Octree size : " << accelerator.getOctreeSize());
204  PLUGIN_INFO(1, "Octree depth : " << accelerator.getOctreeDepth());
205  PLUGIN_INFO(1, "--------------------------------------------");
206  _octreeInitialized = true;
207 }
