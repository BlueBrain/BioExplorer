27 static const float OPENDECK_RADIUS = 2.55f;

29 void CylindricStereoCamera_initRay(uniform Camera* uniform _self, varying Ray& ray, const varying CameraSample& sample)
30 {
31  uniform CylindricStereoCamera* uniform self = (uniform CylindricStereoCamera * uniform) _self;
32 
33  const vec2f screen = Camera_subRegion(_self, sample.screen);
34 
35  float radius = 0.0f;
36  float alpha = 0.0f;
37  float beta = atan(self->interpupillaryDistance / (2.0f * OPENDECK_RADIUS));
38  if (self->side_by_side)
39  {
40  if (screen.x < 0.5)
41  {
42  radius = self->interpupillaryDistance * 0.5f;
43  alpha = 2.0f * screen.x * -pi;
44  beta = -beta;
45  }
46  else
47  {
48  radius = -self->interpupillaryDistance * 0.5f;
49  alpha = 2.0f * (screen.x - 0.5f) * -pi;
50  }
51  }
52  else
53  {
54  radius = self->interpupillaryDistance * 0.5f;
55  alpha = screen.x * -pi;
56  }
57 
58  const float y = self->imgPlane_size_y * (screen.y - 0.5f);
59  const LinearSpace3f cameraSpace = make_LinearSpace3f(self->dir_cam, self->dir_dv, self->dir_du);
60 
61  vec3f dir = normalize(make_vec3f(sinf(alpha + beta), y, -cosf(alpha + beta)));
62  dir = normalize(cameraSpace * dir);
63 
64  vec3f org;
65  org.x = self->org.x + self->dir_cam.x * radius * cosf(alpha) + self->dir_du.x * radius * sinf(alpha);
66  org.y = self->org.y + self->dir_cam.y * radius * cosf(alpha) + self->dir_du.y * radius * sinf(alpha);
67  org.z = self->org.z + self->dir_cam.z * radius * cosf(alpha) + self->dir_du.z * radius * sinf(alpha);
68 
69  const float time = Camera_shutterTime(_self, sample.time);
70  setRay(ray, org, normalize(dir), self->super.nearClip, inf, time);
71 }

73 /*! create a new ispc-side version of a OpenDeckCylinderStereoCamera - with
74  given pointer to the c-class version - and return both class pointer and
75  pointer to internal data back via the supplied reference parameters. */
76 export void* uniform CylindricStereoCamera_create(void* uniform cppE)
77 {
78  uniform CylindricStereoCamera* uniform self = uniform new uniform CylindricStereoCamera;
79  self->super.cppEquivalent = cppE;
80  self->super.initRay = CylindricStereoCamera_initRay;
81  self->super.doesDOF = false;
82  return self;
83 }
