28 static const float OPENDECK_RADIUS = 2.55f;
29 static const float OPENDECK_HEIGHT = 2.3f;
30 static const float OPENDECK_METALSTRIPE_HEIGHT = 0.045f;
31 static const float PI = 3.141592f;
32 static const float OPENDECK_BEZEL_ANGLE = PI / 180.0f * 7.98995f;
33 static const float ANGLE_PER_BORDER_SEGMENT = (PI - 8.0f * OPENDECK_BEZEL_ANGLE) / 7.0f + OPENDECK_BEZEL_ANGLE;
34 static const float FULL_ANGLE = ANGLE_PER_BORDER_SEGMENT + OPENDECK_BEZEL_ANGLE;

46 // Pass 'seed' by reference to keep randomness state
47 __device__ float4 launch(uint& seed, const float2 screen, const bool use_randomness)
48 {
49  float eyeDelta = 0.0f;
50  float alpha = 0.0f;
51  float3 dPx, dPy;
52 
53  float2 sample = make_float2(launch_index) / screen;
54 
55  if (segmentID <= 13 && segmentID % 2 == 0)
56  {
57  eyeDelta = half_ipd;
58  uint angularOffset = segmentID / 2;
59 
60  if (segmentID == 0)
61  alpha = sample.x * FULL_ANGLE;
62  else if (segmentID == 12)
63  alpha = PI - FULL_ANGLE + sample.x * FULL_ANGLE;
64  else
65  alpha = angularOffset * (FULL_ANGLE - OPENDECK_BEZEL_ANGLE) + sample.x * FULL_ANGLE;
66  }
67  else if (segmentID <= 13 && segmentID % 2 == 1)
68  {
69  eyeDelta = -half_ipd;
70  uint angularOffset = segmentID / 2;
71  if (segmentID == 1)
72  alpha = sample.x * FULL_ANGLE;
73  else if (segmentID == 13)
74  alpha = PI - FULL_ANGLE + sample.x * FULL_ANGLE;
75  else
76  alpha = angularOffset * (FULL_ANGLE - OPENDECK_BEZEL_ANGLE) + sample.x * FULL_ANGLE;
77  }
78  else if (segmentID == 14)
79  {
80  eyeDelta = half_ipd;
81  }
82  else if (segmentID == 15)
83  {
84  eyeDelta = -half_ipd;
85  }
86 
87  float3 pixelPos;
88  if (segmentID <= 13)
89  {
90  pixelPos.x = OPENDECK_RADIUS * -cosf(alpha);
91  pixelPos.y = OPENDECK_METALSTRIPE_HEIGHT + OPENDECK_HEIGHT * sample.y;
92  pixelPos.z = OPENDECK_RADIUS * -sinf(alpha);
93 
94  dPx =
95  make_float3(FULL_ANGLE * OPENDECK_RADIUS * sinf(alpha), 0.0f, FULL_ANGLE * OPENDECK_RADIUS * -cosf(alpha));
96  dPy = make_float3(0.0f, OPENDECK_HEIGHT, 0.0f);
97  }
98  else if (segmentID > 13)
99  {
100  pixelPos.x = 2.0f * OPENDECK_RADIUS * (sample.x - 0.5f);
101  pixelPos.y = 0.0f;
102  pixelPos.z = -OPENDECK_RADIUS * sample.y;
103 
104  dPx = make_float3(2.0f * OPENDECK_RADIUS, 0.0f, 0.0f);
105  dPy = make_float3(0.0f, 0.0f, -OPENDECK_RADIUS);
106  }
107 
108  // The tracking model of the 3d glasses is inversed
109  // so we need to negate CamDu here.
110  const float3 eyeDeltaPos = -headUVec * eyeDelta;
111 
112  optix::Matrix3x3 transform;
113  transform.setCol(0, U);
114  transform.setCol(1, V);
115  transform.setCol(2, W);
116 
117  const float3 d = pixelPos - headPos + eyeDeltaPos;
118  const float dotD = dot(d, d);
119  const float denom = pow(dotD, 1.5f);
120  float3 dir = normalize(d);
121 
122  float3 dirDx = (dotD * dPx - dot(d, dPx) * d) / (denom * screen.x);
123  float3 dirDy = (dotD * dPy - dot(d, dPy) * d) / (denom * screen.y);
124 
125  PerRayData_radiance prd;
126  prd.importance = 1.f;
127  prd.depth = 0;
128  prd.rayDdx = transform * dirDx;
129  prd.rayDdy = transform * dirDy;
130  dir = transform * dir;
131 
132  const float3 org = eye + headPos - eyeDeltaPos;
133  float near = sceneEpsilon;
134  float far = INFINITY;
135 
136  applyClippingPlanes(org, dir, near, far);
137  optix::Ray ray(org, dir, radiance_ray_type, near, far);
138 
139  rtTrace(top_object, ray, prd);
140 
141  return make_float4(make_float3(prd.result) * mainExposure, prd.result.w);
142 }

144 RT_PROGRAM void openDeckCamera()
145 {
146  const size_t2 screen = output_buffer.size();
147  const float2 screen_f = make_float2(screen);
148 
149  uint seed = tea<16>(screen.x * launch_index.y + launch_index.x, frame);
150 
151  const int num_samples = max(1, samples_per_pixel);
152  // We enable randomness if we are using subpixel sampling or accumulation
153  const bool use_randomness = frame > 0 || num_samples > 1;
154 
155  float4 result = make_float4(0.f);
156  for (int i = 0; i < num_samples; i++)
157  result += launch(seed, screen_f, use_randomness);
158  result /= num_samples;
159 
160  float4 acc_val;
161  if (frame > 0)
162  {
163  acc_val = accum_buffer[launch_index];
164  acc_val = lerp(acc_val, result, 1.0f / static_cast<float>(frame + 1));
165  }
166  else
167  acc_val = result;
168 
169  output_buffer[launch_index] = make_color(acc_val);
170 
171  if (accum_buffer.size().x > 1 && accum_buffer.size().y > 1)
172  accum_buffer[launch_index] = acc_val;
173 }
