35 __forceinline__ __device__ float luminance(const float3 &rgb)
36 {
37  const float3 ntsc_luminance = {0.30f, 0.59f, 0.11f};
38  return dot(rgb, ntsc_luminance);

77 static __device__ __inline__ float3 traceRadianceRay(float3 origin,
78  float3 direction,
79  int depth,
80  float importance)
81 {
82  RadiancePRD prd;
83  prd.depth = depth;
84  prd.importance = importance;
85 
86  optixTrace(params.handle, origin, direction, params.scene_epsilon, 1e16f,
87  0.0f, OptixVisibilityMask(1), OPTIX_RAY_FLAG_NONE,
88  RAY_TYPE_RADIANCE, RAY_TYPE_COUNT, RAY_TYPE_RADIANCE,
89  float3_as_args(prd.result),
90  /* Can't use __float_as_uint() because it returns rvalue but
91  payload requires a lvalue */
92  reinterpret_cast<unsigned int &>(prd.importance),
93  reinterpret_cast<unsigned int &>(prd.depth));
94 
95  return prd.result;
96 }

98 static __device__ void phongShadowed()
99 {

106 static __device__ void phongShade(float3 p_Kd, float3 p_Ka, float3 p_Ks,
107  float3 p_Kr, float p_phong_exp,
108  float3 p_normal)
109 {
110  RadiancePRD prd = getRadiancePRD();
111 
112  const float3 ray_orig = optixGetWorldRayOrigin();
113  const float3 ray_dir = optixGetWorldRayDirection();
114  const float ray_t = optixGetRayTmax();
115 
116  float3 hit_point = ray_orig + ray_t * ray_dir;
117 
118  // ambient contribution
119  float3 result = p_Ka * params.ambient_light_color;
120 
121  // compute direct lighting
122  BasicLight light = params.light;
123  float Ldist = length(light.pos - hit_point);
124  float3 L = normalize(light.pos - hit_point);
125  float nDl = dot(p_normal, L);
126 
127  // cast shadow ray
128  float3 light_attenuation = make_float3(static_cast<float>(nDl > 0.0f));
129  if (nDl > 0.0f)
130  {
131  OcclusionPRD shadow_prd;
132  shadow_prd.attenuation = make_float3(1.0f);
133 
134  optixTrace(params.handle, hit_point, L, 0.01f, Ldist, 0.0f,
135  OptixVisibilityMask(1), OPTIX_RAY_FLAG_NONE,
136  RAY_TYPE_OCCLUSION, RAY_TYPE_COUNT, RAY_TYPE_OCCLUSION,
137  float3_as_args(shadow_prd.attenuation));
138 
139  light_attenuation = shadow_prd.attenuation;
140  }
141 
142  // If not completely shadowed, light the hit point
143  if (fmaxf(light_attenuation) > 0.0f)
144  {
145  float3 Lc = light.color * light_attenuation;
146 
147  result += p_Kd * nDl * Lc;
148 
149  float3 H = normalize(L - ray_dir);
150  float nDh = dot(p_normal, H);
151  if (nDh > 0)
152  {
153  float power = pow(nDh, p_phong_exp);
154  result += p_Ks * power * Lc;
155  }
156  }
157 
158  if (fmaxf(p_Kr) > 0)
159  {
160  // ray tree attenuation
161  float new_importance = prd.importance * luminance(p_Kr);
162  int new_depth = prd.depth + 1;
163 
164  // reflection ray
165  // compare new_depth to max_depth - 1 to leave room for a potential
166  // shadow ray trace
167  if (new_importance >= 0.01f && new_depth <= params.max_depth - 1)
168  {
169  float3 R = reflect(ray_dir, p_normal);
170 
171  result += p_Kr *
172  traceRadianceRay(hit_point, R, new_depth, new_importance);
173  }
174  }
175 
176  // pass the color back
177  prd.result = result;
178  setRadiancePRD(prd);
179 }

181 extern "C" __global__ void __closesthit__radiance()
182 {
183  const HitGroupData *sbt_data = (HitGroupData *)optixGetSbtDataPointer();
184  const Phong &phong = sbt_data->shading.phong;
185 
186  const float3 object_normal =
187  make_float3(__uint_as_float(optixGetAttribute_0()),
188  __uint_as_float(optixGetAttribute_1()),
189  __uint_as_float(optixGetAttribute_2()));
190 
191  const float3 world_normal =
192  normalize(optixTransformNormalFromObjectToWorldSpace(object_normal));
193  const float3 normal =
194  faceforward(world_normal, -optixGetWorldRayDirection(), world_normal);
195  phongShade(phong.Kd, phong.Ka, phong.Ks, phong.Kr, phong.phong_exp, normal);
196 }
