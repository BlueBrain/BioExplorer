72 ModelDescriptorPtr WhiteMatterLoader::importFromStorage(const std::string& storage, const LoaderProgress& callback,
73  const PropertyMap& properties) const
74 {
75  PropertyMap props = _defaults;
76  props.merge(properties);
77 
78  details::WhiteMatterDetails details;
79  const auto baseName = std::filesystem::path(storage).filename();
80  details.assemblyName = baseName;
81  details.populationName = baseName;
82  details.sqlFilter = props.getProperty<std::string>(LOADER_PROPERTY_DATABASE_SQL_FILTER.name);
83  details.radius = props.getProperty<double>(LOADER_PROPERTY_RADIUS_MULTIPLIER.name);
84  const auto position = props.getProperty<std::array<double, 3>>(LOADER_PROPERTY_POSITION.name);
85  const Vector3d pos = core::Vector3d(position[0], position[1], position[2]);
86  const auto rotation = props.getProperty<std::array<double, 4>>(LOADER_PROPERTY_ROTATION.name);
87  const Quaterniond rot = core::Quaterniond(rotation[3], rotation[0], rotation[1], rotation[2]);
88  const auto scale = props.getProperty<std::array<double, 3>>(LOADER_PROPERTY_SCALE.name);
89  WhiteMatter whiteMatter(_scene, details, pos, rot, callback);
90  return std::move(whiteMatter.getModelDescriptor());
91 }
