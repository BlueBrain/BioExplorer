33 Assembly::Assembly(Scene &scene, const AssemblyDetails &descriptor)
34  : _details(descriptor)
35  , _scene(scene)
36 {
37  if (descriptor.position.size() != 3)
38  PLUGIN_THROW(std::runtime_error(
39  "Position must be a sequence of 3 float values"));
40 
41  if (descriptor.rotation.size() != 4)
42  PLUGIN_THROW(std::runtime_error(
43  "rotation must be a sequence of 4 float values"));
44 
45  if (descriptor.clippingPlanes.size() % 4 != 0)
46  PLUGIN_THROW(std::runtime_error(
47  "Clipping planes must be defined by 4 float values"));
48  const auto &cp = descriptor.clippingPlanes;
49  for (size_t i = 0; i < cp.size(); i += 4)
50  _clippingPlanes.push_back({cp[i], cp[i + 1], cp[i + 2], cp[i + 3]});
51 
52  PLUGIN_INFO << "Adding assembly [" << descriptor.name << "]" << std::endl;
53 }

55 Assembly::~Assembly()
56 {

58  {
59  const auto modelId = protein.second->getModelDescriptor()->getModelID();
60  PLUGIN_INFO << "Removing protein [" << modelId << "] [" << protein.first
61  << "] from assembly [" << _details.name << "]" << std::endl;
62  _scene.removeModel(protein.second->getModelDescriptor()->getModelID());
63  }
64  for (const auto &meshBasedMembrane : _meshBasedMembranes)
65  {
66  const auto modelId =
67  meshBasedMembrane.second->getModelDescriptor()->getModelID();
68  PLUGIN_INFO << "Removing mesh [" << modelId << "] ["
69  << meshBasedMembrane.first << "] from assembly ["
70  << _details.name << "]" << std::endl;
71  _scene.removeModel(
72  meshBasedMembrane.second->getModelDescriptor()->getModelID());
73  }
74  if (_rnaSequence)
75  {
76  const auto modelId = _rnaSequence->getModelDescriptor()->getModelID();
77  PLUGIN_INFO << "Removing RNA sequence [" << modelId
78  << "] from assembly [" << _details.name << "]" << std::endl;

166 void Assembly::_processInstances(
167  ModelDescriptorPtr md, const std::string &name, const AssemblyShape shape,
168  const floats &assemblyParams, const size_t occurrences,
169  const Vector3f &position, const Quaterniond &rotation,
170  const size_ts &allowedOccurrences, const size_t randomSeed,
171  const PositionRandomizationType &randomizationType)
172 {
173  const float offset = 2.f / occurrences;
174  const float increment = M_PI * (3.f - sqrt(5.f));
175 
176  srand(randomSeed);
177  size_t rnd{1};
178  if (occurrences != 0 && randomSeed != 0 &&
179  randomizationType == PositionRandomizationType::circular)
180  rnd = rand() % occurrences;
181 
182  const Quaterniond assemblyrotation = {_details.rotation[0],
183  _details.rotation[1],
184  _details.rotation[2],
185  _details.rotation[3]};
186  const Vector3f assemblyPosition = {_details.position[0],
187  _details.position[1],
188  _details.position[2]};
189 
190  // Shape parameters
191  const auto &params = assemblyParams;
192  const float size = (params.size() > 0 ? params[0] : 0.f);
193  RandomizationInformation randInfo;
194  randInfo.seed = randomSeed;
195  randInfo.randomizationType = randomizationType;
196  randInfo.positionStrength = (params.size() > 2 ? params[2] : 0.f);
197  randInfo.rotationStrength = (params.size() > 4 ? params[4] : 0.f);
198  const float extraParameter = (params.size() > 5 ? params[5] : 0.f);
199 
200  // Shape
201  uint64_t count = 0;
202  for (uint64_t occurence = 0; occurence < occurrences; ++occurence)
203  {
204  if (!allowedOccurrences.empty() &&
205  std::find(allowedOccurrences.begin(), allowedOccurrences.end(),
206  occurence) == allowedOccurrences.end())
207  continue;
208 
209  randInfo.positionSeed =
210  (params.size() > 1 ? (params[1] == 0 ? 0 : params[1] + occurence)
211  : 0);
212  randInfo.rotationSeed =
213  (params.size() > 3 ? (params[3] == 0 ? 0 : params[3] + occurence)
214  : 0);
215 
216  Transformation transformation;
217  switch (shape)
218  {
219  case AssemblyShape::spherical:
220  {
221  transformation = getSphericalPosition(position, size, occurence,
222  occurrences, randInfo);
223  break;
224  }
225  case AssemblyShape::sinusoidal:
226  {
227  transformation =
228  getSinosoidalPosition(position, size, extraParameter, occurence,
229  randInfo);
230  break;
231  }
232  case AssemblyShape::cubic:
233  {
234  transformation = getCubicPosition(position, size, randInfo);
235  break;
236  }
237  case AssemblyShape::fan:
238  {
239  transformation = getFanPosition(position, size, occurence,
240  occurrences, randInfo);
241  break;
242  }
243  case AssemblyShape::bezier:
244  {
245  if ((params.size() - 5) % 3 != 0)
246  PLUGIN_THROW(std::runtime_error(
247  "Invalid number of floats in assembly extra parameters"));
248  Vector3fs points;
249  for (uint32_t i = 5; i < params.size(); i += 3)
250  points.push_back(
251  Vector3f(params[i], params[i + 1], params[i + 2]));
252  const auto assemblySize = assemblyParams[0];
253  transformation =
254  getBezierPosition(points, assemblySize,
255  float(occurence) / float(occurrences));
256  break;
257  }
258  case AssemblyShape::spherical_to_planar:
259  {
260  transformation =
261  getSphericalToPlanarPosition(position, size, occurence,
262  occurrences, randInfo,
263  extraParameter);
264  break;
265  }
266  default:
267  transformation = getPlanarPosition(position, size, randInfo);
268  break;
269  }
270 
271  // Final transformation
272  const Vector3f translation =
273  assemblyPosition +
274  Vector3f(assemblyrotation *
275  Vector3d(transformation.getTranslation()));
276 
277  if (isClipped(translation, _clippingPlanes))
278  continue;
279 
280  Transformation finalTransformation;
281  finalTransformation.setTranslation(translation);
282  finalTransformation.setRotation(
283  assemblyrotation * transformation.getRotation() * rotation);
284 
285  if (count == 0)
286  md->setTransformation(finalTransformation);
287  const ModelInstance instance(true, false, finalTransformation);
288  md->addInstance(instance);
289 
290  ++count;
291  }
292 }

294 void Assembly::setColorScheme(const ColorSchemeDetails &csd)
295 {
296  if (csd.palette.size() < 3 || csd.palette.size() % 3 != 0)
297  PLUGIN_THROW(std::runtime_error("Invalid palette size"));
298 
299  ProteinPtr protein{nullptr};
300  auto itProtein = _proteins.find(csd.name);
301  if (itProtein != _proteins.end())
302  protein = (*itProtein).second;
303  else
304  {
305  auto itMesh = _meshBasedMembranes.find(csd.name);
306  if (itMesh != _meshBasedMembranes.end())
307  protein = (*itMesh).second->getProtein();
308  else if (_membrane)
309  {
310  const auto membraneProteins = _membrane->getProteins();
311  auto it = membraneProteins.find(csd.name);
312  if (it != membraneProteins.end())
313  protein = (*it).second;
314  }
315  }
316 
317  if (protein)
318  {
319  Palette palette;
320  for (size_t i = 0; i < csd.palette.size(); i += 3)
321  palette.push_back(
322  {csd.palette[i], csd.palette[i + 1], csd.palette[i + 2]});
323 
324  PLUGIN_INFO << "Applying color scheme to protein " << csd.name
325  << " on assembly " << csd.assemblyName << std::endl;
326  protein->setColorScheme(csd.colorScheme, palette, csd.chainIds);
327 
328  _scene.markModified();
329  }
330  else
331  PLUGIN_ERROR << "Protein " << csd.name << " not found on assembly "
332  << csd.assemblyName << std::endl;
333 }

361 std::string Assembly::getAminoAcidInformation(
362  const AminoAcidInformationDetails &aasd) const
363 {
364  PLUGIN_INFO << "Returning Amino Acid information from protein " << aasd.name
365  << std::endl;
366 
367  std::string response;
368  auto it = _proteins.find(aasd.name);
369  if (it != _proteins.end())
370  {
371  // Sequences
372  for (const auto &sequence : (*it).second->getSequencesAsString())
373  {
374  if (!response.empty())
375  response += "\n";
376  response += sequence.second;
377  }
378 
379  // Glycosylation sites
380  const auto &sites = (*it).second->getGlycosylationSites({});
381  for (const auto &site : sites)
382  {
383  std::string s;
384  for (const auto &index : site.second)
385  {
386  if (!s.empty())
387  s += ",";
388  s += std::to_string(index + 1); // Site indices start a 1, not 0
389  }
390  response += "\n" + s;
391  }
392  }
393  else
394  PLUGIN_THROW(std::runtime_error("Protein not found: " + aasd.name));
395 
396  return response;
397 }

443 void Assembly::setProteinInstanceTransformation(
444  const ProteinInstanceTransformationDetails &descriptor)
445 {
446  ProteinPtr protein{nullptr};
447  auto itProtein = _proteins.find(descriptor.name);
448  if (itProtein != _proteins.end())
449  protein = (*itProtein).second;
450  else
451  PLUGIN_THROW(std::runtime_error("Protein " + descriptor.name +
452  " not found on assembly " +
453  descriptor.assemblyName));
454 
455  auto modelDescriptor = protein->getModelDescriptor();
456 
457  auto &instances = modelDescriptor->getInstances();
458  if (descriptor.instanceIndex >= instances.size())
459  PLUGIN_THROW(std::runtime_error(
460  "Invalid instance index (" +
461  std::to_string(descriptor.instanceIndex) + ") for protein " +
462  descriptor.name + " in assembly " + descriptor.assemblyName));
463 
464  auto instance = modelDescriptor->getInstance(descriptor.instanceIndex);
465  auto &transformation = instance->getTransformation();
466 
467  if (descriptor.position.size() != 3)
468  PLUGIN_THROW(std::runtime_error(
469  "Invalid number of float for position of protein " +
470  descriptor.name + " in assembly " + descriptor.assemblyName));
471  const Vector3f position{descriptor.position[0], descriptor.position[1],
472  descriptor.position[2]};
473 
474  if (descriptor.rotation.size() != 4)
475  PLUGIN_THROW(std::runtime_error(
476  "Invalid number of float for position of protein " +
477  descriptor.name + " in assembly " + descriptor.assemblyName));
478  const Quaterniond rotation{descriptor.rotation[0], descriptor.rotation[1],
479  descriptor.rotation[2], descriptor.rotation[3]};
480 
481  PLUGIN_INFO << "Modifying instance " << descriptor.instanceIndex
482  << " of protein " << descriptor.name << " in assembly "
483  << descriptor.assemblyName << " with position=" << position
484  << " and rotation=" << rotation << std::endl;
485  Transformation newTransformation = transformation;
486  newTransformation.setTranslation(position);
487  newTransformation.setRotation(rotation);
488  instance->setTransformation(newTransformation);
489  if (descriptor.instanceIndex == 0)
490  modelDescriptor->setTransformation(newTransformation);
491 
492  _scene.markModified();
493 }
