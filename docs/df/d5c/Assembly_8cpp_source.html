37 Assembly::Assembly(Scene &scene, const AssemblyDetails &details)
38  : _details(details)
39  , _scene(scene)
40 {
41  if (details.position.size() != 3)
42  PLUGIN_THROW("Position must be a sequence of 3 float values");
43 
44  if (details.rotation.size() != 4)
45  PLUGIN_THROW("rotation must be a sequence of 4 float values");
46 
47  if (details.clippingPlanes.size() % 4 != 0)
48  PLUGIN_THROW("Clipping planes must be defined by 4 float values");
49  const auto &cp = details.clippingPlanes;
50  for (size_t i = 0; i < cp.size(); i += 4)
51  _clippingPlanes.push_back({cp[i], cp[i + 1], cp[i + 2], cp[i + 3]});
52 
53  PLUGIN_INFO("Adding assembly [" << details.name << "]");
54 }

56 Assembly::~Assembly()

60  const auto modelId = protein.second->getModelDescriptor()->getModelID();
61  PLUGIN_INFO("Removing protein [" << modelId << "] [" << protein.first

63  << "]");
64  _scene.removeModel(protein.second->getModelDescriptor()->getModelID());
65  }
66  for (const auto &meshBasedMembrane : _meshBasedMembranes)
67  {
68  const auto modelId =
69  meshBasedMembrane.second->getModelDescriptor()->getModelID();
70  PLUGIN_INFO("Removing mesh ["

72  << "] from assembly [" << _details.name << "]");
73  _scene.removeModel(
74  meshBasedMembrane.second->getModelDescriptor()->getModelID());
75  }
76  if (_rnaSequence)
77  {
78  const auto modelId = _rnaSequence->getModelDescriptor()->getModelID();
79  PLUGIN_INFO("Removing RNA sequence [" << modelId << "] from assembly ["
80  << _details.name << "]");

86 {

89 
90  const Vector3f position = {details.position[0], details.position[1],
91  details.position[2]};

93  details.rotation[2], details.rotation[3]};

95  _processInstances(modelDescriptor, details.name, details.shape,

97  rotation, details.allowedOccurrences, details.randomSeed,
98  details.positionRandomizationType);
99 
100  _proteins[details.name] = std::move(protein);
101  _scene.addModel(modelDescriptor);
102  PLUGIN_INFO(
103  "Number of instances: " << modelDescriptor->getInstances().size());

169 void Assembly::_processInstances(
170  ModelDescriptorPtr md, const std::string &name, const AssemblyShape shape,
171  const floats &assemblyParams, const size_t occurrences,
172  const Vector3f &position, const Quaterniond &rotation,
173  const size_ts &allowedOccurrences, const size_t randomSeed,
174  const PositionRandomizationType &randomizationType)
175 {
176  const float offset = 2.f / occurrences;
177  const float increment = M_PI * (3.f - sqrt(5.f));
178 
179  srand(randomSeed);
180  size_t rnd{1};
181  if (occurrences != 0 && randomSeed != 0 &&
182  randomizationType == PositionRandomizationType::circular)
183  rnd = rand() % occurrences;
184 
185  const Quaterniond assemblyrotation = {_details.rotation[0],
186  _details.rotation[1],
187  _details.rotation[2],
188  _details.rotation[3]};
189  const Vector3f assemblyPosition = {_details.position[0],
190  _details.position[1],
191  _details.position[2]};
192 
193  // Shape parameters
194  const auto &params = assemblyParams;
195  const float size = (params.size() > 0 ? params[0] : 0.f);
196  RandomizationDetails randInfo;
197  randInfo.seed = randomSeed;
198  randInfo.randomizationType = randomizationType;
199  randInfo.positionStrength = (params.size() > 2 ? params[2] : 0.f);
200  randInfo.rotationStrength = (params.size() > 4 ? params[4] : 0.f);
201  const float extraParameter = (params.size() > 5 ? params[5] : 0.f);
202 
203  // Shape
204  uint64_t count = 0;
205  for (uint64_t occurence = 0; occurence < occurrences; ++occurence)
206  {
207  if (!allowedOccurrences.empty() &&
208  std::find(allowedOccurrences.begin(), allowedOccurrences.end(),
209  occurence) == allowedOccurrences.end())
210  continue;
211 
212  randInfo.positionSeed =
213  (params.size() > 1 ? (params[1] == 0 ? 0 : params[1] + occurence)
214  : 0);
215  randInfo.rotationSeed =
216  (params.size() > 3 ? (params[3] == 0 ? 0 : params[3] + occurence)
217  : 0);
218 
219  Transformation transformation;
220  switch (shape)
221  {
222  case AssemblyShape::spherical:
223  {
224  transformation = getSphericalPosition(position, size, occurence,
225  occurrences, randInfo);
226  break;
227  }
228  case AssemblyShape::sinusoidal:
229  {
230  transformation =
231  getSinosoidalPosition(position, size, extraParameter, occurence,
232  randInfo);
233  break;
234  }
235  case AssemblyShape::cubic:
236  {
237  transformation = getCubicPosition(position, size, randInfo);
238  break;
239  }
240  case AssemblyShape::fan:
241  {
242  transformation = getFanPosition(position, size, occurence,
243  occurrences, randInfo);
244  break;
245  }
246  case AssemblyShape::bezier:
247  {
248  if ((params.size() - 5) % 3 != 0)
249  PLUGIN_THROW(
250  "Invalid number of floats in assembly extra parameters");
251  Vector3fs points;
252  for (uint32_t i = 5; i < params.size(); i += 3)
253  points.push_back(
254  Vector3f(params[i], params[i + 1], params[i + 2]));
255  const auto assemblySize = assemblyParams[0];
256  transformation =
257  getBezierPosition(points, assemblySize,
258  float(occurence) / float(occurrences));
259  break;
260  }
261  case AssemblyShape::spherical_to_planar:
262  {
263  transformation =
264  getSphericalToPlanarPosition(position, size, occurence,
265  occurrences, randInfo,
266  extraParameter);
267  break;
268  }
269  default:
270  transformation = getPlanarPosition(position, size, randInfo);
271  break;
272  }
273 
274  // Final transformation
275  const Vector3f translation =
276  assemblyPosition +
277  Vector3f(assemblyrotation *
278  Vector3d(transformation.getTranslation()));
279 
280  if (isClipped(translation, _clippingPlanes))
281  continue;
282 
283  Transformation finalTransformation;
284  finalTransformation.setTranslation(translation);
285  finalTransformation.setRotation(
286  assemblyrotation * transformation.getRotation() * rotation);
287 
288  if (count == 0)
289  md->setTransformation(finalTransformation);
290  const ModelInstance instance(true, false, finalTransformation);
291  md->addInstance(instance);
292 
293  ++count;
294  }
295 }

297 void Assembly::setColorScheme(const ColorSchemeDetails &csd)
298 {
299  if (csd.palette.size() < 3 || csd.palette.size() % 3 != 0)
300  PLUGIN_THROW("Invalid palette size");
301 
302  ProteinPtr protein{nullptr};
303  auto itProtein = _proteins.find(csd.name);
304  if (itProtein != _proteins.end())
305  protein = (*itProtein).second;
306  else
307  {
308  auto itMesh = _meshBasedMembranes.find(csd.name);
309  if (itMesh != _meshBasedMembranes.end())
310  protein = (*itMesh).second->getProtein();
311  else if (_membrane)
312  {
313  const auto membraneProteins = _membrane->getProteins();
314  auto it = membraneProteins.find(csd.name);
315  if (it != membraneProteins.end())
316  protein = (*it).second;
317  }
318  }
319 
320  if (protein)
321  {
322  Palette palette;
323  for (size_t i = 0; i < csd.palette.size(); i += 3)
324  palette.push_back(
325  {csd.palette[i], csd.palette[i + 1], csd.palette[i + 2]});
326 
327  PLUGIN_INFO("Applying color scheme to protein "
328  << csd.name << " on assembly " << csd.assemblyName);
329  protein->setColorScheme(csd.colorScheme, palette, csd.chainIds);
330 
331  _scene.markModified();
332  }
333  else
334  PLUGIN_ERROR("Protein " << csd.name << " not found on assembly "
335  << csd.assemblyName);
336 }

364 std::string Assembly::getAminoAcidInformation(
365  const AminoAcidInformationDetails &aasd) const
366 {
367  PLUGIN_INFO("Returning Amino Acid information from protein " << aasd.name);
368 
369  std::string response;
370  auto it = _proteins.find(aasd.name);
371  if (it != _proteins.end())
372  {
373  // Sequences
374  for (const auto &sequence : (*it).second->getSequencesAsString())
375  {
376  if (!response.empty())
377  response += "\n";
378  response += sequence.second;
379  }
380 
381  // Glycosylation sites
382  const auto &sites = (*it).second->getGlycosylationSites({});
383  for (const auto &site : sites)
384  {
385  std::string s;
386  for (const auto &index : site.second)
387  {
388  if (!s.empty())
389  s += ",";
390  s += std::to_string(index + 1); // Site indices start a 1, not 0
391  }
392  response += "\n" + s;
393  }
394  }
395  else
396  PLUGIN_THROW("Protein not found: " + aasd.name);
397 
398  return response;
399 }

442 void Assembly::setProteinInstanceTransformation(
443  const ProteinInstanceTransformationDetails &details)
444 {
445  ProteinPtr protein{nullptr};
446  auto itProtein = _proteins.find(details.name);
447  if (itProtein != _proteins.end())
448  protein = (*itProtein).second;
449  else
450  PLUGIN_THROW("Protein " + details.name + " not found on assembly " +
451  details.assemblyName);
452 
453  auto modelDescriptor = protein->getModelDescriptor();
454 
455  auto &instances = modelDescriptor->getInstances();
456  if (details.instanceIndex >= instances.size())
457  PLUGIN_THROW("Invalid instance index (" +
458  std::to_string(details.instanceIndex) + ") for protein " +
459  details.name + " in assembly " + details.assemblyName);
460 
461  auto instance = modelDescriptor->getInstance(details.instanceIndex);
462  auto &transformation = instance->getTransformation();
463 
464  if (details.position.size() != 3)
465  PLUGIN_THROW("Invalid number of float for position of protein " +
466  details.name + " in assembly " + details.assemblyName);
467  const Vector3f position{details.position[0], details.position[1],
468  details.position[2]};
469 
470  if (details.rotation.size() != 4)
471  PLUGIN_THROW("Invalid number of float for position of protein " +
472  details.name + " in assembly " + details.assemblyName);
473  const Quaterniond rotation{details.rotation[0], details.rotation[1],
474  details.rotation[2], details.rotation[3]};
475 
476  PLUGIN_INFO("Modifying instance "
477  << details.instanceIndex << " of protein " << details.name
478  << " in assembly " << details.assemblyName << " with position="
479  << position << " and rotation=" << rotation);
480  Transformation newTransformation = transformation;
481  newTransformation.setTranslation(position);
482  newTransformation.setRotation(rotation);
483  instance->setTransformation(newTransformation);
484  if (details.instanceIndex == 0)
485  modelDescriptor->setTransformation(newTransformation);
486 
487  _scene.markModified();
488 }
