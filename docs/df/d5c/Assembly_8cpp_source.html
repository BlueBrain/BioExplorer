75 Assembly::Assembly(Scene &scene, const AssemblyDetails &details)
76  : _details(details)
77  , _scene(scene)
78 {
79  const auto size = doublesToVector3d(details.shapeParams);
80  _position = doublesToVector3d(_details.position);
81  _rotation = doublesToQuaterniond(_details.rotation);
82  _clippingPlanes = doublesToVector4ds(details.clippingPlanes);
83 
84  switch (details.shape)
85  {
86  case AssemblyShape::empty_sphere:
87  case AssemblyShape::filled_sphere:
88  {
89  _shape = ShapePtr(new SphereShape(details.shape == AssemblyShape::filled_sphere, _clippingPlanes, size.x));
90  break;
91  }
92  case AssemblyShape::sinusoid:
93  {
94  _shape = ShapePtr(new SinusoidShape(_clippingPlanes, size));
95  break;
96  }
97  case AssemblyShape::cube:
98  {
99  _shape = ShapePtr(new CubeShape(_clippingPlanes, size));
100  break;
101  }
102  case AssemblyShape::fan:
103  {
104  _shape = ShapePtr(new FanShape(_clippingPlanes, size.x));
105  break;
106  }
107  case AssemblyShape::plane:
108  {
109  _shape = ShapePtr(new PlaneShape(_clippingPlanes, Vector2f(size.x, size.z)));
110  break;
111  }
112  case AssemblyShape::mesh:
113 #ifdef USE_ASSIMP
114  {
115  _shape = ShapePtr(new MeshShape(_clippingPlanes, size, _details.shapeMeshContents));
116  break;
117  }
118 #else
119  PLUGIN_THROW("BioExplorer was not compiled with the Assimp library");
120 #endif
121  case AssemblyShape::helix:
122  {
123  _shape = ShapePtr(new HelixShape(_clippingPlanes, size.x, size.y));
124  break;
125  }
126  case AssemblyShape::spherical_cell_diffusion:
127  {
128  _shape = ShapePtr(new SphericalCellDiffusionShape(_clippingPlanes, size.x, size.y, size.z));
129  break;
130  }
131  default:
132  _shape = ShapePtr(new PointShape(_clippingPlanes));
133  break;
134  }
135 
136  PLUGIN_INFO(3, "Adding assembly [" << details.name << "] at position " << _position << ", rotation " << _rotation);
137 }

194 {

196  auto modelDescriptor = protein->getModelDescriptor();
197  const auto animationParams = doublesToMolecularSystemAnimationDetails(details.animationParams);

205  _proteins[details.name] = std::move(protein);

207  _scene.addModel(modelDescriptor);

255 void Assembly::_processInstances(ModelDescriptorPtr md, const std::string &name, const size_t occurrences,
256  const Vector3d &position, const Quaterniond &rotation,
257  const uint64_ts &allowedOccurrences,
258  const MolecularSystemAnimationDetails &molecularSystemAnimationDetails,
259  const double offset, const AssemblyConstraints &constraints)
260 {
261  srand(molecularSystemAnimationDetails.seed);
262 
263  // Shape
264  uint64_t count = 0;
265  for (uint64_t occurrence = 0; occurrence < occurrences; ++occurrence)
266  {
267  try
268  {
269  if (!allowedOccurrences.empty() &&
270  std::find(allowedOccurrences.begin(), allowedOccurrences.end(), occurrence) == allowedOccurrences.end())
271  continue;
272 
273  Transformations transformations;
274 
275  Transformation assemblyTransformation;
276  assemblyTransformation.setTranslation(_position);
277  assemblyTransformation.setRotation(_rotation);
278  transformations.push_back(assemblyTransformation);
279 
280  Transformation shapeTransformation =
281  _shape->getTransformation(occurrence, occurrences, molecularSystemAnimationDetails, offset);
282 
283  transformations.push_back(shapeTransformation);
284 
285  const Transformation finalTransformation = combineTransformations(transformations);
286  const auto &translation = finalTransformation.getTranslation();
287 
288  // Assembly constraints
289  bool addInstance = true;
290  for (const auto &constraint : constraints)
291  {
292  if (constraint.first == AssemblyConstraintType::inside && !constraint.second->isInside(translation))
293  addInstance = false;
294  if (constraint.first == AssemblyConstraintType::outside && constraint.second->isInside(translation))
295  addInstance = false;
296  }
297  if (!addInstance)
298  continue;
299 
300  if (count == 0)
301  md->setTransformation(finalTransformation);
302  const ModelInstance instance(true, false, finalTransformation);
303  md->addInstance(instance);
304  ++count;
305  }
306  catch (const std::runtime_error &)
307  {
308  // Instance is clipped
309  }
310  }
311  md->computeBounds();
312 }

314 void Assembly::setProteinColorScheme(const ProteinColorSchemeDetails &details)
315 {
316  if (details.palette.size() < 3 || details.palette.size() % 3 != 0)
317  PLUGIN_THROW("Invalid palette size");
318 
319  ProteinPtr protein{nullptr};
320  const auto itProtein = _proteins.find(details.name);
321  if (itProtein != _proteins.end())
322  protein = (*itProtein).second;
323  else if (_membrane)
324  {
325  const auto membraneLipids = _membrane->getLipids();
326  const auto it = membraneLipids.find(details.assemblyName + '_' + details.name);
327  if (it != membraneLipids.end())
328  protein = (*it).second;
329  }
330 
331  if (protein)
332  {
333  Palette palette;
334  for (size_t i = 0; i < details.palette.size(); i += 3)
335  palette.push_back({details.palette[i], details.palette[i + 1], details.palette[i + 2]});
336 
337  PLUGIN_INFO(3, "Applying color scheme to protein " << details.name << " on assembly " << details.assemblyName);
338  protein->setColorScheme(details.colorScheme, palette, details.chainIds);
339 
340  _scene.markModified();
341  }
342  else
343  PLUGIN_ERROR("Protein " << details.name << " not found on assembly " << details.assemblyName);
344 }

370 const std::string Assembly::getAminoAcidInformation(const AminoAcidInformationDetails &details) const
371 {
372  PLUGIN_INFO(3, "Returning Amino Acid information from protein " << details.name);
373 
374  std::string response;
375  const auto it = _proteins.find(details.name);
376  if (it != _proteins.end())
377  {
378  // Sequences
379  for (const auto &sequence : (*it).second->getSequencesAsString())
380  {
381  if (!response.empty())
382  response += "\n";
383  response += sequence.second;
384  }
385 
386  // Glycosylation sites
387  const auto &sites = (*it).second->getGlycosylationSites({});
388  for (const auto &site : sites)
389  {
390  std::string s;
391  for (const auto &index : site.second)
392  {
393  if (!s.empty())
394  s += ",";
395  s += std::to_string(index + 1); // Site indices start a 1, not 0
396  }
397  response += "\n" + s;
398  }
399  }
400  else
401  PLUGIN_THROW("Protein not found: " + details.name);
402 
403  return response;
404 }

506 ProteinInspectionDetails Assembly::inspect(const Vector3d &origin, const Vector3d &direction, double &t) const
507 {
508  ProteinInspectionDetails result;
509  result.hit = false;
510  result.assemblyName = _details.name;
511 
512  t = std::numeric_limits<double>::max();
513 
514  // Proteins
515  for (const auto protein : _proteins)
516  {
517  const auto md = protein.second->getModelDescriptor();
518  const auto &instances = md->getInstances();
519  const Vector3d instanceHalfSize = protein.second->getBounds().getSize() / 2.0;
520 
521  uint64_t count = 0;
522  for (const auto &instance : instances)
523  {
524  const auto instancePosition = instance.getTransformation().getTranslation();
525 
526  Boxd box;
527  box.merge(instancePosition - instanceHalfSize);
528  box.merge(instancePosition + instanceHalfSize);
529 
530  double tHit;
531  if (rayBoxIntersection(origin, direction, box, 0.0, t, tHit))
532  {
533  result.hit = true;
534  if (tHit < t)
535  {
536  result.proteinName = protein.second->getDescriptor().name;
537  result.modelId = md->getModelID();
538  result.instanceId = count;
539  result.position = {instancePosition.x, instancePosition.y, instancePosition.z};
540  t = tHit;
541  }
542  }
543  ++count;
544  }
545  }
546 
547  // Membrane
548  if (_membrane)
549  {
550  for (const auto protein : _membrane->getLipids())
551  {
552  const auto md = protein.second->getModelDescriptor();
553  const auto &instances = md->getInstances();
554  const Vector3d instanceHalfSize = protein.second->getBounds().getSize() / 2.0;
555 
556  uint64_t count = 0;
557  for (const auto &instance : instances)
558  {
559  const auto instancePosition = instance.getTransformation().getTranslation();
560 
561  Boxd box;
562  box.merge(instancePosition - instanceHalfSize);
563  box.merge(instancePosition + instanceHalfSize);
564 
565  double tHit;
566  if (rayBoxIntersection(origin, direction, box, 0.0, t, tHit))
567  {
568  result.hit = true;
569  if (tHit < t)
570  {
571  result.proteinName = protein.second->getDescriptor().name;
572  result.modelId = md->getModelID();
573  result.instanceId = count;
574  result.position = {instancePosition.x, instancePosition.y, instancePosition.z};
575  t = tHit;
576  }
577  }
578  ++count;
579  }
580  }
581  }
582 
583  return result;
584 }
