50 Assembly::Assembly(Scene &scene, const AssemblyDetails &details)
51  : _details(details)
52  , _scene(scene)
53 {
54  const auto size = doublesToVector3d(details.shapeParams);
55  _position = doublesToVector3d(_details.position);
56  _rotation = doublesToQuaterniond(_details.rotation);
57  _clippingPlanes = doublesToVector4ds(details.clippingPlanes);
58 
59  switch (details.shape)
60  {
61  case AssemblyShape::sphere:
62  {
63  _shape = ShapePtr(new SphereShape(_clippingPlanes, size.x));
64  break;
65  }
66  case AssemblyShape::sinusoid:
67  {
68  _shape = ShapePtr(new SinusoidShape(_clippingPlanes, size));
69  break;
70  }
71  case AssemblyShape::cube:
72  {
73  _shape = ShapePtr(new CubeShape(_clippingPlanes, size));
74  break;
75  }
76  case AssemblyShape::fan:
77  {
78  _shape = ShapePtr(new FanShape(_clippingPlanes, size.x));
79  break;
80  }
81  case AssemblyShape::plane:
82  {
83  _shape =
84  ShapePtr(new PlaneShape(_clippingPlanes, Vector2f(size.x, size.z)));
85  break;
86  }
87  case AssemblyShape::mesh:
88  {
89  _shape = ShapePtr(
90  new MeshShape(_clippingPlanes, size, _details.shapeMeshContents));
91  break;
92  }
93  default:
94  _shape = ShapePtr(new PointShape(_clippingPlanes));
95  break;
96  }
97 
98  PLUGIN_INFO(3, "Adding assembly [" << details.name << "] at position "
99  << _position << ", rotation "
100  << _rotation);
101 }

133  const auto transmembraneParams =
134  doublesToVector2d(details.transmembraneParams);

136  _processInstances(modelDescriptor, details.name, details.occurrences,
137  proteinPosition, proteinRotation,
138  details.allowedOccurrences, animationParams,

143  _scene.addModel(modelDescriptor);
144  PLUGIN_INFO(3, "Number of instances: "
145  << modelDescriptor->getInstances().size());

197 void Assembly::_processInstances(ModelDescriptorPtr md, const std::string &name,
198  const size_t occurrences,
199  const Vector3d &position,
200  const Quaterniond &rotation,
201  const uint64_ts &allowedOccurrences,
202  const AnimationDetails &animationDetails,
203  const double offset,
204  const AssemblyConstraints &constraints)
205 {
206  srand(animationDetails.seed);
207 
208  // Shape
209  uint64_t count = 0;
210  for (uint64_t occurrence = 0; occurrence < occurrences; ++occurrence)
211  {
212  try
213  {
214  if (!allowedOccurrences.empty() &&
215  std::find(allowedOccurrences.begin(), allowedOccurrences.end(),
216  occurrence) == allowedOccurrences.end())
217  continue;
218 
219  Transformations transformations;
220 
221  Transformation assemblyTransformation;
222  assemblyTransformation.setTranslation(_position);
223  assemblyTransformation.setRotation(_rotation);
224  transformations.push_back(assemblyTransformation);
225 
226  Transformation shapeTransformation =
227  _shape->getTransformation(occurrence, occurrences,
228  animationDetails, offset);
229 
230  transformations.push_back(shapeTransformation);
231 
232  Transformation proteinTransformation;
233  proteinTransformation.setTranslation(position);
234  proteinTransformation.setRotation(rotation);
235  transformations.push_back(proteinTransformation);
236 
237  const Transformation finalTransformation =
238  combineTransformations(transformations);
239  const auto &translation = finalTransformation.getTranslation();
240 
241  // Assembly constaints
242  bool addInstance = true;
243  for (const auto &constraint : constraints)
244  {
245  if (constraint.first == AssemblyConstraintType::inside &&
246  !constraint.second->isInside(translation))
247  addInstance = false;
248  if (constraint.first == AssemblyConstraintType::outside &&
249  constraint.second->isInside(translation))
250  addInstance = false;
251  }
252  if (!addInstance)
253  continue;
254 
255  if (count == 0)
256  md->setTransformation(finalTransformation);
257  const ModelInstance instance(true, false, finalTransformation);
258  md->addInstance(instance);
259  ++count;
260  }
261  catch (const std::runtime_error &)
262  {
263  // Instance is clipped
264  }
265  }
266 }

268 void Assembly::setProteinColorScheme(const ProteinColorSchemeDetails &details)
269 {
270  if (details.palette.size() < 3 || details.palette.size() % 3 != 0)
271  PLUGIN_THROW("Invalid palette size");
272 
273  ProteinPtr protein{nullptr};
274  const auto itProtein = _proteins.find(details.name);
275  if (itProtein != _proteins.end())
276  protein = (*itProtein).second;
277  else if (_membrane)
278  {
279  const auto membraneLipids = _membrane->getLipids();
280  const auto it =
281  membraneLipids.find(details.assemblyName + '_' + details.name);
282  if (it != membraneLipids.end())
283  protein = (*it).second;
284  }
285 
286  if (protein)
287  {
288  Palette palette;
289  for (size_t i = 0; i < details.palette.size(); i += 3)
290  palette.push_back({details.palette[i], details.palette[i + 1],
291  details.palette[i + 2]});
292 
293  PLUGIN_INFO(3, "Applying color scheme to protein "
294  << details.name << " on assembly "
295  << details.assemblyName);
296  protein->setColorScheme(details.colorScheme, palette, details.chainIds);
297 
298  _scene.markModified();
299  }
300  else
301  PLUGIN_ERROR("Protein " << details.name << " not found on assembly "
302  << details.assemblyName);
303 }

331 const std::string Assembly::getAminoAcidInformation(
332  const AminoAcidInformationDetails &details) const
333 {
334  PLUGIN_INFO(3, "Returning Amino Acid information from protein "
335  << details.name);
336 
337  std::string response;
338  const auto it = _proteins.find(details.name);
339  if (it != _proteins.end())
340  {
341  // Sequences
342  for (const auto &sequence : (*it).second->getSequencesAsString())
343  {
344  if (!response.empty())
345  response += "\n";
346  response += sequence.second;
347  }
348 
349  // Glycosylation sites
350  const auto &sites = (*it).second->getGlycosylationSites({});
351  for (const auto &site : sites)
352  {
353  std::string s;
354  for (const auto &index : site.second)
355  {
356  if (!s.empty())
357  s += ",";
358  s += std::to_string(index + 1); // Site indices start a 1, not 0
359  }
360  response += "\n" + s;
361  }
362  }
363  else
364  PLUGIN_THROW("Protein not found: " + details.name);
365 
366  return response;
367 }

479 ProteinInspectionDetails Assembly::inspect(const Vector3d &origin,
480  const Vector3d &direction,
481  double &t) const
482 {
483  ProteinInspectionDetails result;
484  result.hit = false;
485  result.assemblyName = _details.name;
486 
487  t = std::numeric_limits<double>::max();
488 
489  // Proteins
490  for (const auto protein : _proteins)
491  {
492  const auto md = protein.second->getModelDescriptor();
493  const auto &instances = md->getInstances();
494  const Vector3d instanceHalfSize =
495  protein.second->getBounds().getSize() / 2.0;
496 
497  uint64_t count = 0;
498  for (const auto &instance : instances)
499  {
500  const auto instancePosition =
501  instance.getTransformation().getTranslation();
502 
503  Boxd box;
504  box.merge(instancePosition - instanceHalfSize);
505  box.merge(instancePosition + instanceHalfSize);
506 
507  double tHit;
508  if (rayBoxIntersection(origin, direction, box, 0.0, t, tHit))
509  {
510  result.hit = true;
511  if (tHit < t)
512  {
513  result.proteinName = protein.second->getDescriptor().name;
514  result.modelId = md->getModelID();
515  result.instanceId = count;
516  result.position = {instancePosition.x, instancePosition.y,
517  instancePosition.z};
518  t = tHit;
519  }
520  }
521  ++count;
522  }
523  }
524 
525  // Membrane
526  if (_membrane)
527  {
528  for (const auto protein : _membrane->getLipids())
529  {
530  const auto md = protein.second->getModelDescriptor();
531  const auto &instances = md->getInstances();
532  const Vector3d instanceHalfSize =
533  protein.second->getBounds().getSize() / 2.0;
534 
535  uint64_t count = 0;
536  for (const auto &instance : instances)
537  {
538  const auto instancePosition =
539  instance.getTransformation().getTranslation();
540 
541  Boxd box;
542  box.merge(instancePosition - instanceHalfSize);
543  box.merge(instancePosition + instanceHalfSize);
544 
545  double tHit;
546  if (rayBoxIntersection(origin, direction, box, 0.0, t, tHit))
547  {
548  result.hit = true;
549  if (tHit < t)
550  {
551  result.proteinName =
552  protein.second->getDescriptor().name;
553  result.modelId = md->getModelID();
554  result.instanceId = count;
555  result.position = {instancePosition.x,
556  instancePosition.y,
557  instancePosition.z};
558  t = tHit;
559  }
560  }
561  ++count;
562  }
563  }
564  }
565 
566  return result;
567 }
