32 Assembly::Assembly(Scene &scene, const AssemblyDescriptor &ad)
33  : _descriptor(ad)
34  , _scene(scene)
35 {
36  if (ad.position.size() != 3)
37  PLUGIN_THROW(std::runtime_error(
38  "Position must be a sequence of 3 float values"));
39  _position = {ad.position[0], ad.position[1], ad.position[2]};
40 
41  if (ad.clippingPlanes.size() % 4 != 0)
42  PLUGIN_THROW(std::runtime_error(
43  "Clipping planes must be defined by 4 float values"));
44  const auto &cp = ad.clippingPlanes;
45  for (size_t i = 0; i < cp.size(); i += 4)
46  _clippingPlanes.push_back({cp[i], cp[i + 1], cp[i + 2], cp[i + 3]});
47 
48  PLUGIN_INFO << "Add assembly " << ad.name << " at position " << _position
49  << std::endl;
50 }

161 void Assembly::_processInstances(
162  ModelDescriptorPtr md, const std::string &name, const AssemblyShape shape,
163  const floats &assemblyParams, const size_t occurrences,
164  const Vector3f &position, const Quaterniond &orientation,
165  const size_ts &allowedOccurrences, const size_t randomSeed,
166  const PositionRandomizationType &randomizationType,
167  const float locationCutoffAngle)
168 {
169  const float offset = 2.f / occurrences;
170  const float increment = M_PI * (3.f - sqrt(5.f));
171 
172  srand(randomSeed);
173  size_t rnd{1};
174  if (occurrences != 0 && randomSeed != 0 &&
175  randomizationType == PositionRandomizationType::circular)
176  rnd = rand() % occurrences;
177 
178 #ifdef DEBUG
179  model.addCone(0, {{0, 0, 0}, {0, 0, 1}, 0.1f, 0.f});
180 #endif
181 
182  uint64_t count = 0;
183  for (uint64_t i = 0; i < occurrences; ++i)
184  {
185  if (!allowedOccurrences.empty() &&
186  std::find(allowedOccurrences.begin(), allowedOccurrences.end(),
187  i) == allowedOccurrences.end())
188  continue;
189 
190  Vector3f pos;
191  Vector3f dir;
192  switch (shape)
193  {
194  case AssemblyShape::spherical:
195  {
196  getSphericalPosition(rnd, assemblyParams[0], assemblyParams[1],
197  randomizationType, randomSeed, i, occurrences,
198  position, pos, dir);
199  break;
200  }
201  case AssemblyShape::sinusoidal:
202  {
203  const auto assemblySize = assemblyParams[0];
204  const auto assemblyHeight = assemblyParams[1];
205  getSinosoidalPosition(assemblySize, assemblyHeight,
206  randomizationType, randomSeed, position, pos,
207  dir);
208  break;
209  }
210  case AssemblyShape::cubic:
211  {
212  const auto assemblySize = assemblyParams[0];
213  getCubicPosition(assemblySize, position, pos, dir);
214  break;
215  }
216  case AssemblyShape::fan:
217  {
218  const auto assemblyRadius = assemblyParams[0];
219  getFanPosition(rnd, assemblyRadius, randomizationType, randomSeed,
220  i, occurrences, position, pos, dir);
221  break;
222  }
223  case AssemblyShape::bezier:
224  {
225  const Vector3fs points = {
226  {1, 391, 0}, {25, 411, 0}, {48, 446, 0}, {58, 468, 0},
227  {70, 495, 0}, {83, 523, 0}, {110, 535, 0}, {157, 517, 0},
228  {181, 506, 0}, {214, 501, 0}, {216, 473, 0}, {204, 456, 0},
229  {223, 411, 0}, {241, 382, 0}, {261, 372, 0}, {297, 402, 0},
230  {308, 433, 0}, {327, 454, 0}, {355, 454, 0}, {389, 446, 0},
231  {406, 433, 0}, {431, 426, 0}, {458, 443, 0}, {478, 466, 0},
232  {518, 463, 0}, {559, 464, 0}, {584, 478, 0}, {582, 503, 0},
233  {550, 533, 0}, {540, 550, 0}, {540, 574, 0}, {560, 572, 0},
234  {599, 575, 0}, {629, 550, 0}, {666, 548, 0}, {696, 548, 0},
235  {701, 582, 0}, {701, 614, 0}, {683, 639, 0}, {653, 647, 0},
236  {632, 651, 0}, {597, 666, 0}, {570, 701, 0}, {564, 731, 0},
237  {559, 770, 0}, {565, 799, 0}, {577, 819, 0}, {611, 820, 0},
238  {661, 809, 0}, {683, 787, 0}, {700, 768, 0}, {735, 758, 0},
239  {763, 768, 0}, {788, 792, 0}, {780, 820, 0}, {770, 859, 0},
240  {740, 882, 0}, {705, 911, 0}, {688, 931, 0}, {646, 973, 0},
241  {611, 992, 0}, {585, 1022, 0}};
242  const auto assemblySize = assemblyParams[0];
243  getBezierPosition(points, assemblySize,
244  float(i) / float(occurrences), pos, dir);
245  break;
246  }
247  default:
248  const auto assemblySize = assemblyParams[0];
249  getPlanarPosition(assemblySize, randomizationType, randomSeed,
250  position, pos, dir);
251  break;
252  }
253 
254  // Clipping planes
255  if (isClipped(pos, _clippingPlanes))
256  continue;
257 
258  // Remove membrane where proteins are. This is currently done according
259  // to the vector orientation
260  bool occupied{false};
261  if (locationCutoffAngle != 0.f)
262  for (const auto &occupiedDirection : _occupiedDirections)
263  if (dot(dir, occupiedDirection.first) >
264  occupiedDirection.second)
265  {
266  occupied = true;
267  break;
268  }
269  if (occupied)
270  continue;
271 
272  const Quaterniond instanceOrientation = glm::quatLookAt(dir, UP_VECTOR);
273 
274  Transformation tf;
275  tf.setTranslation(_position + pos);
276  tf.setRotation(instanceOrientation * orientation);
277 
278  if (count == 0)
279  md->setTransformation(tf);
280  const ModelInstance instance(true, false, tf);
281  md->addInstance(instance);
282 
283  // Store occupied direction
284  if (locationCutoffAngle != 0.f)
285  _occupiedDirections.push_back({dir, locationCutoffAngle});
286 
287  ++count;
288  }
289 }

291 void Assembly::setColorScheme(const ColorSchemeDescriptor &csd)
292 {
293  if (csd.palette.size() < 3 || csd.palette.size() % 3 != 0)
294  PLUGIN_THROW(std::runtime_error("Invalid palette size"));
295 
296  ProteinPtr protein{nullptr};
297  auto itProtein = _proteins.find(csd.name);
298  if (itProtein != _proteins.end())
299  protein = (*itProtein).second;
300  else
301  {
302  auto itMesh = _meshBasedMembranes.find(csd.name);
303  if (itMesh != _meshBasedMembranes.end())
304  protein = (*itMesh).second->getProtein();
305  else if (_membrane)
306  {
307  const auto membraneProteins = _membrane->getProteins();
308  auto it = membraneProteins.find(csd.name);
309  if (it != membraneProteins.end())
310  protein = (*it).second;
311  }
312  }
313 
314  if (protein)
315  {
316  Palette palette;
317  for (size_t i = 0; i < csd.palette.size(); i += 3)
318  palette.push_back(
319  {csd.palette[i], csd.palette[i + 1], csd.palette[i + 2]});
320 
321  PLUGIN_INFO << "Applying color scheme to protein " << csd.name
322  << " on assembly " << csd.assemblyName << std::endl;
323  protein->setColorScheme(csd.colorScheme, palette, csd.chainIds);
324 
325  _scene.markModified();
326  }
327  else
328  PLUGIN_ERROR << "Protein " << csd.name << " not found on assembly "
329  << csd.assemblyName << std::endl;
330 }

358 std::string Assembly::getAminoAcidInformation(
359  const AminoAcidInformationDescriptor &aasd) const
360 {
361  PLUGIN_INFO << "Returning Amino Acid information from protein " << aasd.name
362  << std::endl;
363 
364  std::string response;
365  auto it = _proteins.find(aasd.name);
366  if (it != _proteins.end())
367  {
368  // Sequences
369  for (const auto &sequence : (*it).second->getSequencesAsString())
370  {
371  if (!response.empty())
372  response += "\n";
373  response += sequence.second;
374  }
375 
376  // Glycosylation sites
377  const auto &sites = (*it).second->getGlycosylationSites({});
378  for (const auto &site : sites)
379  {
380  std::string s;
381  for (const auto &index : site.second)
382  {
383  if (!s.empty())
384  s += ",";
385  s += std::to_string(index + 1); // Site indices start a 1, not 0
386  }
387  response += "\n" + s;
388  }
389  }
390  else
391  PLUGIN_THROW(std::runtime_error("Protein not found: " + aasd.name));
392 
393  return response;
394 }
