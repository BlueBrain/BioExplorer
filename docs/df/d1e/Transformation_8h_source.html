67  {
68  Matrix4d matrix;
69  matrix = matrix * glm::translate(_translation);
70  matrix = matrix * glm::translate(_rotationCenter);
71  matrix = matrix * glm::toMat4(_rotation);
72  matrix = matrix * glm::translate(-1.0 * _rotationCenter);
73  if (withScale)
