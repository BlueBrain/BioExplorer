58 double Synapses::_getDisplacementValue(const DisplacementElement& element)
59 {
60  const auto params = _details.displacementParams;
61  switch (element)
62  {
63  case DisplacementElement::morphology_spine_strength:

72 void Synapses::_buildModel()
73 {
74  if (_modelDescriptor)
75  _scene.removeModel(_modelDescriptor->getModelID());
76 
77  auto model = _scene.createModel();
78  ThreadSafeContainer container(*model, _alignToGrid, _position, _rotation);
79 
80  const auto synapses = DBConnector::getInstance().getSynapses(_details.populationName, _details.sqlFilter);
81 
82  const auto nbSynapses = synapses.size();
83  const size_t materialId = 0;
84  uint64_t i = 0;
85  uint64_t progressStep = synapses.size() / 100 + 1;
86  for (const auto& synapse : synapses)
87  {
88  if (_details.representation == SynapseRepresentation::spine)
89  _addSpine(container, synapse.first, synapse.second, materialId);
90  else
91  container.addSphere(synapse.second.preSynapticSurfacePosition,
92  DEFAULT_SPINE_RADIUS * _details.radiusMultiplier, materialId, false, i);
93  if (i % progressStep == 0)
94  PLUGIN_PROGRESS("Loading " << i << "/" << nbSynapses << " synapses", i, nbSynapses);
95  ++i;
96  }
97 
98  container.commitToModel();
99  PLUGIN_INFO(1, "");
100 
101  const ModelMetadata metadata = {{"Number of synapses", std::to_string(nbSynapses)},
102  {"SQL filter", _details.sqlFilter}};
103 
104  _modelDescriptor.reset(new core::ModelDescriptor(std::move(model), _details.assemblyName, metadata));
105  if (_modelDescriptor)
106  _scene.addModel(_modelDescriptor);
107  else
108  PLUGIN_THROW(
109  "Synapse efficacy model could not be created for "
110  "population " +
111  _details.populationName);
112 }

114 void Synapses::_addSpine(ThreadSafeContainer& container, const uint64_t guid, const Synapse& synapse,
115  const size_t SpineMaterialId)
116 {
117  const double radius = _details.radiusMultiplier * DEFAULT_SPINE_RADIUS;
118 
119  const auto spineSmallRadius = radius * spineRadiusRatio * 0.5;
120  const auto spineBaseRadius = radius * spineRadiusRatio * 0.75;
121  const auto spineLargeRadius = radius * spineRadiusRatio * 2.5;
122 
123  const auto direction =
124  Vector3d((rand() % 200 - 100) / 100.0, (rand() % 200 - 100) / 100.0, (rand() % 200 - 100) / 100.0);
125  const auto l = 6.f * radius;
126 
127  const auto origin = synapse.postSynapticSurfacePosition;
128  const auto target = synapse.preSynapticSurfacePosition;
129 
130  // Create random shape between origin and target
131  auto middle = (target + origin) / 2.0;
132  const double d = length(target - origin) / 1.5;
133  const auto id = guid * 4;
134  middle += Vector3f(d * rnd2(id), d * rnd2(id + 1), d * rnd2(id + 2));
135  const float spineMiddleRadius = spineSmallRadius + d * 0.1 * rnd2(id + 3);
136 
137  const auto displacement = Vector3f(_getDisplacementValue(DisplacementElement::morphology_spine_strength),
138  _getDisplacementValue(DisplacementElement::morphology_spine_frequency), 0.f);
139 
140  const bool useSdf =
141  andCheck(static_cast<uint32_t>(_details.realismLevel), static_cast<uint32_t>(MorphologyRealismLevel::spine));
142  Neighbours neighbours;
143  neighbours.insert(container.addSphere(middle, spineMiddleRadius, SpineMaterialId, useSdf, NO_USER_DATA, neighbours,
144  displacement));
145  if (middle != origin)
146  container.addCone(origin, spineSmallRadius, middle, spineMiddleRadius, SpineMaterialId, useSdf, NO_USER_DATA,
147  neighbours, displacement);
148  if (middle != target)
149  container.addCone(middle, spineMiddleRadius, target, spineLargeRadius, SpineMaterialId, useSdf, NO_USER_DATA,
150  neighbours, displacement);
151 }
