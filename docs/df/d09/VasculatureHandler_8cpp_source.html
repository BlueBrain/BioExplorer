32 VasculatureHandler::VasculatureHandler(const VasculatureReportDetails& details)
33  : brayns::AbstractSimulationHandler()
34  , _details(details)
35 {
36  auto& connector = DBConnector::getInstance();
37  _simulationReport =
38  connector.getVasculatureSimulationReport(_details.populationName,
39  _details.simulationReportId);
40  const auto endTime = _simulationReport.endTime;
41  _dt = _simulationReport.timeStep;
42  _unit = _simulationReport.timeUnits;
43  _nbFrames = (_simulationReport.endTime - _simulationReport.startTime) /
44  _simulationReport.timeStep;
45  _frameData = connector.getVasculatureSimulationTimeSeries(
46  _details.populationName, _details.simulationReportId, 0);
47  _frameSize = _frameData.size();
48  PLUGIN_INFO(1, "Report successfully attached");
49  PLUGIN_INFO(1, "- Frame size : " << _frameSize);
50  PLUGIN_INFO(1, "- Number of frames: " << _nbFrames);
51  PLUGIN_INFO(1, "- Start time : " << _simulationReport.startTime);
52  PLUGIN_INFO(1, "- End time : " << _simulationReport.endTime);
53  PLUGIN_INFO(1, "- Time interval : " << _simulationReport.timeStep);
54  PLUGIN_INFO(1, "- Time units : " << _simulationReport.timeUnits);
55 }

57 void* VasculatureHandler::getFrameData(const uint32_t frame)
58 {
59  try
60  {
61  if (_currentFrame != frame && frame < _nbFrames)
62  _frameData =
63  DBConnector::getInstance().getVasculatureSimulationTimeSeries(
64  _details.populationName, _details.simulationReportId,
65  frame);
66  }
67  catch (const std::runtime_error& e)
68  {
69  PLUGIN_ERROR(e.what())
70  }
71  _currentFrame = frame;
72  return _frameData.data();
73 }
