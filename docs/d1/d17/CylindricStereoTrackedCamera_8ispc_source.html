36 void CylindricStereoTrackedCamera_initRay(uniform Camera* uniform _self, varying Ray& ray,
37  const varying CameraSample& sample)
38 {
39  uniform CylindricStereoTrackedCamera* uniform self = (uniform CylindricStereoTrackedCamera * uniform) _self;
40 
41  const vec2f screen = Camera_subRegion(_self, sample.screen);
42 
43  float eyeDelta = 0.0f;
44  float alpha = 0.0f;
45  if (self->bufferId == LEFT_WALL)
46  {
47  eyeDelta = -self->half_ipd;
48  alpha = screen.x * pi;
49  }
50  else if (self->bufferId == RIGHT_WALL)
51  {
52  eyeDelta = self->half_ipd;
53  alpha = screen.x * pi;
54  }
55  else if (self->bufferId == LEFT_FLOOR)
56  {
57  eyeDelta = -self->half_ipd;
58  }
59  else if (self->bufferId == RIGHT_FLOOR)
60  {
61  eyeDelta = self->half_ipd;
62  }
63 
64  vec3f pixelPos;
65  if (self->bufferId == LEFT_WALL || self->bufferId == RIGHT_WALL)
66  {
67  pixelPos.x = self->opendeck_radius * -cosf(alpha);
68  // pixelPos.y = OPENDECK_METALSTRIPE_HEIGHT + OPENDECK_HEIGHT *
69  // screen.y;
70  pixelPos.y = self->opendeck_metal_stripe_height + self->opendeck_height - self->opendeck_height * screen.y;
71  pixelPos.z = self->opendeck_radius * -sinf(alpha);
72  }
73  else if (self->bufferId == LEFT_FLOOR || self->bufferId == RIGHT_FLOOR)
74  {
75  pixelPos.x = 2.0f * self->opendeck_radius * (screen.x - 0.5f);
76  pixelPos.y = 0.0f;
77  pixelPos.z = -self->opendeck_radius + self->opendeck_radius * screen.y;
78  // pixelPos.z = -OPENDECK_RADIUS * screen.y;
79  }
80 
81  const vec3f eyeAxis = self->od_dir_du;
82  const vec3f eyeDeltaPos = eyeAxis * eyeDelta;
83 
84  const LinearSpace3f cameraSpace = make_LinearSpace3f(self->dir_du, self->dir_dv, self->dir_cam);
85 
86  vec3f dir = normalize(pixelPos - self->headPos + eyeDeltaPos);
87  dir = normalize(cameraSpace * dir);
88 
89  const vec3f org = self->org + self->headPos - eyeDeltaPos;
90  const float time = Camera_shutterTime(_self, sample.time);
91  setRay(ray, org, normalize(dir), self->super.nearClip, inf, time);
92 }

94 /*! create a new ::ispc-side version of a OpenDeckCylinderStereoCamera - with
95  given pointer to the c-class version - and return both class pointer and
96  pointer to internal data back via the supplied reference parameters. */
97 export void* uniform CylindricStereoTrackedCamera_create(void* uniform cppE)
98 {
99  uniform CylindricStereoTrackedCamera* uniform self = uniform new uniform CylindricStereoTrackedCamera;
100  self->super.cppEquivalent = cppE;
101  self->super.initRay = CylindricStereoTrackedCamera_initRay;
102  self->super.doesDOF = false;
103  return self;
104 }

106 export void CylindricStereoTrackedCamera_set(void* uniform _self, const uniform vec3f& org,
107  const uniform vec3f& headPos, const uniform vec3f& dir_cam,
108  const uniform vec3f& dir_du, const uniform vec3f& dir_dv,
109  const uniform vec3f& od_dir_du, const uniform float interpupillaryDistance,
110  const uniform uint8 bufferId, const uniform float cameraScaling)
111 {

113  self->org = org;
114  self->headPos = headPos * cameraScaling;
115  self->dir_cam = dir_cam;
116  self->dir_du = dir_du;
117  self->dir_dv = dir_dv;
118  self->od_dir_du = od_dir_du;
119  self->half_ipd = interpupillaryDistance * cameraScaling * 0.5f;
120  self->bufferId = bufferId;
121  self->opendeck_radius = OPENDECK_RADIUS * cameraScaling;
122  self->opendeck_height = OPENDECK_HEIGHT * cameraScaling;

124 }
