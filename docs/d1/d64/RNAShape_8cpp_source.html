36 RNAShape::RNAShape(const Vector4ds& clippingPlanes, const RNAShapeType& shapeType, const uint64_t nbElements,
37  const Vector2f& shapeParams, const Vector2f& valuesRange, const Vector3d& curveParams)
38  : Shape(clippingPlanes)
39  , _shapeType(shapeType)
40  , _shapeParams(shapeParams)
41  , _valuesRange(valuesRange)
42  , _curveParams(curveParams)
43 {
44  _bounds.merge(Vector3d(-curveParams.x / 2.0, -curveParams.y / 2.0, -curveParams.z / 2.0));
45  _bounds.merge(Vector3d(curveParams.x / 2.0, curveParams.y / 2.0, curveParams.z / 2.0));
46 
47  _U = Vector3d(valuesRange.x, valuesRange.y, nbElements);
48  _V = Vector3d(valuesRange.x, valuesRange.y, nbElements);
49 
50  switch (_shapeType)
51  {
52  case RNAShapeType::moebius:
53  _U = {2.0 * M_PI, 4.0 * M_PI, nbElements};
54  _V = {-0.4, 0.4, 1.0};
55  break;
56  case RNAShapeType::heart:
57  _U = {0.0, 2.0 * M_PI, nbElements};
58  _V = {0.0, 1.0, 1.0};
59  break;
60  default:
61  break;
62  }
63 
64  _step = (_U.y - _U.x) / _U.z;
65 }

67 Transformation RNAShape::getTransformation(const uint64_t occurrence, const uint64_t nbOccurrences,
68  const MolecularSystemAnimationDetails& MolecularSystemAnimationDetails,
69  const double offset) const
70 {
71  const double u = _valuesRange.x + _step * occurrence;
72  const double v = _valuesRange.x + _step * (occurrence + 1);
73 
74  Vector3d src, dst;
75  _getSegment(u, v, src, dst);
76 
77  const Vector3d direction = normalize(dst - src);
78  const Vector3d normal = cross(UP_VECTOR, direction);
79  double upOffset = 0.0;
80  if (MolecularSystemAnimationDetails.positionSeed != 0)
81  upOffset = MolecularSystemAnimationDetails.positionStrength *
82  rnd3(MolecularSystemAnimationDetails.positionSeed + occurrence);
83 
84  Vector3d pos = src + normal * (offset + upOffset);
85 
86  if (isClipped(pos, _clippingPlanes))
87  throw std::runtime_error("Instance is clipped");
88 
89  Quaterniond rot = safeQuatlookAt(normal);
90  if (MolecularSystemAnimationDetails.rotationSeed != 0)
91  rot = weightedRandomRotation(rot, MolecularSystemAnimationDetails.rotationSeed, occurrence,
92  MolecularSystemAnimationDetails.rotationStrength);
93 
94  Transformation transformation;
95  transformation.setTranslation(pos);
96  transformation.setRotation(rot);
97  return transformation;
98 }

105 void RNAShape::_getSegment(const double u, const double v, Vector3d& src, Vector3d& dst) const
106 {
107  switch (_shapeType)
108  {
109  case RNAShapeType::moebius:
110  {
111  src = _moebius(u, v);
112  dst = _moebius(u + _step, v);
113  break;
114  }
115  case RNAShapeType::torus:
116  {
117  src = _torus(u);
118  dst = _torus(u + _step);
119  break;
120  }
121  case RNAShapeType::star:
122  {
123  src = _star(u);
124  dst = _star(u + _step);
125  break;
126  }
127  case RNAShapeType::spring:
128  {
129  src = _spring(u);
130  dst = _spring(u + _step);
131  break;
132  }
133  case RNAShapeType::trefoilKnot:
134  {
135  src = _trefoilKnot(u);
136  dst = _trefoilKnot(u + _step);
137  break;
138  }
139  case RNAShapeType::heart:
140  {
141  src = _heart(u);
142  dst = _heart(u + _step);
143  break;
144  }
145  case RNAShapeType::thing:
146  {
147  src = _thing(u);
148  dst = _thing(u + _step);
149  break;
150  }
151  default:
152  PLUGIN_THROW("Undefined shape");
153  break;
154  }
155 }

157 Vector3d RNAShape::_trefoilKnot(const double t) const
158 {
159  return {_shapeParams.x * ((sin(t) + 2.0 * sin(_curveParams.x * t))) / 3.0,
160  _shapeParams.x * ((cos(t) - 2.0 * cos(_curveParams.y * t))) / 3.0,
161  _shapeParams.x * (-sin(_curveParams.z * t))};
162 }

164 Vector3d RNAShape::_torus(const double t) const
165 {
166  return {_shapeParams.x * (cos(t) + _curveParams.x * cos(_curveParams.y * t) * cos(t)),
167  _shapeParams.x * (sin(t) + _curveParams.x * cos(_curveParams.y * t) * sin(t)),
168  _shapeParams.x * _curveParams.x * sin(_curveParams.y * t)};
169 }

171 Vector3d RNAShape::_star(const double t) const
172 {
173  return {_shapeParams.x * (2.0 * sin(3.0 * t) * cos(t)), _shapeParams.x * (2.0 * sin(3.0 * t) * sin(t)),
174  _shapeParams.x * sin(3.0 * t)};
175 }
176 
177 Vector3d RNAShape::_spring(const double t) const
178 {
179  return {_shapeParams.x * cos(t) + (+_curveParams.x * cos(_curveParams.y * t)) * cos(t),
180  _shapeParams.x * sin(t) + (+_curveParams.x * cos(_curveParams.y * t)) * sin(t),
181  _shapeParams.x * t + _curveParams.x * sin(_curveParams.y * t)};
182 }

184 Vector3d RNAShape::_heart(const double u) const
185 {
186  return {_shapeParams.x * 4.0 * pow(sin(u), 3.0),
187  _shapeParams.x * 0.25 * (13.0 * cos(u) - 5.0 * cos(2.0 * u) - 2.0 * cos(3.0 * u) - cos(4.0 * u)), 0.0};
188 }
189 
190 Vector3d RNAShape::_thing(const double t) const
191 {
192  return {_shapeParams.x * (sin(t) + _curveParams.x * sin(_curveParams.y * t)),
193  _shapeParams.x * (cos(t) - _curveParams.x * cos(_curveParams.y * t)),
194  _shapeParams.x * (-sin(_curveParams.z * t))};
195 }

197 Vector3d RNAShape::_moebius(const double u, const double v) const
198 {
199  return {4.0 * _shapeParams.x * (cos(u) + v * cos(u / 2.0) * cos(u)),
200  4.0 * _shapeParams.x * (sin(u) + v * cos(u / 2.0) * sin(u)), 8.0 * _shapeParams.x * (v * sin(u / 2.0))};
201 }
