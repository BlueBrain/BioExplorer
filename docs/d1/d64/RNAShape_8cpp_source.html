33 RNAShape::RNAShape(const Vector4ds& clippingPlanes,
34  const RNAShapeType& shapeType, const uint64_t nbElements,
35  const Vector2f& shapeParams, const Vector2f& valuesRange,
36  const Vector3d& curveParams)
37  : Shape(clippingPlanes)
38  , _shapeType(shapeType)
39  , _shapeParams(shapeParams)
40  , _valuesRange(valuesRange)
41  , _curveParams(curveParams)
42 {
43  _bounds.merge(Vector3d(-curveParams.x / 2.0, -curveParams.y / 2.0,
44  -curveParams.z / 2.0));
45  _bounds.merge(Vector3d(curveParams.x / 2.0, curveParams.y / 2.0,
46  curveParams.z / 2.0));
47 
48  _U = Vector3d(valuesRange.x, valuesRange.y, nbElements);
49  _V = Vector3d(valuesRange.x, valuesRange.y, nbElements);
50 
51  switch (_shapeType)
52  {
53  case RNAShapeType::moebius:
54  _U = {2.0 * M_PI, 4.0 * M_PI, nbElements};
55  _V = {-0.4, 0.4, 1.0};
56  break;
57  case RNAShapeType::heart:
58  _U = {0.0, 2.0 * M_PI, nbElements};
59  _V = {0.0, 1.0, 1.0};
60  break;
61  default:
62  break;
63  }
64 
65  _step = (_U.y - _U.x) / _U.z;
66 }

68 Transformation RNAShape::getTransformation(
69  const uint64_t occurrence, const uint64_t nbOccurrences,
70  const AnimationDetails& animationDetails, const double offset) const
71 {
72  const double u = _valuesRange.x + _step * occurrence;
73  const double v = _valuesRange.x + _step * (occurrence + 1);
74 
75  Vector3d src, dst;
76  _getSegment(u, v, src, dst);
77 
78  const Vector3d direction = normalize(dst - src);
79  const Vector3d normal = cross(UP_VECTOR, direction);
80  double upOffset = 0.0;
81  if (animationDetails.positionSeed != 0)
82  upOffset = animationDetails.positionStrength *
83  rnd3(animationDetails.positionSeed + occurrence);
84 
85  Vector3d pos = src + normal * (offset + upOffset);
86 
87  if (isClipped(pos, _clippingPlanes))
88  throw std::runtime_error("Instance is clipped");
89 
90  Quaterniond rot = safeQuatlookAt(normal);
91  if (animationDetails.rotationSeed != 0)
92  rot = weightedRandomRotation(rot, animationDetails.rotationSeed,
93  occurrence,
94  animationDetails.rotationStrength);
95 
96  Transformation transformation;
97  transformation.setTranslation(pos);
98  transformation.setRotation(rot);
99  return transformation;
100 }

107 void RNAShape::_getSegment(const double u, const double v, Vector3d& src,
108  Vector3d& dst) const
109 {
110  switch (_shapeType)
111  {
112  case RNAShapeType::moebius:
113  {
114  src = _moebius(u, v);
115  dst = _moebius(u + _step, v);
116  break;
117  }
118  case RNAShapeType::torus:
119  {
120  src = _torus(u);
121  dst = _torus(u + _step);
122  break;
123  }
124  case RNAShapeType::star:
125  {
126  src = _star(u);
127  dst = _star(u + _step);
128  break;
129  }
130  case RNAShapeType::spring:
131  {
132  src = _spring(u);
133  dst = _spring(u + _step);
134  break;
135  }
136  case RNAShapeType::trefoilKnot:
137  {
138  src = _trefoilKnot(u);
139  dst = _trefoilKnot(u + _step);
140  break;
141  }
142  case RNAShapeType::heart:
143  {
144  src = _heart(u);
145  dst = _heart(u + _step);
146  break;
147  }
148  case RNAShapeType::thing:
149  {
150  src = _thing(u);
151  dst = _thing(u + _step);
152  break;
153  }
154  default:
155  PLUGIN_THROW("Undefined shape");
156  break;
157  }
158 }

160 Vector3d RNAShape::_trefoilKnot(const double t) const
161 {
162  return {_shapeParams.x * ((sin(t) + 2.0 * sin(_curveParams.x * t))) / 3.0,
163  _shapeParams.x * ((cos(t) - 2.0 * cos(_curveParams.y * t))) / 3.0,
164  _shapeParams.x * (-sin(_curveParams.z * t))};
165 }

167 Vector3d RNAShape::_torus(const double t) const
168 {
169  return {_shapeParams.x *
170  (cos(t) + _curveParams.x * cos(_curveParams.y * t) * cos(t)),
171  _shapeParams.x *
172  (sin(t) + _curveParams.x * cos(_curveParams.y * t) * sin(t)),
173  _shapeParams.x * _curveParams.x * sin(_curveParams.y * t)};
174 }

176 Vector3d RNAShape::_star(const double t) const
177 {
178  return {_shapeParams.x * (2.0 * sin(3.0 * t) * cos(t)),
179  _shapeParams.x * (2.0 * sin(3.0 * t) * sin(t)),
180  _shapeParams.x * sin(3.0 * t)};
181 }
182 
183 Vector3d RNAShape::_spring(const double t) const
184 {
185  return {_shapeParams.x * cos(t) +
186  (+_curveParams.x * cos(_curveParams.y * t)) * cos(t),
187  _shapeParams.x * sin(t) +
188  (+_curveParams.x * cos(_curveParams.y * t)) * sin(t),
189  _shapeParams.x * t + _curveParams.x * sin(_curveParams.y * t)};
190 }
191 
192 Vector3d RNAShape::_heart(const double u) const
193 {
194  return {_shapeParams.x * 4.0 * pow(sin(u), 3.0),
195  _shapeParams.x * 0.25 *
196  (13.0 * cos(u) - 5.0 * cos(2.0 * u) - 2.0 * cos(3.0 * u) -
197  cos(4.0 * u)),
198  0.0};
199 }
200 
201 Vector3d RNAShape::_thing(const double t) const
202 {
203  return {_shapeParams.x *
204  (sin(t) + _curveParams.x * sin(_curveParams.y * t)),
205  _shapeParams.x *
206  (cos(t) - _curveParams.x * cos(_curveParams.y * t)),
207  _shapeParams.x * (-sin(_curveParams.z * t))};
208 }
209 
210 Vector3d RNAShape::_moebius(const double u, const double v) const
211 {
212  return {4.0 * _shapeParams.x * (cos(u) + v * cos(u / 2.0) * cos(u)),
213  4.0 * _shapeParams.x * (sin(u) + v * cos(u / 2.0) * sin(u)),
214  8.0 * _shapeParams.x * (v * sin(u / 2.0))};
215 }
