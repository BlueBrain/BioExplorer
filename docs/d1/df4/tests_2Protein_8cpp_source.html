36std::string getFileContents(const std::string& filename)
37{
38 std::ifstream file(filename);
39 std::string str;
40 if (file)
41 {
42 std::ostringstream ss;
43 ss << file.rdbuf();
44 str = ss.str();
45 }
46 else
47 PLUGIN_THROW("Failed to open " + filename);
48 return str;
49}

76BOOST_AUTO_TEST_CASE(protein)
77{
78 std::vector<const char*> argv{"brayns", "--http-server", "localhost:0",
79 "--plugin", "BioExplorer"};
80 brayns::Brayns brayns(argv.size(), argv.data());
81 auto& scene = brayns.getEngine().getScene();
82 Protein protein(scene, getProteinDescriptor());
83
84 BOOST_CHECK(protein.getAtoms().size() == 21776);
85 BOOST_CHECK(protein.getResidues().size() == 20);
86 BOOST_CHECK(protein.getResidueSequences().size() == 4);
87
88 std::vector<Vector3f> positions;
89 std::vector<Quaterniond> rotations;
90 const std::vector<size_t> siteIndices = {};
91 protein.getGlycosilationSites(positions, rotations, siteIndices);
92 BOOST_CHECK(positions.size() == 24);
93 BOOST_CHECK(rotations.size() == positions.size());
94
95 positions.clear();
96 rotations.clear();
97 const size_ts chainIds = {};
98 protein.getSugarBindingSites(positions, rotations, siteIndices, chainIds);
99 BOOST_CHECK(positions.size() == 0);
100 BOOST_CHECK(rotations.size() == 0);
101
102 const auto sites = protein.getGlycosylationSites(siteIndices);
103 BOOST_CHECK(sites.size() == 4);
104
105 const std::vector<size_t> expectedSizes{5, 7, 5, 7};
106 size_t count = 0;
107 for (const auto& site : sites)
108 {
109 BOOST_CHECK(site.second.size() == expectedSizes[count]);
110 ++count;
111 }
112}
