45 inline float processShadows(const uniform CellGrowthRenderer* uniform self, varying ScreenSample& sample,
46  varying DifferentialGeometry& dg)
47 
48 {
49  // Shadows
50  float shadowIntensity = 0.f;
51  for (uniform int i = 0; shadowIntensity < 1.f && i < self->super.super.numLights; ++i)
52  {
53  const uniform Light* uniform light = self->super.super.lights[i];
54  const vec2f samplingLocation = make_vec2f(0.5f);
55  const varying Light_SampleRes lightSample = light->sample(light, dg, samplingLocation);
56  const vec3f lightColor = lightSample.weight;
57  const float radiance = reduce_max(lightColor);
58  const vec3f lightDirection = lightSample.dir;
59 
60  if (radiance <= 0.f)
61  continue;
62 
63  vec3f ld = lightDirection;
64  if (self->softShadowStrength > 0.f)
65  ld = normalize(ld + self->softShadowStrength * getRandomVector(self->super.useHardwareRandomizer,
66  self->super.super.super.fb->size.x, sample,
67  dg.Ns, self->super.randomNumber));
68 
69  Ray shadowRay;
70  setRay(shadowRay, dg.P, ld);
71  shadowRay.t0 = dg.epsilon;
72  shadowRay.t = self->shadowDistance;
73  shadowRay.time = sample.ray.time;
74 
75  unsigned int iteration = 0;
76  while (shadowIntensity < 1.f && iteration < self->super.maxRayDepth)
77  {
78  ++iteration;
79 
80  shadowRay.geomID = -1;
81  shadowRay.primID = -1;
82  shadowRay.instID = -1;
83 
84  traceRay(self->super.super.super.model, shadowRay);
85 
86  if (shadowRay.geomID < 0)
87  iteration = self->super.maxRayDepth;
88  else
89  {
90  DifferentialGeometry shadowDg;
91  postIntersect(self->super.super.super.model, shadowDg, shadowRay, DG_NG);
92  const vec4f simulationColor = getSimulationValue(&self->super, &shadowDg, shadowRay.primID);
93 
94  if (simulationColor.w > self->simulationThreshold)
95  shadowIntensity += self->shadowIntensity;
96 
97  shadowRay.t0 = shadowRay.t + shadowDg.epsilon;
98  shadowRay.t = self->shadowDistance - shadowRay.t;
99  }
100  }
101  }
102  return shadowIntensity;
103 }

105 inline vec3f CellGrowthRenderer_shadeRay(const uniform CellGrowthRenderer* uniform self, varying ScreenSample& sample)
106 {
107  Ray ray = sample.ray;
108  ray.time = inf;
109  sample.z = inf;
110 
111  float zDepth = 0.f;
112 
113  vec3f pathColor = make_vec3f(0.f);
114  sample.alpha = 0.f;
115 
116  while (true)
117  {
118  vec3f Kd;
119  float Ns = 1.f;
120 
121  traceRay(self->super.super.super.model, ray);
122 
123  if (ray.geomID < 0)
124  {
125  // No intersection
126  vec4f colorContribution =
127  skyboxMapping((Renderer*)self, ray, (AdvancedMaterial*)self->super.super.bgMaterial);
128  colorContribution.w = 1.f;
129  pathColor = make_vec3f(colorContribution);
130  break;
131  }
132 
133  // Intersection found
134  // Retreive information about the geometry, typically geometry ID,
135  // normal to the surface, material ID, texture coordinates, etc.
136  DifferentialGeometry dg;
137  postIntersect(self->super.super.super.model, dg, ray,
138  DG_NG | DG_NS | DG_NORMALIZE | DG_FACEFORWARD | DG_MATERIALID | DG_COLOR);
139 
140  // Material management
141  const uniform Material* material = dg.material;
142  const uniform AdvancedMaterial* advancedMaterial = (const uniform AdvancedMaterial*)material;
143 
144  MaterialShadingMode shadingMode = diffuse;
145 
146  if (!advancedMaterial)
147  Kd = make_vec3f(dg.color);
148  else
149  foreach_unique(mat in advancedMaterial)
150  {
151  shadingMode = mat->shadingMode;
152  Kd = make_vec3f(dg.color) * mat->Kd;
153  Ns = mat->Ns;
154  }
155 
156  const vec4f simulationColor = getSimulationValue(&self->super, &dg, ray.primID);
157 
158  if (simulationColor.w > self->simulationThreshold)
159  {
160  if (self->useTransferFunctionColor)
161  Kd = make_vec3f(simulationColor);
162 
163  // Shading
164  const uniform Light* uniform light = self->super.super.lights[0];
165  const vec2f samplingLocation = make_vec2f(0.5f);
166  const varying Light_SampleRes lightSample = light->sample(light, dg, samplingLocation);
167  const vec3f lightDirection = lightSample.dir;
168 
169  sample.z = ray.t;
170  sample.alpha = 1.f;
171  const vec3f intersection = ray.org + ray.t * ray.dir;
172 
173  float cosNL = 1.f;
174  switch (shadingMode)
175  {
176  case electron:
177  case electron_transparency:
178  cosNL = 1.f - max(0.f, dot(neg(ray.dir), dg.Ns));
179  break;
180  case cartoon:
181  cosNL = max(0.f, dot(lightDirection, dg.Ns));
182  const unsigned int angleAsInt = cosNL * Ns;
183  cosNL = (float)(angleAsInt) / Ns;
184  break;
185  case perlin:
186  float n = noise(intersection * Ns);
187  if (ray.t < dg.epsilon * 1.5e4f)
188  n += worleyNoise(intersection, 2.f);
189  else
190  n *= 2.f;
191 
192  n = clamp(n, 0.f, 1.f);
193  cosNL = n * max(0.f, dot(lightDirection, dg.Ns));
194  break;
195  case diffuse:
196  case diffuse_transparency:
197  cosNL = max(0.f, dot(lightDirection, dg.Ns));
198  }
199 
200  float shadowIntensity = 0.f;
201  if (self->shadowIntensity > 0.f)
202  shadowIntensity = processShadows(self, sample, dg);
203 
204  pathColor = cosNL * Kd * clamp(1.f - shadowIntensity, 0.f, 1.f);
205  break;
206  }
207 
208  // Next ray
209  ray.t0 = ray.t + dg.epsilon;
210  ray.t = inf;
211  ray.primID = -1;
212  ray.geomID = -1;
213  ray.instID = -1;
214  }
215 
216  return make_vec3f(pathColor) * self->super.super.exposure;
217 }

219 void CellGrowthRenderer_renderSample(uniform Renderer* uniform _self, void* uniform perFrameData,
220  varying ScreenSample& sample)
221 {
222  uniform CellGrowthRenderer* uniform self = (uniform CellGrowthRenderer * uniform) _self;
223  sample.ray.time = self->super.super.timestamp;
224  if (self->super.super.anaglyphEnabled)
225  {
226  ScreenSample s = sample;
227  s.ray.org = s.ray.org - self->super.super.anaglyphIpdOffset;
228  const vec3f colorLeft = CellGrowthRenderer_shadeRay(self, s);
229  const vec3f leftAnaglyphColor = getAnaglyphLeftColor(colorLeft);
230 
231  s = sample;
232  s.ray.org = s.ray.org + self->super.super.anaglyphIpdOffset;
233  const vec3f colorRight = CellGrowthRenderer_shadeRay(self, s);
234  const vec3f rightAnaglyphColor = getAnaglyphRightColor(colorRight);
235 
236  sample.rgb = leftAnaglyphColor + rightAnaglyphColor;
237  }
238  else
239  sample.rgb = CellGrowthRenderer_shadeRay(self, sample);
240 }
