32 void DefaultMaterial::commit()
33 {
34  if (ispcEquivalent == nullptr)
35  ispcEquivalent = ispc::DefaultMaterial_create(this);
36 
37  // Opacity
38  d = getParam1f("d", 1.f);
39  map_d = (ospray::Texture2D*)getParamObject("map_d", nullptr);
40  auto xform_d = getTextureTransform("map_d");
41 
42  // Diffuse color
43  Kd = getParam3f("kd", ospray::vec3f(.8f));
44  map_Kd = (ospray::Texture2D*)getParamObject("map_kd", nullptr);
45  auto xform_Kd = getTextureTransform("map_kd");
46 
47  // Specular color
48  Ks = getParam3f("ks", ospray::vec3f(0.f));
49  map_Ks = (ospray::Texture2D*)getParamObject("map_ks", nullptr);
50  auto xform_Ks = getTextureTransform("map_ks");
51 
52  // Specular exponent
53  Ns = getParam1f("ns", 10.f);
54  map_Ns = (ospray::Texture2D*)getParamObject("map_ns", nullptr);
55  auto xform_Ns = getTextureTransform("map_ns");
56 
57  // Bump mapping
58  map_Bump = (ospray::Texture2D*)getParamObject("map_bump", nullptr);
59  auto xform_Bump = getTextureTransform("map_bump");
60  auto rot_Bump = xform_Bump.l.orthogonal().transposed();
61 
62  // Refraction mapping
63  refraction = getParam1f("refraction", 0.f);
64  map_Refraction =
65  (ospray::Texture2D*)getParamObject("map_refraction", nullptr);
66  auto xform_Refraction = getTextureTransform("map_refraction");
67 
68  // Reflection mapping
69  reflection = getParam1f("reflection", 0.f);
70  map_Reflection =
71  (ospray::Texture2D*)getParamObject("map_reflection", nullptr);
72  auto xform_Reflection = getTextureTransform("map_reflection");
73 
74  // Light emission mapping
75  a = getParam1f("a", 0.f);
76  map_a = (ospray::Texture2D*)getParamObject("map_a", nullptr);
77  auto xform_a = getTextureTransform("map_a");
78 
79  // Glossiness
80  glossiness = getParam1f("glossiness", 1.f);
81 
82  ispc::DefaultMaterial_set(
83  getIE(), map_d ? map_d->getIE() : nullptr,
84  (const ispc::AffineSpace2f&)xform_d, d,
85  map_Refraction ? map_Refraction->getIE() : nullptr,
86  (const ispc::AffineSpace2f&)xform_Refraction, refraction,
87  map_Reflection ? map_Reflection->getIE() : nullptr,
88  (const ispc::AffineSpace2f&)xform_Reflection, reflection,
89  map_a ? map_a->getIE() : nullptr, (const ispc::AffineSpace2f&)xform_a,
90  a, glossiness, map_Kd ? map_Kd->getIE() : nullptr,
91  (const ispc::AffineSpace2f&)xform_Kd, (ispc::vec3f&)Kd,
92  map_Ks ? map_Ks->getIE() : nullptr,
93  (const ispc::AffineSpace2f&)xform_Ks, (ispc::vec3f&)Ks,
94  map_Ns ? map_Ns->getIE() : nullptr,
95  (const ispc::AffineSpace2f&)xform_Ns, Ns,
96  map_Bump ? map_Bump->getIE() : nullptr,
97  (const ispc::AffineSpace2f&)xform_Bump,
98  (const ispc::LinearSpace2f&)rot_Bump);
99 }
