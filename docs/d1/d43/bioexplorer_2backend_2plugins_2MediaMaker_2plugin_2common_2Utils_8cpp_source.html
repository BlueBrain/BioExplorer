52 void setCamera(const CameraKeyFrame &keyFrame, Camera &camera, const bool triggerCallback)
53 {
54  camera.setPosition(keyFrame.origin);
55  camera.setTarget(keyFrame.origin + keyFrame.direction);
56  // Orientation
57  const auto q = glm::inverse(glm::lookAt(keyFrame.origin, keyFrame.origin + keyFrame.direction,
58  keyFrame.up)); // Not quite sure why this
59  // should be inverted?!?
60  camera.setOrientation(q);
61 
62  // Aperture
63  if (camera.hasProperty(CAMERA_PROPERTY_APERTURE_RADIUS.name))
64  camera.updateProperty(CAMERA_PROPERTY_APERTURE_RADIUS.name, keyFrame.apertureRadius);
65 
66  // Focus distance
67  if (camera.hasProperty(CAMERA_PROPERTY_FOCAL_DISTANCE.name))
68  camera.updateProperty(CAMERA_PROPERTY_FOCAL_DISTANCE.name, keyFrame.focalDistance);
69 
70  // Stereo
71  if (camera.hasProperty(CAMERA_PROPERTY_STEREO.name))
72  camera.updateProperty(CAMERA_PROPERTY_STEREO.name, keyFrame.interpupillaryDistance != 0.0);
73  if (camera.hasProperty(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name))
74  camera.updateProperty(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name, keyFrame.interpupillaryDistance);
75 
76  camera.markModified(triggerCallback);
77 }

79 CameraKeyFrame getCameraKeyFrame(Camera &camera)
80 {
81  CameraKeyFrame keyFrame;
82  const auto &p = camera.getPosition();
83  keyFrame.origin = {p.x, p.y, p.z};
84  const auto d = glm::rotate(camera.getOrientation(), core::Vector3d(0., 0., -1.));
85  keyFrame.direction = {d.x, d.y, d.z};
86  const auto u = glm::rotate(camera.getOrientation(), core::Vector3d(0., 1., 0.));
87  keyFrame.up = {u.x, u.y, u.z};
88  if (camera.hasProperty(CAMERA_PROPERTY_APERTURE_RADIUS.name))
89  keyFrame.apertureRadius = camera.getProperty<double>(CAMERA_PROPERTY_APERTURE_RADIUS.name);
90  if (camera.hasProperty(CAMERA_PROPERTY_FOCAL_DISTANCE.name))
91  keyFrame.focalDistance = camera.getProperty<double>(CAMERA_PROPERTY_FOCAL_DISTANCE.name);
92  if (camera.hasProperty(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name))
93  keyFrame.interpupillaryDistance = camera.getProperty<double>(CAMERA_PROPERTY_INTERPUPILLARY_DISTANCE.name);
94  return keyFrame;
95 }
