36 Protein::Protein(Scene& scene, const ProteinDetails& details)
37  : Molecule(scene, doublesToVector3d(details.position), doublesToQuaterniond(details.rotation), details.chainIds)
38  , _details(details)
39 {
40  size_t lineIndex{0};
41 
42  std::stringstream lines{_details.contents};
43  std::string line;
44  std::string title;
45  std::string header{_details.name};
46 
47  while (getline(lines, line, '\n'))
48  {
49  if (line.find(KEY_ATOM) == 0)
50  _readAtom(line, _details.loadHydrogen);
51  else if (details.loadNonPolymerChemicals && line.find(KEY_HETATM) == 0)
52  _readAtom(line, _details.loadHydrogen);
53  else if (line.find(KEY_HEADER) == 0)
54  header = _readHeader(line);
55  else if (line.find(KEY_TITLE) == 0)
56  title = _readTitle(line);
57  else if (details.loadBonds && line.find(KEY_CONECT) == 0)
58  _readConnect(line);
59  else if (line.find(KEY_SEQRES) == 0)
60  _readSequence(line);
61  // else if (line.find(KEY_REMARK) == 0)
62  // _readRemark(line);
63  }
64 
65  if (_residueSequenceMap.empty())
66  {
67  // Build AA sequences from ATOMS if no SEQRES record exists
68  size_t previousReqSeq = std::numeric_limits<size_t>::max();
69  for (const auto& atom : _atomMap)
70  {
71  auto& sequence = _residueSequenceMap[atom.second.chainId];
72  if (previousReqSeq != atom.second.reqSeq)
73  sequence.resNames.push_back(atom.second.resName);
74  previousReqSeq = atom.second.reqSeq;
75  }
76  for (auto& sequence : _residueSequenceMap)
77  sequence.second.numRes = sequence.second.resNames.size();
78  }
79 
80  // Build 3d models according to atoms positions (re-centered to origin)
81  if (details.recenter)
82  {
83  core::Boxd newBounds;
84  const auto& center = _bounds.getCenter();
85  for (auto& atom : _atomMap)
86  {
87  atom.second.position -= center;
88  newBounds.merge(atom.second.position);
89  }
90  _bounds = newBounds;
91  }
92 
93  _buildModel(_details.assemblyName, _details.name, _details.pdbId, header, _details.representation,
94  _details.atomRadiusMultiplier, _details.loadBonds);
95 
96  _buildAminoAcidBounds();
97  _computeReqSetOffset();
98 
99  // Trans-membrane information
100  const auto transmembraneParams = doublesToVector2d(details.transmembraneParams);
101  _transMembraneOffset = transmembraneParams.x;
102  _transMembraneRadius = transmembraneParams.y;
103 }

116 void Protein::setColorScheme(const ProteinColorScheme& colorScheme, const Palette& palette, const size_ts& chainIds)
117 {
118  switch (colorScheme)
119  {
120  case ProteinColorScheme::none:
121  for (auto& atom : _atomMap)
122  _setMaterialDiffuseColor(atom.first, atomColorMap[0]);
123  break;
124  case ProteinColorScheme::atoms:
125  _setAtomColorScheme();
126  break;
127  case ProteinColorScheme::chains:
128  _setChainColorScheme(palette);
129  break;
130  case ProteinColorScheme::residues:
131  _setResiduesColorScheme(palette);
132  break;
133  case ProteinColorScheme::amino_acid_sequence:
134  _setAminoAcidSequenceColorScheme(palette);
135  break;
136  case ProteinColorScheme::glycosylation_site:
137  _setGlycosylationSiteColorScheme(palette);
138  break;
139  case ProteinColorScheme::region:
140  _setRegionColorScheme(palette, chainIds);
141  break;
142  default:
143  PLUGIN_THROW("Unknown colorscheme");
144  }
145 }

165 void Protein::_setGlycosylationSiteColorScheme(const Palette& palette)
166 {
167  if (palette.size() != 2)
168  PLUGIN_THROW("Invalid palette size. 2 colors are expected");
169 
170  // Initialize atom colors
171  for (const auto& atom : _atomMap)
172  {
173  const size_t index = static_cast<size_t>(atom.second.chainId[0]) - 63;
174  _setMaterialDiffuseColor(atom.first, palette[0]);
175  }
176 
177  const auto sites = getGlycosylationSites({});
178 
179  for (const auto chain : sites)
180  for (const auto site : chain.second)
181  for (const auto& atom : _atomMap)
182  if (atom.second.chainId == chain.first && atom.second.reqSeq == site)
183  _setMaterialDiffuseColor(atom.first, palette[1]);
184 
185  PLUGIN_INFO(3, "Applying Glycosylation Site color scheme (" << (sites.size() > 0 ? "2" : "1") << ")");

188 const std::map<std::string, size_ts> Protein::getGlycosylationSites(const size_ts& siteIndices) const
189 {
190  std::map<std::string, size_ts> sites;
191  for (const auto& sequence : _residueSequenceMap)
192  {
193  std::string shortSequence;
194  for (const auto& resName : sequence.second.resNames)
195  shortSequence += aminoAcidMap[resName].shortName;
196 
197  for (size_t i = 0; i < shortSequence.length(); ++i)
198  {
199  const auto offsetIndex = i + sequence.second.offset;
200  bool acceptSite{true};
201  if (!siteIndices.empty())
202  {
203  const auto it = find(siteIndices.begin(), siteIndices.end(), i);
204  acceptSite = (it != siteIndices.end());
205  }
206 
207  const char aminoAcid = shortSequence[offsetIndex];
208  if (aminoAcid == 'N' && acceptSite)
209  {
210  if (i < shortSequence.length() - 2)
211  {
212  const auto aminAcid1 = shortSequence[offsetIndex + 1];
213  const auto aminAcid2 = shortSequence[offsetIndex + 2];
214  if ((aminAcid2 == 'T' || aminAcid2 == 'S') && aminAcid1 != 'P')
215  sites[sequence.first].push_back(i);
216  }
217  }
218  }
219  }
220 
221  for (const auto& site : sites)
222  {
223  std::string indices = "[";
224  for (const auto& index : site.second)
225  {
226  if (indices.length() > 1)
227  indices += ",";
228  indices += std::to_string(index + 1); // Indices start at 1, not 0
229  }
230  indices += "]";
231  PLUGIN_INFO(3, "Found " << site.second.size() << " glycosylation sites " << indices << " on sequence "
232  << site.first);
233  }
234  return sites;
235 }

251 void Protein::_getSitesTransformations(Vector3ds& positions, Quaternions& rotations,
252  const std::map<std::string, size_ts>& sitesPerChain) const
253 {
254  for (const auto& chain : sitesPerChain)
255  {
256  const auto itAminoAcids = _aminoAcidBounds.find(chain.first);
257  if (itAminoAcids == _aminoAcidBounds.end())
258  PLUGIN_THROW("Invalid chain");
259 
260  const auto aminoAcidsPerChain = (*itAminoAcids).second;
261  for (const auto bindingSite : chain.second)
262  {
263  // Protein center
264  const Vector3d proteinCenter = _bounds.getCenter();
265  Boxd bindingSiteBounds;
266  Vector3d bindingSiteCenter;
267 
268  const auto offsetBindingSite = bindingSite;
269 
270  // Site center
271  const auto it = aminoAcidsPerChain.find(offsetBindingSite);
272  if (it != aminoAcidsPerChain.end())
273  {
274  bindingSiteBounds = (*it).second;
275  bindingSiteCenter = bindingSiteBounds.getCenter();
276 
277  // rotation is determined by the center of the site and the center of the protein
278  const auto bindRotation = normalize(bindingSiteCenter - proteinCenter);
279  positions.push_back(_position + _rotation * bindingSiteCenter);
280  rotations.push_back(_rotation * safeQuatlookAt(bindRotation));
281  }
282  else
283  PLUGIN_WARN("Chain: " << chain.first << ", Site " << bindingSite + 1
284  << " is not available in the protein source");
285  }
286  }
287 }

299 void Protein::getSugarBindingSites(Vector3ds& positions, Quaternions& rotations, const size_ts& siteIndices,
300  const size_ts& chainIds) const
301 {
302  positions.clear();
303  rotations.clear();
304 
305  std::set<std::string> chainIdsAsString;
306  for (const auto& atom : _atomMap)
307  {
308  bool acceptChain{true};
309  const size_t chainId = static_cast<size_t>(atom.second.chainId[0] - 64);
310  if (!chainIds.empty())
311  acceptChain = (std::find(chainIds.begin(), chainIds.end(), chainId) != chainIds.end());
312 
313  if (acceptChain)
314  chainIdsAsString.insert(atom.second.chainId);
315  }
316 
317  std::map<std::string, size_ts> sites;
318  for (const auto& chainIdAsString : chainIdsAsString)
319  sites[chainIdAsString] = siteIndices;
320 
321  _getSitesTransformations(positions, rotations, sites);
322 }

344 void Protein::_processInstances(ModelDescriptorPtr md, const Vector3ds& positions, const Quaternions& rotations,
345  const Quaterniond& moleculeRotation, const MolecularSystemAnimationDetails& randInfo)
346 {
347  size_t count = 0;
348  const auto& proteinInstances = _modelDescriptor->getInstances();
349  for (const auto& proteinInstance : proteinInstances)
350  {
351  const auto& proteinTransformation = proteinInstance.getTransformation();
352  for (size_t i = 0; i < positions.size(); ++i)
353  {
354  const auto& position = positions[i];
355  auto rotation = rotations[i];
356 
357  Transformation glycanTransformation;
358  glycanTransformation.setTranslation(position);
359 
360  if (randInfo.rotationSeed != 0)
361  rotation = weightedRandomRotation(rotation, randInfo.rotationSeed, i, randInfo.rotationStrength);
362  glycanTransformation.setRotation(moleculeRotation * rotation);
363 
364  const Transformation combinedTransformation = proteinTransformation * glycanTransformation;
365 
366  if (count == 0)
367  md->setTransformation(combinedTransformation);
368  const ModelInstance instance(true, false, combinedTransformation);
369  md->addInstance(instance);
370  ++count;
371  }
372  }
373 }

375 void Protein::addGlycan(const SugarDetails& details)
376 {
377  if (_glycans.find(details.name) != _glycans.end())
378  PLUGIN_THROW("A glycan named " + details.name + " already exists in protein " + _details.name +
379  " of assembly " + _details.assemblyName);
380 
381  Vector3ds glycanPositions;
382  Quaternions glycanRotations;
383  getGlycosylationSites(glycanPositions, glycanRotations, details.siteIndices);
384 
385  if (glycanPositions.empty())
386  PLUGIN_THROW("No glycosylation site was found on " + details.proteinName);
387 
388  // Create glycans and attach them to the glycosylation sites of the target protein
389  GlycansPtr glycans(new Glycans(_scene, details));
390  auto modelDescriptor = glycans->getModelDescriptor();
391  const Quaterniond proteinRotation = doublesToQuaterniond(details.rotation);
392 
393  const auto randInfo = doublesToMolecularSystemAnimationDetails(details.animationParams);
394  _processInstances(modelDescriptor, glycanPositions, glycanRotations, proteinRotation, randInfo);
395 
396  _glycans[details.name] = std::move(glycans);
397  _scene.addModel(modelDescriptor);
398 }

421 
422  _glycans[details.name] = std::move(glucoses);
