32 Protein::Protein(Scene& scene, const ProteinDescriptor& descriptor)
33  : Molecule(scene, descriptor.chainIds)
34  , _descriptor(descriptor)
35 {
36  size_t lineIndex{0};
37 
38  std::stringstream lines{_descriptor.contents};
39  std::string line;
40  std::string title{_descriptor.name};
41  std::string header{_descriptor.name};
42 
43  while (getline(lines, line, '\n'))
44  {
45  if (line.find(KEY_ATOM) == 0)
46  _readAtom(line, _descriptor.loadHydrogen);
47  else if (descriptor.loadNonPolymerChemicals &&
48  line.find(KEY_HETATM) == 0)
49  _readAtom(line, _descriptor.loadHydrogen);
50  else if (line.find(KEY_HEADER) == 0)
51  header = _readHeader(line);
52  else if (line.find(KEY_TITLE) == 0)
53  title = _readTitle(line);
54  else if (descriptor.loadBonds && line.find(KEY_CONECT) == 0)
55  _readConnect(line);
56  else if (line.find(KEY_SEQRES) == 0)
57  _readSequence(line);
58  // else if (line.find(KEY_REMARK) == 0)
59  // _readRemark(line);
60  }
61 
62  if (_sequenceMap.empty())
63  {
64  // Build AA sequences from ATOMS if not SEQRES record exists
65  size_t previousReqSeq = std::numeric_limits<size_t>::max();
66  for (const auto& atom : _atomMap)
67  {
68  auto& sequence = _sequenceMap[atom.second.chainId];
69  if (previousReqSeq != atom.second.reqSeq)
70  sequence.resNames.push_back(atom.second.resName);
71  previousReqSeq = atom.second.reqSeq;
72  }
73  for (auto& sequence : _sequenceMap)
74  sequence.second.numRes = sequence.second.resNames.size();
75  }
76 
77  // Build 3d models according to atoms positions (re-centered to origin)
78  if (descriptor.recenter)
79  {
80  const auto& center = _bounds.getCenter();
81  for (auto& atom : _atomMap)
82  atom.second.position -= center;
83  }
84 
85  _buildModel(_descriptor.assemblyName, _descriptor.name, title, header,
86  _descriptor.representation, _descriptor.atomRadiusMultiplier,
87  _descriptor.loadBonds);
88 }

90 void Protein::setColorScheme(const ColorScheme& colorScheme,
91  const Palette& palette, const size_ts& chainIds)
92 {
93  switch (colorScheme)
94  {
95  case ColorScheme::none:
96  for (auto& atom : _atomMap)
97  _setMaterialDiffuseColor(atom.first, atomColorMap[0]);
98  break;
99  case ColorScheme::atoms:
100  _setAtomColorScheme();
101  break;
102  case ColorScheme::chains:
103  _setChainColorScheme(palette);
104  break;
105  case ColorScheme::residues:
106  _setResiduesColorScheme(palette);
107  break;
108  case ColorScheme::amino_acid_sequence:
109  _setAminoAcidSequenceColorScheme(palette);
110  break;
111  case ColorScheme::glycosylation_site:
112  _setGlycosylationSiteColorScheme(palette);
113  break;
114  case ColorScheme::region:
115  _setRegionColorScheme(palette, chainIds);
116  break;
117  default:
118  PLUGIN_THROW(std::runtime_error("Unknown colorscheme"))
119  }
120 }

122 void Protein::_setRegionColorScheme(const Palette& palette,
123  const size_ts& chainIds)
124 {
125  size_t atomCount = 0;
126  for (auto& atom : _atomMap)
127  {
128  bool applyColor{true};
129  if (!chainIds.empty())
130  {
131  const size_t chainId =
132  static_cast<size_t>(atom.second.chainId[0] - 64);
133  applyColor = (std::find(chainIds.begin(), chainIds.end(),
134  chainId) != chainIds.end());
135  }
136  if (applyColor)
137  _setMaterialDiffuseColor(atom.first, palette[atom.second.reqSeq]);
138  }
139 
140  PLUGIN_INFO << "Applying Amino Acid Sequence color scheme ("
141  << (atomCount > 0 ? "2" : "1") << ")" << std::endl;

144 void Protein::_setGlycosylationSiteColorScheme(const Palette& palette)
145 {
146  if (palette.size() != 2)
147  PLUGIN_THROW(
148  std::runtime_error("Invalid palette size. 2 colors are expected"));
149 
150  // Initialize atom colors
151  for (const auto& atom : _atomMap)
152  {
153  const size_t index = static_cast<size_t>(atom.second.chainId[0]) - 63;
154  _setMaterialDiffuseColor(atom.first, palette[0]);
155  }
156 
157  const auto sites = getGlycosylationSites({});
158 
159  for (const auto chain : sites)
160  for (const auto site : chain.second)
161  for (const auto& atom : _atomMap)
162  if (atom.second.chainId == chain.first &&
163  atom.second.reqSeq == site)
164  _setMaterialDiffuseColor(atom.first, palette[1]);
165 
166  PLUGIN_INFO << "Applying Glycosylation Site color scheme ("
167  << (sites.size() > 0 ? "2" : "1") << ")" << std::endl;

170 std::map<std::string, size_ts> Protein::getGlycosylationSites(
171  const std::vector<size_t>& siteIndices) const
172 {
173  std::map<std::string, size_ts> sites;
174  for (const auto& sequence : _sequenceMap)
175  {
176  std::string shortSequence;
177  for (const auto& resName : sequence.second.resNames)
178  shortSequence += aminoAcidMap[resName].shortName;
179 
180  for (size_t i = 0; i < shortSequence.length(); ++i)
181  {
182  bool acceptSite{true};
183  if (!siteIndices.empty())
184  {
185  const auto it =
186  find(siteIndices.begin(), siteIndices.end(), i + 1);
187  acceptSite = (it != siteIndices.end());
188  }
189 
190  const char aminoAcid = shortSequence[i];
191  if (aminoAcid == 'N' && acceptSite)
192  {
193  if (i < shortSequence.length() - 2)
194  {
195  const auto aminAcid1 = shortSequence[i + 1];
196  const auto aminAcid2 = shortSequence[i + 2];
197  if ((aminAcid2 == 'T' || aminAcid2 == 'S') &&
198  aminAcid1 != 'P')
199  sites[sequence.first].push_back(i);
200  }
201  }
202  }
203  }
204 
205  for (const auto& site : sites)
206  {
207  std::string indices = "[";
208  for (const auto& index : site.second)
209  {
210  if (indices.length() > 1)
211  indices += ",";
212  indices += std::to_string(index + 1); // Indices start at 1, not 0
213  }
214  indices += "]";
215  PLUGIN_INFO << "Found " << site.second.size() << " glycosylation sites "
216  << indices << " on sequence " << site.first << std::endl;
217  }
218  return sites;
219 }

221 void Protein::_getSitesTransformations(
222  std::vector<Vector3f>& positions, std::vector<Quaterniond>& rotations,
223  const std::map<std::string, size_ts>& sites) const
224 {
225  for (const auto chain : sites)
226  {
227  for (const auto site : chain.second)
228  {
229  bool validSite{false};
230  Boxf bounds;
231  for (const auto& atom : _atomMap)
232  if (atom.second.chainId == chain.first &&
233  atom.second.reqSeq == site)
234  {
235  bounds.merge(atom.second.position);
236  validSite = true;
237  }
238 
239  if (validSite)
240  {
241  const auto& center = bounds.getCenter();
242  positions.push_back(center);
243  rotations.push_back(
244  glm::quatLookAt(normalize(center - _bounds.getCenter()),
245  {0.f, 0.f, -1.f}));
246  }
247  }
248  }
249 }

263 void Protein::getSugarBindingSites(std::vector<Vector3f>& positions,
264  std::vector<Quaterniond>& rotations,
265  const size_ts& siteIndices,
266  const size_ts& chainIds) const
267 {
268  positions.clear();
269  rotations.clear();
270 
271  std::set<std::string> chainIdsAsString;
272  for (const auto& atom : _atomMap)
273  {
274  bool acceptChain{true};
275  const size_t chainId = static_cast<size_t>(atom.second.chainId[0] - 64);
276  if (!chainIds.empty())
277  acceptChain = (std::find(chainIds.begin(), chainIds.end(),
278  chainId) != chainIds.end());
279 
280  if (acceptChain)
281  chainIdsAsString.insert(atom.second.chainId);
282  }
283 
284  std::map<std::string, size_ts> sites;
285  for (const auto& chainIdAsString : chainIdsAsString)
286  sites[chainIdAsString] = siteIndices;
287 
288  _getSitesTransformations(positions, rotations, sites);
289 }
