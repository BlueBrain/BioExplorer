49 bool PointCloudMesher::toConvexHull(Model& model, const PointCloud& pointCloud)
50 {
51 #ifdef USE_CGAL
52  bool addModel{false};
53  for (const auto& point : pointCloud)
54  {
55  auto material =
56  model.createMaterial(point.first, std::to_string(point.first));
57  brayns::PropertyMap props;
58  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
59  static_cast<int>(MaterialShadingMode::diffuse)});
60  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
61  material->updateProperties(props);
62 
63  std::vector<Point_3> points;
64  for (const auto& c : point.second)
65  points.push_back({c.x, c.y, c.z});
66 
67  CGAL::Object obj;
68  // compute convex hull of non-collinear points
69  CGAL::convex_hull_3(points.begin(), points.end(), obj);
70  if (const Polyhedron_3* poly = CGAL::object_cast<Polyhedron_3>(&obj))
71  {
72  PLUGIN_INFO << "The convex hull contains "
73  << poly->size_of_vertices() << " vertices" << std::endl;
74 
75  for (auto eit = poly->edges_begin(); eit != poly->edges_end();
76  ++eit)
77  {
78  Point_3 a = eit->vertex()->point();
79  Point_3 b = eit->opposite()->vertex()->point();
80  model.addCylinder(point.first, {{a.x(), a.y(), a.z()},
81  {b.x(), b.y(), b.z()},
82  point.second[0].w});
83  addModel = true;
84  }
85  }
86  else
87  PLUGIN_ERROR << "something else" << std::endl;
88  }
89  return addModel;
90 #else
91  return false;
92 #endif
93 }

95 bool PointCloudMesher::toMetaballs(brayns::Model& model,
96  const PointCloud& pointCloud,
97  const size_t gridSize, const float threshold)
98 {
99  auto& triangles = model.getTriangleMeshes();
100  for (const auto& point : pointCloud)
101  {
102  if (point.second.empty())
103  continue;
104 
105  PLUGIN_INFO << "Material " << point.first
106  << ", number of balls: " << point.second.size()
107  << std::endl;
108 
109  auto material =
110  model.createMaterial(point.first, std::to_string(point.first));
111  brayns::PropertyMap props;
112  props.setProperty({MATERIAL_PROPERTY_SHADING_MODE,
113  static_cast<int>(MaterialShadingMode::diffuse)});
114  props.setProperty({MATERIAL_PROPERTY_USER_PARAMETER, 1.0});
115  material->updateProperties(props);
116 
117  MetaballsGenerator metaballsGenerator;
118  metaballsGenerator.generateMesh(point.first, point.second, gridSize,
119  threshold, triangles);
120  }
121  return !triangles.empty();
122 }
